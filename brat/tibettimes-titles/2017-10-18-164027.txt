ཞི་རྒོལ་བ་ ༥༠ ལྷག་འཛིན་བཟུང་བྱས།

དེ་རིང་བོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་ཀྱིས་ལྡི་ལིའི་རྒྱ་ནག་གཞུང་ཚབ་ཁང་གི་མདུན་དུ་ཞི་རྒོལ་སྐད་འབོད་བྱེད་སྐབས་དབུས་རྒྱུན་ཚུད་པའི་ཞི་རྒོལ་བ་ ༥༠ ལྷག་འཛིན་བཟུང་བཙན་འཁྲིད་བྱས་ཡོད་འདུག དེ་རིང་རྒྱ་ནག་རྒྱལ་ཡོངས་འཐུས་མིའི་ཚོགས་ཆེན་ཐེངས་ ༡༩ པ་མགོ་བཙུགས་ཡོད་པ་དང་བསྟུན། བོད་ཀྱི་གཞོན་ནུ་དབུས་དང་རོ་ཧི་ནི་ས་གནས་གཉིས་ཀྱིས དེ་རིང་སྔ་དྲོའི་ཆུ་ཚོད་ ༡༡ ཐོག་རྒྱ་གར་གྱི་རྒྱལ་ས་ལྡི་ལིའི་རྒྱ་ནག་གཞུང་ཚབ་ཁང་གི་མདུན་དུ་ཞི་རྒོལ་སྐད་འབོད་ཀྱི་ལས་འགུལ་སྤེལ་ཡོད་པ་དང་། སྐབས་དེར་རྒྱ་གར་ཉེན་རྟོག་པས་དབུས་བོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་ཀྱི་ཚོགས་གཙོ་བསྟན་འཛིན་འཇིགས་མེད་ལགས་ཀྱི་གཙོས་རྒྱུན་ལས་མི་ ༥ དང་། ལྡི་ལི་རོ་ཧི་ནི་ས་གནས་གཞོན་ནུ་ལྷན་ཚོགས་ཀྱི་རྒྱུན་ལས་ ༥༠ ལྷག་འཛིན་བཟུང་བཙན་འཁྲིད་བྱས་ཡོད་འདུག འདི་གའི་གསར་འགོད་པས་ཁ་པར་བརྒྱུད་དབུས་བོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་ཀྱི་ཕྱི་འབྲེལ་དྲུང་ཆེ་ཚེ་དབང་སྒྲོལ་མ་ལགས་སུ་དེའི་སྐོར་སྐད་ཆ་འདྲི་སྐབས།  ངོ་རྒོལ་སྐད་འབོད་ཀྱི་ལས་འགུལ་འདི་ནི་རྒྱ་ནག་རྒྱལ་ཡོངས་འཐུས་མིའི་ཚོགས་ཆེན་ཐེངས་ ༡༩ པ་ཚོགས་བཞིན་པ་དང་། ཞི་ཅིན་ཕིན་གྱིས་ལོ་ལྔར་རྒྱ་ནག་གི་སྲིད་འཛིན་བྱས་པའི་རིང་དྲང་བདེན་བྱས་མེད་པ་དང་། ལྷག་པར་དུ་བོད་ལ་རང་དབང་མེད་ཅིང་དྲག་གནོན་བྱེད་ཀྱི་ཡོད་པར་བརྟེན། བོད་ནང་མུ་མཐུད་རང་ལུས་རང་བསྲེགས་བྱུང་གི་ཡོད་པ་རེད་ལ། ཞི་ཅིན་ཕིན་གྱིས་ད་དུང་ལོ་ལྔ་ལ་རྒྱ་ནག་གི་སྲིད་འཛིན་བྱེད་པར་རེ་བ་བརྒྱབ་ཀྱི་ཡོད་པ་དེར་ངོ་རྒོལ་སྐད་འབོད་ཀྱི་ལས་འགུལ་སྤེལ་བ་ཡིན། ད་ལྟ་དབུས་ས་གཉིས་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་ཀྱི་རྒྱུན་ལས་ ༥༠ ལྷག་རྒྱ་གར་གྱི་ཉེན་རྟོག་པས་འཛིན་བཟུང་བཀག་ཉར་བྱས་ཡོད་པ་དང་། ཁོང་ཚོ་ཉེན་རྟོག་པས་ག་དུས་གློད་གྲོལ་གཏོང་གི་ཡོད་མེད་བཤད་ཐུབ་པ་མི་འདུག ཅེས་བརྗོད་བྱུང་། རྒྱ་ནག་དམར་ཤོག་ཚོགས་པའི་ཚོགས་ཆེན་དང་བསྟུན། དེ་རིང་ཕྱི་ཚེས་ ༡༨ ཉིན་ཐའེ་ཝན་དང་ཨ་རི། ཁ་ན་ཌ་སོགས་སུའང་ཞི་རྒོལ་གྱི་ལས་འགུལ་སྤེལ་གྱི་ཡོད་འདུག གསར་འགོད་པ། ཉི་མ་ཚེ་རིང་།
