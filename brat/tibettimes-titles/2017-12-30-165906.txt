ང་དགུ་སྒྱུ་རྩལ་ཁང་སྒོ་འབྱེད་བྱས།

བོད་རིགས་སྒྱུ་རྩལ་བ་བསྟན་འཛིན་རིག་གྲོལ་ལགས་ཀྱིས་ང་དགུ་སྒྱུ་རྩལ་ཁང་ཞེས་པ་ཞིག་སྒོ་འབྱེད་གནང་སྟེ་བོད་ཀྱི་སྒྱུ་རྩལ་ལ་དགོས་དམིགས་ངེས་ཅན་ཞིག་ཡོད་དགོས་སྐོར་བརྗོད་སོང་།  ཁ་སང་ཕྱི་དྲོའི་ཆུ་ཚོད་ ༦ སྟེང་དྷ་ཤོད་ཁན་ཡཱ་ར་ལམ་གྱི་(Expresso Bar)ཉེ་འགྲམ་དུ། བོད་རིགས་ན་གཞོན་སྒྱུ་རྩལ་བ་བསྟན་འཛིན་རིག་གྲོལ་ལགས་ནས་(ང་དགུ་སྒྱུ་རྩལ་ཁང་)ཞེས་པ་ཞིག་སྒོ་འབྱེད་ཀྱི་མཛད་སྒོ་སྟབས་བདེ་ཞིག་ཚོགས་སོང་བ་དང་།  ཚོགས་ཐོག་ཏུ་བོད་རིགས་རྩོམ་པ་པོ་ལྡུམ་ར་བུ་ཆུང་བསོད་ནམས་ལགས་དང་། བོད་ཀྱི་ཐང་བྲིས་སྒྱུ་རྩལ་བ་ཁག་ཅིག དེ་བཞིན་ན་གཞོན་སྙན་ངག་པ་དང་གློག་བརྙན་འཁྲབ་སྟོན་པ་ཁག་ཅིག  ང་དགུ་སྒྱུ་རྩལ་ཁང་གི་ལས་མི་དང་གསར་འགོད་པ་བཅས་འཛོམས་འདུག  སྒྱུ་རྩལ་ཁང་སྒོ་འབྱེད་ཐོག་རྩོམ་པ་པོ་བུ་ཆུང་བསོད་ནམས་ལགས་ཀྱིས། བོད་ཀྱི་དཔལ་ཡོན་འཕེལ་རིམ་གྱི་ལོ་རྒྱུས་ལ་བལྟས་ན།  བོད་ཀྱི་ཆོས་ལུགས་ལ་འཕེལ་རྒྱས་རིམ་པ་བྱུང་ཡོད་པ་དང་། ཆབ་སྲིད་ལའང་འཕོ་འགྱུར་བྱུང་ཡོད།  ཡིན་ན་ཡང་།  འགྱུར་བ་མ་ཕྱིན་པ་གཅིག་ཡོད་པ་དེ་བོད་ཀྱི་སྒྱུ་རྩལ་བྱེད་སྒོ་སྔར་བཞིན་མུ་མཐུད་འགྲོ་བསྡད་ཡོད་པ་རེད་ལ། འདས་པའི་ལོ་ངོ་ལྔ་བཅུའི་རིང་བཙན་བྱོལ་ནང་གཞུང་མང་གཉིས་ནས་ཀྱང་སྒྱུ་རྩལ་བ་གསོ་སྐྱོང་བྱེད་པར་འབད་བརྩོན་དེ་ཙམ་བྱུང་མེད་པ་རེད།  ཅེས་དང་། ད་ལྟའི་ཆར་བོད་མིའི་སྒྲིག་འཛུགས་ནས་བོད་ཀྱི་སྒྱུ་རྩལ་བ་དང་གློག་བརྙན་འཁྲབ་སྟོན་པར་མཐུན་རྐྱེན་སྦྱོར་སྤྲོད་མགོ་འཛུགས་གནང་ཡོད་པ་དང་། བོད་པའི་སྤྱི་ཚོགས་ནང་རྩོམ་པ་པོ་དང་རི་མོ་བ།  དེ་བཞིན། གློག་བརྙན་བཟོ་མཁན་སོགས་མང་པོ་ཞིག་རིམ་གྱིས་ཡོང་གི་ཡོད་སྟབས།  མ་འོངས་པའི་བོད་ཀྱི་སྒྱུ་རྩལ་ལ་རེ་བ་ཆེན་པོ་ཡོད་སྐོར་སོགས་བརྗོད་སོང་།  ང་དགུ་སྒྱུ་རྩལ་ཁང་གི་འགན་འཛིན་བསྟན་འཛིན་རིག་གྲོལ་ལགས་ཀྱིས། བོད་ཀྱི་སྒྱུ་རྩལ་ལ་དགོས་དམིགས་ཤིག་ཡོད་དགོས་སྟབས། ང་དགུ་སྒྱུ་རྩལ་ཁང་དེ་སྒྱུ་རྩལ་བ་ལྷན་འཛོམས་བྱེད་ས་ཞིག་ཡིན་ལ། ད་ལྟའི་ཆར་ང་དགུ་སྒྱུ་རྩལ་ཁང་དུ་ལས་མི་རི་མོ་བ་བདུན་དང་། དེ་བཞིན་རྩོམ་པ་པོ་དང་གློག་བརྙན་འཁྲབ་སྟོན་པ་ཁག་ཅིག་ཡོད་སྐོར་བརྗོད་བྱུང་། དེའི་རྗེས་བོད་ཀྱི་སྒྱུ་རྩལ་བ་བློ་བཟང་བསྟན་འཛིན་ལགས་ཀྱིས་འཁྲབ་སྟོན་གནང་བའི་གློག་བརྙན་(The Lower Rooms)ཞེས་པ་གཟིགས་འབུལ་དང་སྙན་ངག་གྱེས་འདོན་སོགས་བྱས་སོང་། གསར་འགོད་པ། སྐལ་བཟང་སྦྱིན་པ།
