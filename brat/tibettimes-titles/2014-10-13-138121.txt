ཧྥ་རན་སིའི་རྒྱལ་སར་ལས་འགུལ་སྤེལ།

ཁ་སང་ཧྥ་རན་སའི་རྒྱལ་སར་བོད་པ་དང་རྒྱ་རིགས་དམངས་གཙོ་དོན་གཉེར་བ་སོགས་ཚོགས་པ་ཁག་ཅིག་གི་སྣེ་འཁྲིད་འོག་རྒྱལ་སྤྱིའི་རང་དབང་གོམ་བགྲོད་ཐེངས་བདུན་པ་ཞེས་པ་ཞིག་སྤེལ་ཡོད་འདུག་ལ། གཞན། འཆར་ལོའི་གསུམ་བཅུའི་དུས་དྲན་ཉིན་ཡུ་རོབ་ཡོངས་རྫོགས་མཉམ་རུབ་ཀྱི་ལས་འགུལ་གཞི་རྒྱ་ཆེན་པོ་ཞིག་ཀྱང་སྤེལ་འཆར་ཡོད་འདུག དེ་རིང་ཕྱི་ཚེས་ ༡༡ ཉིན་གྱི་ཉིན་རྒྱབ་ཆུ་ཚོད་ ༢།༣༠ སྟེང་། ཕ་རི་སི་བོད་རིགས་ཚོགས་པའི་ཚོགས་གཙོ་ཐུབ་བསྟན་རྒྱ་མཚོ་ལཊ་དང་། ཨེ་ཤ་ཡའི་ཝེ་ཐི་ནམ་དང་ཤིན་ཅང་། འབར་མ། ལོའོ་སི། རྒྱ་ནག་དམངས་གཙོ་དོན་གཉེར་བའི་ཚོགས་པ་ཁག་གི་ཚོགས་གཙོ་རྣམ་པའི་སྣེ་ཁྲིད་འོག བོད་པ་གཙོས་བོད་དོན་རྒྱབ་སྐྱོར་བ་དང་། དེ་བཞིན་མི་རིགས་འདྲ་མིན་གྱི་མི་མང་བརྒྱ་ཕྲག་ལྷག་གིས་རྒྱལ་ས་ཕ་རི་སིའི་(staue de liberté) རང་དབང་འདྲ་སྐུ་མཚོ་ཁ་ནས་(trocadéro)འགྲོ་བ་མིའི་རང་དབང་ཐང་ཆེན་བར། འཕྲེད་དར་སྟེང་འབོད་ཚིག་སྣ་མང་བཀོད་པའི་ཐོག་ཞི་བའི་གོམ་བགྲོད་ལས་འགུལ་ཞིག་སྤེལ་སོང་། འགུལ་དེ་འདྲ་ཞིག་སྤེལ་དགོས་དོན་ནི་བཙན་གནོན་འོག་ཡོད་པའི་སོ་སོའི་སྲིད་གཞུང་ཁག་ལ་དྲག་གནོན་མཚམས་འཇོག་གིས་བོད་རྒྱ་འབྲེལ་མོལ་བསྐྱར་གསོ་དགོས་པ་དང་། གྲངས་ཉུང་མི་རིགས་ཀྱི་ངོ་བོ་སྲུང་སྐྱོབ་དང་དེ་དག་གི་གཞི་རྩའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་བཅས་ལ་བརྩི་བཀུར་དགོས་པ་བཅས་ཀྱི་ཆེད་དུ་ཡིན་འདུག ཕ་རི་སིའི་བོད་རིགས་ཚོགས་པའི་ཚོགས་གཙོ་ཐུབ་བསྟན་རྒྱ་མཚོ་ལགས་ཀྱིས་ངོ་སྤྲོད་གནང་བ་ལྟར་ན། ༢༠༠༥ ལོ་ནས་བཟུང་ཕ་རི་སི་བོད་རིགས་ཚོགས་པས་ལྷོ་ཤར་ཨེ་ཤ་ལའི་རང་དབང་གོམ་བགྲོད་འདིའི་ནང་མཉམ་ཞུགས་གནང་ཡོད་ཅིང་། འཛུལ་ཞུགས་བྱེད་དོན་གཙོ་བོ་རྒྱ་གཞུང་གི་དྲག་གནོན་སྲིད་བྱུས་འོག་བོད་ནང་གི་བོད་མི་ཚོས་མུ་མཐུད་ཚད་ལས་བརྒལ་བའི་སྡུག་སྦྱོང་མནར་གཅོད་མྱངས་དང་མྱོང་བཞིན་ཡོད་པ་དང་། ཐ་ན་རང་ལུས་མེར་བསྲེགས་ཀྱིས་བོད་མིའི་སེམས་ཀྱི་དཀའ་སྡུག་ཕྱི་ལ་སྟོན་བཞིན་ཡོད་སྟབས། དེ་རིང་མི་རིགས་ཁག་མཉམ་རུབ་ཐོག་ལས་འགུལ་འདི་འདྲ་ཞིག་སྤེལ་ཐུབ་ན་གནས་ཚུལ་དྲིལ་བསྒྲགས་ཡོང་བར་ཕན་ཐོགས་ཡོད་རྒྱུ་དང་། ཕྱོགས་གཞན་ཞིག་ནས་ལས་འགུལ་འདི་བརྒྱུད་ཧྥ་རན་སི་གཞུང་དང་ཡུ་རོབ་གྲོས་ཚོགས་སུ་བོད་ནང་གི་ཛ་དྲག་གནས་སྟངས་ཐོག་སྔར་ལས་རྒྱབ་སྐྱོར་ཡོང་ཆེད་དུ་ཡིན་སྐོར་བརྗོད་སོང་། གཞན། ཉེ་ལམ་ཨི་ཊ་ལིར་ཚོགས་པའི་ཡུ་རོབ་བོད་རིགས་ཚོགས་པ་ཁག་གི་ཚོགས་གཙོ་རྣམ་པའི་ཚོགས་འདུའི་སྟེང་། རྒྱལ་སྤྱིའི་སྡིངས་ཆའི་ཐོག་བོད་དོན་ལས་འགུལ་སྔར་ལས་གཞི་རྒྱ་ཆེ་རུ་གཏོང་རྒྱུར་ཐུན་མོང་ངོས་འཛིན་ཐོག མ་འོངས་པར་ཡང་ཡུ་རོབ་ཀྱི་གལ་གནད་ཆེ་བའི་ས་གནས་ཁག་ཏུ། མུ་མཐུད་ཨོ་སི་ཊི་རི་ཡའི་རྒྱལ་ས་ཝེ་ནམ་དང་སྦེལ་ཇམ་རྒྱལ་ས་སྦར་སིལ་ལྟ་བུའི་མི་མང་ལས་འགུལ་སྤེལ་རྒྱུ་གལ་ཆེན་དུ་མཐོང་སྟེ། འཆར་ལོའི་གསུམ་བཅུའི་དུས་དྲན་ཐེངས་ ༥༦ པ་དང་བསྟུན་ཏེ། ཧྥ་རན་སིའི་རྒྱལ་ས་ཕ་རི་སིར་མི་མང་འདུ་འཛོམས་ལས་འགུལ་ཞིག་སྤེལ་རྒྱུའི་གྲོས་ཆོད་བཞག་ཡོད་པ་དང་། བདུན་ཕྲག་རྗེས་མར་ཡུ་རོབ་བོད་རིགས་ཚོགས་གཙོ་རྣམས་སླར་ཡང་རྒྱལ་ས་ཕ་རི་སིར་འཛོམ་སྟེ་མི་མང་ལས་འགུལ་དེའི་ཐད་ཚོགས་འདུ་སྐོང་འཆར་ཡོད་འདུག གནས་ཚུལ་དེའི་སྐོར་ཁོང་ལ་བཅར་འདྲི་ཞུས་དོན་ལྟར་ན། ལོ་རྗེས་མའི་གསུམ་བཅུའི་དུས་དྲན་ཐེངས་ང་དྲུག་པ་དང་བསྟུན་ཏེ། རྒྱལ་ས་ཕ་རི་སིར་སྦར་སིལ་དོན་གཅོད་དང་ཡུ་རོབ་སྤྱི་འཐུས་རྣམ་པ་གཉིས་ཀྱི་ལམ་སྟོན་འོག ཡུ་རོབ་བོད་རིགས་ཚོགས་པ་ཁག་གིས་མཉམ་རུབ་ཐོག་གཞི་རྒྱ་ཆེ་བའི་མི་མང་ལས་འགུལ་ཞིག་སྤེལ་འཆར་ཡོད་པ་དངོས་འབྲེལ་ཡིན་པ་དང་། སྐབས་དེའི་ལས་འགུལ་ནང་བོད་པ་གཙོས་བོད་དོན་རྒྱབ་སྐྱོར་བ་ ༡༠༠༠༠ ལྷག་འཛོམ་རྒྱུར་རེ་བ་རྒྱག་གི་ཡོད་པ་མ་ཟད། སྐབས་དེར་རྒྱ་གར་ནང་གི་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་མགོ་ཁྲིད། ཡུ་རོབ་གྲོས་ཚོགས་དང་ཧྥ་རན་སིའི་གཞུང་གི་གནད་དོན་མི་སྣ་གདན་ཞུའི་ཐོག་བོད་དོན་གཏམ་བཤད་གནང་འཇུག་རྒྱུ། དེ་བཞིན་ཡུ་རོབ་སྐད་གྲགས་ཡོད་པའི་གཞས་པ་ཁག་གདན་ཞུ་གནང་སྟེ་བོད་དོན་གླུ་གཞས་གཏོང་རྒྱུ་བཅས་ཀྱི་མཛད་རིམ་ཡོད་ཀྱང་། ཞིབ་ཕྲའི་ལས་རིམ་དང་ལས་འགུལ་ངོ་བོའི་སྐོར་ཚོགས་གཙོ་རྣམ་པ་ཕེབས་ཏེ། གོ་བསྡུར་ལྷུག་པོར་གནང་རྗེས་ཡོངས་བསྒྲགས་ཞུ་རྒྱུ་ཡིན་སྐོར་སོགས་བརྗོད་སོང་། གནས་ཚུལ་མཁོ་གཏོང་བ་དཔལ་ལྡན།
