བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་སྔོན་རྩིས།

ཕྱི་ལོ་ ༢༠༡༧ ནས་ ༢༠༡༨ ལོའི་བར་གྱི་སྔོན་རྩིས་ཡོང་འཆར་བསྡོམས་ཧིན་སྒོར་ ༢༤༠༧༨༡༤༩༧༨ དང་། འགྲོ་འཆར་བསྡོམས་ཧིན་སྒོར་ ༢༤༣༩༨༠༢༩༩༧ ཆགས་འདུག དེ་རིང་ཉིན་རྒྱབ་བོད་མི་མང་སྤྱི་འཐུས་གྲོས་ཚོགས་ནང་། བོད་མིའི་སྒྲིག་འཛུགས་དཔལ་འབྱོར་བཀའ་བློན་ཀརྨ་ཡེ་ཤེས་མཆོག་གིས་རྩིས་ལོ་ ༢༠༡༧ ནས་ ༢༠༡༨ ལོའི་སྔོན་རྩིས་སྙན་སྒྲོན་ཞུས་སོང་། སྔོན་རྩིས་ནང་བཀོད་པ་ལྟར་ན། ཐེངས་འདི་ནི་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་བཀའ་ཤག་སྐབས་བཅོ་ལྔ་བའི་སྔོན་རྩིས་དང་པོ་ཆགས་ཡོད་སྟབས། ལས་འཆར་གལ་ཆེན་མང་པོ་ཞིག་གི་ཐོག་སྔོན་རྩིས་ཞུས་ཡོད་སྐོར་འཁོད་འདུག བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་མ་དངུལ་ཕྱོགས་བསྡོམས་ནང་། ཕྱི་ལོ་ ༢༠༡༦ ཟླ་ ༣ ཚེས་ ༣༡ ཉིན་ཧིན་སྒོར་ས་ཡ་ ༢,༨༠༣་༠༢ ཡོད་ཅིང་། དེ་ནི་ ༢༠༡༥ ཟླ་ ༣ ཚེས་ ༣༡ ཉིན་གྱི་གནས་བབ་ལས་སྒོར་ས་ཡ་ ༣༦༦་༦༩ ཙམ་མང་བ་ཆགས་ཡོད་འདུག བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་གཏན་འཇགས་མ་དངུལ་ནང་། ༢༠༡༦ ཟླ་ ༣ ཚེས་ ༣༡ ཉིན་ཧིན་སྒོར་ས་ཡ་ ༢༥༢་༨༩ ཡོད་པའི་ཐོག འདི་ལོར་གསར་དུ་བྱུང་བ་སྒོར་ས་ཡ་ ༡༠༦༡ བསྣན་ན་ ༢༠༡༧ ཟླ་ ༡ ཚེས་ ༣༡ ཉིན་གྱི་གཏན་འཇགས་མ་དངུལ་སྒོར་ས་ཡ་ ༢༦༣་༥༡ ཆགས་ཡོད་པ་དང་། དེའི་ཐོག་ལྷན་ཁང་སོ་སོའི་གཏན་འཇགས་མ་དངུལ་བསྣན་ཏེ། ད་ལྟ་ཁྱོན་བསྡོམས་ས་ཡ་ ༥༦༨་༨༤ ཆགས་ཡོད་འདུག བཀའ་ཤག་གིས་ ༢༠༡༧ ནས་ ༢༠༡༨ ལོའི་བར་གྲོས་ཚོགས་ལ་སྔོན་རྩིས་འཆར་འབུལ་ཞུས་ནི། ༡༽ རྒྱལ་བའི་སྐུ་ཕྱྭ་དང་བོད་བསྟན་སྲིད་ཞབས་རིམ་ཆེད་ཧིན་སྒོར་ས་ཡ་ ༢༣་༤༠ ། ༢༽ རྒྱལ་སྤྱིའི་འབྲེལ་ལམ་དང་དབུ་མའི་ལམ་གྱི་སྲིད་བྱུས་ངོ་སྤྲོད་སོགས་ལ་སྒོར་ས་ཡ་ ༥༥་༡༩ ། ༣༽ བཙན་བྱོལ་བོད་མིའི་འཚོ་རྟེན་ཡར་རྒྱས་དང་སྡོད་སྒར་ཁག་གི་འཛིན་སྐྱོང་ཐད་སྒོར་ས་ཡ་ ༡༥༠་༤༧ ། ༤༽ བོད་མིའི་རིག་གཞུང་དང་ངོ་བོ་སྲུང་སྐྱོབ་ཐད་སྒོར་ས་ཡ་ ༣༠༧་༨༢ ། ༥༽ བོད་མིའི་འཕྲོད་བསྟན་ཐད་ལ་སྒོར་ཡ་ ༣༡་༧༦ ། ༦ ༽ བུད་མེད་ཀྱི་ནུས་སྟོབས་གོང་སྤེལ་ཐད་ལ་སྒོར་ས་ཡ་ ༢་༣༩ ། ༧༽ ཚོང་ལས་དོན་གཉེར་ཅན་ལ་རམ་འདེགས་ཆེད་སྒོར་ས་ཡ་ ༡༥༣་༠༨ དང་། དངུལ་ཁང་ལྟ་བུའི་དཔལ་འབྱོར་སྡེ་ཚན་གསར་འཛུགས་དང་བོད་མི་སུ་ཚོང་བར་བུན་གཏོང་ཆེད་སྒོར་ས་ཡ་ ༧༩་༤༧ ། ༨༽ ཕྱི་རྒྱལ་ཁག་ནས་ཕེབས་པའི་བོད་རིགས་དང་བླངས་ཞབས་ཞུ་བའི་ཆེད་སྒོར་ས་ཡ་ ༣་༨༠ ། ༩༽ བོད་ཀྱི་སྲིད་ཇུས་ཉམས་ཞིབ་ཁང་གི་ལས་དོན་ཆེད་སྒོར་ས་ཡ་ ༧་༣༡ ། ༡༠༽ བོད་མིའི་སྒྲུག་འཛུགས་ཀྱི་ད་ཡོད་ལས་བྱེད་ཚོར་ཟབ་སྦྱོང་ཆེད་སྒོར་ས་ཡ་ ༢༡་༥༢ ། ༡༡༽ བོད་ནས་གསར་དུ་འབྱོར་བ་ཚོར་སྣེ་ལེན་ཆེད་སྒོར་ས་ཡ་ ༤་༨༠ དང་། ཤེས་རབ་དགའ་ཚལ་སློབ་གླིང་ལ་སྒོར་ས་ཡ་ ༡༡་༢༢ ། ཆབ་སྲིད་བཙོན་ཟུར་དང་འབྲེལ་ཆགས་ལ་སྒོར་ས་ཡ་ ༢་༦༧ ། ལག་ཤེས་དང་ཤེས་ཡོན། འཚོ་སྣོན་བཅས་ལ་སྒོར་ས་ཡ་ ༣༤་༦༥ ། ༡༢༽ བོད་མི་ཉམ་ཐག་ཚོའི་ཆེད་སྒོར་ས་ཡ་ ༩༦་༨༡ ། ༡༣༽ གངས་སྐྱིད་དུ་ལས་ཁུངས་གསར་རྒྱག་དང་ལམ་བཟོའི་ཆེད་སྒོར་ས་ཡ་ ༨༣་༡༠ ། ༡༤༽ སྲི་ཞུ་བ་ཚོར་ལས་ཤག་འཛུགས་སྐྲུན་ཆེད་སྒོར་ས་ཡ་ ༥༢་༨༥ དགོས་སྐོར་འཁོད་འདུག གཞན། བོད་མི་སྤྱི་འཐུས་ལྷན་ཚོགས་ཀྱི་སྔོན་རྩིས་ཚོད་དཔག་འཆར་རྩིས་ཚོགས་ཆུང་གི་སྙན་ཐོའི་ནང་། ཐེངས་འདིར་སྤར་གཅོག་གཙོ་བོར་བཟུང་མེད་ཀྱང་། སྤྱི་འཐུས་ཚོགས་དྲུང་ལས་ཁུངས་ཀྱི་ཟུར་བཀོལ་ཁོངས་ས་འཐུས་རྣམས་ལ་ཟབ་སྦྱོང་གི་ཆ་ཤས་དང་། སྐབས་བཅུ་དྲུག་པའི་སྤྱི་འཐུས་རྣམས་ལ་ཐབས་བྱུས་ཚོགས་འདུའི་དགོས་དངུལ་ཆ་ཚང་། བོད་ཀྱི་སྲིད་བྱུས་ཉམས་ཞིབ་ཁང་གི་དམིགས་བསལ་ཁོངས་འབུམ་རམས་པ་དང་ཉེ་བའི་འབུམ་རམས་པའི་ཚོགས་འདུའི་རྩིས་མགོར་ཕྱི་རྒྱལ་མཁས་དབང་གསུམ་གྱི་ལམ་གྲོན་བཅས་ལས་ཁུངས་སོ་སོས་རེ་ཞུས་ལྟར། ཁྱོན་སྒོར་ ༥,༢༢༠་༢༥༠ གཅོག་ཆ་ཞུས་ཡོད་པ་དང་། ལས་བྱེད་འདེམས་བསྐོ་ལྷན་ཁང་གི་ཚོགས་མི་གཅིག་གི་སྡོད་གནས་རྩིས་ལོའི་ནང་རྡ་ས་ནས་སྦིར་དུ་སྤོ་འཆར་ལ་བརྟེན་རྒྱུན་གྲོན་ཁོངས་འགྲིམ་འགྲུལ་དང་ཕྱོགས་སྐྱོད་ཟུར་ཕོགས་ནང་སྒོར་ ༣༦,༠༠༠ ལྷན་ཁང་གི་རེ་ཞུས་ལྟར་སྤར་ཆ་བྱས་ཡོད། ཅེས་སོགས་འཁོད་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
