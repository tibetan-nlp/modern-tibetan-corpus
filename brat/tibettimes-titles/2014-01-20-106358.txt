ཝུ་ཀི་ཤེས་བྱ་ཀུན་བཏུས་ཕམ་མེད་ཟེར།

དེ་སྔོན་དྲ་དམངས་ཚོས་ཆེས་དགའ་བསུ་བྱེད་ཡུལ་ཝུ་ཀི་པི་ཌི་ཡཱ་(Wikipedia)ཞེས་པའི་ཤེས་བྱ་ཀུན་བཏུས་དྲ་གནས་ལ་ཉེ་བའི་ལོ་འགའི་ནང་གྷོ་གྷལ་རིག་མཛོད་(Google’s Knowledge Graghs)གྱིས་བཙན་འཛུལ་ཐེབས་བཞིན་ཡོད་འདུག རྒྱལ་སྤྱིའི་ཐོག་དྲ་དམངས་ས་ཡ་མང་པོས་དགའ་མོས་བྱེད་པའི་ཤེས་བྱ་ཀུན་བཏུས་ཚིག་མཛོད་ཆེན་མོ་ཝུ་ཀི་པི་ཌི་བཀོལ་སྤྱོད་བྱེད་ཚད་ ༢༠༡༢ ལོའི་ཟླ་ ༡༢ ནས་ ༢༠༡༣ ལོའི་ཟླ་ ༡༢ པའི་བར་ཐེར་འབུམ་ ༢ ཙམ་ཇེ་ཉུང་དུ་ཕྱིན་ཡོད་ཅིང་། ལོ་སྔོན་མ་དག་དང་བསྡུར་ན་བརྒྱ་ཆ་ ༡༠ ཙམ་ཉམས་ཆག་ཕྱིན་ཡོད་འདུག དེ་དག་ལས་དབྱིན་ཡིག་ཐོག་འཚོལ་ཞིབ་བྱེད་ཚད་བརྒྱ་ཆ་ ༡༢ དང་འཇར་མན་སྐད་ཐོག་བརྒྱ་ཆ་ ༡༧། ཉི་ཧོང་སྐད་ཐོག་བརྒྱ་ཆ་ ༩ ཙམ་ཉམས་ཆག་ཕྱིན་ཡོད་འདུག གནས་འཕྲིན་ཆེད་མཁས་པ་ཚོས་གནད་དོན་དེར། ༢༠༡༣ ལོའི་ཟླ་ ༡ པོའི་ནང་གྷོ་གྷལ་རིག་མཛོད་གསར་གཏོད་བྱས་ཡོད་སྟབས། རྒྱལ་སྤྱིའི་སྤྱི་ཚོགས་ཀྱིས་ཧ་ཅང་སྟབས་བདེ་ཡིན་པའི་གྷོ་གྷལ་རིག་མཛོད་བེད་སྤྱོད་གཏོང་གི་ཡོད་སྟབས་ཝུ་ཀི་པི་ཌི་ཡཱ་བཀོལ་སྤྱོད་བྱེད་མཁན་ཉུང་དུ་ཕྱིན་ཚུལ་རྗོད་ཀྱི་ཡོད་ཀྱང་། ཝུ་ཀི་པི་ཌི་ཡཱའི་སློབ་སྟོན་པ་ཇེའི་ཝོ་ཤི་(Jay Walsh)ལགས་ཀྱིས་གནས་ཚུལ་དེར་ངོས་ལེན་མེད་པར། གནས་ཚུལ་དེ་རྩིས་ནོར་དང་བསྐྱར་ཞིབ་ཡག་པོ་མ་བྱས་པའི་གནས་ཚུལ་ཞིག་ཡིན་སྐོར་བརྗོད་འདུག ནེའུ་ཡོག་གཙུག་ལག་སློབ་གྲྭ་ཆེན་མོའི་སློབ་དཔོན་དང་དྲ་ཐོག་བདེ་འཇགས་སྐོར་གྱི་ཆེད་མཁས་པ་ཁཱེ་ཤེར་ཀི་ (ClayShirky)ལགས་ཀྱིས། འཛམ་གླིང་སྤྱི་ཡོངས་ནང་ཝུ་ཀི་པོ་ཌི་ཡཱ་བཀོལ་སྤྱོད་བྱེད་མཁན་ཉུང་དུ་འགྲོ་དགོས་དོན་ནི་ ༢༠༡༣ ལོའི་ཟླ་ ༡ པོའི་ནང་གྷོ་གྷལ་རིག་མཛོད་དྲ་བ་གསར་ཐོན་བྱུང་སྟེ། དེ་སྔ་སྐད་གྲགས་ཡོད་པའི་ཝུ་ཀི་ཤེས་བྱ་ཀུན་བཏུས་གཙོས་པའི་དྲ་གནས་མང་པོ་ཞིག་གི་ཁྲོམ་རྭར་བཙན་འཛུལ་ཐེབས་ཡོད་སྐོར་བརྗོད་འདུག འདས་པའི་ལོ་གཉིས་གསུམ་གོང་ནས་ཝུ་ཀི་ཤེས་བྱ་ཀུན་བཏུས་བེད་སྤྱོད་བྱེད་མཁན་ཉམས་ཆག་འགྲོ་བཞིན་ཡོད་ཀྱང་། ཚིག་མཛོད་དེའི་ནང་གི་རྒྱ་གར་གྱི་སྐད་ཡིག་ཁག་བེད་སྤྱོད་གཏོང་མཁན་ཇེ་མང་དུ་འགྲོ་བཞིན་ཡོད་པ་དང་། ༢༠༡༡ ལོ་དང་བསྡུར་ན་ཧིན་རྡུའི་སྐད་བེད་སྤྱོད་གཏོང་མཁན་བརྒྱ་ཆ་ ༣༤ དང་། བྷང་གྷ་ལི་སྐད་བེད་སྤྱོད་གཏོང་མཁན་བརྒྱ་ཆ་ ༤༦། མཱ་ལ་ཡཱ་ལམ་ (Malayalam)ནང་བརྒྱ་ཆ་ ༣༦། མཱ་ར་ཌི་ ( Marati) ནང་བརྒྱ་ཆ་ ༤། ཏི་ལུ་གྷུ་ ( Telugu)ནང་བརྒྱ་ཆ་ ༡། བཅས་འཕར་ཡོད་འདུག ༢༠༡༢ ལོའི་ཟླ་ ༡༡ པ་ནས་ ༢༠༡༣ ལོའི་ཟླ་ ༡༡ པའི་བར་ཝུ་ཀི་པི་ཌི་ཡཱ་ནང་དབྱིན་ཡིག་ཐོག་རྩོམ་ཡིག་འབྲི་མཁན་དང་རྩོམ་སྒྲིག་གནང་མཁན་བརྒྱ་ཆ་ ༡༥ ཙམ་ཉམས་ཆག་ཕྱིན་ཡོད་ཅིང་། དེ་ནི་ལག་ཐོག་ཁ་པར་ཐོག་ཝུ་ཀི་པི་ཌི་ཡཱ་འཚོལ་ཞིབ་བྱེད་མཁན་ལོ་སྔོན་མ་དང་བསྡུར་ན་བརྒྱ་ཆ་ ༥༡ ཙམ་འཕར་ཡོད་སྟབས། ལག་ཐོག་ཁ་པར་སྟེང་ཡི་གེ་གཏག་པར་སྟབས་བདེ་བོ་མེད་པ་དེས་ཤུགས་རྐྱེན་གང་འཚམ་ཐེབས་ཡོད་པར་ངོས་འཛིན་བྱེད་བཞིན་ཡོད་འདུག དྲ་ཐོག་ཀུན་བཏུས་ཚིག་མཛོད་ཝུ་ཀི་པི་ཌི་ཡཱ་ཞེས་པ་ནི་ ༢༠༠༡ ལོར་གསར་བཟོ་བྱས་ཏེ་འཛམ་གླིང་གི་སྐད་ཡིག་འདྲ་མིན་ ༢༣༧ ཙམ་ནང་ཡོད་ཅིང་། ད་བར་བཀོལ་སྤྱོད་བྱེད་མཁན་ས་ཡ་ ༥༢༣ ཙམ་བྱུང་ཡོད་འདུག གྷོ་གྷན་རིག་མཛོད་དེ་ ༢༠༡༣ ཟླ་ ༡ པོའི་ནང་གསར་དུ་ཐོན་པ་དང་། དེར་གནས་ཚུལ་ཅི་རིགས་འཚོལ་ཞིབ་བྱེད་ཐུབ་པ་མ་ཟད་རྒྱབ་ལྗོངས་གནས་ཚུལ་དང་པར་རིས། འབྲེལ་འཐག་བཅས་འཛོམ་པའི་ཤེས་རིགས་བང་མཛོད་ཅིག་ཡིན་འདུག
