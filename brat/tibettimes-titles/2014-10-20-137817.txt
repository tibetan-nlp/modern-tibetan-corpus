ཨ་རིས་མཁའ་ཐོག་ནས་དྲག་སྤྱོད་པར་འཐབ།

ཁ་སང་ཁ་ཉིན་གཉིས་ལ་ཨ་རིའི་མཁའ་དམག་གིས་སི་རི་ཡའི་དྲག་སྤྱོད་ཤོག་གིས་བཙན་བཟུང་བྱེད་འཆར་ཡོད་པའི་སི་རི་ཡིའི་ཁཱོ་བྷེན་གྲོང་འཁྱེར་བརྟན་སྲུང་བྱས་ཡོད་འདུག   ཟླ་འདིའི་ཚེས་ ༡༨ དང་ ༡༩ ཉིན་ཨ་རིས་སི་རི་ཡའི་ཀར་ཌི་ཤི་ཤོག་ཁག་ (Kurdish group) ལ་རྒྱབ་སྐྱོར་བྱས་ཏེ། སི་རི་ཡའི་གྲོང་རྡལ་ཁཱོ་བྷེན་ (Kobane) བཙན་བཟུང་བྱེད་འཆར་ཡོད་ཡོད་པའི་ཨི་རག་དང་སི་རི་ཡའི་མཉམ་འབྲེལ་དྲག་སྤྱོད་ཤོག་ཁག་ཐོག་མཁའ་ཐོག་ནས་དྲག་རྒོལ་ཐེངས་ ༡༡ ཙམ་བྱས་ཏེ་ཁ་ཆེ་དྲག་སྤྱོད་པ་ ༣༡ བསད་པ་མ་ཟད། དྲག་སྤྱོད་པའི་ས་ཚིགས་ ༢༠ དང་དམག་རྒྱག་རླངས་འཁོར་ ༥། དམག་ཁུལ་ཁང་ ༢ བཅས་གཏོར་བཤིག་བཏང་ཡོད་ཀྱང་། ཁ་སང་ཀྲར་ཀིའི་ (Turkey)སྲིད་འཛིན་ཨེར་ཌོ་གྷན་ (Recep Tayyip Erdogan) ལགས་ཀྱིས་ཨ་རིས་སི་རི་ཡའི་ཀར་ཌི་ཤི་ཤོག་ཁག་ལ་རྒྱབ་སྐྱོར་བྱས་པ་དེར་མ་འདོད་པའི་རྣམ་འགྱུར་བསྟན་ཡོད་འདུག  འདིའི་ཐད་ཨ་རིའི་དམག་ཁུལ་ཁང་གིས། “ཨ་རི་དང་སི་རི་ཡའི་ཀར་ཌི་ཤི་ཤོག་ཁག་གིས་ཨི་རག་དང་སི་རི་ཡའི་མཉམ་འབྲེལ་དྲག་སྤྱོད་ཤོག་ཁག་ལ་མཁའ་རྒོལ་བྱས་པ་དེས། དྲག་སྤྱོད་པ་མང་པོ་ཞིག་ཤི་རྨས་དང་། ཁོ་ཚོའི་གོ་མཚོན་དང་དམག་རྒྱག་རླངས་འཁོར་མང་པོ་ཞིག་གཏོར་བཤིག་བཏང་ཡོད་ཀྱང་། ད་དུང་དྲག་སྤྱོད་པས་སི་རི་ཡའི་གྲོང་རྡལ་ཁཱོ་བྷེན་ནང་འཇབ་རྒོལ་བྱས་ནས་མང་ཚོགས་ལ་དཀའ་ངལ་བཟོ་བཞིན་ཡོད།”ཅེས་བརྗོད་འདུག  ཉེ་ལམ་ཨ་རི་དང་ཨ་རིའི་མཐུན་གྲོགས་རྒྱལ་ཁབ་ཚོས་ཀྲར་ཀི་སྲིད་གཞུང་ལ་གནོན་ཤུགས་སྤྲད་དེ། ཀྲར་ཀི་གཞུང་གིས་སི་རི་ཡའི་ཀར་ཌི་ཤི་ཤོག་ཁག་ལ་རྒྱབ་སྐྱོར་བྱས་ཏེ། ཨི་རག་དང་སི་རི་ཡའི་མཉམ་འབྲེལ་དྲག་སྤྱོད་ཤོག་ཁག་གཏོར་ཕམ་གཏོང་དགོས་པའི་སྐུལ་མ་བྱས་ཡོད་ཀྱང་། ཀྲར་ཀི་གཞུང་གིས་སི་རི་ཡའི་ཀར་ཌི་ཤི་ཤོག་ཁག་གིས་འདས་པའི་ལོ་ ༣༠ ལྷག་རིང་ཀྲར་ཀི་ཡི་ལྷོ་ཕྱོགས་ཁུལ་དུ་དྲག་སྤྱོད་ལས་འགུལ་སྤེལ་ཏེ། འབྲེལ་ཡོད་རྒྱལ་ཁབ་ལ་དཀའ་ངལ་ཆེན་པོ་བཟོས་ཡོད་སྟབས། ཁོ་ཚོར་རྒྱབ་སྐྱོར་བྱ་རྒྱུ་མིན་སྐོར་བསྒྲགས་ཡོད་ཀྱང་། ཨ་རིའི་སྲིད་འཛིན་ཨོ་བྷ་མས་ཀྲར་ཌི་སྲིད་འཛིན་དང་གྲོས་མོལ་བྱས་ཏེ་རྒྱལ་ཁབ་གཉིས་པོས་མཉམ་འབྲེལ་ཐོག་སི་རི་ཡའི་ཀར་ཌི་ཤི་ཤོག་ཁག་ལ་རོགས་རམ་བྱས་ནས་ཁ་ཆེ་དྲག་སྤྱོད་ཤོག་ཁག་ལ་དཔུང་འཇུག་བྱ་རྒྱུའི་ཐག་གཅོད་བྱས་ཡོད་ཀྱང་། ཁ་སང་ཨ་རིས་ཀྲར་ཀི་སྲིད་གཞུང་ལ་གྲོས་མོལ་ཙམ་ཡང་མ་བྱས་པར་སི་རི་ཡའི་དྲག་སྤྱོད་ཤོག་ཁག་ལ་མཁའ་རྒོལ་བྱས་ཡོད་སྟབས། ཀྲར་ཀི་གཞུང་གིས་ཨ་རིའི་བྱ་སྤྱོད་དེར་སྐྱོན་བརྗོད་བྱས་འདུག  ཁ་སང་ཀྲར་ཀིའི་སྲིད་འཛིན་ཨེར་ཌོ་གྷན་ (Recep Tayyip Erdogan) ལགས་ཀྱིས་གསར་འགྱུར་བརྒྱུད་ལམ་ཁག་ཏུ་བརྗོད་པར། “ང་ཚོས་སི་རི་ཡའི་ཀར་ཌི་ཤི་ཤོག་ཁག་ལ་ཨ་རིའི་རོགས་སྐྱོར་གོ་མཚོན་ལེན་འཇུག་རྒྱུ་མིན། གང་ཡིན་ཞེ་ན། ཤོག་ཁག་འདི་ནི་ཀྲར་ཀི་ཡི་བརྟན་ལྷིང་མེད་པ་བཟོ་མཁན་འཇིགས་སྐུལ་རིང་ལུགས་པའི་ ཚོགས་པ་ (Kurdistan Workers’ Party) ཞེས་པ་དང་གཅིག་མཚུངས་ཆགས་ཡོད།”ཅེས་བརྗོད་འདུག ཆབ་སྲིད་དབྱེ་ཞིབ་པ་ཚོས། སི་རི་ཡའི་འཇིགས་སྐུལ་རིང་ལུགས་པ་མེད་པ་བཟོ་དགོས་ན། ཀྲར་ཀི་དང་མཉམ་ལས་བྱེད་རྒྱུ་གལ་འགངས་ཆེན་པོ་ཆགས་ཡོད་སྟབས། གལ་ཏེ་ཀྲར་ཀི་ཡིས་མཉམ་བཞུགས་མ་བྱས་ཚེ། དྲག་སྤྱོད་ཤོག་ཁག་རྩ་མེད་གཏོང་ཐུབ་རྒྱུ་ཁག་པོ་ཆགས་ཡོད་སྐོར་བརྗོད་འདུག ད་བར་ཨ་རིའི་མཁའ་དམག་གིས་སི་རི་ཡའི་གྲོང་རྡལ་ཁཱོ་བྷེན་ (Kobane)བཙན་བཟུང་བྱེད་རྩིས་ཡོད་པའི་ཁ་ཆེ་དྲག་སྤྱོད་ཤོག་ཁག་ལ་མཁའ་རྒོལ་ཐེངས་ ༡༣༥ ཙམ་བྱས་ཏེ། ཁོ་ཚོའི་དམག་རྒྱག་རླངས་འཁོར་ ༣༠ ཙམ་དང་། དྲག་སྤྱོད་པ་ ༢༥༧ བསད་ཡོད་པ་མ་ཟད། དམག་ཁུལ་ཁང་ ༢༤ ལྷག་གཏོར་རྒོལ་བཏང་ཡོད་འདུག  གསར་འགོད་པ། དོན་གྲུབ།
