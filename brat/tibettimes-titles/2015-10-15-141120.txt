གཞུང་ཚབ་ཁག་ ༤༢ ལ་བོད་དོན་སྙན་ཞུ་ཕུལ།

  ཁ་སང་བོད་པའི་ཁྱིམ་ཚང་བཞི་ཡིས་བོད་དོན་ཞུ་གཏུག་གི་ལས་འགུལ་ཐེངས་ ༤༦ པ་སྤེལ་ཏེ་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་གཙོས་ཨ་ཧྥི་རི་ཁའི་གཞུང་ཚབ་ཁག་ ༤༢ ལ་བོད་དོན་ཞུ་གཏུག་གི་སྙན་ཞུ་ཕུལ་འདུག འདི་གར་གནས་ཚུལ་མཁོ་གཏོང་གནང་བ་ལྟར་ན། ཁ་སང་ཕྱི་ཚེས་ ༡༤ ཉིན་སུད་སི་ཇི་ནི་ཝ་མཉམ་འབྲེལ་རྒྱ་ཚོགས་ཀྱི་མདུན་ཐང་དུ། བོད་པའི་ཁྱིམ་ཚང་བཞི་ཡི་བོད་དོན་ཞུ་གཏུག་གི་ལས་འགུལ་ཐེངས་༤༦ པ་སྤེལ་ཡོད་པ་དང་། ཐེངས་འདིའི་ལས་འགུལ་ནང་ཡུ་རོབ་སྤྱི་ཟུར་ཚེ་རིང་རྡོ་རྗེ་ལགས་ཀྱིས་དབུས་རིན་ཆེན་ལགས་དང་སྟག་ཚང་བསྟན་འཛིན་རིག་འཛིན་ལགས་བཅས་མཉམ་ཞུགས་གནང་སྟེ། མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་གཙོས་སུད་སིའི་ནང་ཡོད་པའི་ཨ་ཧྥི་རི་ཁ་ཁུལ་གྱི་གཞུང་ཚབ་ཁག་ ༤༢ ལ་བོད་དོན་ཞུ་གཏུག་གི་སྙན་ཞུ་ཕུལ་ཡོད་འདུག   ཨ་ཧྥི་རི་ཁའི་གཞུང་ཚབ་ཁག་ལ་དམིགས་བསལ་གྱིས་བོད་ཀྱི་འཚེ་མེད་ཞི་བའི་ལས་འགུལ་ཁག་ལ་རྒྱབ་སྐྱོར་དང་། དབུས་བོད་མིའི་སྒྲིག་འཛུགས་དང་རྒྱ་ནག་གཞུང་བར་ཞི་མོལ་ཡོང་ཐབས་ཐད་གླེང་སློང་བྱ་དགོས་པའི་སྐོར་གྱི་འབོད་སྐུལ་སྙན་ཞུ་དང་། དབུས་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱིས་བཏོན་པའི་བོད་ནང་གི་འགྲོ་བ་མིའི་ཐོབ་ཐང་གི་སྙན་ཐོ་སོགས་ཕུལ་ཡོད་པ་མ་ཟད། ནམ་རྒྱུན་ལྟར་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཆེས་མཐོའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལྷན་ཁང་དང་། མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་དྲུང་ཆེ་བྷན་ཁེ་མོན་ལགས་ཀྱི་མཚན་ཐོག་སོགས་ལ་བོད་དོན་ཞུ་གཏུག་གི་སྙན་ཞུ་ཕུལ་ཡོད་འདུག   དེ་ཡང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་གྱི་དྲུང་ཆེ་བྷན་ཁེ་མོན་ལགས་ཀྱི་མཚན་ཐོག་ལ་བཏང་བའི་སྙན་ཞུའི་ནང་། མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱིས་བོད་ཐོག་བཞག་པའི་གྲོས་ཆོད་སྔ་ཕྱི་ཁག་ལ་བདག་སྤྲོད་གནང་དགོས་པ་དང་། བོད་ནང་ཞི་བདེ་དང་རང་དབང་སླར་གསོ་གནང་དགོས་པ། ༧གོང་ས་མཆོག་ཆ་རྐྱེན་གང་ཡང་མེད་པའི་ཐོག་བོད་ནང་ཕྱིར་ཕེབས་འཇུག་དགོས་པ། རྒྱ་བོད་འབྲེལ་མོལ་བསྐྱར་མཐུད་བྱ་དགོས་པ། ད་ལོར་སྤྲུལ་སྐུ་བསྟན་འཛིན་བདེ་ལེགས་རིན་པོ་ཆེ་མཆོག་གློ་བུར་བཙོན་ནང་སྐུ་བཀྲོངས་བ་སོགས་བོད་ཀྱི་ཆབ་སྲིད་བཙོན་པ་རྣམས་ཀྱི་ཚེ་སྲོག་ལ་སྲུང་སྐྱོབ་གནང་དགོས་པའི་སྐོར་སོགས་བཀོད་ཡོད་འདུག   རྒྱལ་ཚོགས་མདུན་བོད་དོན་ཞུ་གཏུག་གི་ལས་འགུལ་འདི་ནི་ཕྱི་ལོ་༢༠༡༢ ཟླ་༡༢ ཚེས་༡༠ ཉིན་ཁྱིམ་ཚང་གཉིས་ཀྱིས་དབུ་བཙུགས་ཤིང་། རིམ་གྱིས་བཀྲ་བསྟན་ཚང་པ་སངས་ལགས་དང་། ངག་དབང་ཚང་ངག་དབང་དཔལ་ལྡན། རྒྱལ་བསྟན་ཚང་བཀྲ་ཤིས་ལྷ་མོ་བཅས་མཉམ་ཞུགས་གནང་སྟེ་ད་ཆར་ཁྱིམ་ཚང་བཞི་ཡི་ལས་འགུལ་དུ་གྱུར་ཡོད་འདུག ལས་འགུལ་འདི་སུད་སི་ཇི་ནི་ཝར་རྟེན་གཞི་བྱས་པའི་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ལས་ཁུངས་མདུན་དང་། མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་འགྲོ་བ་མིའི་ཐོབ་ཐང་མདུན་ཐང་བཅས་སུ་བོད་ཕྱི་ནང་གཉིས་སུ་རང་ལུས་མེར་སྲེག་གནང་མཁན་རྒྱལ་གཅེས་དཔའ་བོ་དཔའ་མོ་ཚོའི་གྲངས་དང་མཉམ་པའི་བོད་ཀྱི་རྒྱལ་དར་སྒྲེང་འཛུགས་དང་། བོད་ནང་གི་དཀའ་སྡུག་གི་གནས་བབ་བཀོད་པའི་ཡིག་ཆ་ཁག་འགྲེམས་སྤེལ། སྐུ་ལུས་མེར་སྲེག་གནང་མཁན་རྒྱལ་གཅེས་དཔའ་བོ་དཔའ་མོ་ཚོའི་སྐུ་པར་འགྲེམས་འཇོག   རང་ལུས་མེར་བསྲེགས་གནང་དགོས་པའི་རྒྱབ་ལྗོངས་འགྲེལ་བརྗོད་དང་། འབོད་སྒྲའམ་དགོས་འདུན་བཀོད་པའི་ཡིག་ཆ་ཁག་དང་འཕྲེད་དར། སྐུ་ལུས་མེར་བསྲེགས་གནང་བའི་ཐད་༧གོང་ས་མཆོག་དང་དབུས་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་ལངས་ཕྱོགས་དང་དགོངས་ཚུལ་ག་རེ་ཡིན་པའི་སྐོར། དེ་བཞིན་བོད་ནང་གི་འགྲོ་བ་མིའི་ཐོབ་ཐང་གི་སྙན་ཐོ་དང་ཆོས་ལུགས་དད་མོས་རང་དབང་གི་སྙན་ཐོ།   གསར་འགོད་གསལ་བསྒྲགས། གཞན་ཡང་། ༧གོང་ས་མཆོག་གིས་དམིགས་བསལ་རྒྱ་རིགས་ཚོར་བསྩལ་བའི་བཀའ་སློབ་ཡིག་ཆ་སོགས་རྒྱ་ཡིག ཨིན་ཡིག་དང་ཧྥ་རན་ཆི་(FRENCH) བཅས་ཀྱི་ནང་ཕབ་བསྒྱུར་ཞུས་ཏེ། ཕྱི་རྒྱལ་མི་རིགས་དང་རྒྱལ་ཚོགས་ཀྱི་ལས་བྱེད། ལྷག་དོན་རྒྱ་རིགས་ཡུལ་སྐོར་བ་མང་པོར་འགྲེམ་སྤེལ་བྱེད་ཀྱི་ཡོད་པ་དང་། ལས་འགུལ་སྤེལ་ཐེངས་རེ་རེ་བཞིན་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་ལས་ཁུངས་དང་། མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ཆེས་མཐོའི་ལས་ཁུངས་གཉིས་ལ་སླར་ཡང་དོན་ཚན་བརྒྱད་ཅན་གྱི་དགོས་འདུན་སྙན་ཞུ་དང་། བོད་མིའི་སྒྲིག་འཛུགས་ཕྱི་དྲིལ་ལས་ཁུངས་ཀྱིས་བཏོན་པའི་བོད་ནང་གི་འགྲོ་བ་མིའི་ཐོབ་ཐང་གི་ཟླ་རེའི་སྙན་ཐོ་སོགས་ཕུལ་བཞིན་ཡོད་འདུག
