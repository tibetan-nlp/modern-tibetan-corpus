བླ་བྲང་འཇིཊ་མེད་ལ་ལོ་ ༥ ཁྲིམས་ཆད།

རྒྱ་གཞུང་གིས་རྒྱལ་ཁབ་ཁ་བྲལ་གཏོང་བར་ངན་སྐུལ་བྱས་པའི་ཉེས་མིང་འོག་བླ་བྲང་དགོན་པའི་གྲྭ་འཇིགས་མེད་རྒྱ་མཚོ་ལགས་སུ་ལོ་ལྔའི་ཁྲིམས་ཐག་བཅད་འདུག བོད་ནས་ཁུངས་ལྡན་གནས་ཚུལ་འབྱོར་གསལ་ལྟར་ན། རྒྱ་ནག་གཞུང་གིས་དེ་སྔོན་ཀན་སུའུ་ཞིང་ཆེན་ཀན་ལྷོ་ཁུལ་བླ་བྲང་བཀྲ་ཤིས་འཁྱིལ་དགོན་པའི་བདག་གཉེར་ལྷན་ཁང་གི་འགན་འཛིན་གཞོན་པའི་ལས་འགན་འཁུར་མྱོང་མཁན་གྲྭ་འཇིགས་མེད་རྒྱ་མཚོ་ལགས། ཕྱི་ལོ་ ༢༠༡༡ ཟླ་ ༨ པར་འཛིན་བཟུང་གིས་ད་བར་བཀག་ཉར་བྱས་ཡོད་པར་མ་ཟད། ཟླ་འདིའི་ཚེས་ ༥ ཉིན་ཀན་སུའུ་ཞིང་ཆེན་ལན་ཀྲུའུ་གྲོང་ཁྱེར་དུ་ཡོད་པའི་ཁྲིམས་ཁང་ཞིག་གིས། ཁོང་ལ་རྒྱལ་ཁབ་ཁ་བྲལ་གཏོང་བའི་ངན་སྐུལ་བྱས་ཡོད་པའི་ཉེས་མིང་འོག་ལོ་ ༥ དུས་བཀག་བཙོན་འཇུག་གི་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག ཉེ་ལམ་ཁྲིམས་ཆད་གསལ་བསྒྲགས་གནང་རྗེས་ཁོང་བཙོན་ཁང་གང་དུ་བཙོན་འཇུག་བྱས་ཡོད་མེད་སོགས་ཀྱི་གནས་ཚུལ་ཅི་ཡང་མ་ཤེས་པར་ལྷག་ཡོད་འདུག གྲྭ་འཇིགས་མེད་རྒྱ་མཚོ་འམ་ཡོངས་གྲགས་སུ་འཇིགས་མེད་མགོ་རིལ་འབོད་པ་ཁོང་ནི། ཀན་ལྷོ་བོད་རིགས་རང་སྐྱོང་ཁུལ་བླ་བྲང་བཀྲ་ཤིས་འཁྱིལ་དགོན་པའི་ཉེ་འདབ་ཕུག་སྡེ་ཞེས་པ་ནས་ཡིན་པ་དང་། ད་ལྟ་དགུང་ལོ་ ༤༨ ཡིན་འདུག ཕྱི་ལོ་ ༢༠༠༦ ཟླ་ ༡ ཁོང་བོད་ནས་རྒྱ་གར་ལྷོ་ཕྱོགས་ཨ་མྲར་ཝ་ཏིར་༧གོང་ས་མཆོག་གི་དུས་འཁོར་དབང་ཆེན་ཞུ་རུ་ཡོང་ཡོད་པ་དང་། ཕྱིར་ལོག་རྗེས་ས་གནས་ཉེན་རྟོག་པས་ཟླ་ ༤ ལྷག་གི་རིང་བཀག་ཉར་བྱས་ཡོད་འདུག ༢༠༠༨ ལོའི་ཕྱི་ཟླ་ ༣ ཚེས་ ༢༢ ཉིན་ཁོང་སྐབས་དེའི་བསང་ཆུ་རྫོང་གི་ཞི་རྒོལ་ནང་ཞུགས་མེད་ཀྱང་། རྒྱ་གཞུང་གིས་ཁོང་གློ་བུར་འཛིན་བཟུང་གིས་གཅར་རྡུང་འདྲི་གཅོད་སོགས་བྱས་ཏེ། གཟུགས་པོའི་གནས་སྟངས་ཛ་དྲག་གནས་སུ་གྱར་རྗེས་སྨན་ཁང་དུ་སྨན་བཅོས་དང་མཉམ་ཀློད་གྲོལ་བཏང་ཡོད་འདུག ༢༠༠༨ ཟླ་ ༡༡ ཚེས་ ༤ ཉིན་ཁོང་གིས་བརྙན་རིས་བརྒྱུད་རྒྱ་གཞུང་གིས་བོད་ནང་གི་དྲག་གནོན་དབང་བསྒྱུར་གྱི་སྲིད་ཇུས་ལག་བསྟར་བྱེད་བཞིན་པའི་དངོས་ཡོད་གནས་ཚུལ་ཕྱིར་བསྒྲགས་བྱས་སྟབས། དམག་མི་དང་ཉེན་རྟོག་པ་ ༧༠ ཙམ་འབྱོར་ཏེ་འཛིན་བཟུང་གིས་ཟླ་ ༦ ལྷག་གི་རིང་ཆབ་སྲིད་སློབ་གསོ་སྤེལ་ཡོད་པ་དང་། ༢༠༡༡ ཟླ་ ༨ ཚེས་ ༢༠ ཉིན་ཀན་ལྷོ་གཙོད་གྲོང་ཁྱེར་དུ་ཉེན་རྟོག་པ་ ༥༠ ལྷག་དང་། དྲག་ཆས་དམག་མི་བཅས་ཀྱིས་ཁོང་འཛིན་བཟུང་བྱས་རྗེས་ཁོང་གི་གྲ་ཤག་སྔོག་བཤེར་བྱས་ཏེ། ༧གོང་ས་མཆོག་གི་སྐུར་པར་དང་། འོད་སྡེར་སོགས་ཅ་དངོས་ཁག གློག་ཀླད་བཅས་གཞུང་བཞེས་བཏང་ཡོད་པ་དང་། དུས་དེ་ནས་བཟུང་དུས་ཡུན་གང་འཚམ་རིང་གར་ཡོད་ཅི་བྱུང་གི་གནས་ཚུལ་མ་ཤེས་པར་ལྷག་འདུག ༢༠༡༢ ཟླ་ ༦ པའི་ཟླ་མཇུག་ཁོང་ལ་རྒྱལ་ཁབ་ཁ་བྲལ་ལ་ངན་སྐུལ་བྱས་པའི་ཉེས་མིང་འོག་ཁྲིམས་ཆད་གཅོད་རྒྱུ་ཡིན་པ་དང་། གཟུགས་པོའི་གནས་སྟངས་སྐྱོ་བོ་ཡིན་སྟབས། ལན་ཀྲུའུ་གྲོང་ཁྱེར་གྱི་སྨན་ཁང་ཞིག་ཏུ་སྨན་བཅོས་བྱེད་ཀྱི་ཡོད་པའི་གནས་ཚུལ་ཙམ་ལས་གོ་ཐུབ་མེད་ཀྱང་། ཉེ་ལམ་རྒྱ་གཞུང་གིས་ནང་མི་ཉེ་འབྲེལ་དང་ཁྲིམས་རྩོད་པ་སོགས་ཅི་ཡང་མེད་པའི་ཐོག་ཁོང་ལ་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག གསར་འགོད་པ། དོན་གོ
