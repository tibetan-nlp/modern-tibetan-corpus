རྩ་ཁྲིམས་ལ་བརྩི་འཇོག་བྱེད་དགོས།

  རྒྱ་ནག་གི་ཕྱི་སྲིད་བློན་ཆེན་ཝང་ཡེས་ཐའེ་ཝན་མཚོ་འགག་གི་སྲིད་འཛིན་གསར་བ་ཚའེ་ཡེན་ཝན་གྱིས་རྒྱ་ནག་རྒྱལ་ཁབ་གཅིག་གི་ལམ་ལུགས་འོག་འདུ་བའི་ཐའེ་ཝན་མཚོ་འགག་གི་རྩ་ཁྲིམས་ལ་བརྩ་འཇོག་དགོས་ལུགས་བཤད་འདུག   རྒྱ་ནག་གི་ཕྱི་སྲིད་བློན་ཆེན་ཝང་ཡེས། ཐའེ་ཝན་གྱི་དཔོན་རིགས་སུ་ཞིག་གིས་ཐའེ་ཝན་མཚོ་འགག་གི་སྲིད་དབང་བཟུང་བ་དང་། ཁོའམ་མོས་ཐའེ་རྒྱ་གཉིས་ཀྱི་འབྲེལ་ལམ་འཛིན་སྐྱོང་བྱེད་མིན། ཞི་བའི་སྒོ་ནས་ཐའེ་རྒྱ་གཉིས་ཀྱི་འབྲེལ་ལམ་རྒྱུན་འཁྱོང་བྱེད་མིན། ཡང་ན། ཐའེ་རྒྱ་གཉིས་ཀྱི་རྩ་བའི་ཆབ་སྲིད་འབྲེལ་ལམ་ལ་འགལ་འཛོལ་བཟོ་མིན་གང་ལྟར་ཡང་། ང་ཚོས་དོ་སྣང་དང་ཡིད་གཟབ་དགོས་ས་གཙོ་བོ་དེ་རྒྱ་ནག་རྒྱལ་ཁབ་གཅིག་གི་ལམ་ལུགས་ཡིན། ངའི་རེ་བ་ལ། འདི་ལོའི་ཕྱི་ཟླ་ ༥ བའི་ནང་ལྕམ་ཚའེ་ཨན་ལྕམ་ཚའེ་ཡེན་ཝིན་གྱིས་ཐའེ་ཝན་མཚོ་འགག་གི་སྲིད་དབང་མ་བཟུང་སྔོན་ལ། ཁོ་མོས་ཞི་བའི་སྒོ་ནས་ཐའེ་རྒྱ་གཉིས་ཀྱི་འབྲེལ་ལམ་ཡར་རྒྱས་དང་། རྒྱ་ནག་སྐམ་ས་ཆེན་མོ་དང་ཐའེ་ཝན་མཚོ་འགག་གཉིས་དོན་དངོས་སུ་རྒྱ་ནག་རྒྱལ་ཁབ་གཅིག་གི་ལམ་ལུགས་འོག་འདུ་བའི་ཐའེ་ཝན་མཚོ་འགག་གི་རྩ་ཁྲིམས་ལ་མོས་མཐུན་བྱེད་པའི་རྣམ་འགྱུར་སྟོན་དགོས། རྒྱལ་ཁབ་ཞིག་གི་རྩ་ཁྲིམས་གཞིར་བཟུང་ཐོག་འོས་འདེམས་བྱེད་པའི་གང་ཟག་ཅིག་གིས་རང་གི་རྒྱལ་ཁབ་ཀྱི་རྩ་ཁྲིམས་རྡོག་རོལ་གཏོང་རྩིས་བྱེད་པ་ནི་དངོས་གནས་བློ་ངོར་མི་ཤོང་བ་ཞིག་ཡིན་ལུགས་སོགས་བརྗོད་འདུག   ལྕམ་ཚའེ་ཡེན་ཝན་གྱིས། ཁོ་མོས་ཐའེ་རྒྱ་གཉིས་ཀྱི་ཞི་བའི་འབྲེལ་ལམ་རྒྱུན་འཁྱོང་དང་། རྒྱ་ནག་གཞུང་འབྲེལ་བརྒྱུད་ལམ་ཁག་གིས་ཁོ་མོའི་ད་ལྟའི་ཐའེ་རྒྱ་གཉིས་ཀྱི་གནས་བབ་རྒྱུན་འཁྱོང་བྱེད་པའི་ཁག་ཐག་ལ་དོ་སྣང་དགོས་སྐོར་བརྗོད་ཡོད་པ་རེད།   གནས་ཚུལ་ལྟར་ན། ཁ་སང་ཕྱི་ཚེས་ ༢༥ ཉིན་ཨ་རིའི་རྒྱལ་ས་ཝ་ཞིན་ཊོན་དུ་རྟེན་གཞི་བྱེད་པའི་རྒྱལ་སྤྱིའི་འཐབ་ཇུས་ཞིབ་འཇུག་ལྟེ་གནས་ཁང་ (CSIS) དུ། སྔོན་ཆད་ཐའེ་རྒྱ་གཉིས་ཀྱི་ལས་དོན་ཁང་གི་འགན་འཛིན་དང་དེང་སྐབས་རྒྱ་ནག་གི་ཕྱི་སྲིད་བློན་ཆེན་ཝང་ཡེས། ཐའེ་ཝན་གྱི་སྲིད་འཛིན་འོས་བསྡུ་དེ་བཞིན་རྒྱུན་ལྡན་གྱི་ཆབ་སྲིད་འཕེལ་རིམ་ཙམ་ཡིན་པ་ལས། དེ་ལ་ངོ་མཚར་དགོས་དོན་གང་ཡང་མེད་ལུགས་དང་། རྒྱ་ནག་གི་མངའ་ཁོངས་སུ་འདུ་བའི་ཐའེ་ཝན་མཚོ་འགག་གི་སྲིད་འཛིན་ལ་སུ་ཞིག་བསྡད་ཀྱང་། དེ་ལ་ང་ཚོས་དོ་སྣང་དེ་ཙམ་བྱེད་ཀྱི་མེད་ཚུལ། དེ་བཞིན། ཐའེ་ཝན་མི་མང་ཕྱོགས་གཉིས་ཀྱི་བཟང་ཕྱོགས་ཀྱི་འབྲེལ་ལམ་དང་། རྒྱ་ནག་ནང་ཡུལ་སྐོར་སྤྲོ་འཆམ་དང་དཔལ་འབྱོར་ཁེ་ལས་གཉེར་ཏེ་ཞི་བདེ་ཞོད་འཇགས་ཀྱི་འཚོ་བ་རོལ་འདོད་ཡོད་སྐོར་སོགས་བཤད་ཅིང་། ཐའེ་ཝན་དམངས་གཙོ་ཡར་ཐོན་ཚོགས་པའི་མགྲིན་ཚབ་པ་རུན་ཁྲོ་ཞོང་ (Ruan Chao-hsiung) ཟེར་བ་ཞིག་གིས། རྒྱ་ནག་གི་ཕྱི་སྲིད་བློན་ཆེན་ཝང་ཡེའི་སྐད་ཆ་དེ་དག་ལྕམ་ཚའེ་ཡེན་ཝན་གྱིས་ཐའེ་ཝན་མཚོ་འགག་གི་རྩ་ཁྲིམས་སྒྲོམ་གཞིའི་འོག་ཏུ་ད་ལྟའི་ཐའེ་རྒྱ་གཉིས་ཀྱི་འབྲེལ་ལམ་གནས་བབ་རྒྱུན་འཁྱོང་བྱེད་པའི་ལངས་ཕྱོགས་ལ་དམིགས་ནས་བཤད་པ་ཞིག་ཡིན་སྐོར་དང་། ང་ཚོས་རྒྱ་ནག་སྐམ་ས་ཆེན་མོ་དང་ཐའེ་ཝན་མཚོ་འགག་གཉིས་བར་སྔར་བས་ཞི་འགྲིག་གི་འབྲེལ་མོལ་ཡོང་བའི་རེ་བ་བྱེད་ཀྱི་ཡོད་སྐོར་སོགས་བརྗོད་འདུག   གསར་འགོད་པ། པདྨ་རྒྱལ།
