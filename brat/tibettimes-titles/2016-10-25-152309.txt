རྒྱ་ནག་ནང་འབར་གས་དོན་རྐྱེན་བྱུང་།

ཧྲན་ཞི་ཞིང་ཆེན་ཡས་ལིན་གྲོང་ཁྱེར་ཧྥོའུ་ཀོ་རྫོང་ཁོངས་སུ་འབར་གས་དོན་རྐྱེན་བྱུང་སྟེ་ཉུང་མཐར་མི་གྲངས་ ༡༠ རྐྱེན་འདས་དང་། མི་ ༡༥༠ ལྷག་ལ་རྨས་སྐྱོན་བཟོས་འདུག ཁ་སང་པེ་ཅིན་གྱི་ཕྱི་དྲོའི་ཆུ་ཚོད་ ༢ ཙམ་གྱི་ཐོག རྒྱ་ནག་ཧྲན་ཞི་ཞིང་ཆེན་ཡས་ལིན་གྲོང་ཁྱེར་ཧྥོའུ་ཀོ་རྫོང་སྡོད་མང་གསར་པའི་གྲོང་ཚོ་རུ། གནས་སྐབས་ཀྱི་འཛུགས་སྐྲུན་ཁང་ཞིག་ཏུ་འབད་གས་དོན་རྐྱེན་བྱུང་སྟེ། ཉེ་འཁོར་གྱི་སྡོད་མང་གསར་པའི་སྨན་ཁང་དང་བྱིས་སྙོར་ཁང་། སྡོད་ཁང་སོགས་ལ་གནོད་སྐྱོན་ཚབས་ཆེན་བཟོས་ཡོད་པ་མ་ཟད། ཉུང་མཐར་མི་གྲངས་ ༡༠ རྐྱེན་འདས་དང་། མི་ ༡༥༠ ལྷག་ལ་རྨས་སྐྱོན་ཕོག་ཡོད་འདུག ས་གནས་མི་མང་གིས་ཐེངས་འདིའི་འབར་གས་དོན་རྐྱེན་དེ་འཛུགས་སྐྲུན་ཁང་དེའི་འོག་ཁང་གི་འབར་རྫས་གསོག་ཉར་དང་འབྲེལ་བ་ཡོད་ལུགས་དང་། ས་གནས་དེར་རྡོ་སོལ་ཉོ་འཚོང་དང་། མི་ཁ་ཤས་ཀྱིས་ནག་ཚོང་ཁྲོམ་ར་ནས་འབར་རྫས་རིགས་ཉོ་འཚོང་བྱེད་ཀྱི་ཡོད་སྐོར་བརྗོད་འདུག་ལ། དོན་རྐྱེན་བྱུང་རྗེས་ཧྥོའུ་ཀོ་རྫོང་སྲིད་གཞུང་གིས། རྡོ་སོལ་ཁེ་ལས་ཀྱིས་གསོག་ཉར་འབར་རྫས་རིགས་ལ་འབར་གས་བྱུང་ཉེན་ཡོད་སྟབས། ཉེ་འཁོར་གྱི་རྡོ་སོལ་ཁེ་ལས་ཁག་ཅིག་ལ་ལས་མཚམས་འཇོག་དགོས་པའི་སྐུལ་མ་བྱས་ཡོད་འདུག གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ལོ་ངོ་ ༡༠ ཡི་ཡར་སྔོན་དུ་ཧྥོའུ་ཀོ་རྫོང་ལོ་གོ་ཁྲན་གྲོང་ཚོའི་ནང་འབར་གས་དོན་རྐྱེན་བྱུང་སྟེ། ཁང་པ་ ༢༣ ལ་གཏོར་སྐྱོན་ཐེབས་པ་མ་ཟད། མི་ ༡༢ རྐྱེན་འདས་དང་མི་ ༡༨ ལ་རྨས་སྐྱོན་ཕོག་ཡོད་པ་དང་། རྟོག་ཞིབ་བྱས་རྗེས་ས་གནས་སྡོད་མང་གིས་འབར་རྫས་གསོག་ཉར་བྱས་རྐྱེན་འབར་གས་བྱུང་ཡོད་སྐོར་བརྗོད་ཡོད་པ་དང་། འདས་ལོར་ཐེན་ཅིན་འཛུགས་སྐྲུན་ཁུལ་དུ་འབར་གས་དོན་རྐྱེན་ཚབས་ཆེན་བྱུང་སྐབས། ཉུང་མཐར་མི་གྲངས་ ༡༠༠ ལྷག་རྐྱེན་འདས་དང་། མི་ ༤༠༠ ཡས་མས་ལ་རྨས་སྐྱོན་བཟོས་ཡོད་པ་རེད། གསར་འགོད་པ། པདྨ་མཚོ།
