ཧོང་ཀོང་རང་བཙན་ལ་རེ་བ་མ་བྱེད།

ཁ་སང་ཧོང་ཀོང་གྲོང་ཁྱེར་སྤྱི་ཁྱབ་པ་ལུང་ཁྲུན་ཡེན་གྱིས་ཧོང་ཀོང་མི་མང་གིས་ལོངས་སུ་སྤྱོད་བཞིན་པའི་རང་སྐྱོང་དེ་སྐམ་ས་ཆེན་མོས་གཏན་འབེབས་བྱས་པའི་རྩ་ཁྲིམས་དང་བསྟུན་དགོས་པ་ལས་རང་སྣང་གང་དྲན་བྱས་མི་ཆོག་ཟེར། ཁ་སང་ཧོང་ཀོང་གྲོང་ཁྱེར་སྤྱི་ཁྱབ་པ་ལུང་ཁྲུན་ཡེན་གྱིས། ཧོང་ཀོང་དམིགས་བསལ་སྲིད་འཛིན་ས་ཁུལ་གྱི་ལོ་འཁོར་སྲིད་གྲོས་ཚོགས་ཆེན་ཐོག རྒྱ་ནག་གི་སྲིད་ཇུས་དང་དཔལ་འབྱོར་འཕེལ་རྒྱས་ལ་བརྟེན་ཏེ། ཧོང་ཀོང་མི་མང་ལ་ཁེ་ཕན་འཐོབ་པའི་གོ་སྐབས་མང་པོ་ཡོད་སྟབས། ཧོང་ཀོང་མི་མང་གིས་ཧོང་ཀོང་གྲོང་ཁྱེར་དེ་མེས་རྒྱལ་དང་ནམ་ཡང་ཁ་འབྲལ་ཐབས་མེད་པར་ངོས་འཛིན་བྱེད་དགོས། ཧོང་ཀོང་གྲོང་ཁྱེར་དེ་རྒྱ་ནག་ལ་ཁ་འབྲལ་བའམ་རང་བཙན་སོགས་གང་ཡང་བྱེད་ཐབས་མེད། ཧོང་ཀོང་མི་མང་གིས་ལོངས་སུ་སྤྱོད་བཞིན་པའི་ཚད་མཐོའི་རང་སྐྱོང་དེ་གཞི་རྩའི་རྩ་ཁྲིམས་གཏན་འབེབས་དང་བསྟུན་དགོས། རང་སྐྱོང་ཟེར་བ་དེ་གང་ཟག་རེ་གཉིས་ཀྱི་གཏན་གཏན་འབེབས་བྱེད་པའམ་རང་སྣང་གང་དྲན་བྱེད་ཆོག་པ་ཞིག་མིན་པ་མ་ཟད། དེ་ལ་ལག་ལེན་བསྟར་ཕྱོགས་གཞན་པ་དང་བསྒྱུར་བཅོས་སོགས་གང་ཡང་བྱེད་ཆོག་གི་མ་རེད། དེང་སྐབས་རྒྱལ་སྤྱིའི་འབྲེལ་ལམ་རྙོག་འཛིང་ཆེན་པོ་ཆགས་ཡོད་སྟབས། ཧོང་ཀོང་མི་མང་ལ་མེས་རྒྱལ་གྱི་བདེ་འཇགས་དང་མངའ་ཁོངས་སྲུང་སྐྱོབ། རྒྱལ་མང་ཆིག་སྒྲིལ་བཅས་ཀྱི་འོས་འགན་ཡོད། ཅེས་སོགས་བཤད་འདུག ཕྱི་ལོ་ ༡༩༩༧ ལོར་དབྱིན་ཇིའི་གཞུང་གིས་རྒྱལ་ཁབ་གཅིག་ལ་ལམ་ལུགས་གཉིས་ཀྱི་སྲིད་བྱུས་ཐོག ཧོང་ཀོང་ལ་ཚད་མཐོའི་རང་སྲིད་རང་སྐྱོང་གི་བདག་དབང་སྤྲོད་སྤྱིའི་ཁས་ལེན་འོག དབྱིན་ཇིའི་འོག་ནས་ཧོང་ཀོང་རྒྱ་ནག་ལ་རྩིས་སྤྲོད་བྱས་ཡོད་ཀྱང་། འདས་པའི་ལོ་ངོ་ ༡༠ བརྒྱུད་རིམ་ཁྲོད་རྒྱ་ནག་གཞུང་གིས་ཐོག་མའི་གྲོས་དོན་ལ་རྡོག་རོལ་བཏང་སྟེ། ཧོང་ཀོང་ལ་རྒྱ་ནག་གཞུང་གིས་སྣོན་ཤུགས་སྣ་ཚོགས་སྤྲད་དེ། ཧོང་ཀོང་མི་མང་གི་རང་དབང་དང་དམངས་གཙོ་སོགས་དཀའ་ཁག་གི་གནས་སུ་གྱར་ཡོད། འབྱུང་འགྱུར་ཕྱི་ཟླ་ ༣ པའི་ནང་པེ་ཅིན་ཕྱོགས་གཏོགས་ཀྱི་མི་སྣ་ ༡༢༠༠ གྲུབ་པའི་འོས་བསྡུའི་ཚོགས་ཆུང་བརྒྱུད་དེ། ཧོང་ཀོང་དམིགས་བསལ་སྲིད་འཛིན་ས་ཁུལ་གྱི་སྤྱི་ཁྱབ་པ་འདེམས་བསྐོ་བྱ་རྒྱུ་ཡིན་པ་རེད། གསར་འགོད་པ། པདྨ་རྒྱལ།
