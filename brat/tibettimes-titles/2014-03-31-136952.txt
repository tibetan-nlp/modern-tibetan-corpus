འབའ་བ་ཕུན་ཚོཊ་དབང་རྒྱལ་སྐུ་གྲོངས་འདུག

དེ་རིང་ཞོགས་པར་བོད་མི་གསར་བརྗེ་བ་འབའ་བ་ཕུན་ཚོགས་དབང་རྒྱལ་མཆོག་རྒྱ་ནག་པེ་ཅིན་སྨན་ཁང་དུ་སྐུ་གྲོངས་ཡོད་འདུག དེ་རིང་ཕྱི་ཚེས་ ༣༠ ཉིན་གྱི་པེ་ཅིན་གྱི་ཞོགས་པའི་ཆུ་ཚོད་ ༧།༡༠ ཡས་མས་སུ་བོད་མི་གསར་བརྗེ་བ་འབའ་བ་ཕུན་ཚོགས་དབང་རྒྱལ་མཆོག རྒྱ་ནག་པེ་ཅིན་སྨན་ཁང་དུ་སྐུ་གྲོངས་ཡོད་པའི་གནས་ཚུལ་འབྱོར་ཡོད་པ་དང་། ད་ལྟ་དགུང་ལོ་ ༩༢ ཡིན་འདུག   འབའ་བ་ཕུན་ཚོགས་དབང་རྒྱལ་མཆོག་ནི། ཕྱི་ལོ་ ༡༩༢༢ ལོའི་བོད་ཟླ་ ༡༠ ཚེས་ ༢༥ ཉིན་སྟེ་དགའ་ལྡན་ལྔ་མཆོད་ཉིན་ཁམས་འབའ་ཐང་གྲོང་རྡལ་དུ་སྐུ་འཁྲུངས་པ་དང་། རང་ལོ་བདུན་གྱི་ཐོག་རྒྱ་གཞུང་གིས་བཙུགས་པའི་འབའ་ཐང་དེང་རབས་སློབ་གྲྭའི་ནང་ཞུགས་ཏེ་སློབ་གཉེར་གནང་། ༡༩༣༦ ལོར་ཅང་ཅེ་ཧྲིའི་ཆ་སྲིད་སློབ་གྲྭའི་ཡན་ལག་བོད་སོག་ཨུ་ཡོན་ལྷན་ཁང་གིས་གོ་མིན་ཏང་སྲིད་གཞུང་ནང་ལས་ཞབས་ཞུ་མཁན་བོད་དང་གྲངས་ཉུང་མི་རིགས་གཞན་དག་གི་བུ་ཕྲུག་སྦྱོང་བརྡར་ཞུ་སོའི་སློབ་གྲྭར་སློབ་གཉེར་གནང་ཡོད། ནན་ཅིན་དང་ཁྲོན་ཆིན་བཅས་སུ་སློབ་གཉེར་གནང་སྐབས་རྩོམ་སྒྲིག་པ་སོགས་ཀྱི་ཕྱག་ལས་གནང་བ་དང་སློབ་གྲྭའི་ནང་གསང་བའི་དམར་ཤོག་ཚོགས་པ་གསར་འཛུགས་དང་། ༡༩༣༩ ལོར་བོད་རིགས་དམར་ཤོག་ཚོགས་པ་གསར་འཛུགས་བཅས་ཀྱི་ཕྱག་ལས་གནང་བ་དང་། ༡༩༥༡ ལོའི་ཕྱི་ཟླ་ ༤ བར་ང་ཕོད་ངག་དབང་འཇིགས་མེད་དང་མཉམ་དུ་པེ་ཅིན་དུ་ཕྱིན་སྟེ། སྐབས་དེའི་བཀའ་ཤག་དང་པེ་ཅིན་སྲིད་གཞུང་གི་འབྲེལ་མོལ་དང་༼གྲོས་མཐུན་དོན་ཚན་བཅུ་བདུན༽འཇོག་སྐབས་ཀྱི་སྐད་བསྒྱུར་བའི་ཕྱག་ལས་གནང་ཡོད། ༡༩༥༣ ལོར་ཀྲུང་དབྱང་དམངས་ཨུའི་ཆབ་སྲིད་དང་རིག་གནས་སྡེ་ཁག་གི་འགན་འཛིན་དང་མི་རིགས་དཔེ་བསྐྲུན་ཁང་གི་འགན་འཛིན་གཞོན་པ་བཅས་ཀྱི་ཕྱག་ལས་གནང་བར་མ་ཟད། པཎ་ཆེན་སྐུ་ཕྲེང་བཅུ་བ་མཆོག་དང་མཉམ་རུབ་ཐོག་བོད་ཀྱི་ཆོས་དང་རིག་གཞུང་ཁོར་ཡུག་དང་བཅས་པའི་སྲུང་སྐྱོབ་ཐད་ཕྱག་ལས་རབ་དང་རིམ་པ་གནང་བ་དང་། ༸གོང་ས་མཆོག་གི་སྐུ་ཚབ་རིམ་པར་སློབ་སྟོན་དང་རྒྱབ་སྐྱོར་བཅས་བོད་ཀྱི་ལས་དོན་ཁག་གི་ཐོག་རླབས་ཆེན་གྱི་ཕྱག་ལས་གནང་ཡོད། འབའ་བ་ཕུན་ཚོགས་དབང་རྒྱལ་མཆོག་གིས། ༡༩༥༡ ལོ་ནས་ ༡༩༥༨ ལོའི་བར་བོད་རིགས་མགོ་ཁྲིད་ལས་བྱེད་པའི་ཕྱག་ལས་གནང་ཡོད་པ་དང་། ཁོང་ནི་ལོ་རབས་སུམ་ཅུ་བའི་དུས་མཇུག་ཙམ་དུ་དེང་རབས་ཀྱི་ཤེས་ཡོན་ཐོབ་ཟིན་ཅིང་། མི་རིགས་རིང་ལུགས་པ་ངར་མ་ཞིག་ཡིན་འདུག་ལ། ཁོང་གི་མི་ཚེའི་དེ་བོད་གཞུང་གི་མངའ་ཞབས་དང་། ཁམས། ཨ་མདོ་བཅས་ཀྱི་བོད་རིགས་ཡོངས་ལ་ཁྱབ་ལ། བོད་མི་རང་གི་འཛིན་སྐྱོང་བྱེད་པའི་སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་བོད་ལྗོངས་ཤིག་བསྐྲུན་ཆེད་འབད་འཐབ་བྱེད་རྒྱུའི་དོན་ལ་ལེགས་སྐྱེས་སུ་ཕུལ། ཁོང་གིས་དམིགས་ཡུལ་འདི་སྒྲུབ་ཆེད་རང་སྒེར་ལ་ཉེན་ཁ་ཧ་ཅང་ཆེན་པོ་ཡོད་པའི་གནས་སྟངས་འོག་ངལ་བ་འཛེམས་མེད་ཀྱིས་ཐོག་མར་རང་ཉིད་ཀྱིས་བཙུགས་པའི་བོད་རིགས་དམར་ཤོག་ཚོགས་པ་དང་། དེ་ནས་རྒྱ་ནག་དམར་ཤོག་ཚོགས་པ་བརྒྱུད་འབད་བརྩོན་བྱས་ཡོད། མཐའ་མར་ཁོང་དང་ཁོང་གི་བློ་མཐུན་ཚོ་ཕམ་ཉེས་བྱུང་སྟེ། པེ་ཅིན་བྱང་ཕྱོགས་ཆིན་ཁྲེན་བཙོན་ཁང་ཨང་དང་པོའི་ནང་ལོ་ ༡༨ བསྐྱལ་དགོས་བྱུང་། ༡༩༧༨ ལོར་བཙོན་གྲོལ་ཡོད་པ་དང་ལོ་གཉིས་ཀྱི་རྗེས་ཆབ་སྲིད་ཀྱི་དབང་ཆ་བསྐྱར་གསོ་བྱུང་རྗེས།  ཁོང་གིས་མུ་མཐུད་རྒྱ་ནག་ནང་མི་རིགས་ཀྱི་སྲིད་བྱུས་སྐོར་ལ་སྐད་ཆ་བཤད་པ་དང་རྩོམ་ཡིག་བྲིས་ཏེ། རྒྱ་ནག་དམར་ཤོག་ཚོགས་པའི་ནང་གི་བོད་ཀྱི་ས་ཁུལ་ཁག་ཏུ་རྒྱ་ནག་གི་སྲིད་བྱུས་འཛིན་ཕྱོགས་ཐད་དག་ཐེར་དཔྱད་ཞིབ་བྱེད་མཁན་ཞིག་ཏུ་གྱུར་ཡོད། འབའ་བ་ཕུན་ཚོགས་དབང་རྒྱལ་མཆོག་གིས་ད་ལྟའི་བར་ཕྱག་ལས་ཀྱི་ཞོར་དུ། ༼རྩོད་སྒྲུབ་རིག་པ་གསར་དཔྱད༽དང་༼ཟླ་བའི་གོ་ལའི་སླ་གཟུགས་ཀྱི་རྣམ་བཤད༽། ༼རང་བྱུང་རྩོད་སྒྲུབ་རིག་པ་གསར་དཔྱད༽། ༼རང་རྒྱལ་གྱི་མི་རིགས་གནད་དོན་དང་མི་རིགས་ལས་དོན་སྐོར་གྱི་ཕྱིར་རྟོག༽ བཅས་བརྩམས་ཆོས་ཁག་ཅིག་རྩོམ་གནང་མཛད་ཡོད།  གསར་འགོད་པ། དོན་གོ
