གཞོན་ནུས་སློབ་མ་ ༡༦ ལ་གཟེངས་རྟཊ་ཕུལ།

བོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་ནས་འདི་ལོའི་རྒྱ་གར་རྒྱལ་ཡོངས་འཛིན་རིམ་བཅུ་གཉིས་ཀྱི་ཡིག་རྒྱུགས་ཁྲོད་བོད་ཡིག་ལ་བརྒྱ་ཆ་ ༩༩ ལེན་མཁན་གྱི་སློབ་མ་ ༡༦ ལ་བཀའ་དྲིན་རྗེས་དྲན་གྱི་བྱ་དགའ་ཧིན་སྒོར་ ༢༠༠༠ རེ་དང་ཕྱག་འཁྱེར་བཅས་གནང་འདུག བོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་ཀྱི་འགན་འཛིན་ལྷན་ཚོགས་ཐེངས་༤༥ པའི་ཐོག བོད་རིགས་གཞོན་སྐྱེས་རྣམས་ནས་རང་མི་རིགས་ཀྱི་སྐད་ཡིག་ལ་འབད་པ་ལྷོད་མེད་བྱེད་པར་སྐུལ་འདེད་གཏོང་ཆེད། རྒྱ་གར་རྒྱལ་ཡོངས་འཛིན་རིམ་བཅུ་གཉིས་ཀྱི་ཡིག་རྒྱུགས་ནང་བོད་ཡིག་རྩེ་ཕུད་ལོན་མཁན་སློབ་མ་གཅིག་ལ་བཀའ་དྲིན་རྗེས་དྲན་གྱི་བྱ་དགའ་ཧིན་སྒོར་༢༠༠༠༠ ཐམ་པ་དང་ཕྱག་འཁྱེར་འབུལ་བའི་གྲོས་ཆོད་བྱུང་བ་ལྟར། དེ་རིང་དབུས་གཞོན་ནུའི་ཚོགས་ཁང་དུ་འདི་ལོའི་རྒྱལ་ཡོངས་ཡིག་རྒྱུགས་ཁྲོད་བོད་ཡིག་བརྒྱ་ཆ་ ༩༩ ལེན་མཁན་ཁྱོན་སློབ་ཕྲུག་ ༡༦ བྱུང་ཡོད་པ་ཚང་མར་ཧིན་སྒོར་ ༢༠༠༠ རེ་དང་ལག་འཁྱེར་བཅས་གནང་ཡོད་པ་དང་། སློབ་གྲྭ་ཁག་ཚོགས་བཞིན་པའི་སྐབས་ཡིན་པས་སློབ་ཕྲུག་ཁག་ཅིག་དངོས་སུ་བྱ་དགའ་ལེན་དུ་ཡོང་ཐུབ་མེད་པ་རེད། ད་ལོར་བོད་ཡིག་བརྒྱ་ཆ་ ༩༩ ལེན་མཁན་གྱི་སློབ་མ་བཅུ་དྲུག་ནི། ཧ་སྤུར་ཉིན་སློབ་ཀྱི་སློབ་མ་རྟ་མགྲིན་རྡོ་རྗེ་དང་བསྟན་འཛིན་ཀུན་དགའ། མ་སུ་རི་གཏན་སློབ་ཀྱི་སློབ་མ་བསྟན་འཛིན་ནོར་བུ་དང་ཕོ་ཀ་བཀྲིས། བསྟན་འཛིན་དོན་འགྲུབ། བཀྲིས་རྒྱ་མཚོ། བདེ་སྐྱིད་ཀུན་སྒྲོལ། བསྟན་འཛིན་འཆི་མེད། ནོར་མཚོ། རྟ་མགྲིན་ཚེ་བརྟན། ཚེ་རིང་སྒྲོལ་མ། གསང་བདག་རྡོ་རྗེ། སེ་ལ་ཀུའི་བོད་ཁྱིམ་གྱི་སློབ་མ་པདྨ་ནོར་བུ་དང་ངག་དབང་སྒྲོལ་མ། བསྟན་འཛིན་སྒྲོལ་དཀར། དོན་གྲུབ་གླིང་སློབ་གྲྭའི་སློབ་མ་ཉི་ཟླ་བཅས་ཡིན་འདུག གཟེངས་རྟགས་འདི་ནི་ཕྱི་ལོ་ ༢༠༡༥ ལོར་༧གོང་ས་མཆོག་དགུང་གྲངས་བརྒྱད་ཅུར་ཕེབས་པར་དམིགས་བསལ་གྱིས་བཀའ་དྲིན་རྗེས་དྲན་ཞུ་ཆེད་དང་། བོད་སྐད་ཡིག་ཉེན་འཕྲང་ཅན་དུ་གྱུར་བའི་དུས་སྐབས་འདིར་བོད་རིགས་སློབ་མ་ཚོས་སོ་སོའི་སྐད་ཡིག་ལ་དུང་ཞེན་སྔར་ལྷག་གིས་སློབ་སྦྱོང་ཧུར་ཐག་བྱེད་པར་སེམས་ཤུགས་སྤར་ཆེད་ཐོག་མར་མགོ་འཛུགས་གནང་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༡༥ ལོའི་གཟེངས་རྟགས་དང་པོ་དེ་བརྒྱ་ཆ་ ༩༨ ལེན་མཁན་སེ་ལ་ཀུའི་བོད་ཁྱིམ་གྱི་སློབ་མ་བདེ་འཛོམས་ལགས་སུ་ཐོབ། ཕྱི་ལོ་ ༢༠༡༦ ལོར་མོན་གྷོ་གཏན་སློབ་ཀྱི་སློབ་མ་བདེ་ཆེན་བཟང་མོ་དང་བསྟན་འཛིན་མགོན་པོ་གཉིས་ནས་བརྒྱ་ཆ་ ༩༨ རེ་བླངས་ཏེ་ཁོང་གཉིས་ལ་གཟེངས་རྟགས་ཕུལ་ཡོད་པ་དང་། ལོ་རེ་བཞིན་བོད་ཡིག་གི་བརྒྱ་ཆ་མཐོ་བོ་ལེན་མཁན་མང་དུ་འགྲོ་བཞིན་ཡོད་པ་དང་བསྟུན་རིམ་གྱིས་བྱ་དགའི་དངུལ་ཡང་འཕར་སྣོན་གནང་འཆར་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
