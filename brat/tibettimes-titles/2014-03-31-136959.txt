རང་སྲེག་གནང་མཁན་བཙུན་མ་སྒྲོལ་མ་རེད།

གཟའ་སྤེན་པའི་ཉིན་འབའ་ཐང་དུ་རང་ལུས་མེར་བསྲེཊ་གནང་མཁན་བཙུན་མ་ཁོང་གི་མཚན་ལ་སྒྲོལ་མ་ཞུ་བཞིན་ཡོད་པ་དང་། ད་ལྟ་རང་ལོ་ ༣༡ ཡིན་འདུག མདོ་སྟོད་སྤྱི་འཐུས་འབའ་བ་སྐལ་བཟང་རྒྱལ་མཚན་ལགས་ཀྱིས་འདི་གར་གནས་ཚུལ་ངོ་སྤྲོད་གནང་བ་ལྟར་ན། ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༩ ཉིན་གྱི་བོད་ནང་གི་ཕྱི་དྲོའི་ཆུ་ཚོད་ ༣།༠༠ ཡས་མས་སུ། འབའ་ཐང་རྫོང་འབའ་ཆོས་སྡེ་དགོན་པའི་ཉེ་འགྲམ་དུ་འབའ་ཐང་སྤྱི་སྲུང་སྒང་ཞོག་སྡེ་བའི་རང་ལོ་ ༣༡ ཡིན་པའི་བཙུན་མ་སྒྲོལ་མ་ལཊ་ཀྱིས། རང་ལུས་མེར་བསྲེཊ་ཀྱིས་རྒྱ་གཞུང་ལ་ངོ་རྒོལ་སྐད་འབོད་གནང་ཡོད་པ་དང་། དེ་མ་ཐག་དགོན་པར་བསྐོར་བ་རྒྱག་མཁན་གྱི་བོད་མི་ཚོས་ཁོ་མོའི་ལུས་སྟེང་གི་མེ་བསད་དེ་འབའ་ཐང་རྫོང་མི་མང་སྨན་ཁང་དུ་འཁྱེར་ཏེ། ད་ལྟའི་ཆར་བཙུན་མ་ཁོང་སྲོག་ཉེན་ལས་ཐར་ཡོད་པའི་གནས་ཚུལ་ཐོན་ཡོད་འདུག གནས་ཚུལ་མཁོ་སྤྲོད་པས་བརྗོད་པ་ལྟར་ན་བཙུན་མ་སྒྲོལ་མ་ལགས་ནི། འབའ་ཐང་རྫོང་འབའ་ཐང་སྤྱི་སྲུང་སྒང་ཞོག་གྲོང་སྡེ་ནས་ཡིན་པ་དང་། ཕ་མིང་ལ་ཉི་མ་དང་ཨ་མ་མིང་ལ་དབང་མོ་ཞུ་བཞིན་ཡོད་འདུག ཨ་མ་ལོ་གཅིག་ཙམ་གྱི་སྔོན་འདས་གྲོངས་སུ་ཕྱིན་ཡོད་འདུག ཁོང་མོར་སྤུན་བཙུན་མ་སྤུན་མཆེད་གཉིས་ཡོད་པ་ལས་ཆེ་བ་ཁོང་དང་ཆུང་བའི་མིང་ལ་ཨ་ཆུབ་ཞུ་བཞིན་ཡོད་པ་དང་། ད་ལྟ་རང་ལོ་ ༢༤ ཡིན་འདུག རང་ལུས་མེར་བསྲེགས་མ་གནང་གོང་ཁོང་འབའ་ཐང་རྫོང་རི་སྟོད་རྐ་མགོར་ཕྱག་འཚལ་བཞིན་བསྡད་ཡོད་པ་དང་། ཁ་སང་རང་ལུས་མེར་བསྲེཊ་ཀྱི་གནས་ཚུལ་བྱུང་རྗེས་། རྒྱ་གཞུང་གིས་ཉེ་ལམ་ཁོང་དང་མཉམ་དུ་སྡོད་མཁན་གྱི་བཙུན་མ་གསུམ་འཛིན་བཟུང་གིས་གར་ཡོད་མ་ཤེས་པར་ལྷག་ཡོད་འདུག ཁོང་ཚོའི་ཁྲོད་ཀྱི་ཨ་ནེ་གཅིག་དེ་སྔོན་རྒྱ་གར་ནས་ཕྱིར་བོད་ནང་ལོག་མཁན་ཞིག་ཡིན་འདུག དེ་སྔོན་གྱི་ལོ་རྒྱུས་ལྟར་ན་ཁོང་མོ་འཚར་ལོངས་བྱུང་སའི་སྤྱི་སྲུང་སྒང་ཞོག་སྡེ་བ་དེ་འབའ་ཐང་རྫོང་གི་ཁོངས་གཏོགས་ཡིན་ཡང་། ཕྱིས་སུ་རྒྱ་ནག་གི་ས་ཆ་བགོ་སྐབས་སྨར་ཁམས་ཀྱི་ཁོངས་སུ་བཅུག་ཡོད་འདུག ད་ཆར་རྒྱ་གཞུང་གིས་ས་གནས་སུ་དམག་མི་འགྲེམས་འཇོག་གིས་དམིགས་བསལ་དམ་དྲག་བྱེད་ཀྱིན་མེད་ཀྱང་། ས་གནས་ཀྱི་དྲ་རྒྱ་ཁག་ལ་ལྟ་མི་ཐུབ་པར་གྱུར་ཡོད་པ་དང་ཁ་པར་སོགས་བརྒྱུད་ལམ་ཁག་ལ་སྟངས་འཛིན་ནན་མོ་བྱེད་ཀྱི་ཡོད་འདུག གསར་འགོད་པ། དོན་གོ
