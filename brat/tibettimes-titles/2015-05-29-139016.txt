རྒྱ་ནག་གི་སློབ་མ་ ༨༠༠༠ ཕྱིར་འབུད་བཏང་།

ཨ་རིའི་ནང་དུ་ཡོད་པའི་རྒྱ་ནག་གི་སློབ་མ་ ༨༠༠༠ རིག་གཞུང་ཞིབ་འཇུག་གི་སྤྱོད་ལམ་མ་གཙང་བ་སོགས་ཀྱིས་སློབ་གྲྭ་ནས་ཕྱིར་འབུད་བཏང་ཡོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༦ ཉིན་ཨ་རིའི་ཧཱ་རིན་(Huaren Education)ཤེས་ཡོན་ཞིབ་འཇུག་ལྟེ་གནས་ཁང་གིས། (༢༠༡༥ ལོའི་ཨ་རིའི་རྒྱ་ནག་གི་སློབ་མའི་གནས་ཚུལ་སྐོར་གྱི་རྒྱབ་ཤ་དཀར་པོའི་དེབ)ཞེས་པ་ཞིག་ཕྱིར་བསྒྲགས་བྱས་པ་ལྟར་ན། ཕྱི་ལོ་ ༢༠༡༤ ལོར་ཨ་རིའི་ནང་ཡོད་པའི་རྒྱ་ནག་གི་སློབ་མ་ ༨༠༠༠ སློབ་གྲྭ་ནས་ཕྱིར་འབུད་བཏང་ཡོད་པ་དང་། དེའི་ནང་ནས་བརྒྱ་ཆའི་ ༢༢.༩༨ རིག་གཞུང་ཞིབ་འཇུག་གི་སྤྱོད་ལམ་མ་གཙང་བ་དང་། བརྒྱ་ཆའི་ ༥༧.༥༦ རིག་གཞུང་གི་སྣང་ཚུལ་ཞན་དྲགས་པའི་རྐྱེན་གྱིས་ཡིན་ལ། ཕྱིར་འབུད་བཏང་བའི་སློབ་མ་དེ་དག་ལས་བརྒྱ་ཆའི་ ༨༨.༧ ནི་ལོ་ན་ ༡༨ ནས་ ༢༥ བར་ཡིན་ཞིང་། བརྒྱ་ཆའི་ ༧ ལྷག་དངོས་གཞིའི་སློབ་ཆེན་པ་དང་འབུམ་རམས་པ་རྐྱང་རྐྱང་ཡིན་འདུག ཤེས་ཡོན་ཞིབ་འཇུག་ལྟེ་གནས་ཁང་དེས་འདས་ལོར་བཏོན་པའི་རྒྱབ་ཤ་དཀར་པོའི་དེབ་ཀྱི་ནང་། རིག་གཞུང་ཞིབ་འཇུག་གི་སྤྱོད་ལམ་མ་གཙང་བ་སྟེ། ལས་བྱ་ཕན་ཚུན་བཤུ་བ་དང་རྒྱུགས་ཚབ་སྤྲོད་པ། རྒྱུགས་འབྲས་རྫུན་བཟོ་དང་། རྒྱུགས་འབྲས་རེའུ་མིག་ལ་བཟོ་བཅོས་རྒྱག་པའི་གནད་དོན་རྣམས་རྒྱ་ནག་གི་སློབ་མའི་ནང་རྒྱུན་དུ་བྱུང་གི་ཡོད་པ་མ་ཟད། རྒྱའི་སློབ་མ་མང་པོ་སློབ་གྲྭ་ཡར་ཚོགས་སྐབས་ཐེངས་གཅིག་སླེབ་པ་ཙམ་ལས་སློབ་གྲྭ་ནང་ཡོང་གི་མེད་ཀྱང་། ཐབས་ལམ་སྣ་ཚོགས་སྤྱད་དེ་རྒྱུགས་འཕྲོད་ཐབས་ཀྱི་ཡོད་སྐོར་དང་། ཡང་རྒྱ་ནག་གི་འབྱོར་ལྡན་པའི་བུ་དང་བུ་མོས་རྒྱུན་དུ་གཞན་ལ་སྒོར་མོ་སྤྲད་དེ་ལས་བྱ་འབྲི་རུ་འཇུག་གི་ཡོད་པ་མ་ཟད། རྒྱུགས་སྤྲོད་སྐབས་ཀྱང་གཞན་གྱིས་ཚབ་བྱེད་ཀྱི་ཡོད་སྐོར་བརྗོད་འདུག དེ་ཡང་ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༡ ཉིན་ཨ་རིའི་མངའ་སྡེ་པེན་སོ་ཝྭ་ནི་ཡ་(Pennsylvania)ལྷོ་ཁུལ་གྱི་མཉམ་འབྲེལ་ ཞིབ་བཤེར་ཁང་གིས། མངའ་སྡེ་དེའི་ཕི་ཊི་སི་བྷར་(Pittsburgh)ས་ཁུལ་ནང་དུ། རྒྱ་ནག་གི་མི་སེར་ཡིན་པའི་མི་ ༡༥ ཡིས་ཕྱི་སྐྱོད་ལག་ཁྱེར་རྫུན་མ་བཟོས་པ་དང་ལྐོག་དངུལ་སྤྲད་དེ། ཨ་རིའི་སློབ་ཆེན་ཁག་གི་ནང་འཛུལ་ཞུགས་བྱས་པའི་དོགས་ཡོད་ཉེས་དོན་ ༣༥ ཡར་ཞུ་བྱས་ཡོད་པ་དང་། དོགས་ཡོད་ཉེས་ཅན་དེ་དག་གི་ནང་ནས་མི་ ༡༢ ཀྱི་མིང་ཕྱིར་བསྒྲགས་བྱས་ཡོད་པའི་ཁྲོད། མི་ ༡༠ ཨ་རིའི་ནང་གནས་སྡོད་བྱས་ཡོད་པ་དང་། མི་ ༢ རྒྱ་ནག་ནང་ཡོད་འདུག ད་དུང་མི་ ༣ གྱི་མིང་དང་ཐོབ་ཐང་ཕྱིར་བསྒྲགས་བྱས་མེད་པ་རེད། དོགས་ཡོད་ཉེས་ཅན་དེ་ཚོས་ཕྱི་སྐྱོད་ལག་ཁྱེར་རྫུན་བཟོ་དང་མྱུར་འཕྲིན་གྱིས་མགོ་སྐོར་བཏང་བ་སོགས་ཀྱི་ནག་ཉེས་འོག ཁག་ཅིག་ལ་ལོ་ངོ་ ༢༠ ཡི་བཙོན་འཇུག་ཁྲིམས་ཆད་དང་། ཨ་སྒོར་ཁྲི་ ༢༥ ཡི་ཉེས་ཆད་ཕོག་སྲིད་སྐོར་བསྒྲགས་འདུག ལྷོ་ཁུལ་གྱི་མཉམ་འབྲེལ་ཞིབ་བཤེར་ཁང་དེའི་ཁྲིམས་ལུགས་རིག་པའི་ཆེད་མཁས་པ་ལྕམ་ཌཱ་ཝེ་ཌི་ཧི་ཁི་ཊོན་གྱིས། དོགས་ཡོད་ཉེས་ཅན་དེ་དག་ལ་ཨ་རིའི་རྒྱལ་ཡོངས་ཀྱི་སློབ་ཆེན་ཁག་གི་སློབ་འཇུག་བརྡ་ཐོ་འབྱོར་ཏེ། ཁོ་ཚོར་ཨ་རིའི་ནང་འཛུལ་བའི་སློབ་མའི་ཐོངས་མཆན་རག་ཡོད་ཀྱང་། གལ་སྲིད་ཁོ་ཚོས་མགོ་སྐོར་གཏོང་བའི་ཐབས་ལམ་ལ་བརྟེན་ནས་སློབ་འཇུག་བརྡ་ཐོ་དེ་རག་པ་ཡིན་ན། དོན་དངོས་ཐོག་ཁོ་ཚོ་ཨ་རིའི་སློབ་ཆེན་པ་ཞིག་ཏུ་གྱུར་ཐབས་མེད་ལ། ཁོ་ཚོའི་ཐོངས་མཆན་ཡང་མགོ་སྐོར་གྱི་བྱེད་ཐབས་ལ་བརྟེན་ནས་རག་པ་རེད། གནད་དོན་འདིའི་ཐོག་ནས་ཨ་རིས་ནུས་པ་ཡོད་ཚད་བཏོན་ནས་ཕྱི་སྐྱོད་ལག་ཁྱེར་དང་ཐོངས་མཆན་གྱི་འགྲོ་ལུགས་ཚུལ་མཐུན་ཡོང་བར་སྲུང་སྐྱོབ་དགོས་པ་མ་ཟད། ཨ་རིའི་མཐོ་རིམ་གྱི་ཤེས་ཡོན་མ་ལག་ལ་སྲུང་སྐྱོབ་དགོས་པའི་གལ་གནད་རང་བཞིན་མཚོན་གྱི་ཡོད་སྐོར་བརྗོད་འདུག ཡང་རྒྱབ་ཤ་དཀར་པོ་དེའི་ནང་ད་ལྟ་ཨ་རིའི་ནང་རྒྱ་ནག་གི་སློབ་མ་ཁྲི་ ༥༠ ཙམ་ཡོད་པ་དང་། ཁོ་ཚོ་ལས་མང་ཆེ་བ་ཨ་རིའི་སྐད་གྲགས་ཅན་གྱི་སློབ་ཆེན་མང་པོ་ཡོད་པའི་རྒྱལ་ས་ཝ་ཤིང་ཊོན་དང་ཁ་ལི་ཧྥོར་ནི་ཡ་(California)། ཧྥི་ལོ་རི་ཌཱ་(Florida)སོགས་ཀྱི་ནང་ཡོད་པ་དང་། ཨ་རི་ནི་རྒྱ་ནག་གི་སློབ་མ་ཡོང་ས་གཙོ་ཤོས་ཤིག་ཡིན་ཡང་། རྒྱའི་སློབ་མ་མང་ཆེ་བའི་ཆེད་སྦྱོང་སློབ་ཚན་ནི་ཁྱིམ་མི་དང་གྲོགས་པོ་སོགས་ཀྱི་རེ་འདུན་ལྟར་འདེམ་གྱི་ཡོད་པ་ལས། སློབ་མའི་ནུས་པར་བསམ་བློ་གཏོང་གི་མེད་སྟབས། སློབ་སྦྱོང་ཐོག་དཀའ་ངལ་འཕྲད་ཀྱི་ཡོད་སྐོར་བརྗོད་ཡོད་པ་རེད། ཞིབ་འཇུག་བྱས་པ་ལྟར་ན། རྒྱ་ནག་ནང་དེང་རབས་འཕྲུལ་རིགས་སྣ་ཚོགས་ལ་བརྟེན་ནས་རྒྱུགས་སྤྲོད་སྐབས་ཕབ་བཤུ་བྱེད་མཁན་ཧ་ཅང་མང་པོ་ཡོད་སྟབས། ཉེ་བའི་ལོ་ཤས་ནས་བཟུང་རྒྱ་གཞུང་གིས་རྒྱུགས་སྤྲོད་ཀྱི་སྒྲིག་ལམ་ལ་དོ་ནན་ཆེ་རུ་བཏང་ཡོད་པ་མ་ཟད། སློབ་གྲྭ་ཆེ་འབྲིང་གི་སློབ་མ་མང་པོ་རྒྱུགས་སྤྲོད་སྒྲིག་ལམ་དང་འགལ་བའི་ནག་ཉེས་འོག་ཉེས་ཆད་ཕོག་ཡོད་འདུག། དེང་སང་རྒྱ་ནག་ནང་རྒྱུགས་ཚབ་སྤྲོད་པ་དང་དཔྱད་རྩོམ་འབྲི་རོགས་བྱེད་པ་སོགས་དང་འབྲེལ་བའི་དྲ་རྒྱ་ ༣༡༠༠ ལྷག་ཡོད་པ་དང་། དེ་ཚོས་རྒྱ་ནག་ནང་ལོགས་ཀྱི་མཐོ་སློབ་ཁག་གི་ནང་ཚོང་ལས་དེ་གཉེར་གྱི་ཡོད་པ་མ་ཟད། ཨ་རི་དང་ཁེ་ན་ཌ། དབྱིན་ལན་སོགས་ཀྱི་ནང་ཡོད་པའི་རྒྱ་ནག་གི་སློབ་ཕྲུག་ཚོར་རྒྱུགས་ཚབ་སྤྲོད་རོགས་བྱས་ཏེ། སྒོར་མོ་མང་པོ་ལེན་གྱི་ཡོད་པ་དང་མཉམ་དུ། རྒྱལ་ཁབ་དེ་ཚོའི་ནང་གནས་སྡོད་བྱེད་པའི་གོ་སྐབས་ཀྱང་འཚོལ་གྱི་ཡོད་པ་དང་། དེ་བཞིན། ཕྱིའི་རྒྱལ་ཁབ་ཁག་ནས་རྒྱ་ནག་ནང་རྒྱུགས་ཚབ་སྤྲོད་དུ་ཡོང་མཁན་ཡང་མང་པོ་ཡོད་འདུག ད་ལྟའི་ཆར་རྒྱ་ནག་ནང་རྒྱུགས་ཚབ་སྤྲོད་མཁན་ལ་རིན་གོང་མ་འདྲ་བ་མང་པོ་ཡོད་ཀྱང་། རྒྱུགས་ཚབ་པ་གཅིག་ལ་རྒྱ་སྒོར་ ༢༠༠༠ ནས་ ༡༠༠༠༠ བར་སྤྲོད་དགོས་པ་དང་། གལ་ཏེ་ཕྱིའི་རྒྱལ་ཁབ་ནང་ཕར་འགྲོ་དགོས་པའམ་ཡང་ན་རྒྱ་ནག་ནང་ཚུར་ཡོང་དགོས་པ་ཡིན་ན། རྒྱུགས་ཚབ་པར་གླ་ཆ་མཐོན་པོ་སྤྲོད་དགོས་པ་མ་ཟད། ཕར་ཚུར་གྱི་གནམ་གྲུའི་གླ་ཆ་དང་མགྲོན་ཁང་སོགས་ཀྱི་འགྲོ་གྲོན་ཚང་མ་གཏོང་དགོས་ཀྱི་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
