ཡུ་སྒོར་ཁྲི་ ༨ ཀྱི་ཉེས་ཆད་གཅོད་འཆར།

ཅེག་སྤྱི་མཐུན་རྒྱལ་ཁབ་ཀྱི་རྒྱལ་སའི་ནང་༧གོང་ས་མཆོག་དང་ཅེག་སྲིད་འཛིན་རྣམ་གཉིས་ཀྱི་སྐུ་པར་བཤམས་པར་ཡུ་སྒོར་ཁྲི་ ༨ ཀྱི་ཉེས་ཆད་གཅོད་རྒྱུ་ཡིན་ལུགས་བརྗོད་ཡོད་འདུག ཨ་ནུ་ཐང་བསོད་ནམས་ལགས་ཀྱིས་གནས་ཚུལ་མཁོ་གཏོང་གནང་བ་ལྟར་ན། འདི་ལོའི་ཕྱི་ཟླ་ ༤ ཚེས་ ༢༩ ཉིན་ཡུ་རོབ་ནང་གི་ཅེག་སྤྱི་མཐུན་རྒྱལ་ཁབ་ཀྱི་རྒྱལ་ས་ཕ་རལ་གྷོ་(Prague)གྱི་དཔལ་འབྱོར་དང་ཤེས་རིག ཆབ་སྲིད་ཀྱི་ལྟེ་གནས་ཁང་རྩེག་ཅིག་གི་ངོས་སུ། ཅེག་སྤྱི་མཐུན་རྒྱལ་ཁབ་ཀྱི་ཁོར་ཡུག་བློན་ཆེན་ཟུར་པ་དང་བོད་དོན་རྒྱབ་སྐྱོར་བ་མཱར་ཅིན་དང་ཁོང་གི་བཟའ་ཟླ་གཉིས་ཀྱིས།༸གོང་ས་མཆོག་དང་ཅེག་སྲིད་འཛིན་ཧ་ཝལ་རྣམ་གཉིས་ཀྱི་སྐུ་པར་རིང་པོ་ཞིག་བཤམས་ཡོད་པ་མ་ཟད། སྐུ་ཞབས་སྲིད་བློན་ལགས། ཁྱེད་ཀྱིས་ནོར་འདུག ཅེག་གིས་བོད་ལ་རོགས་སྐྱོར་བྱེད། ཅེས་བྲིས་ཏེ་འབོད་སྐུལ་གྱི་ལས་འགུལ་སྤེལ་ཡོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༦ ཉིན་གནའ་མཁར་སྲུང་སྐྱོབ་ལས་ཁུངས་ཀྱིས་ལས་འགུལ་དེ་གནའ་མཁར་སྲུང་སྐྱོབ་ཀྱི་ཁྲིམས་ལུགས་དང་འགལ་ཚུལ་བརྗོད་དེ། ཉིན་ ༢༠ ཙམ་གྱི་ནང་མར་བླངས་ཚེ། ཡུ་སྒོར་ཁྲི་ ༨ ཀྱི་ཉེས་ཆད་གཅོད་རྒྱུ་ཡིན་ལུགས་བརྗོད་ཡོད་སྟབས། ཁ་སང་ཕྱི་ཚེས་ ༢༩ ཉིན་ས་གནས་བོད་མི་དང་བོད་དོན་རྒྱབ་སྐྱོར་བ་ཚོས་སྐུ་པར་མར་བླངས་ཏེ། ཆུ་ཚོད་ ༡།༣༠ ཙམ་གྱི་རིང་གོམ་བགྲོད་བྱས་ཏེ་ཧ་ཝལ་ལགས་ཀྱི་སྤུན་མཆེད་ཅིག་གི་ཁང་པའི་རྒྱབ་ཏུ་སྤོས་ཡོད་པ་དང་། ས་གནས་དེ་སྔར་ལས་མི་མང་འདུ་འཛོམས་དང་སྤྲོ་འཆམ་པ་ཡོང་སའི་ལྟེ་གནས་ཤིག་ཡིན་པས། ཡང་བསྐྱར་ཆོག་མཆན་རེག་རྒྱུར་དཀའ་ངལ་འཕྲད་རྒྱུ་ཡིན་སྟབས། བོད་དོན་རྒྱབ་སྐྱོར་བ་དང་འགྲོ་བ་མིའི་ཐོབ་ཐང་དོན་གཉེར་བ་ཚོས་ཅེག་སྲིད་གཞུང་གི་ངོས་ནས་རྒྱ་ནག་གཞུང་ལ་བསྟུན་ཐབས་བྱས་ཏེ། བོད་ཀྱི་རང་དབང་དང་ཐོབ་ཐང་བློས་བཏང་བྱས་པ་ནི་དྲང་བདེན་མིན་པའི་སྐྱོན་བརྗོད་ཤུགས་ཆེན་བྱེད་བཞིན་ཡོད་པ་མ་ཟད། བོད་མི་དང་བོད་ཀྱི་གྲོགས་པོ་རྣམས་འདུ་འཛོམས་ཀྱིས་ངོ་རྒོལ་དང་མཛད་སྒོ་མཉམ་དུ་སྤེལ་འཆར་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
