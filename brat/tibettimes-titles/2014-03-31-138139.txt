མུ་ཤི་ར་ཧྥི་རྒྱལ་ཚོང་བ་རེད་ཟེར།

པ་ཀི་སི་ཐན་ཁྲིམས་ཁང་གིས་སྲིད་འཛིན་ཟུར་བ་མུ་ཤི་ར་ཧྥིར་རྒྱལ་ཚོང་བའི་ནག་ཉེས་བཙུགས་ཡོད་འདུག དེ་རིང་ཞོགས་པར་པ་ཀི་ཐན་རྒྱལ་སའི་དམིགས་བསལ་ཁྲིམས་ཁང་གི་ཁྲིམས་དཔོན་ཧྥ་སེལ་ (Faisal Arab) ལགས་ཀྱིས། སྲིད་འཛིན་ཟུར་བ་མུ་ཤི་ར་ཧྥིས་ཕྱི་ལོ་ ༢༠༠༧ ལོའི་ཟླ་ ༡༡ ཚེས་ ༣ ཉིན་ཆེས་མཐོའི་ཁྲིམས་ཁང་གི་ཁྲིམས་དཔོན་ཕྱིར་འབུད་དང་། རྩ་ཁྲིམས་རྩིས་མེད། རྒྱལ་ཁབ་ནང་སྲིད་དབང་སྒེར་གཅོད་བཅས་བྱས་པར་བརྟེན།  ཁོ་རང་པ་ཀི་སི་ཐན་གྱི་རྒྱལ་ཚོང་བ་ཡིན་པའི་ཉེས་འཛུགས་བྱས་ཡོད་འདུག ཐེངས་འདིར་ཁྲིམས་ཁང་གིས་ཉེས་འཛུགས་བྱེད་སྐབས། མུ་ཤི་ར་ཧྥི་ཁྲིམས་རྭའི་རྐུབ་སྟེགས་ལས་ཡར་ལངས་ཏེ་ནག་ཉེས་ཁས་ལེན་བྱས་མེད་པ་མ་ཟད་། སྲིད་དོན་གྱི་ངན་སྐུལ་ཁོ་ན་རེད་ཅེས་ལན་འདེབས་བྱས་འདུག ད་དུང་ཧྥ་རན་སིའི་ (AFP) གསར་ཁང་གི་གསར་འགོད་པར། “ ང་རང་རྒྱལ་ཚོང་བ་རེད་ཟེར། ཉེ་བའི་ལོ་ ༤༥ རིང་ངས་པ་ཀི་སི་ཐན་གྱི་སྤྱི་ཁྱབ་དམག་སྤྱི་དང་དམག་མི་བྱས་མྱོང་། ང་རང་རྒྱལ་ཚོང་བ་ཆགས་པ་ཡིན་ནམ། ” ཞེས་བརྗོད་ཡོད་འདུག སྲིད་དོན་ཉམས་ཞིབ་པ་ཚོས། ཁྲིམས་ཁང་གིས་མུ་ཤི་ར་ཧྥིར་ཉེས་འཛུགས་བྱས་པ་ར་སྤྲོད་ཐུབ་ཚེ་སྲོག་ཁྲིམས་གཅོད་འཆར་ཡོད་ཀྱང་།  ད་ལྟའི་པ་ཀི་སི་ཐན་གྱི་སྲིད་བློན་ནྰ་ཝ་རྫི་ཞྰར་ཧྥི་ (Nawaz Sharif) དང་ཁོ་གཉིས་བར་སྲིད་དོན་ཀྱི་འཐབ་རྩོད་དང་འཁོན་སེམས་ཡོད་སྟབས། དྲང་བདེན་གྱི་ཁྲིམས་བཤེར་བྱེད་རྒྱུ་ཁག་པོ་ཡིན་ཚུལ་བརྗོད་ཀྱི་ཡོད་འདུག རྒྱ་གར་གྱི་གནད་དོན་དབྱེ་ཞིབ་པ་ཚོས། སྲིད་འཛིན་ཟུར་བ་མུ་ཤི་ར་ཧྥི་པ་ཀི་སི་ཐན་གྱི་དམག་མི་དང་འབྲེལ་བ་དམ་ཟབ་ཡོད་སྟབས། དམག་མིས་ཁྲིམས་ཁང་གི་ཐག་གཅོད་ལ་ཐེ་གཏོགས་བྱེད་སྲིད་པ་དང་། དེ་མིན་རྒྱལ་ཁབ་ཀྱི་དམག་དཔུང་དང་ཁྲིམས་ཁང་། གཞུང་འཛིན་ཚོགས་པ་བཅས་ཀྱི་འབྲེལ་ལམ་ཇེ་སྡུག་ཏུ་འགྲོ་བར་ཚོད་དཔག་བྱེད་ཀྱི་ཡོད། གསར་འགོད་པ། ལྷ་སྐྱབས་རྒྱལ།
