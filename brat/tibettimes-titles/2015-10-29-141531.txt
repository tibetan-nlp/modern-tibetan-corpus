ས་གནས་ཁག་ ༤༩ ཡི་སྲིད་སྐྱོང་འོས་གྲངས།

  འདི་གའི་གསར་འགོད་པས་ཁ་པར་བརྒྱུད་དེ་ཡུ་རོབ་སོགས་ས་གནས་ཁག་ ༤༩ ཙམ་གྱི་འོས་བསྡུའི་གྲུབ་འབྲས་རྒྱུས་ལོན་བྱུང་བ་ལྟར་ན། ཁ་སང་ཕྱི་དྲོའི་བར་སྲིད་སྐྱོང་ལས་ཐོག་བློ་བཟང་སེང་གེ་མཆོག་ལ་འོས་གྲངས་  ༢༢༡༣༥ དང་། སྤྱི་འཐུས་ཚོགས་གཙོ་སྤེན་པ་ཚེ་རིང་མཆོག་ལ་འོས་གྲངས་ ༨༡༠༤ ། ཨ་ཚོགས་ཀླུ་མཁར་བྱམས་ལགས་སུ་འོས་གྲངས་ ༡༥༩༦ ། བཀྲ་ཤིས་དབང་འདུས་ལགས་སུ་འོས་གྲངས་ ༡༢༨༧  ། བཀྲ་ཤིས་སྟོབས་རྒྱལ་ལགས་སུ་འོས་གྲངས་ ༢༨ བཅས་ཐོབ་འདུག   ཐེངས་འདིར་ཁྱོན་བསྡོམས་འོས་བསྡུའི་དེབ་སྐྱེལ་གནང་མཁན་༨༧༡༧༩ བྱུང་ཡོད་སྟབས། ཕྱི་ལོ་ ༢༠༡༠ ལོའི་འོས་བསྡུར་དེབ་སྐྱེལ་གནང་མཁན་ ༨༢༨༠༠ བྱུང་བ་དང་བསྡུར་ན། ད་རེབ་མི་ ༤༣༧༩ ཡིས་མང་བ་ཆགས་ཡོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
