བསམ་བློ་བ་ ༥ རྒྱ་གར་བ་ཡིན་ཟེར།

ད་ལོ་འཛམ་གླིང་ཐོག་འབུར་ཐོན་བྱས་རྗེས་འཇོག་མཁན་གྱི་བསམ་བློ་བ་ལྔ་བཅུའི་ཁྲོད་ཀྱི་ལྔ་རྒྱ་གར་གྱི་མི་སེར་ཡིན་འདུག ཁ་སང་དབྱིན་ཡུལ་ནང་རྟེན་གཞི་བྱས་པའི་བསམ་རྒྱ་དུས་དེབ་ ( Prospect Magazine) ཀྱིས་འདས་པའི་ལོ་གཅིག་ཙམ་རིང་། དྲ་རྒྱ་བརྒྱུད་འཛམ་གླིང་གི་ལས་རིགས་འདྲ་མིན་ཐོག་ཕུལ་བྱུང་ཕྱག་ལས་གནང་མཁན་དང་བྱས་རྗེས་འཇོག་མཁན་ཡེ་ཤུའི་བླ་ཆེན་ཧྥན་སེ་སེ་ ( Pope Francis) གཙོས་གྲགས་ཅན་མི་སྣ་ ༥༠ བདམས་ཡོད་འདུག་ལ། དེ་དག་ཁྲོད་རྒྱ་གར་གྱི་དཔལ་འབྱོར་རིག་པ་བ་ཨ་མར་ཌེ་སན་ (Amartya Sen) ལགས་དང་། ཀུ་ཤིག་བྷ་སུ་ ( Kaushik Basu) ། དྷ་སེ་གྷུབ་ཌ ( Partha Dasgupta)། ར་གྷུ་རམ་ (Raghuram Rajan)། རྩོམ་པ་པོ་ཨ་རུན་དྷ་ཌེ་རོཡེ (Arundhati Roy) ལགས་བཅས་རྒྱ་གར་བ་མི་ ༥ བདམས་ཡོད་འདུག འདིའི་ཐད་བསམ་རྒྱ་དུས་དེབ་ཀྱི་རྩོམ་སྒྲིག་པ་མེ་ཌོག་སེ་ (Bronwen Maddox) ལགས་ཀྱིས། ང་ཚོས་འཛམ་གླིང་གི་ལས་རིགས་འདྲ་མིན་ཐོག་བླ་མེད་མཛད་རྗེས་འཇོག་མཁན་གྱི་མི་སྣ་དག་ལ་ཡིད་རངས་དང་། མི་རབས་གསར་པ་ཚོས་དམིགས་དཔེ་དང་སློབ་སྦྱོང་ཞུ་ཆེད་ཙམ་མ་ཟད། ཁོང་ཚོས་གང་གནང་བ་དག་སྔར་ལས་ལྷག་པའི་འཕེལ་རྒྱས་དང་མང་ཚོགས་ཚོར་ཕན་བདེ་བསྐྲུན་རྒྱུར་སྐུལ་འདེད་བྱ་ཆེད་ཡིན་སྐོར་བརྗོད་འདུག ༢༠༠༤ ལོར་མང་ཚོགས་ལ་ཤེས་རྟོགས་ཡོང་ཆེད་དབྱིན་ཡུལ་གྱི་གྲགས་ཅན་མི་སྣ་ ༥༠ འདེམས་སྒྲུག་གིས་ཐོག་མར་མགོ་འཛུགས་བྱས་ཡོད་པ་དང་། ༢༠༠༥ ལོ་ནས་བཟུང་དྲ་རྒྱ་བརྒྱུད་འཛམ་གླིང་ལས་རིགས་འདྲ་མིན་གྱི་མི་སྣ་འོས་འདེམས་བྱེད་མགོ་བཙུགས་ཏེ། ལོ་རེར་འཛམ་གླིང་གི་དཔལ་འབྱོར་རིག་པ་བ་ ༡༧ དང་། ལྟ་གྲུབ་སྨྲ་བ་ ༡༣། ཚན་རིག་པ་ ༣། འཕྲུལ་ལས་མཁས་ཅན་ ༣། ཚོང་ལས་མཁས་ཅན་ ༥ བཅས་དང་། དེ་བཞིན་རྩོམ་པ་པོ་དང་ལས་འགུལ་བ་སོགས་ཀྱང་འདེམས་སྒྲུག་བྱེད་ཀྱི་ཡོད་འདུག འདི་ལོའི་འདེམས་ཐོན་བྱུང་བའི་གྲགས་ཅན་མི་སྣ་ཁག་ནི་ཡེ་ཤུའི་བླ་ཆེན་ཧྥན་སེ་སེ་དང་ཨ་རིའི་སེམས་ཁམས་ཚན་རིག་པ་ཐོ་མ་སེ་ཎེ་གྷལ་ (Thomas Nagel)། ཨ་རིའི་རྒྱལ་སྤྱིའི་ཁྲིམས་ལུགས་མཁས་ཅན་ཨེ་ཎེ་མ་རེ་ ( Anne-Marie Slaughter)། ཁེ་ན་ཌའི་རྩོམ་པ་པོ་ (Naomi Klein)། སི་ལོ་ཝེ་ནི་ཡའི་ལྟ་གྲུབ་མཁས་ཅན་སཱ་ཝ་ཆི་ (Slavoj Žižek)། རྒྱ་ནག་གི་དཔལ་འབྱོར་རིག་པ་མཁས་ཅན་མའོ་ཡུ་ཤི་ (Mao Yushi)། འབར་མའི་ལོ་རྒྱུས་མཁས་ཅན་ཐཱན་ཌི་ (Thant Myint-U)། ཨི་སི་རལ་གྱི་སེམས་ཁམས་ཚན་རིག་པ་ཁཱ་ནེ་མན་ (Daniel Kahneman)། ལྷོ་ཀོ་རི་ཡའི་དཔལ་འབྱོར་རིག་པ་མཁས་ཅན་ཨཱ་ཇོན་ (Ha-Joon Chang)། ལྷོ་ཨ་ཧྥེ་རི་ཀའི་ཚོང་ལས་མཁས་ཅན་ཨེ་ལོན་ (Elon Musk)། དབྱིན་ཡུལ་གྱི་དངོས་ཁམས་ཚན་རིག་པ་པེ་ཌར་ (Peter Higgs)། ནེ་ཇི་རི་ཡའི་དཔལ་འབྱོར་རིག་པ་མཁས་ཅན་ (Ngozi Okonjo-Iweala)། ཨི་ཌ་ལའི་དངོས་ཁམས་ཚན་རིག་པ་ (Fabiola Gianotti)། ཧྥ་རན་སིའི་དཔལ་འབྱོར་རིག་པ་མཁས་ཅན་ཨེ་སེ་ཌར་ (Esther Duflo)། སེང་གྷ་པུར་གྱི་སློབ་གསོ་མཁས་ཅན་ཀེ་ཤོར་ (Kishore Mahubani)། བྷར་ཛིལ་གྱི་ལྟ་གྲུབ་མཁས་ཅན་རོ་བྷར་ཌོ་ (Roberto Unger) སོགས་ཡོད་འདུག ༢༠༠༥ ལོའི་བསམ་རྒྱ་དུས་དེབ་ཀྱི་སྙན་གྲགས་ཆེ་ཤོས་ཀྱི་མི་སྣ་ཨ་རིའི་ལྟ་གྲུབ་མཁས་ཅན་ཎོམ་ཆོམ་སེ་ཁེའེ་ (Noam Chomsky) ལགས་དང་། ༢༠༡༣ ལོར་དབྱིན་ཡུལ་གྱི་སྐྱེ་དངོས་ཚན་རིག་པ་དང་ལྷའི་འཁྲུལ་རྟོག་ (The God Delusion) ཅེས་པའི་དཔེ་དེབ་ཀྱི་རྩོམ་པ་པོ་རེ་ཆར་ཌི་ (Richard Dawkins) ལགས་ཡིན་འདུག འདི་ལོའི་སྙན་གྲགས་ཆེ་ཤོས་ཀྱི་མི་སྣ་མིང་ཐོ་ཟླ་ ༤ པའི་ཚེས་ ༡༡ ཉིན་ཁྱབ་བསྒྲགས་གནང་འཆར་ཡོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
