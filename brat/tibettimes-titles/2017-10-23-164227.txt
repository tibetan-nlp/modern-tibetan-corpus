ཤིན་ཛོ་ཨ་བྷི་སླར་ཡང་སྲིད་བློན་དུ་བདམས།

ཉི་ཧོང་གྲོས་ཚོགས་འོག་པའི་འོས་བསྡུའི་ཁྲོད་སྲིད་བློན་ལས་ཐོག་ཤིན་ཛོ་ཨ་བྷི་ལགས་ཀྱི་ཚོགས་པར་རྒྱལ་ཁ་ཆེན་པོ་ཐོབ་སྟེ་ཁོང་གིས་ཡང་བསྐྱར་སྲིད་བློན་གྱི་ལས་འཁུར་ཞུ་རྒྱུ་ཡིན་འདུག སི་ཨན་ཨན་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཁ་སང་ཉི་ཧོང་རྒྱལ་ཡོངས་གྲོས་ཚོགས་འོག་མའི་འོས་བསྡུའི་གྲུབ་འབྲས་གསལ་བསྒྲགས་གནང་ཡོད་པ་དང་། ཐེངས་འདིའི་འོས་བསྡུའི་ཁྲོད་གྲོས་ཚོགས་འོག་མར་ཞབས་བཀྱག་ ༤༦༥ ཡོད་པ་ལས་ ༣༡༣ ད་ལྟའི་སྲིད་བློན་ལས་ཐོག་པ་ཤིན་ཛོ་ཨ་བྷི་ལགས་ཀྱི་སྲིད་དོན་ཚོགས་པ་དང་དེའི་ཟླ་སྒྲིལ་ཚོགས་པ་གཉིས་ལ་ཐོབ་སྟེ་རྒྱལ་ཁ་ཆེན་པོ་བླངས་ཡོད་སྟབས། མི་འགྱང་བར་ཁོང་གིས་སྲིད་བློན་གྱི་ལས་འཁུར་དམ་བཅའ་འབུལ་རྒྱུ་ཡིན་འདུག འོས་བསྡུའི་གྲུབ་འབྲས་གསལ་བསྒྲགས་བྱས་རྗེས། སྲིད་བློན་ཤིན་ཛོ་ཨ་བྷི་ལགས་ཀྱིས་གང་མྱུར་ངས་ལས་དོན་གལ་ཆེན་པོ་ཞིག་བསྒྲུབ་དགོས་པ་ནི་བྱང་ཀོ་རི་ཡ་དང་འབྲེལ་བའི་དཀའ་རྙོག་དེ་ཡིན་པ་དང་། མི་མང་གིས་མངོན་འདོད་དང་དེ་སྔོན་འོས་བསྡུའི་ཁྱབ་བསྒྲགས་ལས་འགུལ་སྐབས་དམ་བཅའ་བྱས་པ་བཞིན། བྱང་ཀོ་རི་ཡ་ལ་ཁ་གཏད་གཅོག་པར་ལངས་ཕྱོགས་བརྟན་པོ་བཟུང་སྟེ་ཐབས་བྱུས་ནུས་ཤུགས་ལྡན་པ་ཞིག་བྱེད་ཀྱི་ཡིན། ཞེས་གསུངས་ཡོད་འདུག ཡང་ཁོང་གིས། ཉི་ཧོང་འཇིགས་པ་ལས་སྲུང་སྐྱོབ་ཆེད་བདེ་སྲུང་དམག་གི་སྟོབས་ཤུགས་རྒྱ་སྐྱེད་དང་ཉི་ཧོང་གི་དམག་དོན་སྲ་བརྟན་བཟོ་རྒྱུ་མ་ཟད། བྱང་ཀོ་རི་ཡའི་གནད་དོན་དང་འབྲེལ་བའི་ཐོག་ཨ་རིའི་སྲིད་འཛིན་ཌོ་ནལ་ཊོ་རུམ་ཕི་ལགས་དང་གྲོས་བསྡུར་བྱ་རྒྱུ་ཡིན་སྐོར་གསུངས་ཡོད་པདང་། འདི་ལོའི་ཕྱི་ཟླ་ ༡༡ ནང་སྲིད་འཛིན་ཏུམ་ཕི་ཉི་ཧོང་ནང་ཐེངས་དང་པོའི་གཞུང་འབྲེལ་གཟིགས་སྐོར་གནང་འཆར་ཡོད་འདུག ཉི་ཧོང་དང་བྱང་ཀོ་རི་ཡ་གཉིས་ཀྱི་བར་ཆབ་སྲིད་དང་དཔལ་འབྱོར་སོགས་ཕྱོགས་གང་ཐད་ནས་འབྲེལ་ལམ་ཞན་པོ་ཡོད་སྟབས། ཉི་ཧོང་གིས་བྱང་ཀོ་རི་ཡར་ཁ་གཏད་གཅོག་པར་ཨ་རི་དང་འབྲེལ་ལམ་བཟང་པོ་ཡོད་པ་མུ་མཐུད་རྒྱུན་སྐྱོང་དགོས་པ་དེ་ཤིན་ཏུ་གལ་ཆེན་པོ་ཆགས་ཡོད་པ་རེད། སྲིད་བློན་ཤིན་ཛོ་ཨ་བྷི་ལགས་ནི་ད་ལྟ་རང་ལོ་ ༦༣ ཡིན་ཞིང་། ཕྱི་ལོ་ ༢༠༡༢ ལོར་ཐོག་མར་ཉི་ཧོང་གི་སྲིད་བློན་ལ་བདམ་ཐོན་བྱུང་ཡོད་པ་དང་། ཁོང་གི་དབང་འཛིན་དུས་སྐབས་གཉིས་པ་ཁོང་གིས་སྲིད་བློན་གྱི་ལས་ཁུར་དམ་འབུལ་ཞུ་རྒྱུ་ཡིན་པ་དང་། ཁོང་ནི་ཉི་ཧོང་གི་སྲིད་བློན་རིམ་པའི་ལོ་རྒྱུས་ནང་ཞབས་ཞུའི་དུས་ཡུན་རིང་ཤོས་གསུམ་པ་དེ་ཆགས་ཡོད་པ་མ་ཟད། ཁོང་མ་འོངས་པར་སྲིད་བློན་དུས་ཡུན་རིང་ཤོས་ཆགས་སྲིད་སྐོར་རྗོད་ཀྱི་འདུག གསར་འགོད་པ། ཉི་མ་ཚེ་རིང་།
