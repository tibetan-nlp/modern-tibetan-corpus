བོད་ནང་གི་མི་ ༢༠ རྐྱེན་འདས་དང་མི་ ༥༨
ལ་རྨས་སྐྱོན་ཕོག

དེ་རིང་ཕྱི་དྲོར་དྷ་སའི་གཙུག་ལག་ཁང་དུ་བོད་མིའི་སྒྲིག་འཛུགས་བཀའ་ཤག་གིས་བལ་ཡུལ་ས་ཡོམ་ཐད་མཆོད་འབུལ་སྨོན་ལམ་ཚོགས་པའི་སྟེང་སྲིད་སྐྱོང་མཆོག་གིས་བོད་ནང་མི་ ༢༠ ཙམ་རྐྱེན་འདས་སུ་སོང་ཡོད་སྐོར་གསུངས་སོང་། དེ་རིང་ཕྱི་དྲོའི་ཆུ་ཚོད་ ༤ པའི་སྟེང་། བོད་མིའི་སྒྲིག་འཛུགས་བཀའ་ཤག་གི་གོ་སྒྲིག་འོག དྷ་ས་ཐེག་ཆེན་ཆོས་གླིང་གཙུག་ལག་ཁང་དུ་། ཕྱི་ཚེས་ ༢༥ ཉིན་བལ་ཡུལ་དུ་ས་ཡོམ་ཚད་རིམ་ ༧།༩ ཅན་ཞིག་བརྒྱབ་སྟེ་མི་རྒྱུ་གཉིས་ཀར་གོད་ཆག་ཆེན་པོ་བྱུང་བའི་ཐད། མཆོད་སྤྲིན་བརྒྱ་ཚར་འབུལ་ཤོམ་གྱིས་མཆོད་འབུལ་སྨོན་ལམ་ཞིག་ཚོགས་པའི་སྟེང་ཀིརྟི་རིན་པོ་ཆེ་མཆོག་དབུ་བཞུགས་ཐོག སྲིད་སྐྱོང་བློ་བཟང་སེང་གེ་མཆོག་གིས་དབུས་པའི་དམངས་གཙོའི་ཀ་བ་གསུམ་གྱི་མཛད་གཙོ་རྣམ་པ་དང་། བོད་མི་མང་སྤྱི་འཐུས་རྒྱུན་ལས་རྣམ་པ། ལྷན་ཁང་ཁག་གི་དྲུང་ཆེ་ཆུང་དང་སྲི་ཞུ་ལས་བྱེད། བཞུགས་སྒར་ཁུལ་གྱི་ཆོས་སྡེ་ཁག་གི་དགེ་བཙུན་སྔགས་གསུམ། བོད་ཁྱིམ་སློབ་ཕྲུག་དང་ས་གནས་མང་ཚོགས་བཅས་མི་མང་ ༡༠༠༠ ལྷག་འདུ་འཛོམས་གནང་སོང་། སྐབས་དེར་སྲིད་སྐྱོང་མཆོག་གིས་མཆོད་འབུལ་སྨོན་ལམ་དང་འབྲེལ་བའི་གསུང་བཤད་ནང་དོན། ད་རེབ་བལ་ཡུལ་གཙོས་པའི་རྒྱལ་ཁབ་བཞིའི་ནང་ས་ཡོམ་ཤུགས་ཆེན་པོ་བྱུང་ཡོད་པ་དང་། དེའི་ནང་བལ་ཡུལ་ཁྱིམ་མཚེས་རྒྱ་གར་དང་བྷང་ལ་ཌེ་ཞི། འབྲུག དེ་བཞིན་དུ་བོད་ནང་བཅས་ལ་ས་ཡོམ་རྐྱེན་གྱིས་མི་མང་པོ་འདས་གྲོངས་དང་། མི་རྒྱུ་གཉིས་ལ་གནོད་སྐྱོན་ཆེན་པོ་ཐེབས་པའི་ཐད་ཁ་སང་བཀའ་ཤག་ལྷན་ཚོགས་བྱས་ཏེ། ཕྱི་རྒྱལ་ཁག་གི་དོན་གཅོད་དང་ས་གནས་མགོ་འཛིན་ཚང་མས་འགན་ཁུར་ཏེ། བལ་ཡུལ་དང་འབྲེལ་བའི་ཐོག་རོགས་དངུལ་གནང་མཁན་ཡོད་ན་དོན་གཅོད་དང་ས་གནས་མགོ་འཛིན་གྱིས་འགན་འཁུར་དགོས་པའི་འབོད་སྐུལ་ཞུས་ཡོད། དེ་བཞིན། བལ་ཡུལ་ནང་ཡོད་པའི་དགོན་སྡེ་ཁག་དང་མི་མང་། སྐྱིད་སྡུག་ཁག་བཅས་ཀྱིས་ གང་ཐུབ་ཅི་ཐུབ་ཀྱིས་མཐུན་སྦྱོར་གནང་རྒྱུར་འབོད་སྐུལ་ཞུས་ཡོད་པ་མ་ཟད། བཀའ་ཤག་གིས་བལ་སྒོར་འབུམ་ ༢༠ ཙམ་བལ་ཡུལ་གཞུང་ལ་ངལ་སེལ་ཆེད་འབུལ་རྒྱུའི་གྲོས་ཐག་བཅད་ཡོད། ཅེས་གསུངས་སོང་། སྲིད་སྐྱོང་མཆོག་གིས་ས་ཡོམ་གྱི་གནས་སྟངས་སྐོར་ངོ་སྤྲོད་གནང་དོན། དེ་རིང་འདིར་འཛོམས་དགོས་དོན་གཙོ་བོ་འདས་གྲོངས་སུ་ཕྱིན་པ་ཚོར་ཐུགས་སྨོན་གནང་རྒྱུ་དང་། རྨས་སྐྱོན་ཕོག་པ་རྣམས་གང་མགྱོགས་དྲག་སྐྱེད་ཡོང་བའི་སྨོན་འདུན་ཞུ་པ་དང་། ནང་མི་སྤུན་ཆེད་ཚོར་གདུང་སེམས་མཉམ་སྐྱེ་བྱེད་བཞིན་ཡོད། གསར་འགྱུར་བརྒྱུད་ལམ་ཁག་གཞིར་བྱས་ན། བལ་ཡུལ་ནང་ལ་མི་ ༣༦༠༠ ལྷག་འདས་གྲོངས་དང་། མི་ ༦༥༠༠ ལྷག་ལ་རྨས་སྐྱོན་བྱུང་ཡོད་པ་དང་། དེ་བཞིན། རྒྱ་གར་ནང་མི་ ༦༦ འདས་གྲོངས་སུ་ཕྱིན་ཡོད་པ་དང་། མི་ ༡༠༠ ལྷག་ལ་རྨས་སྐྱོན་ཕོག་ཡོད་འདུག བྷང་ལ་ཌེ་ཞི་ལ་མི་ ༤ འདས་གྲོངས་སུ་སོང་པའི་གནས་ཚུལ་འདུག བློ་ཕམ་པ་ཞིག་ལ་བོད་ནང་སྐྱིད་གྲོང་དང་གཉའ་ལམ། དིང་རི་ཁག་ལ་ས་ཡོམ་དེའི་རྐྱེན་གྱིས་ཁང་པ་མང་པོར་གནོད་སྐྱོན་ཐེབས་པ་དང་། བོད་ནང་གི་མི་ ༢༠ འདས་གྲོངས་སུ་ཕྱིན་པ་དང་མི་ ༥༨ ལ་རྨས་སྐྱོན་བྱུང་པའི་གནས་ཚུལ་ཐོན་འདུག བལ་ཡུལ་ནང་དུ་ཡོད་པའི་བོད་མི་ཚོའི་གནས་ཚུལ་ར་འཕྲོད་པའི་ནང་དུ་མཆོད་རྟེན་བྱ་རུང་ཁ་ཤོར་ཁུལ་དུ་བཞུགས་མཁན་གཅིག་དང་། ས་མཚམས་ཁག་ཏུ་བཙུན་མ་གཅིག་བཅས་བལ་ཡུལ་ནང་ཡོད་པའི་བོད་མི་ ༢ འདས་གྲོངས་སུ་ཕྱིན་པའི་གནས་ཚུལ་བྱུང་འདུག བལ་ཡུལ་གྱི་དགོན་པ་ཁག་དང་སློབ་གྲྭ་ཁག་ལ་སྐྱོན་ཏོག་ཙམ་བྱུང་ཡོད་ཀྱང་། སྐྱོན་ཆེན་པོའི་གནས་ཚུལ་དེ་འདྲ་བྱུང་མི་འདུག ཅེས་གསུངས་སོང་། གཞན། དེ་རིང་གཙུག་ལག་ཁང་གི་ཕྱི་ལོགས་སུ་གཞུང་འབྲེལ་མ་ཡིན་པའི་ཚོགས་པ་ཁག་གིས་ཞལ་འདེབས་བསྡུ་སློང་བྱས་ཏེ་རོགས་སྐྱོར་ལས་འགུལ་སྤེལ་རྒྱུར་གྲ་སྒྲིག་བྱེད་བཞིན་འདུག གསར་འགོད་པ། འཇིགས་མེད་རྒྱ་མཚོ།
