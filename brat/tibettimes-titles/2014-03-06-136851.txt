རྒྱ་གར་རྒྱལ་ཡོངས་འོས་བསྡུའི་ལས་རིམ།

ཁ་སང་རྒྱ་གར་དབུས་འོས་བསྡུ་ལྷན་ཁང་གིས་གསལ་བསྒྲགས་སྤེལ་ཏེ། སྐབས་བཅུ་དྲུག་པའི་རྒྱལ་ཡོངས་འོས་བསྡུ་ཆེན་མོ་དེ་ཕྱི་ཟླ་ ༤ པའི་ཚེས་ ༧ ཉིན་མགོ་ཚུགས་བྱ་རྒྱུ་ཡིན་ཚུལ་བསྒྲགས་འདུག ཁ་སང་སྔ་དྲོའི་ཆུ་ཚོད་ ༡༠།༣༠ ཐོག རྒྱ་གར་དབུས་འོས་བསྡུ་ལྷན་ཁང་གི་འགན་འཛིན་སཱམ་པཱ་ཐཱ་ (V S Sampath)ལགས་ཀྱི་གཙོས་འོས་བསྡུའི་འགན་འཛིན་ལས་རོགས་གཞན་གཉིས་དང་ལྷན་དུ་གསར་འགོད་གསལ་བསྒྲགས་སྤེལ་ཏེ། སྐབས་ ༡༦ པའི་རྒྱ་གར་རྒྱལ་ཡོངས་འོས་བསྡུ་ཆེན་མོ་དེ་ཕྱི་ཟླ་ ༤ པའི་ཚེས་ ༧ ནས་ཟླ་ ༥ པའི་ཚེས་ ༡༢ བར། སྡེ་ཚན་ ༩ ལ་ཆ་བགོས་ཀྱིས་འོས་བསྡུ་བྱ་རྒྱུ་ཡིན་སྐོར་ཁྱབ་བསྒྲགས་གནང་འདུག འོས་བསྡུའི་འགན་འཛིན་སཱམ་པཱ་ཐཱ་ལགས་ཀྱིས། “རྒྱ་གར་རྒྱལ་ཡོངས་ཆབ་སྲིད་ཚོགས་པ་ཁག་གིས་སོ་སོའི་འོས་མི་ཁྱབ་བསྒྲགས་གནང་སྐབས་ཚུལ་མཐུན་བག་ཡོད་ཀྱི་འབོད་ཚིག་སྒྲོག་པ་ལས་མི་གཞན་དང་ཚོགས་པ་གཞན། སྤྱི་ཚོགས་བཅས་ལ་གནོད་གཞིའི་ཚིག་ངན་རིགས་དང་། གཞན་སྡེ་ཕམ་རྒྱུའི་ངན་བྱུས་རིགས་མི་བཤམ་པར། ཚུལ་མཐུན་དང་དྲང་བདེན་ཐོག་ནས་རྒྱལ་ཕམ་རྩོད་དགོས་”ཞེས་བརྗོད་འདུག ཐེངས་འདིར་རྒྱ་གར་རྒྱལ་ཡོངས་སུ་འོས་ཆོས་ཚང་བའི་མི་གྲངས་ས་ཡ་ ༨༡༤  ལྷག་གིས་དེབ་སྐྱེལ་བྱས་ཡོད་འདུག་ཅིང་། འདི་ནི་རྒྱ་གར་རང་བཙན་ཐོབ་པའི་རྗེས་ཀྱི་དེབ་སྐྱེལ་བྱེད་མཁན་མང་ཤོས་དེ་ཆགས་ཡོད་པ་མ་ཟད།  ༢༠༠༩ ལོའི་རྒྱལ་ཡོངས་འོས་བསྡུ་ཆེན་མོ་དང་བསྡུར་ན་དེབ་སྐྱེལ་བྱེད་མཁན་ས་ཡ་ ༡༠༠ ལྷག་མང་དུ་ཕྱིན་ཡོད་འདུག གནད་དོན་དབྱེ་ཞིབ་པ་ཚོས་བརྗོད་པ་ལྟར་ན། འགྲན་རྩོད་ཤུགས་ཆེན་བྱེད་ཀྱིན་པའི་རྒྱལ་ཡོངས་འོས་བསྡུ་ཆེན་མོའི་ཁྲོད། རྒྱ་གར་རྒྱལ་ཡོངས་སུ་ཆབ་སྲིད་ཚོགས་པ་མང་པོ་ཞིག་ཡོད་པ་ལས་ཀོང་རེ་སེ་ཆབ་སྲིད་ཚོགས་པ་དང་། བྷེ་ཇཱའི་པི་ཆབ་སྲིད་ཚོགས་པ། མི་སེར་དཀྱུས་མའི་ཆབ་སྲིད་ཚོགས་པ། འཐབ་ཕྱོགས་གསུམ་པའི་ཆབ་སྲིད་ཚོགས་པ་བཞི་ལ་རེ་བ་ཆེན་པོ་ཡོད་པ་དང་། ཀོང་རེ་སེ་ཚོགས་པའི་སྲིད་བློན་འོས་མི་ར་ཧུལ་གྷན་དྷི་ལགས་སུ་མང་ཚོགས་ཀྱི་བློ་བཀུག་ཐུབ་པའི་ཉམས་མྱོང་དང་ནུས་པ་ཞན་པ། མི་སེར་དཀྱུས་མའི་ཚོགས་པའི་སྲིད་བློན་འོས་མི་ཨར་ཝུན་ཁེ་ཅེ་རེ་ཝལ་ལགས་ཀྱིས་ཉེ་ལམ་ལྡི་ལི་ས་གནས་གཞུང་ཉིན་གྲངས་ཉུང་ངུ་ལས་འཛིན་སྐྱོང་བྱེད་ཐུབ་མེད་པ་དང་འོས་བསྡུའི་ཁྱབ་བསྒྲགས་ཐད་དཔལ་འབྱོར་འདང་ངེས་མེད་པ། འཐབ་ཕྱོགས་གསུམ་པའི་ཚོགས་པའི་མགོ་ཁྲིད་དག་གི་ལྟ་བ་ལངས་ཕྱོགས་མཐུན་པོ་མེད་པ་སོགས་ལ་བརྟེན་ཏེ། ཐེངས་འདིར་བྷེ་ཇཱའི་པི་ཆབ་སྲིད་ཚོགས་པར་རྒྱལ་ཁ་ཐོབ་རྒྱུ་ཡིན་པའི་ཚོད་དཔག་བྱེད་བཞིན་ཡོད་འདུག རྒྱ་གར་གྱི་ཐབས་ཇུས་མཁས་པ་མོ་ཧན་(Mohan Guruswamy) ལགས་ཀྱིས། “ངའི་བསམ་པར་སྤྱི་འབངས་ཚོས་རྒྱལ་ཁབ་ཀྱི་མགོ་ཁྲིད་དེ་བློ་སྟོབས་ལྡན་པ་ཞིག་བསླེབ་རྒྱུར་རེ་བ་འཆང་གི་ཡོད་སྟབས། སྐུ་ཞབས་ཎ་རིན་ཌ་མུ་དྷི་ལགས་ནི་མང་ཚོགས་ཀྱིས་རེ་སྒུག་བྱ་ཡུལ་གྱི་མགོ་ཁྲིད་གཅིག་པུ་དེ་ཆགས་ཡོད། ” ཅེས་བརྗོད་འདུག རྒྱ་གར་གྱི་སྐབས་ ༡༦ པའི་རྒྱལ་ཡོངས་འོས་བསྡུའི་ཁྲོད་གྲོས་ཚོགས་འཐུས་མི་ ༥༤༣ འདེམ་དགོས་ཤིང་། དེ་དག་མངའ་སྡེ་ཁག་ལ་ཆ་བགོས་བྱས་ན་ཨན་དྷ་ར་པར་དྷི་ཤིར་གྲོས་ཚོགས་འཐུས་མི་ ༤༢ དང་ཨ་རུ་ན་ཅལ་ལ་འཐུས་མི་ ༢། ཨ་སམ་ལ་འཐུས་མི་ ༡༤། བྷེ་ཧར་ལ་ ༤༠ ། ཆཱ་དྷིསེ་གྷར་ལ་ ༡༡། གྷོ་ཝ་ལ་ ༢།  གྷོ་ཅུ་ར་ཌེ་ལ་ ༢༦། ཧར་ཡཱཎ་ལ་ ༡༠། ཧེ་མ་ཅལ་ལ་ ༤། ཇམ་མུ་དང་ཀཱ་ཤི་མེར་ལ་ ༦། ཇར་ཁནཌི་ལ་ ༡༤། ཀཱ་ཎཱ་ཀྲ་ཁཱར་ ༢༨། ཀེ་ར་ལཱར་ ༢༠། མཱ་དྷེཡ་པར་དྷི་ཤིར་ ༢༩། མཱཧ་ར་ཤི་ཌར་ ༤༨། མ་ནི་པུར་ ༢། མེ་གྷ་ལཱ་ཡཱར་ ༢། མེ་ཛོ་རམ་ལ་ ༡། ཎ་གྷ་ལན་ཌིར་ ༡། ཨོ་རི་སར་ ༢༡། པན་ཇབ་ལ་ ༡༣། ར་ཇེ་སི་ཐན་ལ་ ༢༥། འབྲས་ལྗོངས་ལ་ ༡། ཏ་མིལ་ཎཌོར་ ༣༩། ཏེར་པུ་རར་ ༢། ཡུ་པི་ལ་ ༨༠། ཨུ་དྷ་ར་ཁན་ཌིར་ ༥། ནུབ་བྷང་གྷར་ ༤༢། ཨན་དྷ་མན་དང་ཎ་ཁཱོ་བྷར་གླིང་ཕྲན་ལ་ ༡། ཅན་དྷི་གྷར་ ༡། དྷ་ར་དང་ཎ་གྷར་ཨཝུ་ལིར་ (Dadra and Nagar Haveli) ༡། དྷ་མན་དང་དྷེཝུ་ལ་ ༡། ལག་ཤི་དྷེཝུབ་ལ་ ༡། རྒྱལ་ས་ལྡི་ལིར་ ༡། པོན་ཌོ་ཆུ་རིར་ ༡ བཅས་ཡོད་འདུག ཕྱི་ཟླ་ ༤ པའི་ཚེས་ ༧ ཉིན། ཨ་སམ་དང་ཏེར་པུ་ར་མངའ་སྡེ་ ༢ དང་། ཚེས་ ༩ ཉིན་ཨ་རུ་ན་ཅལ་དང་མེ་ཛོ་རམ། མ་ནི་པུར། མེ་གྷ་ལཱ་ཡཱ། ཎ་གྷ་ལན་ཌི་བཅས་མངའ་སྡེ་ ༥། ཚེས་ ༡༠ ཉིན་ཇམ་མུ་དང་ཀཱ་ཤི་མེར་དང་། ཧར་ཡཱཎ། ཀེ་ར་ལཱ། མཱཧ་ར་ཤི་ཌ། ཇར་ཁནཌི། ཨན་དྷ་མན་དང་ཎ་ཁཱོ་བྷར་གླིང་ཕྲན། རྒྱལ་ས་ལྡི་ལི། ལག་ཤི་དྷེཝུབ་ (Lakshadweep) ཅན་དྷི་གྷ། ཆཱ་དྷིསེ་གྷ། ཡུ་པི། ཨོ་རི་ས། མཱ་དྷེཡ་པར་དྷི་ཤི བྷེ་ཧར་བཅས་མངའ་སྡེ་ ༡༤། ཚེས་ ༡༢ ཉིན་འབྲས་ལྗོངས་དང་ཨ་སམ། ཏེར་པུ་ར་བཅས་མངའ་སྡེ་ ༣། ཚེས་ ༡༧ ཉིན་གྷོ་ཝ་དང་ཇམ་མུ་དང་ཀཱ་ཤི་མེར། མཱཧ་ར་ཤི་ཌ། ནུབ་བྷང་གྷ་ལཱ། ཇར་ཁནཌི། ཆཱ་དྷིསེ་གྷ། ཡུ་པི། ར་ཇེ་སི་ཐན། ཨོ་རི་ས། མ་ནི་པུ། མཱ་དྷེཡ་པར་དྷི་ཤི ཀཱ་ཎཱ་ཀྲ་ཁཱ། བྷེ་ཧར་བཅས་མངའ་སྡེ་ ༡༣ ནང་འོས་བསྡུ་བྱ་རྒྱུ་ཡིན་འདུག ཚེས་ ༢༤ ཉིན་ཨ་སམ་དང་ཇམ་མུ་དང་ཀཱ་ཤི་མེར། མཱཧ་ར་ཤི་ཌ། ནུབ་བྷང་གྷ་ལཱ། ཇར་ཁནཌི། པོན་ཌོ་ཆུ་རིར། ཆཱ་དྷིསེ་གྷ། ཡུ་པི། ཏ་མིལ་ཎཌོ། ར་ཇེ་སི་ཐན། མཱ་དྷེཡ་པར་དྷི་ཤི བྷེ་ཧར་བཅས་མངའ་སྡེ་ ༡༢། ཚེས་ ༣༠ ཉིན་ཨན་དྷ་ར་པར་དྷི་ཤི་དང་ཇམ་མུ་དང་ཀཱ་ཤི་མེར། པན་ཇབ། ནུབ་བྷང་གྷ་ལཱ། དྷ་ར་དང་ཎ་གྷར་ཨཝུ་ལི། དྷ་མན་དང་དྷེཝུ། གྷོ་ཅུ་ར་ཌེ། བྷེ་ཧར་བཅས་མངའ་སྡེ་ ༩། ཟླ་ ༥ པའི་ཚེས་ ༧ ཉིན་ཨན་དྷ་ར་པར་དྷི་ཤི་དང་ཇམ་མུ་དང་ཀཱ་ཤི་མེར། ནུབ་བྷང་གྷ་ལཱ། ཨུ་དྷ་ར་ཁནཌི། ཡུ་པི། ཧེ་མ་ཅལ། བྷེ་ཧར་བཅས་མངའ་སྡེ་ ༧། ཚེས་ ༡༢ ཉིན་ནུབ་གྷང་བྷ་ལཱ་དང་ཡུ་པི། བྷེ་ཧར་བཅས་མངའ་སྡེ་ ༣ ནང་འོས་བསྡུ་བྱ་རྒྱུ་ཡིན་འདུག གསར་འགོད་པ། དོན་གྲུབ།
