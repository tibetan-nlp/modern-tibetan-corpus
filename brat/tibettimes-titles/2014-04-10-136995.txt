སྡེ་དགེར་བོད་མི་ཞིག་འཛིན་བཟུང་།

ཟླ་འདིའི་ཕྱི་ཚེས་ ༨ ཉིན་སྡེ་དགེ་རྫོང་ཡིད་ལྷུང་གྲོང་ཚོ་མ་ཎི་གད་འགོའི་ཁྲོམ་གཞུང་དུ་བོད་མི་ཞིག་གིས་ངོ་རྒོལ་སྐད་འབོད་བྱས་ཏེ་འཛིན་བཟུང་བྱས་ཡོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༨ ཉིན་གྱི་བོད་ནང་གི་ཉིན་གུང་ཆུ་ཚོད་ ༣ པ་ཙམ་ལ། དཀར་མཛེས་ཁུལ་སྡེ་དགེ་རྫོང་ཡིད་ལྷུང་གྲོང་ཚོ་མ་ཎི་གད་འགོའི་ཁྲོམ་གཞུང་དུ། བོད་མི་གཞོན་སྐྱེས་ཤིག་གིས་༧རྒྱལ་བ་བསྟན་འཛིན་རྒྱ་མཚོ་སྐུ་ཚེ་ཁྲི་ལོར་བརྟན་པར་ཤོག བོད་ལ་རང་དབང་དགོས་སོགས་ཀྱི་སྐད་འབོད་གནང་ཡོད་པ་དང་། ཕྱིས་སུ་ཉེན་རྟོག་པས་ཁོང་འཛིན་བཟུང་བྱས་ཡོད་པའི་གནས་ཚུལ་ཐོན་ཡོད་ཀྱང་། ད་ལྟའི་ཆར་བོད་མི་དེའི་མཚན་དང་གནས་སྟངས་གང་ཡང་ཤེས་རྟོགས་ཐུབ་ཀྱི་མི་འདུག སྡེ་དགེ་རྒྱལ་ཡོངས་ལྷན་ཚོགས་ཀྱི་ཚོགས་གཙོ་འཆམ་ར་དྲི་མེད་རྒྱལ་མཚན་ལགས་ཀྱིས་འདི་གར་བརྗོད་པ་ལྟར་ན། མ་ཎི་གད་འགོ་ནི་ཡིད་ལྷུང་གྲོང་ཚོའི་ལས་ཁུངས་ཡོད་ས་དེ་ཡིན་ལ། སྐད་འབོད་དངོས་སུ་མཐོང་མཁན་ཚོས་སྐབས་དེར་ཕོ་གསར་དེ་བྲོས་སོང་ན་ཐར་འགྲོ་ཡང་། གཏན་ནས་མ་བྲོས་པར་ཉེན་རྟོག་པས་འཇུ་བཟུང་མ་བྱས་བར་དུ་སྐད་འབོད་གནང་ཡོད་པ་དང་། དེ་ནས་ཉེན་རྟོག་པས་འཇུ་བཟུང་བྱས་ཏེ། དགོང་མོ་དེར་སྡེ་དགེ་རྫོང་ཐོག་ཏུ་བཙན་ཁྲིད་བྱས་ཡོད་སྐོར་འགྲེལ་བརྗོད་གནང་འདུག ཕྱི་ཕྱོགས་སུ་འབྱོར་བའི་སྐད་འབོད་གནང་སྐབས་ཀྱི་བརྙན་ཐུང་ནང་དུ། བོད་མི་ན་གཞོན་དེས་༧རྒྱལ་བ་བསྟན་འཛིན་རྒྱ་མཚོ་སྐུ་ཚེ་ཁྲི་ལོར་བརྟན་པར་ཤོག་ཅེས་ཡིག་ཆ་མང་པོ་གཏོར་འགྲེམས་གནང་བཞིན་ཡོད་པ་དང་། སྲང་ལམ་གཡས་གཡོན་གྱི་བོད་མི་ཚོས་ཀྱང་ཀི་བཏབ་སྟེ། ཕོ་གསར་ཟེར་ན་ཕོ་གསར་རང་རེད་ཟེར་ནས་ཁོང་ལ་རྒྱབ་སྐྱོར་གནང་བཞིན་འདུག གཞན་ཡང་། བརྒྱུད་ལམ་ཁག་ནས་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། ༢༠༠༦ ལོའི་དབྱར་ཁར་རྒྱ་ནག་གི་ཁེ་ལས་ཁང་ཞིག་གིས། མ་ཎི་གད་འགོར་བཤས་རའི་ལས་སྣོན་བཟོ་གྲྭ་ཞིག་གསར་དུ་བརྒྱབ་པར་ས་གནས་བོད་མི་ཚོས་ངོ་རྒོལ་བྱས་པར་བརྟེན། བོད་མི་ཁག་ཅིག་འཛིན་བཟུང་བྱས་ཏེ་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
