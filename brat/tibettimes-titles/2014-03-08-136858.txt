བོད་སྐད་ཡིག་གི་ལས་དབང་གླེང་པ།

གནས་ཚུལ་བཏང་འབྱོར་བྱུང་བར། གཟའ་མིག་དམར་ནས་ཕུར་བའི་བར་གྱི་ཉིན་ ༣ རིང་། ནགས་ལྷ་རྩོམ་སྒྲིག་ཁང་གི་གོ་སྒྲིག་འོག བཞུགས་སྒར་ངྷ་རམ་ས་ལའི་ཀིརྟི་དགེ་རྩ་ཚོགས་པའི་ཚོགས་ཁང་དུ། ནགས་ཤོད་འབྲི་རུའི་སློབ་མ་རྣམས་ཀྱི་བརྡ་སྤྲོད་རིག་པའི་བགྲོ་གླེང་ཐེངས་གཉིས་པ་དེ་ཚོགས་ཡོད་འདུག ཐེངས་འདིའི་བགྲོ་གླེང་སྟེང་། བོད་ཀྱི་དཔེ་མཛོད་ཁང་གི་རྩོམ་སྒྲིག་པ་ན་ག་སངས་རྒྱས་བསྟན་དར་ལགས་དང་། རྒན་བསྟན་འཛིན་ཆོས་གྲགས་ལགས། ནགས་ལྷ་རྩོམ་སྒྲིག་ཁང་གི་འགན་འཛིན་འབྲི་སྐལ་བཟང་བཅས་ཀྱིས་གཏམ་བཤད་གནང་ཡོད་ཅིང་། གཞན། ས་རཱ་མཐོ་སློབ་དང་ནོར་གླིང་། ཤེས་རབ་དགའ་ཚལ་སློབ་གླིང་། སུ་ཇཱ། གྷོ་པལ་པུར། དེ་མིན་ངྷ་ས་ཉེ་འཁོར་གྱི་གྲྭ་བཙུན་སྔགས་གསུམ་གྱི་དགོན་སྡེ་ཁག་གི་རིགས་གཅིག་རྐང་འཛིན་པ་བཅས་ཁྱོན་སློབ་མ་ ༥༠ ཙམ་མཉམ་ཞུགས་གནང་ཡོད་འདུག བགྲོ་གླེང་གི་བརྗོད་གཞི་ཁག་ནི། ཐོན་མི་སམ་བྷོ་ཊ་མ་ཕེབས་པའི་སྔོན་དུ་བོད་ལ་ཡི་གེ་ཡོད་མེད་དང་། དེའི་སྐབས་ཀྱི་མི་རྣམས་ཕན་ཚུན་བརྡ་སྤྲོད་བྱེད་སྟངས་སོགས་ཀྱི་སྐོར། བོད་ཀྱི་ཡི་གེ་འདི་འཛམ་གླིང་ཐོག་རྩ་ཆེ་ཆུང་ག་འདྲ་ཡིན་པ་དང་། འདི་ལ་དམིགས་བསལ་གྱི་ཁྱད་ཆོས་གང་དག་ཡོད་མེད། རྒྱུན་དུ་ང་ཚོས་བོད་ཀྱི་སྐད་ཡིག་ལ་བདག་གཅེས་ག་འདྲ་བྱེད་དགོས་པའི་སྐོར་སོགས་ཡིན་འདུག གནས་ཚུལ་མཁོ་སྤྲོད་པས་ངོ་སྤྲོད་གནང་བ་ལྟར་ན། ནགས་ལྷ་རྩོམ་སྒྲིག་ཁང་གིས་འདས་པའི་ལོ་ལྔའི་ནང་། <<�ནགས་ལྷའི་སྒྲོན་མེ།>>ཞེས་པའི་དུས་དེབ་འདོན་ཐེངས་ ༥ དང་། གནའ་དེང་ཟུང་གི་རིག་གཞུང་དང་འབྲེལ་བའི་རྩོམ་རིག་སྣ་མང་། ནང་ཆོས་དང་བརྡ་སྤྲོད་རིག་པའི་མཁས་པའི་སློབ་ཁྲིད་ལྟ་ཉན་ཆོག་པའི་རིག་གཞུང་དྲ་མཛོད། www.naglha.net བོད་ཀྱི་བརྡ་སྤྲོད་སྐོར་གྱི་སྙན་རྩོམ་འགྲན་བསྡུར། སློབ་མ་རྣམས་ཀྱི་བརྡ་སྤྲོད་རིག་པའི་བགྲོ་གླེང་སོགས་གོ་སྒྲིག་ཞུས་ཡོད་འདུག
