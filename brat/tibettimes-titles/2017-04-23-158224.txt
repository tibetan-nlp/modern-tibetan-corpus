པར་ཆས་སྒོ་བརྒྱབ་ནས་མི་སྲོག་སྐྱོབ་པ།

སི་ཨན་ཨན་གསར་ཁང་གིས་ ༢༠༡༧།༤།༡༨ ཉིན་སྤེལ། སི་ཨན་ཨན་གསར་འགོད་པས་ཉེ་ལམ་སེ་རཱི་ཡར་འབར་རྫས་དོན་རྐྱེན་བྱུང་བའི་ཁྲོད་ཀྱི་རྨས་མ་ཚོར་རོགས་རམ་བྱེད་མཁན་པར་པ་གཉིས་ལ་དངོས་སུ་ཐུག་འཕྲད་བྱས་ཏེ་ཡུལ་དངོས་ཀྱི་གནས་ཚུལ་རྒྱུས་ལོན་བྱས། སེ་རཱི་ཡའི་ནང་བྱུང་དང་འབྱུང་བཞིན་པའི་ཡ་ང་བའི་དམག་འཁྲུག་གི་པར་དེ་དག་གིས་ང་ཚོའི་སྙིང་ཁར་གྲི་བཙུག་པ་ལྟ་བུའི་ན་ཟུག་སྟེར། ང་ཚོས་ ༢༠༡༥ ལོར་བྱིས་པ་ཆུང་ཆུང་ཨཱ་ལཱན་ཀཱུར་ཌི་(Alan Kurdi)ཡིས་རང་གི་རྒྱལ་ཁབ་བཞག་ནས་བྲོས་བྱོལ་དུ་འགྲོ་སྐབས་རྒྱ་མཚོའི་ནང་བྱིང་ནས་ཤི་བའི་རྗེས་སུ་ཊར་ཀེའི་རྒྱ་མཚོའི་འགྲམ་ངོགས་ཀྱི་ཕྱེ་ཐང་ན་རོ་ཁ་སྦུབ་ཏུ་ལོག་ཡོད་པའི་པར་དེ་མཐོང་ཡོད། ལོ་སྔོན་མར་ཨ་ལིབ་པཱོ་(Aleppo)ལུང་པའི་ཁྱིམ་ཚང་ཞིག་ཏུ་འབར་རྫས་ཐོར་བྱུང་སྟེ་བྱིས་པ་ཆུང་ཆུང་ཨོམ་རིང་ཌ་ནྀཥ་(Omran Daqneesh)ཡི་ལུས་ཡོངས་ཁྲག་དང་ཐལ་རྡུལ་གྱིས་གཡོགས་ཏེ་སྨན་ཁང་གི་རླངས་འཁོར་ནང་ཡོད་པའི་པར་ཡང་ང་ཚོས་མཐོང་ཡོད། ང་ཚོས་ད་དུང་། བདུན་ཕྲག་སྔོན་མར་སླར་ཡང་སེ་རཱི་ཡའི་གྲོང་གསེབ་ཅིག་ཏུ་མི་མང་པོ་ཞིག་ཉེན་གཡོལ་དུ་འགྲོ་སྐབས་སྤྱི་སྤྱོད་རླངས་འཁོར་ནང་འབར་རྫས་གས་གཏོར་བྱུང་སྟེ་མི་ ༡༢༦ འདས་གྲངས་སུ་ཕྱིན་པའང་མཐོང་ཡོད། དོན་རྐྱེན་དེ་བྱུང་སྐབས་པར་པ་ཨ་ཝཊ་ཨལ་ཀཱ་ཌར་ཧ་ཝཀ་(Abd Alkarder Habak)ལགས་ཉེ་འགྲམ་དུ་ཡོད་འདུག ཁོང་དང་ཁོང་གི་རོགས་པ་ཚོས་འབར་རྫས་གས་ཐོར་བྱུང་མ་ཐག་རྨས་མ་ཚོར་རོགས་རམ་བྱས་ཡོད་འདུག ཧ་ཝཀ་གིས་སི་ཨན་ཨན་གསར་ཁང་ལ། “སྐབས་དེར་ཧ་ཅང་འཇིགས་སུ་རུང་བ་ཞིག་རེད། དམིགས་བསལ་གྱིས་བྱིས་པ་ཆུང་ཆུང་ཚོས་ངུ་འབོད་བྱེད་བཞིན་མིག་གི་མཐོང་སར་ཤི་འགྲོ་བ་དེ་སེམས་ཀྱིས་བཟོད་ཀྱི་མི་འདུག ང་དང་ངའི་རོགས་པ་ཚོས་པར་ཆས་ཁ་བརྒྱབ་ནས་རྨས་མར་རོགས་པ་བྱས་པ་ཡིན།” ཞེས་བཤད། ཁོ་རང་ཐོག་མར་ལག་པ་འཆིང་པའི་བྱུས་པ་དེ་ཤི་ཚར་ཡོད་པ་དང་། དེ་ནས་བྱིས་པ་གཞན་དག་སྐྱོབ་པར་ཕྱིན་སྐབས་མི་ཞིག་གིས་སྐད་ཤུགས་ཆེན་པོས། བྱིས་པ་དེ་དག་ལ་ལག་པ་མ་འཆིང་། སྔ་མོ་ནས་ཤི་ཚར་འདུག་ཟེར། འོན་ཀྱང་། ཁོས་སར་འགྱེལ་ཡོད་པའི་བྱིས་པ་ཞིག་ཡར་བཀྱགས་ནས་ལྟ་སྐབས་དབུགས་ཅུང་ཟད་འདུག་སྟབས། བྱུས་པ་དེ་བང་དུ་བཟུང་ནས་བདེ་འཇགས་ཀྱི་ས་ཆ་ཞིག་ཏུ་བསྐྱལ་བ་ཡིན་ཟེར། ཁོས། “སྐབས་དེར་བྱིས་པ་དེས་མིག་ཧྲིག་ཧྲིག་གིས་ང་ལ་ལྟ་བཞིན་ཤུགས་ཀྱིས་འཐམ་ཡོང་གི་འདུག” ཅེས་བརྗོད་བྱུང་། ཧ་ཝཀ་གིས་བྱིས་པ་དེ་པང་དུ་བཟུང་ནས་རྒྱུག་བཞིན་པའི་པར་དེ་རྒྱག་མཁན་ཁོའི་རོགས་པ་མཱུ་ཧམ་མ་ཌི་ཨལ་རཱ་གྷིབ་(Muhammad Alrageb)ལགས་ཡིན་འདུག ཨལ་རཱ་གྷིབ་ཀྱིས། “ཧ་ཝཀ་གིས་པར་ཆས་དཔུང་པར་བསྐོན་ཏེ་དཔའ་ངར་ཆེན་པོས་བྱིས་པ་བྲང་ཁར་སྦྱར་ནས་སྨན་ཁང་གི་རླངས་འཁོར་ཡོད་སར་རྒྱུག་གིན་འདུག”ཟེར། ཨལ་རཱ་གྷིབ་ལགས་ཀྱིས་ཀྱང་ཐོག་མར་རྨས་མ་རྣམས་ལ་རོགས་པ་བྱས་ཡོད་ཀྱང་། རྗེས་སུ་སླར་ཡང་པར་རྒྱག་མགོ་བརྩམས་འདུག ཁོས་“ ངས་འཇིགས་སུ་རུང་བའི་གནས་ཚུལ་དེ་དག་ཚང་མ་བརྙན་ཕབ་བྱས་ན་འདོད་སོང་། གང་ཡིན་ཟེར་ན། བརྙན་དེ་དག་གིས་རྗེས་སུ་གནད་དོན་དེའི་སྐོར་ལ་ཕན་ཐོགས་ཡོང་གི་རེད་དྲན་སོང་།”ཞེས་ཟེར། ད་དུང་ཁོང་གིས་ “གསར་འགོད་པ་ལོ་ཆུང་ཞིག་(ཧ་ཝཀ་)ཞིག་གིས་རྨས་མ་རྣམས་ཀྱི་ཚེ་སྲོག་སྐྱོབ་བཞིན་པ་དེ་མཐོང་ནས་ང་རང་སྤོབས་པ་སྐྱེས་སོང་།” ཞེས་ཁ་སྣོན་བྱས་བྱུང་། ཧ་ཝཀ་ལགས་ཀྱིས་བྱིས་པ་དེ་སྨན་ཁང་གི་རླངས་འཁོར་ནང་བཅུག་མ་ཐག་ཕྱིར་འབར་རྫས་ཐོར་སར་རྒྱུག་ཡོང་སྐབས། ཡང་བསྐྱར་བྱིས་ཆུང་ཆུང་ཞིག་ཤི་བའི་རོ་ཁ་སྦུབ་ཏུ་ལོག་ཡོད་པ་མཐོང་སྟེ། ཧ་ཝཀ་སེམས་པ་དྲག་ཏུ་སྐྱོ་ནས་པུས་མོ་ས་རུ་བཙུགས་ནས་ངུ་འབོད་བྱས་ཡོད་འདུག ཧ་ཝཀ་ལགས་སེམས་ཚོར་དྲག་པོས་ངུ་ངག་འདོན་བཞིན་པའི་པར་དེ་ཁོའི་རོགས་པ་ཨལ་རཱ་གྷིབ་ལགས་ཀྱིས་བརྒྱབ་པ་ཡིན་འདུག ཧ་ཝཀ་ལགས་ཀྱིས་སི་ཨན་ཨན་གསར་ཁང་ལ་ “ང་སྐབས་དེར་སྐྱོ་གདུང་གི་ཚོར་བ་དྲག་པོས་མནན་སོང་།” “ང་དང་ངའི་རོགས་པ་ཚོས་མཐོང་བའི་གནས་ཚུལ་དེ་དག་ཚིག་གིས་མཚོན་ཐུབ་པ་ཞིག་མ་རེད།”ཅེས་བརྗོད་སོང་། གསར་འགོད་པ། ཉི་མ་ཚེ་རིང་།
