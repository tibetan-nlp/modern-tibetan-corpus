སུད་སིར་སྐུའི་གྱ་སྟོན་མཛད་སྒོ་བསྡུས།

སུད་སིའི་རི་ཁོན་དུ་མདོ་སྨད་ས་རྒྱུན་གྱི་གོ་སྒྲིག་འོག་༧གོང་ས་མཆོག་དགུང་གྲངས་ ༨༠ ཞེས་པའི་སྐུའི་གྱ་སྟོན་མཛད་སྒོ་ཞིག་ཚོགས་ཡོད་འདུག ཕྱི་ཟླ་ ༤ ཚེས་ ༡༢ ཉིན་སུད་སི་ས་གནས་མདོ་སྨད་རྒྱུན་ལས་ཀྱིས་གོ་སྒྲིག་འོག སུད་སིའི་རི་ཁོན་ནང་༧གོང་ས་མཆོག་དགུང་གྲངས་བརྒྱད་བཅུར་ཕེབས་པའི་སྐུའི་གྱ་སྟོན་མཛད་སྒོ་ཞིག་ཚོགས་ཡོད་པ་དང་། མཛད་སྒོའི་སྐུ་མགྲོན་དུ་ཡུ་རོབ་སྤྱི་འཐུས་བཀྲེས་མཐོང་དབང་པོ་དང་། སུད་སི་སྐུ་ཚབ་དོན་གཅོད། ཡུ་རོབ་སྤྱི་ཟུར་ཁག་ཅིག་བཅས་བོད་ཆོལ་ཁ་གསུམ་གྱི་མི་མང་ ༦༠༠ ལྷག་འདུ་འཛོམས་གནང་ཡོད་འདུག ཉིན་དེའི་མཛད་རིམ་གྱི་ཐོག་མར་བསང་གསོལ་བཏང་བ་དང་། རི་ཁོན་ཆོས་འཁོར་དགོན་གྱི་དགེ་འདུན་པ་རྣམས་ཀྱིས་གཙོས་ཞབས་རིམ་སྨོན་ལམ་གསུངས་བ། སྐུ་པར་མཐོང་གྲོལ་རིན་པོ་ཆེ་གདན་ཞུ། མནྜལ་རྟེན་གསུམ་ཕུལ་བ་བཅས་གྲུབ་རྗེས། ས་གནས་མདོ་སྨད་རྒྱུན་ལས་ཚོགས་གཞོན་དཀོན་མཆོག་བཀྲ་ཤིས་ལགས་ཀྱིས་མཛད་རིམ་ངོ་སྤྲོད་དང་། ས་གནས་མདོ་སྨད་རྒྱུན་ལས་ཚོགས་གཙོ་དབང་རྒྱལ་ལགས་ཀྱིས་དགའ་སྟོན་མཛད་སྒོ་ངོ་སྤྲོད་སོགས་གནང་རྗེས། ཆོལ་ཁ་གསུམ་གྱི་གཞས་སྣ་འཁྲབ་སྟོན་དང་། རྒྱན་གོས་འཁྲབ་སྟོན། བྱིས་པའི་འཁྲབ་སྟོན་སོགས་གནང་སོང་། སྐབས་དེར་མཛད་སྒོ་གོ་སྒྲིག་པ་དང་ས་གནས་མདོ་སྨད་རྒྱུན་ལས་ཚོགས་གཙོ་ཚེ་རིང་དབང་རྒྱལ་ལགས་ཀྱིས་གཏམ་བཤད་གནང་དོན། མདོ་སྨད་ཁུལ་གྱི་ཡུལ་སྲོལ་གོམས་གཤིས་ལ་གྱ་སྟོན་གྱི་འགྲོ་ལུགས་ཡོད་པ་ལྟར། སུད་སི་ས་གནས་སུའང་མཛད་སྒོ་ཞིག་གོ་སྒྲིག་ཞུ་ཐུབ་ན་ལེགས་ཚུལ་གྱི་དགོངས་ཚུལ་བྱུང་བ་ལྟར། དེ་རིང་སུད་སིའི་ནང་བཞུགས་པའི་ཆོལ་ཀ་གསུམ་གྱི་མི་མང་དགྱེས་འཛོམས་གནང་བ་ལྟར། དེ་རིང་འདིར་ཆོལ་ཁ་གསུམ་གྱི་མི་མང་ ༦༠༠ ལྷག་ཕེབས་འདུག ས་གནས་མདོ་སྨད་མི་མང་གིས་ཐེངས་འདིའི་མཛད་སྒོའི་ཞལ་ལག་རིན་མེད་གཏོང་དགོས་པའི་བསམ་ཚུལ་བྱུང་བ་ལྟར། མཛད་སྒོའི་འགྲོ་གྲོན་སོགས་མང་གིས་ཞལ་འདེབས་ཕུལ་ཏེ། མི་མང་ཡོངས་ལ་ཞལ་ལག་རིན་མེད་བཏང་པ་དང་། ཉིན་མོ་དམིགས་བསལ་འདི་ལ་མི་མང་ཧ་ཅང་དགའ་བོ་དང་སྤྲོ་བོ་བྱུང་འདུག ཅེས་བརྗོད་སོང་།   སུད་སི་ས་གནས་ཀྱི་མི་ཧ་ཅང་མང་པོ་ཞིག་རྒྱ་གར་དྷ་རམ་ས་ལར་སྐུའི་གྱ་སྟོན་ལ་བཅར་གྱི་ཡོད་ཀྱང་། ལས་ཀ་དང་གནས་ཚུལ་འདྲ་མིན་གྱི་དབང་གིས་ཕར་འགྲོ་མི་ཐུབ་མཁན་གང་འཚམས་ཡོང་གི་ཡོད་པ་དང་། ཁོང་ཚོར་ཡང་སྐུའི་གྱ་སྟོན་ལ་དགའ་སྟོན་གྱི་རྣམ་པ་ཞིག་སྟོན་འདོད་ཡོད་སྟབས། ང་ཚོས་དུས་ཚོད་སྔ་ཚུད་ནང་གོ་སྒྲིག་ཞུ་མ་ཐུབ་པ་ཡིན་ན། སྐུའི་འཁྲུངས་སྐར་ཟིན་ཚར་རྗེས་གོ་སྒྲིག་ཞུས་ན་མི་མང་ལ་འདོད་པ་མ་ཁེངས་པ་ཡོང་རྒྱུ་དང་། འཁྲུངས་སྐར་ཉིན་རང་ལ་གོ་སྒྲིག་ཞུ་དགོས་ན་མི་འཛོམ་མིན་བཅས་ལ་བརྟེན་ཏེ་དེ་རིང་གོ་སྒྲིག་ཞུས་པ་ཡིན། ཞེས་བརྗོད་བྱུང་། མཛད་སྒོ་འདིའི་ཐོག་སུད་སི་ས་གནས་མདོ་སྨད་རྒྱུན་ལས་གྱི་འགྲོ་གྲོན་དང་རྩོམ་སྒྲིག་བྱས་པའི་༧གོང་ས་མཆོག་དགུང་ན་ཕྲ་དུས་ནས་ད་ལྟའི་བར་བོད་ཀྱི་བསྟན་སྲིད་མི་རིགས་ཆེད་མཛད་འཕྲིན་བསྐྱངས་དང་སྐྱོང་བཞིན་པའི་འོད་སྡེར་ཞིག་ཀྱང་འགྲེམ་སྤེལ་གནང་ཡོད་འདུག གནས་ཚུལ་གཏོང་མཁན། ཙ་ཡུས་ཕུན་ཚོགས།
