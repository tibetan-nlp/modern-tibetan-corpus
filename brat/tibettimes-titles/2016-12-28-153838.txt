རྒྱ་གར་བ་ཞིག་གིས་སྤྱི་ལེ་ ༡༢༠༠
ལ་བོད་དོན་ཁྱབ་བསྒྲཊ་སྐོར་སྐྱོད་གནང་།

རྒྱ་གར་བོད་དོན་རྒྱབ་སྐྱོར་བ་སྐུ་ཞབས་ཞི་རེ་སན་ཌེ་ཤི་མ་ཤི་རམ་ལགས་ཀྱིས་རྒྱ་གར་གྱི་གྲོང་ཁྱེར་ནག་པུར་ནས་གྲོང་ཁྱེར་གྲོང་གསེབ་མང་པོ་བརྒྱུད་སྤྱི་ལེ་ ༡༢༠༠ ཙམ་ལ་བོད་དོན་ཁྱབ་བསྒྲགས་རྐང་འཁོར་སྐོར་སྐྱོད་གནང་། རྒྱ་གར་བོད་དོན་རྒྱབ་སྐྱོར་བ་སྐུ་ཞབས་ཞི་རེ་སན་ཌེ་ཤི་མ་ཤི་རམ་(shri sandesh meshram)ལགས་ཀྱིས། འདི་ལོའི་ཕྱི་ཟླ་ ༡༢ པའི་ཚེས་ ༡༠ སྟེ་༧གོང་ས་མཆོག་ལ་ཞི་བདེ་གཟེངས་རྟགས་ཐོབ་པའི་ཉིན་མོ་དང་བསྟུན་ཏེ། རྒྱ་གར་གྱི་གྲོང་ཁྱེར་ནག་པུར་(Nagpur)ནས་གནས་མཆོག་རྡོ་རྗེ་གདན་གྱི་གྲོང་ཁྱེར་བྷོ་ཌི་གྷ་ཡ་སོགས་བརྒྱུད་སྤྱི་ལེ་ ༡༢༠༠ ཙམ་ལ་རྐང་འཁོར་སྐོར་སྐྱོད་ཀྱིས་བོད་དོན་ཁྱབ་བསྒྲགས་ཀྱི་ལས་འགུལ་སྤེལ་ཡོད་འདུག ཟླ་འདིའི་ཚེས་ ༢༡ ཉིན་ཝ་ར་ནཱ་སའི་གྲོང་ཁྱེར་ས་ར་ན་ཐི་རུ་འབྱོར་སྐབས། ས་གནས་དེ་གའི་བོད་པའི་རང་བདེན་ཚོགས་ཆུང་དང་ཝར་ན་མཐོ་སློབ་ཀྱི་བོད་རིགས་སློབ་མའི་སྐུ་ཚབ། གཞོན་ནུ་དང་བུད་མེད། བོད་དགོན་བཤད་སྒྲུབ་འདོད་རྒུ་འཁྱིལ་སོགས་ཀྱིས་ཁོང་ལ་ཕེབས་བསུ་ཞུས་ཡོད་འདུག ཚེས་ ༢༤ ཉིན་བོད་པ་ཚོའི་ཕེབས་སྐྱེལ་ཁྲོད་ཝ་ར་ན་སི་ནས་ཐོན་ཏེ་ཕྱིར་རྡོ་རྗེ་གདན་གྱི་ཕྱོགས་ལོག་ཡོད་པ་དང་། དེ་རིང་ཕྱི་ཚེས་ ༢༨ ཉིན་གནས་མཆོག་རྡོ་རྗེ་གདན་དུ་འབྱོར་རྒྱུ་ཡིན་ཞིང་། རྡོ་རྗེ་གདན་ནས་དུས་འཁོར་དབང་ཆེན་རིང་སྤྱི་ཚོགས་ཞབས་ཞུའི་ཕྱག་ལས་གནང་རྒྱུ་ཡིན་འདུག འདི་ནི་ཁོང་གི་རྐང་འཁོར་སྐོར་སྐྱོད་ཀྱི་བོད་དོན་ཁྱབ་བསྒྲགས་ལས་འགུལ་ཐེངས་གཉིས་པ་ཡིན་ཞིང་། ཐེངས་དང་པོར་ཕྱི་ལོ་ ༢༠༡༤ ལོའི་ཟླ་ ༡༢ ཚེས་ ༡༠ ཉིན། རྒྱལ་སྤྱིའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་གི་ཉིན་མོ་དང་བསྟུན་ཏེ་ནག་པུར་ནས་མགོ་བཙུགས་ཏེ་རྒྱ་གར་གྱི་མངའ་སྡེ་ལྔ་བརྒྱུད་ཁྱོན་སྤྱི་ལེ་ ༤༠༠༠ ཙམ་ལ་རྐང་འཁོར་སྐོར་སྐྱོད་གནང་སྟེ་བོད་དོན་དྲིལ་བསྒྲགས་ལས་འགུལ་སྤེལ་ཡོད་པ་རེད། སྐུ་ཞབས་ཞི་རེ་སན་ཌེ་ཤི་མ་ཤི་རམ་ལགས་ཀྱིས་འདས་པའི་ལོ་ངོ་ ༤༠ ལྷག་རིང་བོད་དོན་ཐོག་རྒྱབ་སྐྱོར་གནང་ཡོད་ཅིང་། ༡༩༩༦ ལོའི་ཟླ་ ༢ པའི་ནང་བཞུགས་སྒར་དྷ་རམ་ས་ལ་ནས་རྒྱ་གར་རྒྱལ་ས་ལྡི་ལིའི་བར་ཞི་བའི་གོམ་བགྲོད་གནང་སྐབས་ཁོ་རང་མཉམ་ཞུགས་གནང་ཡོད་པ་མ་ཟད་གོམ་བགྲོད་དེའི་ཁྲོད་རྒྱ་གར་བ་ཁོ་རང་གཅིག་པུ་ཡིན་པ་དང་། ལོ་དེར་ཧ་མ་ར་ཞེ་ཊར་ནས་རྒྱ་གར་རྒྱལ་ལ་ལྡི་ལིའི་བར་ཞི་བའི་གོམ་བགྲོད་ཐེངས་གཉིས་པ་གནང་སྐབས་ཀྱང་མཉམ་ཞུགས་གནང་ཡོད་པ་རེད། སྐུ་ཞབས་ཞི་རེ་སན་ཌེ་ཤི་མ་ཤི་རམ་ལགས་ནི། ནག་པུར་འཕགས་བོད་མཛའ་འབྲེལ་ཚོགས་པའི་སྤྱི་ཁྱབ་དྲུང་ཆེ་ཡིན་འདུག
