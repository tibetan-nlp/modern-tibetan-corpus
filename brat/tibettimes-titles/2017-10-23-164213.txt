རྒས་ཡོལ་ཏང་ཡོན་ཚོ་མྱུལ་མར་གྱུར།

རྒྱ་ནག་གཞུང་གིས་དཔུང་རྟགས་དམར་པོ་ཐོགས་པའི་རྒས་ཡོལ་ཏང་ཡོན་འབུམ་ ༨ དང་ཁྲི་ ༥  གླས་ཏེ་ཚོགས་ཆེན་ཕྱི་ལོགས་ཀྱི་གནས་ཚུལ་ལ་མྱུལ་ཞིབ་བྱེད་དུ་འཇུག་བཞིན་ཡོད་འདུག དེ་རིང་ཧོང་ཀོང་རང་དབང་གསར་ཁང་གི་གནས་ཚུལ་ལྟར་ན། རྒྱ་ནག་དམར་ཤོག་ཚོགས་པའི་ཚོགས་ཆེན་འཚོགས་བཞིན་པའི་སྐབས་འདིར། རྒྱ་ནག་གཞུང་གིས་མི་དམངས་ཚོགས་ཁང་ཆེན་མོའི་ཕྱི་ཕྱོགས་དང་དོགས་ཡོད་ས་གནས་ཁག་ཏུ་ཞིབ་བཤེར་ས་ཚིགས་ཆེད་འཛུགས་ཀྱིས། ཉེན་རྟོག་པ་དང་དམག་མི་འཕོན་ཆེན་གྱིས་ལྟ་རྟོག་ཞིབ་འཇུག་བྱེད་བཞིན་ཡོད་པར་མ་ཟད། དཔུང་རྟགས་དམར་པོ་ཐོགས་པའི་རྒས་ཡོལ་ཏང་ཡོན་ ༨༥༠༠༠༠ ལྷག་གླས་ཏེ། མྱུལ་ཞིབ་བྱེད་དུ་འཇུག་བཞིན་ཡོད་པ་དང་། ཁོ་ཚོའི་ལས་འགན་གཙོ་བོ་ནི་ཚོགས་ཆེན་ཕྱི་ལོགས་ཀྱི་གནས་ཚུལ་ལ་མྱུལ་ཞིབ་དང་བདེ་འཇགས་ལ་ལྟ་རྟོག ལྷག་པར་དུ་ཕྱིའི་རྒྱལ་ཁབ་ཀྱི་གསར་འགོད་པ་ཚོའི་འགྲོ་འདུག་སྤྱོད་གསུམ་ལ་གསང་བའི་སྒོ་ནས་ལྟ་ཞིབ་བྱ་རྒྱུ་དེ་ཡིན་ལ། ཟླ་གཅིག་རིང་ཏང་ཡོན་གཅིག་ལ་རྒྱ་སྒོར་ ༣༠༠༠ གྱི་གླ་ཆ་སྤྲོད་ཀྱི་ཡོད་པ་དང་། རྒྱལ་ཞེན་རིང་ལུགས་ཀྱི་ཏང་ཡོན་ཁག་ཅིག་གླ་ཆ་མེད་པའི་དང་བླངས་པ་ཡིན་འདུག རང་མིང་བརྗོད་མེད་པའི་ཏང་ཡོན་བུད་མེད་ཅིག་གིས། གོང་རིམ་ནས་ང་ཚོར་དཔུང་རྟགས་དམར་པོ་འདི་སྤྲད་རྗེས། ང་ཚོ་ལས་ཀར་འཇུག་བཞིན་ཡོད། ཅེས་བཤད་འདུག རུས་མིང་ལ་རཱིན་ (Ren) ཟེར་བའི་ཏང་ཡོན་ཞིག་རྒན་པ་ཞིག་གིས། ད་ལྟ་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་གནད་དོན་སེལ་ཐུབ་མེད་ནའང་། ཕྱི་ཚུལ་ནས་བལྟས་ན་ཧ་ཅང་ལྷིང་འཇགས་ཡིན་པའི་ཚོར་སྣང་སྟེར་བ་དང་། དེ་སྔོན་གུང་གསེང་སྐབས་སྣ་འཛོམས་ཁྲོམ་རར་འགྲོ་དུས། དཔོན་རིགས་ཚོས་རྒྱུ་ངོམས་བྱེད་པ་མཐོང་རྒྱུ་འདུག་ལ། ཁོ་ཚོས་མི་མང་གི་དངུལ་བེད་སྤྱོད་གཏོང་གི་ཡོད་པ་མི་མང་ཚང་མས་ཧ་གོ་ཡི་ཡོད་པ་རེད། ཅེས་བཤད་འདུག དམར་ཤོག་ཚོགས་པའི་ཚོགས་ཆེན་མགོ་བཙུགས་པ་དང་བསྟུན། རྒྱ་ནག་རྒས་ཡོལ་ཏང་ཡོན་ལས་བྱེད་ལྟ་སྐྱོང་ཁང་གི་གོ་སྒྲིག་འོག ཏང་ཁག་ ༡༦ དང་། ཏང་ཡོན་རྒས་ཡོལ་བ་ ༣༤༠ ལ་ཚོགས་ཆེན་གྱི་གནས་ཚུལ་དང་། ཞི་ཅིན་ཕིན་གྱི་གཏམ་བཤད་སློབ་སྦྱོང་བྱེད་དུ་བཅུག་སྟེ། མིག་གིས་མཐོང་བའི་འོན་པ་ཚོར་གཟུགས་མཐོང་གློག་བརྙན་ལྟ་རུ་འཇུག་གི་ཡོད་པ་དང་། ཞ་ར་རྣ་བས་གོ་མཁན་ཚོར་རླུང་འཕྲིན་སོགས་ལ་ཉན་དུ་འཇུག་བཞིན་ཡོད་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
