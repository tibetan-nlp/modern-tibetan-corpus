གད་སྙིཊ་ཤིག་རེད་བློ་བདེ་པོ་བྱོས།

ཁ་སང་ཨ་ཅག་ཅིག་གིས་གླིང་སྐོར་ལམ་ནས་རྙེད་པའི་ཅ་དངོས་དེར་རྒྱ་གར་བདེ་སྲུང་ལས་ཁུངས་ཀྱིས་བརྟག་དཔྱད་བྱས་ཏེ་འབར་རྫས་རིགས་སོགས་ཉེན་ཚབས་ཅན་གྱི་དངོས་པོ་གང་ཡང་མིན་པར་གད་སྙིགས་ཤིག་ཡིན་སྐོར་འགྲེལ་བརྗོད་གནང་འདུག དེ་རིང་འདི་གའི་གསར་འགོད་པས་ཁ་པར་བརྒྱུད་བདེ་སྲུང་དྲུང་འཕར་ཀརྨ་རིན་ཆེན་ལགས་སུ། ཁ་སང་གླིང་སྐོར་ལམ་ནས་རྙེད་པའི་ཅ་དངོས་དེར་རྒྱ་གར་གཞུང་གིས་བརྟག་དཔྱད་གནང་རྗེས་གསལ་བཤད་ག་རེ་གནང་ཡོད་དྲིས་པར་ཁོང་གིས། ཅ་དངོས་དེར་རྒྱ་གར་བདེ་སྲུང་ལས་ཁུངས་ཀྱིས་ཆེད་ལས་པ་སྐད་བཏང་ནས་བརྟག་དཔྱད་གནང་བ་བརྒྱུད། འབར་རྫས་རིགས་སོགས་ཉེན་ཚབས་ཅན་གྱི་ཅ་དངོས་གང་ཡང་མིན་པར་གད་སྙིགས་ཤིག་ཡིན་སྐོར་འགྲེལ་བརྗོད་གནང་སོང་པས། ཚང་མས་བློ་བདེ་པོ་གནང་རོགས། ཞེས་བརྗོད་བྱུང་། ཡང་ཁོང་གིས་གནས་ཚུལ་འདི་རིགས་ལ་བརྟག་དཔྱད་མ་བྱས་པར་གང་བྱུང་བརྒྱུད་བསྒྲགས་བྱས་པ་ཡིན་ན། སྤྱི་ཚོགས་ཀྱི་ལྷིང་འཇགས་ལ་གནོད་པ་མ་ཟད། བརྒྱུད་བསྒྲགས་བྱེད་མཁན་དོ་བདག་ལའང་ཁྲིམས་ཐོག་ནས་དཀའ་ངལ་འཕྲད་སྲིད་སྐོར་བརྗོད་དོན། སྤྱི་ཚོགས་ཀྱི་དྲ་ལམ་ཁག་གི་ནང་གནས་ཚུལ་ཞིབ་ཕྲ་མ་བརྟག་མ་དཔྱད་པར་གང་དྲན་བཀོད་རྒྱུ་དེ་རྩ་བ་ནས་ཡག་པོ་མི་འདུག གནས་ཚུལ་ཆ་ཚང་རྟོགས་རྗེས་མང་ཚོགས་ལ་བརྒྱུད་བསྒྲགས་གནང་བ་ཡིན་ན་འོས་པ་ཞིག་རེད། དེ་མིན་གང་བྱུང་བརྒྱུད་བསྒྲགས་གནང་བ་ཡིན་ན་སྤྱི་ཚོགས་ནང་དོགས་འདྲོག་ཡོང་རྒྱུ་དང་། དེ་ཡང་འབར་རྫས་རིགས་དང་། ལྷག་པར་བཞུགས་སྒར་གླིང་སྐོར་ཉེ་འཁོར་དུ་རྙེད་པ་རེད་ཅེས་ཧ་ལམ་ཆ་ཚང་ཧ་གོ་གི་ཡོད་པ་ལྟ་བུ་བྱས་ཏེ། སོ་སོའི་ངོས་ནས་ཞིབ་འཇུག་མཐིལ་ཕྱིན་བྱས་པ་ལྟ་བུ་གནང་ན་རྩ་བ་ནས་ཡག་པོ་མི་འདུག དེས་ང་ཚོའི་སྤྱི་ཚོགས་ཀྱི་ལྷིང་འཇགས་ལ་གནོད་པ་མ་ཟད། དོ་བདག་རང་ཉིད་ལ་ཁྲིམས་ཐོག་ནས་དཀའ་ངལ་འཕྲད་སྲིད་པ་རེད། གང་ཡིན་ཞེ་ན་གནས་ཚུལ་འདི་རིགས་ཧ་ཅང་ཉེན་ཚབས་ཆེན་པོ་ཡིན་པར་བརྟེན། མ་འོངས་པར་ཚང་མས་ཐུགས་གཟབ་གནང་རོགས་ཞུ་རྒྱུ་ཡིན། ཞེས་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
