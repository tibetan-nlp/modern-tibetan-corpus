ཆུ་ཤུཊ་གློག་ཁང་ ༤༩ འཛུཊ་འཆར་ཡོད།

ཁ་སང་ནས་བོད་ཁུལ་གྱི་ཆུ་ཤུགས་གློག་ཁང་ཆེ་ཤོས་གཉིས་པ་ཆབ་མདོ་མགོ་རྡོག་ཆུ་ཤུགས་གློག་ཁང་གི་འཕྲུལ་ཚོ་དང་པོའི་གློག་འདོན་འཁོར་སྐྱོད་མགོ་བརྩམས་ཡོད་འདུག   ཁ་སང་ཕྱི་ཚེས་ ༢༧ ཉིན། ཆབ་མདོ་གྲོང་ཁྱེར་ཁ་རུབ་ཁུལ་ཚེ་དབལ་གྲོང་ཚོའི་ཉེ་འཁྲིས་ཀྱི་རྫ་ཆུའི་བརྒྱུད་དུ། རྒྱ་གཞུང་གིས་འདས་པའི་ལོ་ངོ་ ༣ གྱི་སྔོན་ནས་བཟུང་འཛུགས་སྐྲུན་བྱེད་བཞིན་པའི་མགོ་རྡོག་ཆུ་ཤུཊ་གློག་ཁང་གི་འཕྲུལ་ཚོ་དང་པོའི་གློག་འདོན་འཁོར་སྐྱོད་མགོ་བརྩམས་ཡོད་པ་དང་། ཨར་ལས་བཟོ་སྐྲུན་དེར་ཁྱོན་རྒྱ་སྒོར་དུང་ཕྱུར་ ༣༨།༣ བཏང་ཡོད་ཅིང་། སྒྲིག་སྦྱོར་བྱས་པའི་འཕྲུལ་འཁོར་ནས་གློག་གཏོང་ཚད་ཆན་ཝ་ཁྲི་ ༡༦།༥ ཟིན་གྱི་ཡོད་སྟབས། བོད་ཁུལ་གྱི་རྫམ་ཆུ་ཤུགས་གློག་ཁང་གི་འཕྲོས་ཀྱི་གློག་ཁང་ཆེ་ཤོས་ཨང་གཉིས་པ་དེ་ཡིན་སྐོར་བསྒྲགས་འདུག   ཧྭ་ནེང་མགོ་རྡོག་ཆུ་ཤུགས་གློག་ཁང་ཚད་ཡོད་ཚོང་གཉེར་ཁང་གི་སྤྱི་ཁྱབ་ཏོན་ཁྲོན་གྱིས་གསར་འགོད་པར། མགོ་རྡོག་ཆུ་ཤུགས་གློག་ཁང་གི་འཕྲུལ་ཚོ་དང་པོའི་འཁོར་སྐྱོད་དེས་ཆབ་མདོ་གྲོང་ཁྱེར་གྱི་ཐོན་སྐྱེད་དང་འཚོ་བའི་ཐད་གློག་དཀོན་པའི་དཀའ་གནད་ཐག་གཅོད་ནུས་པ་ལྡན་པ་བྱ་ཐུབ་རྒྱུ་ཡིན་ལ། ཚོད་དཔག་ལྟར་ན། ཕྱི་ལོ་ ༢༠༡༦ ལོའི་ཟླ་ ༦ པའི་ནང་ཆུ་ཤུགས་གློག་ཁང་དེའི་འཕྲུལ་ཚོ་ཚང་མ་འཁོར་སྐྱོད་ཐུབ་རྒྱུ་ཡིན་པ་དང་། སྐབས་དེར་ཆུ་ཚོད་རེར་གློག་གཏོང་ཚད་ཆན་ཝ་དུང་ཕྱུར་ ༨།༣༡༩ ཟིན་གྱི་ཡོད་པས། ཆབ་མདོ་གྲོང་ཁྱེར་གྱི་གློག་གི་གནད་དོན་ཐག་གཅོད་དཔྱིས་ཕྱིན་པ་ཐུབ་རྒྱུ་ཡིན་སྐོར་བརྗོད་འདུག   དེ་ཡང་ཆབ་མདོ་གྲོང་ཁྱེར་ཐོན་ཁུངས་གསར་སྤེལ་ཁང་གི་ལས་གཉེར་བ་སླེ་ཁྲའོ་ཡིས། དེང་སྐབས་ཆབ་མདོ་གྲོང་ཁྱེར་གྱིས་འབྲི་ཆུ་དང་ཟླ་ཆུ། རྒྱལ་མོ་རྔུལ་ཆུ་བཅས་ཀྱི་ཆུ་རྒྱུད་དངོས་དང་ཆུ་ལག་ཐོག སྤྱིའི་གློག་འདོན་ཚད་ཝ་སྟོང་ཚན་ཁྲི་ཚོ་ ༣༥༨༥ ཟིན་པའི་སྐས་རིམ་ཆུ་ཤུགས་གློག་ཁང་ ༤༩ འཛུགས་རྒྱུའི་འཆར་འགོད་བྱེད་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
