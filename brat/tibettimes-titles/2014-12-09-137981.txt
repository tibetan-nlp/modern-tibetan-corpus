ལྷོ་ཀོ་རི་ཡ་དང་རྒྱ་གར་གྱི་མཉམ་འབྲེལ།

རྒྱ་གར་དང་ལྷོ་ཀོ་རི་ཡ་གཉིས་ཀྱིས་རྒྱལ་སྲུང་མཉམ་ལས་ཀྱི་ཚོགས་འདུ་ཞིག་ལྷོ་ཀོ་རི་ཡའི་ནང་ཚོགས་ཏེ་ཚོང་དོན་དང་ལོ་རྗེས་མར་རྒྱ་གར་སྲིད་བློན་ལྷོ་ཀོ་རི་ཡར་སྐྱོད་ཕྱོགས་བཅས་ལ་གྲོས་མོལ་བྱ་རྒྱུ་ཡིན་འདུག ཁ་སང་ལྷོ་ཀོ་རི་ཡའི་ཕྱི་སྲིད་ལྷན་ཁང་གི་ལྷོ་ཤར་ཨེ་ཤ་ཡའི་སྤྱི་ཁྱབ་འགན་འཛིན་སུ་ཇཱན་ཨང་ (Suh Jeong-in) གིས་གསལ་བསྒྲགས་སྤེལ་བའི་ནང་། ཟླ་འདིའི་ཚེས་ ༢༩ ཉིན་ལྷོ་ཀོ་རི་ཡའི་རྒྱལ་ས་སཱོལ་ནང་རྒྱ་གར་དང་ལྷོ་ཀོ་རི་ཡ་བར་རྒྱལ་སྲུང་མཉམ་ལས་གྲོས་མོལ་ཐེངས་ ༨ པ་དེ་ཚོགས་རྒྱུ་ཡིན་པ་དང་། ཚོགས་འདུ་དེར་རྒྱ་གར་ཕྱི་སྲིད་བློན་ཆེན་ལྕམ་སུ་ཤི་མཱ་སཱོ་ཝ་ར་ཇཱ་ཕེབས་ཏེ། ལྷོ་ཀོ་རི་ཡའི་སྲིད་འཛིན་ལྕམ་པག་གྷེན་ཧེ་དང་ཐུག་འཕྲད་ཀྱིས་མ་འོངས་པར་རྒྱལ་ཁབ་གཉིས་བར་རྒྱལ་སྲུང་དང་དཔལ་འབྱོར་ཚོང་དོན་མཉམ་ལས་གོང་འཕེལ་གཏོང་ཐབས་དང་། ༢༠༡༥ ལོའི་ལོ་མགོར་རྒྱ་གར་སྲིད་བློན་ནཱ་རེན་ཌ་མུ་དྷེ་ལྷོ་ཀོ་རི་ཡར་སྐོར་སྐྱོད་བྱེད་ཕྱོགས་བཅས་ལ་གྲོས་མོལ་བྱ་རྒྱུ་ཡིན་སྐོར་བསྒྲགས་འདུག འདིའི་ཐད་སུ་ཇཱན་ཨང་གིས། རྒྱ་གར་གྱི་སྲིད་བློན་ནཱ་རེན་ཌ་མུ་དྷེ་ཡི་ལྷོ་ཀོ་རི་ཡར་སྐོར་སྐྱོད་བྱ་འཆར་ཡོད་པ་དེས། འདས་པའི་ལོ་ཤས་ནས་བཟུང་རྒྱལ་ཁབ་གཉིས་ཀྱི་འབྲེལ་ལམ་སྔར་ལྷག་འཕེལ་རྒྱས་འགྲོ་བཞིན་ཡོད་པ་གསལ་བཤད་བྱས་ཡོད་སྟབས། ད་རེབ་རྒྱ་གར་གྱི་ཕྱི་སྲིད་བློན་ཆེན་ལྷོ་ཀོ་རི་ཡར་བསྐྱོད་དེ་རྒྱལ་ཁབ་གཉིས་བར་གྱི་གྲོས་མཐུན་ཁག་ཅིག་ལ་མིང་རྟགས་འགོད་འཆར་ཡོད། ཅེས་བརྗོད་འདུག ཁོང་གིས་མུ་མཐུད་དུ། ཟླ་འདིའི་ཟླ་མཇུག་ཏུ་རྒྱལ་ཁབ་གཉིས་བར་རྒྱལ་སྲུང་གི་འབྲེལ་ལམ་གོང་མཐོར་གཏོང་ཐབས་ཀྱི་གྲོས་མོལ་ཐེངས་ ༨ པ་དེ་བྱ་རྒྱུ་ཡིན་པ་མ་ཟད། མ་འོངས་པར་གྲོས་མོལ་འདི་བརྒྱུད་རྒྱལ་ཁབ་གཉིས་ཀྱི་ཆབ་སྲིད་དང་། རྒྱལ་སྲུང་འབྲེལ་ལམ་ཡག་ཏུ་ཕྱིན་ཏེ་མཐུན་གྲོགས་རྒྱལ་ཁབ་ཅིག་ཏུ་འགྱུར་ཐུབ་པ་བྱ་རྒྱུ་ཡིན། ཞེས་བརྗོད་འདུག ༢༠༡༠ ལོར་ལྷོ་ཀོ་རི་ཡ་དང་རྒྱ་གར་བར་ཁྱབ་རྒྱ་ཆེ་བའི་དཔལ་འབྱོར་མཉམ་ལས་གྲོས་མཐུན་ཞེས་པ་ཞིག་བཞག་རྗེས། ༢༠༡༡ ལོར་ཨ་སྒོར་ས་ཡ་ ༩༢ དང་། ༢༠༡༢ ལོར་ཨ་སྒོར་ས་ཡ་ ༩༨། ༢༠༡༣ ལོར་ཨ་སྒོར་ཐེར་འབུམ་ ༡ བཅས་ཀྱི་ཚོང་འབྲེལ་བྱས་ཡོད་པ་མ་ཟད། ཉེ་ཆར་རྒྱ་གར་གཞུང་གིས་བལ་ཡུལ་ནང་ཚོགས་པའི་ལྷོ་ཤར་ཨེ་ཤ་ཡའི་དཔལ་འབྱོར་མཉམ་འབྲེལ་ལྷན་ཚོགས་ཐོག་ལྷོ་ཀོ་རི་ཡའི་སྐུ་ཚབ་ཟུར་ཉན་དུ་སྐད་བཏང་ཡོད་པ་དང་ཐུག་འཕྲད་ཀྱིས་ལྷོ་ཀོ་རི་ཡ་ནས་རྒྱ་གར་གཞུང་གིས་K9 རྟགས་ཅན་གྱི་མེ་སྒྱོགས་དང་། དམག་རྒྱག་རླངས་འཁོར་ཉོ་རྒྱུ་བྱས་ཡོད་འདུག དེ་བཞིན་འདི་ལོའི་ཕྱི་ཟླ་ ༡༡ པའི་ནང་འབར་མའི་ནང་ལྷོ་ཤར་ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་ཁག་གི་ལྷན་ཚོགས་ཐེངས་ ༣༦ པ་ཚོགས་སྐབས། རྒྱ་གར་སྲིད་བློན་དང་ལྷོ་ཀོ་རི་ཡའི་སྲིད་འཛིན་གཉིས་ཐུག་འཕྲད་བྱས་ཏེ་དམག་དོན་དང་རྒྱལ་སྲུང་། དཔལ་འབྱོར་མཉམ་ལས་བཅས་ཀྱི་ཐོག་གྲོས་མོལ་བྱས་ཡོད་པ་དང་། རྒྱལ་ཁབ་གཉིས་ཀྱི་རྒྱལ་སྲུང་ལྷན་ཚོགས་ལྷོ་ཀོ་རི་ཡར་འཚོགས་རྒྱུ་དང་། རྒྱ་གར་གྱི་སྲིད་བློན་ལྷོ་ཀོ་རི་ཡར་རྒྱལ་དོན་སྐོར་སྐྱོད་གནང་རྒྱུ་བཅས་ཀྱི་ཐག་གཅོད་བྱས་ཡོད་འདུག ཆབ་སྲིད་དབྱེ་ཞིབ་པ་ཚོས། འདས་པའི་ལོ་ཤས་ནས་བཟུང་རྒྱ་ནག་གཞུང་གིས་ཨེ་ཤ་ཡའི་མངའ་ཁུལ་ནང་དབང་ཤུགས་ཆེ་རུ་གཏོང་ཐབས་བྱས་ཏེ། མཚོ་གླིང་རྩོད་ཉོག་དང་རྒྱལ་ཁབ་གཞན་གྱི་བདག་དབང་ལ་ཐེ་ཇུས་བྱེད་བཞིན་ཡོད་སྟབས། ད་རེབ་ལྷོ་ཀོ་རི་ཡས་རྒྱ་གར་དང་མཉམ་ལས་བྱས་ཏེ་རྒྱ་ནག་གཞུང་གི་གནོན་ཤུགས་འགོག་ཐབས་བྱེད་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
