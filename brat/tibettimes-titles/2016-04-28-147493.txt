གསར་འགོད་རང་དབང་གི་སྙན་ཐོ་བཏོན།

ཕྱི་ལོར་ ༢༠༡༥ ལོའི་གསར་འགོད་རང་དབང་གི་སྙན་ཐོའི་ནང་འཛམ་གླིང་གི་མི་འབོར་ལས་བརྒྱ་ཆ་ ༤༦ གསར་འགོད་རང་དབང་མེད་པའི་ཁོར་ཡུག་ནང་འཚོ་སྡོད་བྱེད་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག དེ་རིང་ཨ་རིའི་རང་དབང་ཁང་(Freedom House)ཞེས་པའི་གཞུང་འབྲེལ་མ་ཡིན་པའི་ཚོགས་པ་ཞིག་གིས། ཕྱི་ལོ་ ༢༠༡༥ ལོའི་འཛམ་གླིང་སྤྱི་ཡོངས་ཀྱི་གསར་འགོད་རང་དབང་ལ་ཞིབ་འཇུག་བྱས་པའི་སྙན་ཐོ་ཞིག་ཁྱབ་བསྒྲགས་བྱས་ཏེ། ༢༠༡༥་ལོའི་གསར་འགོད་རང་དབང་གི་གནས་བབ་འདས་པའི་ལོ་ངོ་ ༡༢ རིང་གི་སྡུག་ཤོས་ཆགས་ཡོད་པ་དང་། ཚོགས་པ་དེས་ཞིབ་འཇུག་བྱས་པའི་རྒྱལ་ཁབ་ཁག་ ༡༩༩ ཡི་ནང་ནས་རྒྱ་ནག་ནི་འཛམ་གླིང་ཐོག་གསར་འགོད་རང་དབང་ལ་བཀག་སྡོམ་ཐེབས་པ་ཚབས་ཆེ་ཤོས་ཤིག་ཏུ་ངོས་འཛིན་བྱས་ཡོད་པ་མ་ཟད། སྲིད་འཛིན་ཞི་ཅིན་ཕིན་སྲིད་སྟེགས་སུ་བསླེབས་རྗེས་གསར་འགོད་རང་དབང་ཐོག་སྔར་ལས་དམ་དྲག་ཤུགས་ཆེ་རུ་ཕྱིན་ཡོད་སྐོར་བརྗོད་ཡོད་པ་རེད། སྙན་ཐོའི་ནང་ཕྱི་ལོ་ ༢༠༡༥ ལོའི་གསར་འགོད་རང་དབང་གི་གནས་བབ་འདས་པའི་ལོ་ངོ་ ༡༢ རིང་གི་སྡུག་ཤོས་ཆགས་པའི་རྒྱུ་མཚན་ནང་འདས་ལོར་དཀྱིལ་ཤར་ཁུལ་དང་མེག་སི་ཁོ་(Mexico)ནང་གསར་འགོད་པར་འཇིགས་སྐུལ་གནོད་འཚེ་ཆེན་པོ་ཐེབས་ཡོད་པ་དང་། རྒྱ་གཞུང་གིས་ཧོང་ཀོང་ནང་སྨྲ་བརྗོད་རང་དབང་ལ་གནོད་འཚེ་བཏང་བ་སོགས་ཐེར་འདོན་བྱས་འདུག དེ་བཞིན་འཛམ་གླིང་གི་རྒྱལ་ཁབ་མང་པོའི་ནང་གསར་འགོད་རང་དབང་མྱུར་ཚད་མཐོན་པོས་ཇེ་ཞན་དུ་གྱུར་བའི་རྒྱུ་རྐྱེན་ནི། གསར་འགྱུར་བརྒྱུད་ལམ་ཁག་ལ་ཕྱོགས་རེས་དབྱེ་འབྱེད་ཆེ་ཞིང་ཐལ་དྲགས་རང་བཞིན་གྱི་གནས་བབ་ཚབས་ཆེན་ཡིན་པ་དང་། གསར་འགོད་པར་ཁྲིམས་སྒྲིག་ལས་བརྒལ་བའི་འཇིགས་སྐུལ་དང་གནོད་འཚེ་གཏོང་བ་ཚབས་ཆེ་རུ་ཕྱིན་པ་སོགས་ཀྱིས་ཡིན་སྐོར་བརྗོད་འདུག ཡང་སྙན་ཐོའི་ནང་འཛམ་གླིང་ཐོག་གསར་འགོད་རང་དབང་ཆེས་ཆེ་བའི་རྒྱལ་ཁབ་ནི་ནོར་ཝེ་དང་སྤེལ་ཇམ། ཧྥིན་ལན་ཌི། ཧོ་ལན་ཌི། སི་ཝེ་ཌན་སོགས་ཀྱི་ནང་། སྒྲིག་གཞི་ཡང་དག་གི་སྒོ་ནས་ཆབ་སྲིད་དང་འབྲེལ་བའི་གསར་འགྱུར་སྤེལ་བ་དང་གསར་འགོད་པའི་བདེ་འཇགས་ལ་སྲུང་སྐྱོབ་ཡོད་པ། སྲིད་གཞུང་གིས་གསར་འགྱུར་བརྒྱུད་ལམ་ཁག་ལ་ཇུས་གཏོགས་ཉུང་ཞིང་། བརྒྱུད་ལམ་ཁག་ལ་བཅའ་ཁྲིམས་དང་དཔལ་འབྱོར་གྱི་གནོན་ཤུགས་ཆུང་བ་ཡོད་སྐོར་བརྗོད་ཡོད་པ་རེད། དེ་ཡང་སྙན་ཐོ་ལྟར་ན། ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་ཁག་གི་ནང་ནས་ཕཱ་ལོ་(Palau) གསར་འགོད་རང་དབང་ཆེ་ཤོས་ཨང་དང་པོ་དང་ནེའུ་ཛི་ལན་ཌི་ཨང་རིམ་གསུམ་པ། ཨོ་སི་ཊོ་ལི་ཡ་ ༥ པ། ཉི་ཧོང་དང་ཐའེ་ཝན་ ༧ པ། ལྷོ་ཀོ་རི་ཡ་ ༡༥ པ། ཧོང་ཀོང་ ༡༨ པ་བཅས་ཡིན་ལ། སེང་གྷ་ཕོར་དང་མ་ལེ་ཞེ་ཡ་ ༣༡ པ། རྒྱ་ནག་ཨང་ ༣༩ པ་དང་བྱང་ཀོ་རི་ཡ་ ༤༠ པ་བཅས་གསར་འགོད་རང་དབང་མེད་པའི་རྒྱལ་ཁབ་ཀྱི་ནང་ཚུད་ཡོད་པ་དང་། འཛམ་གླིང་ཐོག་གསར་འགོད་རང་དབང་མེད་པ་ཚབས་ཆེ་ཤོས་ཀྱི་རྒྱལ་ཁབ་ནི་ཡུ་རོབ་ཤར་རྒྱུད་ཀྱི་རྒྱལ་ཁབ་བྷ་ལ་རོ་སི་ (Belarus)དང་ཡུ་ཁི་རིན་གྱི་གླིང་ཕྲན་ཁིར་མེ་ཡ་( Crimea) ། ཁུ་བྷ། སི་རི་ཡ། བྱང་ཀོ་རི་ཡ། དབྱི་རག་སོགས་ཡིན་ལ། འཛམ་གླིང་ཁྱོན་ཡོངས་སུ་གསར་འགོད་རང་དབང་ལོངས་སུ་སྤྱོད་མཁན་བརྒྱ་ཆ་ ༡༣ དང་། གསར་འགོད་རང་དབང་ཅུང་ཟད་ཡོད་པ་བརྒྱ་ཆ་ ༤༡ ལས་ཟིན་མེད་སྟབས། འཛམ་གླིང་གི་མི་འབོར་ལས་བརྒྱ་ཆ་ ༤༦ གསར་འགོད་རང་དབང་མེད་པའི་ཁོར་ཡུག་ནང་འཚོ་སྡོད་བྱེད་བཞིན་ཡོད་པ་རེད། གསར་འགོད་པ། པདྨ་མཚོ།
