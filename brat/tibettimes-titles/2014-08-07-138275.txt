བུད་མེད་ཚོ་སྲུང་སྐྱོབ་ཀྱི་དམ་བཅའ་འཇོག

དབུས་བོད་ཀྱི་བོད་མེད་ལྷན་ཚོགས་ཀྱིས་གོ་སྒྲིག་བྱས་པའི་༼བཙན་བྱོལ་བོད་ཀྱི་བུད་མེད་ལ་ཁྲིམས་དོན་གྱི་ནུས་སྟོབས་གོང་འཕེལ།༽ཞེས་པའི་ཉིན་ལྔའི་རིང་གི་ཟབ་སྦྱོང་དེ་རིང་མཇུག་བསྒྲིལ་ཡོད་འདུག བུད་མེད་ལྷན་ཚོགས་ཀྱིས་མཇུག་བསྡོམས་གནས་ཚུལ་བཏང་འབྱོར་ལྟར་ན། ཐེངས་འདིའི་ཟབ་སྦྱོང་ལ་མཉམ་ཞུགས་གནང་མཁན་མི་གྲངས་ ༦༠ ཡིས་ང་ཚོའི་བཙན་བྱོལ་སྤྱི་ཚོགས་ནང་ཕོ་མོ་དབྱེད་འབྱེད་དང་ཁྲིམས་དོན་ནུས་སྟོབས་གོང་སྤེལ་གཏོང་ཐབས་ལ་ཤུགས་རྒྱག་རྒྱུའི་དམ་བཅའ་བཞག་ཡོད་པ་དང་། སྐྱེས་པ་ ༨ ཀྱིས་བོད་པའི་སྤྱི་ཚོགས་ནང་བདེ་འཇགས་ཀྱི་ཁོར་ཡུག་བསྐྲུན་ཐབས་ཐོག་ཤུགས་རྒྱག་རྒྱུའི་དམ་བཅའ་བཞག་ཡོད་འདུག ཟབ་སྦྱོང་ཉིན་བཞི་བར་ཟབ་སྦྱོང་ལ་ཞུགས་མཁན་ཡོངས་ལ། ༧གོང་ས་མཆོག་ལ་མཇལ་བཅར་གོ་སྐབས་བྱུང་ཡོད་པ་དང་། སྐབས་དེར་༧གོང་ས་མཆོག་གིས། “འདི་ལྟ་བུ་ཁྲིམས་དོན་ནུས་སྟོབས་གོང་སྤེལ་གྱི་ཟབ་སྦྱོང་རིགས་མུ་མཐུད་སྤེལ་དགོས་པ་རེད་ལ། བུད་མེད་གཅིག་པུར་མ་ཡིན་པར་སྐྱེས་པ་ཕོ་ལའང་དེ་མཚུངས་སུ་གོ་རྟོགས་སྤེལ་དགོས་པ་རེད། སྤྱི་ཚོགས་ཀྱི་ཕོ་དང་བུད་མེད་གཅིག་པུར་ཕན་ཐོགས་སྒྲུབ་པ་ཙམ་མ་ཡིན་པར། མ་གྱུར་སེམས་ཅན་ཐམས་ཅན་ལ་ཕན་པ་སྒྲུབ་ཐབས་གནང་དགོས” ཞེས་བཀའ་གནང་ཡོད་འདུག ཟབ་སྦྱོང་སྔོན་འགྲོའི་ཉིན། དཔལ་རྒྱལ་བ་ཀརྨ་པ་མཆོག་ལ་མཇལ་བཅར་ཁོང་གིས། ཁོང་གི་ངོས་ནས་ནང་བསྟན་ཆོས་ལུགས་ཀྱི་ཆོ་ཁ་དང་ཕྱག་ལེན་ཁྲོད་ན་ཕོ་མོ་དབྱེ་འབྱེད་ཏོག་ཙམ་ཡོད་པའི་དོ་སྣང་བྱུང་མྱོང་ཡོད་སྐོར་དང་། ཐེངས་འདིའི་ཟབ་སྦྱོང་ལ་ཞུགས་མཁན་ཚོས་མཐོང་རྒྱ་དང་ཐོས་རྒྱ་སྐྱེད་ནས་སྤྱི་ཚོགས་གཞན་དང་གཞན་གྱི་སླད་དུའང་རོགས་ཕན་དང་ཐུགས་ཁུར་གནང་དགོས་པ་དང་། བུད་མེད་ཚོགས་སྡེ་གཞན་དང་མཉམ་རུབ་ཀྱི་ཕྱག་ལས་གནང་ཐབས་ལ་འབད་དགོས་སྐོར་སོགས་གསུངས་ཡོད་འདུག བོད་ཀྱི་བུད་མེད་ལྷན་ཚོགས་ཀྱིས་ཁྲིམས་དོན་ཀྱི་སྦྱོང་བརྡར་དང་ཟབ་སྦྱོང་ཐོབ་རིགས་ལ་དཔང་རྟགས་ཕྱག་འཁྱེར་སྤྲོད་བཞིན་ཡོད་པ་དང་། ཐེངས་འདིའི་ཟབ་སྦྱོང་ཐོག་ ༼རྒྱ་གར་ཁྲིམས་ཁ་ན་བུད་མེད་དང་བྱིས་པ་སྲུང་སྐྱོབ་གང་ཡོད་པ་ངོ་སྤྲོད་ཀྱི་རྒྱུན་སྤྱོད་ལག་དེབ།༽ཅེས་དབྱིན་བོད་གཉིས་ནང་ཡོད་པའི་ལག་དེབ་ཅིག་ཀྱང་བཀྲམས་ཡོད་འདུག ཐེངས་འདིའི་ཟབ་སྦྱོང་དེ་ཚེས་ ༤ ནས་ ༧ བར་ཉིན་བཞིའི་རིང་སྤྲོད་ཡོད་པ་དང་། ས་གནས་བུད་མེད་ ༡༣ དང་ལྡི་ལི་མཐོ་སློབ་གཙོས་པའི་མཐོ་སློབ་ཁག་ཅིག་བཅས་ནས་མི་ ༦༠ ཙམ་མཉམ་ཞིག་གནང་ཡོད་འདུག
