གསང་ཉུལ་རྒོལ་རྡུང་བར་ཞིབ་བཤེར།

རྒྱ་ནག་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ཁང་གིས་རྒྱལ་ཁབ་བདེ་འཇགས་ལྷན་ཁང་གི་འགན་འཛིན་གཞོན་པ་མཱ་ཅན་ཟེར་བར་གསང་བའི་ཞིབ་བཤེར་བྱེད་བཞིན་ཡོད་འདུག ཁ་སང་ཧོང་ཀོང་གི་ལན་ཧྭ་ཞོགས་པའི་ཚགས་པར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། རྒྱ་ནག་རྒྱལ་ཁབ་བདེ་འཇགས་ལྷན་ཁང་གི་འགན་འཛིན་གཞོན་པ་མཱ་ཅན་ཟེར་བར་རྒྱ་ནག་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ཁང་གིས་གསང་བའི་ཞིབ་བཤེར་བྱེད་བཞིན་ཡོད་པ་མ་ཟད། ཁོ་བོའི་གཉེན་འབྲེལ་ཁག་ཅིག་ཡང་འཛིན་བཟུང་བྱས་ཏེ་འདྲི་གཅོད་བྱེད་བཞིན་ཡོད་འདུག རྒྱ་ནག་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ཁང་གིས་ད་ལྟའི་བར་མཱ་ཅན་གྱི་སྐོར་ལ་གནས་ཚུལ་གང་ཡང་ཕྱིར་བསྒྲགས་བྱས་མེད་ཀྱང་། ཉེ་སྔོན་ནས་བདེ་འཇགས་ལྷན་ཁང་གི་ལས་བྱེད་མང་པོ་འཕོ་འགྱུར་བཏང་ཡོད་པ་དང་། མཱ་ཡི་འབྲེལ་ཡོད་ཅིག་གིས་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་དོགས་གཞིའི་འོག་མཱ་ཅན་ལ་ཞིབ་བཤེར་བྱེད་བཞིན་ཡོད་ཅིང་། ཁོ་བོ་རྒྱ་ནག་ཆབ་སྲིད་ལྷན་ཁང་གི་མགོ་ཁྲིད་ཟུར་པ་ཀྲུའུ་ཡོང་ཁང་དང་འཐབ་ཕྱོགས་གཅིག་གྱུར་ཁང་གི་འགན་འཛིན་ཟུར་པ་ལིན་ཅི་ཧྭ་གཉིས་ཀྱི་ཉེས་དོན་དང་འབྲེལ་བ་ཡོད་སྲིད་སྐོར་བརྗོད་འདུག བྷི་བྷི་སི་སོགས་ཕྱི་ཕྱོགས་གསར་ལམ་ཁག་ཏུ་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། མཱ་ཅན་གྱིས་རྒྱལ་ཁབ་བདེ་འཇགས་ལྷན་ཁང་ནང་ལོ་ངོ་ ༡༠ ཙམ་གྱི་རིང་གསང་འཕྲིན་རྒོལ་རྡུང་གི་ལས་འགན་འཁུར་ཡོད་པ་དང་། ཁོ་བོའི་ཁྱིམ་མི་དང་བཅས་ལིན་ཅི་ཧྭ་དང་རྒྱུན་ལྡན་གྱི་འགྲོ་འོང་འབྲེལ་བ་ཆེན་པོ་ཡོད་པ་མ་ཟད། ཀྲུའུ་ཡོང་ཁང་ཚང་དང་ཁོའི་ཕྱོགས་གཏོགས་ཚོས་ཡུན་ནན་ཞིང་ཆེན་ནང་གཉེར་བའི་རྒྱ་སྒོར་དུང་ཕྱུར་བཅུ་གྲངས་ཀྱི་ཚོང་ལས་དང་འབྲེལ་བ་ཡོད་སྲིད་པས། ཁོ་བོར་ཞིབ་བཤེར་བྱས་པ་དེས་རྒྱ་ནག་རྒྱལ་སྲུང་ལས་ཁུངས་དང་རྒྱལ་ཁབ་གསང་བའི་ལས་ཁུངས། བདེ་འཇགས་ལྷན་ཁང་བཅས་ཀྱི་ནང་ལྷག་པའི་ཀྲུའུ་ཡོང་ཁང་གི་ཤུགས་རྐྱེན་རྣམས་གཙང་བཟོ་བྱེད་ཀྱི་ཡོད་པ་གསལ་བོ་མཚོན་གྱི་ཡོད་སྐོར་བརྗོད་འདུག མཱ་ཅན་ཟེར་བ་དེ་རྒྱ་ནག་ཅང་ཞི་ཞིང་ཆེན་གྱི་མི་ཡིན་པ་དང་། ཕྱི་ལོ་ ༡༩༨༥ ལོར་ལྷོ་ནུབ་སྲིད་ཁྲིམས་རིག་པའི་སློབ་ཆེན་དུ་འཛུལ་ཞུགས་བྱས་ཡོད་པ་དང་། མཐར་ཕྱིན་ནས་ཡུན་རིང་མ་འགོར་བར་རྒྱལ་ཁབ་བདེ་འཇགས་ཚན་པའི་ནང་ལས་ཀ་བྱས་པ་མ་ཟད། རྒྱལ་ཁབ་བདེ་འཇགས་ལྷན་ཁང་གི་སྐབས་བཅུ་པའི་གཙོ་འགན་འཁུར་ཏེ། རྒྱ་ནག་གི་ཕྱི་རྒྱལ་གཞུང་ཚབ་ཁང་དང་ཕྱི་སྐྱོད་སློབ་མར་རྟོག་ཞིབ་དང་། ཕྱི་ཕྱོགས་ཀྱི་རྒོལ་རྡུང་ལས་འགུལ་ལ་མྱུར་ཞིབ་ཀྱི་ལས་ཀ་བྱས་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༠༦ ལོར་རྒྱལ་ཁབ་བདེ་འཇགས་ལྷན་ཁང་གི་འགན་འཛིན་གཞོན་པར་བསྐོ་བཞག་བྱས་ཏེ། གསང་ཉུལ་རྒོལ་རྡུང་གི་གཙོ་འགན་འཁུར་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
