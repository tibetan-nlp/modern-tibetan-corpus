དཔལ་འབྱོར་གྱི་བསླུ་བྲིད།

  རྒྱ་གཞུང་གིས་དཔལ་འབྱོར་ཡར་རྒྱས་གཏོང་བཞིན་ཡོད་པའི་བོད་ཀྱི་ས་ཁུལ་ཁག་གི་མི་གྲངས་ཀྱི་བརྒྱ་ཆའི་ ༨ ཧན་མི་རིགས་ཡིན་སྐོར་བརྗོད་འདུག   ཁ་སང་ཨེ་ཕེ་གསར་ཁང་གིས་བོད་ཕྱི་ནང་གི་བོད་མི་དང་རྒྱ་རིགས། དེ་བཞིན་ཕྱི་རྒྱལ་གྱི་བོད་དོན་ཉམས་ཞིབ་པ་མང་པར་བོད་ནང་གི་གནས་སྟངས་སྐོར་ལ་བཅར་འདྲི་བྱས་པ་བརྒྱུད་གནས་ཚུལ་སྤེལ་བའི་ནང། ཕྱི་ལོ་ ༡༩༥༡ ལོ་ནས་ད་ལྟའི་བར་རྒྱ་གཞུང་གིས་བོད་ནང་ཨ་སྒོར་ས་ཡ་ ༡༠༠ ལྷག་གི་འགྲོ་གྲོན་བཏང་ནས་དཔལ་འབྱོར་ཡར་རྒྱས་སུ་བཏང་ཡོད་ཡོད་ཀྱང་། བོད་ཀྱི་སྤྱི་ཚོགས་བརྟན་ལྷིང་བཟོ་བའི་དམ་དྲག་གི་སྲིད་ཇུས་ཀྱིས་བོད་མིའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་རྡོག་རོལ་བཏང་སྟེ་གནས་སྟངས་ཚབས་ཆེ་རུ་འགྲོ་བཞིན་ཡོད་ཀྱང་། རྒྱ་གཞུང་གིས་མུ་མཐུད་དངུལ་དངོས་ཁོ་ནས་མཐུན་འགྱུར་བསྐྲུན་ཏེ། རྒྱལ་སྤྱིའི་སྟེང་ནས་བོད་ནང་འཛིན་པའི་རྒྱ་གཞུང་གི་དབང་སྒྱུར་སྲིད་ཇུས་མྱུར་དུ་མཚམས་འཇོག་དགོས་པའི་གནོན་ཤུགས་སྤྲོད་བཞིན་ཡོད་པ་ཟློག་ཐབས་དང་། བོད་མི་མང་གི་ཡིད་སེམས་བསླུ་ཐབས་བྱེད་ཀྱི་ཡོད་སྐོར་བཀོད་འདུག   དེ་ཡང་བཅར་འདྲིའི་ནང་ཁ་ན་ཌའི་ལྷོ་ནུབ་གྲོང་ཁྱེར་བན་ཁོ་བར་ནང་ཡོད་པའི་ཁོ་ལུམ་བྷེ་ཡ་གཙུག་ལག་སློབ་ཆེན་གྱི་སློབ་དཔོན་མཁས་དབང་ཚེ་རིང་ཤཀྱ་ལགས་སུ་བཅར་འདྲི་བྱེད་སྐབས། ད་ལྟའི་ཆར་རྒྱ་གཞུང་གིས་བོད་ནང་དྲག་གནོན་གྱི་ཐབས་ཇུས་ལ་འགྱུར་བ་བཏང་སྟེ་བོད་མིའི་སྤྱི་ཚོགས་ལ་མགོ་སྐོར་གཏོང་གི་ཡོད་པ་མ་ཟད། རྒྱ་གཞུང་གི་དཔོན་རིགས་ཚོར་ངན་པ་ལག་འབྲེལ་བྱེད་མཁན་ཁག་ཅིག་དང་ལྷན་དུ་བོད་དོན་སྒྲུབ་པའི་ཐབས་ཤེས་བྱེད་ཀྱི་ཡོད་ཅེས་དང་། བོད་རྒྱའི་གནད་དོན་མ་སེལ་བར་དུ། བོད་ནང་གི་དཔལ་འབྱོར་ཡར་རྒྱས་དང་རྒྱ་ནག་ནང་གི་དཔལ་འབྱོར་ཡར་རྒྱས་གཉིས་བར་གྱི་ཁྱད་པར་ཆེ་རུ་འགྲོ་གི་རེད། ཅེས་གསུངས་འདུག   ཡང་ནེའུ་ཡོག་གྲོང་ཁྱེར་དུ་རྟེན་གཞི་བྱས་པའི་ཁོ་ལུམ་བྷེ་ཡ་གཙུག་ལག་སློབ་ཆེན་གྱི་བོད་རིག་པའི་ཞིབ་འཇུག་སྡེ་ཚན་གྱི་འགན་འཛིན་རའོ་བྷིར་ཊི་བྷར་ནི་ཊ་ལགས་ཀྱིས། འདས་པའི་མི་ལོ་ ༢༠ རིང་རྒྱ་ནག་གིས་བོད་ནང་ལག་བསྟར་བྱེད་པའི་སྲིད་ཇུས་དེ་དོན་དངོས་སུ་བོད་མི་མང་ལ་ཕན་ཐོགས་ཅི་ཡང་མེད་ཀྱང་། འདས་པའི་ལོ་ངོ་བཅུ་ཕྲག་རིང་གི་དཔལ་འབྱོར་གྱི་ཁེ་ཕན་ལ་བརྟེན་ནས་བོད་ཁུལ་དུ་གནས་སྤོར་བྱས་པའི་ཧན་མི་རིགས་ཀྱི་འཚོ་གནས་ལ་ཡར་རྒྱས་ཕྱིན་ཡོད་ཅིང་། དཔལ་འབྱོར་ཡར་རྒྱས་གཏོང་བཞིན་པའི་བོད་ཁུལ་གྱི་མི་གྲངས་ས་ཡ་ ༣.༢ ལྷག་གི་བརྒྱ་ཆའི་ ༨ རྒྱ་ནག་ཧན་མི་རིགས་ཡིན་ལ། ལྷག་པར་དུ་ཉེ་བའི་ཆར་རྒྱ་གཞུང་གིས་བོད་ཁུལ་དུ་དྲག་གནོན་བྱེད་མགོ་བརྩམས་ཡོད་པ་རེད། ཅེས་དང་།   ཡང་ཁོང་གིས། གལ་ཏེ་རྒྱ་ནག་གིས་དངོས་ནས་སྤྱི་ཚོགས་མཐུན་འགྲིག་དང་དཔལ་འབྱོར་དོ་སྙོམ་མེད་པའི་ས་ཁུལ་དུ་དངུལ་དངོས་ཀྱི་མཐུན་འགྱུར་བསྐྲུན་ཚེ། རྒྱ་ནག་གིས་བོད་ཁུལ་དུ་རྒྱ་མི་རིགས་གནས་སྤོ་བ་དང་ཁེ་ལས་ཁོ་ན་གཉེར་བ། ལྡོག་ཕྱོགས་ནས་ཤན་ཤུགས་སྤྲོད་པ། ལྷག་པར་དུ་དཔལ་འབྱོར་གྱི་དབང་སྒྱུར་སྲིད་ཇུས་སོགས་ཅི་ནས་ཀྱང་མཚམས་འཇོག་དགོས་ཡོད། ཅེས་བརྗོད་འདུག   ཡང་གནས་ཚུལ་ནང་། ཕྱི་ལོ་༢༠༠༩ ལོ་ནས་བཟུང་ད་ལྟའི་བར་བོད་ནང་བོད་མི་སེར་སྐྱ་ཕོ་མོ་ ༡༤༠ ལྷག་གིས་རང་ལུས་མེར་བསྲེགས་གནང་སྟེ་རྒྱ་གཞུང་གི་དབང་བསྒྱུར་སྲིད་བྱུས་ལ་ངོ་རྒོལ་དང་། ༧གོང་ས་མཆོག་བོད་དུ་གདན་འདྲེན་ཞུ་དགོས་པའི་དགོས་འདུན་བཏོན་ཏེ་མུ་མཐུད་ངོ་རྒོལ་བྱེད་བཞིན་ཡོད་སྐོར་སོགས་བཀོད་འདུག   གསར་འགོད་པ། པདྨ་རྒྱལ།
