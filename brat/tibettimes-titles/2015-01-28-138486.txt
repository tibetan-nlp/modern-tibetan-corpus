དྲག་སྤྱོད་པས་ཚོང་གཏམས་སྟངས།

ཁ་སང་སླར་ཡང་ཁ་ཆེ་དྲག་སྤྱོད་པས་བརྙན་ཐུང་བརྒྱུད་ཉི་ཧོང་གིས་ཇོར་ཌན་ལ་གནོན་ཤུགས་སྤྲད་དེ་ཇོར་ཌན་གྱི་འཛིན་བཟུང་འོག་ཡོད་པའི་དྲག་སྤྱོད་པ་ཞིག་ཀློད་གྲོལ་གཏོང་དགོས་པའི་དགོས་འདུན་བཏོན་འདུག ཁ་སང་ཨི་རག་དང་སི་རི་ཡའི་མཉམ་འབྲེལ་དྲག་སྤྱོད་ཤོག་ཁག་གིས་ཉི་ཧོང་གི་གསར་འགོད་པ་ཁཱན་ཇེ་གྷོ་ཌོ་ཡིས་ཇོར་ཌན་གྱི་གནམ་གྲུ་གཏོང་མཁན་མུ་ཋ་ཨལ་ཁཱ་སེ་སི་བྷེའི་འདྲ་པར་བཟུང་ཡོད་པའི་བརྙན་ཐུང་ཞིག་ཕྱིར་བསྒྲགས་བྱས་ཏེ། གལ་ཏེ་ཇོར་ཌན་གཞུང་གིས་དེ་སྔོན་ ༢༠༠༥ ལོར་ཇོར་ཌན་ནང་དྲག་སྤྱོད་སྤེལ་ཏེ་འཛིན་བཟུང་ཐེབས་པའི་སཱ་ཇེ་ཌ་ཨལ་རེ་ཤཱ་ཝེ་ (Sajida al-Rishawi) ཀློད་གྲོལ་མ་བཏང་ཚེ། ཆུ་ཚོད་ ༢༤ ནང་ཚུད་དུ་ལོ་ན་ ༤༧ སོན་པའི་ཁཱན་ཇེ་གྷོ་ཌོ་ (Kenji Goto) དང་། མུ་ཋ་ཨལ་ཁཱ་སེ་སི་བྷེ་(Muath al-Kasaesbeh) དམར་གསོད་གཏོང་རྒྱུ་ཡིན་སྐོར་འཇིགས་སྐུལ་བྱས་འདུག གཞུང་གིས་ ༢༠༠༥ ལོར་ཇོར་ཌན་རྒྱལ་ས་ཨམ་མན་ནང་དྲག་སྤྱོད་ལས་འགུལ་སྤེལ་མཁན་གྱི་དྲག་སྤྱོད་ བརྙན་ཐུང་དེའི་ནང་མི་ཞིག་གིས། “ང་ལ་ད་ལྟ་དུས་ཚོད་ཉུང་ངུ་ལས་ལྷག་མེད་པ་མ་ཟད། ད་དུང་ང་དང་ཁོ་མོ་བརྗེ་པོ་རྒྱག་དགོས་སྟབས། དེ་ནི་བསམ་ཤེས་བྱེད་དཀའ་བ་ཞིག་རེད། ཁོ་མོ་བཙོན་འཇུག་བྱས་ཏེ་ལོ་བཅུ་ལྷག་འགྲོ་བཞིན་ཡོད་པ་དང་། ང་བཙན་ཁྲིད་བྱས་ཏེ་ཟླ་ཤས་ལས་འགྲོ་གི་མེད་ཀྱང་། ཁོ་ཚོས་ཕན་ཚུན་བརྗེ་པོ་རྒྱག་དགོས་སྐོར་བརྗོད་སྟབས། གལ་སྲིད་ཇོར་ཌན་གཞུང་གིས་དུས་ཐོག་ཐག་གཅོད་མ་བྱས་ཚེ་ངེད་གཉིས་དམར་གསོད་གཏོང་རྒྱུ་ཡིན་སྟབས། ཉི་ཧོང་གཞུང་གིས་ཇོར་ཌན་གཞུང་ལ་གནོན་ཤུགས་སྤྲད་དེ་ང་ཚོ་སྐྱོབ་ཐབས་བྱེད་དགོས།”ཞེས་བརྗོད་འདུག ཉི་ཧོང་དང་ཇོར་ཌན་གཞུང་གིས་དྲག་སྤྱོད་པས་ཕྱིར་བསྒྲགས་བྱས་པའི་བརྙན་ཐུང་དེའི་ནང་གི་སྐད་གདངས་དེ་གསར་འགོད་པ་ཁཱན་ཇེ་གྷོ་ཌོའི་ཡིན་མིན་ཞིབ་འཇུག་བྱས་པ་བརྒྱུད། མདང་དགོང་ཉི་ཧོང་གི་སྤྱི་ཁྱབ་དྲུང་ཆེ་ཡོ་ཤི་ཧེ་ཌི་སོ་གྷ་ (Yoshihide Suga) ཡིས་བརྙན་ཐུང་ནང་གི་སྐད་གདངས་དེ་གསར་འགོད་པ་ཁཱན་ཇེ་ཡི་སྐད་སྒྲ་འདྲ་པོ་འདུག་ཅེས་བརྗོད་ཡོད་པ་མ་ཟད། ཇོར་ཌན་གྱི་བརྙན་འཕྲིན་ཚན་པས་གཅིག་མཚུངས་བསྒྲགས་གཏམ་སྤེལ་ཡོད་འདུག ཉི་ཧོང་གི་སྲིད་བློན་ཤིང་ཛོ་ཨ་བྷེས་བརྙན་ཐུང་དེ་ནི་གཞན་ལ་བརྙས་བཅོས་བྱེད་ཆེད་ཕྱི་བསྒྲགས་བྱས་པ་ཞིག་ཡིན་སྟབས། ང་ཁོང་ཁྲོ་ཆེན་པོ་ལངས་ཀྱི་འདུག་ཅེས་བརྗོད་ཡོད་པ་དང་། ཁོང་གིས་གྲོས་ཚོགས་གོང་མའི་ནང་ཇོར་ཌ་གཞུང་དང་འབྲེལ་བ་བྱས་ཏེ་ཁཱན་ཇེ་ཀློད་གྲོལ་གཏོང་རྒྱུར་གྲོས་མོལ་བྱེད་བཞིན་ཡོད། ང་ཚོས་འཇིགས་སྐུལ་རིང་ལུགས་པར་ཁ་གཏད་གཅག་སྟེ་འཛམ་གླིང་ཞི་བདེ་དང་བརྟན་ལྷིང་ཡོང་བ་བྱེད་དགོས་པ་མ་ཟད། ཉི་ཧོང་གི་བདེ་འཇགས་ལ་སྲུང་སྐྱོབ་བྱེད་དགོས་ཞེས་སོགས་བརྗོད་འདུག དེ་རིང་སྔ་དྲོར་ཉི་ཧོང་གི་ཕྱི་སྲིད་བློན་ཆེན་ཧྥོ་མེ་ (Fumio Kishida) ཡིས་གསར་འགོད་པར་ང་ཚོས་ཇོར་ཌན་གཞུང་དང་གྲོས་མོལ་བྱས་ཏེ། ཁཱན་ཇེ་གང་མགྱོགས་ཀློད་གྲོལ་ཡོང་རྒྱུར་འབད་བརྩོན་བྱེད་བཞིན་ཡོད་ཅེས་དང་། ཇོར་ཌན་གྲོས་ཚོགས་ཀྱི་ཕྱི་སྲིད་ཚོགས་ཆུང་གི་ཚོགས་གཙོ་བྷ་སམ་ (Bassam Al-Manaseer) གྱིས་རྒྱལ་ས་ཨམ་མན་ནང་རྟེན་གཞི་བྱས་པའི་ཨ་རིའི་བྷོ་ལོམ་བྷག་ (Bloomberg) གསར་ཁང་གི་བཅར་འདྲིའི་ལན་དུ། ད་ལྟ་ང་ཚོས་ཨི་རག་གི་ཚོ་ཁག་སོ་སོའི་སྡེ་དཔོན་དང་ཆོས་ལུགས་ཀྱི་མགོ་ཁྲིད་ཁག་བརྒྱུད་དྲག་སྤྱོད་པ་དང་གྲོས་མོལ་བྱས་ཏེ། ཁཱན་ཇེ་དང་ཁཱ་སེ་སི་བྷེ་ཀློད་གྲོལ་ཐོབ་རྒྱུར་ཐབས་ཤེས་བྱེད་བཞིན་ཡོད་སྟབས། ད་ལྟ་ང་ཚོས་གང་མགྱོགས་གནས་ཚུལ་བཟང་པོ་ཞིག་འཐོན་རྒྱུར་རེ་སྒུག་བྱེད་བཞིན་ཡོད་ཅེས་སོགས་བརྗོད་འདུག ཁ་སང་མང་ཚོགས་ ༡༠༠༠ ལྷག་གིས་ཇོར་ཌན་གྱི་སྲིད་བློན་ལས་ཁུངས་མདུན་དུ་གཞུང་གིས་དྲག་སྤྱོད་པས་དགོས་འདུན་བཏོན་པ་དག་ལ་དང་ལེན་བྱས་ཏེ། ད་ལྟ་ཁོ་ཚོའི་བཀག་བསྐྱིལ་འོག་ཡོད་པའི་གནམ་གྲུ་གཏོང་མཁན་གྱི་ཚེ་སྲོག་སྐྱོབ་དགོས་པའི་ངོ་རྒོལ་བྱས་ཡོད་པ་དང་། མུ་ཋ་ཨལ་ཁཱ་སེ་སི་བྷེ་ཡི་པ་ཕ་སཱ་ཧྥེ་ (Safi Kasaesbeh) ཡིས་ཇོར་ཇན་མང་ཚོགས་ཀྱིས་གཞུང་ལ་གནོན་ཤུགས་སྤྲད་དེ་མུ་ཋ་གང་མགྱོགས་ཀློད་གྲོལ་ཐོབ་རྒྱུར་རྒྱབ་སྐྱོར་ཡོང་བའི་རེ་སྐུལ་ཞུས་འདུག དེ་བཞིན། གསར་འགོད་པ་ཁཱན་ཇེ་ཡི་ཨ་མ་ཇན་ཁོ་ (Junko Ishido) ཡིས་ཉི་ཧོང་གི་རྒྱལ་ཡོངས་བརྙན་འཕྲིན་བརྒྱུད་ལམ་ (NHK)ཐོག་གཞུང་གིས་དྲག་སྤྱོད་པས་ངའི་བུ་ཀློད་གྲོལ་ཡོང་རྒྱུར་གང་ཡོང་ཤུགས་སྣོན་རྒྱག་རོགས་ཞེས་ཞུ་རྒྱུ་ཡིན་ཞེས་སོགས་བརྗོད་འདུག ༢༠༡༤ ལོའི་ཟླ་ ༡༠ ནང་ཉི་ཧོང་གི་གསར་འགོད་པ་ཁཱན་ཇེ་གྷོ་ཌོ་ཡིས་ལོ་དེའི་ཟླ་ ༨ ནང་དྲག་སྤྱོད་པས་བཙན་ཁྲིད་བྱས་པའི་ཁོང་གི་གྲོགས་པོ་ཧཱ་རུ་ཎཱ་ཡུ་ཁཱ་ཝ་ (Haruna Yukawa) ཀློད་གྲོལ་ཐོབ་ཆེད་སི་རི་ཡའི་ནང་ཕྱིན་སྐབས་དྲག་སྤྱོད་པས་འཛིན་བཟུང་བྱས་ཡོད་པ་དང་། ཟླ་འདིའི་ཚེས་ ༢༠ ཉིན་དྲག་སྤྱོད་པས་བརྙན་ཐུང་ཞིག་ཕྱིར་བསྒྲགས་བྱས་ཏེ། གལ་སྲིད་ཉི་ཧོང་སྲིད་གཞུང་གིས་ཆུ་ཚོད་ ༧༢ ནང་ཚུད་དུ་ཨ་སྒོར་དུང་ཕྱུར་ ༢ ཀྱི་མ་སྤྲད་ཚེ། ཉི་ཧོང་བ་དེ་གཉིས་དམར་གསོད་གཏོང་རྒྱུ་ཡིན་སྐོར་བརྗོད་ཡོད། མུ་ཋ་ཨལ་ཁཱ་སེ་སི་བྷེ་(Muath al-Kasaesbeh) ནི་འདས་ལོའི་ཟླ་ ༡༢ ནང་གནམ་གྲུ་བཏང་སྟེ་དྲག་སྤྱོད་པར་མཁའ་རྒོལ་བྱེད་སྐབས་སི་རི་ཡའི་བྱང་ཕྱོགས་ས་ཁུལ་ནང་གནམ་གྲུ་ཟགས་ཏེ་དྲག་སྤྱོད་པས་བཙན་ཁྲིད་བྱས་ཡོད་འདུག གནད་དོན་དབྱེ་ཞིབ་པ་ཚོས། དེ་སྔོན་དྲག་སྤྱོད་པས་ཉི་ཧོང་གཞུང་ལ་ཧཱ་རུ་ཎཱ་ཡུ་ཁཱ་ཝ་དང་གསར་འགོད་པ་ཁཱན་ཇེ་ཀློད་གྲོལ་གཏོང་རྒྱུར་ཨ་སྒོར་དུང་ཕྱུར་གཉིས་སྤྲོད་དགོས་པའི་ཉེན་བརྡ་བཏང་བ་ལྟར་ཉི་ཧོང་གཞུང་གིས་དུས་ཐོག་དངུལ་སྤྲད་མེད་ཀྱང་། ཧཱ་རུ་ཎཱ་ཡུ་ཁཱ་ཝ་ཡི་གནས་སྟངས་ཤེས་རྟོགས་དཀའ་ཡང་ཁཱན་ཇེ་གྷོ་ཌོ་བསད་མེད་པ་བཞིན། གལ་སྲིད་ད་རེབ་ཀྱང་ཇོར་ཌན་དང་ཉི་ཧོང་གཞུང་གིས་དྲག་སྤྱོད་པའི་དགོས་འདུན་ཁས་ལེན་མ་བྱས་ཀྱང་མཇུག་འབྲས་ཐག་གིས་གཅོད་ཐུབ་པ་ཞིག་མ་རེད་ཟེར། གསར་འགོད་པ། དོན་གྲུབ།
