ཕ་རེ་སིར་མི་ ༡༢༣ དྲག་སྤྱོད་པས་བསད།

  མདང་ནུབ་ཧྥ་རན་སིའི་རྒྱལ་ས་ཕ་རེ་སིའི་ས་གནས་ཁ་ཤས་ནང་འཇིགས་སྐུལ་དྲག་སྤྱོད་པས་མེ་མདའ་བརྒྱབ་པ་དང་འབར་རྫས་གཏོང་བ་བཅས་ཀྱི་འོག་མི་ ༡༢༣ ཙམ་འདས་གྲོངས་དང་མི་ ༢༠༠ ལྷག་ལ་རྨས་སྐྱོན། གཞན་མི་མང་པོ་ཞིག་དྲག་སྤྱོད་པས་གཏའ་མར་བཟུང་ཡོད་འདུག   མདང་ནུབ་ཆུ་ཚོད་ ༡༠ ཡས་མས་ལ། ཡུ་རོབ་ཧྥ་ར་སིའི་རྒྱལ་ས་ཕ་རེ་སིའི་རྫོང་བཅུ་བའི་ཟ་ཁང་ཞིག་ཏུ་དྲག་སྤྱོད་པས་གང་བྱུང་མེ་མདའ་བརྒྱབ་ཡོད་པ་དང་། དེ་དང་མཉམ་དུ་ས་གནས་གཞན་ཞིག་གི་རོལ་དབྱངས་ཚོགས་ཁང་ཏུ་དྲག་སྤྱོད་པས་གང་བྱུང་མེ་མདའ་བརྒྱབ་སྟེ་མི་མང་པོ་དམར་གསོད་བཏང་ཡོད་འདུག་ལ་མི་མང་པོ་ཞིག་བཙན་འཁྲིད་བྱས་ཏེ་གཏའ་མར་ཉར་ཡོད་འདུག གཞན། ཕ་རེ་སིའི་གྲོང་ཁྱེར་བྱང་ཕྱོགས་ཀྱི་རྐང་སྤོལ་རྩེ་ཐང་ཞིག་གི་ཉེ་འཁྲིས་ཀྱི་ཆང་ཁང་ཞིག་ཏུ་འབར་རྫས་གཏོར་བ་བཅས་ཀྱི་འོག་དེ་རིང་ཞོགས་པའི་ཆུ་ཚོད་ ༡༡ བར་མི་ ༡༢༣ ཙམ་འདས་གྲོངས་དང་། མི་ ༢༠༠ ལྷག་ལ་རྨས་སྐྱོན་ཕོག་ཡོད་ཤེས་རྟོགས་བྱུང་ཡོད་ཚུལ་ཧྥ་རན་སིའི་གཞུང་གིས་གསལ་བསྒྲགས་བྱས་འདུག   ཐེངས་འདིའི་དོན་རྐྱེན་བཟོ་མཁན་དྲག་སྤྱོད་པ་ག་ཚོད་ཡོད་མེད་དང་དེ་དག་སུ་དང་འབྲེལ་བ་ཡོད་མེད་ད་སྔ་གསལ་ཁ་མེད་ཀྱང་། ཁྱོན་ད་རེབ་ཕ་རེ་སིའི་ས་གནས་ཁག་ ༦ ནང་དྲག་སྤྱོད་གཏོར་རྒོལ་བཏང་ཡོད་པ་དང་། བརྒྱུད་ལམ་རེ་ཟུང་ཐོག་ཐེངས་འདིའི་དོན་རྐྱེན་བྱུང་ཡུལ་གཅིག་ལ་ཁ་ཆེ་དྲག་སྤྱོད་པ་མི་ ༨ ཡོད་པ་དང་། མི་ ༧ གྱིས་རང་སྲོག་བློས་གཏོང་གི་དྲག་སྤྱོད་ཁྲོད་སྲོག་ཤོར་ཡོད་པ་དང་། མི་ ༡ ལ་ཉེན་རྟོག་པས་མེ་མདའ་བརྒྱབ་སྟེ་བསད་ཡོད་སྐོར་སོགས་བཀོད་འདུག ཁྱོན་ད་རེབ་དོན་རྐྱེན་བྱུང་ཡུལ་ཁག་ཏུ་དྲག་སྤྱོད་པ་ཧ་ཅང་མང་པོ་ཞིག་བྱུང་ཡོད་པ་དང་རེ་ཟུང་ཉེན་རྟོག་པས་བསད་ཡོད་པ་དང་ཁ་ཤས་ཤིག་འཛིན་བཟུང་བྱས་ཡོད་འདུག   མདང་ནུབ་ཧྥ་རན་སི་དང་འཇར་མན་གྱི་རྐང་སྤོལ་རུ་ཁག་གཉིས་བར་རྐང་སྤོལ་འགྲན་བསྡུར་བྱེད་ཀྱི་ཡོད་པ་དང་། སྐབས་དེར་ཧྥ་རན་སིའི་སྲིད་འཛིན་ཧོ་ལན་ཌི་ཡང་རྐང་སྤོལ་འགྲན་བསྡུར་ལ་གཟིགས་བཞིན་ཡོད་འདུག གནས་ཚུལ་དེ་བྱུང་རྗེས་སྲིད་འཛིན་གྱིས་བརྙན་འཕྲིན་བརྒྱུད། ད་ལྟ་ཧྥ་རན་སིའི་མཐའ་མཚམས་ཁུལ་ཛ་དྲག་གིས་བཀག་སྡོམ་བྱས་ཡོད་པ་དང་། ད་རེབ་ཀྱི་དོན་རྐྱེན་བཟོ་མཁན་དྲག་སྤྱོད་པ་དག་ལ་ཐར་ས་ཡས་ནས་མེད་ཅེས་བརྗོད་ཡོད་འདུག་ལ། ཨ་རིའི་སྲིད་འཛིན་ཨོ་བྷ་མས་ཀྱང་། ཕ་རེ་སིའི་དོན་རྐྱེན་དེ་ཧྥ་རན་སི་ཙམ་མ་ཡིན་པར་འཛམ་གླིང་ཡོང་གི་དོན་རྐྱེན་ཞིག་ཡིན་སྟབས། ང་ཚོས་མཉམ་འབྲེལ་ཐོག་གདོང་ལེན་བྱེད་དགོས་སྐོར་སོགས་བསྒྲགས་འདུག   ད་རེབ་ཀྱི་ཧྥ་རན་སིའི་དོན་རྐྱེན་དེར་འཛམ་གླིང་གི་རྒྱལ་ཁབ་མང་པོའི་གཞུང་གིས་གདུང་སེམས་མཉམ་སྐྱེད་བྱེད་བཞིན་ཡོད་ཅིང་། ཉེ་ལམ་ཕྱི་ཟླ་ ༨ པའི་ནང་ཁ་ཆེ་དྲག་སྤྱོད་པ་གཉིས་ཧྥ་རན་སིའི་རྒྱལ་ས་ཕ་རེ་སིར་གཏོར་རྒོལ་གཏོང་རྩིས་ཡོད་སྐབས་ཨ་རིའི་དམག་མིག་ཉིས་ཀྱིས་འཇུས་ཡོད་པ་དང་། ད་ལོའི་ཕྱི་ཟླ་ ༡ ཚེས་ ༧ ཉིན་ཧྥ་རན་སིའི་རྒྱལ་ས་ཕ་རེ་སིའི་བདུན་རེའི་གསར་དེབ་ཤཱག་ལུ་(Charlie Hebdo)ལ་ཁ་ཆེ་དྲག་སྤྱོད་པས་གཏོར་རྒོལ་བཏང་སྟེ། གསར་ཁང་དེའི་རྩོམ་སྒྲིག་པ་ཚུད་པའི་མི་ ༡༢ ཀྱི་ཚེ་སྲོག་ཤོར་ཡོད།
