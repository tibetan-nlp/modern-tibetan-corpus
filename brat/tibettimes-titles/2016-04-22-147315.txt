བོད་ཀྱི་རྒྱལ་དར་རྐུ་མ་ཕྱིར་འཕུད་བཏང་།

ཅེཀ་སྤྱི་མཐུན་རྒྱལ་ཁབ་ཀྱི་ཁྲིམས་ཁང་གིས་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱི་སྐོར་སྐྱོད་སྐབས་བོད་པའི་རྒྱལ་དར་རྐུ་འཁྱེར་བྱེད་མཁན་རྒྱ་རིགས་ཤིག་སྤྱི་མང་གི་བདེ་འཇགས་དང་རྒྱུ་ནོར་ཁེ་ཕན་ལ་ཉེན་ཚབས་བཟོས་པའི་ཉེས་མིང་ཐོག་ཕྱིར་བསྐྲད་བཏང་ཡོད་འདུག ཁ་སང་ཅེཀ་སྤྱི་མཐུན་རྒྱལ་ཁབ་གྱི་ (Prague Daily Monitor) གསར་འགྱུར་བརྒྱུད་ལམ་དུ་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཟླ་སྔོན་མའི་ཕྱི་ཚེས་ ༢༨ ཉིན་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་ཅེཀ་སྤྱི་མཐུན་རྒྱལ་ཁབ་ནང་སྐོར་སྐྱོད་སྐབས། རྒྱ་རིགས་ཤིག་གིས་བོད་པའི་རྒྱལ་དར་རྐུ་འཁྱེར་བྱས་ཡོད་པར་ཅེཀ་སྤྱི་མཐུན་རྒྱལ་ཁབ་ཀྱི་ཉེན་རྟོག་པས་འཇུ་བཟུང་བཀག་བསྐྱིལ་བྱས་ཡོད་པ་དང་། ཕྱི་ཟླ་ ༤ ཚེས་ ༡ ཉིན་ཅེཀ་སྤྱི་མཐུན་རྒྱལ་ཁབ་ཀྱི་རྒྱལ་ས་ཕ་ར་གྷུ་ (Prague) རྫོང་ཁྲིམས་ཁང་དྲུག་པ་ནས་ཁྲིམས་བཤེར་བྱས་ཏེ། སྤྱི་མང་གི་བདེ་འཇགས་དང་རྒྱུ་ནོར་ཁེ་ཕེན་བཅས་ལ་ཉེན་ཚབས་བཟོས་པའི་ཉེས་མིང་འོག ཉེས་ཅན་པ་དེ་ཅེཀ་སྤྱི་མཐུན་རྒྱལ་ཁབ་ཀྱི་མི་སེར་མིན་སྟབས། རྒྱལ་ཁོངས་ནས་ལོ་གཅིག་ནས་བཅུ། ཡང་ན་དུས་ངེས་མེད་དུ་ཕྱིར་བསྐྲད་བྱས་ཡོད་འདུག རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱི་ཉིན་གསུམ་རིང་གི་སྐོར་སྐྱོད་སྐབས་སུ། ཅེཀ་སྤྱི་མཐུན་རྒྱལ་ཁབ་ཀྱི་རྒྱལ་ས་ཕ་ར་གྷུའི་སྲང་ལམ་གང་སར་ཉེན་རྟོག་པ་ཆིག་སྟོང་ལྷག་གིས་སྐོར་གཡེང་བྱས་ཡོད་པ་དང་། སྐབས་དེར་ཞི་རྒོལ་ལས་འགུལ་བ་མི་ ༢༣ འཇུ་བཟུང་བཀག་བསྐྱིལ་བྱས་ཡོད་ནའང་གནད་དོན་ཚབས་ཆེན་བྱུང་མེད། ཅེཀ་སྤྱི་མཐུན་རྒྱལ་ཁབ་ཀྱི་ཉེན་རྟོག་མགོ་གཙོ་ཞིག་གིས་ཕྱིའི་རྒྱལ་ཁབ་ཀྱི་ཆབ་སྲིད་མི་སྣ་གལ་ཆེན་རིགས་རྒྱལ་དོན་སྐོར་སྐྱོད་དུ་ཡོང་སྐབས་སྲིད་ཆབ་མི་སྣ་དེའི་བདེ་འཇགས་སྲུང་སྐྱོབ་ཐད་ཅེཀ་སྤྱི་མཐུན་གྱི་དངུལ་ལོ་ཁུ་རོ་ཝན་ (crown) ས་ཡ་ ༡༦ གཏོང་གི་ཡོད་སྐོར་བརྗོད་འདུག ཞི་ཅིན་ཕིན་གྱི་སྐོར་སྐྱོད་སྐབས་སུ་ཅེཀ་སྤྱི་མཐུན་རྒྱལ་ཁབ་ཀྱི་ཉེན་རྟོག་པས་ཐོག་རྩེག་ཁང་ཁག་ཅིག་གི་སྟེང་སྒྲེང་འཛུགས་བྱས་ཡོད་པའི་བོད་པའི་རྒྱལ་དར་ཁག་ཅིག་མར་བླངས་ཡོད་པར་མི་མང་གིས་སྐྱོན་བརྗོད་ཤུགས་ཆེན་བྱས་ཡོད་ནའང་། ཁོ་ཚོས་བོད་པའི་རྒྱལ་དར་མར་བླངས་པ་དེ་བཙན་ཤུགས་ཀྱི་སྒོ་ནས་མ་ཡིན་པར་ཞི་མཐུན་གྱི་སྒོ་ནས་བླངས་ཡོད་སྐོར་བརྗོད་ཡོད་པ་རེད། གཞན། གནས་ཚུལ་འདིའི་ནང་བོད་པའི་རྒྱལ་དར་རྐུ་འཁྱེར་བྱེད་མཁན་རྒྱ་རིགས་དེའི་སྐྱེས་ལོ་ཙམ་མ་གཏོགས་ཁོའི་མིང་དང་ཕ་ཡུལ་སོགས་གསལ་བོར་འཁོད་མེད་ཅིང་། དོན་རྐྱེན་དེ་ཕྱི་ཟླ་གསུམ་པའི་ཚེས་ ༢༨ ཉིན་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་ཅེཀ་སྤྱི་མཐུན་རྒྱལ་ཁབ་ཀྱི་གནམ་ཐང་ནས་ཨེ་བ་རོབ་སི་ཀ་ (Evropska) སྲང་ལམ་བརྒྱུད་སྐབས་བྱུང་བ་ཞིག་ཡིན་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
