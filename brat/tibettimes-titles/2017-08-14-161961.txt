སྒོ་དམར་ཆོས་འཕེལ་ཀློད་གྲོལ་བཏང་།

ཁྱིམ་མི་གཉེན་འབྲེལ་བཙོན་ཁང་ནང་ཆབ་སྲིད་བཙོན་པར་ཐུག་འཕྲད་དུ་སོང་ཚེ་རྒྱ་སྐད་བཤད་ན་སྐར་མ་ ༣༠ དང་བོད་སྐད་བཤད་ན་སྐར་མ་ ༥ ལས་ཁ་བརྡ་བྱེད་དབང་མི་འདུག་ཟེར། འདི་གར་འབྲེལ་ཡོད་ཅིག་གིས་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། དགེ་འདུན་པ་སྒོ་དམར་ཆོས་འཕེལ་ལགས་ཟླ་སྔོན་མའི་ནང་མཚོ་སྔོན་མུན་ཡན་རྫོང་གི་བཙོན་ཁང་ཞིག་ནས་ཀློད་གྲོལ་བཏང་ཡོད་པ་དང་། ད་ལྟ་ཁོང་རོང་པོ་དགོན་ཆེན་གྱི་ཚོགས་དང་ཆོས་ར་སོགས་སུ་ཕེབས་ཀྱི་ཡོད་ཀྱང་། གནས་སྐབས་རིང་ས་གནས་གཞན་ལ་གང་འདོད་དུ་འགྲོ་བསྐྱོད་མི་ཆོག་པར་ས་གནས་ཉེན་རྟོག་པས་ལྟ་རྟོག་འོག་བཞག་ཡོད་པ་དང་། ཁོང་ཀློད་གྲོལ་ཐོབ་སྟེ་རང་ཡུལ་དུ་འབྱོར་སྐབས་ས་གནས་ཀྱི་དགེ་འདུན་པ་དང་ནང་མི་སོགས་ཀྱིས་ཕེབས་བསུ་ཞུས་ཡོད་ཀྱང་། ཁོང་ཚོར་པར་རྒྱག་ཏུ་བཅུག་མེད་པ་རེད། ཕྱི་ལོ་ ༢༠༡༥ ཟླ་ ༧ ་ཚེས་ ༡༠ ཙམ་ལ་རེབ་གོང་ཐུན་རིན་རྫོང་ཉེན་རྟོག་པས་ཁོང་གི་བཞུགས་ཤག་ཏུ་སྔོག་བཤེར་བྱས་ཏེ།  ༧གོང་ས་མཆོག་གི་སྐུ་པར་བཤམས་ཡོད་པ་ནག་ཉེས་སུ་རྩིས་ཏེ་འཛིན་བཟུང་བཙན་ཁྲིད་བྱས་ཏེ་ཟླ་ངོ ༦ ལྷག་ལ་བཀག་ཉར་བྱས། ཕྱི་ལོ ༢༠༡༦ ཟླ ༡ ཚེས་ ༡༧ ཉིན་རྨ་ལྷོ་ཁུལ་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་ཟེར་བས་ཆ་འཕྲིན་བརྒྱུད་ལམ་ཁག་ཏུ་༧གོང་ས་མཆོག་གི་སྐུ་པར་བརྒྱུད་སྐུར་དང་ཉར་ཚགས་བྱས་ཏེ། སྤྱི་ཚོགས་བརྟན་ལྷིང་དང་རྒྱལ་ཁབ་ཁ་བྲལ་གཏོང་བའི་བྱ་སྤྱོད་སྤེལ་ཡོད་པའི་ཉེས་མིང་འོག དགེ་འདུན་པ་སྒོ་དམར་ཆོས་འཕེལ་ལགས་ལ་ལོ་གཉིས་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་དེ། མཚོ་སྔོན་ཞིང་ཆེན་མུན་ཡོན་རྫོང་ཁོངས་ཀྱི་ཞིང་ལས་ར་བ་ཞེས་བཏགས་པའི་བཙོན་ཁང་ཞིག་དུ་བཅུག བཙོན་ཁང་དེ་དེ་སྔོན་ངལ་རྩོལ་བསྒྱུར་བཀོད་ར་བ་ཞིག་ཡིན་ཡང་། ན་ནིང་ལོ་ནས་བཟུང་ངལ་རྩོལ་བསྒྱུར་བཀོད་ར་བ་མེད་པ་བཟོ་རྒྱུ་ཡིན་ལུགས་བསྒྲགས་ཡོད་པས་ད་ལྟ་དེ་ལ་ཞིང་ལས་ར་བ་ཞེས་འབོད་ཀྱི་ཡོད་འདུག བཙོན་ཁང་དུ་ཡོད་སྐབས་ནང་མི་ཉེ་འབྲེལ་ཚོར་ཟླ་རེར་ཐེངས་གཅིག་རེར་ཐུག་ཏུ་བཅུག་ཡོད་པ་དང་། ཐུག་འཕྲད་སྐབས་རྒྱ་སྐད་བཤད་ན་སྐར་མ་ ༣༠ ལ་ཁ་བརྡ་བྱེད་ཆོག་པ་དང་། བོད་སྐད་བཤད་ན་སྐར་མ་ ༥་ལས་བཤད་བཅུག་མེད་པ་རེད། ལྷག་པར་དུ་ཁོང་ལ་ཆབ་སྲིད་ནག་ཉེས་བཙུགས་ཡོད་སྟབས། ཐུག་འཕྲད་སྐབས་བཙོན་པ་གཞན་དང་མི་འདྲ་བར་ཉེ་འཁྲིས་སུ་ཉེན་རྟོག་པས་ལྟ་རྟོག་བརྙན་ལེན་སོགས་བྱས་ཡོད་པ་དང་།   སྐབས་རེར་ཐུག་འཕྲད་དུ་ཡོང་བའི་རྒྱ་སྐད་མི་ཤེས་པའི་ཉེ་འབྲེལ་ཚོར་ཁ་བརྡའང་བྱེད་དུ་བཅུག་མེད་པ་རེད། ཁོང་གིས་བཙོན་ཁང་དུ་ཉིན་རེར་ཆུ་ཚོད་ ༡༥ རེར་དམག་ལྭ་དང་། བྱ་སྤུའི་གྱོན་ཆས་རིགས་བཟོས་པ་དང་། འབྲོག་པའི་ལྕགས་རྭ་ཡི་འཛེར་མ་ར་མགོ་བཟོ་བ་སོགས་ཀྱི་ལས་ཀ་བྱས་ཡོད་པ་དང་། བཙོན་ཁང་དེ་ངལ་རྩོལ་བསྒྱུར་བཀོད་ཁང་ཞིག་ཡིན་ཡང་འདས་ལོ་ནས་ངལ་རྩོལ་བསྒྱུར་བཀོད་ཁང་རྣམས་མེད་པ་བཟོ་རྒྱུ་ཡིན་ལུགས་བསྒྲགས་ཡོད་སྟབས། ད་ལྟ་བཙོན་ཁང་དེའི་མིང་ལ་ཞིང་ལས་ར་བ་ཞེས་འབོད་ཀྱི་ཡོད་འདུག དགེ་འདུན་པ་སྒོ་དམར་ཆོས་འཕེལ་ལགས་ནི་སྒོ་དམར་སྡེ་བའི་ཕ་ཀླུ་བྱམས་ཀྱི་བུ་ཡིན་ཞིང་། ད་ལྟ་དགུང་ལོ་ ༤༨ ཡིན་ལ།  རེབ་གོང་རོང་པོ་དགོན་ཆེན་ཐོས་བསམ་ནོར་གླིང་གྲྭ་ཚང་ནས་ཡིན་པ་དང་། རྒྱུན་དུ་ཡ་རབས་ཚུལ་མཐུན་དང་སློབ་སྦྱོང་ལ་བརྩོན་ཞིང་དགོན་པ་ནས་བསླབ་པ་མཐར་སོན་པའི་དགེ་འདུན་པ་རྒན་གྲས་ཤིག་ཡིན་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
