མི་རབས་རྒན་པ་རྣམས་བཀའ་དྲིན་ཆེ།༼པར།༽

  སུ་ཇ་བོད་ཁྱིམ་གྱི་སློབ་ཕྲུག་ཁག་ཅིག་གིས་མི་རབས་རྒན་པ་རྣམས་ཀྱི་བཀའ་དྲིན་རྗེས་སུ་དྲན་ཏེ་རྒན་རྒོན་ཚོའི་ཕྱིའི་ཁོར་ཡུག་དང་ནང་གི་ཉལ་ཆས་སོགས་གཙང་མ་བཟོས་ཏེ་བཀའ་དྲིན་རྗེས་དྲན་ཞུས་འདུག   དེ་རིང་ཉིན་རྒྱབ་སུ་ཇ་བོད་ཁྱིམ་སློབ་གྲྭའི་སློབ་ཕྲུག་གཉིས་འདི་གའི་གསར་ཁང་ལ་ཕེབས་ཏེ། ཁོ་རང་ཚོས་དགུན་ཁའི་གུང་གསེང་དང་བསྟུན་ཏེ་དྷ་རམ་ས་ལའི་ཉེ་འཁོར་གྱི་རྒས་གསོ་ཁང་ཁག་ཅིག་ལ་དང་བླངས་ཞབས་ཞུ་བསྒྲུབས་པའི་སྐོར་ངོ་སྤྲོད་གནང་སོང་།   ཁོང་གཉིས་ཀྱིས་བརྗོད་པ་ལྟར་ན། སུ་ཇ་སློབ་གྲྭའི་འཛིན་རིམ་བཅུ་གཅིག་གི་སློབ་ཕྲུག་ཁག་ཅིག་གིས་གྲོས་བསྡུར་བྱས་ཏེ། ༧གོང་ས་མཆོག་དགུང་གྲངས་ ༨༠ ཕེབས་པ་དང་བསྟུན་ཏེ། རྒན་པའི་བཀའ་དྲིན་རྗེས་དྲན་དང་། བོད་པའི་ཡ་རབས་བཟང་སྤྱོད། མི་རབས་གསར་པ་ཚོར་མིག་དཔེ་བཟང་པོ་འཇོག་ཐབས་བཅས་ཀྱི་ཆེད་དུ། ད་ལོའི་དགུན་ཁའི་གུང་གསེང་དང་བསྟུན་ཏེ་སློབ་ཕྲུག་ ༩ ཡིས་མ་སྐུལ་དང་བླངས་ཐོག་དྷ་རམ་ས་ལའི་ཉེ་འཁོར་གྱི་རྒས་གསོ་ཁང་ ༣ ལ་ཞབས་ཞུ་བསྒྲུབ་རྒྱུའི་ཐག་གཅོད་བྱས་པ་ཡིན་འདུག   སུ་ཇ་བོད་ཁྱིམ་སློབ་གྲྭ་ཟླ་སྔོན་མའི་ཚེས་ ༡༦ ཉིན་ནས་གུང་གསེང་བཏང་ཡོད་པ་དང་། ཚེས་ ༢༠ ཉིན་ནས་སློབ་ཕྲུག་ཕོ་མོ་ ༧ གྱིས་སུ་ཇ་སློབ་གྲྭའི་རྒན་རྒོན་ ༣ ལ་ཞབས་ཞུ་བསྒྲུབས་པ་ནས་མགོ་བཙུགས་ཏེ། ཅོན་ཏ་ར་རྒས་གསོ་ཁང་དང་། དྷ་སྟེང་རྒས་གསོ་ཁང་། དྷ་ས་བྱམས་པ་གླིང་རྒས་གསོ་ཁང་བཅས་སུ་བདུན་ཕྲག་རེ་རེ་བྱས་ཏེ་དང་བླངས་ཞབས་ཞུ་བསྒྲུབས་ཡོད་འདུག   སློབ་ཕྲུག་དེ་དག་གི་དང་བླངས་ཞབས་ཞུའི་རྣམ་གྲངས་ ༥ ཡོད་པ་ནི། དང་པོ། རྒས་གསོ་ཁང་གི་ཁོར་ཡུག་གཙང་མ་བཟོ་བ་དང་། གཉིས་པ། རྒན་རྒོན་ཚོའི་ཉལ་ཁང་གཙང་མ་བཟོང་བ། གསུམ་པ། ཉལ་ཆས་དང་གྱོན་གོས་གཙང་མ་བཟོ་བ། བཞི་པ། གཟུགས་པོ་འཁྲུད་པ་དང་སྐྲ་དང་སེན་མོ་གཙང་མ་བཟོས་པ། ལྔ་པ། གཟིགས་པོར་འཕུར་མཉེད་དང་སྨན་བཅོས། ཁ་བརྡ་སེམས་གསོ་སོགས་བྱས་ཏེ་ཁམས་དྭངས་མོ་བཟོ་རྒྱུ་སོགས་ཡིན་འདུག   ཕ་ཡུལ་རྒྱལ་རོང་ནས་ཡིན་པའི་སློབ་ཕྲུག་བཀྲ་ཤིས་ལགས་སུ། ཐེངས་འདིར་རྒས་གསོ་ཁང་ཁག་ཏུ་དང་བླངས་ཞབས་ཞུ་བསྒྲུབས་པའི་མྱོང་ཚོར་དྲིས་པར་ཁོང་གིས། “ཚོར་བ་ཤུགས་ཆེན་པོ་ཞིག་གང་བྱུང་སོང་ཟེར་ན། རྒན་རྒོན་དེ་ཚོར་བོད་དོན་ལ་སེམས་ཤུགས་ཆེན་པོ་དང་། སྐུ་མདུན་ལ་དད་པ་དམ་ཚིག་ཆེན་པོ་ཧ་ལས་པ་འདུག ཁོ་རང་ཚོས་ང་ཚོར་བཤད་རྒྱུ་གཙོ་བོ་ཨང་དང་པོ་སྐུ་མདུན་གྱི་བཀའ་གསུང་ཅི་སྒྲུབ་ཞུ་དགོས་པ་དང་། གཉིས་པ་དེ་རང་རང་གི་གནས་སྟངས་འགྲིག་ཙམ་བྱུང་ན་བསྡད་འཇོག་རྒྱུ་མ་ཡིན་པར་བོད་སྤྱི་དོན་ལ་ལས་ཀ་བྱེད་དགོས་པ་ཤུགས་ཆེན་པོ་གསུང་གི་འདུག” ཅེས་བརྗོད་སོང་།   ཕ་ཡུལ་གཞིས་ཀ་རྩེ་ནས་ཡིན་པའི་བཀྲ་ཤིས་ལགས་ཀྱིས། “ང་ཚོ་རྒས་གསོ་ཁང་ཁག་ཏུ་འགྲོ་སྐབས། རྒན་རྒོན་ཁ་ཤས་ཀྱི་གཤིས་ཀ་གྱོང་པོ་ཞེ་དྲགས་འདུག ཁ་ཤས་ཀྱིས་ཁོ་རང་ཚོའི་ཅ་ལག་དང་གྱོན་ཆས་དེ་ཚོར་ལག་པ་འཆང་འཇུག་གི་མི་འདུག གང་ཡིན་ཟེར་ན། ག་ཚོར་དོགས་པ་ཟ་བཞིན་ཡོད་ས་རེད། སྔོན་མ་ཁོ་རང་ཚོར་འབྲེལ་བ་ཡོད་མཁན་གཞོན་སྐྱེས་དེ་འདྲས་དངུལ་དང་ཅ་དངོས་རྐུ་མ་བརྒྱབ་པ་དེ་འདྲ་བྱུང་ཡོད་ས་རེད། ང་ཚོས་གཙང་མ་ཡག་པོ་བཟོས་ཚར་བའི་རྗེས་ལ་ཁོ་རང་ཚོས་དགའ་བོ་ཞེ་དྲགས་གནང་གི་འདུག” ཅེས་བརྗོད་བྱུང་།   ཁོས་མུ་མཐུད་དུ་བརྗོད་དོན། “རྒན་རྒོན་དེ་ཚོས་ཅ་ལག་འདྲ་མི་འདྲ་ཉར་འདུག ཁ་ཤས་ལ་ཤིང་བཟོ་བའི་ཆ་ལག་ཆ་ཚང་བ་དེ་འདྲ་འདུག ང་ལ་ཉམས་མྱོང་ག་རེ་བྱུང་སོང་ཟེར་ན། རྒན་རྒོན་དེ་ཚོའི་ཅ་ལག་སྔོན་མ་གང་དུ་བཞག་ཡོད་པ་དེ་རང་ལ་འཇོག་རྒྱུ་མ་གཏོགས་འཇོག་ས་བརྗེ་བོ་བརྒྱབ་པ་ཡིན་ན། ཁོ་རང་ཚོས་མ་རྙེད་པ་དང་། དྲན་པ་རྗེད་འགྲོ་ཡི་འདུག དེ་ནས་ཁོ་རང་ཚོའི་ཅ་ལག་དེ་ལ་འགྱུར་བ་རྐང་གཅིག་ཀྱང་གཏོང་ཆོག་གི་མི་འདུག དཔེར་ན། ཁ་པར་ཞིག་ཡིན་ན། ཁོ་རང་ཚོའི་ཁ་པར་དེ་ཡག་པོ་མེད་ནའང་དེ་ལས་ཡག་ཅིག་སྤྲད་ན་ཁྱོན་ནས་ལེན་རོགས་བྱེད་ཀྱི་མི་འདུག” ཅེས་བརྗོད་སོང་།   ཁོང་གཉིས་ལ་ཐེངས་འདིའི་ཞབས་ཞུ་བ་ཚོ་སྡོད་ས་དང་ཟ་མ། ཕར་ཚུར་འགྲུལ་བཞུད་ཀྱི་འགྲོ་སོང་སོགས་ཇི་ལྟར་བཏང་བ་ཡིན་སྐོར་དྲིས་པར། ཁོང་ཚོས་ཐོག་མ་སུ་ཇ་བོད་ཁྱིམ་གྱི་དགེ་ལས་སློབ་ཕྲུག་ཚོར་ཞལ་འདེབས་རེ་བསླངས་ཏེ་དངུལ་ཏོག་ཙམ་རག་ཡོད་པ་དང་། རྗེས་ལ་ཁོ་རང་ཚོའི་ལས་དོན་ལ་རྒྱབ་སྐོར་གནང་མཁན་མི་སྒེར་བ་རེ་ཟུང་གིས་ཀྱང་ཞལ་འདེབས་སྤྲད་དེ་འགྲོ་སོང་ཚང་མ་དེ་ནས་བཏང་ཡོད་པ་མ་ཟད། དངུལ་དེས་རྒས་གསོ་ཁང་ཁག་ལ་ཞལ་ལག་སྟབས་བདེ་རེའང་བཏང་ཡོད་སྐོར་ངོ་སྤྲོད་གནང་སོང་།   སློབ་ཕྲུག་དང་བླངས་ཞབས་ཞུ་ཚོགས་ཆུང་དེའི་ལས་བསྡོམས་ནང་འཁོད་པ་ལྟར་ན། ཅོན་ཏ་ར་རྒས་གསོ་ཁང་དུ་རྒན་རྒོན་ ༢༤ དང་། དྷ་སྟེང་བོད་ཁྱིམ་དུ་རྒན་འཁོགས་ ༢༧ ། བྱམས་པ་གླིང་རྒས་གསོ་ཁང་དུ་རྒན་རྒོན་ ༡༦༦ བཅས་ཡོད་འདུག དང་བླངས་སློབ་ཕྲུག་ ༧ ནི། བསྟན་འཛིན་དང་ནམ་མཁའ། སྤེན་པ། མཁའ་འགྲོ། ངག་དབང་དར་རྒྱས། ཉི་མ། བཀྲ་ཤིས་བཅས་ཡིན་འདུག གསར་འགོད་པ། དོན་འགྲུབ་བཀྲ་ཤིས།
