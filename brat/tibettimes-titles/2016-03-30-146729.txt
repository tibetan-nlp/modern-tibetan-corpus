བོད་མི་ ༦༨༦ ཁེ་ན་ཌར་བསླེབས་ཚར་ཡོད།

ཨ་རུ་ན་ཅལ་ནས་བོད་མི་ ༡༠༠༠ ཁེ་ན་ཌར་གཞུང་འབྲེལ་གནས་སྤོར་གཏོང་བའི་ལས་གཞིའི་ཁྲོད་སྐྱབས་བཅོལ་ལག་དེབ་མངའ་སྡེ་གཞན་ནས་བཟོས་པ་དང་རྒྱ་གར་གྱི་འོས་བསྡུའི་ཆ་ཤས་ལེན་པའི་ཡིག་ཆ་བཟོས་པ་སོགས་ཀྱིས་དཀའ་ངལ་འཕྲད་ཀྱི་ཡོད་སྐོར་གསུངས་སོང་། བོད་མི་མང་སྤྱི་འཐུས་ལྷན་ཚོགས་ཀྱི་གྲོས་ཚོགས་ཚོགས་དུས་ ༡༡ པའི་ལས་ཉིན་ ༩ པའི་སྔ་དྲོར། བོད་མིའི་སྒྲིག་འཛུགས་ལས་བྱེད་འདེམས་བསྐོ་ལྷན་ཁང་གི་སྒྲིག་གཞི་དང་། ལས་ཤག་དང་བགྲེས་ཤག་བགོ་འགྲེམས་ཀྱི་སྒྲིག་གཞི། ལས་བྱེད་ལས་དོན་གཟེངས་ཐོན་གྱི་གཟེངས་རྟགས་གནང་ཕྱོགས་སོགས་གཞུང་འབྲེལ་སྒྲིག་གཞི་ཁག་ཅིག་བསྐྱར་བཅོས་གཏན་འབེབས་གནང་གྲུབ་རྗེས། ནང་སྲིད་བཀའ་བློན་རྒྱ་རི་སྒྲོལ་མ་མཆོག་ནས་རྒྱ་གར་བྱང་ཤར་ཨ་རུ་ན་ཅལ་མངའ་སྡེ་ནས་བོད་མི་ ༡༠༠༠ ཁེ་ན་ཌར་གཞིས་སྤོར་གྱི་ལས་གཞི་དང་འབྲེལ་བའི་ཁ་སང་གི་དྲི་བ་ཞིག་ལ་བཀའ་ལན་གནང་སྐབས། ལས་གཞི་དེའི་ནང་ཁྱོན་བོད་མི་ ༨༩༠ ལ་རྒྱན་བརྒྱབ་ཟིན་པའི་ཁོངས་ནས་མི་གྲངས་ ༦༨༦ ཕར་བསླེབས་ཚར་ཡོད་སྐོར་གསུངས་སོང་། ནང་སྲིད་བཀའ་བློན་མཆོག་ནས་དེའི་སྐོར་གསུངས་དོན། མི་འགྲོ་ ༡༠༠༠ གི་ནང་ནས་མི་ ༨༩༠ ཞིག་རྒྱན་ཤོག་ནང་བསླེབས་ཡོད་པ་དང་། ཕྲུ་གུ་གསར་དུ་སྐྱེས་པ་དང་ཆང་ས་བརྒྱབ་པ་སོགས་ཀྱི་ཐོག་ནས་གྲངས་ཀ་ཡར་མར་འགྲོ་མི་སྲིད་པ་མིན་པས་ ༡༡༠ ལ་རྒྱན་མ་བརྒྱབ་པར་བཞག་ཡོད་ལ། ད་ལྟའི་ཆར་ཕྲུ་གུ་གསར་དུ་སྐྱེས་པ་ ༢༡ དང་། འདས་གྲོངས་ཕྱིན་པ་ ༦ འདུག ད་ལྟའི་བར་མི་འགྲོ་ ༨༩༠ ཡི་ནང་ནས་མི་ ༨༧༩ ལ་གཞུང་ཚབ་ནས་སྨན་བཅོས་བརྟག་དཔྱད་དང་བཅར་འདྲི་བྱས་ཚར་ཡོད་པའི་ནང་ནས་ཐོན་ཚར་བ་མི་ ༦༨༦ ཡོད་པ་དང་། ཕལ་ཆེར་སང་ཉིན་ཡིན་ས་ཡོད་འཐོན་མཁན་མི་ ༣ ཡོད། དེ་ནས་འདི་ལོའི་ཕྱི་ཟླ་ ༤ པའི་ནང་མི་ ༦༠ ཞིག་འཐོན་ཐུབ་པ་བྱེད་ཀྱི་ཡོད་ལ། ཕྱི་ཟླ་ ༥ པའི་ནང་འཐོན་མཁན་ཁག་ཅིག་བང་རིམ་བསྒྲིགས་བཞག་ཡོད། ཅེས་དང་། ཡང་ཁོང་མོས། ལས་གཞི་དེ་དུས་བཀག་གི་ལས་གཞི་ཞིག་ཡིན་པས་དུས་བཀག་ནང་ཚར་ཐུབ་དང་མི་ཐུབ་ཟེར་བ་དེ་དེའི་སྔོན་གྱི་གྲོས་ཚོགས་ནང་དུའང་ཞུས་པ་ཡིན། ང་ཚོའི་ཐོག་ནས་དེར་བློ་ཚབ་ཡོད། ལས་གཞིའི་འགན་གཙོ་བོ་འཁུར་མཁན་དེ་ཁེ་ན་ཌ་བོད་ཀྱི་ལས་འཆར་འགན་འཛིན་ཉི་མ་རྡོ་རྗེ་ལགས་ཆགས་བསྡད་ཡོད་ལ། ཁོང་ལའང་དེ་ཞུ་གཏུག་ཅིག་བྱེད་དགོས་ཀྱི་འདུག ང་ཚོ་ཕར་བཅར་དགོས་ཀྱི་ཡོད་ན་རེད། ཡང་མིན་ན་དོན་གཅོད་དང་སྤྱི་མཐུན་ཚོགས་པ་བརྒྱུད་ནས་ཡིན་ནའང་རེད། དུས་ཡུན་འཕར་འགྱངས་ཤིག་ཞུ་དགོས་ཀྱི་འདུག་ཅེས་བཀའ་བསྡུར་བྱུང་སྐབས། ཕྱི་ཟླ་ ༥ པའི་བར་དུ་མི་ཕར་བསླེབས་ཚར་དགོས་པ་མིན་པར་ཕྱི་ཟླ་ ༥ པ་བར་དུ་སྙན་ཞུ་དེ་ཚོ་ཚང་མ་ཡར་འབུལ་ཐུབ་པ་དགོས་ཀྱི་འདུག་ལ། ལས་རིམ་དང་བཅར་འདྲི་ལེན་པ་དེ་ཚོ་དངོས་གནས་དྲང་གནས་ཁེ་ན་ཌའི་གཞུང་ཚབ་ཀྱི་ཐོག་ནས་ནམ་རྒྱུན་ཟླ་བ་ ༦ འགོར་ས་ལ་ང་ཚོར་བདུན་ཕྲག་གཉིས་ལས་མ་འགོར་བའི་གནས་སྟངས་དེ་འདྲ་གནང་གི་ཡོད་སྟབས། ཕྱི་ཟླ་ ༥ པ་བར་དུ་སྙན་ཞུ་ཚང་མ་ཡར་འབུལ་ཐུབ་ངེས་རེད། གལ་སྲིད་མ་ཐུབ་པར་ཉི་མ་ཁ་ཤས་ཕར་འགྱངས་བྱེད་དགོས་བྱུང་བ་ཡིན་ནའང་དཀའ་ངལ་ཡོང་གི་མ་རེད་ཅེས་ང་ཚོར་ཚུར་གསུངས་བྱུང་། ཞེས་གསུངས་སོང་། ཁེ་ན་ཌའི་གཞུང་ཚབ་ནས་ངོས་བཞེས་མ་གནང་བའི་དཀའ་ངལ་ཁག་ཅིག་ཡོད་སྐོར་ཁོང་མོས། གཞུང་ཚབ་ནས་དགག་པ་བརྒྱབ་བཞག་པའི་དཀའ་ངལ་ཡོད་པའི་གནས་སྟངས་ལ་ཞིབ་དཔྱོད་བྱེད་དགོས་པ་ཞིག་ཆགས་ཡོད། ཞིབ་དཔྱོད་མང་བ་ཆེ་སྐྱབས་བཅོལ་ལག་དེབ་ཨ་རུ་ན་ཅལ་ནས་མིན་པར་ཕྱི་ལོགས་ནས་བཟོས་ཡོད་པ་དེ་ཚོའི་ནང་ནས་ཁ་ཤས་ཤིག་ལ། མི་དེ་ཨ་རུ་ན་ཅལ་རང་ལ་སྡོད་མཁན་ཡིན་དང་མིན་གྱི་དོགས་གཞིའི་ཐོག་ནས་ཞིབ་དཔྱོད་བྱེད་དགོས་པ་ཆགས་ཀྱི་འདུག དེ་བཞིན་བར་ལམ་ཞིག་ནས་རྒྱ་གར་གྱི་འོས་བསྡུ་ཆེན་མོའི་ནང་ཆ་ཤས་ལེན་ཆོག་གི་ཡོད་པ་རེད་ཅེས་རེ་ཟུང་ཞིག་གིས་ཡིག་ཆ་དེ་བཟོས་ཡོད་པ་དང་། གཞུང་ཚབ་ཀྱིས་ག་འདྲ་ཁོ་རང་ཚོས་ཕར་ཚུར་མཁྱེན་གྱི་ཡོད་མེད་ང་ཚོས་གསལ་པོ་ཞུ་ཐུབ་ཀྱི་མི་འདུག་སྟེ། གང་ལྟར་དེ་འདྲའི་གནས་ཚུལ་ཐོན་པའི་རྒྱུ་མཚན་ལ་བརྟེན། ཁེ་ན་ཌར་གཞིས་སྤོར་ལས་གཞིའི་ནང་རྒྱལ་ཁབ་གཞན་ཞིག་གི་ཡུལ་མི་བྱེད་ཐུབ་པའི་གནས་སྟངས་ཡོད་པ་ཡིན་ན། སྐྱབས་བཅོལ་གྱི་ལས་གཞི་དེའི་ནང་འོས་ཆོས་ཚང་གི་མེད་ཅེས་པའི་ཚིག་ཅིག་འདུག དེ་འདྲའི་ཐོག་ནས་ཅིག་གསུང་གི་འདུག ཡིན་ནའང་། ང་ཚོས་དེ་ཚོའི་ནང་ནས་རེ་ཟུང་གཙང་མ་བྱེད་ཐུབ་སོང་ལ། རེ་ཟུང་ད་དུང་ཡང་གཙང་མ་མ་ཆགས་པར་ལྷག་བསྡད་ཡོད་པ་རེད། ཅེས་དང་། མཐའ་མའི་ཐག་གཅོད་དེ་ཁོ་རང་ཚོའི་ལག་པར་ཡོད་ཀྱང་ནང་སྲིད་ནས་ཞུ་གཏུག་བྱེད་དགོས་པ་ཚང་མ་བྱེད་བཞིན་ཡོད་སྐོར་གསུངས་སོང་། གསར་འགོད་པ། པདྨ་མཚོ།
