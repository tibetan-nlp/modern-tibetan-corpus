རི་ཉིལ་བྱུང་སྟེ་དགོན་པ་བརྡིབས།

ཁ་སང་ཡུལ་ཤུལ་ཁུལ་འདམ་དཀར་དགོན་རྙིང་དུ་རི་ཉིལ་བྱུང་སྟེ་དགོན་པའི་མཆོད་རྟེན་སོགས་ལ་རྡིབ་སྐྱོན་བྱུང་ཡོད་འདུག ཁ་སང་ཕྱི་ཚེས་ ༢ ཉིན་གྱི་ཕྱི་དྲོ་བོད་ནང་གི་ཆུ་ཚོད་ ༦ པ་ཡས་མས་སུ། ཡུལ་ཤུལ་ཁུལ་འདམ་དཀར་དགོན་རྙིང་ཆགས་སའི་རི་ལྡེབས་ཏུ་ས་ཉིལ་བྱུང་སྟེ། དགོན་པའི་མཆོད་རྟེན་དང་ཁང་པ་སོགས་ལ་རྡིབ་སྐྱོན་བྱུང་ཡོད་འདུག རི་ཉིལ་བྱུང་རྗེས་ཡུལ་ཤུལ་གྲོང་ཁྱེར་མི་མང་སྲིད་གཞུང་གིས་ཉེན་བརྡ་བཏང་སྟེ། འདམ་དཀར་དགོན་རྙིང་གི་ཉེ་འཁོར་དུ་འགྲོ་འོང་བྱེད་སྐབས་བདེ་འཇགས་ལ་ཡིད་གཟབ་དགོས་སྐོར་དང་། གནས་སྐབས་སུ་སྐྱེལ་འདྲེན་རླངས་འཁོར་སོགས་ལྗིད་ཚད་ཆེ་བའི་རླངས་འཁོར་རིགས་འགྲོ་འོང་བྱེད་པར་བཀག་འགོག་བྱ་རྒྱུ་ཡིན་ལུགས་བསྒྲགས་འདུག འདི་གར་མིང་བརྗོད་འདོད་མེད་མཁན་ཞིག་གིས་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། དེ་སྔོན་རྒྱ་རིགས་ས་དཔྱད་པ་ཞིག་གིས། འདམ་དཀར་དགོན་རྙིང་ཆགས་པའི་ས་ཆ་དེར་ས་རུད་བྱུང་བའི་ཉེན་ཁ་ཡོད་སྐོར་བརྗོད་ཡོད་པ་ལྟར། དགོན་རྙིང་ནས་སྤྱི་ལེ་ ༣ ཙམ་གྱི་སར་དགོན་པ་གསར་རྒྱག་བྱས་ཏེ་ཁྱོན་རྒྱ་སྒོར་ས་ཡ་ ༨༠ ཙམ་གྱི་འགྲོ་གྲོན་བཏང་ཡོད་པ་དང་། འདི་ལོའི་ཕྱི་ཟླ་ ༦ པ་ཙམ་ལ་དགོན་གསར་ལེགས་འགྲུབ་བྱུང་སྟེ་དགེ་འདུན་པ་ཚང་མ་དགོན་གསར་ནང་གནས་སྤོས་ཡོད་འདུག ཐེངས་འདིའི་རི་ཉིལ་ཁྲོད་དགོན་པའི་ནང་རྟེན་ལ་གནོད་སྐྱོན་ཆེན་པོ་ཐེབས་ཡོད་མེད་གསལ་བོ་མེད་ཀྱང་། མིའི་ཚེ་སྲོག་ལ་གནོད་སྐྱོན་བྱུང་མེད་པ་རེད། འདམ་དཀར་ལྷུན་གྲུབ་བདེ་ཆེན་ཆོས་འཁོར་གླིང་ནི་སློབ་དཔོན་ཆེན་པོ་འབར་ཚ་སྟག་རྡེལ་བའམ་ཉི་མ་རས་ཆེན་ཞུ་བས་ཕྱི་ལོ་ ༡༡༩༠ ནང་གསར་དུ་ཕྱག་བཏབ་པ་དང་། དགོན་པའི་ལོ་རྒྱུས་མདོར་བསྡུས་ནང་ད་ལྟའི་ཆར་དགོན་པ་དེར་རྒྱ་གཞུང་གིས་ཆོག་མཆན་ཐོབ་པའི་གྲྭ་པ་ ༣༠ དང་གྲྭ་ཤག་ ༩༠ ཡོད་ལུགས་བཀོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
