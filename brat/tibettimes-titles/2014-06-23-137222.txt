བོད་མི་ཞིག་བསད་དེ་ཆུའི་ནང་གཡུཊ།

ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༨ ཉིན་རྔ་བ་ཁུལ་ཁྱུང་མཆུ་རྫོང་གི་བོད་མི་སངས་ཡས་ཟེར་བ་ཞིག་ཉེན་རྟོག་པས་བསད་དེ་ཆུའི་ནང་དུ་གཡུགས་ཡོད་འདུག འདི་གར་ཆབ་སྲིད་བཙོན་ཟུར་དགེ་འདུན་ཚེ་རིང་ལགས་ཀྱིས་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༨ ཉིན། རྔ་བ་ཁུལ་ཁྱུང་མཆུ་རྫོང་གསེར་དེའུ་ཡུལ་ཚོ་ནས་ཡིན་པའི་སངས་ཡས་ཟེར་བ་ཞིག་གིས། ཁྲིན་ཏུའུ་གྲོང་ཁྱེར་ནས་རླངས་འཁོར་བཏང་སྟེ་སྤོ་དཀར་རི་བོའི་ཕུག་ལམ་དུ་འབྱོར་སྐབས། བཀྲ་ཤིས་གླིང་(li xian)ཉེན་རྟོག་པས་རྗེས་འདེད་བྱས་ཏེ་མེ་མདའ་བརྒྱབ་ནས་བསད་པ་མ་ཟད། ལྐོག་ནས་ཕུང་པོ་རྒྱལ་མོའི་རྔུལ་ཆུའི་ནང་གཡུགས་ཡོད་ཀྱང་། ལམ་བཟོ་བ་ཁ་ཤས་ཀྱིས་ཕུང་པོ་རྙེད་དེ་འབྲེལ་ཡོད་ནས་བརྟག་དཔྱད་བྱས་རྗེས། ཉེན་རྟོག་པས་བསད་པ་ཤེས་རྟོགས་བྱུང་བ་ལྟར་ངོས་ལེན་བྱས་ཡོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༩ ཙམ་ལ་སངས་ཡས་ཀྱི་ཁྱིམ་མི་གཉེན་འབྲེལ་སོགས་མི་ ༥༠ ལྷག་བཀྲ་ཤིས་གླིང་དུ་ཕྱིན་ཏེ། འབྲེལ་ཡོད་ལས་ཁུངས་ལ་ཞུ་གཏུག་བྱས་རྗེས། དེ་རིང་ཕྱི་ཚེས་ ༢༣ ཉིན་བཀྲ་ཤིས་གླིང་ཉེན་རྟོག་ཁང་གིས་ཁྱིམ་མི་ཚོར་རྒྱ་སྒོར་ཁྲི་ ༦ སྤྲད་དེ། གནས་ཚུལ་ལ་རིམ་གྱིས་ཞིབ་འཇུག་གསལ་བོ་བྱས་ཏེ་རོ་སྐྱིན་སྤྲོད་རྒྱུ་ཡིན་པས། གཞན་ལ་བཤད་མི་ཆོག་ཅེས་བཀའ་རྒྱ་བཏང་ཡོད་པ་མ་ཟད། ཕུང་པོ་ལ་པར་རྒྱག་ཏུ་མ་བཅུག་པར་འཕྲལ་དུ་མེར་བསྲེགས་བཏང་ཡོད་འདུག གནས་ཚུལ་མཁོ་སྤྲོད་པས། “ད་ལྟའི་ཆར་སངས་ཡས་གསོད་རྐྱེན་ཅི་ཡིན་གསལ་བོ་བྱུང་མེད་ཀྱང་། བཀྲ་ཤིས་གླིང་ཉེན་རྟོག་ཁང་གིས་ཁོ་བོའི་རླངས་འཁོར་དེ་བརྐུས་པ་ཡིན་ལུགས་དང་། ཡིག་ཆ་མེད་པའི་ཉེས་འཛུགས་བྱས་འདུག ཡིན་ནའང་། རྐུན་རྫས་ཡིན་པའི་དཔང་རྟགས་ཅི་ཡང་སྟོན་རྒྱུ་རག་མི་འདུག” ཅེས་བརྗོད་བྱུང་། སངས་ཡས་ནི་རྔ་བ་ཁུལ་ཁྱུང་མཆུ་རྫོང་གསེར་དེའུ་ཡུལ་ཚོའི་རུ་ཁག་དང་པོ་ནས་ཡིན་ཞིང་། ད་ལྟ་ལོ་ ༤༠ མ་ཟིན་ཙམ་ཡིན་པ་དང་། རྒྱུན་དུ་འབྲོག་ལས་གཉེར་ཞོར་དུ་ཚོང་བརྒྱབ་ནས་བསྡད་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
