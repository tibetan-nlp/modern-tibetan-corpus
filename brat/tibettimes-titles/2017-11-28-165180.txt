སློབ་ཕྲུག་གི་གུང་གསེང་ཞབས་ཞུ།

སྦེལ་ཀོབ་བོད་ཁྱིམ་གྱི་སློབ་ཕྲུག་ཁག་ཅིག་གིས་གཙང་སྦྲ་རོགས་རམ་བྱེད་བཞིན་པའི་ལུགས་བསམ་རྒས་གསོ་ཁང་གི་རྒན་འཁོགས་ཚོས་སོ་སོའི་བུ་ཕྲུག་ཡིན་ནའང་རོགས་རམ་ལྟ་སྐྱོང་འདི་འདྲ་བྱེད་ཁག་པོ་ཡིན་སྐོར་བརྗོད་བྱུང་། རྒྱ་གར་ལྷོ་ཕྱོགས་སྦེལ་ཀོབ་བོད་ཁྱིམ་གྱི་འཛིན་རིམ་དགུ་པ་དང་བཅུ་པའི་སློབ་ཕྲུག་ཚོ་རེས་མོས་ཀྱིས་གཟའ་ཉི་མའི་སྔ་དྲོའི་ཆུ་ཚོད་ ༧ ནས་ ༡༡ ་བར་འཆར་ཅན་ལྟར།  ལུགས་བསམ་རྒས་གསོ་ཁང་གི་རྒན་འཁོགས་ཚོར་གཟུགས་པོ་དང་གྱོན་གོས་འཁྲུད་པ་དང་། ནང་ལ་གཙང་སྦྲ།  རྐང་ལག་གི་སེན་མོ་གཏུབ་པ། སྐྲ་བཞར་བ་སོགས་ཀྱི་དྭང་བླངས་རོགས་རམ་བྱེད་བཞིན་ཡོད་པ་མ་ཟད། སློབ་ཕྲུག་རང་ངོས་ནས་དངུལ་ཕྲན་བུ་བསྡུས་ཏེ་རྒན་འཁོགས་ཚོར་སྐྲ་བཞར་འཕྲུལ་ཆས་དང་། ཕྱགས་མ། གད་སྙིགས་སྒམ། རྩ་གཏུབ་བྱེད་སོགས་ཀྱང་ཉོས་འདུག འདི་གའི་གསར་འགོད་པས་ལུགས་ཟུང་བསམ་འགྲུབ་གླིང་རྒས་གསོ་ཁང་གི་འགན་འཛིན་ཐུབ་བསྟན་ཆོས་འགྲུབ་ལགས་སུ། སྦེལ་ཀོབ་བོད་ཁྱིམ་གྱི་སློབ་མ་ཚོའི་རོགས་རམ་ཐད་གཟིགས་སྣང་ཅི་བྱུང་དྲིས་པར་ཁོང་གིས། སློབ་ཕྲུག་ཚོ་གཟའ་ཉི་མ་རྟག་པར་འདིར་ཡོང་ནས་བགྲེས་སོང་ཚོ་ལ་གོན་པ་དང་གཟུགས་པོ་འཁྲུད་རོགས་དང་། ཁང་པའི་ནང་གད་ཕྱིས་ཆ་ཚང་བྱེད་ཀྱི་ཡོད་པ་རེད། རྒན་འཁོགས་ཚོ་ལ་ཧང་སང་བ་བྱམས་པོ་གནང་གི་འདུག  ཁོ་རང་ཚོའི་ཕ་མའི་ཐོག་ལ་ག་འདྲ་ཡོད་མེད་མི་ཤེས་ཏེ་འདིར་ཡོང་དུས་ཧང་སང་བ་ཡག་པོ་འདུག ཅེས་བརྗོད་བྱུང་། ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༦ རེས་གཟའ་ཉི་མ་ཉིན། གསར་འགོད་པ་རྒས་གསོ་ཁང་དུ་ཕྱིན་ནས་དངོས་སུ་རྒྱུས་ལོན་བྱེད་སྐབས། སློབ་ཕྲུག་ཁག་ཅིག་གིས་རྒན་འཁོགས་ཚོར་སྐྲ་བཞར྄་བ་དང་། ནང་ལ་གཙང་སྦྲ་སོགས་བྱས་ཏེ་བྲེལ་བ་རེད་འདུག དྭང་བླངས་ཞབས་ཞུ་བ་འཛིན་རིམ་བཅུ་པའི་སློབ་ཕྲུག་ལྷ་འཛོམས་ཀྱིས་གསར་འགོད་པར། ང་ཚོས་དྭང་བླང་ལས་ཀ་འདི་ཕྱི་ལོ་ ༢༠༠༧ནས་མགོ་འཛུགས་གནང་ཡོད་པ་དང་། དེ་ཡང་ལོ་རེར་འཛིན་རིམ་བཅུ་པ་ཐོན་མཁན་ཚོས་ཡང་འཛིན་རིམ་དགུ་པའི་སློབ་ཕྲུག་ཚོ་ལ་ལས་འགན་དེ་རྩིས་སྤྲོད་བྱས་ཏེ། སློབ་ཕྲུག་གཉིས་རེར་བགྲེས་སོང་ཕོ་མོ་རེ་བདག་པོ་རྒྱག་ཏུ་འཇུག་གི་ཡོད་པ་རེད། ཅེས་བརྗོད་བྱུང་། ཕ་ཡུལ་ཀོང་པོ་ནས་ཡིན་པའི་བགྲེས་སོང་གཙང་རྒྱས་ལགས་སུ་སློབ་ཕྲུག་ཚོས་རོགས་རམ་ག་འདྲ་བྱེད་ཀྱི་འདུག་ཅེས་དྲིས་པར་ཁོང་གིས། སློབ་ཕྲུག་ཚོ་ཧ་ཅང་ཡག་པོ་འདུག གཟུགས་པོ་འཁྲུད་རོགས་དང་གོན་པ། ད་དུང་ཁང་པ་ནང་སོགས་ལ་གད་ཕྱིས་ཧ་ཅང་ཡག་པོ་བྱེད་ཀྱི་འདུག ང་ཚོས་ནམ་ཡིན་ནའང་སྨོན་ལམ་བརྒྱབ་བཞིན་ཡོད། དངོས་ནས་རང་གི་བུ་ཕྲུག་ཡིན་ནའང་དེ་ལས་ལྷག་ཞིག་ཁག་པོ་རེད་ལབ་ཀྱི་ཡོད། ཅེས་བརྗོད་བྱུང་། སྤྱིར་བཏང་རྒས་གསོ་ཁང་འདི་བོད་མིའི་སྒྲིག་འཛུགས་ནང་སྲིད་ལས་ཁུངས་ཀྱི་ཁྱབ་ཁོངས་སུ་ཡོད་ཀྱང་། ཕྱི་ལོ་ ༡༩༩༦ ནས་༢༠༡༣་བར་དུ་ཨི་ཌ་ལིའི་མི་སེར་ཡིན་པའི་བོད་རིགས་བུད་མེད་བསྟན་འཛིན་ཚེ་ལྷ་ལགས་ཀྱིས་འགྲོ་སོང་སོགས་ཀྱི་འགན་གཙོ་བོ་འཁུར་གནང་ཡོད་སྟབས། གསར་འགོད་པས་བསྟན་འཛིན་ཚེ་ལྷ་ལགས་སུ། ད་ལྟ་རྒས་གསོ་ཁང་ལ་རོགས་རམ་ལྟ་སྐྱོང་མུ་མཐུད་གནང་གི་ཡོད་མེད་དྲིས་བར་མོས། སྔོན་མ་ཡིན་ན་ངས་ང་རང་གི་ཁང་པ་དང་དཔལ་འབྱོར་ཡོད་ཚད་བཙོང་ནས་རྒས་གསོ་ཁང་འདི་ལ་རོགས་སྐྱོར་བྱས་པ་ཡིན། ཕྱི་ལོ་༢༠༡༣ ལོ་ནས་ང་རང་རྒན་ཡོལ་དུ་ཕྱིན་ཏེ་དངུལ་ཀྱི་ཡོངས་ཁུངས་གང་ཡང་མེད་སྟབས། ཕར་འགྲོ་ཚུར་འོང་གི་ལམ་གླ་སོགས་ལ་ཡང་དཀའ་ངལ་འཕྲད་ཡོང་གི་འདུག་པས། ཕྱི་ལོ་ ༢༠༡༣ ལོ་ནས་རོགས་སྐྱོར་མཚམས་བཞག་པ་ཡིན། ཞེས་བརྗོད་བྱུང། རྒས་གསོ་ཁང་འདི་ཐོག་མར་ཕྱི་ལོ་ ༡༩༩༨ལོར་བཙུགས་ཡོད་པ་དང་།  ད་ལྟ་སྨན་བཅོས་ཀྱི་ལས་བྱེད་གཅིག་དང་། རྒན་འཁོགས་ཚོ་ལ་ལྟ་རྟོག་རོགས་རམ་དང་གཙང་སྦྲ་བྱེད་མཁན་ལས་བྱེད་གསུམ། མ་བྱན་གཅིག སྤྱི་ཡོངས་ལ་ལྟ་རྟོག་གནང་མཁན་འགན་འཛིན་གཅིག་བཅས་ཁྱོན་ལས་བྱེད་ ༦ དང་། རྒན་འཁོགས་ཕོ་མོ་ ༣༤ འདུག  ལྷོ་ཕྱོགས་ས་གནས་གསར་འགོད་པ་དོན་འགྲུབ་བཀྲ་ཤིས།
