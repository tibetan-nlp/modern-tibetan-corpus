ཨི་བྷོ་ལ་ནད་ཡམས་ཉེན་ཁ་ཆེ་རུ་ཕྱིན།

ཁ་སང་རྒྱལ་སྤྱིའི་འཕྲོད་བསྟེན་ལས་ཁུངས་ཀྱིས་གསར་འགོད་གསལ་བསྒྲགས་ཤིག་སྤེལ་ཏེ་ད་བར་ཨི་བྷོ་ནད་ཡམས་ཕོག་མཁན་ ༢༧༩༣ རྐྱེན་འདས་སུ་ཕྱིན་སྐོར་བསྒྲགས་འདུག ཁ་སང་རྒྱལ་སྤྱིའི་འཕྲོད་བསྟེན་ལྷན་ཚོགས་ཀྱིས་ཨི་བྷོ་ལ་ནད་ཡམས་དང་འབྲེལ་བའི་གསར་འགོད་གསལ་བསྒྲགས་ཤིག་སྤེལ་བའི་ནང་། འདི་ལོའི་ཟླ་ ༣ ནས་བཟུང་ཨི་བྷོ་ལ་ནད་ཡམས་ཁྱབ་བརྡལ་ཕྱིན་ཏེ་ཟླ་འདིའི་ཚེས་ ༡༨ ཉིན་བར་མི་ ༢༧༩༣ རྐྱེན་འདས་ཕྱིན་ཡོད་པ་དང་། མི་ ༥༧༦༢ ལ་ནད་ཡམས་དེ་ཕོག་ཡོད་སྐོར་བསྒྲགས་ཡོད་འདུག གསར་འགོད་བསྒྲགས་གཏམ་ནང་། “འདི་ལོའི་ཕྱི་ཟླ་ ༣ ཙམ་ནས་ཨ་ཧྥི་རི་ཀའི་ནུབ་ཕྱོགས་རྒྱལ་ཁབ་ཁག་ཅིག་ལ་ཨི་བྷོ་ལ་ནད་ཡམས་ཁྱབ་བརྡལ་ཕྱིན་ཏེ། ཟླ་འདིའི་ཚེས་ ༡༨ ཉིན་བར་ལེ་བྷི་རི་ཡའི་མི་ ༡༥༧༨ རྐྱེན་འདས་ཕྱིན་ཡོད་པ་དང་། མི་ ༣༠༢༢ ལ་ནད་ཡམས་ཕོག་ཡོད་པ་མ་ཟད། དེ་མིན་སི་ནེ་གྷལ་ནང་ནད་ཡམས་ཕོག་མཁན་ ༡ དང་། ནེ་ཇི་རི་ཡ་ནང་མི་ ༢༡ བྱུང་ཡོད་པ་ལས་མི་ ༨ འདས་གྲོངས་ཕྱིན་ཡོད”  ཅེས་འཁོད་འདུག ད་དུང་མུ་མཐུད་གསར་འགོད་བསྒྲགས་གཏམ་ནང་། “ད་ལྟ་ཨི་བྷོ་ལ་ནད་ཡམས་ནི་རྒྱལ་སྤྱིའི་སྒྲིག་འཛུགས་དང་མང་ཚོགས་ཚོས་སེམས་འཚབ་བྱེད་གཞི་ཞིག་ཆགས་ཡོད་སྟབས། འབྲེལ་ཡོད་གཞུང་ཁག་གིས་ནད་གཞི་འདི་རྒྱལ་ཁབ་གཞན་ནང་ཁྱབ་བརྡལ་མི་འགྲོ་བའི་ཆེད་ནད་ཡམས་ཕོག་མཁན་མི་སྣ་དག་ལ་རྟོག་ཞིབ་ནན་པོ་བྱས་ཏེ་རྒྱལ་ཁབ་གཞན་ནང་འགྲོ་སྐྱོད་ཐོངས་མཆན་བཀག་འགོག་བྱེད་དགོས” ཞེས་བརྗོད་འདུག རྒྱལ་སྤྱིའི་གཞུང་ཁག་དང་འབྲེལ་ཡོད་སྒྲིག་འཛུགས་ཚོས་ཨི་བྷོ་ལ་ནད་པར་ལྟ་རྟོག་ཇི་ལྟར་བྱེད་དགོས་པའི་བསྒྲགས་གཏམ་ནང་།”འབྲེལ་ཡོད་གཞུང་གིས་དགོས་འགལ་མ་བྱུང་བར་ཨི་བྷོ་ལ་ནད་གཞི་ཕོག་པ་དག་ཟུར་བཀག་བྱེད་མི་ཆོག་པ་དང་། ཁོང་ཚོའི་ཟས་གོས་གནས་མལ་ལ་ལྟ་རྟོག་དང་། ཉིན་རེའི་གནས་སྟངས་དག་འབྲེལ་ཡོད་གཞུང་དང་རྒྱལ་སྤྱིའི་འཕྲོད་བསྟེན་ལས་ཁུངས་ཁག་ལ་སྙན་སེང་ཞུ་དགོས་པ་མ་ཟད། ནད་ཡམས་ཁྱབ་བརྡལ་འགྲོ་ཡུལ་རྒྱལ་ཁབ་ཁག་གིས་རྒྱལ་སྤྱིའི་འཕྲོད་བསྟེན་ལས་ཁུངས་ཀྱི་རྟོག་ཞིབ་པ་དང་ཆེད་ལས་མི་སྣའི་བདེ་འཇགས་ལ་སྲུང་སྐྱོབ་བྱེད་དགོས” ཞེས་སོགས་འཁོད་འདུག བདུན་ཕྲག་སྔོན་མའི་ནང་ཨི་བྷོ་ལ་ནད་ཡམས་གོ་རྟོག་སྤེལ་མཁན་གྱི་འཕྲོད་ལས་བྱེད་དང་གསར་འགོད་པ་ཁག་ཅིག་ལྷོ་གྷི་ནིའི་ (Guinea) ས་ཁུལ་གྱི་མང་ཚོགས་ཚོར་གོ་རྟོག་སྤེལ་སྐབས། ཡུལ་མི་ཚོས་འཇུ་བཟུང་བཀག་བསྐྱིལ་བྱས་ཏེ་བསད་ཡོད་འདུག གནད་དོན་དབྱེ་ཞིབ་པ་ཚོས། ད་ལྟ་ཨི་བྷོ་ལ་ནད་ཡམས་ནི་རྒྱལ་སྤྱིའི་ཐོག་ཉེན་ཁ་ཆེ་ཤོས་གྲས་ཀྱི་ནད་གཞི་ཞིག་ཆགས་ཡོད་སྟབས། གལ་ཏེ་ནད་གཞི་འདི་གང་མགྱོགས་བཅོས་ཐབས་མ་བྱས་ཚེ། མ་འོངས་པར་གནོད་ཚབས་སྔར་ལས་ཆེ་རུ་འགྲོ་རྒྱུ་ཡིན་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
