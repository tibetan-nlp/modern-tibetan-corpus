རྒྱ་མིར་གཉེན་སྒྲིག་ན་བྱ་དགའ་ཡོད།

རྒྱ་ནག་གཞུང་གིས་ཤིན་ཅང་ཡུ་གུར་རིཊ་རང་སྐྱོང་ཁུལ་ཆེ་མུའུ་རྫོང་དུ་དངུལ་དངོས་སོཊ་ཀྱི་བྱ་དགའ་སྤྲད་དེ་རྒྱ་རིགས་དང་གཉེན་སྒྲིག་རྒྱུའི་སྐུལ་སློང་བྱེད་ཀྱི་ཡོད་འདུག དེ་རིང་ཕྱི་ཚེས་ ༣ ཉིན་ཨ་རིའི་ནེའུ་ཡོག་དུས་བབ་ (TheNewyorkTimes) སྟེང་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཉེ་ལམ་རྒྱ་ནག་གཞུང་གིས་རྒྱ་ནག་ནུབ་ཕྱོགས་ཤིན་ཅང་ཡུ་གུར་རིགས་རང་སྐྱོང་ཁུལ་གྱི་རྒྱ་རིགས་དང་འཁྲུག་འཛིང་དོན་རྐྱེན་ཐེངས་མང་བྱུང་སའི་ཆེ་མུའུ་རྫོང་དུ། ཡུ་གུར་མང་ཚོགས་ལ་སྐུལ་སློང་གིས་རྒྱ་རིགས་དང་མཉམ་དུ་གཉེན་སྒྲིག་བྱས་ཚེ། ནང་མིའི་ཚོང་ལས་ཙག་ཙིག་གཉེར་ཆེད་དང་ཁྱིམ་ཚང་གསར་རྒྱག་བཅས་ཀྱི་ཆེད་ལོ་ལྔའི་རིང་བསྟུད་མར་རྒྱ་སྒོར་ཁྲི་ ༡ རེའི་བྱ་དགའ་སྤྲོད་རྒྱུ་ཡིན་ཚུལ་བསྒྲགས་ཡོད་པ་མ་ཟད། རྒྱ་རིགས་དང་གཉེན་སྒྲིག་བྱས་པའི་ཁྱིམ་ཚང་ལ་གཞུང་ཕྱོགས་ནས་སྡོད་གནས་དང་ལས་གནས་སོགས་དམིགས་བསལ་གོ་སྒྲིག་གནང་རྒྱུ་དང་། ལོ་རེར་ཉུང་མཐར་རྒྱ་སྒོར་ཁྲི་ ༢ སྨན་བཅོས་རོགས་སྐྱོར་གནང་རྒྱུ། ཕྲུ་གུ་བུ་བཅོལ་ཁང་ནས་མཐོ་འབྲིང་བར་རིན་མེད་ཐོག་བསྐྱལ་རྒྱུ་དང་། ཆེད་ལས་རང་བཞིན་གྱི་སློབ་གྲྭར་བསྐྱོད་མཁན་ཚོར་ལོ་རེར་རྒྱ་སྒོར་ ༣༠༠༠ གི་ཁ་གསབ་དངུལ་སྤྲོད་རྒྱུ། སློབ་ཆེན་དུ་རྒྱུགས་འཕྲོད་མཁན་ཚོར་ལོ་རེར་རྒྱ་སྒོར་ ༥༠༠༠ གི་ཁ་གསབ་ཀྱི་དངུལ་སྤྲོད་རྒྱུ་ཡིན་སྐོར་སོགས་བསྒྲགས་ཡོད་འདུག ཐེངས་འདིའི་རྒྱ་གཞུང་གི་སྲིད་ཇུས་དེ་ཤིན་ཅང་ཡུ་གུར་རིགས་རང་སྐྱོང་ཁུལ་ཆེ་མུའུ་རྫོང་གི་གཞུང་འབྲེལ་དྲ་རྒྱར་བཀོད་ཡོད་སྟབས། རྒྱལ་སྤྱིའི་ཆབ་སྲིད་དཔྱད་བརྗོད་པ་ཚོས། དེ་ནི་རྒྱ་ནག་གཞུང་གི་ཉེ་བའི་ལོ་མང་རིང་འཆར་གཞི་ལྡན་པའི་ཐོག་བོད་ཀྱི་ས་ཁུལ་ཁག་ལ་ཉམས་ཞིབ་རིམ་པ་བྱས་རྗེས་བོད་ནང་གི་བོད་མི་ཚོ་རྒྱ་རིགས་དང་གཉེན་སྒྲིག་བྱེད་རྒྱུའི་སྲིད་ཇུས་ལག་བསྟར་བྱེད་ཀྱི་ཡོད་པ་དང་གཅིག་མཚུངས་ཆགས་ཡོད་ཚུལ་དང་། ཤིན་ཅང་ཆེ་མུའུ་རྫོང་དུ་ལག་བསྟར་བྱེད་དོན། ཉེ་བའི་ལོ་འགའི་རིང་ས་གནས་དེར་ཡུ་གུར་མི་རིགས་དང་རྒྱ་རིགས་བར་འཁྲུག་འཛིན་ཤོར་ཏེ། མི་རིགས་ཕན་ཚུན་གྱི་འཁོན་འཛིང་ཇེ་ཆེར་འགྲོ་བཞིན་པ་གནོན་ཐབས་སུ་ལག་བསྟར་བྱེད་ཀྱི་ཡོད་སྐོར་བརྗོད་འདུག ཤིན་ཅང་ཡུ་གུར་རིགས་རང་སྐྱོང་ལྗོངས་ཆེ་མུའུ་རྫོང་ཆོན་ཁི་ལེ་ཤང་གི་དམར་ཤོག་མགོ་གཙོ་ཡྰ་སེན་ནྰ་སི་ཨར་གྱིས། རྒྱ་ནག་ནང་མི་རིགས་ཕན་ཚུན་བར་གཉེན་སྒྲིག་བྱེད་ཀྱིན་པ་དེ་ནི། མི་རིགས་ཕན་ཚུན་མཐུན་སྒྲིལ་མཉམ་གནས་ཐོག་འཚོ་བ་སོགས་ཕྱོགས་ཡོངས་ཡར་རྒྱས་གོང་འཕེལ་དུ་འགྲོ་ཆེད་ཡིན་ཚུལ་བརྗོད་ཡོད་པ་མ་ཟད། མི་རིགས་ཕན་ཚུན་གཉེན་སྒྲིག་བྱེད་པའི་སྲིད་ཇུས་དེ་ནུས་པ་ངོ་མ་ཞིག་དང་། ཤིན་ཅང་ཡུ་གུར་རིགས་རང་སྐྱོང་ལྗོངས་ལ་ཀྲུང་གོའི་རྨི་ལམ་མངོན་འགྱུར་ཡོང་བའི་ཐབས་ལམ་ཞིག་ཡིན་ཚུལ་བརྗོད་འདུག ཕྱི་ལོ་ ༢༠༠༠ ལོའི་རྒྱ་ནག་གཞུང་གིས་མི་འབོར་ཞིབ་བཤེར་ལྟར་ན། ཤིན་ཅང་ཡུ་གུར་རིགས་རང་སྐྱོང་ལྗོངས་ཀྱི་སྤྱིའི་མི་འབོར་ལས་བརྒྱ་ཆའི་ ༤༣ ཡུ་གུར་བ་དང་། བརྒྱ་ཆའི་ ༤༠ རྒྱ་རིགས་ཡིན་ཚུལ་བསྒྲགས་ཡོད་འདུག་ལ། ད་ལྟའི་ཆར་ཆེ་མུའུ་རྫོང་དུ་ཁྱོན་མི་ཁྲི་ ༡༠ གནས་སྡོད་བྱས་ཡོད་པ་ལས། ཡུ་གུར་བ་བརྒྱ་ཆའི་ ༧༣ དང་། རྒྱ་རིགས་བརྒྱ་ཆའི་ ༢༧ ཡིན་འདུག ཨོ་སི་ཊོ་ལི་ཡའི་ལྰ་ཌོར་བྷེ་ (La Trobe University) སློབ་ཆེན་གྱི་རྒྱ་ནག་གི་མི་རིགས་སྲིད་ཇུས་ཞིབ་འཇུག་པ་ཇེ་མུའུ་སི་ལེ་བྷོར་ཊི་ (James Leibold) ཡིས། ཉེ་ཆར་གསར་དུ་ཐོན་པའི་ཞིབ་འཇུག་གྲངས་འབོར་ལྟར་ན། ད་ལྟའི་ཆར་ཡུ་གུར་བའི་མི་ལས་བརྒྱ་ཆའི་གཅིག་དང་། བོད་པའི་བརྒྱ་ཆའི་བརྒྱད་དེ་མི་རིགས་གཞན་དང་མཉམ་དུ་གཉེན་སྒྲིག་བྱས་ཟིན་ཡོད་པ་དང་། རྒྱ་ནག་ནང་གི་མི་རིགས་ཁག་གི་ཆ་སྙོམས་ཀྱི་ཚད་བརྒྱ་ཆའི་གསུམ་དང་། རྒྱ་མི་རིགས་མི་རིགས་གཞན་དང་གཉེན་སྒྲིག་གནང་ཚད་བརྒྱ་ཆའི་གཅིག་ཚེག་ལྔ་ཡིན་སྐོར་བརྗོད་འདུག འདི་ལོའི་ཕྱི་ཟླ་ ༦ པའི་ནང་བོད་ལྗོངས་ཉིན་རེའི་ཚགས་པར་ཐོག བོད་རང་སྐྱོང་ལྗོངས་ཀྱི་དམར་ཤོག་དྲུང་ཆེ་ཁྲིན་ཆོན་ཀོ་བོད་རང་སྐྱོང་ལྗོངས་སུ་རྒྱ་རིགས་དང་གཉེན་སྒྲིག་བྱས་པའི་ཁྱིམ་ཚང་ ༡༩ དང་ཐུག་ཏེ། རྒྱ་རིགས་དང་གཉེན་སྒྲིག་བྱས་རྗེས་བོད་མི་ཚོས་བདེ་སྐྱིད་ལྡན་པའི་འཚོ་བའི་རོལ་བཞིན་པའི་གནས་སྟངས་ངོ་སྤྲོད་གནང་བ་ཁག་ལ་ཉན་ཡོད་ཚུལ་བསྒྲགས་ཡོད་པ་མ་ཟད། ཁོས་ཁོ་རང་དངོས་སུ་ཐོན་ཏེ་བོད་རིགས་མི་མང་ལ་རྒྱ་རིགས་དང་གཉེན་སྒྲིག་བྱེད་པའི་གུ་ཡངས་སྲིད་ཇུས་སྐུལ་སློང་བྱས་རྗེས། ༢༠༠༨ ལོར་བོད་རང་སྐྱོང་ལྗོངས་སུ་བོད་རྒྱ་གཉེན་སྒྲིག་བྱེད་མཁན་མི་ ༦༦༦ བྱུང་བ་དང་། ༢༠༡༣ ལོར་བོད་རྒྱ་གཉེན་སྒྲིག་བྱེད་མཁན་གྱི་མི་ ༤༧༩༥ བྱུང་སྟེ་འཕར་སྣོན་བྱུང་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། དོན་གོ
