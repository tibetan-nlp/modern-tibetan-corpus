ལྷོ་བྱང་ཀོ་རི་ཡའི་ནང་མི་ཐུག་འཕྲད།

ལྷོ་བྱང་ཀོ་རི་ཡའི་དམག་འཁྲུག་ཁྲོད་ཁ་བྲལ་དུ་ཕྱིན་པའི་ནང་མི་བརྒྱ་སྐོར་ཞིག་དེ་རིང་ཐུག་འཕྲད་ཀྱི་གོ་སྐབས་སྤྲད་དེ་རྒྱལ་ཁབ་གཉིས་བར་གྱི་འབྲེལ་ལམ་ཇེ་ལགས་སུ་གཏོང་ཐབས་བྱས་ཡོད་འདུག དེ་རིང་ཞོགས་པར་ལྷོ་ཀོ་རི་ཡའི་མི་ཚང་དྲུག་བཅུ་ལས་གྲུབ་པའི་མི་གྲངས་ ༡༠༠ ཙམ་སྤྱི་སྤྱོད་རླངས་འཁོར་ལ་བསྡད་དེ། བྱང་ཀོ་རི་ཡའི་ཀུམ་སྒང་ཡུལ་སྐོར་སྤྲོ་འཆམ་ས་གནས་སུ་བསྐྱོད་དེ། སྔོན་མ་ལྷོ་བྱང་ཀོ་རི་ཡའི་དམག་འཁྲུག་སྐབས་ཁ་བྲལ་དུ་ཕྱིན་པའི་ནང་མི་སྤུན་མཆེད་ཚོར་ཐུག་འཕྲད་ཐེངས་གཉིས་པ་དེ་གནང་ཡོད་འདུག ལྷོ་ཀོ་རི་ཡ་གཞུང་གི་ནང་མི་ཐུག་འཕྲད་ཀྱི་སྒུག་ཐོའི་ནང་ཁྱོན་མི་ ༧༢༠༠༠ ལྷག་ཡོད་པ་དང་། ཁོང་ཚོ་མང་ཆེ་བ་ལོ་ན་ ༧༠ ནས་༨༠ བར་གྱི་རྒན་འཁོགས་ཡིན་འདུག སྤུན་མཆེད་ཐུག་འཕྲད་འགྲོ་བཞིན་པའི་ལྷོ་ཀོ་རི་ཡའི་རྒན་པོ་ཞིག་གིས་བྷེ་བྷེ་སི་གསར་འགོད་པར། ངས་མི་ཚེ་གང་བོར་རང་གི་སྤུན་མཆེད་དྲན་བསྡད་པ་ཡིན་སྟབས། ད་ཐེངས་ཁོ་ལ་ཐུག་པའི་དགའ་ཚོར་དེ་བཤད་མི་ཤེས་པ་ཞིག་རེད། ང་ཁོ་ལ་ཐུག་རྗེས་སང་ཉིན་ཤི་ཡང་འགྱོད་པ་མེད་ཅེས་བརྗོད་འདུག ཨ་རི་དང་ལྷོ་ཀོ་རི་ཡ་རྒྱལ་ཁབ་གཉིས་ཀྱི་དམག་སྦྱོང་བྱེད་མགོ་འཛུགས་ལ་ཉེ་སྐབས། བྱང་ཀོ་རི་ཡས་ལྷོ་ཀོ་རི་ཡ་གཞུང་ལ་ནང་མི་ཁ་བྲལ་དུ་ཕྱིན་པ་རྣམས་ཕན་ཚུན་ཐུག་འཕྲད་བྱེད་དགོས་པའི་གྲོས་མགོ་བཏོན་པ་ལྟར། ཟླ་འདིའི་ཚེས་ ༡༢ ནས་ ༡༤ ཉིན་བར་ཀོ་རི་ཡ་ལྷོ་བྱང་གཉིས་ཀྱི་བློན་ཆེན་རིམ་པའི་ཚོགས་འདུ་ཚོགས་ཏེ། བྱང་ཀོ་རི་ཡིས་ནང་མི་ལྷན་འཛོམས་བྱེད་དགོས་ཚེ་ལྷོ་ཀོ་རི་ཡས་ངེས་པར་དུ་ཨ་རི་མཉམ་དུ་དམག་སྦྱོང་བྱ་རྒྱུ་ཕྱི་འགྱངས་བྱེད་དགོས་པའི་ཆ་རྐྱེན་བཏོན་པར་ལྟར། གྲོས་གཞི་ལ་མོས་མཐུན་བྱུང་སྟེ་དུས་ཚོད་གཏན་འཁེལ་བྱས་པ་ཞིག་ཡིན་འདུག གནས་ཚུལ་དེར་འཛམ་གླིང་རྒྱལ་ཁབ་ཁག་གི་མགོ་ཁྲིད་ཚོས་ཀོ་རི་ཡ་ལྷོ་བྱང་བར་ནང་མི་ཐུག་འཕྲད་ལ་དགའ་བསུ་དང་། འབྱུང་འགྱུར་ལྷོ་བྱང་ཀོ་རི་ཡའི་ཁ་བྲལ་དུ་སོང་བའི་ནང་མི་གང་མང་ཐུག་འཕྲད་བྱེད་འཇུག་པར་སྐུལ་འདེད་བྱས་ཡོད་པ་དང་། གནད་དོན་དབྱེ་ཞིབ་པ་ཚོས། བྱང་ཀོ་རི་ཡས་གོ་སྐབས་བཟུང་སྟེ་ལྷོ་ཀོ་རི་ཡའི་ཕྱོགས་ནས་དཔལ་འབྱོར་རོགས་སྐྱོར་རག་ཐབས་བྱེད་ཀྱི་ཡོད་ཀྱང་། བྱང་ཀོ་རི་ཡས་རྡུལ་ཕྲན་མཚོན་ཆ་བཟོ་སྐྲུན་དང་ཚོད་ལྟ་སོགས་མཚམས་འཇོག་མ་བྱས་བར། ལྷོ་ཀོ་རི་ཡའི་ཕྱོགས་ནས་དཔལ་འབྱོར་རོགས་སྐྱོར་རག་རྒྱུ་ཁག་པོ་ཡིན་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། ལྷ་སྐྱབས་རྒྱལ།
