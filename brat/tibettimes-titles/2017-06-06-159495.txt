
ཇམ་སི་ཁའོ་མེ་ཡིས་སྲིད་འཛིན་ཁྲི་རོམ་ཕུ་གང་ལ་བསྐྱེལ་སྲིད།

གཅན་སྣང་བ་ཚེ་རིང་བཀྲ་ཤིས། མི་འགྱངས་བར་(གཟའ་ཕུར་བུའི་ཉིན)ཨ་རིའི་དབུས་ཞིབ་དཔྱོད་ལས་ཁུངས་(FBI)ཀྱི་འགན་འཛིན་ཟུར་བ་ཇམ་སི་ཁའོ་མེ་(James Comey)ལགས་ཨ་རིའི་དབུས་ཁྲིམས་བཟོ་གསང་བའི་ལས་དོན་ཚོགས་ཆུང་(Senate intelligence committee)གི་མདུན་དུ་ཐོན་ཏེ་སྐད་ཆ་བཤད་རྒྱུ་རེད། ཁོང་གིས་སྐད་ཆ་ག་རེ་བཤད་མིན་དེས་ཨ་རིའི་སྲིད་འཛིན་ཊི་རུམ་ཕི་(President Trump)སྔར་བཞིན་སྲིད་འཛིན་གྱི་ལས་གནས་ཐོག་སྡོད་ཐུབ་མིན་ཐག་གཅོད་བྱེད་ངེས་རེད། གལ་ཏེ། ཇམ་སི་ཁའོ་མེ་ལགས་ཀྱིས་སྲིད་འཛིན་ཊི་རུམ་ཕིས་ཁོ་རང་དབུས་ཞིབ་དཔྱོད་ལས་ཁུངས་ཀྱི་འགན་འཛིན་ལས་གནས་དབྱུང་བཏང་བ་བདེན་དཔང་ར་སྤྲོད་བྱས་ཚེ། དེ་ནི་ཨ་རིའི་ཆབ་སྲིད་ལོ་རྒྱུས་ཁྲོད་ཀྱི་ཆེས་རྙོག་འཛིང་དང་འཁྲུག་ཆ་དོད་ཤོས་ཀྱི་དུས་སྐབས་ཤིག་ཆགས་ངེས་རེད། ཨ་རིའི་དབུས་ཁྲིམས་བཟོ་གསང་བའི་ལས་དོན་ཚོགས་ཆུང་གིས་ཇམ་སི་ཁའོ་མེ་ལགས་ལ། ཁོས་དེ་སྔ་སྲིད་འཛིན་ཊི་རུམ་ཕི་ཡི་སློབ་སྟོན་པ་དང་ར་ཤི་ཡའི་བར་གྱི་གསང་བའི་འབྲེལ་བར་ཞིབ་བཤེར་བྱེད་ཀྱིན་ཡོད་པ་དེར་སྲིད་འཛིན་ཊི་རུམ་ཕིས་མཚམས་འཇོག་བྱེད་དགོས་པའི་རེ་སྐུལ་དང་གནོན་ཤུགས་སྤྲད་ཡོད་མེད་འདྲི་རྩད་བྱ་རྒྱུ་རེད་ལ། ཇམ་སི་ཁའོ་མེ་ལགས་ལས་གནས་ནས་ཕུད་པ་དེ་ཁོས་ཊི་རུམ་ཕིའི་རེ་སྐུལ་དང་ལེན་མ་བྱས་པའི་རྒྱུ་མཚན་ཡིན་མིན་ཡང་གསལ་ཁ་བཟོ་རྒྱུ་རེད། སྲིད་འཛིན་ཊི་རུམ་ཕིས། དེང་སྐབས་ཁོས་ཇམ་སི་ཁའོ་མེ་ཡི་ཞིབ་བཤེར་ལས་དོན་དེ་མཚམས་འཇོག་དགོས་པའི་རེ་སྐུལ་ཞུས་མེད་ཅེས་ཉེས་འཛུགས་དེར་ངོས་ལེན་བྱེད་ཀྱི་མེད། འོན་ཀྱང་། ཇམ་སི་ཁའོ་མེ་ལགས་ཀྱིས་དེ་ལས་ལྡོག་པ་ཞིག་ཤོད་བཞིན་ཡོད་པ་དང་། དེ་དང་འབྲེལ་བའི་གནས་ཚུལ་ཁག་ལ་བདེན་དཔང་བྱེད་ཐུབ་ཚེ། སྲིད་འཛིན་ཊི་རུམ་ཕི་ལ་ཨ་རིའི་ཁྲིམས་ཐོག་ནས་དྲང་བདེན་བཀག་འགོག་བྱས་པའི་ནག་ཉེས་( obstruction of justice)ཕོག་སྲིད་ལ། ནག་ཉེས་དེས་ཊི་རུམ་ཕི་སྲིད་འཛིན་རྐུབ་སྟེགས་ལས་རྒྱང་འབུད་(impeachment)གཏོང་སྲིད། ཐེངས་འདིར་ཇམ་སི་ཁའོ་མེ་ལགས་ཀྱིས་སྐད་ཆ་ཅི་ཞིག་བཤད་མིན་དེས། ཊི་རུམ་ཕིའི་སྲིད་འཛིན་གོ་གནས་ཐག་གཅོད་བྱེད་ངེས་ཡིན་སྟབས། གནད་དོན་འདི་ནི་ཧ་ཅང་གནད་འགག་ཆེན་པོ་ཞིག་ཆགས་ཡོད་ལ། ད་དུང་དེས་མ་ཚད་ ༢༠༡༧ ལོའི་ཟླ་ ༥ པའི་ཚེས་ ༡༧ ཉིན་ཨ་རིའི་དྲང་བདེན་ལྷན་ཚོགས་ (The Department of Justice)ཀྱིས་ཇམ་སི་ཁའོ་མེ་སྔོན་གྱི་ཨ་རིའི་དབུས་ཞིབ་དཔྱོད་ལས་ཁུངས་ཀྱི་འགན་འཛིན་རོ་སྦིར་ཌི་མུའུ་ལེར་(Robert Mueller)མཆོག་ ༢༠༡༦ ལོའི་ཨ་རིའི་སྲིད་འཛིན་ཁྱབ་བསྒྲགས་ལས་འགུལ་དང་ར་ཤི་ཡའི་བར་གྱི་རྙོག་གླེང་ཐོག་ཞིབ་བཤེར་བྱེད་པའི་དམིགས་བསལ་གྱི་ཁྲིམས་དོན་སློབ་སྟོན་པར་(special counsel)བསྐོ་བཞག་གནང་ཡོད། མི་ཁ་ཤས་ཀྱིས་གཅིག་བྱས་ན་སྲིད་འཛིན་ཊི་རུམ་ཕིས་སྲིད་འཛིན་གྱི་དམིགས་བསལ་དབང་ཚད་བེད་སྤྱོད་བཏང་སྟེ་ཇམ་སི་ཁའོ་མེ་ལགས་ཨ་རིའི་དབུས་ཁྲིམས་བཟོ་གསང་བའི་ལས་དོན་ཚོགས་ཆུང་གི་མདུན་སར་ཐོན་པར་བཀག་འགོག་བྱེད་སྲིད་སྐོར་རྗོད་ཀྱི་ཡོད་ཀྱང་། མང་པོ་ཞིག་གིས་ད་ལྟ་ཊི་རུམ་ཕི་ལ་རྩེད་མོ་རྩེད་བྱེད་ཀྱི་ཤོག་སྦག་(card)གང་ཡང་ལྷག་མེད་པའི་ཚོད་དཔག་བྱེད་ཀྱིན་ཡོད། གང་ཡིན་ཞེ་ན། སྲིད་འཛིན་ལ་ཞིབ་བཤེར་བྱེད་བཞིན་པའི་ཐོག་ལ་སྲིད་འཛིན་གྱི་དམིགས་བསལ་དབང་ཚད་བེད་སྤྱོད་གཏོང་ས་མེད་པ་དེ་རེད། གཉིས་ནས་ཇམ་སི་ཁའོ་མེ་ལགས་ད་ཆ་མི་སེར་དཀྱུས་མ་ཞིག་ཡིན་སྟབས། མི་སེར་དཀྱུས་མ་ཞིག་གིས་གནད་དོན་གང་ཞིག་ར་སྤྲོད་བྱེད་འདོད་པ་དེ་བཀག་འགོག་བྱེད་མི་ཆོག དེར་མ་ཟད། རོ་སྦིར་ཌི་མུའུ་ལེར་(Robert Mueller)་ལ་ཁྲིམས་ཐོག་ནས་དབང་ཚད་ཆེན་པོ་ཡོད་སྟབས། སྲིད་འཛིན་ཊི་རུམ་ཕིས་ཁོ་རང་གང་འདོད་དུ་གནས་དབྱུང་གཏོང་མི་ཐུབ་ལ། ཁོས་ར་ཤི་ཡས་ ༢༠༡༦ ཨ་རིའི་སྲིད་འཛིན་གྱི་འོས་བསྡུའི་གྲུབ་འབྲས་ལ་ཤུགས་རྐྱེན་སྤྲད་ཡོད་མེད་ཞིབ་ཏུ་འཚོལ་ཞིབ་བྱེད་རྒྱུ་ཡིན་པ་རེད། གང་ལྟར། ཊི་རུམ་ཕི་དང་ར་ཞི་ཡའི་བར་འབྲེལ་བ་ཡོད་པའི་དཔང་རྟགས་མང་པོ་ཕྱིར་མངོན་ཡོང་གི་ཡོད་ལ། ཊི་རུམ་ཕི་ཡི་ཕྱག་རོགས་དག་གིས་ཁྲིམས་འགལ་གྱི་འབྲེལ་བ་དེ་སྦ་གསང་བྱེད་ཀྱི་ཡོད་པའི་བདེན་དཔང་ཡང་མང་པོ་ཕྱིར་འཐོན་ཡོང་གི་ཡོད་སྟབས། ཨ་རིའི་ཕོ་བྲང་དཀར་པོ་དེ་ད་ཆར་ཆབ་སྲིད་དང་ཁྲིམས་ལུགས་གཉིས་ཐད་ནས་ཉེན་ཁར་ཐུག་ཡོད། རོ་སྦིར་ཌི་མུའུ་ལེར་(Robert Mueller)མཆོག་གི་ཞིབ་བཤེར་དེ་ཟླ་བ་མང་པོ་དང་ཐ་ན་ལོ་གཅིག་ཀྱང་འགོར་སྲིད། འོན་ཀྱང་། ཊི་རུམ་ཕི་ལ་སྐབས་ཐོག་གི་ཉེན་ཁ་དེ་ཇམ་སི་ཁའོ་མེ་ཡི་ཕྱོགས་ནས་བསླེབས་ངེས་སོ།། གནས་ཚུལ་འདིའི་སྐོར་ལ་རྒྱས་པ་ཀློག་འདོད་ན་དྲ་ཚིགས་འདི་དག་ལ་གཟིགས་རོགས།   [ 1 ] [ 2 ]  [ 3 ]  [ 4 ]  [ 5 ]
