སྦོབས་སེམས་དང་ཡིད་ཆེས་དགོས་པ་རེད།

ཁ་སང་༧གོང་ས་མཆོག་གིས་ཉི་ཧོང་གི་གནོད་ཐེབས་མང་ཚོཊ་ལ་བཀའ་སློབ་གནང་སྟེ། དཀའ་ངལ་གང་འདྲ་ཞིག་འཕྲད་ཀྱང་ངས་ཐུབ་ཀྱི་རེད་སྙམ་པའི་སྤོབས་སེམས་དང་ཡིད་ཆེས་དགོས་རྒྱུ་ཧ་ཅང་གལ་ཆེན་ཡིན་སྐོར་གསུངས་འདུག ཟླ་འདིའི་ཚེས་ ༦ ཉིན་༧གོང་ས་མཆོག་ཉི་ཧོང་གི་ཚོཊ་པ་ཁག་གསུམ་གྱིས་ཇི་ལྟར་གསོལ་འདེབས་ཞུས་དོན་ལྟར་ཆིབས་བསྒྱུར་གནང་ཡོད་པ་དང་།  ཁ་སང་ཚེས་ ༧ ཉིན་ཉི་ཧོང་ཤིན་ཏའོ་ཆོས་ལུཊ་ཀྱིས་གསོལ་འདེབས་སྨོན་ལམ་སྟེང་ཆིབས་བསྒྱུར་གནང་སྟེ། ༢༠༡༡ ལོར་ཉི་ཧོང་དུ་ས་ཡོམ་དང་ཚུ་ནཱ་མི་ (Tsunami) ཞེས་པའི་རྒྱ་མཚོའི་རྦ་རླབས་ཀྱིས་འབྱུང་བའི་གོད་ཆག་དང་། རྡུལ་ཕྲན་གནོད་སྐྱོན་གྱི་རྐྱེན་པས་སེམས་ལ་དཀའ་ངལ་འཕྲད་བཞིན་པ་དག་གི་ཆེད་མགོན་པོ་ཀླུ་སྒྲུབ་ཀྱིས་མཛད་པའི་སྟོད་ཕྱག་གསོལ་འདེབས་སྨོན་ལམ་དང་ལམ་སྟོན་བཀའ་སློབ་བཅས་གནང་ཡོད་འདུག ཁ་སང་དེ་སྔོན་རང་བྱུང་གོད་ཆག་ཕོག་མཁན་གྱི་མི་མང་ལ་ཐུག་འཕྲད་དང་ལམ་སྟོན་བཀའ་སློབ་གནང་སྐབས་༧གོང་ས་མཆོག་གིས། སྤྱིར་མིའི་མི་ཚེའི་ནང་ངས་ཐུབ་ཀྱི་རེད་སྙམ་པའི་སྤོབས་སེམས་དང་ཡིད་ཆེས་དགོས་རྒྱུ་དེ་ཧ་ཅང་གལ་ཆེན་པོ་རེད། ལྷག་པར་དུ་དཀའ་ངལ་འཕྲད་སྐབས་དཀའ་ངལ་དེ་སེལ་དགོས་བསམ་པའི་སྤོབས་སེམས་དང་སྙིང་སྟོབས་དེ་ཡང་ཧ་ཅང་གལ་ཆེན་པོ་རེད། ས་ཡོམ་དང་ཆུ་ལོག རྡུལ་ཕྲན་གྱིས་འོད་རླབས་སོགས་ཀྱི་དཀའ་ངལ་འཕྲད་པ་ཚོར་ངོས་ནས་དམིགས་བསལ་གདུང་སེམས་མཉམ་སྐྱེད་ཡོད། དེ་སྔོན་ས་ཡོམ་སོགས་ཀྱི་དཀའ་ངལ་འཕྲད་སར་ངོས་རང་ཐེངས་གཅིག་བསླེབ་ཏེ་ཐུག་པ་རེད། སྐབས་དེར་ཁོང་ཚོས་མིག་ཆུ་གཏོང་གི་འདུག ངོས་ཀྱང་མིག་ཆུ་ཤོར་གྲབས་བྱས་སོང་། ཧ་ཅང་སེམས་སྐྱོ་བོ་བྱུང་། སྐབས་དེར་ངོས་ནས་སེམས་ཁྲལ་བྱས་ཏེ་སེམས་ཤུགས་ཆགས་བསྡད་ན་ཕན་ཐོགས་མེད། སེམས་མ་སྐྱིད་པ་བྱས་ཏེ་བསྡད་ན་དཀའ་ངལ་གོད་ཆག་སྟེང་ལ་ཁ་སྣོན་ལྟ་བུ་ཆགས་ཀྱི་ཡོད། དཀའ་ངལ་དང་གོད་ཆག་ཆེན་པོ་བྱུང་དེ་སེལ་རྒྱུར་འབད་བརྩོན་བྱེད་རྒྱུ་གལ་ཆེན་པོ་རེད་ཅེས་བརྗོད་རྒྱུ་བྱུང་། འཛམ་གླིང་དམག་ཆེན་གཉིས་པའི་སྐབས་འཇར་མན་དང་ཉི་ཧོང་གཉིས་ལ་བརྗོད་དཀའ་བའི་གཏོར་བཤིག་ཕོག་ཡོད་ཀྱང་། རྒྱལ་ཁབ་གཉིས་ཀྱིས་མི་མང་གིས་གཏོར་སྐྱོན་གྱི་ཐལ་རྡུལ་ཁྲོད་ནས་རྒྱལ་ཁབ་བསྐྱར་སྐྲུན་བྱས་ཡོད་པའི་སྐོར་ངོས་རང་འཛམ་གླིང་ས་ཕྱོགས་གང་དུ་ཕྱིན་ཡང་བརྗོད་ཀྱི་ཡོད། རྒྱལ་ཁབ་འདི་གཉིས་རིམ་པས་མང་གཙོ་དང་དཔལ་འབྱོར། ཆབ་སྲིད་སོགས་སྟོབས་ཤུགས་ཆེན་པོ་ཆགས་ཐུབ་པ་བྱུང་ཡོད་པར་མ་ཟད། རང་ཉིད་ལ་བློ་སྟོབས་དང་གདེང་ཚོད་ཡོད་སྟབས། གཏོར་བཤིག་ག་ཚོད་ཕོག་བསྐྱར་སྐྲུན་ལེགས་པོ་ཐུབ་ཡོད་པ་རེད། ཁྱེད་རང་ཚོས་དེ་དག་དཔེ་མཚོན་དུ་བཟུང་ནས་སྙིང་སྟོབས་སྐྱེད་དགོས་ཞེས་སོགས་བཀའ་སློབ་གནང་འདུག སང་ཉིན་ཚེས་༩ ཉིན་༧གོང་ས་མཆོག་ཉི་ཧོང་གི་སེ་ཧྥི་གྷ་ཀེན་ (Seifu Gaken) སློབ་གྲྭའི་སློབ་ཕྲུག་ཚོར་བཀའ་སློབ་དང་། ཚེས་ ༡༡ ནས་ ༡༢ བར་གྱི་ཉིན་གཉིས་རིང་ཉི་ཧོང་གི་ཀྱོ་ཊོ་ཆེས་མཐོའི་གཙུག་ལག་སློབ་གཉེར་ཁང་གི་ཞིབ་འཇུག་སྡེ་ཚན་དང་སེམས་དང་སྲོག་གི་བསྟི་གནས་ཁང་ཐུན་མོང་གི་གོ་སྒྲིག་ཞུས་པའི་༼སེམས་ཀྱི་བཀོད་པ༽ ཞེས་པའི་བརྗོད་གཞིའི་ཐོག་ཚན་རིག་པ་ཁག་ཅིག་ལྷན་བགྲོ་གླེང་གནང་རྒྱུ་ཡིན་འདུག གསར་འགོད་པ། དོན་གོ
