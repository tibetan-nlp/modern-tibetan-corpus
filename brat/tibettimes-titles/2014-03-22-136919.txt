ཁྲིམས་ཐག་བཅད་ནས་ལོ་ལྔ་ལྷག་ཕྱིན།

ཀན་ལྷོ་ཁུལ་བསང་ཆུ་རྫོང་ཁོངས་ནས་ཡིན་པའི་བོད་མི་ ༣ ལ་ལོ་ ༥ ནས་ ༡༢ བར་གྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་དེ་ལོ་ ༥ ལྷག་ཕྱིན་འདུག ༢༠༠༩ ལོའི་ཧོར་ཟླ་ ༤ ཚེས་༢༤ ཉིན་ཀན་ལྷོ་ཁུལ་གཙོད་གྲོང་ཁྱེར་དུ། རྒྱ་གཞུང་གིས་བསང་ཆུ་རྫོང་ཁོངས་ཀྱི་བོད་མི་རིག་འཛིན་ཚེ་རིང་ཟེར་བ་ཞིག་ལ་ལོ་ ༡༢ ཀྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་པ་དང་། དེ་བཞིན། ཁོང་དང་མཉམ་དུ་འཛིན་བཟུང་བྱས་པའི་སྙིང་ལྕགས་རྒྱལ་དང་བཀྲ་ཤིས་ཚེ་རིང་ཟེར་བ་གཉིས་ལའང་ལོ་ ༥ དང་ ༧ བཅས་ཀྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་དེ་ལོ་ ༥ ལྷག་ཕྱིན་ཡོད་ཀྱང་། རྒྱ་གཞུང་གིས་བོད་ནང་གི་བརྒྱུད་ལམ་སོགས་ལ་དམ་དྲག་ཆེན་པོ་བྱེད་བཞིན་ཡོད་སྟབས། ད་ལྟའི་བར་གནས་ཚུལ་དེ་ཕྱིར་བསླེབས་མེད་པ་རེད། བོད་ཀྱི་འགྲོ་བ་མིའི་ཐོབ་ཐང་དང་མང་གཙོ་འཕེལ་རྒྱས་ལྟེ་གནས་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། རིག་འཛིན་ཚེ་རིང་ལགས་ནི་བོད་མདོ་སྨད་ཀན་ལྷོ་ཁུལ་བསང་ཆུ་རྫོང་ཐང་དཀར་ཡུལ་ཚོ་རོང་ངར་རུ་ཆེན་རོང་ཚ་སྡེ་བ་ནས་ཡིན་པ་དང་། ༢༠༠༨ ལོའི་ཕྱི་ཟླ་ ༤ བའི་ནང་རྒྱ་གཞུང་ཉེན་རྟོག་པས་འཇུ་བཟུང་གིས་བསང་ཆུ་རྫོང་དུ་བཙན་འཁྲིད་བྱས་ཏེ། གར་ཡོད་ཅི་བྱུང་མེད་པར་ལོ་གཅིག་ཙམ་ཕྱིན་རྗེས། ༢༠༠༩ ལོའི་ཧོར་ཟླ་ ༤ ཚེས་ ༢༤ ཉིན་ཀན་ལྷོ་ཁུལ་གཙོད་གྲོང་ཁྱེར་ནས་ཁོང་ལ་ལོ་ ༡༢ ཀྱི་ཁྲིམས་ཐག་བཅད་དེ། ཀན་སུའུ་ཞིང་ཆེན་ཐན་ཧྲའེ་གྲོང་ཁྱེར་གྱི་བཙོན་ཁང་ཞིག་ཏུ་བཅུག་ཡོད་པ་དང་། ཁོང་དང་མཉམ་དུ་འཛིན་བཟུང་བྱས་པའི་རང་ལོ་ ༤༤ ཡིན་པའི་སྙིང་ལྕགས་རྒྱལ་དང་། རང་ལོ་ ༣༤ ཡིན་པའི་བཀྲ་ཤིས་ཚེ་རིང་གཉིས་ལའང་ ༥ དང་ ༧ གྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་ཀྱང་། ད་ལྟའི་ཆར་བཙོན་ཁང་གང་དུ་ཡོད་མེད་སོགས་ཤེས་རྟོགས་ཐུབ་མི་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
