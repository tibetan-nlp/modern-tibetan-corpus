བློ་བཟང་རྣམ་རྒྱལ་ཀློད་གྲོལ།

རྒྱ་ནག་གཞུང་གིས་ན་གཞོན་རྩོམ་པ་པོ་བློ་བཟང་རྣམ་རྒྱལ་ལམ་གསང་མིག་ལགས་བཙོན་གྲོལ་བཏང་ཡོད་འདུག དེ་རིང་ཕྱི་ཚེས་ ༢༦ ཉིན་གྱི་སྔ་དྲོ་རྒྱ་ནག་གཞུང་གིས། ཕྱི་ལོ་ ༢༠༡༢ ལོའི་ཟླ་ལྔ་བར་འཛིན་བཟུང་བཙོན་འཇུག་བྱས་ཡོད་པའི་ཀན་ལྷོ་རྨ་ཆུ་རྫོང་གི་གཞོན་སྐྱེས་རྩོམ་པ་པོ་བློ་བཟང་རྣམ་རྒྱལ་ལམ་སྨྱུག་མིང་ལ་གསང་མིག་འབོད་པ་ཀློད་གྲོལ་བཏང་ཡོད་འདུག བློ་བཟང་རྣམ་རྒྱལ་ལགས་ནི། ཀན་ལྷོ་ཁུལ་རྨ་ཆུ་རྫོང་དབལ་པན་སྡེ་བའི་སྐོང་གནས་རུ་ཆེན་ཡག་ཤུ་རོ་ཞེས་པ་ནས་ཡིན་པ་དང༌། ཕ་མིང་ལ་ཚེ་བརྟན་དང་མ་མིང་ལ་པདྨ་ཞུ་བཞིན་ཡོད་འདུག་ལ། ཁོང་ད་ལྟ་རང་ལོ་༢༨  ཡས་མས་ཡིན་འདུག ༢༠༠༨ལོར་རྨ་ཆུ་རྫོང་དུ་བོད་མི་སྐྱ་སེར་གཉིས་ཀྱི་ཞི་རྒོལ་སྐབས། རྒྱ་གཞུང་གིས་ཁོང་གཙོས་པའི་མི་ ༡༨ འཛིན་བཟུང་གིས་གཅར་རྡུང་འདྲི་གཅོད་བྱས་རྗེས་ཀློད་གྲོལ་བཏང་ཡོད་ཀྱང་། ༢༠༡༢ ལོའི་ཟླ་ ༥ བར་རྒྱ་ནག་ཁྲིན་ཏུའུ་གྲོང་ཁྱེར་གྱི་བྷུ་དྷ་རིག་གནས་ལས་ཁང་དུ་ཞོར་ལས་བྱེད་སྐབས་གསང་སྟབས་ཀྱིས་འཛིན་བཟུང་བྱས་ཏེ། ཟླ་ ༩ ལྷག་གི་རིང་གར་ཡོད་ཅི་བྱུང་གི་གནས་ཚུལ་ཅི་ཡང་མ་ཤེས་པར་ལྷག་འདུག ༢༠༡༣ ལོའི་ཕྱི་ཟླ་ ༣ པར་ནང་མི་ཉེ་འབྲེལ་ཚོས་ཁོང་༸གོང་ས་མཆོག་གི་བཀའ་སློབ་དང་། ཆབ་སྲིད་ཀྱི་ཡིག་ཆ་དཔར་བསྐྲུན་བྱས་པའི་ཉེས་མིང་འོག་ཁྲིན་ཏུའུ་ཁྱེར་ལྟ་སྲུང་ཁང་དུ་བཀག་ཉར་བྱས་ཡོད་པ་ཤེས་རྟོགས་བྱུང་ཡོད་ཀྱང་། ཉེན་རྟོག་པ་དང་དམག་མིས་ནང་མིར་ཐུག་འཕྲད་ཕར་ཞོག ཐ་ན་ཟ་མ་དང་གྱོན་གོས་སོགས་ཀྱང་བསྐྱལ་མ་བཅུག་པར་དམ་དྲག་བཀག་སྡོམ་བྱས་ཡོད་འདུག བློ་བཟང་རྣམ་རྒྱལ་ལགས་སློབ་གྲྭ་ཆུང་འབྲིང་འགྲིམས་མྱོང་མཁན་ཞིག་ཡིན་པ་མ་ཟད། ༢༠༡༡ལོར་མགོ་ལོག་ར་རྒྱར་ཡོད་པའི་གངས་ལྗོངས་ཤེས་རིག་ནོར་བུའི་གླིང་ནས་སློབ་མཐར་ཕྱིན་ཡོད་འདུག་ལ། སྐབས་དེར་ ༼ འཚོལ༽ཞེས་པའི་སྙན་དེབ་ཅིག་སྤྱི་ཚོགས་ནང་འགྲེམས་སྤེལ་བྱས་ཡོད་པ་དང་། གླུ་ཚིག་དང་བརྩམས་ཆོས་གང་མང་ཞིག་ཀྱང་སྤྱི་ཚོགས་ཐོག་བཀྲམས་ཡོད་འདུག གསར་འགོད་པ། དོན་གོ
