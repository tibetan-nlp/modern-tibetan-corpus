ཤིན་ཅང་ནང་ས་ཡོམ་གནོད་སྐྱོན་ཐེབས།

ད་ནང་ཤིན་ཅང་ནུབ་ཁུལ་དུ་ས་ཡོམ་ཐེངས་ ༥༡ བརྒྱབ་སྟེ་ད་ལྟའི་བར་མི་གྲངས་ ༨ རྐྱེན་འདས་དང་མི་ ༢༠ ལྷག་ལ་རྨས་སྐྱོན་བཟོས་ཡོད་འདུག དེ་རིང་སྔ་དྲོའི་ཆུ་ཚོད་ ༦ པ་ཙམ་ལ་ཤིན་ཅང་ནུབ་ཁུལ་གྱི་(Taxkorgan Tajik Autonomous County)རྫོང་ཁོངས་སུ་ཤུགས་ཚད་ ༥.༥ ཅན་གྱི་ས་ཡོམ་བརྒྱབ་སྟེ། མི་གྲངས་ ༨ རྐྱེན་འདས་དང་མི་ ༢༠ ལྷག་ལ་རྨས་སྐྱོན་ཕོག སྡོད་ཁང་ ༢༠༠ མ་ཟིན་ཙམ་ལ་རྡིབ་སྐྱོན་བྱུང་ཡོད་འདུག ཞོགས་པ་ཆུ་ཚོད་༨།༣༠ ཐོག་ཤིན་ཅང་ས་ཡོམ་དྲ་བའི་ཐོག་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། དེ་རིང་ཞོགས་པའི་ཆུ་ཚོད་ ༨ པ་བར་དུ་ཁུལ་དེར་ཤུགས་ཚད་འདྲ་མིན་གྱི་ས་ཡོམ་ཐེངས་ ༥༡ བརྒྱབ་ཡོད་པ་དང་། རྫོང་ཁུལ་དུ་ཁང་པ་རྙིང་པ་མང་པོ་གཏོར་ཐུབ་མེད་པར་ལྷག་ཡོད་སྟབས་མི་ལ་ཤི་རྨས་བྱུང་བ་ཅུང་མང་བ་དང་། ད་ལྟའི་ཆར་མི་གྲངས་ ༨ རྐྱེན་འདས་དང་། མི་ ༢༠ ལྷག་ལ་རྨས་སྐྱོན་བཟོས། སྡོད་ཁང་ ༡༨༠ ལྷག་ལ་རྡིབ་སྐྱོན་བྱུང་སྟེ། ཐེངས་འདིའི་ས་ཡོམ་གནོད་སྐྱོན་ཁྲོད་མི་གྲངས་ ༧༨༩ ལ་གནོད་སྐྱོན་བཟོས། ད་ལན་རྫོང་དང་གྲོང་ཁྱེར་ཁག་ ༦ ཙམ་ནང་ས་ཡོམ་རིམ་པ་འདྲ་མིན་བྱུང་ཡོད་ཀྱང་།  ཐག་སི་ཁོར་གྷོན་ཐོའུ་ཇིག་(Toxkhürghon Tujik)རྫོང་ཁོངས་སུ་གནོད་སྐྱོན་ཚབས་ཆེ་ཤོས་ཐེབས་ཡོད་འདུག ཤིན་ཅང་ནི་གཏིང་ཚད་ཟབ་ཅིང་ཁྱབ་རྒྱ་ཆེ་བའི་ས་ཡོམ་མང་ཤོས་ཀྱི་ས་ཁུལ་ཞིག་ཆགས་ཡོད་པ་དང་། པའེ་ཁུ་རིག་མཛོད་ནང་བཀོད་པའི་ཤིན་ཅང་གི་ས་དཔྱད་ལོ་རྒྱུས་ལྟར་ན།  དུས་རབས་ཉི་ཤུ་པའི་ནང་ནས་བཟུང་སྟེ་ཤིན་ཅང་ནང་ཤུགས་ཚད་ ༦ ཡན་གྱི་ས་ཡོམ་ ༡༠༤ བརྒྱབ་ཡོད་པ་དང་། དེའི་ནང་དུ་ཤུགས་ཚད་ ༧ ཅན་ཐེངས་ ༡༤། ཤུགས་ཚད་ ༨ ཅན་ཐེངས་གཉིས་བཅས། ཆ་སྙོམས་བྱས་ན་ལོ་རེར་ཤུགས་ཚད་ ༦ ལས་བརྒལ་བའི་ས་ཡོམ་ཐེངས་གཅིག་རེ་འབྱུང་བཞིན་ཡོད་འདུག   གསར་འགོད་པ། པདྨ་མཚོ།
