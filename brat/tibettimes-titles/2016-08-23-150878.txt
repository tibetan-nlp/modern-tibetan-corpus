ཟླ་དྲུག་ནང་བྱིས་པ་སྐྱབས་བཅོལ་བ་ ༢༦༠༠༠
།

མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་བྱིས་པའི་ཐེབས་རྩ་ཁང་གིས་འདས་པའི་ཟླ་བ་ ༦ ནང་ཨ་རིའི་མཐའ་མཚམས་སུ་བྱིས་པ་སྐྱབས་བཅོལ་བ་ ༢༦༠༠༠ འཛིན་བཟུང་བྱས་ཡོད་སྐོར་བརྗོད་འདུག ཁ་སང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་བྱིས་པའི་ཐེབས་རྩ་ཁང་གིས་སྙན་ཐོ་ཞིག་བཏོན་པའི་ནང་། འདས་པའི་ཟླ་བ་ ༦ རིང་ཨ་རིའི་ཁྱིམ་མཚེས་རྒྱལ་ཁབ་ཁག་གི་བྱིས་པ་སྐྱབས་བཅོལ་བ་ ༢༦༠༠༠ ཨ་རིའི་དབུས་མཐའ་མཚམས་ནས་འཛིན་བཟུང་བྱས་ཡོད་པ་མ་ཟད། ཁྱིམ་ཚང་དང་བཅས་པ་བྲོས་བྱོལ་དུ་ཡོང་མཁན་མི་མང་ ༢༩༧༠༠ ཡོད་པའི་གྲས་ཀྱི་མང་ཆེ་བ་བྱིས་པ་དང་བྱིས་པའི་ཨ་མ་ཡིན་ཞིང་། ཁོ་ཚོ་ཨ་རིའི་ཁྱིམ་མཚེས་རྒྱལ་ཁབ་ཨེད་སད་བ་ཌོར་ (El Salvador) དང་། གྷུ་ཊི་མ་ལ་ (Guatemala)། ཧོན་ཌུ་ར་སི་ (Honduras) བཅས་ནས་ཡིན་ཞིང་། རྒྱལ་ཁབ་དེ་དག་ནང་བཙན་གཡེམ་དང་ལྐོག་གསོད། དབུལ་ཕོངས་བཅས་ཀྱི་གནས་སྟངས་ཚབས་ཆེན་ཆགས་ཡོད་སྟབས། ཁོ་ཚོ་ཨ་རིའི་ནང་སྐྱབས་བཅོལ་དུ་ཡོང་མཁན་ཡིན་སྐོར་བརྗོད་འདུག སྙན་ཐོ་དེའི་ནང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་བྱིས་པའི་ཐེབས་རྩ་ཁང་གི་འཛིན་སྐྱོང་འགན་འཛིན་གཞོན་པ་ཧྥོར་སེ་ཐི་ (Forsyth) ཟེར་བ་ཞིག་གིས། བྱིས་པ་སྐྱབས་བཅོལ་བ་དེ་ཚོའི་གནས་སྟངས་ལ་བསམ་བློ་བཏང་ཚེ། ཧ་ཅང་ཡིད་སྐྱོ་བ་ཞིག་རེད། བྱིས་པ་དེ་ཚོས་ཉེན་ཚབས་ཆེ་བའི་བྲོས་བྱོལ་བརྒྱུད་ནས་བདེ་འཇགས་ཀྱི་འཚོ་གནས་འཚོལ་བཞིན་ཡོད་པ་དང་། ཁོ་ཚོ་སོ་སོའི་རྒྱལ་ཁབ་ནང་གི་བཙན་གཡེམ་དང་འཁྲུག་ཟིང་། དབུལ་ཕོངས་བཅས་ཀྱི་གནས་སྟངས་འོག་ནས་ཨ་རིའི་ནང་སྐྱབས་བཅོལ་དུ་ཡོང་གི་ཡོད་སྟབས། ང་ཚོས་རྒྱལ་ཁབ་དེ་དག་གི་དབུལ་ཕོངས་དང་འཁྲུག་ཟིང་འགོག་ཐབས་བྱེད་དགོས་པ་ཁ་ཚ་དགོས་གཏུགས་ཆགས་ཡོད་ལ། ང་ཚོར་བྱིས་པ་དེ་དག་སྲུང་སྐྱོབ་བྱེད་པའི་འོས་འགན་ཡོད། ཅེས་སོགས་བརྗོད་འདུག སྙན་ཐོ་དེའི་ནང་ད་དུང་། གལ་ཏེ་བྱིས་པ་སྐྱབས་བཅོལ་བ་དེ་དག་ཕྱིར་སྐྲོད་བཏང་བ་ཡིན་ན། རྒྱལ་ཁབ་དེ་དག་ནང་ཁྲིམས་འགལ་ནག་ཉེས་བཟོ་མཁན་ཚོས་བྱིས་པ་དེ་ཚོར་བཙན་གཡེམ་དང་དམར་གསོད་གཏང་སྲིད། བྱིས་པ་སྐྱབས་བཅོལ་བ་དང་ཡུལ་གྱར་བ་ཚོ་མཐའ་མཚམས་ནས་འཛིན་བཟུང་དང་བཀག་བསྐྱིལ་བྱས་མི་ཆོག ཁོ་ཚོར་འཕྲོད་བསྟེན་ལྟ་སྐྱོང་གི་ཞབས་ཞུ་ཆ་ཚང་སྤྲོད་དགོས། ཁོ་ཚོ་སོ་སོའི་ཕ་མ་དང་ལྷན་དུ་འཚོ་སྡོད་བྱེད་དུ་འཇུག་དགོས། ཞེས་བཀོད་འདུག བྱིས་པའི་ཐེབས་རྩ་ཁང་གིས་རྒྱལ་སྤྱིའི་ཉེ་ཡངས་ཚོགས་པའི་ (Amnesty International) སྙན་ཐོ་ཞིག་ལུང་འདྲེན་བྱས་པའི་ནང་། མེག་སེ་ཁོ་ (Mexico) དང་གྷུ་ཊི་མ་ལ་ (Guatemala) རྒྱལ་ཁབ་གཉིས་ནང་བུ་མོ་མང་པོ་ཞིག་བཙན་དབང་གི་སྒོ་ནས་གཞང་ཁང་དང་ཆང་ཁང་ནང་ལས་ཀ་བྱེད་དུ་འཇུག་གི་ཡོད་པ་དང་། རྒྱལ་ཁབ་དེ་དག་ནས་བྲོས་བྱོལ་དུ་ཡོང་མཁན་བུ་མོ་ ༡༠ གྲས་ནས་བུ་མོ་ ༦ ལ་ལམ་བར་ནས་བཙན་གཡེམ་གཏོང་གི་འདུག ཅེས་བརྗོད་འདུག གནས་ཚུལ་ལྟར་ན། ཕྱི་ལོ་ ༢༠༡༤ ལོར་ཨ་རིའི་དབུས་རྒྱུད་མཐའ་ཚམས་སུ་བྱིས་པ་ཡུལ་གྱར་བ་ ༤༤༥༠༠ ལྷག་འབྱོར་རྗེས། ཨ་རིའི་གཞུང་གིས་མེག་སེ་ཁོ་སྲིད་གཞུང་ལ་མཐའ་མཚམས་སུ་བྱོར་བའི་བྱིས་པ་ཡུལ་གྱར་བ་ཚོ་བཀག་བསྐྱིལ་བྱས་མི་ཆོག་པའི་གནོན་ཤུགས་སྤྲད་ནའང་། ཕྱི་ལོ་ ༢༠༡༥ ལོར་མེག་སེ་ཁོ་སྲིད་གཞུང་གིས་བྱིས་པ་ཡུལ་གྱར་བ་ ༣༦༠༠༠ མཐའ་མཚམས་ནས་འཛིན་བཟུང་བྱས་ཡོད་འདུག ཕྱི་ལོ་ ༢༠༡༣ ལོར་མེག་སེ་ཁོ་སྲིད་གཞུང་གིས་བྱིས་པ་ཡུལ་གྱར་བ་ ༩༦༠༠ བཀག་བསྐྱིལ་བྱས་ཡོད་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
