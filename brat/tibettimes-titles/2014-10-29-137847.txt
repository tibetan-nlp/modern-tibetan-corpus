རྙི་སྟོད་ཡུལ་ཚོའི་འབྲོག་པའི་དཀའ་ངལ།

ཉེ་ལམ་རྒྱ་གཞུང་གིས་དཀར་མཛེས་ཁུལ་གསེར་རྟ་རྫོང་རྙི་སྟོད་ཡུལ་ཚོའི་སྡེ་བ་ཁག་གི་འབྲོག་པ་རྣམས་རྒྱ་གཞུང་གིས་གསར་རྒྱག་བྱས་པའི་སྡོད་ཁང་ནང་བཙན་ཤེད་ཀྱིས་སྤོ་རུ་བཅུག་ཡོད་ཀྱང་། སྡོད་ཁང་དེ་དག་ལ་སེར་ཀ་གས་བ་དང་། སྤུས་དག་མེད་པ་བཅས་ལ་བརྟེན་ཏེ་མང་ཚོགས་སྡོད་ཐབས་མེད་པ་ཆགས་ཏེ། གོང་རིམ་ལ་ཞུ་གཏུག་ཡང་ཡང་བྱས་ཀྱང་ཡ་ལན་སྤྲད་མེད་པ་མ་ཟད། དེ་སྔོན་རྒྱ་གཞུང་གིས་མང་ཚོགས་ལ་བཙན་ཤེད་ཀྱིས་བུ་ལོན་ལེན་བཅུག་ནས་སྡོད་ཁང་བརྒྱབ་པའི་དངུལ་བུན་དེ་གང་མགྱོགས་གཞུང་ལ་གཙང་སྤྲོད་བྱེད་དགོས་པའི་བཀའ་བཏང་ཡོད་འདུག བོད་ནང་གི་གནས་ཚུལ་མཁོ་སྤྲོད་པ་ཞིག་གིས་བརྗོད་དོན། ༢༠༠༩ ནས་བཟུང་རྒྱ་གཞུང་གིས་ཞིང་ཆེན་ཁག་གི་མངའ་ཁུལ་ནང་ཡོད་པའི་བོད་པའི་འབྲོག་པ་ཚོར་སྡོད་ཁང་གསར་རྒྱག་གི་ལས་གཞི་ཞིག་མགོ་བཙུགས་བྱས་ཡོད་པ་ལྟར། དཀར་མཛེས་ཁུལ་གསེར་རྟ་རྫོང་རྙི་སྟོད་ཡུལ་ཚོའི་མང་ཚོགས་ལའང་སྡོད་ཁང་གསར་རྒྱག་ཆེད་ཁྱིམ་ཚང་རེར་རྒྱ་སྒོར་ཁྲི་ ༣ རེ་སྤྲད་པ་དང་། གཞན་རྒྱ་སྒོར་ཁྲི་ ༢ རེ་བུ་ལོན་ལེན་དུ་བཅུག་སྟེ། ཡུལ་ཚོའི་མགོ་པ་དང་ཚོ་དཔོན་གཉིས་ལ་འགན་ལེན་དུ་བཅུག་ཅིང་། ཁོ་གཉིས་ཀྱིས་མང་ཚོགས་ཀྱི་ས་ནས་སྲིད་གཞུང་གིས་སྤྲད་པའི་རྒྱ་སྒོར་དང་བུ་ལོན་རྣམས་བསྡུས་ཏེ་སྡོད་ཁང་གསར་རྒྱག་བྱས་ཡོད་ཀྱང་། ད་ཆར་སྡོད་ཁང་དེ་དག་ལ་སེར་ཀ་གས་པ་དང་སྤུས་ཀ་སྐྱོ་བ་བཅས་ལ་བརྟེན་ཏེ་མི་སྡོད་ཐུབ་མེད་པ་ཆགས་འདུག ཅེས་དང་། ཡང་ཁོ་མོས་མུ་མཐུད་བརྗོད་དོན། མང་ཚོགས་ལ་སྡོད་ཁང་གིས་ཕན་ཐོགས་མིན་ལ་མ་ལྟོས་པར། རྒྱ་གཞུང་གིས་མང་ཚོགས་ལ་སྤྲད་པ་དང་བུ་ལོན་གཡར་བའི་དངུལ་དེ་དག་གང་མགྱོགས་ཕྱིར་སྤྲོད་དགོས་པ་དང་། ལོ་རེར་འབྲོག་པ་མང་ཚོགས་ལ་རྩྭ་རིན་དུ་སྤྲོད་བཞིན་པའི་རྒྱ་སྒོར་ཆིག་སྟོང་ལྷག་རེ་ཡོད་པ་དེ་ནས་གཅོག་འཐེན་བྱ་རྒྱུ་ཡིན་སྐོར་སོགས་ཀྱི་བཀའ་བསྒྲགས་འདུག ཅེས་སོགས་བརྗོད། གསར་འགོད་པ། དོན་གྲུབ།
