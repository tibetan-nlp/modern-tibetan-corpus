རིམ་པ་དང་པོའི་མགོ་ཁྲིད་ཅིག་ལ་ཞིབ་བཤེར།

རྒྱ་ནག་རྒྱལ་ཡོངས་སྲིད་གྲོས་ཀྱི་མགོ་ཁྲིད་གཞོན་པ་སུའུ་རོང་གིས་ཁྲིམས་འགལ་གྱི་ནག་ཉེས་ཚབས་ཆེན་བཟོས་ཡོད་པའི་དོགས་གཞིའི་འོག་ཞིབ་བཤེར་བྱེད་བཞིན་ཡོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༤ ཉིན་རྒྱ་ནག་དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ཨུ་ཡོན་ལྷན་ཁང་གི་དྲ་བའི་ཐོག རྒྱ་ནག་རྒྱལ་ཡོངས་སྲིད་གྲོས་ཀྱི་མགོ་ཁྲིད་གཞོན་པ་སུའུ་རོང་གིས། རྒྱལ་ཁབ་ཀྱི་ཁྲིམས་དང་འགལ་བའི་ནག་ཉེས་ཚབས་ཆེན་བཟོས་ཡོད་པའི་དོགས་གཞིའི་འོག སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ཨུ་ཡོན་ལྷན་ཁང་གིས་ཞིབ་བཤེར་བྱེད་བཞིན་ཡོད་སྐོར་བསྒྲགས་ཡོད་པ་དང་། དེ་ནི་རྒྱ་ནག་དམར་ཤོག་ཚོགས་པའི་སྐབས་ ༡༨ པའི་ཚོགས་ཆེན་ཚོགས་རྗེས། རྒྱལ་ཁབ་ཀྱི་རིམ་པ་དང་པོའི་མགོ་ཁྲིད་ཞིག་གི་ནག་ཉེས་ཕྱིར་བསྒྲགས་བྱས་པ་ཐོག་མ་དེ་ཆགས་ཡོད་སྟབས། རྒྱལ་ཕྱི་ནང་ནས་དོ་སྣང་དང་གླེང་སློང་ཆེན་པོ་བྱེད་བཞིན་འདུག རྒྱ་ནག་གཞུང་ཕྱོགས་ནས་སུའུ་ལ་ནག་ཉེས་གང་ཡོད་གསལ་བོ་བརྗོད་མེད་ཀྱང་། གསར་ལམ་གཞན་ནས་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། འདི་ལོའི་ཕྱི་ཟླ་ ༢ ཚེས་ ༢༡ ཉིན། སུའུ་རོང་རྒྱ་ནག་ཆབ་སྲིད་ལྷན་ཁང་གི་མགོ་ཁྲིད་ཟུར་པ་ཀྲུའུ་ཡཱོང་ཁང་དང་ལིན་ཅི་ཧྭ་ཚོགས་པའི་ཚོགས་མི་གཙོ་བོ་ཞིག་ཡིན་པའི་དོགས་གཞིའི་འོག ཞིབ་བཤེར་བྱེད་མགོ་བརྩམས་ཡོད་པ་མ་ཟད། ཁོའི་ཆུང་མ་ཡཱས་ཅེ་ཡིས་ཅི་ལིན་ཞིང་ཆེན་ཞིན་ཡཱས་གྲོང་ཁྱེར་དུ་ས་ཆ་ཉོ་ཚོང་དང་འཛུགས་སྐྲུན་ལས་གཞིའི་ཁྲོད་ལྐོག་ཟ་རུལ་སུངས་ཚབས་ཆེན་བྱས་ཡོད་པའི་ནག་ཉེས་འོག་འཛིན་བཟུང་བྱས་ཡོད་པ་དང་། དེ་བཞིན། སུའུ་ཡི་ཚ་བོ་ཡང་འཛིན་བཟུང་བྱས་ཡོད་སྟབས། གནད་དོན་དེ་ཚོ་སུའུ་དང་ཐད་ཀར་འབྲེལ་བ་ཡོད་ལུགས་བརྗོད་ཀྱི་ཡོད་པ་མ་ཟད། ཕྱི་ལོ་ ༢༠༡༠ ལོར་སུའུ་རོང་ཅང་ཞི་ཞིང་ཆེན་གྱི་མགོ་ཁྲིད་ཡིན་སྐབས། ཅང་ཞི་ཞིང་ཆེན་ཧྥོ་ཀྲུའུ་གྲོང་ཁྱེར་དུ་དྲག་ཆར་གྱི་གནོད་སྐྱོན་བྱུང་སྟེ། མི་མང་པོར་ཤི་རྨས་བྱུང་བའི་གནད་དོན་སྦ་གསང་བྱས་ཏེ་ལས་འགན་ལ་གཡོ་ཟོལ་བྱས་ཡོད་ལུགས་དང་། ལྐོག་ཟ་རུལ་སུངས་ཀྱི་དོགས་གཞིའི་འོག་ཞིབ་བཤེར་བྱེད་བཞིན་པའི་ཧྭ་རིན་ཚོང་གཉེར་ཁང་གི་སྤྱི་ཁྱབ་ཟུར་པ་སུང་ལེན་དང་སྒེར་གྱི་འབྲེལ་བ་ཟབ་མོ་ཡོད་སྐོར་སོགས་བརྗོད་ཀྱི་འདུག སུའུ་རོང་ནི་རྒྱ་ནག་ཅི་ལིན་ཞིང་ཆེན་པའེ་ཀྲོ་གྲོང་ཁྱེར་ནས་ཡིན་པ་དང་། ཕྱི་ལོ་ ༡༩༨༥ ལོར་ཅི་ལིན་ཞིང་ཆེན་ཏང་སློབ་ཀྱི་ཏང་གི་ལས་བྱེད་ཆེད་སྦྱོང་འཛིན་གྲྭ་ནས་མཐར་ཕྱིན་པ་ནས་བཟུང་སྟེ། ཅི་ལིན་ཞིང་ཆེན་པའེ་ཀྲོ་དང་སི་ཕིན་གྲོང་ཁྱེར་སྤྱི་ཁྱབ་ཀྱི་འགན་འཁུར་མྱོང་བ་དང་། ཕྱིས་སུ་ཅི་ལིན་དང་མཚོ་སྔོན། ཀན་སུའུ། ཅང་ཞི་ཞིང་ཆེན་བཅས་ཀྱི་མགོ་ཁྲིད་བྱས་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༡༣ ལོའི་རྒྱ་ནག་སྐབས་ ༡༢ པའི་སྲིད་གྲོས་ཚོགས་ཆེན་ཐོག་རྒྱལ་ཡོངས་སྲིད་གྲོས་ཀྱི་མགོ་ཁྲིད་གཞོན་པར་བསྐོ་བཞག་བྱས་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
