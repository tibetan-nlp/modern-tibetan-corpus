རུང་རྒྱས་ཨ་གྲཊ་ལཊ་ཀློད་གྲོལ་བཏང་།

མདང་དགོང་ཆབ་སྲིད་བཙོན་པ་རུང་རྒྱས་ཨ་གྲགས་ལགས་བཙོན་ཁང་ནས་ཀློད་གྲོལ་བཏང་སྟེ་རང་ཁྱིམ་དུ་བསྐྱལ་ཡོད་འདུག བོད་ཀྱི་ཆབ་སྲིད་བཙོན་པ་རུང་རྒྱས་ཨ་གྲགས་ལགས་བཙོན་འཇུག་དུས་ཡུན་རྫོགས་ཏེ་ཀློད་གྲོལ་བཏང་ཡོད་ཀྱང་། ཁོང་གི་ཁྱིམ་མི་གཉེན་འབྲེལ་ཚོར་བརྡ་ལན་གང་ཡང་མ་སྤྲད་པར་མདང་དགོང་བོད་ནང་གི་མཚན་གུང་ཆུ་ཚོད་དང་པོའི་ཐོག་གློ་བུར་ངང་རང་ཁྱིམ་དུ་བསྐྱལ་ཡོད་འདུག དེ་རིང་འདི་གས་གསར་འགོད་པས་ཁོང་གི་ཚ་བོ་སྤྱི་འཐུས་ཨ་ཀྲུག་ཚེ་བརྟན་ལགས་སུ། རུང་རྒྱས་ཨ་གྲགས་ལགས་ཀྱི་སྐུ་གཟུགས་གནས་སྟངས་གང་འདྲ་ཡོད་སོགས་དྲིས་པར་ཁོང་གིས། རུང་རྒྱས་ཨ་གྲགས་ལགས་སི་ཁྲོན་མེན་ཡཱང་བཙོན་ཁང་ནས་ཀློད་གྲོལ་བཏང་སྟེ། མདང་དགོང་ཆུ་ཚོད་དང་པོའི་ཐོག་གློ་བུར་ངང་རང་ཁྱིམ་དུ་བསྐྱལ་ཡོད་འདུག ད་ལྟ་འབྲེལ་བ་བྱེད་རྒྱུ་ཁག་པོ་ཆགས་འདུག་པས་ཁོང་གི་སྐུ་གཟུགས་གནས་སྟངས་ག་འདྲ་ཡོད་སོགས་གནས་ཚུལ་གསལ་པོ་ཤེས་རྟོགས་ཐུབ་ཀྱི་མི་འདུག ཕལ་ཆེར་སྐུ་གཟུགས་གནས་སྟངས་ཨོ་ཙམ་ཡོད་ས་རེད། ཅེས་བརྗོད་བྱུང་། རུང་རྒྱས་ཨ་གྲགས་ལགས་ནི་ལི་ཐང་གི་འབྲོག་མི་དཀྱུས་མ་ཞིག་ཡིན་ཞིང་། ཕྱི་ལོ་༢༠༠༧ ཟླ་ ༨ ཚེས་ ༡ ཉིན་རྒྱ་ནག་གི་མགོ་ཁྲིད་དང་མི་མང་སྟོང་ཕྲག་མང་པོ་འདུ་འཛོམས་བྱས་པའི་ལི་ཐང་རྟ་རྒྱུགས་ཆེན་མོའི་མཛད་སྒོའི་སྟེང་། ཁོང་མདུན་ཕྱོགས་ཀྱི་སྡིངས་ཆའི་སྒང་ཐོན་ཏེ་༧གོང་ས་མཆོག་བོད་མི་ས་ཡ་དྲུག་གི་མགོན་སྐྱབས་ཡིན་ལ། ཁོང་མགྱོགས་མྱུར་ཏུ་བོད་ལ་གདན་འདྲེན་ཞུ་དགོས་པ། པཎ་ཆེན་སྐུ་ཕྲེང་བཅུ་པའི་ཡང་སྲིད་དགེ་འདུན་ཆོས་ཀྱི་ཉི་མ་མཆོག་དང་། སྤྲུལ་སྐུ་བསྟན་འཛིན་བདེ་ལེགས་རིན་པོ་ཆེ་རྣམ་གཉིས་ཀློད་གྲོལ་གཏོང་དགོས་པ་དང་། དེ་བཞིན་རྒྱ་གཞུང་གི་དྲག་གནོན་འོག་བོད་ནང་རང་དབང་མེད་པའི་དངོས་ཡོད་གནས་སྟངས་སྐོར་གྱི་གཏམ་བཤད་སྤེལ་ཡོད་སྟབས། རྒྱ་གཞུང་གི་ཉེན་རྟོག་པས་དེ་མ་ཐག་ཁོང་འཛིན་བཟུང་བྱས་ཡོད་པ་མ་ཟད། ལོ་དེའི་ཟླ་ ༡༡ པའི་ནང་དཀར་མཛེས་བོད་རིགས་རང་སྐྱོང་ཁུལ་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་ཟེར་བས། རྒྱལ་ཁབ་ཁ་བྲལ་དང་སྤྱི་ཚོགས་བརྟན་བརླིང་ལ་གནོད་སྐྱོན་བཟོས་པའི་ཉེས་མིང་འོག ཁོང་ལ་ལོ་ ༨ ཀྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་དེ་སི་ཁྲོན་མེན་ཡཱང་བཙོན་ཁང་དུ་བཅུག་ཡོད་པ་རེད། གསར་འགོད་པ། པདྨ་མཚོ།
