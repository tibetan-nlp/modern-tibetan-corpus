ཉུང་མཐར་མི་ ༡༠ དམར་གསོད་བཏང་།

ཧིན་རྡུ་ནེ་ཤ་ཡའི་རྒྱལ་སའི་ནང་དྲག་སྤྱོད་པས་དམར་གསོད་དོན་རྐྱེན་བསླངས་ཏེ་ཉུང་མཐར་མི་གྲངས་ ༡༠ ཡི་ཚེ་སྲོག་ཤོར་ཡོད་འདུག   དེ་རིང་སྔ་དྲོར་ཧིན་རྡུ་ནེ་ཤ་ཡའི་རྒྱལ་ས་ཇཱ་ཀར་ཊཱ་ (Jakarta )རུ་ཡོད་པའི་འཛམ་གླིང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཡན་ལག་ལས་ཁུངས་ཀྱི་ཉེ་འགྲམ་ཚིག་ཇ་ཁང་ཞིག་དང་ཉེན་རྟོག་བདེ་སྲུང་ལྟེ་གནས་ཤིག་ཏུ། དྲག་སྤྱོད་པ་ཁག་ཅིག་གིས་དམར་གསོད་དོན་རྐྱེན་བསླངས་ཏེ། ད་ལྟའི་ཆར་ཉེན་རྟོག་པ་ ༣ དང་མི་སེར་དཀྱུས་མ་ ༢། དྲག་སྤྱོད་པ་ ༥ བཅས་ཉུང་མཐར་མི་གྲངས་ ༡༠ ཡི་ཚེ་སྲོག་ཤོར་ཡོད་འདུག   ཉེན་རྟོག་ཕྱོགས་ནས་བརྗོད་པ་ལྟར་ན། ད་ལྟའི་ཆར་དྲག་སྤྱོད་པ་ག་ཚོད་ཡོད་མེད་གསལ་པོ་ཤེས་རྟོགས་ཐུབ་ཀྱི་མེད་ཀྱང་། དྲག་སྤྱོད་པ་ ༤ འཛིན་བཟུང་བྱས་ཡོད་པ་དང་། ཐེངས་འདིའི་དོན་རྐྱེན་ཁྲོད་ཁྱོན་དྲག་སྤྱོད་པ་ ༡༤ ཡོད་པའི་ཚོད་དཔག་བྱེད་བཞིན་ཡོད་འདུག   དོན་རྐྱེན་བྱུང་རྗེས། ཁ་ཆེའི་དྲག་སྤྱོད་ཚོགས་པས་ (IS) ཐེངས་འདིའི་དམར་གསོད་དོན་རྐྱེན་དེ་ཁོ་ཚོས་བསླངས་བ་ཡིན་སྐོར་བསྒྲགས་ཡོད་པ་དང་། ཧིན་རྡུ་ནེ་ཤ་ཡའི་ཉེན་རྟོག་ཕྱོགས་ནས་ཁ་ཆེའི་དྲག་སྤྱོད་ཚོགས་པ་དང་འབྲེལ་བ་ཡོད་པའི་ས་གནས་ཀྱི་ཚོགས་པ་ཞིག་གིས་དོན་རྐྱེན་དེ་བསླངས་ཡོད་པའི་དོགས་པ་བྱེད་ཀྱི་ཡོད་འདུག   ཕྱི་ལོ་ ༢༠༠༩ ནས་བཟུང་ད་ལྟའི་བར་དུ་ཧིན་རྡུ་ནེ་ཤ་ཡའི་ནང་དྲག་སྤྱོད་པས་དམར་གསོད་དོན་རྐྱེན་ཐེངས་ ༦ བསླངས་ཏེ། ཉུང་མཐར་མི་གྲངས་ ༢༧༢ ལྷག་དམར་གསོད་བཏང་ཡོད་པ་རེད།   གསར་འགོད་པ། པདྨ་མཚོ།
