
དོལ་རྒྱལ་གྱི་ངོ་རྒོལ་བ་ཚོ་ཆིབས་འཁོར་འཁྲིས་ལ་བསླེབས་མི་འདུག

  སྲིད་སྐྱོང་མཆོག་གིས་ཉེ་སྔོན་ཨ་རིའི་ནང་བྱུང་བའི་སྐུ་ཕྱྭ་དང་འབྲེལ་བའི་གནད་དོན་ཐོག་གསལ་བཤད་གནང་སྐབས་དོན་གཅོད་དང་སྤྱི་མཐུན་ཚོགས་པ་སོགས་ནས་གནང་དགོས་པ་ཆ་ཚང་གནང་ཡོད་ཀྱང་། ཨར་བྷན་མཐོ་སློབ་ནས་བཀའ་འཁྲོལ་གནང་མེད་པར་བརྟེན་ཕེབས་བསུར་བཅར་ཐུབ་མེད་སྐོར་གསུངས་སོང་།   དེ་རིང་སྐབས་བཅོ་ལྔ་པའི་བོད་མི་མང་སྤྱི་འཐུས་ལྷན་ཚོགས་ཀྱི་གྲོས་ཚོགས་ཚོགས་དུས་བཅུ་པའི་ལས་ཉིན་དྲུག་པའི་སྔ་དྲོར། སྤྱི་འཐུས་བཀྲ་ཤིས་རྣམ་རྒྱལ་ལགས་ཀྱིས་འདི་ལོའི་ཕྱི་ཟླ་ ༧ པའི་ནང་༧གོང་ས་མཆོག་ཨ་རིའི་གྲོང་ཁྱེར་ལོསེ་ཨེནྫི་ལེ་སེ་ཡར་ཆིབས་བསྒྱུར་སྐབས། མཛད་འཆར་གནང་ཡུལ་ཚོགས་ཁང་གི་སྒོ་ཆེན་རྒྱབ་མདུན་གཉིས་ཀར་དོལ་རྒྱལ་གྱི་ངོ་རྒོལ་སྐད་འབོད་བྱེད་མཁན་ཁོ་ན་མ་གཏོགས་ཕེབས་བསུ་ཞུ་མཁན་གྱི་བོད་མི་ཚོ་བཅར་ཐུབ་མེད་པ་མ་ཟད། དོལ་རྒྱལ་རྗེས་འབྲང་པ་ཚོ་ཕེབས་འཁོར་དང་ཐག་ཉེར་སླེབས་ནས་སྐུ་ཕྱྭར་འགལ་རྐྱེན་ཡོང་དོགས་ཀྱི་བློ་འཚབ་བཟོ་གཞིའི་གནས་ཚུལ་བྱུང་ཡོད་སྟབས། དེ་སྐུ་ཕྱྭའི་འགན་སྲུང་དང་འབྲེལ་བའི་གལ་ཆེའི་ལས་དོན་ལ་འགལ་འཛོལ་སོང་བའི་གནད་དོན་ཞིག་ལ་ངོས་འཛིན་གནང་གི་ཡོད་མེད་དང་། དེ་འགན་ཁུར་བ་སུའི་ཐད་ནས་འཐུས་ཤོར་སོང་བར་ངོས་འཛིན་གནང་གི་ཡོད། གལ་སྲིད་འཐུས་ཤོར་ལ་ངོས་འཛིན་གནང་གི་ཡོད་ན། མ་འོངས་པར་འདི་རིགས་ཡང་ཡང་མི་ཡོང་བར་བཀའ་ཤག་གིས་ལས་དོན་གནང་འཆར་གང་ཡོད་མེད་སོགས་སྐར་རྟགས་ཅན་གྱི་དྲི་བ་ཕུལ་བར་སྲིད་སྐྱོང་མཆོག་གིས། བཀའ་ཤག་གིས་གནད་དོན་ཚང་མའི་ནང་ནས་༧གོང་མཆོག་གི་སྐུ་ཕྱྭའི་གནད་དོན་དེ་གལ་ཆེ་ཤོས་སུ་ངོས་འཛིན་ཞུས་དང་ཞུ་བཞིན་ཡོད་ཅིང་། སྔ་ལོ་སན་ཕ་རན་སི་སེ་ཁོར་གནས་ཚུལ་བྱུང་བ་ནས་བཟུང་འབྲེལ་ཡོད་ཆོས་རིག་ལས་ཁུངས་དང་བདེ་སྲུང་ལས་ཁུངས། ཕྱི་དྲིལ་ལས་ཁུངས་བཅས་སོ་སོ་ནས་སྐུ་ཚབ་དོན་གཅོད་ལས་ཁུངས་རྣམས་ལ་སྐུ་ཕྱྭའི་སྲུང་སྐྱོབ་དང་ཉེན་དོགས་དང་འབྲེལ་བའི་ལམ་སྟོན་བཀོད་ཁྱབ་གསལ་ནན་ཞུས་ཡོད་པ་དང་། སྐུ་ཚབ་དོན་གཅོད་ལས་ཁུངས་རྣམས་ནས་ཀྱང་ལམ་སྟོན་དེ་དག་གཞིར་བཟུང་འགན་འཁུར་ལག་བསྟར་ཞུ་བཞིན་ཡོད་ཀྱང་། འདི་ལོའི་ཕྱི་ཟླ་ ༧ པའི་ནང་ལོསེ་ཨེནྫི་ལེ་སེ་ཡར་གནས་ཚུལ་ཞིག་སླར་ཡང་ལྷག་འདུག་པ་དེར་བློ་ཕམ་ཡོད། གནས་ཚུལ་དེའི་འགན་འཁྲི་ནི། ས་གནས་སུ་གདན་ཞུ་བྱེད་མཁན་དང་མཛད་འཆར་གོ་སྒྲིག་བྱེད་མཁན་གྱི་ཚོགས་པ་དང་། ས་གནས་བོད་རིགས་སྤྱི་མཐུན་ཚོགས་པ། འབྲེལ་ཡོད་སྐུ་ཚབ་དོན་གཅོད་ལས་ཁུངས་བཅས་ལ་ཡོད་པ་ཞུ་མ་དགོས་པ་ཞིག་ཡིན། མ་འོངས་པར་གནས་ཚུལ་དེ་རིགས་ཡང་ཡང་མི་ཡོང་སླད་ཉེ་སྔོན་བཀའ་ཤག་གིས་འབྲེལ་ཡོད་ཕྱི་དྲིལ་ལས་ཁུངས་དང་། བདེ་སྲུང་ལས་ཁུངས་གཉིས་ལ་དམིགས་བསལ་བཀོད་ཁྱབ་ཅིག་གནང་ཡོད་པས་དེར་མཇུག་གནོན་བྱས་པའི་སྒོ་ནས་ཞུ་སྒོ་གང་མཆིས་ཞུ་རྒྱུ་ཡིན་སྐོར་སོགས་ལན་འདེབས་གནང་སོང་།   ཡང་འཕྲོས་དོན་དྲི་བའི་ནང་སྤྱི་འཐུས་བཀྲ་ཤིས་རྣམ་རྒྱལ་ལགས་ཀྱིས་སྲིད་སྐྱོང་མཆོག་གི་བཀའ་ལན་ནང་དུ་བཀའ་ཤག་ནས་སྐུའི་ཉེན་སྲུང་ཞུ་ཕྱོགས་ཐད་འགལ་འཛོལ་བྱུང་བའི་ཐུགས་དགོངས་བྱུང་སོང་ཞེས་གསུངས་སྟངས་ལ་བལྟས་པ་ཡིན་ན། འགལ་འཛོལ་བྱུང་བར་ངས་ངོས་འཛིན་ཞུ་ཡི་ཡོད། གལ་སྲིད་འགལ་འཛོལ་སོང་བར་ངོས་འཛིན་གནང་གི་ཡོད་ན། འགལ་འཛོལ་སོང་མཁན་དེ་ལ་ཕྱག་ལེན་ག་རེ་གནང་འཆར་ཡོད་དྲིས་པར་སྲིད་སྐྱོང་མཆོག་གིས། བཀའ་ཤག་གི་ལན་ནང་གསལ་པོ་ཞེ་དྲགས་འཁོད་ཡོད། ད་ལྟའི་བཀུར་འོས་སྤྱི་འཐུས་ཀྱིས་དྲི་བའི་ནང་ག་རེ་བཀོད་འདུག་གསུངས་ན། དོལ་རྒྱལ་རྗེས་འབྲང་པ་ཚོ་ཆིབས་འཁོར་གྱི་འགྲམ་ལ་བསླེབས་པ་རེད་ཅེས་འཁོད་ཡོད་སྟབས། བཀའ་ཤག་ནས་གནས་ཚུལ་ཧ་གོ་པ་དེ་ཞུས་ན་བསམ་བྱུང་། བཀུར་འོས་སྤྱི་འཐུས་མཆོག་ནས་ཕལ་ཆེར་གནས་ཚུལ་ཇི་མ་ཇི་བཞིན་གང་འདྲ་བྱུང་ཡོད་མེད་མཁྱེན་མདོག་ཁ་པོ་རེད། དེའི་སྐོར་ལ་གཞི་རྩའི་ས་གནས་སྤྱི་མཐུན་ཚོགས་པ་ནས་དགའ་ལྡན་ཕོ་བྲང་ལ་ཞིབ་ཅིང་ཕྲ་བ་ཞུས་ཡོད་པ་མ་ཟད། དཔལ་ལྡན་ཚོགས་གཙོ་མཆོག་ལོསེ་ཨེནྫི་ལེ་སེ་ཡ་ནང་ཕེབས་པའི་སྐབས་ལའང་ངག་ཐོག་དང་། ཉེ་ཆར་ཡིག་ཐོག་ལ་ཞུས་ཡོད་པ་མ་ཟད། བདེ་སྲུང་ལས་ཁུངས་ལའང་འབྱོར་སོང་། དོན་གཅོད་དང་དོན་གཅོད་ལས་བྱེད་ཚོས་ཞུ་སྟངས་ལ་བལྟས་པ་ཡིན་ན་ག་རེ་བྱུང་འདུག་ཟེར་ན། ཉིན་མ་དང་པོ་ཧོན་ཌ་བསྟི་གནས་ཁང་དེ་དབྱིན་སྐད་ཐོག་ཞུས་ན་ (Public place) མི་མང་གིས་གླ་ཆོག་པའི་ཚོགས་ཁང་ཞིག་ཡིན་དུས། དེར་དོལ་རྒྱལ་རྗེས་འབྲང་པ་ཚོར་ངོ་རྒོལ་བྱེད་ས་ཞིག་བྱུང་ཡོད་པ་མ་ཟད། བོད་མི་ཚོ་ལའང་ཕེབས་བསུ་ཞུ་ས་ཞིག་ཀྱང་གནང་ཡོད་པ་རེད། དེ་དོན་གཅོད་དང་ས་གནས་སྤྱི་མཐུན་ཚོགས་པས་འབད་བརྩོན་གནང་ནས་ཆོག་མཆན་ཐོབ་པ་ལྟར། ལོསེ་ཨེནྫི་ལེ་སེ་ཡ་དང་སན་ཧྥ་རན་སེ་སི་ཁོ། པོ་ཌི་ལན་ཌི་ནང་གི་བོད་མི་གཙོས་མི་གྲངས་ ༣༠༠ ཙམ་ཞིག་བསླེབས་ནས་ཕེབས་བསུ་ཞུ་ཐུབ་ཡོད་ཀྱང་། ཉིན་མ་གཉིས་པ་དང་གསུམ་པའི་ཚོགས་ཁང་དེ་ཨར་བྷན་གཙུག་ལག་སློབ་ཆེན་ནང་ལ་ཚོགས་ཡོད་པ་རེད། ས་ཆ་དེ་མཐོ་སློབ་སྒེར་གྱི་ཆགས་ཡོད་སྟབས། སྤྱི་མཐུན་ཚོགས་པས་འབད་བརྩོན་གནང་ནས་མཐོ་སློབ་ལ་༧གོང་ས་མཆོག་ཕེབས་བསུ་ཞུ་ས་ཞིག་དགོས་པའི་གློག་འཕྲིན་བཏང་ནས་འབྲེལ་བ་མང་པོ་གནང་ཡོད་སྟབས། མཐོ་སློབ་ངོས་ནས་ང་ཚོ་༧གོང་ས་མཆོག་གདན་འདྲེན་ཞུ་མཁན་གྱི་ཚོགས་པ་དང་མཉམ་དུ་ལས་ཀ་བྱེད་ཀྱི་ཡོད་ཞུས་ཏེ། ཐོག་མར་ཆོག་མཆན་གང་ཡང་གནང་མེད་སྟབས། རྗེས་སུ་དོན་གཅོད་རང་ནས་ལས་བྱེད་རྒན་པ་ཞིག་དང་སྤྱི་མཐུན་ཚོགས་པའི་ཚོགས་གཙོ་མཉམ་དུ་ཕེབས་ནས་ཨར་བྷན་མཐོ་སློབ་ཀྱི་འབྲེལ་ཡོད་ལས་བྱེད་ཕེབས་རོགས་ཞུས་པ་ལྟར་ཕེབས་ཡོད་སྟབས། ང་ཚོར་ཕེབས་བསུ་ཞུ་སའི་ས་ཆ་ཞིག་གནང་རོགས་ཞེས་ཞུ་སྐབས། ཨར་བྷན་མཐོ་སློབ་ཀྱི་མཛད་འཆར་གོ་སྒྲིག་གནང་མཁན་ཚོགས་པ་དེའི་སྐུ་ཚབ་ཕེབས་ཡོད་སྟབས། ཁོ་རང་གིས་བཀའ་བསྐྱོན་གནང་ནས་གོ་སྒྲིག་གནང་མཁན་གྱི་ཚོགས་པ་དེའི་གནོན་ཤུགས་འོག་ལ་ཕེབས་བསུ་གནང་མཁན་བོད་མི་ཚོར་ཆོག་མཆན་གནང་མི་འདུག ཨར་བྷན་མཐོ་སློབ་ཀྱིས་གནང་མི་དགོས་པའི་རྒྱུ་མཚན་གང་ཡིན་གློག་འཕྲིན་ནང་ལ་གོ་སྒྲིག་ཚོགས་པ་དང་འབྲེལ་བ་གནང་རོགས་གནང་ཅེས་འཁོད་ཡོད་སྟབས། གོ་སྒྲིག་ཚོགས་པས་རྐྱེན་བྱས་ནས་མ་གནང་བ་རེད་ཅེས་སྤྱི་མཐུན་ཚོགས་པས་ཡིག་ཐོག་བཀོད་འདུག ཡིན་ཡང་གནས་ཚུལ་བྱུང་བཞག་པ་དེ་༧གོང་ས་མཆོག་ཆིབས་བསྒྱུར་བསྐྱངས་སྐབས་བོད་པ་ཕེབས་བསུ་ཞུ་མཁན་མ་བྱུང་བ་རེད་མ་གཏོགས། ཆིབས་འཁོར་འཁྲིས་ལ་མི་བསླེབས་བཞག་པ་དེ་འདྲ་རེད་མི་འདུག དེ་དགའ་ལྡན་ཕོ་བྲང་དང་འབྲེལ་ཡོད་དོན་གཅོད་ཆིབས་འཁོར་ནང་ཡོད་སྟབས་རྩད་གཅོད་བྱས་པ་ཡིན། ཆིབས་འཁོར་ཕེབས་སྐབས་སྐད་འབོད་རྒྱག་མཁན་ཚོའི་མདུན་ནས་ཕེབས་དགོས་པ་ཆགས་ཡོད་ཀྱང་། མཐོ་སློབ་ཀྱི་སྒོ་འགྲམ་ལ་སླེབ་སྐབས་དོལ་རྒྱལ་བསྟེན་མཁན་ཕར་བསླེབས་ནས་ངོ་རྒོལ་བརྒྱབ་པ་དེ་འདྲ་བྱུང་མི་འདུག ང་ཚོས་གསལ་བཤད་ཞུ་རྒྱུ་ཞིག་ལ་དོན་གཅོད་ལས་ཁུངས་དང་འབྲེལ་ཡོད་ལས་བྱེད། སྤྱི་མཐུན་ཚོགས་པ་བཅས་ནས་གནང་དགོས་པ་ཆ་ཚང་གནང་འདུག་ཀྱང་། ཨར་བྷན་མཐོ་སློབ་ནས་བཀའ་འཁྲོལ་གནང་མི་འདུག་ཞུ་རྒྱུ་ཡིན། ཞེས་གསུངས་སོང་།   སྤྱི་འཐུས་སྒྲོལ་མ་ཚེ་རིང་ལགས་ཀྱིས་འཕྲོས་དོན་བཀའ་འདྲི་ཞུ་སྐབས། བཀའ་ཤག་སྐབས་ ༡༤ པའི་ཐོག་ནས་དོལ་རྒྱལ་རྗེས་འབྲང་པ་ཚོས་༧སྐྱབས་མགོན་རིན་པོ་ཆེ་ག་པར་ཆིབས་བསྒྱུར་བསྐྱངས་ཀྱང་། སུན་བཙེར་བཟོས་དང་བཟོ་བཞིན་ཡོད་པ་དག་ལ་གཟིགས་ཕྱོགས་དང་གནང་ཕྱོགས་ག་རེ་ཡིན་མིན་གྲོས་ཚོགས་འདིའི་ཐོག་ལ་གསལ་པོ་ཞིག་བཟོ་རོགས་གནང་ཞེས་བཀའ་འདྲི་ཞུས་པར། ༧གོང་ས་མཆོག་གི་སྐུ་ཕྱྭ་ལས་གལ་ཆེ་བ་མེད་པ་དེ་དེ་སྔ་ནས་བཀའ་ཤག་གིས་གསལ་པོ་ཞུས་ཡོད་ལ། ད་སྔ་ཡིན་ཡང་སྐུ་ཕྱྭ་དེ་གལ་ཆེ་ཤོས་ལ་བརྩི་ཡི་ཡོད་པ་མ་ཟད། རྩིས་པ་གཞིར་བཟུང་གོང་དུ་ཞུས་པ་བཞིན་བཀའ་ཤག་དང་འབྲེལ་ཡོད་ལས་ཁུངས་བདེ་སྲུང་དང་ཕྱི་དྲིལ་ལས་ཁུངས། ཆོས་རིག་ལས་ཁུངས་བཅས་ཀྱི་ཐོག་ནས་ཀྱང་དོན་གཅོད་ཁག་ལ་ལམ་སྟོན་གནང་རྒྱུ་དང་། དོན་གཅོད་ཁང་གིས་ཀྱང་འབད་བརྩོན་གནང་རྒྱུ་སོགས་གང་ཅིའི་ཐོག་གནང་དང་གནང་མུས་ཡིན། ཉེ་བའི་ཆར་སྲུང་སྐྱོབ་གནད་དོན་ཡིན་ཡང་ཁེ་ན་ཌ་དང་ཨོ་སི་ཀྲོ་ལི་ཡ་ཡིན་ཡང་ཚད་མཐོ་རུ་དང་མང་དུ་བཏང་ཡོད་པ་དག་ནི། བཀའ་ཤག་ཐོག་ནས་འབད་བརྩོན་གནང་བར་བརྟེན་ནས་བྱུང་བཞག་པ་ཞིག་ཡིན་སྟབས། ཧ་ལས་པའི་གལ་ཆེན་པོ་ཞིག་ལ་ངོས་འཛིན་བྱེད་ཀྱི་ཡོད་པ་མ་ཟད། དེ་བཞིན་དོལ་རྒྱལ་རྗེས་འབྲང་པ་ཚོ་ཡིན་ཡང་རྒྱ་ནག་གཞུང་གིས་བེད་སྤྱོད་འོག་ལ་ཆབ་སྲིད་འཁྲབ་ནས་གཞི་མེད་སྐྱོན་འཛུགས་བྱེད་ཀྱི་ཡོད་པར་ཡིག་ངག་གཉིས་ཐོག་ནས་ངོས་འཛིན་བྱེད་ཀྱི་ཡོད་པ་དེ་སྔ་ཞུས་ཡོད་པ་དང་ད་ལྟ་ཡང་ཞུ་བཞིན་ཡོད། ཅེས་གསུངས་སོང་།   སྤྱི་འཐུས་ཡེ་ཤེས་སྒྲོལ་མ་ལགས་ཀྱིས། སྐུ་ཚབ་དོན་གཅོད་དང་བོད་རིགས་སྤྱི་མཐུན་ཚོགས་པས་ཕེབས་བསུ་གནང་ཡུལ་གྱི་ས་ཆ་གནང་རྒྱུའི་ཆོག་མཆན་ཞུས་ཡོད་ཀྱང་། ཕེབས་བསུར་བཅར་མ་ཐུབ་པ་བཟོ་མཁན་གྱི་ཚོགས་པ་དེ་སུ་ཡིན་མིན་དང་། ཁོ་ཚོས་ག་རེ་བྱས་ནས་དེ་འདྲ་གནང་ཡོད་དྲིས་པའི་ལན་དུ་སྲིད་སྐྱོང་མཆོག་གིས། ཚོགས་པ་དེའི་མིང་ལ་ (Friends of Dalai Lama ) ཏཱ་ལའི་བླ་མ་མཆོག་གི་གྲོགས་པོ་ཞེས་པ་ཞིག་ཡོད་པ་དེ་རེད། དེར་རྣམ་པ་ཚོས་ཞིབ་འཇུག་གནང་བ་ཡིན་ན། དེའི་ནང་ལ་དོ་བདག་སུ་ཡིན་མིན་ཚང་མ་བཀོད་ཡོད་པ་རེད་ཞུ་རྒྱུ་ཡིན། རྒྱུ་མཚན་ག་རེ་བགྲངས་འདུག་ལབ་ན། དོན་གཅོད་རང་ནས་ལས་བྱེད་ཅིག་དང་སྤྱི་མཐུན་ཚོགས་པའི་ཚོགས་གཙོ་ཕར་བཅར་ཏེ། སྐུ་ཕྱྭ་དང་འབྲེལ་བའི་ཐོག་ལ་སེམས་ཁྲལ་ཞེ་པོ་ཞིག་འདུག་སྟབས། དོལ་རྒྱལ་རྗེས་འབྲང་པ་ཚོ་ག་པར་བསྡད་ཀྱི་རེད་ཅེས་དྲི་སྐབས། ཨར་བྷན་མཐོ་སློབ་ཀྱི་འབྲེལ་ཡོད་ལས་བྱེད་ཚོས་ས་ཆ་དེ་རྩད་གཅོད་བྱས་འདུག དེ་ནས་ང་ཚོར་ཕེབས་བསུ་ཞུ་སའི་ས་ཆ་ཞིག་དགོས་ཀྱི་འདུག་ཟེར་སྐབས། ཨར་བྷན་མཐོ་སློབ་དང་ཚོགས་པ་དེའི་སྐུ་ཚབ་ཀྱིས་ཨར་བྷན་མཐོ་སློབ་ནང་རྒྱ་རིགས་སློབ་ཕྲུག་མང་པོ་ཞིག་ཡོད་སྟབས། འདིར་བོད་པ་མང་པོ་ཞིག་བསླེབ་པ་ཡིན་ན་རྒྱ་རིགས་སློབ་ཕྲུག་ཚོས་ག་རེ་བྱེད་མིན་ཧ་གོ་བ་མི་འདུག་ཟེར་ནས་སྤྱི་མཐུན་ཚོགས་པས་ཞིབ་ཅིང་ཕྲ་བ་ཡིག་ཐོག་བཀོད་ནས་དགའ་ལྡན་ཕོ་བྲང་དང་ཚོགས་གཙོ། བདེ་སྲུང་ལས་ཁུངས་བཅས་ལ་ཕུལ་ཡོད་པ་རེད། ཅེས་གསུངས་སོང་།   དེ་ནས་སྤྱི་འཐུས་པདྨ་འབྱུང་གནས་ལགས་ཀྱིས། དྲི་བ་དང་པོའི་ནང་འདི་ལོའི་ཕྱི་ཟླ་ ༧ པའི་ནང་ལོསེ་ཨེནྫི་ལེ་སེ་ཡ་ནང་གནས་ཚུལ་ཞིག་སླར་ཡང་ལྷག་འདུག་པ་དེར་བློ་ཕམ་བྱུང་ཞེས་གསུངས་ཡོད་སྟབས། གནས་ཚུལ་སྡུག་པོ་བྱུང་མེད་ཀྱང་བློ་ཕམ་བྱུང་ཟེར་སྐབས་སྐྱོན་ཞིག་ཡོད་པ་གསལ་པོ་རེད། མ་འོངས་པར་ང་ཚོའི་རྣ་བས་ཐོས་མི་རུང་བའི་སྐད་ཆ་དེ་འདྲ་ཞིག་བྱུང་བ་ཡིན་ན། བློ་ཕམ་བྱུང་ཟེར་ནས་བཞག་པ་རེད་དང་། ཡང་ན་དེ་ལས་གཞན་པ་ཞིག་གི་ཐོག་འགྲོ་རྒྱུ་ཡིན། དེའི་འགན་འཁྲི་སུ་ལ་ཡོད་དམ་ལབ་སྐབས། གདན་ཞུ་བྱེད་མཁན་དང་མཛད་འཆར་གོ་སྒྲིག་ཞུ་མཁན་ཚོགས་པ། ས་གནས་བོད་རིགས་སྤྱི་མཐུན་ཚོགས་པ་དང་འབྲེལ་ཡོད་སྐུ་ཚབ་ཚོགས་པ་བཅས་ལ་འགན་འཁྲི་ཡོད་ཅེས་ལབ་འདུག ང་ཚོ་ནམ་རྒྱུན་ཕྱི་འབྲེལ་ལས་ཁུངས་ཀྱི་ལས་བསྡོམས་མཇལ་སྐབས། ཕྱི་རྒྱལ་སྐུ་ཚབ་དོན་གཅོད་དེ་དག་༧རྒྱལ་བ་རིན་པོ་ཆེ་མཆོག་ག་པར་ཆིབས་བསྒྱུར་བསྐྱངས་ཀྱང་། ས་གནས་དེ་དང་དེའི་དོན་གཅོད་ཀྱིས་ཕེབས་བསུ་སྐྱེལ་གནང་རྒྱུ་དང་། སྐུའི་འཕྲིན་ལས་རྒྱས་པ་སོགས་སྐུ་ཚབ་དོན་གཅོད་དེ་ཚོར་ཡོད་བསམ་པ་ཞིག་མཐོང་གི་འདུག ད་རེས་སྐུ་ཚབ་དོན་གཅོད་དང་བོད་རིགས་སྤྱི་མཐུན་ཚོགས་པ། གདན་ཞུ་བྱེད་མཁན་དག་གདན་ཐོབ་ཅིག་སྒང་ལ་བཞག་འདུག མ་འོངས་པར་ཁོ་ཚོའི་གདན་ཐོབ་གཅིག་པ་ཆགས་ཀྱི་རེད་དམ། ཡང་ན་ལས་བསྡོམས་ནང་ཡོང་བ་ལྟར་སྐུ་ཚབ་དོན་གཅོད་དེར་འགན་གཙོ་བོ་འཇོག་གི་ཡོད་དང་མེད། འཇོག་གི་ཡོད་པ་ཡིན་ན་འགལ་འཛོལ་ལ་གཞིགས་པའི་ཐག་གཅོད་གང་གནང་འཆར་ཡོད་མེད་དྲིས་པར་སྲིད་སྐྱོང་མཆོག་གིས། བློ་ཕམ་བྱུང་ཟེར་བ་དེར་ངས་གསལ་བཤད་ཞུས་ཡིན་བསམ་གྱི་འདུག བོད་མི་ཚོས་ཉིན་མ་དང་པོ་དེར་ཕེབས་བསུ་ཞུ་ཐུབ་ཡོད་ཀྱང་། ཉིན་མ་གཉིས་པ་གསུམ་པར་ཕེབས་བསུ་ཞུ་རྒྱུའི་ས་ཆ་མ་རག་པའི་རྐྱེན་གྱིས་ཕེབས་བསུ་ཞུ་མ་ཐུབ་པ་དེར་བཀའ་ཤག་ནས་བློ་ཕམ་ཡོད་ཞུ་རྒྱུ་ཡིན་ཡང་། དོན་གཅོད་དང་ས་གནས་སྤྱི་མཐུན་ཚོགས་པས་གནང་དགོས་པ་ཆ་ཚང་གནང་འདུག་ཅེས་ཞུ་རྒྱུ་ཡིན། དེ་བཞིན་སྐུ་ཕྱྭ་དང་འབྲེལ་བའི་ཐོག་ལ་དོན་གཅོད་ལ་འགན་འཁྲི་ཡོད་ཀྱང་། གནང་སྟངས་གནང་ལུགས་དེ་ས་གནས་སྤྱི་མཐུན་ཚོགས་པས་ཉེ་བའི་ཆར་ནས་ང་ཚོ་བདེ་སྲུང་ལས་ཁུངས་དང་ཕྱི་དྲིལ་ལས་ཁུངས། འདི་ནས་བཀོད་ཁྱབ་ནང་ལའང་དང་བླངས་ཀྱི་སྐུ་སྲུང་བྱེད་མཁན་ཡོད་སྟབས། ཁོ་ཚོར་ཡིན་ཡང་འགན་ཞིག་གནང་གི་ཡོད་པ་མ་ཟད། དེ་བཞིན་གོ་སྒྲིག་ཞུ་མཁན་ཨར་བྷན་མཐོ་སློབ་དང་གདན་འདྲེན་ཞུ་མཁན་ཚོགས་པ་བཅས་ལ་འགན་ཡོད་ཀྱང་། སྤྱི་ཡོངས་ཀྱི་འགན་གཙོ་བོ་དེ་དོན་གཅོད་ལ་ཡོད་པ་རེད། ཡིན་ཡང་སྐུ་ཕྱྭ་དང་འབྲེལ་བའི་ཐོག་ལ་ད་གིན་ངས་ཞུས་པ་ལྟར། ཆིབས་འཁོར་འགྲམ་ལ་དོལ་རྒྱལ་རྗེས་འབྲང་པ་བསླེབས་པ་དེ་འདྲ་བྱུང་མི་འདུག ཅེས་གསུངས་སོང་།   སྤྱི་འཐུས་ཟླ་བ་ཕུན་སྐྱིད་ལགས་ཀྱིས། གནད་དོན་དེའི་ཐད་ཞིབ་འཇུག་ཚོགས་ཆུང་ཞིག་འཛུགས་རྒྱུར་སྤོབས་པ་དང་གདེང་ཚོད། འཛུགས་དགོས་ཀྱི་ཡོད་པ་ཞིག་ལ་ངོས་འཛིན་གནང་གི་ཡོད་དང་མེད་དྲིས་པར་སྲིད་སྐྱོང་མཆོག་གིས། བཀའ་ཤག་ནས་ལོསེ་ཨེནྫི་ལེ་ཡ་དང་ནེའུ་ཡོག་གཙོས་དེ་སྔོན་གྱི་ཆིབས་བསྒྱུར་དང་འབྲེལ་བའི་ཐོག་ནས་བཀོད་ཁྱབ་ཅིག་ཕུལ་ཚར་ཡོད་སྟབས། ད་ལྟའི་ཆར་དེར་ངོས་འཛིན་བྱེད་ཀྱི་ཡོད། ད་ལྟ་དེ་ཕེབས་བསུའི་ཆོག་མཆན་རག་དང་མ་རག་ཡིན་དུས་གྲོས་ཚོགས་ཀྱིས་དེའི་ཐོག་ཞིབ་འཇུག་ཅིག་གནང་དགོས་གསུངས་པ་ཡིན་ན། བཀའ་ཤག་ནས་བཀའ་བསྡུར་གནང་རྒྱུ་རེད། ཅེས་གསུངས་སོང་།   མཇུག་ཏུ་བདེ་སྲུང་བཀའ་བློན་འབྲོང་ཆུང་དངོས་གྲུབ་མཆོག་གིས་དོན་རྐྱེན་དེ་དང་འབྲེལ་བའི་ཐོག་གསལ་བཤད་གནང་དོན། ཕ་གར་གདན་ཞུ་བྱེད་མཁན་གྱི་མི་དེ་དགའ་ལྡན་ཕོ་བྲང་ནས་གསལ་པོ་ཞེ་དྲགས་ལོ་རྒྱུས་རྒྱབ་ལྗོངས་དང་གནས་སྟངས། ད་ལྟ་བར་དུ་ད་རེས་གཅིག་པུ་ཙམ་མིན་པར་མཛད་རིམ་གོ་སྒྲིག་ཆ་ཚང་མཁྱེན་བསྡད་པའི་སྒྲིག་འཛུགས་ཤིག་རེད་ཞུ་རྒྱུ་ཡིན་ལ། ལྷག་པར་དུ་བོད་མིའི་སྒྲིག་འཛུགས་དང་བདེ་སྲུང་ལས་ཁུངས་ནས་དེ་ཆ་ཚང་ཤེས་ཀྱི་ཡོད། དེ་འདྲའི་ཐོག་ལ་དོ་བདག་གྲ་སྒྲིག་བྱེད་མཁན་གྱི་རྒྱབ་ལྗོངས་ད་གིན་བཀུར་འོས་སྤྱི་འཐུས་ཀྱིས་ཐོག་མ་བཀའ་གནང་སྐབས་བཀའ་འཁྲོལ་མ་སྤྲད་པ་དེ་གང་འདྲ་ཡིན་ན། དོལ་རྒྱལ་བསྟེན་མཁན་དང་དེང་སང་དེ་འདྲའི་སྒྲིག་འཛུགས་མང་པོ་ཡོད་པ་དང་འབྲེལ་བ་མེད་འགྲོ་བསམ་པའི་ཐུགས་དོགས་བྱུང་སོང་བསམ་པའི་ཚོར་སྣང་དེ་འདྲ་བྱུང་། དེ་འདྲའི་ཐུགས་དོགས་ཁྱོན་ནས་མ་གནང་རོགས་གནང་། དེ་འདྲའི་ཐོག་ལ་ཐུགས་དོགས་རྩ་བ་ཉིད་ནས་གནང་དགོས་དོན་མེད་ཀྱང་། མི་སོ་སོ་སྒེར་གྱི་གཤིས་ཀ་དང་ལྟོས་གཞི་ཟེར་བ་དེ་ཚོ་གནས་ཚུལ་ལོག་ཁ་ཞིག་རེད་མ་གཏོགས། རྒྱབ་ལྗོངས་དང་གདན་འདྲེན་ཞུ་མཁན་མི་སྒེར་དང་། ཚོགས་པའི་རྒྱབ་ལྗོངས་ཐོག་ལ་ཐུགས་དོགས་གནང་དགོས་པའི་རྒྱུ་མཚན་གང་ཡང་མེད་གསལ་པོ་ཤེས་ཀྱི་ཡོད་ཅེས་སྙན་སེང་ཞུ་རྒྱུ་ཡིན། ཞེས་གསུངས་སོང་།   གསར་འགོད་པ། དོན་གྲུབ།
