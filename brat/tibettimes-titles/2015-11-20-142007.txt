ཡུ་གུར་དྲག་སྤྱོད་པ་བསད་པ་ཡིན་ཟེར།

  ནེའུ་ཡོག་དུས་བབ་སོགས་རྒྱལ་སྤྱིའི་གསར་ལམ་ཁག་ཅིག་ཐོག་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། རྒྱ་ནག་གི་དབང་འཛིན་འོག་ཡོད་པའི་ཞིན་ཅིང་ཡུ་གུར་རིགས་རང་སྐྱོང་ལྗོངས་ཟེར་བའི་ཨ་སི་ཁུ་(Asku) ས་གནས་ཀྱི་རྡོ་སོལ་གསེར་འདོན་ས་ཚིགས་ཤིག་ཏུ། དེ་རིང་སྔ་དྲོར་རྒྱ་ནག་རྒྱལ་སྲུང་དྲག་ཆས་དམག་དཔུང་གིས་གཏོར་རྒོལ་བཏང་སྟེ་ཡུ་གུར་བ་ ༡༦ དམར་གསོད་བཏང་ཡོད་པའི་གནས་ཚུལ་བཀོད་འདུག   གནས་ཚུལ་འདིའི་སྐོར་ཨེ་ཕེ་(AP)གསར་ཁང་གིས་བཀོད་དོན། རྒྱ་ནག་གཞུང་འབྲེལ་ཐེན་ཧྲན་དྲ་ཚིགས་སྟེང་རྒྱ་ནག་རྒྱལ་སྲུང་དམག་དཔུང་གིས་ཤིན་ཅང་ཡུ་གུར་མི་རིགས་རང་སྐྱོང་ལྗོངས་ཁུལ་དུ། ཁ་ཆེ་འཇིགས་སྐུལ་དྲག་སྤྱོད་པ་ཡིན་པའི་དོགས་ཡོད་མི་སྣ་ ༢༩ ལྷག་ཤི་རྨས་བཟོས་ཡོད་སྐོར་བཀོད་ཡོད་ཀྱང་། དོན་དངོས་ཐོག་རྒྱ་ནག་དྲག་ཆས་དམག་དཔུང་གིས་ཐེངས་འདིར་ཤིན་ཅང་ཡུ་གུར་མི་རིགས་རང་སྐྱོང་ལྗོངས་ཀྱི་ཨ་སི་ཁུ་ས་ཁུལ་གྱི་རྡོ་སོལ་གཏེར་སྒོ་ཁུངས་ཞིག་ལ་གཏོར་རྒོལ་བཏང་བའི་ཁྲོད་མི་ ༡༦ དམར་གསོད་བཏང་ཡོད་པ་དང་། དེའི་གྲས་སུ་བུད་མེད་དང་བྱིས་པའང་ཡོད་པ་མ་ཟད། དོགས་གཞིའི་མི་སྣ་གཅིག་ད་ལྟ་འཛིན་བཟུང་བྱས་ཡོད་འདུག ཅེས་བཀོད་འདུག   གནས་ཚུལ་དེའི་ནང་མུ་མཐུད་དུ་བཀོད་དོན། རྒྱལ་སྤྱིའི་བརྒྱུད་ལམ་ཁག་གིས་རྡོ་སོལ་གཏེར་ཁུངས་དེ་འགད་གཏོར་བཏང་བའི་དོན་རྐྱེན་དེའི་དངོས་ཡོད་གནས་ཚུལ་ཕྱིར་བསྒྲགས་བྱས་ཡོད་ཀྱང་། རྒྱ་གཞུང་གིས་གནས་ཚུལ་དེ་བཀག་སྡོམ་བྱས་ཏེ་དོན་རྐྱེན་དེའི་ཁྲོད་ཤི་རྨས་ཐེབས་པ་དག་ཤིན་ཅང་ནང་ཡོད་པའི་ཁ་ཆེ་འཇིགས་སྐུལ་དྲག་སྤྱོད་པ་ཡིན་ལུགས་སྙོན་འཛུགས་བྱས་ཡོད་པ་དང་། ཤིན་ཅང་ཡུ་གུར་མི་རིགས་རང་སྐྱོང་ལྗོངས་ནང་ཕྱི་ཕྱོགས་ཀྱི་ཁ་ཆེ་འཇིགས་སྐུལ་དྲག་སྤྱོད་པར་འབྲེལ་བ་ཡོད་པའི་མི་སྣ་ཡོད་ལུགས་ཤོད་ཀྱི་ཡོད་པ་མ་ཟད། པེ་ཅིན་གཞུང་གིས་ཡུ་གུར་ཁ་ཆེ་གྲངས་ཉུང་མི་རིགས་དང་རྒྱ་ནག་ཧན་མི་རིགས་གཉིས་བར་འབྱུང་བཞིན་པའི་རྩོད་རྙོག་དེ་རྒྱ་ནག་ནང་ཁུལ་གྱི་གནད་དོན་ཞིག་ཏུ་འགྲེལ་བཤད་བྱས་ཏེ། སྔ་རྗེས་སུ་བྱུང་བའི་དོན་རྐྱེན་ཡོད་ཚད་ཀྱི་དངོས་ཡོད་གནས་ཚུལ་གསང་རྒྱ་དང་རྫུན་བཟོ་བྱེད་ཀྱི་ཡོད་པ་རེད། ཅེས་འཁོད་འདུག   གནས་ཚུལ་དེའི་ནང་། ཐེན་ཧྲན་དྲ་ཚིགས་སྟེང་རྒྱ་ནག་རྒྱལ་སྲུང་དམག་དཔུང་གིས་ཁ་ཆེ་འཇིགས་སྐུལ་པ་ ༡༡ དང་ཉེན་རྟོག་པ་ ༥ བཅས་བསད་རྗེས། དྲག་སྤྱོད་པ་དེ་དག་སྦས་སྐོང་བྱེད་སའི་རྡོ་སོལ་གཏེར་ཁུངས་དུ་སོང་ཡོད་ལུགས་བཀོད་འདུག་ཀྱང་། དོན་དངོས་སུ་རྒྱ་ནག་དྲག་ཆས་དམག་དཔུང་གིས་རྡོ་སོལ་གཏེར་སྒོ་དེ་འབར་རྫས་ཀྱིས་བཀལ་གཏོར་བཏང་ཡོད་པ་དང་། རྡོ་སོལ་གཏེར་སྒོ་དེ་ཡང་རྒྱ་ནག་ཧན་མི་རིགས་ཀྱི་གཏེར་འདོན་ཚན་པས་སྟངས་འཛིན་བྱས་ཡོད་པ་ཞིག་ཡིན་ཞིང་། གནས་ཚུལ་དེ་ཨ་རིའི་གཞུང་གིས་ཤེས་རྟོགས་བྱུང་སྟེ། ཁ་པར་སྐུལ་ལམ་ནས་ས་གནས་དཔོན་རིགས་ཁག་ཅིག་ལ་འབྲེལ་གཏུག་བྱས་ཀྱང་། ཁོ་ཚོས་དོན་རྐྱེན་དེའི་དངོས་ཡོད་གནས་ཚུལ་མཁོ་སྤྲོད་བྱས་མེད་པ་མ་ཟད་ཁ་པར་ལ་བརྡ་ལན་ཡག་པོ་ཞིག་ཀྱང་སྤྲད་མེད་སྐོར་བཀོད་འདུག   གཞན། ནེའུ་ཡོག་དུས་བབ་སོགས་བརྒྱུད་ལམ་གཞན་ཁག་ཅིག་ནང་། ཐེངས་འདིའི་དོན་རྐྱེན་ཁྲོད་ཡུལ་མི་ ༡༧ ནས་ ༥༠ བར་ཤི་རྨས་བྱུང་བའི་ཁྲོད་བུད་མེད་དང་བྱིས་པའང་ཡོད་པ་དང་། ཤིན་ཅང་ཡུ་གུར་གྲངས་ཉུང་མི་རིགས་ས་ཁུལ་ཁག་ཏུ་འབྱུང་བཞིན་པའི་གནད་དོན་གྱི་འབྱུང་རྐྱེན་གཙོ་བོ་དེ་ས་གནས་སྲིད་གཞུང་གི་དྲག་ཆས་ཉེན་རྟོག་པས་ནམ་རྒྱུན་ཡུ་གུར་བ་ཚོར་ཚད་ལས་བརྒལ་བའི་དྲག་གནོན་བྱེད་ཀྱི་ཡོད་པས་རེད་འདུག ཡིན་ནའང་། རྒྱ་གཞུང་གིས་ད་དུང་རྒྱལ་ནང་གི་གྲངས་ཉུང་མི་རིགས་ཁག་ལ་ལྷག་བསམ་རྣམ་དག་གི་སྒོ་ནས་བདེ་འཇགས་འགན་སྲུང་དང་འཚོ་གནས་ཡར་རྒྱས་གཏོང་གི་ཡོད་ལུགས་ཤོད་ཀྱི་ཡོད་པ་དང་། ཉེ་ལམ་ཧྥ་རན་སིའི་རྒྱལ་ས་ཕ་རེ་སིའི་ནང་འཇིགས་སྐུལ་བས་མི་གསོད་ཁྲག་སྦྱོར་གྱི་དོན་རྐྱེན་བསླངས་པ་དང་བསྟུན་ཏེ། རྒྱ་ནག་མི་དམངས་བདེ་འཇགས་ལྷན་ཁང་ཟེར་བས་ཐེན་ཧྲན་དྲ་ཚིགས་སྟེང་བསྒྲགས་གཏམ་སྤེལ་ཏེ། ད་རེབ་ཀྱི་རྡོ་སོལ་གཏེར་ཁུངས་ཀྱི་དོན་རྐྱེན་དེ་ཡང་ཁ་ཆེ་འཇིགས་སྐུལ་དྲག་སྤྱོད་པར་ཏན་ཏན་འབྲེལ་བ་ཡོད་ལུགས་བརྗོད་དེ། དོན་རྐྱེན་དེའི་ཁུངས་ལྡན་གནས་ཚུལ་འགེབས་སྲུང་བྱེད་ཡོད་པ་རེད། ཅེས་སོགས་བཀོད་འདུག   གསར་འགོད་པ། པདྨ་རྒྱལ།
