ཀླུ་སྒྲུབ་བསྟན་དར་ལཊ་ཀློད་གྲོལ།

དེ་རིང་སྔ་དྲོར་ཨ་མཆོག་མཚན་ཉིད་དགོན་པའི་སྐྱོར་དཔོན་ཀླུ་སྒྲུབ་བསྟན་དར་ལགས་མེན་ཡཱང་བཙོན་ཁང་ནས་ཀློད་གྲོལ་བཏང་ཡོད་འདུག འབྲེལ་ཡོད་ནས་འདི་གར་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། ཕྱི་ལོ་ ༢༠༠༨ ལོར་རྒྱ་གཞུང་གི་དྲག་གནོན་ལ་ངོ་རྒོལ་སྐད་འབོད་གནང་བའི་ཉེས་མིང་འོག ལོ་ ༧ གྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་པའི་ཨ་མཆོག་མཚན་ཉིད་དགོན་པའི་སྐྱོར་དཔོན་ཀླུ་སྒྲུབ་བསྟན་དར་ལགས། དེ་རིང་ཕྱི་ཚེས་ ༡ ཉིན་གྱི་སྔ་དྲོར་མེན་ཡཱང་བཙོན་ཁང་ནས་ཀློད་གྲོལ་བཏང་ཡོད་པ་དང་། ཁོང་གི་འཛིན་གྲྭ་གཅིག་པ་སོགས་མེན་ཡཱང་དུ་བསུ་རུ་སོང་ཡོད་པ་དང་། མི་འགྱངས་བར་ཕྱིར་ཕ་ཡུལ་དུ་འབྱོར་རྒྱུ་ཡིན་འདུག ཡིན་ནའང་། ད་ལྟའི་ཆར་ཁོང་གི་སྐུ་གཟུགས་གནས་སྟངས་སོགས་ཞིབ་ཕྲའི་གནས་ཚུལ་ཤེས་རྟོགས་ཐུབ་ཀྱི་མེད་པ་རེད། ཕྱི་ལོ་ ༢༠༡༤ ལོར་རྒྱ་གར་ནང་བཙན་བྱོལ་དུ་འབྱོར་བའི་ཀླུ་སྒྲུབ་བསྟན་དར་ལགས་ཀྱི་བཙོན་རོགས། རྔ་པ་སྤྲོ་ཚིགས་དགོན་པའི་གྲྭ་འཇམ་དབྱངས་བརྩོན་འགྲུས་ལགས་ཀྱིས་དེ་སྔོན་འདི་གའི་གསར་འགོད་པར་འགྲེལ་བརྗོད་གནང་བ་ལྟར། ཕྱི་ལོ་ ༢༠༡༠ ཟླ་ ༣ ཚེས་ ༣༡ ཉིན་ཁོང་མེན་ཡཱང་བཙོན་ཁང་ནས་ཀློད་གྲོལ་གཏོང་སྐབས། བཙོན་ཁང་དེའི་ནང་ཁོང་གིས་ངོ་ཤེས་པའི་བོད་རིགས་ཆབ་སྲིད་བཙོན་པ་ཁྱོན་ ༣༩ ཡོད་པ་དང་། ཁོང་ཚོའི་ནང་ནས་རྔ་སྨད་ལོ་རྟའི་སྐར་མ་ཚེ་བཙོན་དང་། ཨ་མཆོག་མཚན་ཉིད་དགོན་པའི་སྐྱོར་དཔོན་ཚུལ་ཁྲིམས་རྒྱ་མཚོར་ལོ་ ༩ ། སྐྱོར་དཔོན་ཀླུ་སྒྲུབ་ཕུན་ཚོགས་ལ་ལོ་ ༡༣། ཀླུ་སྒྲུབ་ཡེ་ཤེས་ལ་ལོ་ ༡༣། ལི་ཐང་རོང་རྒྱས་ཨ་གྲགས་ལ་ལོ་ ༨། ཨ་ཁུ་བྷོར་ལོ་ ༡༠། ཀུན་མཁྱེན་ལ་ལོ་ ༩། ཀིརྟི་དགོན་གྱི་གྲྭ་འཕྲིན་ལས་ཚེ་རིང་ལ་ལོ་ ༩། ཚེ་ཁོར་ལོ་ ༡༣། གཏེར་མཛོད་ལ་ལོ་ ༡༥། རྔ་མ་འ་ཁམས་ཀྱི་དཔལ་རྒྱས་ལ་ལོ་ ༡༡། འཕགས་པར་ལོ་ ༡༢། འོད་ལྡན་ལ་ལོ་ ༩། རིན་ཆེན་ལ་ལོ་ ༡༡།རྟོགས་ལྡན་དགོན་པའི་གྲྭ་སེང་གྷར་ལོ་ ༨། འཕྲིན་ཁོར་ལོ་ ༧། སྨེ་བ་ཞང་གི་ཀུན་རྡོར་ལོ་ ༨ བཅས་ཀྱི་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག ཡང་གྲྭ་འཇམ་དབྱངས་བརྩོན་འགྲུས་ལགས་ཀྱིས། ཕྱི་ལོ་ ༢༠༠༨ ལོའི་ཟླ་ ༣ ཚེས་ ༡༦ ཉིན་རྔ་པ་རྫོང་དུ་བོད་མི་ཚོས་རྒྱ་གཞུང་ལ་ངོ་རྒོལ་གནང་སྐབས། ཁོང་དང་སྤྲོ་ཚིགས་དགོན་པའི་གྲྭ་པ་ ༡༨ ཙམ་གྱིས་མི་མང་དང་མཉམ་དུ་རྔ་པ་རྫོང་གི་སྲང་ལམ་དུ། ༧གོང་ས་མཆོག་བོད་དུ་ཕེབས་དགོས། བོད་མིར་རང་དབང་དང་ཞི་བདེ་དགོས་ཞེས་སྐད་འབོད་རྒྱག་བཞིན་ཁྲོམ་སྐོར་གནང་སྐབས། རྒྱ་གཞུང་ཉེན་རྟོག་པ་འབྱོར་ཏེ་མང་ཚོགས་ཁྲོད་གང་བྱུང་མེ་མདའ་བརྒྱབ་སྟེ། ཁོང་གིས་མཐོང་ས་ནས་མི་སྐྱ་ཕོ་མོ་ ༡༣ ཙམ་དམར་གསོད་དང་། གཞན་མང་པོ་ཞིག་ལ་རྨས་སྐྱོན་ཕོག་ཡོད་སྐོར་བརྗོད་སོང་། གསར་འགོད་པ། པདྨ་མཚོ།
