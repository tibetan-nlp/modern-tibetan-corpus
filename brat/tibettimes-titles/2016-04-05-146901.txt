ཆོས་དབྱིངས་སྐྱིད་ལ་དངུལ་གྱི་རྟཊ་མ་ཐོབ།

བོད་རིགས་ལུས་རྩལ་བ་ཆོས་དབྱིངས་སྐྱིད་ལགས་ལོན་ཏོན་ཨོ་རྩལ་འགྲན་ཚོགས་ཀྱི་འདུར་རྒྱུག་ཁྲོད་ཨང་གཉིས་པ་ཆགས་ཏེ་དངུལ་གྱི་རྟགས་མ་ཐོབ་ཡོད་འདུག དེ་རིང་ཀྲུང་གོའི་བོད་སྐད་རླུང་འཕྲིན་དྲ་བ་ཟེར་བའི་ཐོག་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཉེ་སྔོན་རྒྱལ་སྤྱིའི་ལུས་རྩལ་འདུམ་གཅོད་ཁྲིམས་ཁང་གིས། ཨུ་རུ་སུའི་ལུས་རྩལ་པ་འགའ་ལུས་རྩལ་འགྲན་བསྡུར་གྱི་སྒྲིག་ཁྲིམས་དང་འགལ་ཏེ་ཁྲིམས་ཆད་ཕོག་ཡོད་པ་ཁྱབ་བསྒྲགས་བྱས་ཡོད་པ་དང་།  དེའི་ཁྲོད་ཕྱི་ལོ་ ༢༠༡༢ ལོའི་ལོན་ཏོན་ཨོ་རྩལ་འགྲན་ཚོགས་ཀྱི་སྐྱེས་མའི་སྤྱི་ལེ་ ༢༠ ཅན་གྱི་འདུར་རྒྱུག་ཁྲོད་ཨང་གཉིས་པ་ལེན་མཁན་ཨུ་རུ་སུའི་ལུས་རྩལ་བ་ཁཱ་ནེའུ་ཧྲིན་ཅིན་ནཱ་ཡིས། འགྲན་བསྡུར་སྐབས་ལུས་ངལ་གསོ་བའི་སྨན་བཏུང་ཡོད་པ་ར་འཕྲོད་བྱུང་སྟེ་དངུལ་གྱི་རྟགས་མ་ཕྱིར་བསྡུས་ཡོད་སྟབས། བོད་ཀྱི་འདུར་རྒྱུག་ལུས་རྩལ་བ་ཆོས་དབྱིངས་སྐྱིད་ལགས་ཨང་གཉིས་པར་རིམ་གསབ་བྱས་ཡོད་པ་དང་། ལུས་རྩལ་བ་ལིའུ་ཧོང་ཨང་རིམ་གསུམ་པར་གསབ་སྟེ་ཟངས་ཀྱི་རྟགས་མ་ཐོབ་ཡོད་འདུག བོད་རིགས་ལུས་རྩལ་བ་ཆོས་དབྱིངས་སྐྱིད་ལགས་འདི་ལོའི་ཕྱི་ཟླ་ ༣ ཚེས་ ༦ ཉིན་མཚོ་སྔོན་ཞིང་ཆེན་གྱི་ཚབ་བྱས་ཏེ། ཨན་ཧེ་ཞིང་ཆེན་དུ་སྤེལ་བའི་ ༢༠༡༦ ལོའི་རྒྱལ་ཡོངས་འདུར་རྒྱུག་བྱ་དགའ་ཆེན་མོའི་འགྲན་བསྡུར་དང་ཨོ་རྩལ་འགྲན་ཚོགས་ལེགས་འདེམ་འགྲན་བསྡུར་ནང་ཞུགས་ཡོད་འདུག་ལ། ཁོང་མོ་ཕྱི་ལོ་ ༢༠༡༠ ལོའི་ཟླ་ ༧ པར་རྒྱལ་ཁབ་མཆོང་རྒྱུག་འཕེན་གསུམ་རུ་ཁག་ཏུ་ཞུགས་ཤིང་། ཕྱི་ལོ་ ༢༠༡༥ ལོའི་སི་པིན་རྒྱལ་ཁབ་ཀྱི་ལ་ཁི་ལུའུ་ནི་ཡ་རྒྱལ་སྤྱིའི་མཆོང་རྒྱུག་འཕེན་གསུམ་འགྲན་བསྡུར་གྱི་སྤྱི་ལེ་ ༢༠ ཅན་གྱི་བུད་མེད་འདུར་རྩལ་འགྲན་བསྡུར་ཁྲོད་ཨང་གཉིས་པ་དང་། ཀྲུང་གོའི་རྒྱལ་ཡོངས་རྟིང་རྩོད་འདུར་རྒྱུག་འགྲན་བསྡུར་གྱི་སྤྱི་ལེ་ ༢༠ ཅན་གྱི་སྐྱེད་མའི་འདུར་རྒྱུག་འགྲན་སྡུར་ཁྲོད་ཨང་དང་པོ་ཐོབ་ཡོད་པ་མ་ཟད། ཕྱི་ལོ་ ༢༠༡༢ ལོར་ལོན་ཏོན་དུ་ཚོགས་པའི་ཨོ་རྩལ་འགྲན་ཚོགས་སྐབས་ ༣༠ པའི་སྤྱི་ལེ་ ༢༠ ཅན་གྱི་སྐྱེས་མའི་འདུར་རྒྱུག་འགྲན་བསྡུར་ཁྲོད་ཨང་གཉིས་པ་ཆགས་ཏེ་དངུལ་གྱི་རྟགས་མ་ཐོབ་ཡོད་སྟབས། ཨེ་ཤ་ཡའི་འདུར་རྒྱུག་ལུས་རྩལ་གྱི་ལོ་རྒྱུས་ནང་ཟིན་ཐོ་གསར་པ་ཞིག་བསྐྲུན་ཡོད་པ་མ་ཟད། ཨོ་རྩལ་འགྲན་ཚོགས་ཁྲོད་བྱ་དགའ་ལེན་མཁན་གྱི་བོད་རིགས་ཐོག་མ་དེ་ཆགས་ཡོད་པ་རེད། ཆོས་དབྱིངས་སྐྱིད་ལགས་ཀྱི་གཟེངས་སུ་ཐོན་པའི་གྲུབ་འབྲས་ཀྱིས་བོད་ཁུལ་གྱི་ན་གཞོན་མང་པོར་ལུས་རྩལ་ཐད་རྨི་ལམ་གསར་པ་ཞིག་བསྐྲུན་ཡོད་པ་ལྟར། ཕྱི་ལོ་ ༢༠༡༣ ལོའི་རྒྱ་ནག་རྒྱལ་ཡོངས་སྐབས་བཅུ་གཉིས་པའི་ལུས་རྩལ་འགྲན་ཚོགས་ཐོག མཚོ་བྱང་ཁུལ་མདའ་བཞི་རྫོང་ནས་ཡིན་པའི་བོད་རིགས་ལུས་རྩལ་བ་མཚོ་མོ་སྐྱིད་ལ་གསེར་གྱི་རྟགས་མ་ཐོབ་ཡོད་པ་དང་། གཞན་ཡང་། ཕྱི་ལོ་ ༢༠༡༥ ལོར་ཁོ་ཊར(Qatar) གྱི་རྒྱལ་ས་ཌོ་ཧཱ་རུ་ཚོགས་པའི་ཨེ་ཤ་ཡའི་ན་གཞོན་མཆོང་རྒྱུག་འགྲན་ཚོགས་ཐོག རྨིད་ ༥༠༠༠ ཅན་གྱི་སྐྱེད་མའི་རྒྱུག་འགྲན་ནང་མགོ་ལོག་བོད་རིགས་རང་སྐྱོང་ཁུལ་རྨ་ཆེན་རྫོང་རྟ་བོ་གྲོང་རྡལ་ནས་ཡིན་པའི་བོད་རིགས་བུ་མོ་ཆོས་སྐྱིད་སྒྲོལ་མ་ལགས་ལ་ཨང་གཉིས་པའི་རྒྱལ་ཁ་ཐོབ་ཡོད་པ་རེད། གསར་འགོད་པ། པདྨ་མཚོ།
