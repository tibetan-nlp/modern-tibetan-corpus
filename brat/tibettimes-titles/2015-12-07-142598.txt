རྟོག་ཞིབ་ལས་བྱེད་ཚོར་ཉེས་རྡུང་བཏང་འདུག

  ལྷོ་སོག་པོའི་ཨི་ཇེ་ནཱ་ཇི་ས་ཁུལ་དུ་ཡོད་པའི་རྒྱ་གཞུང་རྟོག་ཞིབ་ས་ཚུགས་ལ་ཁ་སང་དགོང་མོར་མགོ་རས་དཀྲིས་པའི་མི་མང་པོ་ཞིག་གིས་འཇབ་རྒོལ་བྱས་ཏེ་ས་ཚིགས་དེའི་ལས་བྱེད་ ༡༣ ཙམ་ལ་རྨས་སྐྱོན་བཟོས་ཡོད་འདུག   དེ་རིང་རྒྱ་ནག་གི་བརྒྱུད་ལམ་ཁག་ཅིག་ནང་གནས་ཚུལ་ཐོན་པར། ཁ་སང་ཐོ་རིངས་ཆུ་ཚོད་ ༣།༣༠ ཡས་མས་ལ། རྒྱ་ནག་གིས་ནང་སོག་ཏུ་འབོད་པའི་ཨི་ཇེ་ནཱ་ཇི་(Ejinagj)ས་ཁུལ་གྱི་ཁྲིམས་འཛིན་རྟོག་ཞིབ་ས་ཚུགས་སུ། གདོང་རས་གྱོན་པའི་མི་ ༡༠༠ ལྷག་གིས་ལ་འཇབ་རྒོལ་བྱས་ཏེ། ས་ཚིགས་དེའི་ལས་བྱེད་ཚོར་རྡུང་རྡེག་ཚབས་ཆེན་བཏང་པའི་ཁྲོད་མི་ ༦ ལ་རྨས་སྐྱོན་ལྕི་མོ་ཕོག་ནས་སྨན་ཁང་དུ་མྱུར་སྐྱོབ་བྱེད་བཞིན་ཡོད་འདུག   ཨི་ཇེ་ནཱ་ཇི་ས་ཁུལ་གྱི་ས་གནས་སྲིད་གཞུང་གི་མགོ་གཙོ་གཞོན་པ་ཁྲིན་ཐེ་ཅོན་(Chen Tiejun) གྱིས་གསར་འགོད་པར། མགོ་རས་དཀྲིས་པའི་ཉེས་ཅན་དེ་དག་གིས་ཁྲིམས་འཛིན་ཁང་གི་རྟོག་ཞིབ་གློག་ཆས་དང་འབྲེལ་ལམ་ཁག་བཅད་རྗེས་འཇབ་རྒོལ་བྱས་ཡོད་པ་དང་། ལས་བྱེད་ཚོའི་གདོང་ལ་སི་པན་ཁུ་བ་གཏོར་ནས་མགོ་ལ་ཁུག་སྐོན་པ་དང་། དབྱུག་པ་སོགས་ཀྱིས་རྡུང་རྡེག་ཚད་མེད་བཏང་རྗེས་ལག་པ་ཐག་པས་བསྡམས་ཏེ་ཟུར་གྱི་བྱེ་ཐང་དུ་གཡུག་འདུག ད་དུང་ལས་ཁུངས་ཀྱི་བེད་སྤྱོད་ཡོད་པའི་ཅ་དངོས་དང་། ལས་བྱེད་པའི་ཁ་པར་དང་གྱོན་པ་སོགས་ཁྱེར་འདུག ཅེས་བརྗོད་འདུག   རྒྱ་ནག་གི་གོ་ལའི་དུས་བབ་ཀྱིས་ཀྱང་། གོང་གི་གནས་ཚུལ་དེ་དང་འབྲེལ་ཏེ། ལོ་འདིའི་ཕྱི་ཟླ་ ༩ བའི་ནང་དུའང་གདོང་རས་གྱོན་པའི་མི་ ༤༠ ལྷག་གིས་རླངས་འཁོར་བཏང་ཡོང་སྟེ་ས་ཚུགས་དེའི་ལས་བྱེད་ཚོར་འཇིགས་སྐུལ་བྱས་ཡོད་སྐོར་བཀོད་འདུག ཐེངས་འདིར་ལས་བྱེད་ ༡༣ ལ་རྨས་སྐྱོན་ཕོག་པའི་ཁྲོད་མི་ ༦ ལ་རྨས་སྐྱོན་ལྕི་མོ་ཕོག་ཡོད་འདུག་ལ། ཅ་དངོས་བཅག་སྐྱོན་དང་རྐུ་འཕྲོག་བྱས་པའི་གྱོང་གུད་རྒྱ་སྒོར་ས་ཡ་ ༡༠༠༠ ལྷག་ཡིན་སྐོར་སོགས་བཀོད་འདུག   གནས་ཚུལ་དེ་བྱུང་རྗེས་ས་གནས་རྒྱ་གཞུང་ཉེན་རྟོག་པས་ཉེས་ཅན་དག་གང་མགྱོགས་འཛིན་བཟུང་བྱ་རྒྱུ་ཡིན་སྐོར་ཁྱབ་བསྒྲགས་བྱས་ཡོད་ཀྱང་། སྤྱི་ཚོགས་བརྒྱུད་ལམ་ཁག་ཏུ། ཉེན་རྟོག་པས་ཉེས་ཅན་འཛིན་བཟུང་བྱ་རྒྱུ་ཁྱབ་བསྒྲགས་བྱས་ཡོད་ཀྱང་། རྒྱུ་མཚན་ཅིའི་ཕྱིར་འཇབ་རྒོལ་བྱས་པའི་གནས་ཚུལ་གསལ་བཤད་བྱས་མེད་པ་དེར་གནད་དོན་ཆེན་པོ་ཡོད་སྐོར་བརྗོད་འདུག གནས་ཚུལ་དེའི་ཐད་བྷེ་བྷེ་སི་རྒྱ་ཡིག་སྡེ་ཚན་གྱིས་ཉེ་སྔོན་ནང་སོག་མི་རིགས་དང་རྒྱ་མི་རིགས་བར་ལ་གནས་སྟངས་ཛ་དྲག་ཆེ་བའི་སྣང་ཚུལ་ཞིག་བྱུང་ཡོད་པ་རེད། ཅེས་བཀོད་འདུག   གསར་འགོད་པ། འཇིགས་མེད་རྒྱ་མཚོ།
