རྒྱལ་ཕམ་ཐག་གཅོད་ཀྱི་འོས་བསྡུ།

ཁ་སང་རྒྱ་གར་གྱི་མངའ་སྡེ་ ༡༢ ནང་གྲོས་ཚོགས་འོག་མའི་འཐུས་མི་ ༡༢༡ ཙམ་འོས་འདེམས་བྱས་ཏེ་གཞུང་འཛིན་ཚོགས་པ་སུ་ཆགས་ལ་རྒྱལ་ཕམ་གྱི་འགྲན་རྩོད་ཅིག་བྱས་ཡོད་འདུག རྒྱ་གར་རྒྱལ་ཡོངས་གྲོས་ཚོགས་འོག་མའི་འོས་བསྡུའི་ལས་རིམ་དེ་ས་ཁུལ་ཁག་དགུ་ལ་དབྱེ་ཡོད་པ་ལས། ཁ་སང་ཚེས་ ༡༧ ཉིན་ས་ཁུལ་རིམ་པ་ལྔ་པ་ཀཱ་ཎཱ་ཀྲ་ཁཱ་དང་ར་ཇེ་སི་ཐན། མཱཧ་ར་ཤི་ཌ། ཡུ་པི་སོགས་མངའ་སྡེ་ ༡༢ ནང་མགོ་འཛུགས་བྱས་ཏེ། གྲོས་ཚོགས་འཐུས་མི་ ༡༢༡ ཙམ་འོས་འདེམས་བྱས་ཡོད་ཅིང་། ད་རེབ་ནི་གྲོས་ཚོགས་འཐུས་མི་མང་ཤོས་ཡོད་པའི་རྒྱལ་ཕམ་དབྱེ་འབྱེད་ཀྱི་འོས་བསྡུ་དེ་ཆགས་ཡོད་འདུག འོས་བསྡུའི་ཁྲོད་ཀཱ་ཎཱ་ཀྲ་ཁཱ་མངའ་སྡེའི་འཐུས་མི་ ༢༨ དང་། ར་ཇེ་སི་ཐན་གྱི་འཐུས་མི་ ༢༠། མཱཧ་ར་ཤི་ཌ་ཡི་འཐུས་མི་ ༡༩། ཡུ་པི་ཡི་འཐུས་མི་ ༡༡། ཨོ་རི་སའི་འཐུས་མི་ ༡༡། མཱ་དྷེཡ་པར་དྷི་ཤི་ཡི་འཐུས་མི་ ༡༠། བྷེ་ཧར་གྱི་འཐུས་མི་ ༧། ཇར་ཁནཌི་ཡི་འཐུས་མི་ ༦། ནུབ་བྷང་གྷལ་གྱི་འཐུས་མི་ ༤། ཆཱ་དྷིསེ་གྷར་གྱི་འཐུས་མི་ ༣། ཇམ་མུ་དང་ཀཱ་ཤི་མེར་གྱི་འཐུས་མི་ ༡། མ་ནི་པུར་གྱི་གྲོས་ཚོགས་འཐུས་མི་ ༡ བཅས་འོས་འདེམས་བྱས་ཡོད་འདུག ཀཱ་ཎཱ་ཀྲ་ཁཱའི་ནང་འོས་ཤོག་ལྡུག་མཁན་ས་ཡ་ ༤༦།༢ ཡོད་ཅིང་། དེ་དག་གིས་འོས་མི་ ༤༣༥ ལ་འོས་ཤོག་ཕང་སྟེ་འོས་འདེམས་བྱེད་བཞིན་ཡོད་འདུག་ལ། ཀོང་རེ་སི་ཚོགས་པའི་སྙན་གྲགས་ཆེ་བའི་འོས་མི་ཁག་ནི་མངའ་སྡེའི་སྤྱི་ཁྱབ་བློན་ཆེན་སིདྷ་རམ་ (Siddaramaiah) ལགས་དང་། དབུས་གཞུང་གི་བློན་ཆེན་ཝུ་ར་པཱ་ (Veerappa Moily) ལགས། རྒྱ་གར་ཨེན་ཧྥོ་སི་ ( Infosy) ཁེ་ལས་ཀྱི་སྤྱི་ཁྱབ་གཞོན་པ་ཟུར་པ་ཎན་ཌིན་ (Nandan Nilekani) ལགས། བྷེ་ཇཱའེ་པི་ཚོགས་པའི་འོས་མི་ནི་བྷེ་ཨསེ་ (BS Yeddyurappa) ལགས་དང་། ཤེར་ཨ་མུ་ལུ་ (B Sreeramulu) ལགས་བཅས་གཉིས་ཡིན་འདུག ར་ཇེ་སི་ཐན་ནང་འོས་ཤོག་བླུག་མཁན་ས་ཡ་ ༣༨ ཡོད་ཅིང་། དེ་དག་གིས་ཆབ་སྲིད་ཚོགས་པ་འདྲ་མིན་གྱི་འོས་མི་ ༢༣༩ ལ་འོས་ཤོག་བླུག་སྟེ་གྲོས་ཚོགས་འཐུས་མི་འོས་འདེམས་བྱེད་བཞིན་ཡོད་པ་དང་། བྷེ་ཇཱའེ་པི་ཚོགས་པའི་འོས་མི་གཙོ་བོ་དྷུ་ཤི་ཡན་ཌི་སིན་ (Dushyant Singh) ལགས་དང་། ཇོ་ཤི་(C P Joshi)། ར་ཐོར་ (Rajyavardhan Singh Rathore) ལགས། ཀོང་རེ་སི་ཚོགས་པའི་འོས་མི་རྒྱ་གར་དབུས་གཞུང་གི་བློན་ཆེན་སཱ་ཆིན་ (Sachin Pilot) ལགས་དང་། ཀུ་མར་རི་(Chandresh Kumari)། ཇེ་ཌན་ཌ་ར་ (Jitendra Singh) ལགས་བཅས་ཡིན་འདུག མཱཧ་ར་ཤི་ཌ་ནང་འོས་ཤོག་བླུག་མཁན་ས་ཡ་ ༣༢།༥ དང་། གྲོས་ཚོགས་འཐུས་མའི་འོས་མི་ ༣༥༨ ཡོད་འདུག་ལ། ཀོང་རེ་སི་ཚོགས་པའི་འོས་མི་གཙོ་གྲས་ནི་རྒྱ་གར་དབུས་གཞུང་གི་ནང་སྲིད་བློན་ཆེན་སུ་ཤེལ་ཀུ་མར་ (Sushilkumar Shinde) ལགས་དང་། མངའ་སྡེའི་བློན་ཆེན་ཟུར་པ་ཨ་ཤོག་ཆཱ་ཝན་ (Ashok Chavan) ལགས། བྷེ་ཇཱའེ་པི་ཚོགས་པའི་འོས་མི་གྷུ་པི་ཎ་ཌ་ (Gopinath Munde) ལགས་བཅས་ཡིན་འདུག ཡུ་པི་ནང་འོས་ཤོག་ལྡུག་མཁན་ས་ཡ་ ༡༨ དང་། གྲོས་ཚོགས་འཐུས་མིའི་འོས་མི་ ༡༥༡ ཡོད་འདུག་ལ། ཀོང་རེ་ཚོགས་པའི་འོས་མི་གཙོ་བོ་ར་ཧུལ་གྷན་དྷི་ལགས་དང་། པིར་ཡང་གྷ་གྷན་དྷི་ལགས། བྷེ་ཇཱའེ་པི་ཚོགས་པའི་འོས་མི་ནི་མ་ནེ་ཁཱ་གྷན་དྷི་ (Maneka Gandhi) ལགས་བཅས་ཡིན་འདུག བྷེ་ཧར་ནང་འོས་ཤོག་ལྡུག་མཁན་ས་ཡ་ ༣༥།༤༥ དང་། གྲོས་ཚོགས་འཐུས་མིའི་འོས་མི་ ༡༧༡ ཡོད་ཅིང་། བྷེ་ཇཱའེ་པི་འོས་མི་རྒྱ་གར་གྱི་ལྟད་མོ་འཁྲབ་མཁན་སིང་ཧཱ་ (Shatrughan Sinha) ལགས་དང་། སིན་ (RK Singh)། ཨ་ཤི་ཝ་ཎི་(Ashwini Choubey) ལགས། ཀོང་རེ་སི་ཚོགས་པའི་འོས་མི་གཙོ་བོ་ཨ་ཤི་ཤི་ (Ashish Ranjan Sinha) ལགས། ( RDJ) ཚོགས་པའི་འོས་མི་མངའ་སྡེའི་བློན་ཆེན་ཎི་དྷི་ཤི་ཀུ་མར་ལགས་བཅས་ཡིན་འདུག དེ་བཞིན་ཨོ་རི་ས་མངའ་སྡེའི་འོས་མི་ཁག་ནི་ཤེར་ཁན་ཌི་ (Srikant Jena) ལགས་དང་། ཨར་ཇུན་ (Arjun Sethi ) ལགས། ཇེ་པན་ཌ་(Jay Panda) ལགས། ཇར་ཁནཌི་མངའ་སྡེ་ནས་ཁེ་ར་མུན་ཌི་ (Karia Munda) ལགས་དང་། ཨ་མི་དྷབ་ (Amitabh Chowdhury) ལགས། ཇཱ་ཡན་ཌི་ (Jayant Sinha) ལགས། ཇཱ་སི་ཝན་ཌི་སིན་ཧཱ་ (Jaswant Sinha)ལགས། སུབ་བྷོ་ཌ་ (Subodh Kant Sahay) ལགས། ཆཱ་དྷིསེ་གྷར་མངའ་སྡེ་ནས་ཨ་ཇེ་ཌི་(Ajit Jogi) ལགས་དང་། ཨ་བྷེ་ཤིག་ (Abhishek ) ལགས། མཱ་ཧ་ས་ཥམིན་ཌི་(Mahasamund) ལགས། ར་ཇི་ཎ་ཌི་གྷོན་ (Rajnadgaon) ལགས། ནུབ་བྷང་གྷལ་མངའ་སྡེ་ནས་བུ་ཆུང་བྷོ་དྷི་ཡ་ (Baiching Bhutia) ལགས། ཇམ་མུ་དང་ཀཱ་ཤི་མེར་མངའ་སྡེ་ནས་གྷོ་ལམ་ (Gulam Nabi Azad) ལགས་དང་། ཇེན་ཌན་དྷར་ (Jintender Singh) ལགས་བཅས་ཡིན་འདུག ཆབ་སྲིད་དབྱེ་ཞིབ་པ་ཚོས། སྐབས་དྲུག་པའི་རྒྱ་གར་གཞུང་འཛིན་ཚོགས་པ་སུ་ཆགས་མིན་དེ་ཐེངས་འདིའི་འོས་བསྡུར་ཐུག་ཡོད་སྟབས། ཆབ་སྲིད་ཚོགས་པ་ཁག་གིས་ད་རེབ་ཀྱི་འོས་བསྡུའི་ཁྲོད་རྒྱལ་ཁ་འཐོབ་རྒྱུར་འབད་བརྩོན་ཆེན་པོ་བྱས་ཡོད་ཟེར། རྒྱ་གར་གྱི་ཆབ་སྲིད་དབྱེ་ཞིབ་མཁས་ཅན་སཱ་ཌི་ཤི་ (Satish Misra) ལགས་ཀྱིས། “ རྒྱ་གར་ཀོང་རེ་སི་ཚོགས་པས་སྲིད་གཞུང་བཟུང་སྐབས། རྒྱ་གར་རྒྱལ་ཡོངས་ནང་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་ཚབས་ཆེའི་དོན་རྐྱེན་བསྟུད་མར་ཐོན་ཏེ་མང་ཚོགས་ཀྱིས་ངོ་རྒོལ་མང་པོ་བྱུང་ཡོད་སྟབས། ཐེངས་འདིར་བྷེ་ཇཱའེ་པི་ཚོགས་པར་རྒྱལ་ཡོངས་འོས་བསྡུའི་རྒྱལ་ཁ་ཐོབ་རྒྱུ་རེད།’’ ཅེས་བརྗོད་འདུག དེ་རིང་གི་འོས་བསྡུའི་ནང་འོས་མི་ ༡༧༦༩ དང་། འོས་ཤོག་ལྡུག་མཁན་ས་ཡ་ ༡༩༥ ཙམ་བྱུང་ཡོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
