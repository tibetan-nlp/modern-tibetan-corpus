དོན་གྲུབ་དབང་ཆེན་ཀློད་གྲོལ་བཏང་འདུག

འཇིགས་བྲལ་གློག་བརྙན་ཕབ་སྐྲུན་པ་དོན་འགྲུབ་དབང་ཆེན་ལགས་རྒྱ་ནག་གི་བཙོན་འོག་ནས་ཀློད་གྲོལ་ཐོབ་སྟེ། དེ་རིང་ཉིན་གོང་བོད་ནང་གི་ཆུ་ཚོད་ ༣།༠༠ ཙམ་ལ་ཨ་མདོ་བ་ཡན་རྫོང་གི་རང་ཁྱིམ་དུ་འབྱོར་ཡོད་འདུག ད་ལྟ་རང་བོད་ནས་གནས་ཚུལ་འབྱོར་བ་ལྟར་ན། ཕྱི་ལོ་ ༢༠༠༨ ཟླ་ ༣ ཚེས་ ༢༦ ཉིན། རྒྱ་ནག་གི་ཉེན་རྟོག་པས་མཚོ་སྔོན་ཟི་ལིང་ནས་འཛིན་བཟུང་བྱས་པའི་འཇིགས་བྲལ་གློག་བརྙན་ཕབ་སྐྲུན་པ་དོན་འགྲུབ་དབང་ཆེན་ལགས་རྒྱ་ནག་གི་བཙོན་འོག་ནས་ལོ་ ༦ གི་ཁྲིམས་ཆད་དུས་ཡུན་རྫོགས་ཏེ། དེ་རིང་ཉིན་རྒྱབ་བོད་ནང་གི་ཆུ་ཚོད་ ༣།༠༠ ཙམ་ལ་མཚོ་སྔོན་བ་ཡན་རྫོང་ཁོ་ཚེ་སྡེ་བའི་རང་ཁྱིམ་དུ་འབྱོར་ཡོད་འདུག གནས་ཚུལ་ཁོ་སྤྲོད་པས་བརྗོད་པ་ལྟར་ན། ཉིན་འགའི་སྔོན་ལ་རྒྱ་ནག་གི་འབྲེལ་ཡོད་ཚན་པས་བརྡ་ལན་འབྱོར་ཏེ་ཚེས་ ༦ ཉིན་ཀན་ལྷོ་ཁུལ་བསང་ཆུ་རྫོང་ཐོག་ནས་ཀློད་གྲོལ་གཏོང་རྒྱུ་ཡིན་སྐོར་བརྗོད་དེ། ནང་མི་གཉེན་འབྲེལ་ཚོས་གྲ་སྒྲིག་བྱས་ཡོད་ཀྱང་། དེ་རིང་ཉིན་རྒྱབ་ཁོང་གློ་བུར་དུ་རང་ཁྱིམ་ལ་འབྱོར་ཡོད་འདུག ཁོང་གི་ད་ལྟའི་གཟུགས་གཞིའི་གནས་སྟངས་དང་བཙོན་ནས་ཇི་ལྟར་ཀློད་གྲོལ་བཏང་མིན་སོགས་ཀྱི་གནས་ཚུལ་གཞན་ད་སྔ་ཤེས་མ་སོང་ཟེར། ༢༠༠༨ ལོའི་རྒྱ་ནག་པེ་ཅིན་ཨོ་ལིམ་ཕིག་ལུས་རྩེད་འགྲན་འཚོགས་བསླེབ་ལ་ཉེ་སྐབས། དོན་གྲུབ་དབང་ཆེན་ལགས་དང་མགོ་ལོག་འཇིགས་མེད་རྣམ་གཉིས་ཀྱིས་བོད་ནང་གི་ས་ཁུལ་མང་པོར་གསང་སྟབས་ཀྱིས་སྐོར་སྐྱོད་བྱས་ཏེ། ས་གནས་མང་ཚོགས་ལ་པེ་ཅིན་ཨོ་རྩེད་ལ་འདོད་མོས་ཡོད་མེད་དང་བོད་ནང་གི་དངོས་ཡོད་གནས་སྟངས། ༧གོང་ས་མཆོག་དང་བྱེས་གྱར་བོད་མི་ཚོར་དད་པ་དམ་ཚིག་གང་འདྲ་ཡོད་མེད་བཅས་ཀྱི་སྐོར་ལ་གློག་བརྙན་ཕབ་སྐྲུན་བྱས་བར། ༢༠༠༨ ཟླ་ ༣ ཚེས་ ༢༦ ཉིན་རྒྱ་ནག་གཞུང་གི་ཉེན་རྟོག་པས་མཚོ་སྔོན་ཟི་ལིང་ནས་དོན་གྲུབ་དབང་ཆེན་ལགས་འཛིན་བཟུང་བྱས་ཏེ་ལོ་ ༦ གི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་དེ་ཟི་ལིང་གི་བཙོན་ཁང་ཞིག་ཏུ་ད་བར་བཙོན་འཇུག་མནར་གཅོད་མྱངས་ཡོད། མགོ་ལོག་འཇིགས་མེད་ལགས་སྔ་ཕྱི་ཐེངས་ ༣ ཙམ་ལ་རྒྱ་གཞུང་གིས་འཛིན་བཟུང་བྱས་ཏེ་རྡུང་རྡེག་མནར་གཅོད་མང་པོ་བཏང་ཡོད་ཅིང་། ༢༠༡༢ ཟླ་ ༩ ཚེས་ ༣༠ ཉིན་ཀན་ལྷོ་ཁུལ་བསང་ཆུ་རྫོང་བང་དཀར་ཐང་གི་ལྟ་སྲུང་ཁང་ཞིག་ནས་བྲོས་ཐར་བྱུང་སྟེ། འདས་པའི་ལོ་གཅིག་དང་ཕྱེད་ལྷག་རིང་བོད་ནང་གི་ས་གནས་མང་པོར་ཉེན་གཡོལ་བྲོས་བྱོལ་བྱས་མཐར་རྒྱ་གར་ནང་སྐྱབས་བཅོལ་དུ་འབྱོར་ཏེ། ཟླ་སྔོན་མའི་ཚེས་ ༢༨ ཉིན་ངྷ་རམ་ས་ལར་ཁོང་གི་གནས་ཚུལ་སྐོར་ཡོངས་ཁྱབ་གསར་འགོད་གསལ་བསྒྲགས་སྤེལ་ཡོད། དོན་གྲུབ་དབང་ཆེན་ལགས་ཀྱི་ཕ་མ་གཉིས་དང་བཟའ་ཟླ་བུ་ཕྲུག་སོགས་རྒྱ་གར་ནང་སྐྱབས་བཅོལ་དུ་ཡོང་ཡོད་ཅིང་། ཕ་མ་གཉིས་ད་ལྟ་བཞུགས་སྒར་ངྷ་རམ་ས་ལར་བཞུགས་མུས་ཡིན།  འབྲེལ་ཡོད་གནས་ཚུལ་ཁག  དང་པོ། གཉིས་པ། གསུམ་པ། བཞི་པ། ལྔ་པ།
