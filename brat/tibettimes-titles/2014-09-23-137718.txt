མི་ཁྲི་ ༦ ལྷག་སྐྱེལ་འདྲེན་བྱས།

རྒྱ་གཞུང་གིས་ལྷ་ས་ནས་གཞིས་ཀ་རྩེའི་མེ་འཁོར་ལྕགས་ལམ་དངོས་སུ་ཤར་གཏོང་བྱས་པའི་དུས་ཡུན་ཟླ་གཅིག་རིང་མི་ཁྲི་དྲུག་སྐྱེལ་འདྲེན་བྱས་ཡོད་ཟེར། རྒྱ་ནག་གཞུང་གིས་འདི་ལོའི་ཕྱི་ཟླ་ ༨ ཚེས་ ༡༥ ཉིན་མཚོ་བོད་ལྕགས་ལམ་གྱི་སྣུར་སྲིང་ས་ཐིག་དང་པོ། བོད་རང་སྐྱོང་ལྗོངས་ཀྱི་ལྷ་ས་གྲོང་ཁྱེར་ནས་གཞིས་ཀ་རྩེ་བར་མེ་འཁོར་ལྕགས་ལམ་ཤར་གཏོང་བྱས་ཡོད་པ་ལྟར། ཟླ་འདིའི་ཚེས་ ༡༥ བར་གྱི་དུས་ཡུན་ཟླ་གཅིག་རིང་ཡུལ་སྐོར་བ་དང་མི་མང་ ༦༧༠༠༠ ལྷག་སྐྱེལ་འདྲེན་བྱས་ཐུབ་པ་བྱུང་ཡོད་པ་མ་ཟད། ས་གནས་ཀྱི་མང་ཚོགས་དང་ཡུལ་སྐོར་བར་སྟབས་བདེ་གང་ལེགས་བསྐྲུན་ཡོད་ཚུལ་བསྒྲགས་ཡོད་འདུག སྤྱིའི་རིང་ཚད་ལ་སྤྱི་ལེ་ ༢༥༣ ཡོད་པའི་ལྷ་ས་ནས་གཞིས་རྩེ་བར་གྱི་མེ་འཁོར་ལྕགས་ལམ་དེར་ཤེལ་རོང་དང་ཆུ་ཤུར། སྙེ་མོ། རིན་སྤུངས་བཅས་ཁྱོན་ས་ཚིགས་ཁག་ ༡༤ བཀོད་སྒྲིག་བྱས་ཡོད་འདུག་ལ། རྒྱ་ནག་གཞུང་འབྲེལ་གྱི་གསལ་བསྒྲགས་ཤིག་གི་ནང་འདས་པའི་ཟླ་གཅིག་རིང་ལྷ་ས་མེ་འཁོར་ས་ཚུགས་ནས་གཞིས་རྩེར་མི་ ༣༣༠༠༠ སྐྱེལ་འདྲེན་དང་། གཞིས་རྩེ་མེ་འཁོར་ས་ཚུགས་ནས་ལྷ་སར་མི་ ༣༤༠༠༠ ལྷག་ཙམ་སྐྱེལ་འདྲེན། ས་ཚུགས་གཉིས་ཀས་ཉིན་རེར་ཆ་སྙོམས་མི་ ༢༣༠༠ ལྷག་སྐྱེལ་འདྲེན་བྱེད་ཐུབ་ཡོད་སྐོར་བསྒྲགས་ཡོད་འདུག རྒྱ་ནག་གཞུང་གིས་ཕྱི་ལོ་ ༢༠༠༦ ལོར་རྒྱ་མཚོའི་ངོས་ནས་མཐོ་ཚད་མུ་ཊར་ ༥༠༠༠ ལྷག་ཡོད་པའི་ན་གོར་མོ་ནས་ལྷ་ས་བར་ལྕགས་ལམ་ཤར་གཏོང་བྱས་ཡོད་པ་དང་། འདི་ལོའི་ཕྱི་ཟླ་ ༨ པའི་ནང་རྒྱ་མཚོའི་ངོས་ནས་མཐོ་ཚད་མུ་ཊར་ ༣༦༠༠ ནས་ ༤༠༠༠ ཡོད་པའི་ལྷ་ས་དང་གཞིས་རྩེ་བར་གྱི་ལྕགས་ལམ་ཤར་གཏོང་བྱས་ཡོད། འདི་ནི་རྒྱ་ནག་གཞུང་གི་ལོ་ལྔའི་འཆར་གཞི་བཅུ་གཉིས་པའི་གྲས་ཀྱི་བོད་རང་སྐྱོང་ལྗོངས་ཁོངས་ཀྱི་ལས་འཆར་གལ་ཆེ་ཤོས་ཤིག་ཆགས་ཡོད་འདུག་ལ། ༢༠༡༦ ལོར་རྒྱ་གཞུང་གིས་མཚོ་བོད་ལྕགས་ལམ་གྱི་སྣེ་གཉིས་སུ་གྱེས་ཏེ། སྣེ་གཅིག་བལ་ཡུལ་དང་སྣེ་གཞན་དེ་རྒྱ་གར་དང་འབྲུག་གི་ས་མཚམསས་བར་མཐུད་དེ། ༢༠༢༠ ལོར་ཁྱིམ་མཚེས་རྒྱལ་ཁབ་དེ་གཉིས་ཀྱི་ས་མཚམས་བར་མེ་འཁོར་ཤར་གཏོང་ཐུབ་པ་བྱ་འཆར་ཡོད་འདུག ཆབ་སྲིད་དབྱེ་ཞིབ་པ་ཚོས། རྒྱ་གཞུང་གིས་མ་འོངས་པར་ལྕགས་ལམ་རྒྱ་སྐྱེད་ཀྱིས་བལ་ཡུལ་དང་འབྲུག རྒྱ་གར་བཅས་ཀྱི་ས་མཚམས་བར་སླེབས་ཐབས་བྱེད་དོན། ས་ཁུལ་དེ་དག་བོད་དང་ས་འབྲེལ་ཆགས་ཡོད་པ་དང་། ཨེ་ཤ་ཡའི་གནད་འགག་གི་ས་གནས་ཆགས་ཡོད་སྟབས། ཐབས་ཇུས་བཀོད་སྒྲིག་རྒྱ་སྐྱེད་གཏོང་ཆེད་ཡིན་པའི་ཚོད་དཔག་བྱེད་བཞིན་ཡོད་འདུག གསར་འགོད་པ། དོན་གོ
