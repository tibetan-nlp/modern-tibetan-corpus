རྩོམ་པ་པོ་ཞིག་འཛིན་བཟུང་བྱས་འདུག

ཁ་སང་ཕྱི་དྲོའི་ཆུ་ཚོད་ ༤ པ་ཡས་མས་ཤིག་ལ་རྔ་པའི་འབྲོག་སྡེ་རྨེའུ་རུ་མའི་ཡུལ་ཚོའི་རུ་ཆེན་གསུམ་པའི་རྩོམ་པ་པོ་རྨེའུ་བཟོད་པ་ལགས་རླངས་འཁོར་ནང་རྨེའུ་རུ་མའི་ཡུལ་ཚོ་ནས་རྔ་པ་རྫོང་ཐོག་ཏུ་སྐྱོད་བཞིན་པའི་ལམ་བར་དུ། རྔ་པ་རྫོང་གི་བདེ་འཇགས་མི་སྣ་དང་ཉེན་རྟོག་མི་སྣ་མཉམ་རུབ་ཀྱིས་འཛིན་བཟུང་བཙན་ཁྲིད་བྱས་འདུག ཁོང་འཛིན་བཟུང་བྱེད་དགོས་པའི་རྒྱུ་མཚན་ཁ་གསལ་མེད་ཀྱང་། སྡེ་མི་ཚོ་དང་ཁོང་གི་ངོས་ཤེས་ཚོས། རྩོམ་པ་པོ་རྨེའུ་བཟོད་པ་ལགས་ཀྱིས་༼ རྨེའུ་བཟོད་པའི་འཕྲིན་སྟེགས།༽ཟེར་བའི་སྒེར་གྱི་ཟིན་བྲིས་ཤིག་གཉེར་བཞིན་ཡོད་པ་དང་། དེར་ཁོ་རང་དང་མི་གཞན་གྱིས་བྲིས་པའི་བོད་ཀྱི་རང་དབང་དང་རང་ལུས་མེར་བསྲེགས་སྐོར་གྱི་རྩོམ་ཡིག་ ༼རང་དབང་གི་ཆེད་དུ་ལག་པ་གནམ་ལ་ཕྱར་བ། ༽ཞེས་པ་དང་། ཉིན་ཤས་སྔོན་རང་ལུས་མེར་བསྲེགས་གནང་མཁན་རྒྱལ་གཅེས་དཔའ་མོ་ཚེ་པེ་སྐྱིད་ལགས་ཀྱི་སྐོར་ལ་རྒྱ་ཡིག་ཐོག་༼དཔའ་མོ།༽ཞེས་པ་དང་། ༼ ཁྱི་རྒན་ཚང་ད་འདུག་གི་མི་འཇུག་ནས། བཙན་འཛུལ་བྱེད་རྒྱུ་མང་མེད་ན་ཉུང་མེད་༽སོགས་བོད་དོན་དང་འབྲེལ་བའི་རྩོམ་ཡིག་མང་པོ་སྤེལ་ཡོད་སྟབས། རྒྱ་གཞུང་གིས་དེ་དག་ཆབ་སྲིད་དང་འགལ་བའི་ཉེས་འཛུགས་བྱས་པ་ཡིན་སྲིད་པའི་ཚོད་དཔག་བྱེད་བཞིན་ཡོད་འདུག རྒྱ་གཞུང་གིས་ ༢༠༠༨ ལོའི་ཞི་རྒོལ་རྗེས་སུ་རྔ་པ་ཁུལ་ལ་ཆབ་སྲིད་ཀྱི་སློབ་གསོ་༼དགུ་སྣོན་གསུམ་སློབ་གསོ་༽ཟེར་བ་སོགས་ནི་ཞིང་འབྲོག་དམངས་ཚོགས་དཀྱུས་མའི་བྱིས་པ་ལོ་བཅོ་བརྒྱད་ལ་མ་ཟིན་པ་རྣམས་རྒྱ་ནག་ཁྲིན་ཏུའུ་སོགས་ལ་བཙན་ཁྲིད་བྱས་ཏེ་རྒྱ་གཞུང་གི་ལངས་ཕྱོགས་དང་རྒྱ་ནག་གི་གོམས་གཤིས་སོགས་འཇགས་ཐབས་ཀྱི་ཆེད་གཉེར་གྱིས་སྲིད་བྱུས་དེའི་ཁུངས་སུ་བཟོད་པ་ལགས་ཀྱང་ཚུད་དེ། སློབ་གྲྭ་དེར་མི་འགྲོ་ཐབས་མེད་བྱུང་ཡང་རྗེས་སུ་ནུབ་བྱང་མི་རིགས་སློབ་ཆེན་ལ་དངོས་ཞུགས་སློབ་མའི་ཁུངས་སུ་ཞུགས་ཡོད་ཅིང་། གུང་སེང་སོགས་དང་བསྟུན་ཏེ་རང་གི་ཕ་ཡུལ་རྨེའུ་རུ་མའི་ཡུལ་ཚོའི་སྟེང་ནས་ཁོང་དང་ཁོང་གི་གྲོགས་པོ་ཁག་ཅིག་གིས་༼སྨྲ་དབང་མེད་པའི་འུ་ཅག༽ཟེར་བ་སོགས་བགྲོ་གླེང་ཚོགས་འདུ་མང་པོ་ཞིག་གོ་སྒྲིག་བྱེད་མྱོང་ཡོད་འདུག ཁོང་གི་ཕ་ཡུལ་ནི་རྔ་པའི་འབྲོག་སྡེ་རྨེའུ་རུ་མའི་ཡུལ་ཚོའི་རུ་ཆེན་གསུམ་པ་ཡིན་ཞིང་། ཁྱིམ་མིང་ལ་མགོན་ཁོ་ཚང་ཟེར་ཞིང་ཁོང་ད་ལྟ་རང་ལོ་ ༢༡ ཡིན་འདུག ཁོའི་ཕའི་མིང་ལ་མགོན་ཁོ་དང་མའི་མིང་ལ་དཔལ་ཁོ། སྲིང་མོ་གཅིག་བཅས་ཡོད་འདུག རྨེའུ་བཟོད་པ་ལགས་སྔོན་མ་རྔ་པ་ཀིརྟི་དགོན་པའི་གྲྭ་པ་བྱས་མྱོང་ཞིང་། ད་ལྟ་ལན་ཀྲུག་ནུབ་བྱང་མི་རིགས་སློབ་གྲྭ་ཆེན་མོའི་དངོས་ཞུགས་སྦྱོང་འཛིན་གྲྭ་གཉིས་པ་ཡིན་འདུག་ལ། ཁོང་གི་ཆེད་སྦྱོང་སློབ་ཚན་ནི་རྩོམ་རིག་སྐོར་ཡིན་འདུག ཁོང་གི་ཁྱིམ་ཚང་གི་འཚོ་བའི་གནས་སྟངས་སྐྱོ་བོ་ཡིན་སྟབས། སློབ་གྲྭའི་གུང་སེང་དང་བསྟུན་རྨེའུ་རུ་མའི་ཡུལ་ཚོའི་སྟེང་ནས་༼བོད་སྒྲ་བརྙན་གླིང་།༽ཟེར་བའི་འོད་སྡེར་བཙོང་ས་ཞིག་གཉེར་ནས་སློབ་ཡོན་འཚོལ་བཞིན་ཡོད་འདུག ཁོང་ནི་དུས་རྒྱུན་དཔེ་ཀློག་དང་རྩོམ་ཡིག་འབྲི་རྒྱུ། བོད་ཀྱི་རྩ་དོན་བགྲོ་གླེང་སོགས་ལ་དགའ་མོ་ཆེན་པོ་བྱེད་མཁན་ཞིག་ཡིན་པ་དང་། འདི་ལོའི་དབྱར་ཁར་འཕྲུལ་རྟའི་བརྡབ་སྐྱོན་འོག་རྐང་པ་བཅགས་ནས་ད་དུང་དྲག་སྐྱེད་བྱུང་མེད་པར་ལག་རྟེན་ལ་བརྟེན་ཏེ་འགྲོ་བཞིན་ཡོད་འདུག  གོང་གི་གནས་ཚུལ་དེ་དྷ་སར་གནས་སྡོད་རྨེའུ་དཀོན་རྒྱམ་ལགས་ཀྱིས་བཏང་འབྱོར་བྱུང་།
