ལོ་གཅིག་ལ་ཡོང་འབབ་སྒོར་ཁྲི་ ༢༡༢
རེད་ཟེར།

ཧྭ་ཞཱ་དངུལ་ཁང་གི་འགན་འཛིན་གཞོན་པ་ཝང་ཡོ་ཐེན་ཞིབ་བཤེར་མིང་ཐོའི་ནང་ཚུད་ཡོད་པའི་གནས་ཚུལ་དང་འབྲེལ་ཏེ། ད་ཆར་རྒྱ་ནག་གཞུང་གི་རྒོལ་རྡུང་ལས་འགུལ་དེ་དཔལ་འབྱོར་རུལ་སུངས་གཙང་སེལ་གྱི་ཕྱོགས་སུ་ཁ་བསྒྱུར་ཡོད་སྐོར་རྗོད་བཞིན་འདུག ཁ་སང་ཕྱི་དྲོར་ཧྭ་ཞཱ་དངུལ་ཁང་གིས་རྒྱ་ནག་པེ་ཅིན་གྲོང་ཁྱེར་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གིས་ཧྭ་ཞཱ་དངུལ་ཁང་ གི་འགན་འཛིན་གཞོན་པ་ཝང་ཡོ་ཐེན་ (Wang Yaoting) ཁྲིམས་འགལ་གྱོད་དོན་འོག་ཞིབ་བཤེར་ཐོ་འགོད་བྱས་ཡོད་སྐོར་དང་། ཝང་ཡོ་ཐེན་ལ་ ༢༠༡༤ ལོ་གཅིག་པུའི་ནང་ཡོང་འབབ་རྒྱ་སྒོར་ཁྲི་ ༢༡༢ ཡོད་སྐོར་སོགས་བརྗོད་འདུག གནས་ཚུལ་དེ་དང་འབྲེལ་ཏེ་རྒྱ་ནག་ཕྱི་ལོགས་ཀྱི་གསར་ལམ་ཁག་ཏུ། ཞི་ཅིན་ཕིན་དང་ཝང་ཆེ་ཧྲན་གཉིས་ཀྱིས་སྟག་ལ་འཐབ་པའི་ལས་འགུལ་ཁྲོད་ད་ལྟ་དཔལ་འབྱོར་རུལ་སུངས་པ་རྣམས་གཙང་སེལ་བྱེད་ཀྱིན་ཡོད་ཚུལ་བརྗོད་ཀྱིན་འདུག ཚད་ཅིན་ཉིན་རེ་གསར་ཤོག་གིས་ཧྭ་ཞཱ་དངུལ་ཁང་མི་ཞིག་གི་སྐད་ཆ་ལུང་འདྲེན་བྱས་པ་ལྟར་ན། མི་དེས་ཟླ་སྔོན་མའི་ཟླ་མཇུག་ལ་ཝང་ཡོ་ཐེན་ཞིབ་བཤེར་བྱེད་ཀྱིན་ཡོད་པ་དངོས་ཡོད་ཡིན་ཡང་རྒྱུ་རྐྱེན་ག་རེ་ཡིན་མིན་གསལ་བོ་མི་ཤེས་ཀྱང་ཚོད་དཔག་བྱས་ན་ལྐོག་ངན་བླངས་པར་འབྲེལ་ཡོད་པ་གསལ་བོ་རེད་ཅེས་སོགས་བརྗོད་འདུག དེ་རིང་ཀྲུང་གོར་ལྟ་བ་ཞེས་པའི་དྲ་བའི་སྟེང་། ཞི་ཅིན་ཕིན་གྱི་དམག་སྤྱི་ཝང་ཆི་ཧྲན་གྱིས་སྔོན་ཆད་རྒྱ་ནག་དབུས་དངུལ་ཁང་གི་འགན་འཛིན་གཞོན་པ་དང་། འཛུགས་སྐྲུན་དངུལ་ཁང་གི་འགན་འཛིན་སོགས་ཀྱི་འགན་ཁུར་ཡོད་སྟབས་དཔལ་འབྱོར་རུལ་སུངས་ལ་ཆ་རྒྱུས་ཡོད་པ་དང་། ཁོས་དཔལ་འབྱོར་རུལ་སུངས་བ་ཚོར་རྒོལ་རྡུང་གཏིང་ཇེ་ཟབ་ཏུ་གཏོང་གི་ཡོད་པ་མཚོན་ཐུབ། ལོ་སྔོན་མའི་ཟླ་ ༨ པའི་ནང་རྒྱ་ནག་འཕེལ་རྒྱས་དང་བཅོས་སྒྱུར་ཨུ་ཡོན་གྱི་དཔལ་འབྱོར་འགན་འཛིན་ཟུར་བ་ཀྲང་ཏོན་ཧྲིན་ (Zhang Dongshen) ཞིབ་བཤེར་བྱས་ཡོད་པ་དེས་སྲིད་གཞུང་གིས་གྲོང་ཁྱེར་གྱི་རུལ་སུངས་ལ་རྒོལ་རྡུང་མགོ་བཙུགས་ཡོད་པ་མཚོན་ཐུབ་ཅེས་སོགས་འཁོད་འདུག ད་ལོའི་ཕྱི་ཟླ་དང་པོའི་མཇུག་ཏུ། རྒྱ་ནག་གི་མིན་ཧྲིན་དངུལ་ཁང་གི་འགན་འཛིན་མའོ་ཞོ་ཧྥིན་དབུས་སྒྲིག་ཁྲིམས་ཞིབ་ བཤེར་ལྷན་ཁང་གིས་ཞིབ་བཤེར་དུ་ཁྲིད་སྐབས། རྒྱ་གི་བརྒྱུད་ལམ་ཁག་ཏུ་མའོ་ཞོ་ཧྥིན་གྱི་གྱོད་དོན་དེ་ལིན་ཅི་ཧྭ་དང་འབྲེལ་བ་ཡོད་པ་དང་། གནས་ཚུལ་སྤེལ་མཁན་ལཱོ་ཁྲང་ཕིན་གྱིས། མའོ་ཞོ་ཧྥིན་ནི་ལིན་ཅི་ཧཱ་ཡིས་ཕུགས་རྒྱང་རིང་བོར་བསམ་བློ་བཏང་ཡོད་པའི་མི་གལ་ཆེན་ཞིག་ཡིན་སྐོར་བརྗོད་འདུག ཟླ་སྔོན་མའི་ཚེས་ ༨ ཉིན་ཨ་རིའི་བྷོ་བྷེར་གསར་འགྱུར་དྲ་བ་ (Bloomberg News) ཞེས་པར་གནས་ཚུལ་བརྒྱུད་སྤྲོད་བྱེད་མཁན་ཞིག་གིས། རྒྱ་གཞུང་དབུས་དངུལ་ཁང་གི་འགན་འཛིན་ཏེད་ཞིང་ལོན་ (Dai Xianglong) ཞིབ་བཤེར་བྱས་པ་དེས། ཁོས་དབུས་དངུལ་ཁང་གི་འགན་འཛིན་དང་ཐེན་ཅིན་གྲོང་ཁྱེར་གྱི་མགོ་གཙོ། རྒྱལ་ཡོངས་སྤྱི་ཚོགས་འགན་ལེན་དཔལ་འབྱོར་ལས་འཛིན་ལྷན་ཚོགས་ཀྱི་ཚོགས་གཙོ་བཅས་གནང་སྐབས། དབང་ཆ་བེད་སྤྱོད་གཏོང་བའི་ཤུགས་རྐྱེན་དང་དཔོན་རིགས་ནང་ཁུལ་གྱི་གནས་ཚུལ་སྟངས་འཛིན། གཉེན་ཉེ་ཆེད་ཁེ་ཕན་ཧམ་ལེན་བྱས་ཡོད་ཚུལ་སོགས་བརྗོད་འདུག ཧོང་ཀོང་འཐབ་རྩོད་དུས་དེབ་ཀྱིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཟླ་ ༣ པའི་ཟླ་སྟོད་ལ་ཏེད་ཞིང་ལོན་གྱིས་དབུས་སྒྲིག་ཁྲིསམ་ཞིབ་བཤེར་ལྷན་ཁང་ལ་རང་སྐྱོན་ཁས་ལེན་བྱས་ཏེ་མགོ་བཏགས་ཞུས་ཡོད་པ་དང་། ཟླ་ ༤ པའི་ཟླ་མགོར་དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གིས་ཞིབ་བཤེར་བྱེད་བཞིན་པ་ཁྱབ་བསྒྲགས་བྱས་འདུག གསར་འགོད་པ། འཇིགས་མེད་རྒྱ་མཚོ།
