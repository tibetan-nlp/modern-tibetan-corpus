ལྐོག་རྔན་བླངས་པ་ཁས་ལེན་བྱས།

ཁ་སང་རྒྱ་ནག་དབུས་དམག་སྲིད་ལྷན་ཁང་གི་ཚོགས་གཞོན་གྱིས་ལྐོག་ཟ་དང་ལྐོག་རྔན་བླངས་པ་ཁས་ལེན་བྱས་ཡོད་ཚུལ་ཕྱི་བསྒྲགས་བྱས་འདུག ཁ་སང་རྒྱ་ནག་ཤིན་ཧྭ་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། འདི་ལོའི་ཕྱི་ཟླ་ ༣ པའི་ནང་རྒྱ་ནག་ཞིབ་དཔྱོད་ཚན་པས་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་དོགས་གཞིའི་འོག་གནས་དབྱུང་བཏང་བའི་རྒྱ་ནག་དབུས་དམག་སྲིད་ལྷན་ཁང་གི་ཚོགས་གཞོན་ཞའོ་ཚའེ་ཧོ་ (Xu Caihou) ཡིས། ཁ་ཉིན་ཁོས་དེ་སྔོན་རྒྱ་ནག་དབུས་དམག་སྲིད་ལྷན་ཁང་གི་ཚོགས་གཞོན་བྱེད་སྐབས་ལྐོག་རྔན་ཕོན་ཆེན་བླངས་པ་ཁས་ལེན་བྱས་ཡོད་སྐོར་བསྒྲགས་འདུག  འདིའི་ཐད་རྒྱ་ནག་ཞིབ་དཔྱོད་ཚན་པས། “ང་ཚོས་འདས་པའི་ཟླ་བདུན་ཙམ་རིང་ཞའོ་ཚའོ་ཧོ་ལ་འདྲི་རྩད་བྱས་པ་བརྒྱུད། ཁོས་དེ་སྔོན་དབང་ཆ་བེད་སྤྱོད་ལོག་པར་བཏང་སྟེ་ལྐོག་རྔན་ཕོན་ཆེན་བླངས་ནས་མི་གཞན་གྱི་གོ་གནས་སྤར་བ་མ་ཟད། ནང་མི་བརྒྱུད་ལྐོག་རྔན་བླངས་ཏེ་གཞན་ལ་ལས་ཀ་སྤྲད་པ་སོགས་ཀྱི་ནག་ཉེས་བསགས་ ཡོད་པ་ཁས་ལེན་བྱས་ཡོད།”ཅེས་བརྗོད་འདུག ད་ལྟ་ལོ་ ༧༡ ཡིན་པའི་ཞའོ་ཚའེ་ཧོ་ (Xu Caihou) ཡིས་ ༢༠༠༤ ནས་ ༢༠༡༢ ལོའི་བར་རྒྱ་ནག་དབུས་དམག་སྲིད་ལྷན་ཁང་གི་ཚོགས་གཞོན་གྱི་ལས་འགན་འཁུར་བའི་རིང་ལྐོག་ཟ་རུལ་སུངས་དང་དབང་ཆ་ལོག་པར་བེད་སྤྱོད་བཏང་བ་སོགས་བྱས་ཡོད་ཅིང་། ཁོ་ནི་རྒྱ་ནག་གི་དམག་དོན་ཚན་ཁག་ནས་ལྐོག་ཟ་རུལ་སུངས་བྱས་ཏེ་འཇུ་བཟུང་བྱས་པའི་དཔོན་རིགས་མཐོ་གྲས་ཤིག་ཆགས་ཡོད་པ་དང་། ད་ལྟ་རྒྱ་ནག་ཞིབ་དཔྱོད་ཚན་པས་རྒྱ་ནག་དབུས་བདེ་འཇགས་ལྷན་ཁང་གི་མགོ་ཁྲིད་ཟུར་བ་ཀྲུག་ཡོང་ཁང་ལ་ཞིབ་བཤེར་བྱེད་བཞིན་ཡོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
