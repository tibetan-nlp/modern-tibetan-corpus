སྒྲོལ་མ་སྐྱབས་ལཊ་བཙོན་གྲོལ་ཐོབ།

  ཁ་སང་བོད་ཀྱི་ཆབ་སྲིད་བཙོན་པ་རྒྱལ་གཅེས་དཔའ་བོ་སྒྲོལ་མ་སྐྱབས་ལགས་བཙོན་གྲོལ་བཏང་ཡོད་འདུག   ཨ་རིག་འགྱུར་མེད་ལགས་ཀྱིས་འདི་གར་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། ཟླ་འདིའི་ཚེས་ ༨ ཉིན་གྱི་སྔ་དྲོར་རྒྱའི་བཙོན་འོག་ཏུ་ལོ་ངོ་ ༡༠ ལྷག་རིང་མནར་གཅོད་མྱངས་བའི་བོད་ཀྱི་ཆབ་སྲིད་བཙོན་པ་རྒྱལ་གཅེས་དཔའ་བོ་སྒྲོལ་མ་སྐྱབས་ལགས་ལྷ་སའི་ཆུ་ཤུར་བཙོན་ཁང་ནས་ཀློད་གྲོལ་བཏང་སྟེ། ལྷ་ས་ཉེན་རྟོག་པས་ཟི་ལིང་བར་བསྐྱལ་ཡོད་པ་དང་། ཟི་ལིང་ནས་མདོ་ལ་རྫོང་ཉེན་རྟོག་པར་རྩིས་སྤྲད་དེ་མདང་དགོང་མཚན་མོར་ནང་དུ་བསྐྱལ་ཡོད་པ་དང་། ད་ལྟའི་ཆར་ཁོང་གི་སྐུ་གཟུགས་གནས་སྟངས་ཧ་ཅང་སྐྱོ་བོ་ཡོད་འདུག     གཞོན་སྐྱེས་རྩོམ་པ་པོ་སྒྲོལ་མ་སྐྱབས་ལགས་ནི། ཕྱི་ལོ་ ༡༩༧༦ ལོར་བོད་མདོ་སྨད་ཨ་རིག་སྟེ། ད་ལྟའི་རྒྱ་གཞུང་གིས་ས་ཁོངས་ཆ་བགོས་ལྟར་ན་མཚོ་སྔོན་ཞིང་ཆེན་མཚོ་བྱང་ཁུལ་མདོ་ལ་རྫོང་འབྲོང་ལུང་སྡེ་བ་རུ་སྐྱེས་ཤིང་། ཆུང་དུས་ནས་བཟུང་ཨ་རིག་སྡེ་བའི་སློབ་ཆུང་དང་། རྫོང་སློབ་འབྲིང་། མཚོ་སྔོན་ཞིང་ཆེན་དགེ་ཐོན་སློབ་ཆེན་བཅས་རིམ་པར་སློབ་མཐར་ཕྱིན་རྗེས། མདོ་ལ་རྫོང་སློབ་འབྲིང་དུ་དགེ་རྒན་གྱི་ཕྱག་ལས་གཉེར་པ་དང་། རྗེས་སུ་པེ་ཅིན་སློབ་ཆེན་དུ་ཞུགས་ཏེ་ཞིབ་འཇུག་སློབ་མའི་སྡེ་ཁག་ཏུ་སློབ་གཉེར་གནང་ཡོད་པ་རེད། ཕྱི་ལོ་ ༢༠༠༣ ལོར་རྒྱ་གར་ལ་བཙོན་བྱོལ་ཏུ་ཕེབས་ཏེ་ཤེས་རབ་དགའ་ཚལ་སློབ་གླིང་དུ་དབྱིན་ཡིག་སྦྱོང་ཞོར་༼ལྷིང་འཇགས་མེད་པའི་ཧི་མ་ལ་ཡ༽ ཞེས་པའི་དེབ་དེ་རྒྱ་ཡིག་ཐོག་བརྩམས་གནང་ཡོད་ལ། ཕྱི་ལོ་ ༢༠༠༤  ལོར་ཕྱིར་བོད་ནང་དུ་ལོག་སྟེ་ལྷ་ས་གྲོང་ཁྱེར་གྱི་སློབ་འབྲིང་ཞིག་ནང་དགེ་རྒན་གྱི་ཕྱག་ལས་གནང་ཡོད་པ་དང་། ལས་ཞོར་དུ་བོད་ཀྱི་ས་ཁམས་ལོ་རྒྱུས་ཀྱི་དེབ་ཅིག་ཀྱང་རྩོམ་འབྲི་བྱེད་མགོ་བརྩམས་ཡོད་ཀྱང་། ཕྱི་ལོ་ ༢༠༠༥ ལོའི་ཟླ་ ༣ ཚེས་ ༩ ཉིན་རྒྱ་གཞུང་ཉེན་རྟོག་པས་ཁོང་འཇུ་བཟུང་བྱས་ཏེ།  ལོ་དེའི་ཕྱི་ཟླ་ ༡༡ ཚེས་ ༣༠ ཉིན་ལྷ་ས་གྲོང་ཁྱེར་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་ཟེར་བས་རྒྱལ་ཁབ་ཀྱི་བདེ་འཇགས་ལ་གནོན་སྐྱོན་བཏང་བའི་ཉེས་མིང་གཡོགས་ཏེ། གསང་བའི་ཐོག་ནས་ཁོང་ལ་ལོ་ ༡༠ དང་ཟླ་ ༦ གི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་དེ་ལྷ་ས་ཆུ་ཤུར་བཙོན་ཁང་དུ་བཙོན་འཇུག་བྱས་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༠༧ ལོའི་ཟླ་ ༧ པའི་ནང་ཁོང་མཚོ་སྔོན་ཞིང་ཆེན་གྱི་སྲིད་འཛིན་ལྟེ་གནས་གྲོང་ཁྱེར་ཟི་ལིང་གི་བཙོན་ཁང་དུ་གནས་སྤོ་བྱས་ཡོད་འདུག     ཕྱི་ལོ་ ༢༠༠༩ ལོར་རྩོམ་པ་པོ་སྒྲོལ་མ་སྐྱབས་ལགས་ལ་ཧེལ་མན་ཧམ་མེ་ཊ་གཟེངས་རྟགས་(Hellman/Hammett) ཕུལ་ཡོད་པ་མ་ཟད། ཕྱི་ལོ་ ༢༠༡༢ ལོར་རྒྱའི་རང་དབང་སྨྱུག་ཚོགས་ཀྱི་རྒྱུན་ལས་མཐུན་ཚོགས་ཀྱིས་བོད་རིགས་རྩོམ་པ་པོ་བཀྲ་ཤིས་རབ་བརྟན་དང་། དཀོན་མཆོག་ཚེ་བྷེ། གངས་སྐྱེས་སྒྲུབ་པ་སྐྱབས། དེ་བཞིན། རྒྱ་རིགས་རྩོམ་པ་པོ་ཁྲིན་ཞི་དང་ཀོ་ཀྲི་ཧྲིན། རའོ་ཝུན་ཝེ། ལིས་པིས་ཧྥིན་བཅས་འཕྲལ་དུ་བཙོན་གྲོལ་དགོས་པའི་རྩོམ་པ་པོ་གཙོ་བོའི་གྲས་སུ་དམིགས་སུ་བཏོན་ཏེ་རྒྱལ་སྤྱིའི་སྨྱུག་ཚོགས་ལ་སྙན་གསེང་ཞུས་ཡོད་པ་ལྟར། ལོ་དེར་རྒྱལ་སྤྱིའི་སྨྱུག་ཚོགས་ཀྱི་བཙོན་འོག་རྩོམ་པ་པོའི་མཐུན་ཚོགས་དང་རྒྱའི་རང་དབང་སྨྱུག་ཚོགས་ཀྱིས། བཙོན་འོག་རྩོམ་པ་པོའི་གཟེངས་རྟགས་བོད་ཀྱི་གཞོན་སྐྱེས་རྩོམ་པ་པོ་སྒྲོལ་མ་སྐྱབས་ལགས་ལ་ཕུལ་ཡོད་པ་རེད།   རྩོམ་པ་པོ་སྒྲོལ་མ་སྐྱབས་ལགས་ཀྱི་༼བོད་རིགས་སྤུན་ཟླ་ཡོངས་ལ་ཕུལ་བའི་ཡི་གེ་ཞིག༽ཞེས་པའི་ནང་དུ། “ང་ཚོ་ཞེད་སྣང་སྐྱེ་མི་དགོས་ལ། མཆི་མ་ཡང་འདོན་མི་དགོས། ང་ཚོར་ངེས་པར་དུ་རྒྱལ་ཁ་འཐོབ་ཡོང་། ཁྱེད་ཚོ་སྟེ། སྙིང་ཉེ་བའི་བོད་རིགས་སྤུན་ཟླ་རྣམ་པ། ངའི་ཤ་ཉེ་ཚོ། ང་ཚོས་ངེས་པར་དུ་ལས་ཀའི་མཐའ་བསྐྱལ་དགོས། ང་ཚོ་ཉིན་ཞིག་གཟི་བརྗིད་ལྡན་པ་ཞིག་ཏུ་འགྱུར་ཐུབ” ཅེས་བྲིས་ཡོད་པ་རེད། གསར་འགོད་པ། དོན་གྲུབ།
