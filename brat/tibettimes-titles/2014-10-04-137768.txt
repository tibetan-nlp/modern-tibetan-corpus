ངོ་རྒོལ་བ་མང་པོར་རྨས་སྐྱོན་བཟོས།

ཁ་སང་ཕྱི་ཚེས་ ༣ ཉིན་ཧོང་ཀོང་གི་ངོ་རྒོལ་ལས་འགུལ་ཁྲོད་ཟིང་འཁྲུག་བྱུང་ཏེ་མི་མང་པོར་རྨས་སྐྱོན་བཟོས་ཡོད་པ་དང་། ཉེན་རྟོག་པས་མི་ ༢༠ ཙམ་འཛིན་བཟུང་བྱས་ཡོད་འདུག ཧོང་ཀོང་མཐོ་སློབ་ཁག་གི་སློབ་མའི་གཙོས་མི་མང་སྟོང་ཕྲག་མང་པོས་རྒྱ་གཞུང་གྱི་དབང་བསྒྱུར་སྲིད་བྱུས་ལ་ངོ་རྒོལ་གྱིས་ཆར་གདུགས་གསར་བརྗེ་ལས་འགུལ་སྤེལ་ཏེ་དེ་རིང་ནས་ཉིན་གྲངས་ ༧ འགྲོ་བཞིན་ཡོད་པ་དང་། གང་མགྱོགས་ཀྱིས་གྲོས་མོལ་བྱ་རྒྱུར་མོས་མཐུན་བྱུང་སྟེ་མི་ཚོགས་རིམ་གྱིས་ཇེ་ཉུང་དུ་འགྲོ་བཞིན་ཡོད་ཀྱང་། ཁ་སང་ཕྱི་ཚེས་ ༣ ཉིན་ཧོང་ཀོང་ཝང་ཇོ་དང་ཐོང་ལོ་ཝན་ས་ཁུལ་གྱི་ངོ་རྒོལ་ལས་འགུལ་ཁྲོད། གློ་བུར་དུ་གང་ནས་ཡོང་མི་ཤེས་པའི་ལས་འགུལ་བ་ཁག་ཅིག་ཡོང་སྟེ། གང་བྱུང་ངོ་རྒོལ་བ་ཚོར་གཅར་རྡུང་དང་འཇིགས་སྐུལ་བྱས་ཏེ་ངོ་རྒོལ་བ་མང་པོར་རྨས་སྐྱོན་བཟོས་ཡོད་སྟབས། ཉེན་རྟོག་པས་སྐྱེས་པ་ ༢༠ ཙམ་འཛིན་བཟུང་བྱས་ཡོད་པ་དང་། ཁོ་ཚོའི་ནང་ནས་མི་ ༨ ནག་ཚོགས་དང་འབྲེལ་བ་ཡོད་སྐོར་བསྒྲགས་འདུག གཅར་རྡུང་ཕོག་པའི་བུད་མེད་ཅིག་གིས་གསར་འགོད་པར། ལས་འགུལ་བར་རྒོལ་རྡུང་གཏོང་མཁན་དེ་ཚོ་རྒྱ་ནག་སྐམ་ས་ཆེན་མོ་ནས་ཡིན་ཞིང་། ཕྱི་ཚེས་ ༢ ཉིན་གྱི་དགོང་མོ་ནས་བཟུང་ཁོ་ཚོ་མི་ ༣༠༠༠ ལྷག་ལས་འགུལ་ནང་ཞུགས་ཏེ་ཟིང་འཁྲུག་བསླངས་ཡོད་སྟབས། ཁོ་མོས་ཆེད་དམིགས་ཀྱིས་ཁོ་ཚོ་བརྙན་དུ་བླངས་བར་བརྟེན། ཁོ་མོའི་པར་ཆས་བཙན་འཕྲོག་བྱས་ཏེ་སྡིགས་ར་སྐུལ་ཡོད་སྐོར་བརྗོད་ཡོད་པ་མ་ཟད། ཁོ་མོའི་གྲོགས་མོ་ཞིག་གི་མཉེན་ཉེའང་ལས་འགུལ་འགོག་རྒོལ་ནང་ཞུགས་ཡོད་པ་དང་། མི་དེས་ངོ་རྒོལ་བ་རྒན་གཞོན་དང་ཕོ་མོ་དབྱེར་མེད་ཀྱིས་གཅར་རྡུང་གཏོང་བ་དང་སྡིགས་ར་སྐུལ་བ། མི་ཚོགས་ཁྲོད་འབུད་རྒྱག་གཏོང་བ་སོགས་ཀྱི་ཟིང་འཁྲུག་སློང་ཆེད། གཞན་གྱིས་ཁོ་བོར་རྒྱ་སྒོར་ ༥༠༠༠ གི་ལྐོག་རྔན་སྤྲད་ལུགས་བཤད་ཡོད་སྐོར་འགྲེལ་བརྗོད་བྱས་འདུག དེ་རིང་སྔ་དྲོར་ཧོང་ཀོང་ཉེན་རྟོག་ཕྱོགས་ནས་གསར་འགོད་ལྷན་ཚོགས་ཤིག་བསྡུས་ཏེ། ཝང་ཇོ་ས་ཁུལ་གྱི་ཟིང་འཁྲུག་ཁྲོད་ལོ་ན་ ༡༨ ནས་ ༦༧ བར་གྱི་སྐྱེས་པ་ ༢༠ འཛིན་བཟུང་བྱས་ཡོད་ཅིང་། ཁོ་ཚོའི་ནང་ནས་མི་ ༨ ནག་ཚོགས་དང་འབྲེལ་བ་ཡོད་ལ། ཁོ་ཚོས་ཟིང་འཁྲུག་ཁྲོད་མི་མང་དཀྱུས་མ་ ༡༢ དང་ཉེན་རྟོག་པ་ ༦ ལ་རྨས་སྐྱོན་བཟོས་ཡོད་པ་དང་། ཚོགས་པ་དེའི་མི་སྔར་ལས་མང་པོ་འཛིན་བཟུང་བྱ་རྒྱུ་ཡིན་སྐོར་བསྒྲགས་ཡོད་པ་མ་ཟད། ཝང་ཇོ་ས་ཁུལ་གྱི་ངོ་རྒོལ་བ་ཚོས་རང་ཉིད་ཀྱི་བདེ་འཇགས་ལ་དོ་ཁུར་བྱས་ཏེ། ཅིན་ཀྲུང་ངོ་རྒོལ་འདུ་འཛོམས་སར་ཕྱིར་ལོག་དགོས་པའི་འབོད་སྐུལ་ཡང་བྱས་ཡོད་འདུག ཁ་སང་ཕྱི་དྲོར་ཧོང་ཀོང་དམིགས་བསལ་སྲིད་འཛིན་ཉིང་ཁྲིན་ཡེན་གྱིས། ངོ་རྒོལ་བ་ཚོས་སྲིད་གཞུང་ལྟེ་གནས་དང་སྲིད་འཛིན་ལས་ཁུངས་བསྐོར་ནས་ངོ་རྒོལ་བྱེད་བཞིན་པ་དེས། ལས་བྱེད་ ༣༠༠༠ ཙམ་གྱི་ལས་ཀ་ལ་ཤུགས་རྐྱེན་ཐེབས་ཀྱི་ཡོད་པ་དང་། ད་ལྟའི་ཆར་ངོ་རྒོལ་བ་མཐའ་སྐྲོད་གཏོང་བའི་འཆར་གཞི་མེད་ཀྱང་མུ་མཐུད་ངོ་རྒོལ་བྱས་པ་ཡིན་ན་བཟོད་སྒོམ་བྱ་རྒྱུ་མིན་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
