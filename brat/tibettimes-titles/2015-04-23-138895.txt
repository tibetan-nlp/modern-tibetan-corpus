གསར་འགོད་པར་ཉེན་ཁ་ཆེས་ཆེ་དུས་ཤིག་རེད།

རྒྱལ་སྤྱིའི་གསར་འགོད་པ་སྲུང་སྐྱོབ་ལྷན་ཚོགས་ཀྱི་སྙན་ཐོའི་ནང་སྲིད་གཞུང་ལ་ངོ་རྒོལ་བྱས་པ་སོགས་ཀྱི་ཉེས་མིང་འོག་རྒྱ་གཞུང་གིས་ཁྱོན་གསར་འགོད་པ་ ༤༤ བཀག་ཉར་བྱས་ཡོད་སྐོར་བསྒྲགས་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༡ ཉིན་ཨ་རིར་རྟེན་གཞི་བྱས་པའི་རྒྱལ་སྤྱིའི་གསར་འགོད་པ་སྲུང་སྐྱོབ་ལྷན་ཚོགས་(Committee to protect journalists)ཀྱིས། འཛམ་གླིང་རྒྱལ་ཁབ་ཁག་གི་ནང་གསར་འགོད་པ་བཀག་ཉར་དང་སུན་གཙེར་བཟོས་པ། དྲ་རྒྱའི་བརྒྱུད་ལམ་ལ་སྟངས་འཛིན་ནན་པོ་བྱས་པ་སོགས་ཀྱི་ཐོག་ཞིབ་འཇུག་བྱས་པ་བརྒྱུད། ༼གསར་འགྱུར་རང་དབང་གི་གནས་སྟངས༽ཞེས་པའི་སྙན་ཐོ་ཞིག་ཕྱིར་བསྒྲགས་བྱས་ཡོད་པ་དང་། སྙན་ཐོ་དེའི་ནང་སྤྱི་ཚོགས་རིང་ལུགས་ཀྱི་རྒྱལ་ཁབ་བཞི་སྟེ་རྒྱ་ནག་དང་བྱང་ཀོ་རི་ཡ། ཝེ་ནམ། ཀོ་བྷ་བཅས་ཀྱི་གཙོས་འབར་མ་དང་ཨི་རག ཨ་ཛར་བྷ་ཇན་(Azerbaijan)། སོ་ཌི་ཨ་རབ་བྷི་ཡ། ཨ་ཧྥེ་རི་ཀའི་(Eritrea)་དང་(Ethiopia)བཅས་གནས་སྟངས་ཚབས་ཆེ་ཤོས་ཀྱི་རྒྱལ་ཁབ་ ༡༠ ཡི་མིང་ཐེར་འདོན་བྱས་ཡོད་པ་དང་། འདི་ལོར་རྒྱ་ནག་ཨང་རིམ་ ༨ པ་སྟེ་གསར་འགོད་པ་བཀག་ཉར་བྱེད་མཁན་ཚབས་ཆེ་ཤོས་དེ་ཆགས་ཡོད་འདུག གསར་འགོད་པ་སྲུང་སྐྱོབ་ལྷན་ཚོགས་དེས་རྒྱ་ནག་ལོ་བསྟུད་མར་མིང་གཞུང་དེའི་ནང་ཚུད་པའི་རྒྱུ་རྐྱེན་ཐད། སྙན་ཐོའི་ནང་བཏོན་པའི་རྒྱལ་ཁབ་གཞན་ ༧ དང་བསྡུར་ན། རྒྱ་ནག་སྐམ་ས་ཆེན་མོའི་ནང་དྲ་རྒྱའི་བརྒྱུད་ལམ་ཞིབ་བཤེར་བྱེད་པའི་གནས་སྟངས་ཅུང་ལྷོད་པོ་ཡོད་ཀྱང་། འགྲོ་བ་མིའི་ཐོབ་ཐང་ལ་སྣང་མེད་རྡོག་རོལ་གཏོང་བཞིན་ཡོད་པའི་རྐྱེན་གྱིས་ཨང་རིམ་ ༨ པ་དེ་ཆགས་ཡོད་ལ། རྒྱ་ནག་ནང་ཁྱོན་གསར་འགོད་པ་ ༤༤ བཀག་ཉར་བཙོན་འཇུག་བྱས་ཡོད་པ་དང་། དེའི་ནང་ནས་མི་ ༢༩ སྲིད་གཞུང་ལ་ངོ་རྒོལ་བྱས་པའི་ཉེས་མིང་གཡོགས་ཏེ་འཛིན་བཟུང་བྱས་འདུག རྒྱ་ནག་འཛམ་གླིང་ཐོག་གསར་འགོད་པ་བཀག་ཉར་བྱས་པ་མང་ཤོས་དེ་ཆགས་ཡོད་པ་མ་ཟད། དྲ་རྒྱར་ནན་བཤེར་བྱེད་པའི་ཐབས་ཇུས་ལག་བསྟར་བྱས་ཏེ། རྒྱ་ནག་ནང་དྲ་རྒྱ་བེད་སྤྱོད་བྱེད་མཁན་གྱི་མི་གྲངས་དུང་ཕྱུར་ ༦.༤༢ ལྷག་གིས། ལྟ་ཚུལ་མི་མཐུན་པའི་དྲ་རྒྱ་དང་སྤྱི་ཚོགས་འབྲེལ་འདྲིས་ཀྱི་བརྒྱུད་ལམ་ཁག་ལ་ལྟ་ཀློག་བྱེད་པར་བཀག་སྡོམ་བྱེད་བཞིན་ཡོད་པ་དང་འབྲེལ། ཕྱིའི་བརྒྱུད་ལམ་ཁག་ལའང་སིམ་འཛུལ་གྱི་ལག་པ་རྐྱོང་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག ཡང་ལྷན་ཚོགས་དེའི་སློབ་སྟོན་འགན་འཛིན་ལྕམ་ཁོ་ཊི་ནི་ཡིས། ཕྱི་ལོ་ ༢༠༡༤ ལོ་གཅིག་པུའི་ནང་འཛམ་གླིང་ཐོག་ཁྱོན་གསར་འགོད་པ་ ༢༢༡ བཙོན་འཇུག་བྱས་ཡོད་པ་ལྟར། འདས་པའི་ལོ་ ༣ ནི་གསར་འགོད་པར་ཆེས་ཉེན་ཁ་ཆེ་བའི་དུས་སྐབས་ཤིག་སྟེ། གསར་འགོད་པ་དམར་གསོད་དང་བཀག་ཉར་བྱས་པའི་གྲངས་འབོར་ཁོ་ནས་ཟིན་ཐོ་ཞིག་བསྐྲུན་ཡོད་ལུགས་དང་། ཚོགས་པ་དེས་རོགས་སྐྱོབ་བྱས་པའི་གྲངས་འབོར་ལྟར་ན། འདས་པའི་ལོ་ ༥ ནང་ཁྱོན་གསར་འགོད་པ་ ༤༠༠ ལྷག་ཉེན་གཡོལ་དུ་འགྲོ་དགོས་བྱུང་ཡོད་སྐོར་སོགས་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
