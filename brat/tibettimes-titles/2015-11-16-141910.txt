རྒྱལ་ཁབ་ ༢༠ དང་དྲག་སྤྱོད་པ།

ཁ་ཆེའི་དྲག་སྤྱོད་ཚོགས་པར་རྒོལ་རྡུང་གཏོང་བའི་ཐད་མི་མཐུན་པ་ཆེན་པོ་ཡོད་པས་རྒྱལ་ཁབ་ ༢༠ ཡི་དབུ་ཁྲིད་ལྷན་ཚོགས་ཐོག་མཉམ་འབྲེལ་གྱི་ནུས་པ་འཐོན་མི་ཐུབ་སྐོར་བརྗོད་འདུག   ཕྱི་ཚེས་ ༡༥ ཉིན་ནས་བཟུང་ཉིན་གྲངས་གཉིས་རིང་། འཛམ་གླིང་སྟོབས་ལྡན་རྒྱལ་ཁབ་ ༢༠ ཡི་དབུ་ཁྲིད་ལྷན་ཚོགས་ཐེངས་བཅུ་པ་ཊིར་ཀི་ (Turkey)ཡི་ཨིན་ཐལ་ཡཱ་མཚོ་འགྲམ་དུ་ཚོགས་བཞིན་ཡོད་པ་དང་། ཐེངས་འདིའི་ལྷན་ཚོགས་ཐོག འཛམ་གླིང་གི་དཔལ་འབྱོར་འཕེལ་རྒྱས་རྒྱུན་འཁྱོངས་དང་དཔལ་འབྱོར་གྱི་སྟོབས་ཤུགས་སྲ་བརྟན་དུ་གཏོང་ཐབས་སྐོར་ལ་གྲོས་བསྡུར་བྱ་རྒྱུ་ཡིན་ནའང་། ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༣ ཉིན་གྱི་མཚན་མོར་ཧྥ་རན་སིའི་རྒྱལ་ས་ཕ་རི་སའི་ནང་ཁ་ཆེའི་དྲག་སྤྱོད་ཚོགས་པས་(ISIS)གཏོར་རྒོལ་དོན་རྐྱེན་བསླངས་ཏེ། ཉུང་མཐར་མི་གྲངས་ ༡༣༢ དམར་གསོད་བཏང་ཡོད་པ་དང་། ཁ་ཆེའི་དྲག་སྤྱོད་ཚོགས་པས་བསྟུན་མར་སྤེལ་བའི་དོན་རྐྱེན་རྣམས་འཛམ་གླིང་གི་སྟོབས་ལྡན་རྒྱལ་ཁབ་ཁག་ཚང་མར་འབྲེལ་བ་ཆགས་ཀྱི་ཡོད་སྟབས། ཐེངས་འདིའི་ལྷན་ཚོགས་ཐོག་རྒྱལ་ཁབ་ ༢༠ ཡི་དབུ་ཁྲིད་རྣམས་མཉམ་འབྲེལ་གྱི་ཐོག་ནས་ཡོངས་ཁྱབ་གསལ་བསྒྲགས་ཤིག་སྤེལ་ཐུབ་པའི་རེ་བ་དང་འབོད་སྐུལ་བྱེད་བཞིན་ཡོད་འདུག   ཐེངས་འདིའི་ལྷན་ཚོགས་ཐོག་ཞུགས་པའི་ཨ་རིའི་སྲིད་འཛིན་ཨོ་བྷ་མས། ཕ་རི་སིའི་དམར་གསོད་དོན་རྐྱེན་འདི་ནི་ཕ་རི་སི་དང་ཧྥ་རན་སིའི་མི་མང་ལ་དྲག་རྒོལ་བྱས་པ་ཙམ་མིན་པར་འགྲོ་བ་མིའི་རིགས་ཡོངས་དང་། མིའི་རིགས་ཀྱི་ཡོངས་ཁྱབ་རིན་ཐང་ལྟ་བར་དྲག་རྒོལ་བྱས་པ་ཞིག་ཡིན་སྟབས། ང་ཚོས་དྲག་སྤྱོད་པར་རྒོལ་རྡུང་གཏོང་བའི་གྲ་སྒྲིག་ཆ་ཚང་བྱས་ཟིན་ལ། དེ་དང་མཉམ་དུ་ཧྥ་རན་སིའི་གཞུང་དང་མི་མང་ལ་ག་དུས་ཡིན་ནའང་རོགས་རམ་བྱ་རྒྱུ་ཡིན། ཞེས་དང་། ཨ་རི་དང་ཨུ་རུ་སུ་གཉིས་མཉམ་འབྲེལ་དམ་ཟབ་ཀྱིས་ཕ་རི་སིའི་དམར་གསོད་དོན་རྐྱེན་གྱི་ཇུས་འགོད་པ་དང་ཚོགས་པ་རྗེས་འདེད་ཀྱིས་འཛིན་ཐབས་བྱ་རྒྱུ་ཡིན་སྐོར་བརྗོད་འདུག   ཡང་ལྷན་ཚོགས་ཐོག་ཞུགས་པའི་ཨུ་རུ་སུའི་སྲིད་འཛིན་པུ་ཊིན་གྱིས། དྲག་སྤྱོད་རིང་ལུགས་པའི་བྱ་སྤྱོད་ལ་རྒོལ་རྡུང་གཏོང་བ་དེ་རྒྱལ་སྤྱིའི་སྤྱི་ཚོགས་མཉམ་འབྲེལ་བྱས་ན་ད་གཟོད་ནུས་པ་ཐོན་ཐུབ་པ་རེད་ཅེས་བརྗོད་ཡོད་འདུག་ལ། དེ་བཞིན་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱིས། ཕ་རི་སིའི་དམར་གསོད་དོན་རྐྱེན་ཁྲོད་ཚེ་སྲོག་ཤོར་བ་རྣམས་ལ་མྱ་ངན་ཞུ་བ་དང་། འདས་པོའི་ནང་མི་དང་རྨས་སྐྱོན་ཕོག་པ་ཚོར་སེམས་གསོ་ཡོད་ཅེས་བརྗོད་འདུག   ཨ་རིའི་ནང་ཡོད་པའི་ཆབ་སྲིད་དཔྱད་བརྗོད་པ་ཡང་ཅན་ལི་ཡིས། ཁ་ཆེའི་དྲག་སྤྱོད་ཚོགས་པའི་གནད་དོན་དེ་འཛམ་གླིང་ཡོངས་ཀྱི་གནད་དོན་གལ་ཆེན་པོ་ཞིག་ཏུ་གྱུར་ཡོད། གང་ཡིན་ཟེར་ན། ཚོགས་པ་དེས་མི་རྣམས་ཀྱི་འཚོ་བ་སྐྱེལ་སྟངས་ལ་འགྱུར་བ་གཏོང་ཐབས་བྱེད་ཀྱི་ཡོད་པ་རེད། དྲག་སྤྱོད་པར་རྒོལ་རྡུང་བའི་གནད་དོན་འདིའི་ཐད་ཨ་རིའི་གཙོས་དམངས་གཙོའི་རྒྱལ་ཁབ་ཁག་དང་ཨུ་རུ་སུའི་བར་མི་མཐུན་པ་ཆེན་པོ་ཡོད་པ་རེད། ཨུ་རུ་སུས་ཁ་ཆེའི་དྲག་སྤྱོད་ཚོགས་པའི་རྟེན་གཞི་གཙོ་བོ་སི་རི་ཡའི་ནང་ཡོད་སྟབས་དྲག་སྤྱོད་པར་རྒོལ་རྡུང་གཏོང་གིན་ཡོད་པ་དང་། ཨ་རིས་སི་རི་ཡ་ནི་དབང་ཆ་སྒེར་གཅོད་ཀྱི་རྒྱལ་ཁབ་ཅིག་ཡིན་སྟབས། དྲག་སྤྱོད་ལ་རྒོལ་གྱི་ཡོད་པ་མ་ཟད། དབང་ཆ་སྒེར་གཅོད་ལའང་ངོ་རྒོལ་བྱེད་བཞིན་ཡོད་པ་དང་། ཨུ་རུ་སུས་དབང་ཆ་སྒེར་གཅོད་ལ་ངོ་རྒོལ་བྱེད་ཀྱི་མེད། ད་དུང་ཨ་རི་དང་ཨུ་རུ་སུ་གཉིས་ཀྱིས་འཛམ་གླིང་ཡོངས་ལ་ཤུགས་རྐྱེན་འཇོག་པའི་སྟོབས་ཤུགས་འགྲན་རྩོད་བྱེད་ཀྱི་ཡོད་སྟབས། རྒྱལ་ཁབ་གཉིས་བར་མི་མཐུན་པ་ཆེན་པོ་བྱུང་གི་ཡོད་པ་རེད། ཅེས་དང་། ཡང་ཁོང་གིས། ཁ་ཆེའི་དྲག་སྤྱོད་ཚོགས་པར་རྒོལ་རྡུང་གཏོང་བའི་ཐད་ཨ་རིའི་འགན་ཆེ་ཤོས་འཁུར་དགོས། གང་ཡིན་ཟེར་ན། ཨོ་བྷ་མས་སྲིད་འཛིན་གྱི་ལས་འཁུར་ཞུས་རྗེས། དབང་ཆ་སྒེར་གཅོད་དང་འཇིགས་སྐུལ་དྲག་སྤྱོད་རིང་ལུགས་པར་མཐའ་གཅིག་ཏུ་རྒོལ་རྡུང་གཏོང་བའི་སྲིད་བྱུས་འཛིན་གྱི་ཡོད་པ་རེད། དེ་ནས་ཨུ་རུ་སུ་དང་རྒྱ་ནག་གཉིས་ཀྱིས་འགན་ཆེན་པོ་འཁུར་དགོས། གང་ཡིན་ཟེར་ན། རྒྱལ་ཁབ་དེ་གཉིས་ཀྱིས་སི་རི་ཡའི་གནད་དོན་ཐག་གཅོད་བྱེད་པའི་རྒྱལ་སྤྱིའི་སྤྱི་ཚོགས་ཀྱི་བྱ་འགུལ་ལ་བཀག་འགོག་མང་པོ་བྱས་པར་བརྟེན། ཁ་ཆེའི་དྲག་སྤྱོད་ཚོགས་པས་སྨྱོ་ཧམ་ཆེན་པོས་འཇིགས་སྐུལ་དྲག་སྤྱོད་ཀྱི་དོན་རྐྱེན་རིམ་པར་སྤེལ་གྱི་ཡོད་པ་རེད། འཛམ་གླིང་སྟོབས་ལྡན་རྒྱལ་ཁབ་ཁག་དང་། ལྷག་པར་དུ་ཨ་རི་དང་ཨུ་རུ་སུ་གཉིས་དོན་དམ་གྱི་མཉམ་འབྲེལ་བྱུང་བ་ཡིན་ན། ཁ་ཆེའི་དྲག་སྤྱོད་ཚོགས་པ་རྩ་མེད་བཟོ་རྒྱུ་ཁག་པོ་མིན་ལ། སི་རི་ཡའི་ནང་འཁྲུག་འཕྲལ་དུ་མཚམས་འཇོག་ཐབས་བྱས་ཏེ། རྒྱལ་ཁབ་བསྐྱར་སྐྲུན་ཡོང་བར་འབད་པ་ནི་གནད་དོན་ཐག་གཅོད་པའི་ཐབས་ལམ་ངོ་མ་དེ་ཡིན་སྐོར་བརྗོད་འདུག     རྒྱལ་སྤྱིའི་གནད་དོན་དཔྱད་ཞིབ་པ་ཚོས། ཐེངས་འདིའི་ལྷན་ཚོགས་ཐོག་དྲག་སྤྱོད་རིང་ལུགས་པར་རྒོལ་རྡུང་གཏོང་བ་དང་། ཡུ་རོབ་ནང་སི་རི་ཡའི་སྐྱབས་བཅོལ་བ་འབོར་ཆེན་འབྱོར་བཞིན་ཡོད་པ་དེའི་ཐད་གླེང་སློང་བྱུང་བ་མ་ཟད། རྒྱལ་ཁབ་ཁག་གི་དབུ་ཁྲིད་ཚང་མས་དྲག་སྤྱོད་རིང་ལུགས་པར་རྒོལ་རྡུང་གཏོང་རྒྱུ་ཡིན་པའི་ཁས་ལེན་བྱེད་སྲིད། ཡིན་ནའང་། རྒོལ་རྡུང་གཏོང་བའི་གལ་གནད་ཀྱི་བརྗོད་གཞིའི་ཐོག་རྒྱལ་ཁབ་ཁག་གི་ལངས་ཕྱོགས་ལ་སྔར་བཞིན་མི་མཐུན་པ་ཆེན་པོ་ཡོད་སྐོར་དང་། དེ་བཞིན་ཐེངས་འདིའི་དམར་གསོད་དོན་རྐྱེན་གྱིས་སི་རི་ཡའི་སྐྱབས་བཅོལ་བ་རྣམས་ལ་འགོག་རྐྱེན་ཆེན་པོ་བཟོ་རྒྱུ་ཡིན་ལུགས་བརྗོད་ཀྱི་འདུག ཡིན་ནའང་། ཡུ་རོབ་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱིས་ཕ་རི་སིའི་དམར་གསོད་དོན་རྐྱེན་སོགས་གནས་ཚུལ་ག་རེ་བྱུང་བ་ཡིན་ནའང་། རྒྱལ་ཁབ་ཁག་གིས་སི་རི་ཡའི་སྐྱབས་བཅོལ་བའི་གནད་དོན་གླེང་སློང་བྱེད་པར་ཤུགས་རྐྱེན་ཐེབས་མི་འོས་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
