ལོ་རེར་མི་ཁྲི་ ༢༠ ལྷག་རྐྱེན་འདས།

ཁ་སང་འཛམ་གླིང་འཕྲོད་བསྟེན་ལྷན་ཁང་གི་གཞུང་ཚབ་བྷིར་ཧར་ཌི་ལགས་ཀྱིས་རྒྱ་ནག་ནང་ལོ་རེར་མི་ཁྲི་ ༢༠ ལྷག་འགྲིམ་འགྲུལ་དོན་རྐྱེན་ཁྲོད་རྐྱེན་འདས་སུ་འགྲོ་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག ཁ་སང་རྒྱ་ནག་ཉིན་རེའི་ཚགས་པར་གྱི་དབྱིན་ཡིག་དྲ་བའི་ཐོག པེ་ཅིན་དུ་ཡོད་པའི་འཛམ་གླིང་འཕྲོད་བསྟེན་ལྷན་ཁང་གི་གཞུང་ཚབ་བྷིར་ཧར་ཌི་(Bernhard Schwartlaender)ལགས་ཀྱིས། རྒྱ་ནག་ནང་གི་འགྲིམ་འགྲུལ་གནད་དོན་སྐོར་གྱི་གནས་ཚུལ་ཞིག་སྤེལ་ཏེ། རྒྱ་ནག་ནང་འགྲིམ་འགྲུལ་དོན་རྐྱེན་ཁྲོད་ལོ་རེར་ཁྱོན་མི་གྲངས་ཁྲི་ ༢༠ ལྷག་གིས་ཚེ་སྲོག་ཤོར་གྱི་ཡོད་པ་དང་། དེའི་ནང་ལོ་ ༡༥ མ་ལོན་པའི་བྱིས་པ་ཁྲི་ ༡ ལྷག་ཚུད་ཀྱི་ཡོད་པ་མ་ཟད། རྨས་སྐྱོན་ཕོག་པའི་གྲངས་འབོར་དེ་ལས་ཀྱང་མང་བ་ཡོད་སྟབས། ཤི་རྨས་ཀྱི་དོན་རྐྱེན་དེ་ཚོ་རྩ་བ་ནས་མེད་པ་བཟོ་ཐབས་ཡོད་སྐོར་བརྗོད་འདུག གནད་དོན་དཔྱད་ཞིབ་པ་ཚོས། རྒྱ་ནག་ནང་འཛུགས་སྐྲུན་ཤུགས་ཆེན་བྱེད་བཞིན་ཡོད་པའི་དུས་དང་བསྟུན་ཏེ། རླངས་འཁོར་དང་སྦག་སྦག རྐང་འཁོར་སོགས་ཀྱང་མང་དུ་ཕྱིན་ཏེ་གྲངས་འབོར་ཚད་ལས་བརྒལ་ཡོད་སྟབས། རང་བཞིན་གྱིས་འགྲིམ་འགྲུལ་སྒྲིག་ཁྲིམས་ལག་བསྟར་བྱེད་པར་གནོད་པ་ཆེན་པོ་ཐེབས་ཀྱི་ཡོད་པ་དང་། དེ་བཞིན། འགྲུལ་བ་ཚོས་ཀྱང་འགྲིམ་འགྲུལ་སྒྲིག་ལམ་ལ་བརྩི་སྲུང་མི་བྱེད་པར་གང་འདོད་ཀྱིས་འགྲོ་སྐྱོད་བྱས་ཏེ་དོན་རྐྱེན་མང་དུ་འབྱུང་བཞིན་ཡོད་པ་མ་ཟད། ཁ་ལོ་བ་མང་ཆེ་བས་མི་ལ་བརྡུངས་ཏེ་རྨས་སྐྱོན་ཕོག་པ་ལས་འདས་གྲོངས་སུ་ཕྱིན་ན་ལེགས་པའི་བསམ་བློ་འཁུར་ཏེ། བརྡབ་སྐྱོན་ཁྲོད་ཚེ་སྲོག་ཤོར་བའི་གྲངས་འབོར་འཕར་གྱི་ཡོད་སྟབས། ཁ་ལོ་བའི་ཚེ་སྲོག་གི་རིན་ཐང་ལྟ་བ་གོང་མཐོར་བཏང་སྟེ། སྲོག་སྐྱོབ་ཐབས་ཀྱི་སྤྱོད་བཟང་སྐུལ་སློང་བྱེད་དགོས་སྐོར་སོགས་རྗོད་ཀྱི་འདུག འདས་ལོའི་ཕྱི་ཟླ་ ༡༢ པའི་ནང་རྒྱ་ནག་ཉེན་རྟོག་ཁང་གིས་བཏོན་པའི་བསྡོམས་རྩིས་ནང་། རྒྱ་ནག་ནང་འགྲིམ་འགྲུལ་དོན་རྐྱེན་ཁྲོད་ལོ་རེར་མི་ཁྲི་ ༦ ཙམ་གྱིས་ཚེ་སྲོག་ཤོར་གྱི་ཡོད་པ་དང་། མི་ཁྲི་ ༢༠ ལྷག་ལ་རྨས་སྐྱོན་ཕོག་གིན་ཡོད་སྐོར་བཀོད་ཡོད་སྟབས། འཛམ་གླིང་འཕྲོད་བསྟེན་ལྷན་ཁང་གིས་བཏོན་པའི་གྲངས་འབོར་ལས་བཞི་ཆ་གཅིག་ཀྱང་ཟིན་གྱི་མེད་པ་རེད། ཡིན་ནའང་། རྒྱ་ནག་གི་གནད་དོན་ཉམས་ཞིབ་པ་སོགས་ཀྱིས་རྒྱ་གཞུང་གིས་བཏོན་པའི་གྲངས་འབོར་དེ་ལས་གཏན་ནས་ཚད་ཀྱི་མེད་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
