གྲྭ་ཤེར་སྐྱབས་ལཊ་འཛིན་བཟུང་།

ཟླ་འདིའི་ཕྱི་ཚེས་ ༩ ཉིན་དཀར་མཛེས་ཁུལ་གསེར་རྟ་རྫོང་ཐོག་ཏུ་གྲྭ་ཤེར་སྐྱབས་ལགས་ཀྱིས་རྒྱ་གཞུང་ལ་ངོ་རྒོལ་གནང་སྟེ་འཛིན་བཟུང་བྱས་ཡོད་འདུག འདི་གར་གནས་ཚུལ་བཏང་བ་ལྟར་ན། ཟླ་འདིའི་ཚེས་༩ ཉིན་གྱི་བོད་ནང་གི་ཉིན་གུང་ཆུ་ཚོད་ ༡༢ ཙམ་ལ། དཀར་མཛེས་ཁུལ་གསེར་རྟ་རྫོང་རྟ་ཤུལ་སྡེ་བའི་གྲྭ་ཤེར་སྐྱབས་ལགས་ཀྱིས། གསེར་རྟ་རྫོང་གི་ཁྲོམ་གཞུང་དུ་༧གོང་ས་མཆོག་བོད་ལ་ཕེབས་སུ་འཇུག་དགོས། བོད་ལ་རང་དབང་དགོས་ཞེས་སྐད་འབོད་དང་ཡིག་ཆ་གཏོར་འགྲེམས་བྱས་ཏེ་རྒྱ་གཞུང་ལ་ཞི་རྒོལ་གནང་ཡོད་པ་དང་། ངོ་རྒོལ་སྐད་འབོད་གནང་སྟེ་སྐར་མ་ལྔ་དྲུག་ཙམ་ལས་མ་འགོར་བར་ས་གནས་དྲག་ཆས་ཉེན་རྟོག་པ་དང་དམག་མི་མང་པོ་འབྱོར་ཏེ། ཁོང་འཛིན་བཟུང་བྱས་ནས་གསེར་རྟ་རྫོང་ཉེན་རྟོག་ཁང་དུ་བཙན་ཁྲིད་བྱས་ཡོད་པ་མ་ཟད། དེ་མ་ཐག་གསེར་རྟ་རྫོང་ཁོངས་ཀྱི་དྲ་རྒྱ་སོགས་བརྒྱུད་ལམ་ཡོངས་རྫོགས་བཀག་སྡོམ་བྱས་ཡོད་འདུག གྲྭ་ཤེར་སྐྱབས་ལགས་ནི་ད་ལྟ་རང་ལོ་ ༢༠ ཡིན་ཞིང་། གསེར་རྟ་གནུབ་གཟུར་དགོན་པའི་གྲྭ་བ་ཡིན་ལ། དེང་སྐབས་གསེར་རྟ་བླ་རུང་ལྔ་རིག་ནང་བསྟན་སློབ་གླིང་དུ་སློབ་གཉེར་གནང་བཞིན་ཡོད་པ་དང་། འཛིན་གྲྭའི་ནང་སློབ་གཉེར་བ་རྩེར་སོན་གྱི་གྲས་ཤིག་ཡིན་འདུག ཁོང་གི་ཕ་མིང་ལ་སྐྱ་རིས་དང་ཨ་མའི་མིང་ལ་ཚེ་ལྷ་ཟེར་གྱི་ཡོད་པ་དང་། ཁོང་ལ་ཕ་མ་གཅིག་པའི་མིང་སྲིང་ལྔ་ཡོད་པའི་ནང་ནས་ཁོང་ཆེ་ཤོས་དེ་ཡིན་འདུག གནུབ་གཟུར་དབང་ཤུལ་དགོན་ནི་གསེར་རྟ་རྫོང་དང་རྒྱང་ཐག་སྤྱི་ལེ་ ༣༠ ཙམ་གྱི་སར་ཡོད་པ་དང་། དགོན་པ་དེར་སྤྱིར་བཏང་དགེ་འདུན་པ་ ༥༠༠ ཙམ་ཡོད་ནའང་། དགེ་འདུན་པ་མང་པོ་ས་ཁུལ་དང་ཆོས་སྡེ་གཞན་པར་སློབ་གཉེར་དུ་ཕྱིན་ཡོད་སྟབས། ། ནམ་རྒྱུན་དགོན་པའི་ནང་གཏན་སྡོད་དགེ་འདུན་པ་ ༡༠༠ ལྷག་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
