ཉེན་རྟོག་པ་ ༧ གྱིས་ངོ་རྒོལ་བྱས།

ཟླ་འདིའི་ཕྱི་ཚེས་ ༤ ཉིན་ཏེ་རྒྱ་ནག་གི་རྩ་ཁྲིམས་ཉིན་མོར་པེ་ཅིན་གྲོང་ཁྱེར་གྱི་གཞུང་སྒྲུབ་ལྟེ་གནས་ཀྲུང་ནན་ཧེ་རུ་ཉེན་རྟོག་པ་ ༧ གྱིས་ཙིག་དུག་བཏུངས་ཏེ་རང་ཤི་ངོ་རྒོལ་བྱས་ཡོད་འདུག འདི་ལོའི་ཕྱི་ཟླ་ ༡༡ ཚེས་ ༡ ཉིན་རྒྱ་ནག་རྒྱལ་ཡོངས་མི་མང་འཐུས་ཚོཊ་རྒྱུན་ལས་ལྷན་ཁང་གིས། ད་ལོ་ནས་བཟུང་ལོ་རེའི་ཕྱི་ཟླ་ ༡༢ ཚེས་ ༤ ཉིན་ནི་རྒྱལ་ཁབ་རྩ་ཁྲིམས་ཉིན་མོར་གཏན་འབེབས་བྱས་ཡོད་པ་ལྟར། ཟླ་འདིའི་ཕྱི་ཚེས་ ༤ ཉིན་རྩ་ཁྲིམས་ཉིན་མོ་ཐེངས་དང་པོ་སྲུང་གཙི་བྱས་པ་ཡོད་པ་དང་། ཉིན་དེར་པེ་ཅིན་གྲོང་ཁྱེར་གྱི་གཞུང་སྒྲུབ་ལྟེ་གནས་ཁག་དང་འགྲིམ་འགྲུལ་ས་ཚིཊ་ཁག་ཏུ་ཉེན་རྟོག་པ་མང་པོ་འགྲེམས་འཇོག་བྱས་ཏེ། མི་མང་གི་འགྲོ་འོང་ལ་དམ་དྲག་ཆེན་པོ་བྱས་ཡོད་པ་མ་ཟད། ཕྱོཊ་ཡོང་གི་འགྲུལ་བ་སྟོང་ཕྲག་མང་པོ་བཀག་ཉར་གྱིས་མིང་དང་ཁ་པར་ཨང་གྲངས། ཐོབ་ཐང་ལག་ཁྱེར་སོཊ་ཐོ་འགོད་བྱས་ནས་ཞིབ་བཤེར་ནན་པོ་བྱས་ཡོད་འདུག ཉིན་དེའི་ཞོགས་པའི་ཆུ་ཚོད་ ༡༠ མ་ཟིན་ཙམ་ལ། ཧེ་ལུང་ཅང་ཞིང་ཆེན་ཧར་པིན་གྲོང་ཁྱེར་ནས་ཡིན་པའི་ཉེན་རྟོག་པ་ ༧ པེ་ཅིན་གཞུང་གི་ལྟེ་གནས་ཀྲུང་ནན་ཧེ་སྒོ་ཆེན་དུ། ཧར་པིན་གྲོང་ཁྱེར་གྱི་མགོ་ཁྲིད་ཚོས་ཁྲིམས་ལྟར་སྲིད་འཛིན་བྱ་བ་ལེགས་པོ་སྒྲུབ་ཀྱི་མེད་སྟབས། ཉེན་རྟོག་བྱ་བ་དང་འཚོ་བའི་གནད་དོན་ཐག་གཅོད་བྱེད་དགོས་པའི་འཕྲེད་དར་འཕྱར་ཏེ་ཞུ་སྐུལ་བྱེད་སྐབས། སྒྲིག་ཆས་གྱོན་མེད་པའི་ཉེན་རྟོག་པ་ ༢༠ ལྷག་འབྱོར་ཏེ་ཁོང་ཚོའི་རེ་སྐུལ་ལ་ཉན་འཇོག་མ་བྱས་པར་འཕྲེད་དར་བཙན་འཕྲོག་བྱས་ཡོད་པར་བརྟེན། སྔ་ས་ནས་གྲ་སྒྲིག་བྱས་པའི་ཙིག་དུག་བཏུངས་ཏེ་རང་ཤི་བརྒྱབ་ཡོད་ཀྱང་། དེ་མ་ཐག་ཉེན་རྟོག་པས་ཁོང་ཚོ་སྨན་ཁང་དུ་བསྐྱལ་ཏེ་མྱུར་སྐྱོབ་བྱས་ཡོད་འདུག ངོ་རྒོལ་བ་ཉེན་རྟོག་པ་ཞིག་གིས། ཁོང་ཚོས་ཞུ་གཏུག་བྱས་ཏེ་ལོ་ངོ་ ༧ འགྲོ་བཞིན་ཡོད་ཀྱང་། གནད་དོན་ཐག་གཅོད་བྱས་མེད་པ་དང་། འདི་ལོའི་ཕྱི་ཟླ་ ༧ ཚེས་ ༡༡ ཉིན་ཧར་པིན་གྲོང་ཁྱེར་ཉེན་རྟོག་ཁང་གི་མགོ་ཁྲིད་ཀྱིས། ཁྱེད་ཚོའི་གནད་དོན་དེ་ངས་ཐག་གཅོད་བྱེད་ཐབས་མི་འདུག་པས། གང་དུ་ཞུ་གཏུག་བྱེད་པར་འགྲོ་འདོད་ན་དེར་སོང་ཞེས་བརྗོད་པར་བརྟེན། ཆེད་དམིགས་ཀྱིས་པེ་ཅིན་ལ་ཞུ་གཏུག་བྱེད་དུ་ཡོང་ཡོད་ཀྱང་གནད་དོན་ཐག་གཅོད་བྱས་མེད་པ་མ་ཟད། ཁོང་ཚོའི་ནང་ནས་ཉེན་རྟོག་པ་ཧང་ཝེ་ཟེར་བ་ཞིག་ལོ་ཕྱེད་རིང་བཀག་ཉར་བྱས་ཡོད་པ་དང་། ད་ལྟ་ཁོང་ཚོར་གླ་ཕོགས་མེད་ལ། རྒྱལ་ཁབ་ཀྱི་འགན་སྲུང་ཅི་ཡང་མེད་པར་འགྲོ་ཡུལ་སྡོད་ཡུལ་མེད་པར་གྱུར་ཡོད་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
