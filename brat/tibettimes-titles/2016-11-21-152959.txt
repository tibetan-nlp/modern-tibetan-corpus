ཨོ་བྷ་མ་དང་ཕུ་ཊེན་གཉིས་ཐུག་འཕྲད།

ཨ་རིའི་སྲིད་འཛིན་ཨོ་བྷ་མས་ཨུ་རུ་སུའི་ཕྱོགས་ནས་སི་རི་ཡ་དང་ཡུ་ཁུ་རན་གཉིས་ནང་གི་ཟིང་འཁྲུག་ཞི་འཇགས་ཡོང་ཐབས་སུ་རྒྱལ་ཁབ་ཕན་ཚུན་བར་གྱི་ཞི་འགྲིག་གྲོས་ཆོད་ལག་བསྟར་བྱེད་དགོས་ཟེར། ཁ་སང་གཟའ་ཉི་མར་ཕི་རུ་ (Peru)རྒྱལ་ཁབ་གྱི་རྒྱལ་སར་ཨ་ཤ་ཡ་ཞི་བདེ་རྒྱ་མཚོའི་མངའ་ཁུལ་གྱི་དཔལ་འབྱོར་མཉམ་ལས་སྒྲིག་འཛུགས་ཚོགས་འདུ་ཚོགས་ཡོད་པ་དང་། སྐབས་དེ་དང་བསྟུན་ཏེ་ཨ་རིའི་སྲིད་འཛིན་ཨོ་བྷ་མ་དང་ཨུ་རུ་སུའི་སྲིད་འཛིན་ཕུ་ཊེན་གཉིས་ཟུར་དུ་མཇལ་འཕྲད་གནང་སྟེ། སྐར་མ་ ༥ རིང་སི་རི་ཡ་དང་ཡུ་ཁུ་རན་གཉིས་ནང་གི་ཟིང་འཁྲུག་སྐོར་ལ་གྲོས་མོལ་བྱས་ཡོད་ཅིང་། དེ་ནི་ཨ་རིའི་སྲིད་འཛིན་ཨོ་བྷ་མའི་སྲིད་འཛིན་ལས་ཡུན་མ་རྫོགས་སྔོན་གྱི་ཕྱོགས་གཉིས་བར་གྱི་ཐུག་འཕྲད་མཐའ་མ་དེ་ཆགས་རྒྱུ་ཡིན་འདུག ཐེངས་འདིར་སྲིད་འཛིན་རྣམ་གཉིས་ཐུག་འཕྲད་རྗེས་ཨ་རིའི་སྲིད་འཛིན་ཨོ་བྷ་མས་གསར་འགོད་པར། ཐེངས་འདིར་སྲིད་འཛིན་པུ་ཊེན་ལ་ཨུ་རུ་སུ་དང་སི་རི་ཡིའི་མཉམ་འབྲེལ་དམག་དཔུང་གིས་སི་རི་ཡིའི་ནང་མི་གསོད་ཁྲག་སྦྱོར་གྱི་ཟིང་འཁྲུག་སློང་བཞིན་པའི་གནས་སྟངས་ཐད་ཨ་རིའི་གཞུང་ལ་བློ་འཚུབ་ཡོད་པ་དང་། སི་རི་ཡ་དང་ཨུ་རུ་སུ་གཉིས་ནས་ད་ལྟའི་སི་རི་ཡའི་རྒྱལ་ནང་དུ་འབྱུང་བཞིན་པའི་ཟིང་འཁྲུག་མཚམས་བཞག་སྟེ་ཆབ་སྲིད་ཀྱི་གནས་སྟངས་ལ་འཕོ་འགྱུར་གཏོང་དགོས་པ། དེ་བཞིན། ཨ་རིའི་ཕྱོགས་ནས་ནམ་རྒྱུན་ཨུ་རུ་སུ་དང་ཨ་རི་གཉིས་བར་ལྷག་བསམ་རྣམ་དག་འབྲེལ་ལམ་འཛུགས་ཐབས་ལ་འབད་བརྩོན་བྱེད་ཀྱིན་ཡོད་སྐོར་སོགས་བརྗོད་ཡོད་སྐོར་བསྒྲགས་འདུག དེ་བཞིན། ཡུ་ཁུ་རན་ནང་གི་ཟིང་འཁྲུག་ཞི་འཇགས་ཡོང་ཐབས་སུ་ཨུ་རུ་སུའི་གཞུང་གིས་ཧྥ་རན་སི་དང་འཇར་མན། ཡུ་ཁུ་རན་དང་ཨ་རིའི་གཞུང་བཅས་བར་གྱི་ཞི་འགྲིག་གྲོས་ཆོད་ལག་བསྟར་བྱ་དགོས་སྐོར་སོགས་བཤད་ཡོད་པ་རེད། ཨུ་རུ་སུའི་སྲིད་འཛིན་ཕུ་ཊེན་གྱིས་གསར་འགོད་པར། ཐེངས་འདིའི་ཨ་རིའི་སྲིད་འཛིན་ཨོ་བྷ་མ་དང་ཁོང་གཉིས་བར་གྱི་ཐུག་འཕྲད་དེར་གྲུབ་འབྲས་ཡག་པོ་བྱུང་ཡོད་སྐོར་དང་། ཨ་རིའི་སྲིད་འཛིན་ཨོ་བྷ་མ་ཨུ་རུ་སུར་ག་དུས་ཡོང་འདོད་ན། དེ་དུས་ཁོང་གིས་སྲིད་འཛིན་ཨོ་བྷ་མར་ཐུག་འཕྲད་བྱ་འདོད་ཡོད། ཅེས་བཤད་ཡོད་སྐོར་བརྗོད་ཀྱང་། ཡུ་ཁུ་རན་དང་སི་རི་ཡའི་དམག་འཁྲུག་ཐོག་གྲོས་མོལ་ཅི་བྱུང་སྐོར་གསལ་བསྒྲགས་བྱས་མེད། ཨ་རི་ཐེངས་འདིའི་ཞི་བདེ་རྒྱ་མཚོའི་མངའ་ཁུལ་གྱི་ཨེ་ཤ་ཡའི་དཔལ་འབྱོར་མཉམ་ལས་སྒྲིག་འཛུགས་ཀྱི་ཚོགས་ཆེན་ཐོག་མཉམ་ཞུགས་བྱས་པའི་དམིགས་འབེན་གཙོ་བོ་ནི། སྒྲིག་འཛུགས་དེའི་ཁོངས་ཞུགས་རྒྱལ་ཁབ་ ༡༢ ནུབ་རྒྱ་མཚོ་ཆེན་མོའི་ཉོ་འཚོང་གྲོས་ཆིངས་སྒྲིག་འཛུགས་ (TPP) ནང་ཞུགས་ཏེ་ཨེ་ཤ་ཡའི་མངའ་ཁུལ་ཁག་ནང་རང་དབང་གི་ཉོ་འཚོང་བྱ་རྒྱུ་དེ་ཡིན་ནའང་། ད་ལོའི་ཨ་རིའི་སྲིད་འཛིན་དེ་སྤྱི་མཐུན་ཚོགས་པར་ཐོབ་ཡོད་སྟབས། ད་ལྟའི་ཨ་རིའི་དམངས་གཙོའི་གཞུང་གི་དཔལ་འབྱོར་འཚོལ་སྒྲུབ་ཀྱི་ལས་གཞི་དེ་སེམས་འཁྲལ་དུ་གྱུར་ཡོད་ཅིང་། རྒྱ་ནག་གི་དམིགས་འབེན་གཙོ་བོ་ནི། ཞི་བདེ་རྒྱ་མཚོའི་མངའ་ཁུལ་གྱི་ཨེ་ཤ་ཡའི་དཔལ་འབྱོར་མཉམ་ལས་སྒྲིག་འཛུགས་ཀྱི་ཚོགས་ཞུགས་རྒྱལ་ཁབ་ ༢༡ ཚང་མ་ནུབ་རྒྱ་མཚོ་ཆེན་མོའི་ཉོ་འཚོང་གྲོས་ཆིངས་སྒྲིག་འཛུགས་ནང་ཞུགས་ཏེ་ནུབ་རྒྱ་མཚོ་ཆེན་མོའི་མངའ་ཁུལ་དུ་དཔལ་འབྱོར་འཛུགས་སྐྲུན་བྱ་རྒྱུ་དེ་ཡིན་འདུག ཞི་བདེ་རྒྱ་མཚོའི་མངའ་ཁུལ་གྱི་ཨེ་ཤ་ཡའི་དཔལ་འབྱོར་མཉམ་ལས་སྒྲིག་འཛུགས་ཀྱི་ཚོགས་ཆེན་དེ་ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༩ ནས་ ༢༠ བར་ཚོགས་ཡོད་པ་དང་། ཚོགས་ཆེན་ཐོག་ཚོགས་ཞུགས་རྒྱལ་ཁབ་ ༢༡ གི་སྲིད་འཛིན་དང་སྲིད་བློན། དེ་བཞིན་ཚོང་དོན་སྐུ་ཚབ་སོགས་ཁྱོན་མི་གྲངས་ ༡༠༠༠ ལྷག་མཉམ་ཞུགས་བྱས་ཡོད་འདུག ཨེ་ཤ་ཡའི་དཔལ་འབྱོར་མཉམ་ལས་སྒྲིག་འཛུགས་བརྒྱུད་རྒྱལ་སྤྱིའི་ཐོན་སྐྱེད་རྩ་འབོར་བརྒྱ་ཆ་ ༥༧ དང་རྒྱལ་སྤྱིའི་དངོས་ཟོག་ཉོ་འཚོང་བརྒྱ་ཆ་ ༤༨། དེ་བཞིན། རྒྱལ་སྤྱིའི་དངོས་ཟོག་འཛད་སྤྱོད་མི་འབོར་བརྒྱ་ཆ་ ༤༠ ཟིན་གྱི་ཡོད་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
