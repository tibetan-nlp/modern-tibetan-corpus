ད་ལྟ་རྩྭ་ཐང་སྲིད་གཞུང་ལ་བདག་གི་རེད།

  མགོ་ལོག་རྨ་སྟོད་སྐྱ་རང་ས་ཁུལ་གྱི་འབྲོག་པ་རྣམས་གྲོང་བརྡལ་ཅན་དུ་བསྒྱུར་རྗེས། ད་ལྟ་སྲིད་གཞུང་གིས་རྩྭ་ཐང་དོ་དམ་ཚོགས་ཆུང་གི་མིང་ཐོག་ནས་འབྲོག་ཁུལ་ས་ཁུག་དག་ཏུ་གཏེར་ཁ་སྔོག་མགོ་བརྩམས་ཡོད་འདུག   མཚོ་སྔོན་ཞིང་ཆེན་མགོ་ལོག་ཁུལ་རྨ་སྟོད་རྫོང་སྐྱ་རང་ཡུལ་མཚོའི་རུ་ཁག་དྲུག་པའི་འབྲོག་པ་ཚོ་གྲོང་བརྡལ་དུ་གནས་སྤོ་བྱས་ཏེ་ད་ལྟ་ལོ་ ༤ ཙམ་འགྲོ་བཞིན་ཡོད་པ་དང་། དེང་སྐབས་ས་གནས་དེར་སྲིད་གཞུང་གི་རྩྭ་ཐང་དོ་དམ་ཚོགས་ཆུང་ཟེར་བ་ཞིག་བཙུགས་ཡོད་འདུག་ལ། ཚོགས་ཆུང་དེ་ཀྲུང་ཧྭ་མི་དམངས་སྤྱི་མཐུན་རྒྱལ་ཁབ་མཚོ་སྔོན་ཞིང་ཆེན་གྱི་རྩྭ་ཐང་བཅའ་ཁྲིམས་ཟེར་བ་ལག་བསྟར་བྱེད་པར་བཙུགས་པ་ཞིག་ཡིན་འདུག   ཚོགས་ཆུང་དེས་ ༢༠༡༦ ལོའི་ལོ་མགོ་ནས་བཟུང་འབྲོག་ས་ཁུག་ཀྱོག་དག་ཏུ་ཕྱིན་ཏེ། གསེར་དང་ཕྱེ་མའི་གཏེར་ཁ་སྔོག་ནས་དབོར་འདྲེན་བྱེད་མགོ་བཙུགས་ཡོད་ཅིང་། གནས་ཚུལ་དེ་ས་གནས་འབྲོག་པ་ཚོས་རྫོང་སྲིད་གཞུང་ལ་སྙན་གསང་ཞུས་སྐབས་རྫོང་སྲིད་གཞུང་གིས། “ཁྱེད་རང་ཚོ་འབྲོག་ས་ནས་རུ་སྤོར་ཏེ་ད་ལྟ་གྲོང་བརྡལ་དུ་བསྡད་ཡོད་པ་དང་། སྲིད་གཞུང་གིས་ཁྱེད་ཅག་ལ་ཁང་པ་དང་སྒོར་མོ་སོགས་སྤྲད་ཡོད་པ་རེད། རྩྭ་ཐང་གི་བདག་དབང་ད་ལྟ་སྲིད་གཞུང་ལ་ཡོད་སྟབས། ཁྱེད་རང་ཚོས་སེམས་ཁྲལ་བྱེད་དགོས་པ་མ་རེད།” ཅེས་སོགས་བརྗོད་ཡོད་འདུག   རྨ་སྟོད་རྫོང་སྲིད་གཞུང་གིས། རྩྭ་ཐང་དོ་དམ་ཚོགས་ཆུང་དེའི་ལས་དོན་གཙོ་བོ་ནི་ལས་ཁུང་དང་མི་སྒེར་སུ་ཡིན་རུང་ཚང་མར་རྩྭ་ཐང་བཅའ་ཁྲིམས་དང་ཁྲིམས་སྲོལ་སྒྲིག་གཞི་ཁག་ལ་འགལ་བ་དང་། རྩྭ་ཐང་གཏོར་བརླག་བྱེད་བའི་བྱ་སྤྱོད་རིགས་ལ་ལྟ་སྐུལ་དང་མཚང་འདོན་ཞུ་གཏུག་བྱེད་པའི་དབང་ཆ་ཡོད་སྐོར་སྒྲོག་གི་ཡོད་འདུག   དེ་ཡང་སྔོན་ས་ཆ་དེར་འབྲོག་ཁྱིམ་ ༢༧༠ ཡས་མས་ཡོད་ཀྱང་ད་ལྟ་ཕྱུག་ཟོག་སྐྱོང་མཁན་འབྲོག་པ་ ༢༠ ཙམ་ལས་ལྷག་མེད་པ་དང་། རྩྭ་ཐང་བཅའ་ཁྲིམས་ཟེར་བ་དེ་ལྟར་ན། འབྲོག་པའི་ཁྱིམ་ཚང་རེར་ཟོག་ ༡༢ དང་ལུག་ ༣༠ ལས་ཉར་མི་ཆོག་པ་དང་། དེ་ལས་བརྒལ་ན་ཉིན་རེར་ནོར་གཅིག་རེར་སྒོར་ ༤༠ དང་ལུག་རེར་སྒོར་ ༣༠ ཡི་ཉེས་ཆད་གཅོད་རྒྱུ་ཡིན་སྐོར་འཁོད་ཡོད་འདུག   སྦེལ་ཀོབ་ས་གནས་གསར་འགོད་པ་དོན་འགྲུབ་བཀྲ་ཤིས།
