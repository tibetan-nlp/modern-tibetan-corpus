བོད་ནང་རྟོག་ཞིབ་བྱ་བ་མགོ་འཛུཊ།

ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༧ ཙམ་ནས་བཟུང་དུས་ཡུན་ཟླ་ ༢ རིང་རྒྱ་ནག་དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གི་རྟོག་ཞིབ་ཚན་པས་བོད་རང་སྐྱོང་ལྗོངས་དང་མཚོ་སྔོན་སོགས་སུ་ལྐོག་ཟ་རུལ་སུངས་ལ་རྟོག་ཞིབ་བྱེད་མགོ་བཙུགས་འདུག ཉེ་སྔོན་རྒྱ་ནག་དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གིས་ཞིང་ཆེན་ཁག་ ༡༣ ནང་རྟོག་ཞིབ་ཐེངས་དང་པོ་བྱས་པ་མཇུག་སྒྲིལ་ཏེ། ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༦ ཉིན་འདི་ལོའི་རྟོག་ཞིབ་ཐེངས་གཉིས་པའི་གྲ་སྒྲིག་ཚོགས་འདུ་པེ་ཅིན་དུ་བསྡུས་ཡོད་པ་དང་། ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༦ ཉིན་འདི་ལོའི་ཐེངས་གཉིས་པའི་རྟོག་ཞིབ་བྱ་ཡུལ་ཞིང་ཆེན་ ༡༠ ཡི་མིང་གཞུང་ཕྱིར་བསྒྲགས་བྱས་པའི་ནང་དུ། བོད་རང་སྐྱོང་ལྗོངས་དང་མཚོ་སྔོན། སི་ཁྲོན། ཀང་ཞི། ཧྲང་ཧེ། ཀྲེ་ཅང་། ཧུའུ་པེ། ཧེ་ལོང་ཅང་སོགས་ཡོད་པ་དང་། དེ་ཚོ་དང་མཉམ་དུ་རྒྱལ་ཁབ་ལུས་རྩལ་སྤྱི་གཉེར་ཁང་དང་རྒྱ་ནག་ཚན་རིག་གླིང་། རྒྱ་ནག་རླངས་འཁོར་བཟོ་ཚོང་ཁང་དང་པོ་བཅས་ལའང་དུས་ཡུན་ཟླ་ ༢ རིང་གི་རྟོག་ཞིབ་བྱ་བ་མགོ་འཛུགས་བྱས་ཡོད་འདུག རྟོག་ཞིབ་ཐེངས་གཉིས་པའི་གྲ་སྒྲིག་ཚོགས་འདུའི་ཐོག དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གི་མགོ་ཁྲིད་ཝཱང་ཅི་ཧྲན་གྱིས། ལྐོག་ཟ་རུལ་སུངས་ཀྱི་གནད་དོན་རྟོག་ཞིབ་བྱེད་སྐབས་གཏེར་ཁའི་ཐོན་ཁུངས་དང་ས་ཆ་ཉོ་ཚོང་། འཛུགས་བསྐྲུན་ལས་རིགས། དམངས་ཕན་དངུལ་དངོས། ཆེད་ལས་འགྲོ་གྲོན་དོ་དམ་སོགས་ཀྱི་ཐོག་ལ་དམིགས་བསལ་གྱིས་དོ་ནན་བྱས་ཏེ། གང་ལ་གནད་དོན་ཡོད་ན་དེ་ལ་རྟོག་ཞིབ་ནན་ཏན་བྱེད་སྐོར་བརྗོད་ཡོད་འདུག ཕྱི་ལོ་ ༢༠༡༢ ཟླ་ ༡༢ ནས་བཟུང་ད་ལྟའི་བར་རྒྱ་ནག་གི་ཞིང་ཆེན་དང་རང་སྐྱོང་ལྗོངས་སོགས་ ༣༡ ཙམ་གྱི་ནང་རྟོག་ཞིབ་བྱས་ཏེ། ལས་བྱེད་ཁྱོན་ ༥༢༡ ལྷག་ལ་ནག་ཉེས་ཕོག་ཡོད་པ་དང་། དེའི་ནང་རྒྱལ་ཁབ་རིམ་པའི་མཐོ་རིམ་དཔོན་རིགས་ ༢ དང་ཞིང་ཆེན་རིམ་པའི་དཔོན་རིགས་ ༣༧ ཡོད་འདུག དཔོན་རིགས་ནག་ཉེས་ཕོག་པ་མང་ཤོས་སི་ཁྲོན་དང་ཧུའུ་པེ། ཧྥི་ཅན་བཅས་ཡིན་ཞིང་། སི་ཁྲོན་དུ་མི་གྲངས་ ༤༡ དང་ཧུའུ་པེར་ ༣༦། ཧྥི་ཅན་དུ་ ༣༢ བཅས་ལ་ཁྲིམས་འགལ་གྱི་ནག་ཉེས་ཕོག་ཡོད་འདུག འདས་ལོའི་ལོ་མཇུག་ནས་བཟུང་གསར་འགྱུར་བརྒྱུད་ལམ་ཁག་ཏུ། རྒྱ་ནག་དབུས་ཆབ་སྲིད་ལྷན་ཁང་གི་མགོ་ཁྲིད་ཟུར་པ་ཀྲུའུ་ཡཱོང་ཁང་དང་ཁོ་བོའི་འབྲེལ་ཡོད་མི་སྣ་མང་པོ་བཀག་ཉར་བྱས་ཡོད་པའི་གནས་ཚུལ་རིམ་པར་ཐོན་ཡོད་ཀྱང་། རྒྱ་ནག་གཞུང་ཕྱོགས་ནས་ཀྲུའུ་ཡཱོང་ཁང་བཀག་ཉར་འོག་ཡོད་པའི་གནས་ཚུལ་ཐད་བདེན་རྫུན་གྱི་གསལ་ཁ་གང་ཡང་བཟོས་མེད་ཀྱང་། ཁ་སང་རྒྱ་ནག་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གི་དྲ་རྒྱའི་ཐོག ཀྲུའུ་ཡཱོང་ཁང་གིས་ཁྲིམས་འགལ་གྱི་ནག་ཉེས་ཚབས་ཆེན་བཟོས་པའི་དོགས་གཞིའི་འོག ༼རྒྱ་ནག་གུང་ཁྲན་ཏང་གི་སྒྲིག་ཁྲིམས༽དང་༼གུང་ཁྲན་ཏང་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གི་ཉེས་དོན་ཞིབ་བཤེར་བྱ་བའི་གཏན་འབེབས༽ཞེས་པ་གཞིར་བཟུང་སྟེ། ཀྲུའུ་ཡི་ཉེས་དོན་ཞིབ་བཤེར་བྱ་རྒྱུ་གཏན་འཁེལ་བྱས་ཡོད་ལུགས་གཞུང་འབྲེལ་ཐོག་ཕྱིར་བསྒྲགས་ཐོག་མ་བྱས་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
