འཕྲིན་ལས་རྒྱ་མཚོ་ལཊ་ཀློད་གྲོལ།

ཁ་སང་ཕྱི་ཚེས་ ༤ ཉིན་རྔ་བ་རྫོང་རྟོགས་ལྡན་དགོན་གྱི་གྲྭ་འཕྲིན་ལས་རྒྱ་མཚོ་ལགས་བཙོན་འཇུག་དུས་ཡུན་རྫོགས་ཏེ་ཀློད་གྲོལ་བཏང་ཡོད་འདུག འདི་གར་འབྲེལ་ཡོད་ཅིག་གིས་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། རྔ་བ་རྫོང་རྟོགས་ལྡན་དགོན་གྱི་གྲྭ་འཕྲིན་ལས་རྒྱ་མཚོ་ལགས་ཀྱིས། ཕྱི་ལོ་ ༢༠༠༨ ལོར་རྔ་བ་རྫོང་ཁོངས་སུ་བྱུང་བའི་ཞི་རྒོལ་གྱི་དངོས་ཡོད་གནས་ཚུལ་ཕྱི་ཕྱོགས་སུ་མཁོ་སྤྲོད་གནང་བའི་ནག་ཉེས་འོག རྒྱ་གཞུང་གིས་ཁོང་ལ་ལོ་ ༧ གྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་དེ་སི་ཁྲོན་མེན་ཡཱང་བཙོན་ཁང་དུ་བཅུག་ཡོད་པ་ལྟར། ཁ་སང་ཕྱི་ཚེས་ ༤ ཉིན་བཙན་འཇུག་དུས་ཡུན་རྫོགས་ནས་བཙོན་གྲོལ་བཏང་ཡོད་པ་དང་།  དེ་རིང་སྔ་དྲོར་རྟོགས་ལྡན་དགོན་གྱི་བླ་སྤྲུལ་གྲྭ་མང་གི་གཙོས་སེར་སྐྱ་མི་མང་གི་ཕེབས་བསུའི་ཁྲོད་ཕྱིར་རང་དགོན་དུ་འབྱོར་ཡོད་པ་མ་ཟད། ༼མི་དཀར་དཀར། ཁྲིམས་ནག་ནག༽ཅེས་པའི་འཕྲེད་དར་འཕྱར་ཏེ། རྒྱ་གཞུང་གིས་ཉེས་མེད་བསྙོན་འཛུགས་བྱས་ནས་མི་ལོ་ཧྲིལ་པོ་ ༧ རིང་བཙོན་ཁང་དུ་དཀའ་སྡུག་མནར་གཅོད་མྱངས་བ་དང་། ལྷག་པར་དུ་ཁོང་གིས་རང་མི་རིགས་ལ་བཅིང་བའི་ལར་ཞེན་དང་དྲང་བདེན་ལ་ཕྱོགས་པའི་ཆོད་སེམས་སོགས་ལ་རྗེས་སུ་ཡི་རང་དང་དགའ་གུས་མཚོན་པར་བྱས་ཡོད་འདུག ད་ལྟ་དགུང་ལོ་ ༣༧ ཡིན་པའི་གྲྭ་འཕྲིན་ལས་རྒྱ་མཚོའམ་འཕྲིན་ཁོ་ལགས་ནི། རྔ་བ་རྫོང་རྔ་སྨད་ཕྱུག་ཚང་ཡུལ་ཚོའི་ཕ་ནས་པེ་དང་མ་ཀུན་རིག་གཉིས་ཀྱི་བུ་ཡིན་པ་དང་། ཁོང་ལ་ཕ་མ་གཅིག་པའི་མིང་སྲིང་ལྔ་ཡོད་པའི་ནང་ནས་ཆེ་བ་དེ་ཡིན་ལ། ཆུང་དུས་ནས་གཡུང་དྲུང་བོན་གྱི་གདན་ས་ཆེན་མོ་རྟོགས་ལྡན་དགོན་དུ་ཞུགས་ཏེ་སློབ་གཉེར་གནང་ཡོད་པ་དང་། ནམ་རྒྱུན་གཤིས་སྤྱོད་ལེགས་ཤིང་འཛིན་གྲྭའི་ནང་གི་སློབ་སྦྱོང་ཡག་ཤོས་ཤིག་ཡིན་སྟབས། ཁོང་གིས་ཁྲིམས་འགལ་གྱི་བྱ་སྤྱོད་ཅིག་སྤེལ་འདུག་ཟེར་ན་ཀུན་གྱིས་ཡིད་ཆེས་བྱེད་ཐབས་མེད་ཀྱང་། ཕྱི་ལོ་ ༢༠༠༨ ཟླ་ ༤ ཚེས་ ༤ ཉིན་རྒྱ་གཞུང་ཉེན་རྟོག་པ་མང་པོ་རྟོགས་ལྡན་དགོན་དུ་ཡོང་སྟེ། གྲྭ་མང་ཚོགས་ཚོགས་བཞིན་པའི་སྐབས་སུ་ཁོང་ལ་ཉེས་མེད་བསྙོན་འཛུགས་ཀྱིས་བཙན་ཁྲིད་བྱས་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
