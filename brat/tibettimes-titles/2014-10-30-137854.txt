ཉི་ཧོང་དང་རྒྱ་ནག་གི་འབྲེལ་མོལ།

ཁ་སང་པེ་ཅིན་མི་མང་ཚོགས་ཁང་ཆེན་མོའི་ནང་ཉི་ཧོང་གི་སྲིད་བློན་ཟུར་པ་ཧྥི་ཐན་ཁང་ཧྥོ་ལགས་ཀྱིས་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་ལ་ཐུག་འཕྲད་གནང་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༨ ཉིན་ཉི་ཧོང་གི་སྲིད་བློན་ཟུར་པ་ཧྥི་ཐན་ཁང་ཧྥོ་ལགས། བྷོ་ཨོ་ཨེ་ཤ་ཡའི་གླེང་སྟེགས་ལས་འཛིན་དཔོན་གྱི་མཚན་ཐོག་ནས་ལས་འཛིན་ལྷན་ཚོགས་ཀྱི་གནད་ཡོད་མི་སྣ་དང་མཉམ་དུ་རྒྱ་ནག་ནང་ཕྱིན་ཏེ། ཁ་སང་ཕྱི་ཚེས་ ༢༩ ཉིན་པེ་ཅིན་མི་མང་ཚོགས་ཁང་ཆེན་མོའི་ནང་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་ལ་ཐུག་འཕྲད་གནང་ཡོད་འདུག སྐབས་དེར་ཞི་ཅིན་ཕིན་གྱིས་ཧྥི་ཐན་ཁང་ཧྥོ་ལགས་སོགས་ཀྱི་འབད་བརྩོན་འོག ཉེ་བའི་ལོ་ཤས་ནང་བྷོ་ཨོ་གླེང་སྟེགས་ཀྱིས་ཡ་གླིང་མཉམ་འབྲེལ་ལ་ཡར་སྐུལ་དང་ལོ་རེ་བཞིན་ཤུགས་རྐྱེན་ཆེ་རུ་འགྲོ་བཞིན་ཡོད་སྐོར་བརྗོད་ཡོད་པ་དང་། ཧྥི་ཐན་ཁང་ཧྥོ་ཡིས་ཡ་གླིང་འཕེལ་རྒྱས་ཕྱིན་པ་དང་བསྟུན་ནས་རྒྱ་ནག་གིས་ནུས་པ་ཆེན་པོ་འདོན་བཞིན་ཡོད་པ་ལྟར་དེར་ལས་འགན་ཡང་ཡོད་པས། ལས་འགན་དེ་ཚོ་ཇི་ལྟར་ལག་བསྟར་བྱེད་མིན་དེ་རྒྱ་ནག་ལ་འཕྲད་པའི་གལ་འགངས་ཆེ་བའི་གནད་དོན་ཞིག་ཆགས་ཡོད་སྐོར་བརྗོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༢ ཉིན། ཉི་ཧོང་གི་སྲིད་བློན་གཞོན་པ་ཐཱ་རོ་ཨེ་སོ་(Tarō Asō)ལགས་དང་རྒྱ་ནག་གི་སྲིད་བློན་གཞོན་པ་ཀྲང་ཀོ་ལི་གཉིས་ཐུག་འཕྲད་བྱས་ཏེ། འདི་ལོའི་ཕྱི་ཟླ་ ༡༡ པའི་ནང་། པེ་ཅིན་དུ་འཚོགས་འཆར་ཡོད་པའི་ཡ་གླིང་ཞི་བདེ་རྒྱ་མཚོའི་དཔལ་འབྱོར་མཉམ་ལས་ལྷན་ཚོགས་སྐབས། རྒྱལ་ཁབ་གཉིས་ཀྱི་སྲིད་འཛིན་དང་སྲིད་བློན་ཐུག་འཕྲད་བྱས་ཏེ། ཕྱོགས་གཉིས་ཀྱི་འཐབ་ཇུས་རོགས་ཕན་གྱི་འབྲེལ་བ་བསྐྱར་གསོ་བྱེད་དགོས་པའི་འབོད་སྐུལ་བྱས་ཡོད་པ་ལྟར། ཁ་སང་གསར་འགོད་ལྷན་ཚོགས་ཤིག་གི་ཐོག་རྒྱ་ནག་ཕྱི་འབྲེལ་ལྷན་ཁང་གི་མགོ་ཁྲིད་ཝང་ཡས་ཀྱིས། རྒྱ་ནག་གི་མི་ལ་ཚུར་ཡོང་ཐམས་ཅད་མགྲོན་པོར་རྩི་བའི་ལུགས་སྲོལ་ཡོད་སྐོར་དང་། ཉི་ཧོང་གཞུང་གིས་རྒྱ་ནག་དང་ཉི་ཧོང་བར་གྱི་གནད་དོན་ལ་གཙིགས་འཛིན་བྱས་ཏེ། ལྷག་བསམ་རྣམ་དག་གིས་གནད་དོན་ཐག་གཅོད་བྱེད་པའི་རེ་བ་ཡོད་སྐོར་བརྗོད་འདུག གནད་དོན་དབྱེ་ཞིབ་པ་ཚོས་ཝང་གི་སྐད་ཆ་དེའི་ནང་དོན་ནི། ཞི་བདེ་རྒྱ་མཚོའི་ཁུལ་གྱི་མཚོ་གླིང་གནད་དོན་དང་ལོ་རྒྱུས་ངོས་འཛིན་ཐད་ཤིན་ཛོ་ཨ་བྷི་ཡིས་བསམ་ཤེས་དགོས་པའི་སྔོན་བརྡ་བཏང་བ་ཡིན་སྐོར་བརྗོད་ཀྱི་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
