དགེ་རྒན་ ༥༠༠༠ ལྷག་གི་ངོ་རྒོལ།

རྒྱ་ནག་ཧི་ནན་ཞིང་ཆེན་ལོ་ཧི་གྲོང་ཁྱེར་གྱི་རྫོང་ཁག་གཉིས་ཀྱི་སློབ་གྲྭ་ཆུང་འབྲིང་གི་དགེ་རྒན་ ༥༠༠༠ ལྷག་གིས་ལས་མཚམས་བཞག་སྟེ་ངོ་རྒོལ་བྱེད་བཞིན་ཡོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༨ ཉིན་ནས་རྒྱ་ནག་ཧི་ནན་ཞིང་ཆེན་ལོ་ཧི་གྲོང་ཁྱེར་གྱི་ཝུའུ་ཡཱང་དང་ལིན་ཡེན་རྫོང་གཉིས་ཀྱི་སློབ་གྲྭ་ཆུང་འབྲིང་གི་དགེ་རྒན་ཚོས་ལས་མཚམས་བཞག་སྟེ། ཅིའི་ཕྱིར་དགེ་རྒན་གྱི་གླ་ཆ་ཁྲིམས་ལྟར་ལག་བསྟར་མི་བྱེད་པ་རེད་ཅེས་པའི་འཕྲེད་དར་བཟུང་ནས་རྫོང་སྲིད་གཞུང་གི་ལས་ཁུངས་སྒོ་དང་། རླངས་འཁོར་འབབ་ཚིགས་སོགས་སུ་ཁྲོམ་སྐོར་དང་ཟས་བཅད་ངོ་རྒོལ་བྱས་ནས་ཉིན་ ༡༤ འགྲོ་བཞིན་ཡོད་པ་དང་། སློབ་གྲྭ་དེ་དག་གི་སློབ་མ་ཚོས་རང་སྦྱོང་བྱས་ནས་བསྡད་ཡོད་འདུག ལས་འགུལ་ཁྲོད་རྒྱ་གཞུང་ཉེན་རྟོག་པས་སྣེ་ཁྲིད་དགེ་རྒན་མང་པོར་གཅར་རྡུང་དང་ཁག་ཅིག་འཛིན་བཟུང་བྱས་ཡོད་འདུག ལས་འགུལ་ནང་ཞུགས་པའི་དགེ་རྒན་ཞིག་གིས་གསར་འགོད་པར་བརྗོད་རྒྱུར། “སྲིད་གཞུང་གིས་རྣམ་འགྱུར་མཁྲེགས་པོ་བཟུང་སྟེ་ཁྲིམས་ལྟར་གནད་དོན་ཐག་གཅོད་མི་བྱེད་པ་མ་ཟད། ངོ་རྒོལ་བྱས་ན་མེ་མདའ་རྒྱག་རྒྱུ་ཡིན་ལུགས་དང་། མུ་མཐུད་ལས་འགུལ་ནང་འཛུལ་ཞུགས་བྱེད་མཁན་ཚོ་འཛིན་བཟུང་བྱེད་རྒྱུ་ཡིན་པའི་འཇིགས་སྐུལ་བྱེད་ཀྱི་འདུག ད་དུང་གྲོང་ཁྱེར་ནང་ངོ་རྒོལ་ལས་འགུལ་ལ་སྐྱོན་བརྗོད་བྱེད་མཁན་གྱི་ཚོགས་པ་བཙུགས་ཏེ། སློབ་གྲྭ་ཁག་ཏུ་ཆེད་སྐྱོད་ཀྱིས་སློབ་སྤྱི་ཚོར་དགེ་རྒན་རྣམས་ངོ་རྒོལ་ནང་ཞུགས་པར་བཀག་འགོག་དགོས་པའི་བཀའ་རྒྱ་བཏང་འདུག ཡིན་ནའང་། གནད་དོན་འདི་ཐག་གཅོད་མ་བྱས་ན་ད་རེས་གཟའ་མཇུག་ཏུ་ངོ་རྒོལ་ལས་འགུལ་སྔར་ལས་ཤུགས་ཆེན་པོ་སྤེལ་སྲིད་པ་རེད། ”ཅེས་བརྗོད་འདུག སློབ་མའི་ཁྱིམ་བདག་ཞིག་གིས། “ཟླ་འདིའི་ཕྱི་ཚེས་ ༨ ཉིན་ནས་བཟུང་དགེ་རྒན་ཚོས་ལས་མཚམས་བཞག་ཡོད་སྟབས། བྱིས་པ་སློབ་གྲྭར་བསྐྱལ་ཡང་སློབ་ཁྲིད་བྱེད་མཁན་མི་འདུག དཀའ་ངལ་འདི་ང་ཚོ་ཚོད་ལྟའི་སློབ་འབྲིང་དང་པོ་གཅིག་པུར་འཕྲད་པ་མ་ཡིན་པར་རྫོང་ཡོངས་ཀྱི་སློབ་གྲྭ་ཚང་མར་འཕྲད་ཀྱི་འདུག དགེ་རྒན་ཚོར་གླ་ཆ་ཉུང་བའི་ཁར་ཁ་གཡར་ས་མང་པོ་བཙལ་ཏེ་གླ་ཆ་གཅོག་གི་འདུག སྲིད་གཞུང་གི་བྱ་སྤྱོད་ཚད་ལས་བརྒལ་མེད་ན་དགེ་རྒན་ཚོས་ལས་མཚམས་གཏན་ནས་འཇོག་གི་མ་རེད། ”ཅེས་བརྗོད་འདུག ཐེངས་འདིར་དགེ་རྒན་ཚོས་དྲ་ཐོག་ཏུ་སྤེལ་བའི་སྙན་ཞུའི་ནང་། ཕྱི་ལོ་ ༢༠༡༠ ལོར་འགྲེམས་སྤེལ་བྱས་པའི་ཡིག་ཆ་ལྟར་ན། ལོ་དེའི་ཕྱི་ཟླ་ ༧ པ་ནས་བཟུང་ལས་དོན་སྡེ་ཚན་ཁག་གི་ལས་བྱེད་པའི་གླ་ཆར་ལྟ་ཞིབ་བྱས་ཏེ། ལོ་ཧི་གྲོང་ཁྱེར་གྱི་དགེ་རྒན་ཡོངས་ལ་ཟླ་རེར་གླ་ཕོགས་རྒྱ་སྒོར་ ༣༠༠ རེ་འཕར་སྣོན་བྱ་རྒྱུ་ཡིན་པའི་གཏན་འབེབས་བྱུང་ཡོད་ཀྱང་། ཝུའུ་ཡཱང་རྫོང་དུ་ ༢༠༡༢ ཕྱི་ཟླ་ ༧ ནང་གཞི་ནས་ལག་བསྟར་བྱས་ཏེ་ཟླ་རེར་རྒྱ་སྒོར་ ༡༥༠ འཕར་སྣོན་བྱུང་ཡོད། ཡིན་ནའང་། འདས་པའི་ལོ་གཉིས་རིང་མི་རེར་ཁྱོན་བསྡོམས་རྒྱ་སྒོར་ ༣༦༠༠ ཆད་ཡོད། ༢༠༡༢ ལོའི་སྲིད་གཞུང་གི་ཡིག་ཆ་ལྟར་ན། ལོ་ཧི་གྲོང་ཁྱེར་གྱི་དགེ་རྒན་གཅིག་ལ་ལོ་རེར་གླ་ཕོགས་རྒྱ་སྒོར་ ༢༢༠༠༠ སྤྲོད་དགོས་པའི་གཏན་འབེབས་བྱས་ཡོད་ཀྱང་། ཝུའུ་ཡཱང་རྫོང་གི་དགེ་རྒན་རེར་རྒྱ་སྒོར་ ༩༨༠༠ ལས་མེད་པ་དང་། དེ་བཞིན། ༢༠༡༣ ལོར་གྲོང་ཁྱེར་ཡོངས་ཀྱི་དགེ་རྒན་རེར་རྒྱ་སྒོར་ ༢༥༦༠༠ སྤྲོད་དགོས་ཀྱང་། དགེ་རྒན་རེར་རྒྱ་སྒོར་ ༡༢༠༠༠ ལས་སྤྲད་མེད་ཅེས་བཀོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
