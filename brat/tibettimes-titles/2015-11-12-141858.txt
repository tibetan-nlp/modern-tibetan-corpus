མནར་གཅོད་གཏོང་སྟངས་གསར་པ་རེད།

  རྒྱ་ནག་གི་བཙོན་ཁང་ཁག་ཏུ་བཙོན་པ་ཚོར་ཟས་བཅད་གཉིད་བཀག་གི་མནར་གཅོད་འོག་འདྲི་གཅོད་བྱེད་ཀྱི་ཡོད་པའི་གནས་ཚུལ་སྐོར་རྒྱལ་སྤྱིའི་ཉེ་ཡངས་ཚོགས་པའི་ཞིབ་འཇུག་སྙན་ཐོ་བཏོན་འདུག   དེ་རིང་ཨེ་ཕེ་ (AP) གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། རྒྱལ་སྤྱིའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལྟ་ཞིབ་ཚོགས་པས་ཞིབ་འཇུག་སྙན་ཐོ་ཞིག་བཏོན་པའི་ནང་། རྒྱ་ནག་གི་བཀག་བསྐྱིལ་ཁང་ཁག་ཏུ་བཙོན་པ་ཚོར་དུས་ཡུན་རིང་བོར་ལག་ལྕགས་དང་རྐང་ལྕགས་རྒྱག་པ་དང་། གཉིད་ཉལ་དུ་མི་འཇུག་པ། བཟའ་བཏུང་མི་སྤྲོད་པ། འཕྲོད་བསྟེན་ལ་བརྟག་དཔྱད་སྨན་བཅོས་མི་བྱེད་པ་བཅས་ཀྱི་གནས་སྟངས་འོག བཙན་ཤེད་ཀྱིས་འདྲི་གཅོད་དང་མནར་གཅོད་གཏོང་གི་ཡོད་འདུག   ཕྱི་ལོ་ ༢༠༡༠ ནས་བཟུང་རྒྱ་གཞུང་གིས་ཉེས་དོན་བཅའ་ཁྲིམས་ལ་བསྒྱུར་བཅོས་བཏང་སྟེ། རྒྱལ་ནང་གི་བཙོན་པ་ཚོའི་ལུས་སེམས་གཉི་གར་གནོད་འཚེ་ཚབས་ཆེན་ཐེབས་ཀྱི་ཡོད་པར་དོགས་གཞི་བྱེད་ཀྱི་ཡོད་པ་མ་ཟད། འདས་ལོའི་ཕྱི་ཟླ་ ༥ ནང་རྒྱལ་སྤྱིའི་ཉེས་ཡངས་ཚོགས་པས་རྒྱ་ནག་ནང་འགྲོ་བ་མིའི་ཐོབ་ཐང་སྐོར་གྱི་སྙན་ཐོ་ཞིག་བཏོན་པའི་ནང་དུའང་གོང་མཚུངས་ཀྱི་ངོས་འཛིན་བྱུང་ཡོད་འདུག   རྒྱལ་སྤྱིའི་ཉེས་ཡངས་ཚོགས་པ་དང་རྒྱལ་སྤྱིའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལྟ་ཞིབ་ཚོགས་པ་གཉིས་ཀྱི་ཞིབ་འཇུག་སྙན་ཐོ་གཉིས་ནང་། རྒྱ་གཞུང་གིས་སྔར་བཞིན་རྒྱལ་ནང་གི་བཙོན་པ་ཚོར་ཁྲིམས་མེད་ལུགས་ལྡོག་དང་མི་སྤྱོད་ལས་འདས་པའི་མནར་གཅོད་གཏོང་སྟངས་ལག་བསྟར་བྱེད་ཀྱི་ཡོད་སྐོར་འཁོད་ཡོད་འདུག   འདི་ལོའི་ཕྱི་ཟླ་ ༩ ནང་རྒྱ་ནག་གི་གསར་འགོད་པ་ལེའུ་ཧུ་ཟེར་བ་ཞིག་གིས་ཨེ་ཕེ་གསར་ཁང་ལ་བརྗོད་དོན། ཁོས་ཉེས་དོན་ཅི་ཡང་བསགས་མེད་ནའང་། ཁོ་རང་གཉིད་ལས་སད་དུས་ཁོ་བཀག་བསྐྱིལ་ཁང་ཞིག་གི་ནང་བཅུག་འདུག ཅེས་དང་། ཡང་རྒྱ་ནག་ནང་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལྟ་ཞིབ་ཚན་པའི་འགན་འཛིན་ལྕམ་སོ་ཧྥེ་རེ་ཆར་ཌི་སོན་ (Sophie Richardson) ལགས་ཀྱིས་བསྒྲགས་གཏམ་ཞིག་སྤེལ་ཏེ། རྒྱ་གཞུང་གིས་འཛམ་གླིང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་བསྐྱར་ཞིབ་གཞིར་བཟུང་ཐོག རྒྱལ་ནང་གི་བཙོན་པ་ཚོའི་ལུས་སེམས་གཉི་གར་མནར་གཅོད་གཏོང་གི་ཡོད་པ་དེར་དྲང་བདེན་གྱི་སྒོ་ནས་གྲོས་མོལ་བྱེད་དགོས། རྒྱ་ནག་གིས་བཙོན་པ་ཚོར་མི་སྤྱོད་ལས་འདས་པའི་མནར་གཅོད་གཏོང་གི་ཡོད་པ་དོན་དངོས་ཡིན་སྟབས། རྒྱ་ནག་གཞུང་གིས་གནད་དོན་འདིར་ཁ་གསལ་གྱི་ལན་སྤྲོད་ཐུབ་ཀྱི་མེད། མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་བསྐྱར་ཞིབ་སྙན་ཐོའི་ནང་རྒྱ་ནག་གིས་བཙོན་པ་ཚོའི་ལུས་སེམས་གཉི་གར་འཚེ་བ་བྱེད་ཀྱི་ཡོད་པའི་གནས་ཚུལ་དེ་ལ་རྒྱ་ནག་གི་དབང་འཛིན་པས་ཁས་ལེན་མི་བྱེད་པར་གཡོ་རྫུ་བྱེད་ཀྱི་ཡོད་པ་རེད། ཅེས་སོགས་བརྗོད་འདུག   གནས་ཚུལ་ལྟར་ན། རྒྱལ་སྤྱིའི་ཉེས་ཡངས་ཚོགས་པས་རྒྱ་ནག་རྒྱལ་ཡོངས་ཀྱི་ཁྲིམས་རྩོད་པ་ ༣༧ དང་ཁྲིམས་ཁང་ཁག་ནས་ཐག་བཅད་ཟིན་པའི་ཁྲིམས་ཡིག་ ༥༩༠། དེ་བཞིན། རྒྱ་ནག་གི་ཁྲིམས་ལུགས་ལ་ཞིབ་འཇུག་དང་བཅར་འདྲི་སོགས་བརྒྱུད་དེ་རྒྱ་ནག་ནང་གི་འགྲོ་བ་མིའི་ཐོབ་ཐང་གནས་སྟངས་སྐོར་ལ་ཞིབ་འཇུག་སྙན་ཐོ་བཏོན་པ་ཡིན་འདུག   ༢༠༡༤ ལོའི་ཕྱི་ཟླ་ ༤ ནང་རྒྱ་ནག་མི་དམངས་བདེ་འཇགས་ལྷན་ཁང་གི་ཉེན་རྟོག་བཀག་བསྐྱིལ་རྟོག་ཞིབ་ཚན་པའི་དབང་འཛིན་པ་ཀྲོ་ཁྲིན་ཀང་ཟེར་བ་དེས། བཙོན་པར་མནར་གཅོད་དང་འཚེ་བ་མི་ཆོག་པའི་བཅའ་ཁྲིམས་གསར་བ་ལག་བསྟར་བྱས་པ་ནས་བཟུང་། བཀག་བསྐྱིལ་ཁང་ཁག་ཏུ་བཙན་ཤེད་ཀྱིས་འདྲི་གཅོད་དང་མནར་གཅོད་བཏང་བའི་གནད་དོན་གཅིག་ཀྱང་མེད་ལུགས་བརྗོད་ཡོད་ནའང་། རྒྱལ་སྤྱིའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལྟ་ཞིབ་ཚོགས་པས་བཏོན་པའི་ཞིབ་འཇུག་སྙན་ཐོའི་སྐོར་ལ། རྒྱ་ནག་ཕྱི་སྲིད་ལྷན་ཁང་གི་མགྲིན་ཚབ་པ་ཧྭ་ཁྲིན་ཡེན་ཟེར་བ་ཞིག་གིས། རྒྱ་ནག་ནང་འགྲོ་བ་མིའི་ཐོབ་ཐང་གི་གནས་སྟངས་རྟོག་ཞིབ་སྐབས་སུ། བཙོན་པ་ཚོར་མནར་གཅོད་གཏོང་རྒྱུ་མཚམས་བཞག་ཡོད་སྟབས། དོན་དངོས་སུ་ཁོ་ཚོར་སུས་མནར་གཅོད་གཏོང་གི་ཡོད་མེད་རྟོགས་ཐབས་མེད་ལུགས་བརྗོད་པ་ཙམ་ལས། རྒྱལ་སྤྱིའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལྟ་ཞིབ་ཚོགས་པའི་ཞིབ་འཇུག་སྙན་ཐོར་བརྡ་ལན་ཁ་གང་ཡང་སྤྲད་མེད།   གསར་འགོད་པ། པདྨ་རྒྱལ།
