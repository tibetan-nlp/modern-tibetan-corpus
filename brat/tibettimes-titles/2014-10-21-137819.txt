འདས་ལོར་མི་ ༢༤༠༠ ལ་སྲོག་ཁྲིམས་བཅད།

གླེང་མོལ་ཐེབས་རྩ་ཚོགས་པའི་ཞིབ་འཇུག་སྙན་ཐོའི་ནང་ཕྱི་ལོ་ ༢༠༡༣ ལོར་རྒྱ་གཞུང་གིས་སྲོག་ཁྲིམས་བཅད་པའི་ཉེས་ཅན་གྱི་གྲངས་འབོར་དེ་རྒྱལ་ཁབ་གཞན་དག་གི་བསྡོམས་འབོར་ལས་མང་བ་ཡོད་ལུགས་བསྒྲཊ་འདུག ཉེ་སྔོན་ཨ་རིའི་ནང་རྟེན་གཞི་བྱས་པའི་གླེང་མོལ་ཐེབས་རྩ་ཚོགས་པས་ཞིབ་འཇུག་སྙན་ཐོ་ཞིག་ཕྱིར་བསྒྲགས་གནང་བ་ལྟར་ན། ཕྱི་ལོ་ ༢༠༠༧ ལོར་རྒྱ་ནག་ཆེས་མཐོའི་ཁྲིམས་ཁང་གིས་རྒྱལ་ཡོངས་ཀྱི་ས་གནས་ཁྲིམས་ཁང་ཁག་གི་སྲོག་ཁྲིམས་ལག་བསྟར་གྱི་དབང་ཆ་ཕྱིར་བསྡུ་བྱས་རྗེས། ལོ་རེ་བཞིན་སྲོག་ཁྲིམས་བཅད་པའི་གྲངས་འབོར་མངོན་གསལ་གྱིས་ཉུང་དུ་ཕྱིན་ཡོད་ཀྱང་། ད་ལྟའང་རྒྱ་གཞུང་གིས་ལོ་རེར་སྲོག་ཁྲིམས་བཅད་པའི་ཉེས་ཅན་གྱི་གྲངས་འབོར་དེ། རྒྱལ་ཁབ་ཆུང་ཆུང་ཁག་ཅིག་གི་གྲངས་འབོར་གཅིག་མཚུངས་མེད་པ་ཙམ་མ་གཏོགས་འཛམ་གླིང་གི་རྒྱལ་ཁབ་གཞན་ཡོངས་ཀྱི་བསྡོམས་འབོར་ལས་ཀྱང་རྦད་དེ་མང་བ་ཡོད་འདུག སྙན་ཐོའི་ནང་དུ། འདས་ལོར་རྒྱ་གཞུང་གིས་ཁྱོན་མི་གྲངས་ ༢༤༠༠ ལ་སྲོག་ཁྲིམས་བཅད་ཡོད་པ་དང་། གྲངས་འབོར་དེ་ཕྱི་ལོ་ ༢༠༠༢ ལོའི་མི་གྲངས་ ༡༢༠༠༠ དང་། ༢༠༡༢ ལོར་མི་ ༣༠༠༠ ལ་སྲོག་ཁྲིམས་བཅད་པའི་གྲངས་འབོར་གཉིས་དང་བསྡུར་ན་བརྒྱ་ཆ་ ༢༠ ཉུང་དུ་ཕྱིན་ཡོད། ཡིན་ནའང་། རྒྱ་གཞུང་གིས་ཤིན་ཅང་ནང་གཙོ་བོར་གྱུར་བའི་རྒྱ་ནག་རྒྱལ་ཡོངས་སུ་འཇིགས་སྐུལ་བར་རྒོལ་རྡུང་དྲག་པོ་གཏོང་བའི་སྲིད་བྱུས་ལག་བསྟར་བྱེད་བཞིན་ཡོད་སྟབས། འདི་ལོར་སྲོག་ཁྲིམས་བཅད་པའི་གྲངས་འབོར་ཉུང་དུ་མི་འགྲོ་བར་འདས་ལོ་དང་གཅིག་མཚུངས་ཆགས་སྲིད་པ་རེད། རྒྱ་གཞུང་གིས་སྲོག་ཁྲིམས་བཅད་པའི་གྲངས་འབོར་དེ་རྒྱལ་ཁབ་ཀྱི་གསང་བ་ཞིག་ཏུ་ངོས་འཛིན་གྱིས་གཏན་ནས་ཕྱིར་བསྒྲགས་བྱེད་ཀྱི་མེད་ཀྱང་། རྒྱ་ནག་གི་ཁྲིམས་འཛིན་མི་སྣ་ཞིག་གི་ས་ནས་གནས་ཚུལ་ཤེས་རྟོགས་བྱུང་ཡོད་སྐོར་བརྗོད་འདུག སྙན་ཐོའི་ནང་རྒྱ་ནག་ལྷོ་ཕྱོགས་བདུན་རེའི་ཚགས་པར་ཁང་གི་གནས་ཚུལ་ཞིག་ལུང་འདྲེན་བྱས་པའི་ནང་། ཕྱི་ལོ་ ༢༠༡༣ ཟླ་ ༧ ནས་འདི་ལོའི་ཕྱི་ཟླ་ ༩ པ་བར་དུ། རྒྱ་ནག་ཆེས་མཐོའི་ཁྲིམས་ཁང་གི་དྲ་བའི་ཐོག་ཁྱོན་སྲོག་ཁྲིམས་བཅད་པའི་ཉེས་དོན་ ༡༥༢ ཕྱིར་བསྒྲགས་བྱས་ཡོད་པ་དང་། དེའི་ནང་ལྐོག་གསོད་ཀྱི་ཉེས་དོན་ ༡༢༩ དང་། སྨྱོས་རྫས་ལྐོག་ཚོང་གི་ཉེས་དོན་ ༡༧ ཚུད་ཡོད་ལ། སྲོག་ཁྲིམས་ཀྱི་ཉེས་དོན་བསྐྱར་ཞིབ་ཁྲོད་བརྒྱ་ཆ་ ༡༣ ལ་མ་གཏོགས་དག་འབུད་ཁྲིམས་རྩོད་པ་མཉམ་ཞུགས་བྱེད་དུ་བཅུག་མེད་ལུགས་དང་། ཕྱི་ལོ་ ༢༠༡༣ ལོར་རྒྱ་ནག་ཆེས་མཐོའི་ཁྲིམས་ཁང་གིས་ས་གནས་ཁྲིམས་ཁང་ཁག་གིས་ཐག་གཅོད་བྱས་པའི་སྲོག་ཁྲིམས་ཀྱི་བརྒྱ་ཆ་ ༣༩ ཕྱིར་སློག་བྱས་ཏེ། དཔང་རྟགས་བཙལ་ནས་བསྐྱར་དུ་ཁྲིམས་བཤེར་བྱེད་དུ་བཅུག་ཡོད་པ་མ་ཟད། དེའི་ནང་གི་སྲོག་ཁྲིམས་ཁག་ཅིག་ལ་དཔང་རྟགས་འདང་ངེས་མེད་པར་ཁྲིམས་ཆད་རྩིས་མེད་གཏོང་དགོས་ཁེལ་ཡོད་པ་དང་། སྤྱིའི་ཆ་ནས་ས་གནས་ཁྲིམས་ཁང་གི་ཉེས་དོན་གཏན་འབེབས་ཀྱི་དབང་ཚད་བརྒྱ་ཆ་བརྒྱ་ཡིན་པ་མ་ཟད། ནག་ཉེས་བཙན་འགེལ་གྱི་དྲང་བདེན་མིན་པའི་གནད་དོན་མང་དུ་བྱུང་བཞིན་ཡོད་སྐོར་སོགས་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
