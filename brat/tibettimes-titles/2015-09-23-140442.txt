རྡོ་སོལ་འཕྲོག་རྩོད་ཀྱིས་མི་ ༤༠
ལྷག་ལ་ཤི་རྨས་བཟོས།

  ཤིན་ཅང་ཡུ་གུར་མི་རིགས་རང་སྐྱོང་ལྗོངས་ཁོངས་ཀྱི་ཡུ་གུར་བ་དང་ས་སྲུང་ཉེན་རྟོག་པའི་བར་རྡོ་སོལ་གཏེར་འདོན་ས་ཚུགས་འཕྲོག་རྩོད་བྱས་ཏེ་མི་ ༤༠ ལྷག་ལ་ཤི་རྨས་ཚབས་ཆེན་བྱུང་འདུག   རྒྱ་ནག་གཞུང་འབྲེལ་མི་སྣས་གནས་ཚུལ་བསྒྲགས་པ་ལྟར་ན། ཤིན་ཅང་ཡུ་གུར་མི་རིགས་རང་སྐྱོང་ལྗོངས་ཨ་ཁི་སུ་ (Aksu) གྲོང་ཁྱེར་གྱི་བྷེ་ཁྲིན་ (Baicheng) རྫོང་ཁོངས་སུ་ཡོད་པའི་རྡོ་སོལ་གཏེར་འདོན་ས་ཚུགས་ཤིག་ཏུ། ཁ་སང་ཕྱི་ཚེས་ ༢༢ ཉིན་ས་གནས་ཡུ་གུར་བ་དང་རྒྱ་ནག་ཉེན་རྟོག་དམག་མིའི་བར་འཁྲུག་འཛིང་ཤོར་ཏེ། ཉུང་མཐར་མི་ ༤༠ ལྷག་ལ་ཤི་རྨས་ཚབས་ཆེན་བྱུང་ཡོད་པ་དང་། ཤི་རྨས་ཐེབས་པའི་ནང་ཉེན་རྟོག་དམག་མི་ ༥ ཡོད་པ་དང་། དེ་མིན་ཚང་མ་ཡུ་གུར་བ་ཡིན་སྐོར་བརྗོད་འདུག   ས་གནས་ཉེན་རྟོག་པ་ཇེ་མལ་ཨེ་ས་ (Jamal Eysa) ཟེར་བ་ཞིག་གིས། ལག་ཏུ་གྲི་ཐོགས་པའི་ཡུ་གུར་ཁ་བྲལ་བས་རྡོ་སོལ་གཏེར་འདོན་ས་ཚུགས་ཀྱི་ས་སྲུང་ཉེན་རྟོག་པའི་ཐོག་ལ་དྲག་རྒོལ་བྱས་པ་དང་། ཉིན་དེའི་ཕྱི་དྲོར་ང་ལ་གྲོང་རྡལ་ཉེན་རྟོག་ཁང་ནས་ཁ་པར་ཞིག་འབྱོར་ཏེ། རྡོ་སོལ་གཏེར་འདོན་ས་ཚུགས་བདེ་སྲུང་ལ་གྲ་སྒྲིག་དགོས་སྐོར་མངགས་པ་ལྟར། ང་ཚོ་ས་སྲུང་ཉེན་རྟོག་པ་ ༢༠ ལྷག་གིས་རྡོ་སོལ་གཏེར་འདོན་ས་ཚུགས་བརྟན་སྲུང་བྱས་པ་ཡིན། སྤྱིར་བཏང་ཡུ་གུར་ཁ་བྲལ་བས་རྡོ་སོལ་གཏེར་ཁའི་བདག་པོ་དམིགས་འབེན་དུ་བཟུང་ཡོད་ནའང་། འཁྲུག་འཛིང་ཞི་འཇགས་སུ་གཏོང་དུ་ཡོང་བའི་ཉེན་རྟོག་པའི་ཐོག་ལ་དྲག་རྒོལ་བྱས་པ་རེད། ཡུ་གུར་ཁ་བྲལ་བ་ཚོས་དུས་རྒྱུན་ང་ཚོ་ཉེན་རྟོག་ཚན་པ་དང་རྒྱ་ནག་ཧན་མི་རིགས་ཀྱི་ཚོང་པ། བཟོ་གྲྭ་ཁག་གི་སྦྱིན་བདག་སོགས་ལ་དྲག་རྒོལ་དང་ཆག་སྒོ་སློང་གི་ཡོད་སྐོར་བརྗོད་འདུག   འཇར་མན་དུ་རྟེན་གཞི་བྱས་ཡོད་པའི་འཛམ་གླིང་ཡུ་གུར་མི་རིགས་མཐུན་ཚོགས་ཀྱིས་གཙོས་ཕྱི་རྒྱལ་ཁག་ཏུ་ཡོད་པའི་ཡུ་གུར་བ་ཚོས། སྔར་རྗེས་སུ་བྱུང་བའི་འཁྲུག་འཛིང་གི་དོན་རྐྱེན་ཡོད་ཚད་པེ་ཅིན་གཞུང་གི་ཉེན་རྟོག་དམག་མིའི་བྱ་སྤྱོད་ལས་བྱུང་བ་དང་། རྒྱ་གཞུང་གིས་གྲངས་ཉུང་མི་རིགས་ལ་དབྱེ་འབྱེད་དང་ཆོས་ལུགས་བྱེད་སྒོར་དྲག་གནོན། རིག་གནས་ཡུལ་སྲོལ་བཀག་འགོག་བྱེད་ཀྱི་ཡོད་པའི་སྐྱོན་བརྗོད་བྱས་འདུག   ཉེ་ལམ་ནས་རྒྱ་ནག་གཞུང་གིས་ཤིན་ཅང་ཡུ་གུར་མི་རིགས་རང་སྐྱོང་ལྗོངས་ནང་ཆོས་ལུགས་ཀྱི་ལྟ་བ་མཁྲེགས་འཛིན་མཁན་དང་ཁ་བྲལ་རིང་ལུགས་པ། དེ་བཞིན་འཇིགས་སྐུལ་རིང་ལུགས་པ་སོགས་ཀྱི་མིང་བཏགས་ཏེ། རྒྱལ་ཁབ་ཀྱི་ གནོད་འཚེ་ཅན་གསུམ་དུ་ངོས་འཛིན་གྱིས་དྲག་གནོན་ཤུགས་ཆེན་བྱེད་བཞིན་ཡོད་པ་དང་། ཕྱི་ཕྱོགས་ཀྱི་གནད་དོན་དབྱེ་ཞིབ་པ་མང་པོས་རྒྱ་གཞུང་གིས་ཡུ་གུར་བ་ཚོར་བསྙོན་འཛུགས་མང་པོ་བྱེད་ཀྱི་ཡོད་པ་མ་ཟད། གཞི་མེད་འཁྱོག་བརྗོད་དང་བཙན་གནོན་ཤུགས་ཆེན་བྱས་པར་བརྟེན། ཡུ་གུར་བ་ཚོ་འུ་ཐུག་ཐབས་ཟད་ཀྱི་གནས་སུ་བསྐྱལ་ཏེ་མི་གསོད་ཁྲག་སྦྱོར་གྱི་དྲག་རྒོལ་མང་པོ་བསྟུན་མར་བྱུང་དང་འབྱུང་བཞིན་ཡོད་པས། དྲག་སྤྱོད་ཀྱི་འགན་ཡོད་ཚད་རྒྱ་གཞུང་གིས་འཁུར་དགོས་སྐོར་བརྗོད་ཀྱི་ཡོད་པ་རེད།   དེ་བཞིན་གནད་དོན་དེའི་ཐད་རྒྱལ་སྤྱིའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལྷན་ཚོགས་ཁག་གིས། རྒྱ་ནག་གི་དབང་འཛིན་པ་ཚོས་ཤིན་ཅང་རང་སྐྱོང་ལྗོངས་ནང་མི་རིགས་གཉའ་གནོན་དང་དབང་སྒྱུར་སྲིད་ཇུས་ལག་བསྟར་བྱས་ཏེ། དྲག་ཆས་ཉེན་རྟོག་པས་ཡུ་གུར་མི་མང་ལ་རྒོལ་རྡུང་གཏོང་བ་དང་ཁ་ཆེ་ཆོས་ལུགས་ཀྱི་བྱེད་སྒོ་ཁག་ལ་དམ་དྲག་བྱེད་ཀྱི་ཡོད་པ་མ་ཟད། ཡུ་གུར་མི་མང་གི་སྐད་ཡིག་རིག་གཞུང་ཉམས་རྒུད་དུ་གཏོང་བཞིན་ཡོད། ཅེས་སོགས་བརྗོད་འདུག   གསར་འགོད་པ། པདྨ་རྒྱལ།
