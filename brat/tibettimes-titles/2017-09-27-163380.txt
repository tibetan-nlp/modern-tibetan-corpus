ཆོས་དབྱིངས་སྐྱིད་ལ་གསེར་གྱི་རྟཊ་མ་ཐོབ།

རྒྱ་ནག་ནང་བསྡུས་པའི་རྒྱལ་སྤྱིའི་འདུར་རྒྱུག་འགྲན་སྡུར་ཁྲོད་བོད་རིགས་ཨོ་རྩལ་པ་ཆོས་དབྱིངས་སྐྱིད་ལགས་ཀྱིས་ཨང་དང་པོ་བླངས་ཏེ་གསེར་གྱི་རྟགས་མ་ཐོབ་ཡོད་འདུག ཁ་སང་ཕྱི་དྲོར་བོད་ལྗོངས་མི་མང་དྲ་བས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༥ ཉིན་རྒྱ་ནག་སུའུ་ཀྲུའུ་གྲོང་ཁྱེར་ཝུ་ཀྲུང“ཐའེ་ཧུའུ་མཚེའུ་བསྐོར་བའི”རྒྱལ་སྤྱིའི་འདུར་རྒྱུག་འགྲན་བསྡུར་སྐབས། བུད་མེད་ཀྱི་སྤྱི་ལེ་ ༡༠ ཡི་འདུར་རྒྱུག་འགྲན་བསྡུར་ཁྲོད་བོད་རིགས་ཨོ་རྩལ་པ་ཆོས་དབྱིངས་སྐྱིད་ལགས་ཀྱིས་སྐར་མ་ ༤༢། ༤༦ གི་མགྱོགས་ཚད་གྲུབ་འབྲས་བླངས་ཏེ་ཨང་དང་པོའི་གསེར་གྱི་རྟགས་མ་ཐོབ་ཡོད་པ་དང་། དེ་བཞིན་འདུར་རྒྱུག་དེའི་ཁྲོད་བོད་རིགས་ལུས་རྩལ་པ་ཀླུ་མོ་མཚོ་ཡིས་ཨང་བདུན་པ་བླངས་ཡོད་འདུག ཐེངས་འདིའི་རྒྱལ་སྤྱིའི་འདུར་རྒྱུག་འགྲན་བསྡུར་དེ་ནི་ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༤ ནས་ ༢༨ བར་ཉིན་ ༤ རིང་འཚོགས་ཀྱི་ཡོད་པ་དང་། རྒྱུག་རྩལ་འགྲན་བསྡུར་གྱི་རྣམ་གྲངས་ལ་ཆེད་ལས་ཚོགས་ཆུང་དང་། གཞོན་ནུ་ལོ་ཆུང་ཚོགས་ཆུང་སོགས་རྣམ་གྲངས་མི་འདྲ་བ་གསུམ་དང་། འགྲན་བསྡུར་ས་ཚིགས་སུ་ཐའེ་ཧུའུ་མཚེའུ་ས་ཚིགས་དང་། མུའུ་ཏུའུ་ས་ཚིགས། ཡོན་པོ་ས་ཚིགས། ཏུང་ཧྲན་ས་ཚིགས་བཅས་སུ་འཚོགས་ཀྱི་ཡོད་པ་དང་། འགྲན་བསྡུར་ཁྲོད་ཨོ་སི་ཊོ་ལེ་ཡ་དང་། ལྷོ་ཨ་ཧྥི་རི་ཁ། བྷིར་རྫིལ། ཨི་ཊ་ལི། སི་པན། ཁི་ལོམ་བྷི་ཡ་སོགས་རྒྱལ་ཁབ་དང་ས་ཁུལ་ཁག་ ༢༨ ཀྱི་ལུས་རྩལ་པ་ ༩༠ དང་། རྒྱ་ནག་གི་ལུས་རྩལ་པ་ ༢༦༠ བཅས་ཁྱོན་ལུས་རྩལ་པ་ ༡༧༠༠༠ ཙམ་མཉམ་ཞུགས་བྱས་ཡོད་འདུག ཕྱི་ཚེས་ ༢༦ ཉིན། ཆོས་དབྱིངས་སྐྱིད་དང་ཀླུ་མོ་མཚོ་གཉིས་ཡོན་པོ་ས་ཚིགས་ནས་བུད་མེད་ཀྱི་འདུར་རྒྱུག་སྤྱི་ལེ་ ༡༠.༥ ཡི་འགྲན་སྡུར་ལ་ཞུགས་ཡོད་འདུག གཞན་ཡང་། ཟླ་འདིའི་ཚེས་ ༢༤ ཉིན་རྒྱ་ནག་ཧུའུ་པའེ་ཀོ་པེ་ཡི་གྲོང་ཁྱེར་དུ་ཚོགས་པའི་རྒྱལ་ཡོངས་ན་གཞོན་བྲག་འཛེག་འགྲན་བསྡུར་ཚོགས་ཡོད་པ་དང་། འགྲན་བསྡུར་དེའི་ནང་དུ་ཁྱོན་རུ་ཁག་ ༣༨ དང་། ལུས་རྩལ་པ ༡༣༠ ལྷག་མཉམ་ཞུགས་བྱས་ཡོད་པ་དང་། བོད་ལྗོངས་བྲག་འཛེག་རུ་ཁག་ལ་ཨང་དང་པོའི་གསེར་གྱི་རྟགས་མ་ཐོབ་ཡོད་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
