ཕྱི་རྒྱལ་གྱི་བཏུང་བྱ་ཉོ་ཚོང་མི་ཆོག

རྒྱ་གར་གྱི་མངའ་སྡེ་ཁག་ཅིག་ནང་ཁུ་ནག་(Pepsi)དང་(Coca Cola)སོགས་ཉོ་ཚོང་བྱེད་པར་བཀག་འགོག་བྱས་ཏེ་རྒྱ་གར་རང་གི་བཏུང་བྱའི་རིགས་སྤྱོད་པར་སྐུལ་མ་གཏོང་གི་ཡོད་འདུག རྒྱ་གར་ལྷོ་ཕྱོགས་ཏ་མིལ་ན་ཏུ་ (Tamil Nadu )མཉམ་འབྲེལ་ཚོང་པའི་མཐུན་ཚོགས་ (The Federation of Tamil Nadu Traders Associations )དང་ཏ་མིལ་ན་ཏུ་ཚོང་པའི་མཐུན་ཚོགས་གླེང་སྟེགས་(The Tamil Nadu Traders Associations Forum) གཉིས་ཀྱིས་སྣེ་འཁྲིད་དེ། ཏ་མིལ་ན་ཏུ་མངའ་སྡེའི་ནང་ཕྱི་རྒྱལ་གྱི་ཚོང་རྟགས་ཅན་ཁུ་ནག་(Pepsi)དང་(Coca Cola)སོགས་ཉོ་ཚོང་བྱེད་པར་བཀག་འགོག་བྱས་ཏེ། བེ་ཏའི་སིལ་ཆུ་(Coconut water)སོགས་རྒྱ་གར་རང་ནས་ཐོན་པའི་བཏུང་བྱའི་རིགས་ཉོ་ཚོང་དགོས་པའི་ལས་འགུལ་སྤེལ་བཞིན་ཡོད་པ་དང་། བརྒྱུད་ལམ་དུ་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། ཏ་མིལ་ན་ཏུ་ཡི་ཚོང་ཁང་བདག་པོ་ས་ཡ་གཅིག་ལྷག་གིས་ལས་འགུལ་དེར་རྒྱབ་སྐྱོར་གྱིས་བཏུང་བྱ་དེ་རིགས་ཉོ་ཚོང་མཚམས་འཇོག་བྱས་ཡོད་འདུག དེ་བཞིན་རྒྱ་གར་ནུབ་ཕྱོགས་དབུས་མེ་འཁོར་འགྲིམ་འགྲུལ་ཁང་གི་བཀོད་ཁྱབ་འོག མེ་འཁོར་འབབ་ཚིགས་ ༣༠༠ ནས་(Pepsi, Coca Cola, Sprite, 7Up, Mountain Dew )ཉོ་ཚོང་འཕྲལ་དུ་མཚམས་འཇོག་བྱེད་དུ་འཇུག་གི་ཡོད་པ་དང་། འགྲིམ་འགྲུལ་ཁང་དེའི་འགན་འཛིན་གཞོན་པ་ས་ཆིན་ཤུག་ལཱ་(Sachin Shukla)ལགས་ཀྱིས། བཏུང་བྱ་དེ་རིགས་ཐོན་སྐྱེད་ཀྱི་ཆོག་མཆན་དུས་ཡུན་རྫོགས་ཚར་ཡོད་པ་དང་། འཕྲོད་བསྟེན་ལས་ཁུངས་ནས་ཐོན་དངོས་ལ་ཞིབ་འཇུག་བྱས་པ་བརྒྱུད་ཆོག་མཆན་གསར་པ་མ་རག་བར་དུ་བཏུང་བྱ་དེ་རིགས་ཚོང་འགྲེམས་མཚམས་བཞག་པ་ཡིན་སྐོར་བརྗོད་འདུག ཏ་མིལ་ན་ཏུ་ཁུལ་གྱི་ལས་འགུལ་སྣེ་འཁྲིད་ཚོགས་པ་ཚོས། ཕྱི་རྒྱལ་གྱི་བཏུང་བྱ་ཁུ་ནག་(Pepsi)དང་(Coca Cola)རིགས་ཀྱི་ནང་མངར་མོ་དང་རྫས་འགྱུར་གྱི་སྦྱོར་བ་ཤུགས་ཆེན་པོ་ཡོད་པའི་རྐྱེན་གྱིས་མིའི་བདེ་ཐང་ལ་གནོད་པ་ཆེན་པོ་ཡོད་སྟབས། རྒྱ་གར་རང་ནས་ཐོན་སྐྱེད་བྱས་པའི་བཏུང་བྱའི་རིགས་ཉོ་ཚོང་བྱེད་པར་སྐུལ་འདེད་བཏང་སྟེ་ཤིང་ཏོག་ཁུ་བ་རིགས་བྲིན་པོ་ཡོང་ཐབས་དགོས་སྐོར་དང་། སྒོ་རྩིས་ཚོང་ཁང་(Supermarket)དང་ཟ་ཁང་། མགྲོན་ཁང་བཅས་ཀྱིས་ལས་འགུལ་དེ་ལ་རྒྱབ་སྐྱོར་བྱས་ཏེ། ས་གནས་ཚོང་པ་དང་ཞིང་པ་རྣམས་ཀྱི་དཔལ་འབྱོར་གོང་འཕེལ་དུ་འགྲོ་བར་རོགས་འདེགས་དགོས་པའི་འབོད་སྐུལ་བྱས་ཡོད་འདུག ཡང་ལས་འགུལ་སྣེ་འཁྲིད་པས་བཏུང་བྱའི་རིགས་ལ་ཆུ་ཧ་ཅང་མང་པོ་དགོས་ཀྱི་ཡོད་ལ། ལོ་ཤས་སྔོན་ནས་བཟུང་ཡུལ་དེའི་ནང་ཐན་སྐྱོན་རིམ་པར་བྱུང་སྟེ་དཀའ་རྙོག་མི་ཉུང་བ་ཞིག་འཕྲད་ཡོད་ལ། སྤྱིར་བཏང་ལས་འགུལ་དེ་ཟླ་ཤས་གོང་ནས་བཟུང་མགོ་བཙུགས་ཡོད་པ་དང་། འདི་ལོའི་ཕྱི་ཟླ་ ༡ ནང་གཞོན་སྐྱེས་མང་པོ་ཁྲོམ་གཞུང་དུ་ཐོན་ཏེ། རྒྱ་གར་ཁྲིམས་ཁང་ནས་ཡུལ་དེའི་སྲོལ་རྒྱུན་བྱེད་སྒོ་བ་གླང་ཁྲོམ་འདུའི་དུས་སྟོན་(Jallikattu)བཀག་སྡོམ་བྱས་པར་ངོ་རྒོལ་གྱི་ལས་འགུལ་སྤེལ་སྐབས། ཕྱི་རྒྱལ་གྱི་བཏུང་བྱ་རིགས་བཀག་འགོག་གི་ལས་འགུལ་ལ་རྒྱབ་སྐྱོར་ཆེན་པོ་ཐོབ་སྐོར་བརྗོད་ཡོད་པ་རེད། ཡིན་ནའང་། ས་གནས་དེར་ཡོད་པའི་བཏུང་བྱའི་རིགས་ཀྱི་བཟོ་ཚོང་ཚོགས་པ་(The Indian Beverage Association )ཡིས། ཁུ་ནག་དང་(Coca Cola)་ཚོང་འགྲེམས་བྱེད་པར་བཀོག་འགོག་བྱེད་པ་ནི་དཔལ་འབྱོར་འཕེལ་རྒྱས་ཀྱི་རྨང་གཞིར་འགོག་རྐྱེན་བཟོས་པ་ཞིག་ཡིན་སྐོར་བརྗོད་འདུག་པ་མ་ཟད། བཟོ་ཚོགས་ཚོགས་པ་དེས་ཏ་མིལ་ན་ཏུ་ཡི་དུད་ཁྱིམ་ ༢༠༠༠ ལ་ཐད་ཀར་ལས་ཞུགས་ཀྱི་མཐུན་རྐྱེན་སྦྱོར་གྱི་ཡོད་པ་དང་། དུད་ཁྱིམ་ ༥༠༠༠ ལྷག་གི་ལས་ཞུགས་དང་འབྲེལ་བ་ཆགས་བསྡད་ཡོད་པས་ལས་ཀ་དེ་མུ་མཐུད་འགྲོ་ཐུབ་པའི་རེ་བ་དང་། ཏ་མིལ་ན་ཏུ་ཡི་འཛད་སྤྱོད་པ་རྣམས་ལ་རང་དབང་གི་གདམ་ག་བྱེད་པའི་ཐོབ་ཐང་ཡོད་སྐོར་བརྗོད་འདུག ཨ་རིའི་(PepsiCo)ཚོང་གཉེར་ཁང་གི་འཛིན་སྐྱོང་འགན་འཛིན་ལྕམ་ཨིན་ཌི་ར་ནུ་ཡའི་(Indra Nooyi)ལགས་ཀྱིས་དེང་སང་རྒྱ་གར་ནང་སྐོར་སྐྱོད་བྱེད་བཞིན་ཡོད་པ་དང་། ཁ་སང་རྒྱ་གར་གྱི་དཔལ་འབྱོར་བློན་ཆེན་ཨ་རིན་ཇིཌ་ལི་(Arun Jaitley)ལགས་དང་མཇལ་འཕྲད་ཀྱིས་འཕྲུལ་ཆས་ཅན་གྱི་དངུལ་གཏོང་དང་དེའི་ཤུགས་རྐྱེན་སྐོར་ལ་གླེང་མོལ་གནང་ཡོད་པ་དང་། དེ་བཞིན་བདུན་ཕྲག་འདིའི་ནང་རྒྱ་གར་གྱི་སྲིད་བློན་མུ་དྷེ་མཆོག་ལ་མཇལ་འཕྲད་ཞུས་ཏེ་རྒྱ་གར་དང་ཨ་རིའི་བར་གྱི་འབྲེལ་ལམ་སྐོར། དེ་བཞིན་ཚོང་འབྲེལ་བཟའ་བཏུང་རིགས་ཀྱི་བདེ་འཇགས་སྐོར་ལ་བཀའ་མོལ་གནང་རྒྱུ་ཡིན་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
