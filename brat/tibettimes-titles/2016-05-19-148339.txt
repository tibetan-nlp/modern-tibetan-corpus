རྒྱ་ནག་གཞུང་ཚབ་ལ་ཁ་རྩོད་བརྒྱབ།

དབྱིན་ཡུལ་རྒྱ་ནག་གཞུང་ཚབ་པ་ལུའུ་ཞོ་མེན་གྱིས་གཏམ་གླེང་སྐབས་བོད་མི་དང་བོད་དོན་རྒྱབ་སྐྱོར་བ་ཚོས་བོད་ནང་གི་གནས་སྟངས་དང་བོད་མིའི་རང་སྲེག་སོགས་ཀྱི་སྐོར་ལ་དྲི་བ་བཏོན་ཡོད་འདུག ཧོང་ཀོང་རང་དབང་གསར་ཁང་ (hkfp) གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། གཟའ་མིག་དམར་ཉིན། བྷ་ཐ་རྒྱལ་པོའི་དཔེ་ཁང་ (Bath Royal Literary) དང་ཚན་རིག་སློབ་གཉེར་ཁང་ (Scientific Institution) གཉིས་ཐུན་མོང་གི་གོ་སྒྲིག་འོག ༼རྒྱ་ནག ང་ཚོའི་མིག་མདུན་གྱི་རྒྱལ་སྤྱིའི་ལས་དོན༽ཞེས་པའི་བརྗོད་གཞིའི་ཐོག་གཏམ་གླེང་ཞིག་གོ་སྒྲིག་བྱས་ཡོད་པ་དང་། གཏམ་གླེང་དེའི་སྟེང་དབྱིན་ཡུལ་གྱི་རྒྱ་ནག་གཞུང་ཚབ་པ་ལུའུ་ཞོ་མེན་ (Liu Xiaoming) ཟེར་བ་གདན་ཞུ་བྱས་ཡོད་པ་དང་། གཏམ་གླེང་ཚོགས་འདུ་མ་ཚུགས་པའི་སྔོན་ལ་གོ་སྒྲིག་པས་རྒྱ་ནག་གི་གཞུང་ཚབ་པར་རྙོག་ཟིང་མི་བཟོ་བའི་ཆེད་དུ། བོད་ཀྱི་མཚོན་རྟགས་འཁོད་པའི་གྱོན་ཆས་རིགས་འཕུད་དགོས་པའི་རེ་སྐུལ་བཏོན་ཡོད་འདུག གཏམ་གླེང་ཚོགས་འདུར་ཞུགས་པའི་བོད་མི་དང་བོད་དོན་རྒྱབ་སྐྱོར་བ་ཚོས་ཚོགས་ཁང་གི་ཕྱི་ལོགས་ནས་ཞི་བའི་རྔོམ་སྟོན་ཁྲོམ་སྐོར་བྱས་པ་མ་ཟད། བོད་མི་ཤེས་རབ་རྒྱ་མཚོ་ཞུ་བ་ཞིག་གིས་རྒྱ་ནག་གཞུང་ཚབ་པར་བོད་མིའི་དཀའ་རྙོག་དང་བོད་མིའི་རང་ལུས་མེར་བསྲེགས། དེ་བཞིན། པཎ་ཆེན་ཆུང་སྲིད་རིན་པོ་ཆེ་གར་ཁྲིད་ཆ་མེད་དུ་གྱུར་བའི་གནས་ཚུལ་སོགས་དྲིས་སྐབས། ལུའུ་ཞོ་མེན་གྱིས་ཆུང་སྲིད་རིན་པོ་ཆེ་ནི་བྱིས་པ་དཀྱུས་མ་ཞིག་ཡིན་ལ། ཁོང་པཎ་ཆེན་རིན་པོ་ཆེའི་ཡང་སྲིད་དུ་ངོས་འཛིན་རྩ་བ་ནས་བྱས་མེད་སྐོར་དང་། བོད་ནང་དུ་བོད་མི་ས་ཡ་གསུམ་ཡོད་པ་དང་། ཁོ་ཚོ་ཚང་མས་བོད་མི་མང་སྔར་ལས་སྐྱིད་ཅིང་སྔར་ལས་ཚེ་ཐག་རིང་བ་དང་། སྔར་ལས་རང་དབང་ལྡན་པ་ངོས་ལེན་བྱེད་ཀྱིན་ཡོད་ལུགས་སོགས་བརྗོད་པའི་ལན་དུ། བོད་མི་ཤེས་རབ་རྒྱ་མཚོ་ལགས་ཀྱིས་ཁོ་རང་གི་ནང་མི་ཚོར་དེ་འདྲའི་བདེ་སྐྱིད་ཅིག་མེད་སྐོར་བཤད་སྐབས། ལུའུ་ཞོ་མེན་གྱིས་ཁོར་མུ་མཐུད་དུ་བཤད་བཅུག་མི་འདུག དེའི་རྗེས་ལུའུ་ཞོ་མེན་གྱིས་བོད་དོན་རྒྱབ་སྐྱོར་བ་ཚོས་མི་མང་ཚོགས་ཁང་ནང་འགྲོ་བ་མིའི་ཐོབ་ཐང་རྡོག་རོལ་གཏོང་གི་ཡོད་ལུགས་བཤད་སྐབས། འགྲོ་བ་མིའི་ཐོབ་ཐང་དོན་གཉེར་བ་ཞིག་གིས་དེ་ནི་རྒྱ་ནག་ནང་གི་མནར་གཅོད་དང་བསྡུར་ཐབས་མེད་ཅེས་བཤད་པའི་ལན་དུ། ལུའུ་ཞོ་མེན་གྱིས་དེ་སྐྱག་རྫུན་འབའ་ཞིག་ཡིན་ལུགས་དང་། བོད་དོན་འཐབ་རྩོད་ལ་ཕམ་ཁ་མ་གཏོགས་མེད་ལུགས་དང་། བོད་དོན་རྒྱབ་སྐྱོར་བ་ཚོས་བོད་རྒྱ་ནག་ནས་ཁ་བྲལ་གཏོང་གི་ཡོད་ནའང་། བོད་དོན་རྒྱབ་སྐྱོར་ལ་དུས་ནམ་ཡང་རྒྱལ་ཁ་འཐོབ་མི་སྲིད་ལུགས་བཤད་འདུག གནས་ཚུལ་འདིའི་ནང་མུ་མཐུད་དུ། བོད་མི་ཤེས་རབ་རྒྱ་མཚོས། ངས་ཚེ་གང་བོར་རྒྱ་ནག་གི་དྲིལ་བསྒྲགས་ཐོས་མྱོང་ལ། དུས་ད་ལྟ་ད་དུང་ཡང་རྒྱ་ནག་གཞུང་གི་རྫུན་གཏམ་དེ་དག་རྒྱ་ནང་གི་རྒྱ་རིགས་ཚོར་བཤད་དུས། ཁོ་ཚོ་ཧ་ལས་བཞིན་ཡོད་པ་རེད། གཞུང་ཚབ་པ་ཁྱེད་རང་ལྟར་ངས་ཀྱང་བོད་ནང་གི་བོད་མི་ཚོར་བདེ་སྐྱིད་མེད་པ་ཤེས་ཀྱི་ཡོད། རྒྱུ་མཚན་བོད་ནང་གི་བོད་མི་ཚོས་རྒྱ་ནག་གཞུང་གི་གཉའ་གནོན་ལ་བཟོད་བསྲན་བྱས་ནས་བསྡད་ཡོད་སྐོར་བཤད་ཀྱིན་འདུག བོད་མི་ཚོར་རང་དབང་དགོས། བོད་མི་ཚོར་སོ་སོའི་རང་དབང་གི་མངོན་འདོད་བརྗོད་པའི་ཐོབ་ཐང་ཡོད་པ་མ་རེད། མ་འོངས་པར་བོད་རང་དབང་རྒྱལ་ཁབ་ཆགས་ཀྱི་རེད་ཅེས་སོགས་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
