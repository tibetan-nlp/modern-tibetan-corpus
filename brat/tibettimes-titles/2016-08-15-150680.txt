རྒྱ་གར་རང་བཙན་ཉིན་མོ་སྲུང་གཙི་ཞུས།

དེ་རིང་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱིས་རྒྱ་གར་ལ་རང་བཙན་ཐོབ་སྟེ་ལོ་ངོ་ཧྲིལ་པོ་ ༧༠ འཁོར་བའི་ཉིན་མོ་སྲུང་གཙི་ཞུས་ཡོད་པ་རེད། དེ་རིང་སྔ་དྲོའི་ཆུ་ཚོད་ ༩ ཐོག་བོད་མིའི་སྒྲིག་འཛུགས་བཀའ་ཤག་གི་མདུན་ཐང་དུ། དབྱིན་ཇིའི་དབང་བསྒྱུར་འོག་ནས་རྒྱ་གར་ལ་རང་བཙན་ཐོབ་སྟེ་ལོ་ངོ་ཧྲིལ་པོ་ ༧༠ འཁོར་བའི་རང་བཙན་ཉིན་མོ་སྲུང་གཙི་ཞུས་ཡོད་པ་དང་། ཉིན་མོ་སྲུང་གཙིའི་ཐོག་སྲིད་ཚབ་ཆོས་རིག་བཀའ་བློན་གཡུ་ཐོག་ཀརྨ་དགེ་ལེགས་མཆོག་གི་དབུས་བཀའ་བློན་རྣམ་པ་དང་། ལྷན་ཁང་ཁག་གི་དྲུང་འཕར་ཡན་གྱི་སྲི་ཞུ་ལས་བྱེད་བཅས་འཛོམས་ཏེ། རྒྱ་གར་གྱི་རྒྱལ་གླུ་གྱེར་ཏེ་རྒྱལ་དར་སྒྲེང་འཛུགས་གནང་ཡོད་པ་རེད། སྐབས་དེར་སྲིད་ཚབ་མཆོག་གིས་རྒྱ་གར་གཞུང་མང་གཉིས་ལ་འཚམས་འདྲི་གསུང་བཤད་གནང་དོན། དེ་རིང་རྒྱ་གར་དབྱིན་ཇིའི་དབང་འོག་ནས་རང་བཙན་ཐོབ་སྟེ་མི་ལོ་ ༧༠ འཁོར་བའི་དུས་ཆེན་ཁྱད་པར་ཅན་ཞིག་ཆགས་ཡོད། ངས་དུས་ཆེན་ཁྱད་པར་ཅན་གྱི་སྐབས་འདིར་བོད་མིའི་སྒྲིག་འཛུགས་སྤྱི་དང་འཛིན་སྐྱོང་བཀའ་ཤག དེ་བཞིན་བོད་མི་ཡོངས་ཀྱི་ཚབ་ཞུས་ནས་རྒྱ་གར་གཞུང་མང་བཅས་ཚང་མར་འཚམས་འདྲི་བཀྲ་ཤིས་བདེ་ལེགས་ཞུ་བཞིན་ཡོད། དེ་དང་ཆབས་ཅིག་རྒྱ་གར་དབུས་གཞུང་གཉིས་དང་། རྒྱ་གར་རྒྱལ་ཡོངས་མི་མང་།  ཕྱོགས་མཚུངས་ཧི་མཱ་ཅལ་མངའ་སྡེའི་གཞུང་མང་གཉིས་ཀྱི་གཙོས་བོད་དོན་ཐོག་འདས་པའི་ལོ་ ༦༠ ནང་ཧ་ཅང་བཀའ་དྲིན་ཆེ་བའི་མངའ་སྡེ་ཁག་ལའང་དེ་རིང་གོ་སྐབས་འདི་བརྒྱུད། ངས་ཚང་མར་སྙིང་ཁོང་རུས་པའི་གཏིང་ནས་ཐུགས་རྗེ་ཆེ་ཞུ་རྒྱུ་ཡིན། རྒྱ་གར་རྒྱལ་ཁམས་ཆེན་པོ་འདི་འབྱུང་འགྱུར་གྱི་ལོ་གཅིག་ནང་གང་ས་ཅི་ཐད་ནས་དར་རྒྱས་ཕུན་སུམ་ཚོགས་པ་དང་། ཆར་ཆུ་དུས་སུ་འབབ་པ། ལོ་ཕྱུག་རྟག་ཏུ་ལེགས་པ། མང་ཚོགས་རྣམས་ཀྱིས་བདེ་སྐྱིད་ཀྱི་དཔལ་ལ་རོལ་བ་དང་། སྲིད་གཞུང་མངའ་ཐང་ལྡན་པ་ཞིག་ཡོང་བར་བོད་མི་ཚོའི་ངོས་ནས་སྨོན་འདུན་ཞུ་རྒྱུ་ཡིན། ཞེས་གསུངས་ཡོད་པ་རེད། བཀའ་ཤག་མདུན་ཐང་གི་མཛད་སྒོ་གྲོལ་རྗེས་མགྲོན་འབོད་གནང་བ་ལྟར། བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་དཔལ་འབྱོར་བཀའ་བློན་ཀརྨ་ཡེ་ཤེས་མཆོག་དང་འཕྲོད་བསྟེན་བཀའ་བློན་ཆོས་སྐྱོང་དབང་ཕྱུག་མཆོག དེ་བཞིན། བོད་མི་མང་སྤྱི་འཐུས་ཚོགས་གཞོན་ཨ་ཅཱརྱ་ཡེ་ཤེས་ཕུན་ཚོགས་ལགས་ལགས་ཀྱི་དབུས་སྤྱི་འཐུས་ཁག་ཅིག་དང་། ལྷན་ཁང་ཁག་གི་དྲུང་ཆེ་དྲུང་འཕར་བཅས་དྷ་ས་ཉེན་རྟོག་པའི་རྩེད་ཐང་ཆེན་མོར་ཚོགས་པའི་དམག་དཔུང་རུ་སྒྲིག་གི་མཛད་སྒོར་མཉམ་ཞུགས་གནང་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
