ཨི་ཇིབ་ཁྲིམས་ཁང་ལ་སྐྱོན་བརྗོད།

ཁ་སང་ཨི་ཇིབ་ཁྲིམས་ཁང་ཞིག་གིས་མི་ ༦༨༣ ལ་སྲོག་ཁྲིམས་བཅད་པར་རྒྱལ་སྤྱིའི་གཞུང་དང་ཚོགས་སྡེ་ཁག་གིས་སྐྱོན་བརྗོད་ཤུགས་ཆེན་བྱེད་བཞིན་ཡོད་འདུག ཁ་སང་ཕྱི་ཚེས་ ༢༨ ཉིན་ཨི་ཇིབ་རྒྱལ་ས་ཁེད་རོབ་ལྷོ་ཕྱོགས་མངའ་སྡེ་མེན་ཡཱ་(Minya)ཁྲིམས་ཁང་གིས། དེ་སྔའི་སྲིད་འཛིན་མོར་སི་ (Mohammed Morsi) ལ་རྒྱབ་སྐྱོར་བྱེད་མཁན་ཁ་ཆེ་སྤུན་ཟླ་ཚོགས་པའི་ཚོགས་གཙོ་བྷེ་ཌེ་ (Mohammed Badie)གཙོ་བྱས་པའི་མི་གྲངས་ ༦༨༣  ཡིས་ན་ནིང་ཕྱི་ཟླ་ ༨ ཚེས་ ༡༤ ཉིན་ཉེན་རྟོག་ལས་ཁུངས་ལ་འབར་རྫས་གས་གཏོར་བྱས་ཏེ་མི་གསོད་དོན་རྐྱེན་བཟོས་པ་དང་། མི་མང་ལ་ངན་སྐུལ་བྱས་པ་སོགས་ཀྱི་ཉེས་མིང་འོག་སྲོག་ཁྲིམས་བཅད་ཡོད་འདུག གནས་ཚུལ་དེ་ཕྱི་བསྒྲགས་བྱས་མ་ཐག ཁ་ཆེ་སྤུན་ཟླ་ཚོགས་པར་རྒྱབ་སྐྱོར་བྱེད་མཁན་མི་མང་བརྒྱ་ཕྲག་ལྷག་གིས་ཁྲིམས་ཁང་གི་ཐག་གཅོད་ལ་ངོ་རྒོལ་སྐད་འབོད་བྱས་ཡོད་པ་དང་། ཨི་ཇིབ་ཀྱི་ལུགས་སྲོལ་ལྟར་ན། ཁྲིམས་ཁང་གིས་ཉེས་ཅན་ལ་སྲོག་ཁྲིམས་བཅད་རྗེས་ཨི་ཇིབ་ཁ་ཆེ་ཆོས་ལུགས་ཀྱི་ཁྲིམས་ལ་མཁས་པའི་ལྷན་ཚོགས་ཀྱིས་མཐའ་མའི་སྲོག་ཁྲིམས་གཅོད་མིན་ལ་ལམ་སྟོན་བྱེད་ཀྱི་ཡོད་ཀྱང་། འགྲོ་སྟངས་དེ་རྒྱལ་ཁབ་ཀྱི་རྩ་ཁྲིམས་ནང་གསལ་མེད་སྟབས། ཐེངས་འདིར་ལམ་སྲོལ་དེ་སྣང་མེད་ཐོག་སྲོག་ཁྲིམས་ལག་བསྟར་བྱེད་པའི་ཉེན་ཁ་ཡོད་སྟབས། རྒྱལ་ཁབ་མང་པོའི་གཞུང་དང་ཚོགས་སྡེ་ཁག་གི་སེམས་འཚབ་བྱེད་ཀྱིན་ཡོད་འདུག དབྱིན་ཡུལ་རྒྱལ་ས་ལོན་ཊོན་ནང་རྟེན་གཞི་བྱས་པའི་ཁ་ཆེ་སྤུན་ཟླ་ཚོགས་པའི་གསལ་བསྒྲགས་སྤེལ་བའི་ནང་། ཨི་ཇིབ་བར་བརྒལ་གཞུང་གིས་དམག་མིའི་རྒྱབ་རྟེན་འོག་ནས་ཨི་ཇིབ་ནང་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལ་རྡོག་རོལ་དང་། དྲང་བདེན་མིན་པའི་ཁྲིམས་གཞི་ལག་བསྟར་གྱིས་མི་མང་དམར་གསོད་བྱེད་ཀྱི་ཡོད་པ་དེར་རྒྱལ་སྤྱིའི་སྤྱི་ཚོགས་ཀྱིས་གདོང་ལེན་བྱེད་དགོས་ཚུལ་སོགས་འཁོད་ཡོད་འདུག ཨི་ཇིབ་ཁྲིམས་ཁང་གི་ཐག་གཅོད་དེར་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་གཙོས་ཡུ་རོབ་མཐུན་ཚོགས་དང་ཁ་ཆེ་མཐུན་ཚོགས། རྒྱལ་སྤྱིའི་ཉེས་ཡང་ཚོགས་པ། རྒྱལ་སྤྱིའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་རྟོག་ཞིབ་ཚོགས་པ། དེ་བཞིན་ནུབ་ཕྱོགས་རྒྱལ་ཁབ་ཚོས་ཨི་ཇིབ་བར་བརྒལ་གཞུང་ལ་སྐྱོན་བརྗོད་དང་། ཐག་གཅོད་དེར་ངེས་པར་དུ་བསྐྱར་ཞིབ་བྱ་དགོས་པའི་འབོད་སྐུལ་ཞུ་བཞིན་ཡོད་འདུག དཀྱིལ་ཤར་ཨེ་ཤ་ཡ་དང་བྱང་ཨ་ཧྥེ་རི་ཀའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལྟ་རྟོག་ཚོགས་པའི་འགན་འཛིན་ལྕམ་སཱ་རྰ་ལེད་(Sarah Leah Whitson)ལགས་ཀྱིས་གསར་འགོད་པར། འཛམ་གླིང་གི་ལོ་རྒྱུས་ཐོག་འདི་ལྟ་བུའི་སྲོག་ཁྲིམས་བཅད་པ་བྱུང་མྱོང་མེད་ལ། ཁྲིམས་ཁང་གིས་ཁྲིམས་རྩོད་པར་གྱོད་གཞི་གསལ་བཤད་བྱེད་པའི་གོ་སྐབས་སྤྲད་མེད་པ་ནི་རྒྱལ་སྤྱིའི་ཁྲིམས་ལུགས་དང་འགལ་ཡོད་ཅེས་བརྗོད་འདུག དེ་བཞིན། དཀྱིལ་ཤར་ཨེ་ཤ་ཡ་དང་བྱང་ཨ་ཧྥེ་རི་ཀའི་རྒྱལ་སྤྱིའི་ཉེས་ཡང་ཚོགས་པའི་འགན་འཛིན་ཧཱ་སེ་བྷ་ (Hassiba Hadj Sahraouni) ལགས་ཀྱིས་གསལ་བསྒྲགས་སྤེལ་བའི་ནང་། ཨི་ཇིབ་ཁྲིམས་ཁང་གིས་དྲང་བདེན་ཁྲིམས་བཤེར་བྱེད་པའི་རྩ་དོན་དང་འགལ་ཏེ་ཁྲིམས་ཁང་གི་ཆེ་མཐོང་མེད་པ་བཟོས་ཡོད་སྐོར་བརྗོད་འདུག མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་དྲུང་ཆེ་བྷན་ཁེ་མོན་དང་ཨ་རིའི་རྒྱལ་དོན་དྲུང་ཆེ་ཇོན་ཁད་རེ་གཉིས་ཀྱིས་ཁྲིམས་ཁང་གི་ཐག་གཅོད་དེ་དྲང་བདེན་མིན་སྟབས། གཟའ་འཁོར་འདིའི་ནང་ཨི་ཇིབ་ཕྱི་སྲིད་བློན་ཆེན་ལ་ཐུག་འཕྲད་གནང་སྟེ་གནད་དོན་འདིའི་སྐོར་ལ་གྲོས་མོལ་བྱ་འཆར་ཡོད་སྐོར་བསྒྲགས་འདུག ཟླ་རྗེས་མའི་ཚེས་ ༢༦ དང་ ༢༧ བར་གྱི་ཉིན་གཉིས་རིང་། ཨི་ཇིབ་ཀྱི་སྲིད་འཛིན་འོས་བསྡུ་བྱ་འཆར་ཡོད་སྟབས། བར་བརྒལ་སྲིད་གཞུང་གིས་ཟླ་འདིའི་ཚེས་ ༦ ཉིན་ནས་རྒྱལ་ཁབ་ནང་ཆོས་ལུགས་དང་འབྲེལ་བའི་དུས་དྲན་སྲུང་གཙི་བྱེད་མི་ཆོག་པའི་བཀའ་རྒྱ་བཏང་ཡོད་འདུག གནད་དོན་དབྱེ་ཞིབ་པ་ཚོས། འབྱུང་འགྱུར་ཨི་ཇིབ་སྲིད་གཞུང་སྲ་བརྟན་དང་རྒྱལ་ཁབ་ནང་བརྟན་བརླིང་ཡོང་ཆེད་བར་བརྒལ་སྲིད་གཞུང་གིས་ཁ་ཆེ་སྤུན་ཟླ་ཚོགས་པ་འཇིགས་སྐུལ་རིང་ལུགས་པར་ངོས་འཛིན་གྱིས་རྩ་མེད་བཟོ་བཞིན་ཡོད་ཀྱང་། མཇུག་འབྲས་དེ་ལས་ལྡོག་སྟེ་རྒྱལ་ཁབ་ནང་མི་གསོད་ཁྲག་སྦྱོར་གྱི་དོན་རྐྱེན་མང་དུ་འགྲོ་རྒྱུ་ཡིན་པའི་ཚོད་དཔག་བྱེད་བཞིན་ཡོད་འདུག གསར་འགོད་པ། ལྷ་སྐྱབས་རྒྱལ།
