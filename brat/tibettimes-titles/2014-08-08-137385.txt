མི་ ༣ གྱིས་རང་ཤི་ངོ་རྒོལ་བྱས།

ཁ་སང་ཡང་པེ་ཅིན་གྱི་ན་གཞོན་ཚགས་པར་ཁང་གི་ར་བའི་ནང་མི་ ༣ གྱིས་རང་ཤི་ངོ་རྒོལ་བྱས་ཡོད་པ་དང་། དེ་མ་ཐག་མྱུར་སྐྱོབ་རུ་ཁག་གིས་ཁོང་ཚོ་སྨན་ཁང་དུ་བསྐྱལ་ཡོད་འདུག ཁ་སང་ཉིན་རྒྱབ་པེ་ཅིན་གྱི་ཆུ་ཚོད་ ༢།༣༠ ཙམ་ལ། པེ་ཅིན་ན་གཞོན་ཚོགས་པར་ཁང་གི་ར་བའི་ནང་བུད་མེད་གཅིག་དང་སྐྱེས་པ་ ༢ གཉིས་ཀྱིས་སྨན་རྫས་བཏུང་སྟེ་རང་ཤི་ངོ་རྒོལ་བྱས་ཡོད་པ་དང་། དེ་མ་ཐག་མྱུར་སྐྱོབ་རུ་ཁག་གིས་ཁོང་ཚོ་སྨན་ཁང་དུ་བསྐྱལ་ཡོད་ཀྱང་། ད་ལྟའི་ཆར་ཤི་གསོན་གྱི་གནས་ཚུལ་གང་ཡང་གོ་ཐོས་མེད་པར་གྱུར་ཡོད་འདུག ཉེས་དོན་ཞིབ་བཤེར་ཁང་གིས་རང་ཤི་རྒྱག་མཁན་དེ་ཚོ་གང་ནས་ཡོང་བ་དང་། ཅིའི་ཕྱིར་རང་ཤི་བརྒྱབ་པ་ཡིན་ཞིབ་བཤེར་བྱས་རྗེས། པེ་ཅིན་གྲོང་ཁྱེར་ཉེན་རྟོག་ཁང་ནས་གཅིག་གྱུར་གྱིས་གསལ་བཤད་བྱ་རྒྱུ་ཡིན་ལུགས་བརྗོད་འདུག གནད་དོན་དེ་འདས་པའི་ཟླ་གཅིག་ནང་པེ་ཅིན་ན་གཞོན་ཚགས་པར་ཁང་གི་ཉེ་འགྲམ་དུ་བྱུང་བའི་རང་ཤི་ངོ་རྒོལ་ཐེངས་གཉིས་པ་ཡིན་ལ། འདི་ལོའི་ཕྱི་ཟླ་ ༧ ཚེས་ ༡༦ ཉིན་གྱི་སྔ་དྲོར་ཚགས་པར་ཁང་དེའི་སྒོ་འགྲམ་དུ། རྒྱ་ནག་ཅང་སུའུ་ཞིང་ཆེན་སི་ཧོང་རྫོང་ནས་ཡོང་བའི་བུད་མེད་ ༢ དང་སྐྱེས་པ་ ༥ ཡིས་ཞིང་སྨན་བཏུང་སྟེ་རང་ཤི་བརྒྱབ་ཡོད་པ་དང་། ཕྱིས་སུ་ཅང་སུའུ་ཞིང་ཆེན་སྲིད་གཞུང་གིས་སི་ཧོང་རྫོང་གི་འབྲེལ་ཡོད་མགོ་ཁྲིད་ཚོ་ཏང་གི་སྒྲིག་འཛུགས་ནས་གནས་དབྱུང་བཏང་བ་ཙམ་གྱི་ཐག་གཅོད་དང་། རང་ཤི་རྒྱག་མཁན་ཚོ་བསྙད་འདོགས་ཟིང་སློང་གི་ཉེས་མིང་འོག་བཀག་ཉར་བྱས་ཡོད་འདུག ན་གཞོན་ཚགས་པར་ཁང་གིས་མི་ ༧ གྱིས་རང་ཤི་ངོ་རྒོལ་བྱས་པའི་རྒྱུ་རྐྱེན་སྐོར་བརྗོད་པ་ལྟར་ན། ས་གནས་སྲིད་གཞུང་གིས་བཙན་གྱིས་ཁོང་ཚོའི་སྡོད་ཁང་གཏོར་བཤིག་བཏང་ཡོད་པ་མ་ཟད། སྲིད་གཞུང་གིས་སྤྲད་པའི་གུན་གསབ་ལ་འདོད་བློ་གཏན་ནས་མ་ཚིམ་པ་དང་། དེ་ལས་ཚབས་ཆེ་བའི་རྒྱུ་རྐྱེན་ཞིག་ནི། དེ་སྔོན་ས་གནས་སྲིད་གཞུང་གིས་ཁོང་ཚོ་བཀག་ཉར་བྱས་ཏེ། ཆབ་སྲིད་སློབ་སྦྱོང་འཛིན་གྲྭ་ཚོགས་ཡོད་པ་མ་ཟད། མགོ་ལ་རས་ནག་བསྐོན་ཏེ་གཅར་རྡུང་བཏང་བ་དང་ཟ་མ་མ་སྤྲད་པ། མཐོང་ཆུང་དམའ་འབེབས་སོགས་བྱས་ཡོད་སྟབས། ཁོང་ཚོ་ཆེད་དུ་པེ་ཅིན་ལ་ཡོང་སྟེ་རང་ཤི་ངོ་རྒོལ་བྱས་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
