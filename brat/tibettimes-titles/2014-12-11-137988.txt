རྒྱལ་སྤྱིའི་ཧི་མ་ལ་ཡའི་དུས་སྟོན།

དེ་རིང་དྷ་བོད་ཀྱི་ཟློས་གར་ཁང་དུ་རྒྱལ་སྤྱིའི་ཧི་མ་ལ་ཡའི་དུས་སྟོན་མགོ་འཛུགས་དང་འབྲེལ་རྒྱལ་ཁབ་མི་འདྲ་བ་ཁག་ཅིག་དང་ཧི་མ་ཅལ་མངའ་སྡེའི་ཡུལ་མི་ཚོས་གཞས་དང་ཞབས་བྲོ་འཁྲབ་སྟོན་བྱས་སོང་། དེ་རིང་ཕྱི་དྲོའི་ཆུ་ཚོད་ ༣།༠༠ ཐོག་བོད་ཀྱི་ཟློས་གར་ཚོགས་པའི་ཐང་ཆེན་དུ་ཧི་མ་ཅལ་མངའ་སྡེའི་ཡུལ་སྐོར་སྤྲོ་འཆམ་ལས་ཁུངས་ཀྱི་གོ་སྒྲིག་འོག ཉིན་གཉིས་རིང་གི་‘རྒྱལ་སྤྱིའི་ཧི་མ་ལ་ཡའི་དུས་སྟོན’ཞེས་པ་མགོ་འཛུགས་གནང་ཡོད་ཅིང་། སྐུ་མགྲོན་གཙོར་བོ་ཧི་མ་ཅལ་མངའ་སྡེའི་གྲོང་སྡེ་ལྷན་ཁང་གི་བློན་ཆེན་སུ་དྷེར་ཤཱར་མཱ་ ( Sudhir Sharma) མཆོག་གིས་དུས་སྟོན་དབུ་འབྱེད་གནང་ཡོད་ཅིང་། དེར་ཧི་མ་ཅལ་མངའ་སྡེའི་དཔོན་རིགས་ཁག་ཅིག་དང་། བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་སྲིད་སྐྱོང་བློ་བཟང་སེང་གེ་མཆོག་གཙོས་ནང་སྲིད་དང་ཤེས་རིག འཕྲོད་བསྟེན་བཅས་ཀྱི་བཀའ་བློན་རྣམ་པ་དང་། བོད་མི་མང་སྤྱི་འཐུས་ཚོགས་གཙོ་མཆོག འོས་བསྡུ་དང་འདེམས་ལྷན་གྱི་འགན་འཛིན། ཧིན་བོད་མཛའ་འབྲེལ་ཚོགས་པའི་ཚོགས་གཙོ་དང་། དྷ་སའི་གྲོང་སྡེ་ལས་ཁུངས་ཀྱི་ཚོགས་གཞོན་བཅས་ཕེབས་ཡོད་པ་མ་ཟད། གཞན་ཡང་དྷ་ས་ཁུལ་གྱི་རྒྱ་གར་དང་བོད་མི་བཅས་ཁྱོན་མི་ ༡༠༠༠ ཡས་མས་འདུ་འཛོམས་གནང་སོང་། དུས་སྟོན་གྱི་ཐོག་མར་སྐུ་མགྲོན་གཙོ་བོ་མཆོག་གིས་རྒྱ་གར་དང་བོད་མིའི་ཚོགས་པ་སྒྲིག་འཛུགས་ཁག་གིས་འགྲེམས་སྟོན་ཞུས་པའི་ཡུལ་ལུང་སོ་སོའི་ཐོན་རྫས་དང་འགྲེམས་སྟོན་ཁག་ལ་གཟིགས་ཞིབ་གནང་བ་འབྲེལ། སྐུ་མགྲོན་གཙོ་བོ་མཆོག་དང་སྲིད་སྐྱོང་མཆོག་གིས་སྣང་གསལ་ཞལ་འབྱེད་གནང་རྗེས། ཧི་མ་ཅལ་མངའ་སྡེའི་ལའོ་སི་པེ་དྷི་གཙོས་ཡུལ་ལུང་ཁག་དང་བོད་ཀྱི་ཟློས་གར་ཚོགས་པ། ཇ་མུ་ཀཱ་ཤི་མེར། ཨུ་དྷར་ཁན་ཌི་དང་། ཨུ་རུ་སུའི་རང་སྐྱོང་ས་ཁུལ་ཁལ་མིག་བཅས་ཀྱི་གཞས་པ་གཞས་མ་རྣམས་ཀྱིས་གཞས་དང་ཞབས་བྲོ་འཁྲབ་སྟོན་གནང་སོང་། ཉེ་ལམ་གོང་གནས་གནས་ཚུལ་སྐོར་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་དྲ་བའི་སྟེང་འཁོད་པ་ལྟར་ན། ཉིན་གཉིས་རིང་དུས་སྟོན་དེའི་སྟེང་། ཧི་མཱ་ཅལ་དང་ཨ་རུ་ནཱ་ཅལ། འབྲས་ལྗོངས། ཨུཏྟར་ཁཎྜ། ཇམ་མུ་ཀ་ཤི་མིར། པན་ཇབ། ར་ཇི་སི་ཐན་བཅས་མངའ་སྡེ་ ༧ ནས་རིག་གཞུང་འཁྲབ་སྟོན་པ་མི་གྲངས་ ༢༣༠ ཙམ་མཉམ་ཞུགས་གནང་རྒྱུ་མ་ཟད། ཨུ་རུ་སུ་དང་། སོག་པོ། འབྲུག ཁལ་མུཀ རྒྱ་གར། བལ་པོ་བཅས་ཡུལ་གྲུ་ཁག་ནས་ཀྱང་རིག་གཞུང་འཁྲབ་སྟོན་གནང་རྒྱུ་ཡིན་འདུག གསར་འགོད་པ། དོན་གྲུབ།
