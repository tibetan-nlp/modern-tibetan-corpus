བོད་རྒྱ་འབྲེལ་མོལ་བསྐྱར་གསོ་དགོས།

ཁ་སང་གཟའ་ཕུར་བུའི་ཉིན་གྱི་ཕྱི་དྲོ་ཧྥ་རན་སིའི་རྒྱལ་ས་ཕ་རི་སིར་བོད་མི་དང་བོད་དོན་རྒྱབ་སྐྱོར་བ་ཚོས་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་ལ་ངོ་རྒོལ་གནང་ཡོད་འདུག ཁ་སང་ཕྱི་ཚེས་ ༢༧ ཉིན་གྱི་ཕྱི་དྲོའི་ཆུ་ཚོད་ ༣།༠༠ པའི་ཐོག ཧྥ་རན་སིའི་རྒྱལ་ས་ཕ་རི་སིའི་བོད་རིགས་ཚོགས་པའི་གོ་སྒྲིག་འོག རྒྱལ་སའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ཐང་ཆེན་དུ་ཕ་རི་སིར་གནས་སྡོད་བོད་མི་དང་ཡུ་གུར་བ། རྒྱ་རིགས་མང་གཙོ་དོན་གཉེར་བ་བཅས་མི་ ༥༠༠ ཙམ་གྱིས། ལག་ཏུ་བོད་ཀྱི་རྒྱལ་དར་དང་འཕྲེད་དར་སོཊ་འཕྱར་བཞིན་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་ལ་ངོ་རྒོལ་གྱི་ལས་འགུལ་ཞིག་སྤེལ་ཡོད་པར་མ་ཟད། འགྲོ་བ་མིའི་ཐོབ་ཐང་ཐང་ཆེན་དུ་འཛོམས་ཏེ་བོད་དོན་དང་འབྲེལ་བའི་གཏམ་བཤད་སོགས་གནང་ཡོད་འདུག འདུ་འཛོམས་དེར་ཧྥ་རན་སིའི་གྲོས་ཚོགས་གོང་མའི་འཐུས་མི་སྐུ་ཞབས་ཨོ་ཊག་གྷ་ཌོ་ལིན་ (Audre Gattolin) ལགས་དང་། ཕ་རི་སི་ས་གནས་གྲོས་ཚོགས་འཐུས་མི་ཇོན་མག་བྷུ་ཝོལ་ (Jeun MarcBrule) ལགས་གཙོས། འབྲེལ་ཡོད་མི་སྣ་ཁག་ཆེད་ཕེབས་ཀྱིས་བོད་དོན་རྒྱབ་སྐྱོར་གྱི་གཏམ་བཤད་གནང་སྟེ། ཧྥ་རན་སིའི་སྲིད་འཛིན་གྱིས་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་ལ་བོད་རྒྱ་འབྲེལ་མོལ་མགོ་འཛུགས་རྒྱུར་སྐུལ་མ་གནང་དགོས་སྐོར་སོགས་བརྗོད་སོང་། སྐབས་དེར་ཕ་རི་སིའི་བོད་རིགས་ཚོགས་པའི་ཚོགས་གཙོ་ཐུབ་བསྟན་རྒྱ་མཚོ་ལགས་ཀྱིས། ད་བར་རྒྱ་ནག་གི་སྲིད་འཛིན་རིམ་པར་བརྗེ་བོ་བརྒྱབ་ཡོད་ཀྱང་། བོད་ཐོག་འཛིན་པའི་སྲིད་ཇུས་ལ་འགྱུར་བ་གང་མེད་པར། རྒྱ་གཞུང་གིས་མུ་མཐུད་བོད་ནང་མི་རིགས་གཉའ་གནོན་དང་དཔལ་འབྱོར་དབྱེ་འབྱེད། རིག་གཞུང་རྩ་གཏོར། ཁོར་ཡུག་གཏོར་རླག་གཏོང་བཞིན་པའི་སྲིད་ཇུས་ངན་པ་དང་། བོད་ཡུལ་རྒྱ་ཡུལ་སྒྱུར་བའི་སྲིད་བྱུས་ངན་པ་མཚམས་འཇོག་གིས་བོད་རྒྱ་འབྲེལ་མོལ་བསྐྱར་གསོ་དགོས་པའི་ནན་སྐུལ་ཞུས་སོང་། ཁ་སང་ཕྱི་དྲོ་བོད་མི་ཚོས་རྒྱལ་སའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ཐང་ཆེན་ནས་རྒྱ་ནག་གཞུང་ཚབ་ཁང་བར་གོམ་བགྲོད་གནང་སྟེ་འགྲོ་རྩིས་བྱས་ཀྱང་། ས་གནས་ཉེན་རྟོག་པས་བཀག་འགོག་བྱས་དེ་བསྐྱོད་ཐུབ་མི་འདུག་ལ། དེ་བཞིན་བོད་མིས་ངོ་རྒོལ་བྱ་རྒྱུར་དོགས་ཏེ། སྲིད་འཛིན་ཞིས་ཅིན་ཕིན་ཕེབས་ལམ་གྱི་ས་འོག་མེ་འཁོར་ཁག་ཅིག་ཀྱང་གནས་སྐབས་རིང་བཀག་འགོག་བྱས་ཡོད་འདུག ཧྥ་རན་སི་ནས། གནས་ཚུལ་མཁོ་གཏོང་བ་རིག་བཟང་།
