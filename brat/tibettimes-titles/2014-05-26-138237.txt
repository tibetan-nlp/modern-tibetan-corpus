རྒྱུན་ལས་ཁ་གསབ་ཀྱི་འོས་འདེམས།

དེ་རིང་ཕྱི་དྲོར་བོད་ཀྱི་དགུ་བཅུ་གསུམ་ལས་འགུལ་ཚོགས་པའི་ཛ་དྲག་ཚོགས་ཆེན་ཐོག་རྒྱུན་ལས་ ༡༡ གསར་འདེམས་བྱས་སོང་། ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༤ ནས་ཉིན་གཉིས་རིང་བོད་ཀྱི་དགུ་བཅུ་གསུམ་ལས་འགུལ་ཚོགས་པའི་ཚོགས་ཁང་དུ། ཚོགས་པའི་སྐབས་དྲུག་པའི་ཛ་དྲག་ཚོགས་ཆེན་སྐོང་ཚོགས་གནང་ཡོད་པ་དང་། ཐེངས་འདིའི་ཚོགས་ཆེན་ཐོག དྷ་ས་གཙོ་བོར་གྱུར་བའི་རྒྱ་བལ་འབྲུག་གསུམ་ནས་ཚོགས་མི་ ༡༠༠ ལྷག་འདུ་འཛོམས་བྱུང་ཡོད་པ་དང་། དེ་རིང་ཕྱི་ཚེས་ ༢༥ ཉིན་གྱི་ཉིན་རྒྱབ་ཏུ། བོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་དང་བུད་མེད་ལྷན་ཚོགས། བོད་རང་བཙན་སློབ་ཕྲུག་ཚོགས་པ། རྒྱལ་ཡོངས་དམངས་གཙོ་ཚོགས་པ་བཅས་ཀྱི་ཚོགས་གཙོ་རྣམ་པ་ཆེད་དུ་གདན་འདྲེན་བྱས་ཏེ། ལས་འགུལ་ཚོགས་པའི་སྐབས་དྲུག་པའི་དྲུང་ཆེ་སོགས་རྒྱུན་ལས་ ༡༡ ཨོ་སི་ཊོ་ལི་ཡར་གཞིས་སྤོར་གྱི་རྒྱན་ཁོངས་ཚུད་པར་ཁ་གསབ་ཀྱི་རྒྱུན་ལས་གསར་པ་འོས་འདེམས་བྱས་སོང་། གསར་དུ་འདེམས་པའི་རྒྱུན་ལས་ ༡༡ ལས་དང་པོ་རྒྱལ་ཐང་སྤྲུལ་སྐུ། གཉིས་པ། རྡོ་རྗེ་ཚེ་རིང་། གསུམ་པ། ཟླ་བ་བཀྲ་ཤིས། བཞི་པ། ཀླུ་མཁར་བྱམས། ལྔ་པ། པ་སངས་དབང་རྒྱལ། དྲུག་པ། སངས་རྒྱས། བདུན་པ། བློ་བཟང་བསྟན་འཛིན། བརྒྱད་པ། བློ་བཟང་ཉི་མ། དགུ་པ།  འཆི་མེད་རྡོ་རྗེ། བཅུ་པ། བཀྲ་ཤིས་རྒྱལ་པོ། བཅུ་གཅིག ངག་དབང་ཁྲི་རལ་བཅས་ཡིན་པ་དང་། ནང་ཁུལ་ནས་རྒྱུན་ལས་ཀྱི་ལས་འགན་གཏན་འབེབས་བྱ་རྒྱུ་ཡིན་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
