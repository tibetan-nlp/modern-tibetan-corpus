དུས་དབང་སོ་བཞི་པ་གསལ་བསྒྲཊ་གནང་།

ཕྱི་ལོ་ ༢༠༡༧ ཟླ་ ༡ ཚེས་ ༣ ནས་ ༡༣ བར་གནས་མཆོག་རྡོ་རྗེ་གདན་དུ་༧གོང་ས་མཆོག་ནས་དཔལ་དུས་ཀྱི་འཁོར་ལོའི་དབང་ཆེན་ཐེངས་སོ་བཞི་པ་གནང་གཏན་འཁེལ་འདུག འདི་ལོའི་ཕྱི་ལོ་གསར་ཚེས་སུ་༸གོང་ས་༸སྐྱབས་མགོན་ཆེན་པོ་མཆོག་ནས་རྒྱ་གར་ལྷོ་ཕྱོགས་བཀྲ་ཤིས་ལྷུན་པོ་དགོན་དུ་བཀའ་ཆོས་བཀའ་དྲིན་སྐྱོང་སྐབས། ཕྱི་ལོ་ ༢༠༡༧ ཟླ་ ༡ པོའི་ནང་དུས་འཁོར་དབང་ཆེན་བསྩལ་རྒྱུ་ཡིན་པའི་གསལ་བསྒྲགས་མཛད་ཡོད་པ་ལྟར། ཕྱི་ལོ་ ༢༠༡༧ ཟླ་ ༡ ཚེས་ ༣ ནས་ ༡༣ བར་གནས་མཆོག་རྡོ་རྗེ་གདན་དུ་དཔལ་དུས་ཀྱི་འཁོར་ལོའི་དབང་ཆེན་ཐེངས་ ༣༤ པ་གནང་གཏན་འཁེལ་བ་གསལ་བསྒྲགས་བྱེད་འཐུས་ཀྱི་བཀའ་བཟང་ཐོབ་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
