
ཐའེ་རྒྱ་གཉིས་ཀྱིས་གསང་ཉུལ་བ་བརྗེ་རེས་བྱས།

  གཟའ་ཉི་མར་ཐའེ་ཝན་དང་རྒྱ་ནག་གཉིས་ཀྱིས་ཕན་ཚུན་གྱི་གསང་ཉུལ་བ་སྤྲོད་ལེན་བྱས་ཡོད་ཅིང་། དེ་ནི་ཉེ་ལམ་ཐའེ་རྒྱ་གཉིས་ཀྱི་སྲིད་འཛིན་ཐུག་འཕྲད་བྱས་རྗེས་ཀྱི་བཟང་ཕྱོགས་ཀྱི་འགྱུར་བ་ཞིག་ཏུ་ངོས་འཛིན་བྱེད་ཀྱིན་ཡོད་འདུག   དེ་རིང་རོའུ་ཊར་ཁང་གིས་གནས་ཚུལ་སྤེལ་དོན་ལྟར་ན། ཁ་སང་རེས་གཟའ་ཉི་མའི་ཉིན་ཐའེ་རྒྱ་གཉིས་ནས་དེ་སྔོན་འཇུ་བཟུང་བྱས་ཡོད་པའི་ཕན་ཚུན་གྱི་གསང་ཉུལ་བ་དག་སྤྲོད་ལེན་བྱས་ཡོད་ཅིང་། སྐབས་དེར་རྒྱ་ནག་སྐམ་ས་ཆེན་མོས་ཕྱི་ལོ་ ༢༠༠༦ ལོར་འཛིན་བཟུང་བྱས་པའི་ཐའེ་ཝན་གྱི་དམག་དོན་རྟོག་ཞིབ་པ་ཀྲུ་ཀོང་ཞུན་ (Zhu Gongxun) དང་ཞུ་ཁྲང་ཀོ་ (Xu Changguo) ཟེར་བ་གཉིས་ཀློད་གྲོལ་བཏང་ཡོད་པ་དང་། ཐའེ་ཝན་མཚོ་འགག་གིས་ཀྱང་ཕྱི་ལོ་ ༡༩༩༩ ལོར་འཛིན་བཟུང་བྱས་པའི་རྒྱ་ནག་གི་གསང་ཉུལ་བ་ལི་ཀྲི་ཧོ་ (Li Zhihao) ཟེར་ཞིག་ཀློད་གྲོལ་བཏང་ཡོད་འདུག   རྒྱ་ནག་གི་གསང་ཉུལ་བ་ལི་ཀྲི་ཧོ་ཟེར་བ་དེས་འདས་པའི་ལོ་ངོ་ ༡༠ རིང་ཐའེ་ཝན་མཚོ་འགག་གི་དམག་དོན་ལ་ཉུལ་ཞིབ་བྱས་པའི་ནག་ཉེས་འོག་ཐའེ་པེ་གྲོང་ཁྱེར་དུ་ཚེ་བཙོན་གྱི་ཁྲིམས་ཐག་བཅད་པ་དང་། ཐའེ་ཝན་གྱི་དམག་དོན་རྟོག་ཞིབ་པ་ཀྲུ་ཀོང་ཞུན་དང་ཞུ་ཁྲང་ཀོ་གཉིས་ཀྱིས་རྒྱ་ནག་ནང་གསང་ཉུལ་བྱས་པའི་ནག་ཉེས་འོག་སྲོག་ཁྲིམས་བཅད་ཟིན་ནའང་། ད་ལྟའི་བར་སྲོག་ཁྲིམས་ལག་བསྟར་མ་བྱས་པར་རྒྱ་ནག་གི་བཙོན་འོག་ཏུ་ལོ་ ༩ འགྲོ་བཞིན་ཡོད་འདུག   ཐའེ་ཝན་སྲིད་འཛིན་ལས་ཁུངས་ཀྱི་མགྲིན་པ་ཆར་ལི་ཁྲིན་ (Charles Chen) ཟེར་བ་ཞིག་གིས་བསྒྲགས་གཏམ་སྤེལ་བའི་ནང་། ཕྱོགས་གཉིས་ཀྱི་གསང་ཉུལ་བ་བརྗེ་རེས་བྱས་དོན་ནི་ཉེ་ཆར་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་དང་། ཐའེ་ཝན་གྱི་སྲིད་འཛིན་མཱ་ཡོན་ཇོ་གཉིས་ཐུག་འཕྲད་དང་འབྲེལ་ཏེ། བཟང་ཕྱོགས་ཀྱི་འགྱུར་བ་ཞིག་ཡིན་ཞིང་།  སྲིད་འཛིན་མ་ལགས་ཀྱིས་སྐམ་མཚོ་གཉིས་ཀྱི་འབྲེལ་མོལ་རྒྱུན་མཐུད་དང་འབྱུང་འགྱུར་གྱི་དམིགས་འབེན་མངོན་འགྱུར་ཡོང་བར་རེ་བ་བྱེད་ཀྱི་ཡོད་སྐོར་སོགས་བརྗོད་ཀྱང་། ཐའེ་ཝན་རྒྱ་ནག་གཞུང་ཚབ་ཁང་གིས་ཐེངས་འདིའི་གསང་ཉུལ་བ་བརྗེ་རེས་བྱས་པའི་ཐད་བཟང་ངན་གྱི་གླེང་སློང་གང་ཡང་བྱས་མི་འདུག   རྒྱལ་སྤྱིའི་བརྒྱུད་ལམ་ཁག་ཏུ་སྲིད་དོན་དབྱེ་ཞིབ་པ་ཚོས་ཉེ་ལམ་སང་གྷ་པོར་ནང་མ་ཞི་གཉིས་ཐུག་འཕྲད་བྱས་པ་དེ་ནི་ཆབ་སྲིད་ཀྱི་ཟློས་གར་ཞིག་དང་། ད་རེབ་ཕན་ཚུན་གྱི་གསང་ཉུལ་བ་བརྗེ་རེས་བྱས་པ་འདི་ནི་རྒྱ་ནག་སྐམ་ས་ཆེན་མོ་དང་ཐའེ་ཝན་མཚོ་འགག་གཉིས་བར་ཆབ་སྲིད་ཀྱི་འབྲེལ་ལམ་བསྐྱར་མཐུད་ཡོང་ཐབས་གཞན་ཞིག་ཡིན་པ་མ་ཟད། སྔ་རྗེས་སུ་ཕྱོགས་གཉིས་ཀྱི་སྲིད་འཛིན་ཐུག་འཕྲད་དང་གསང་ཉུལ་བ་བརྗེ་རེས་བྱས་པའི་ཐད་ནས་ཐའེ་རྒྱ་མི་མང་ཁྲོད་གླེང་ཕྱོགས་སྣ་ཚོགས་ཡོང་སྲིད་ནའང་། རྒྱ་ནག་དམར་ཤོག་ཚོགས་པ་དང་ཐའེ་ཝན་གོ་མིན་ཏང་སྲིད་གཞུང་གཉིས་ལ་མཚོན་ན། འདི་ནི་ཕྱོགས་གཉིས་ཀྱི་དམིགས་འབེན་གལ་ཆེན་ཞིག་མངོན་འགྱུར་ཡོང་བའི་སྔ་ལྟས་ཤིག་ཡིན་སྲིད་པར་ཚོད་དཔག་བྱེད་ཀྱི་ཡོད་འདུག   གསར་འགོད་པ། པདྨ་རྒྱལ།
