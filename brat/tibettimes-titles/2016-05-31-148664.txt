བློ་གྲོས་རྒྱ་མཚོ་སླར་ཡང་འཛིན་བཟུང་།

ཟླ་ ༥ ཚེས་ ༡༤ མཚན་མོར་སོག་རྫོང་གི་ཆབ་སྲིད་བཙོན་ཟུར་བློ་གྲོས་རྒྱ་མཚོ་(སོག་མཁར་བློ་གྲོས) བསྐྱར་དུ་ལྷ་ས་ནས་འཛིན་བཟུང་བཀག་ཉར་བྱས་ཡོད་འདུག ཁོང་འཛིན་བཟུང་བྱེད་དགོས་དོན་དང་གང་དུ་བཙོན་འཇུག་བྱས་མིན་ཤེས་རྟོགས་ཐུབ་མེད། ནག་ཆུ་ས་ཁུལ་སོག་རྫོང་གི་ཆབ་སྲིད་བཙོན་པ་བློ་གྲོས་རྒྱ་མཚོ་ལགས་ཐོག་མར་སྒེར་འཁོན་ལ་བརྟེན་ནས་ ༡༩༩༣ ལོའི་ཟླ་ ༡ ཚེས་ ༡༧ ཉིན་ནག་ཆུ་འབྲིང་རིམ་ཁྲིམས་ཁང་ཟེར་བས་ལོ་བཅོ་ལྔའི་ཁྲིམས་ཐག་བཅད་རྗེས། ལྷ་ས་གྲྭ་བཞི་བཙོན་ཁང་དུ་བཙོན་འཇུག་བྱས་ཡོད་པ་དང་། ༡༩༩༥ ལོའི་ཟླ་ ༣ ཚེས་ ༤ ཉིན་ཁོས་བཙོན་ཁང་ནས་བཙོན་པ་ཡོངས་རྫོགས་ལ་སྐུལ་སློང་བྱས་པ་དང་། ཁོང་གིས་“བོད་རང་བཙན་ཡིན། རྒྱ་མི་བོད་ནས་ཕར་རྒྱུགས། རྒྱལ་བ་བསྟན་འཛིན་རྒྱ་མཚོ་སྐུ་ཚེ་ཁྲི་ལོ་བརྟན་པར་ཤོག བོད་མི་ས་ཡ་དྲུག་རྡོག་རྩ་གཅིག་སྒྲིལ་ཡིན།”ཞེས་བཙོན་ཁང་དུ་སྐད་འབོད་བྱས་པ་དང་། སོ་སོའི་ལག་པས་བྲིས་པའི་ཡིག་ཆ་ ༣༠༠ ལྷག་ཙམ་གཏོར་བ་སོགས་ལ་རྐྱེན་བྱས་ཏེ། བཙོན་སྲུང་མགོ་ཁྲིད་རྣམས་ཀྱིས་ཁྲིམས་ཁང་ལ་སྙན་ཞུ་བརྒྱབ་ནས། སྐབས་དེར་ཉིན་གསུམ་ལས་མ་འགྱངས་བར་དམར་གསོད་གཏོང་རྒྱུའི་བཀའ་རྒྱ་འབྱོར་ཡོད་པ་རེད། དེ་དང་དུས་མཚུངས་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་དམིགས་བསལ་བལྟ་རྟོག་ཚོགས་པའི་འགན་འཛིན་ནས། བློ་གྲོས་གསོད་མི་ཆོག་པའི་གནོན་ཤུགས་རྒྱ་གཞུང་ལ་སྤྲད་པ་མ་ཟད། རྒྱལ་སྤྱིའི་ཉེས་ཡང་ཚོགས་པ་ཁག་གཅིག་གིས་ཀྱང་ཡིག་འཕྲིན་དང་གནོན་ཤུགས་སྤྲད་རྐྱེན་རྒྱ་གཞུང་གི་ཁྲིམས་ཁང་ནས་ལོ་དྲུག་སྲོག་ཁྲིམས་ཕར་འགྱངས་དང་། ཆབ་སྲིད་ཐོབ་ཐང་ལོ་ ༣ འཕྲོག་ནས་ཁྱོན་དུས་བཀག་བཙོན་འཇུག་མི་ལོ་ ༢༡ ཁྲིམས་ཐག་བཅད་ཡོད་པ་རེད། བཙོན་ནས་སྐད་འབོད་བྱས་པའི་ལོ་དེའི་ཟླ་ ༣ ཚེས་ ༤ ནས་ཟླ་ ༤ ཚེས་ ༥ བར་སྡུག་སྦྱོང་མནར་གཅོད་མི་སྤྱོད་ལས་འདས་པ་མྱངས་པ་མ་ཟད། ཆུ་ཇ་ཕོར་དོ་དང་ཀྲིན་མོག་གཅིག་ཙམ་ལས་མ་སྤྲད་པ་མ་ཟད། མགོ་བོ་བཀྱག་ས་མེད་པའི་ལྕགས་ཀྱི་དྲ་བའི་ནང་བཅུག་ནས་ལུས་པོ་སྒུར་པོ་ཆགས་པ་བཟོས་པ་དང་། ལག་པའི་མཐེ་བོང་གཉིས་ལ་ལྕགས་རྒྱབ་ནས་མཚན་གང་གནམ་ལ་བཏགས་པ་སོགས་སྡུག་སྦྱོང་ཚད་མེད་བཏང་ཡོད་པ་རེད། དེ་བཞིན་བཙོན་སྲུང་མགོ་པས་ངོས་ལེན་མ་བྱས་ན་གསོན་པོར་སྡོད་རྩིས་མ་བྱེད་ཅེས་ཐ་ཚིག་བསྒྲགས་པ་དང་། བློ་གྲོས་ཀྱིས་ཁྱོད་ཚོས་ང་བསད་ཀྱང་ངས་ནམ་ཡང་ཁས་ལེན་བྱེད་རྒྱུ་མིན་ཞེས་བཤད་ཡོད་པ་རེད། མཐར་ཕྱི་ལོ་ ༢༠༡༣ ཟླ་ ༥ ཚེས་ ༢ ཉིན་ཀློད་གྲོལ་ཐོབ་ཡོད་འདུག བློ་གྲོས་རྒྱ་མཚོ་ནི་སོག་རྫོང་ཚ་དབྲོག་ཡུལ་ཚོའི་སོག་མཁར་གྲོང་སྡེ་ནས་ཡིན་པ་དང་། ཆུང་དུས་ཕའི་ས་ནས་གཏམ་ཡིག་རྩིས་སྦྱང་བའི་ཐོག ལུས་པོ་བདེ་པོ་ཡོད་པས་སོག་རྫོང་ཟློས་གར་ཚོགས་པའི་འཁྲབ་སྟོན་པ་བྱེད་མྱོང་ཡོད་ལ། ཤེད་ཤུགས་ལྡན་པས་ནག་ཆུ་ས་ཁུལ་སུ་ཤེད་རྡོ་བཀྱགས་ནས་ཨང་དང་པོའི་བྱ་དགའ་ཐེངས་གཉིས་རག་མྱོང་ཡོད། མི་རིགས་ཀྱི་ལ་རྒྱ་དང་དཔའ་བློ་ཁོག་ཆེ་ལ། མདོར་ན། བཙན་པོ་གཉའ་ནས་མནན་ཉམ་ཆུང་མཐའ་ནས་སྐྱོང་ཐུབ་མཁན་ཞིག་རེད་ཅེས་བཙོན་རོགས་རྣམས་ཀྱིས་གླེང་བཞིན་ཡོད་པ་དང་། ཁོང་ད་ལྟ་ལོ་ ༥༥ ཡིན་འདུག ཁོང་ལ་ད་ལྟའི་རང་ལོ་ ༥༥ ཡིན་པ་དང་ཨ་མ་དང་། སྲིང་མོ་ལྔ་དང་སྤུན་ཆུང་བ་གཅིག་བཅས་ཡོད་འདུག སྔོན་མ་བཙོན་དུ་མ་བཟུང་སྔོན་གྱི་སྐྱེ་དམན་དགའ་སྐྱིད་ལ་ལ་བུ་གཅིག་ཡོད་པ་དེ་ཡང་སློབ་ཆེན་ཐོན་རྗེས་ན་ཚ་ལྕི་མོ་ཕོགས་ནས་གྲོངས་ཡོད་འདུག ངག་དབང་ཐར་པས་མཁོ་སྤྲོད་བྱས།
