རྩོམ་པ་པོ་སྟག་སྨིག་བཙོན་གྲོལ་ཐོབ།

རྩོམ་པ་པོ་སྟག་སྨིག་མི་ལོ་བཞི་ཡི་ཁྲིམས་ཐག་གི་དུས་ཡུན་རྫོགས་ནས་དེ་རིང་བཙོན་ཁང་ནས་ཀློད་གྲོལ་ཐོབ་སྟེ་ཕྱིར་རང་ཁྱིམ་དུ་ལོག་ཡོད་འདུག རྒྱལ་གཅེས་པ་སྒྱིས་ཚང་སྟག་སྨིག་གམ། སྐལ་བཟང་ཚུལ་ཁྲིམས་ལགས་ཀྱིས་༼བོད་ལ་རང་དབང་མེད་པའི་སྐོར་གྱི་སྙན་ཞུ༽ཞེས་པ་ ༢༥༠༠ ཙམ་ཀན་སུའུ་དང་མཚོ་སྔོན། སི་ཁྲོན་བཅས་སུ་འགྲེམས་སྤེལ་བྱས་མཐར། རྒྱ་ནག་གཞུང་གིས་རྒྱལ་ཁབ་ཀྱི་ཆབ་སྲིད་དང་འགལ་ཡོད་པ་དང་། ཕྱི་ཕྱོགས་ཁ་བྲལ་བ་དང་ངན་འབྲེལ་གྱིས་རྒྱལ་ཁབ་མགོ་རྟིང་སློག་པའི་བྱ་ངན་བརྩམས་ཡོད་པ་བཅས་ཀྱི་ཉེ་འཛུགས་བྱས་ཏེ། ལོ་གཅིག་རིང་བྲོས་བྱོལ་གྱི་འཚོ་བ་བསྐྱལ་ཡོད། ༢༠༠༧ ལོའི་ཕྱི་ཟླ་ ༧ པའི་ཚེས་ ༢༧ ཉིན་གྱི་དགོང་མོར་སི་ཁྲོན་མཛོད་དགེ་རྫོང་ཐོག་ཏུ་འཛིན་བཟུང་གིས་དུས་ཡུན་གང་ཙམ་རིང་བཀག་ཉར་བརྒྱུད་གནས་སྐབས་རིང་ཀློད་གྲོལ་ཙམ་བཏང་ཡོད་ཀྱང་། ༢༠༡༠ ལོའི་ཕྱི་ཟླ་ ༡༢ བའི་ཚེས་ ༡༣ ཉིན་ཡང་བསྐྱར་འཛིན་བཟུང་གིས་ལོ་གཅིག་ལྷག་རིང་ཀན་ལྷོ་གཙོད་གྲོང་ཁྱེར་གྱི་བཙོན་ཁང་ཞིག་ཏུ་བཀག་ཉར་གྱིས་ནང་མི་གཉེན་འབྲེལ་སུ་གང་ལའང་ཐུག་འཕྲད་བྱས་བཅུག་མེད་པ་མ་ཟད། ༢༠༡༡ ལོའི་ཕྱི་ཟླ་ ༡༢ པའི་ཚེས་ ༣༠ ཉིན་ཀན་ལྷོ་ཉེན་རྟོག་པས་ཁོང་གི་ཕ་རྒན་ལ་ལོ་ ༤ དུས་བཀག་བཙོན་འཇུག་གི་ཁྲིམས་ཆད་བྱས་ཡོད་པའི་བརྡ་ལན་སྤྲད། རྒྱ་ནག་གི་བཙོན་འོག་ནས་འདས་པའི་མི་ལོ་ ༤ རིང་དཀའ་སྡུག་མནར་གཅོད་མྱངས་མཐར། དེ་རིང་བློ་བུར་དུ་༼བཙོན་ཁང་ལན་ཀྲུག་ཙང་ཧྲང་པེ་ཙ་ཡེ་ཅང་ཡེ་ཟེར༽བཙོན་ཁང་ནས་ཀློད་གྲོལ་བཏང་སྟེ་ཕ་ཡུལ་དུ་འབྱོར་ཡོད་འདུག རྩོམ་པ་པོ་ཁོང་ནི་མདོ་སྨད་བསང་ཆུ་རྫོང་སྒྱིས་ཚང་མང་དབང་ལུང་སྡེ་བའི་ཕ་བསམ་འཕེལ་དང་མ་སྐྱིད་རིས་གཉིས་ཀྱི་བུར་ ༡༩༨༠ སྐྱེས། ལོ་དགུའི་སྟེང་ས་གནས་སློབ་གྲྭར་འགྲིམས་ནས་བོད་རྒྱ་གཉིས་ཀྱི་སྐད་ཡིག་ལ་སྦྱངས། རང་ལོ་བཅུ་གསུམ་སྟེང་སྒྱིས་ཚང་དགའ་ལྡན་ཆོས་འཁོར་གླིང་དུ་ཆོས་སྒོར་ཞུགས། ༢༠༠༥ ལོའི་བར་གྲྭ་ཚང་གི་འདོན་ཆོག་གིས་བཙོས་ཐ་སྙད་རིག་གནས་སོགས་ལ་སྦྱང་བརྩོན་བྱས། ལོ་དེར་རྒྱུགས་སྤྲོད་ཀྱི་ལམ་ནས་བསང་ཆུ་རྫོང་ཀན་སི་ཞིང་ཆེན་ནང་བསྟན་སློབ་གླིང་ལ་ཞུགས་ཏེ་ ༢༠༠༩ ལོར་སློབ་མཐར་ཕྱིན་ཡོད།
