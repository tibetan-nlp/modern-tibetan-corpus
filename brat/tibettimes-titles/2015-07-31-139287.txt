ཨ་སྒོར་དུང་ཕྱུར་ ༦༠ ལྷག་གི་ཆུ་བེད།

རྒྱ་ནག་གཞུང་གིས་ཨ་སྒོར་དུང་ཕྱུར་ ༦༠ ལྷག་གི་འགྲོ་གྲོན་བཏང་སྟེ་བོད་ནང་འཐུང་ཆུ་ཐོན་སྐྱེད་བྱ་ཡུལ་གྱི་བཟོ་གྲྭ་ཞིག་གསར་རྒྱག་བྱ་རྒྱུ་ཡིན་འདུག ཁ་སང་རྒྱ་གར་གྱི་དཔལ་འབྱོར་དུས་བབ་ཐོག་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། རྒྱ་ནག་གཞུང་གིས་བོད་རང་སྐྱོང་ལྗོངས་ཀྱི་ལྷན་ཚོགས་ཤིག་གི་སྟེང་། མ་འོངས་པའི་ལོ་གསུམ་ནས་ལྔ་ནང་ཚུད་དུ་ཨ་སྒོར་དུང་ཕྱུར་ ༦༠ དང་ས་ཡ་ ༤༤ ཡི་འགྲོ་གྲོན་བཏང་སྟེ། བོད་རང་སྐྱོང་ལྗོངས་ནང་བོད་ཀྱི་དྭངས་ཤིང་གཙང་ལ་རྒྱུ་སྤུས་དག་པའི་ཆུ་བོའི་འབྱུང་ཁུངས་ཁག་སྤྱད་ནས་འཐུང་ཆུའི་ཆུ་བེད་བཟོ་གྲྭ་ཞིག་གསར་འཛུགས་བྱ་རྒྱུ་ཡིན་སྐོར་བཀོད་འདུག བོད་རང་སྐྱོང་ལྗོངས་ཆུ་བེད་ལས་ཁུངས་ཀྱི་འགན་འཛིན་གཞོན་པ་ཤིན་ཏུང་ལིང་ (Xiang Tongliang) གིས། ང་ཚོས་བོད་རང་སྐྱོང་ལྗོངས་ཀྱི་ཆུ་ཁུངས་ ༢༧ གྱི་སྤུས་ཚད་ལ་ཞིབ་འཇུག་བྱས་པ་བརྒྱུད། ཆུ་ཁུངས་ ༢༢ ཀྱི་ཆུ་རྒྱ་དཔངས་གྲུ་བཞི་མ་ས་ཡ་ ༡༩༠ འཐུང་འོས་པ་ར་འཕྲོད་བྱུང་ཡོད་སྟབས། གང་མགྱོགས་ལས་གྲྭ་མགོ་འཛུགས་ཀྱིས་ཆུ་ཁུངས་ཁག་བེད་སྤྱོད་གཏོང་ཐུབ་པ་བྱ་རྒྱུ་ཡིན། ཞེས་བརྗོད་འདུག རྒྱ་ནག་གི་གཞུང་འབྲེལ་བརྒྱུད་ལམ་ཤིན་ཧྭ་གསར་ཁང་གིས། བོད་ནི་རྒྱ་ནག་རྒྱལ་ཡོངས་ནང་ནས་ཆུ་བོའི་ཐོན་ཁུངས་ཕུན་སུམ་ཚོགས་ཤོས་ཤིག་ཡིན་པ་གང་ཞིག རྒྱལ་ཡོངས་ཀྱི་ཆུ་རྒྱ་དཔངས་གྲུ་བཞི་མ་ཐེར་འབུམ་ ༤༣༩།༤ ཟིན་གྱི་ཡོད་སྟབས། རྒྱལ་ཡོངས་ལས་ལྡབ་ ༦༠ ཡི་མང་བ་ཆགས་ཡོད་པ་མ་ཟད། དེ་དག་ནི་འབག་བཙོག་གི་དྲི་མ་ཙམ་ཡང་མེད་པའི་རང་བྱུང་ཁམས་ལས་བྱུང་བའི་དྭངས་གཙང་གི་ཆུ་ཡིན་སྟབས། ད་ལྟ་འཛམ་གླིང་ཐོག་གི་ཆུ་གཙང་ཤོས་དང་ཡག་ཤོས་ཆགས་ཡོད། ཅེས་བརྗོད་འདུག ༢༠༡༡ ལོར་རྒྱ་གཞུང་གིས་ཆུའི་སྤུས་ཚད་ཐད་རྟོག་ཞིབ་བྱས་སྐབས། རྒྱ་ནག་ནང་ཡོད་པའི་ཆུའི་ཐོན་ཁུངས་ཏེ། མཚོ་དང་ཆུ་མཛོད་ཆེ་ཁག་གི་ཆུ་མིས་བཏུང་བ་ཡིན་ན་ལུས་ཟུང་ལ་ཉེན་ཁ་ཡོད་པ་ཤེས་རྟོག་བྱུང་ཡོད་པ་མ་ཟད། ད་བར་ཡུལ་མི་ཚོས་བཀོལ་སྤྱོད་བྱེད་བཞིན་པའི་རྒྱ་ནག་རྒྱལ་ནང་གི་ས་ཁུལ་མི་འདྲ་བ་ ༤༧༠༠ ཙམ་གྱི་ཆུ་ལ་བརྟག་དཔྱད་གཏོང་སྐབས་བརྒྱ་ཆ་ ༥།༣ ཙམ་ལ་གཟུགས་པོར་གནོད་པའི་རྫས་སྦྱོར་ཡོད་སྟབས། མང་ཚོགས་ཀྱིས་མུ་མཐུད་བཏུང་ན་ངན་ཕྱོགས་ཀྱི་མཇུག་འབྲས་འཐོན་རྒྱུ་ཡིན་པའི་ཉེན་བརྡ་བཏང་ཡོད། ད་ལྟ་རྒྱ་ནག་ནང་འཛམ་གླིང་གི་འཐུང་ཆུ་བརྒྱ་ཆ་ ༧ ལས་མེད་སྟབས། ཉེ་ལམ་རྒྱ་ནག་གཞུང་གིས་བོད་ཀྱི་ཆུ་བོ་ཁག་ལ་ཆུ་རགས་བརྒྱབ་སྟེ་རྒྱ་ནག་ནང་ཁ་ཕྱོགས་བསྒྱུར་གྱིན་ཡོད། ༢༠༡༤ ལོར་རྒྱ་ནག་གཞུང་གིས་བོད་ཀྱི་འཐུང་ཆུའི་འབྱུང་ཁུངས་ཁག་སྤྱད་དེ། རྒྱལ་ཡོངས་ཀྱི་མང་ཚོགས་བརྒྱ་ཆ་ ༦༠ ཙམ་གྱི་ཆུའི་དཀའ་ངལ་སེལ་ཡོད་སྟབས། ཟླ་འདིའི་ཚེས་ ༢༩ ཉིན་རྒྱ་ནག་གཞུང་གིས་བོད་རང་སྐྱོང་ལྗོངས་ཀྱི་ལྷན་ཚོགས་ཤིག་གི་སྟེང་། མ་འོངས་པའི་ལོ་གསུམ་ནས་ལྔ་བར་ཨ་སྒོར་དུང་ཕྱུར་ ༦༠ ལྷག་གི་འགྲོ་གྲོན་བཏང་ནས་བོད་རང་སྐྱོང་ལྗོངས་ནང་ཆུ་བེད་བཟོ་གྲྭ་འཛུགས་ཏེ། འཐུང་ཆུའི་ཐོན་སྐྱེད་བྱེད་ཤུགས་ལྡབ་ ༣༠ ཙམ་འཕར་སྣོན་ཐུབ་པ་བྱས་ནས་ལོ་རེར་འཐུང་ཆུ་ཌོན་ ༡༥༣༠༠༠ ལྷག་མཁོ་འདོན་བྱ་རྒྱུ་ཡིན་འདུག བོད་རང་སྐྱོང་ལྗོངས་ཀྱི་མགོ་གཙོ་བློ་བཟང་རྒྱལ་མཚན་གྱིས། བོད་ནང་ཆུ་བེད་བཟོ་གྲྭ་གསར་རྒྱག་བྱེད་སྐབས། རང་བྱུང་ཁོར་ཡུག་ལ་གནོད་སྐྱོན་གཏང་ན་འགྲིག་གི་མེད་སྟབས། བོད་རང་སྐྱོང་ལྗོངས་སྲིད་གཞུང་གིས་འཐུང་ཆུའི་སྲོལ་སྒྲིག་ལ་སྲུང་སྐྱོབ་བྱས་ཏེ། ཁྲིམས་འགལ་བྱེད་རིགས་ལ་ཆད་པ་ནན་མོ་གཅོད་དགོས་ཞེས་བརྗོད་འདུག དེ་རིང་དབྱིན་ཡུལ་གྱི་ཉིན་རེའི་གསར་འཕྲིན་ཐོག་རྒྱལ་སྤྱིའི་ཁོར་ཡུག་སྲུང་སྐྱོབ་པ་ལྕམ་ཧྥེ་ར་པན་སི་ལགས་ཀྱིས། རྒྱ་ནག་གི་མགོ་ཁྲིད་ཚོས་བོད་ནང་གི་ཁོར་ཡུག་ལ་སྲུང་སྐྱོབ་བྱེད་ཀྱི་ཡོད་ཚུལ་རྗོད་བཞིན་ཡོད་ཀྱང་། དོན་དངོས་ཐོག་བོད་ནང་ཤིང་ནགས་གཅོག་འབྲེག་དང་ཁོར་ཡུག་འབག་བཙོག  གཙང་པོ་ཁ་སྒྱུར་དང་བཟོ་གྲྭ་གསར་རྒྱག་སོགས་བྱེད་བཞིན་ཡོད་སྟབས། བོད་ཀྱི་གནམ་གཤིས་ལ་འགྱུར་བ་ཐེབས་ཏེ། ད་ལྟ་འཛམ་གླིང་གི་ཚ་དྲོད་འཕེལ་ཏེ་འཁྱགས་རོམ་བཞུར་བ་དང་། ཆར་ཆུ་དུས་སུ་མ་བབ་པར་ཐན་སྐྱོན་ཚབས་ཆེན་འབྱུང་བཞིན་ཡོད་སྐོར་སོགས་བཀོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
