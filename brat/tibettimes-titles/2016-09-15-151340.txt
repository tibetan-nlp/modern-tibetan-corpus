བླ་རུང་སྒར་དང་ཡ་ཆེན་སྒར་གཏོར་བཤིག

དེ་རིང་ཧོང་ཀོང་རང་དབང་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བའི་ནང་། རྒྱ་ནག་གཞུང་གི་བཀའ་ཁྱབ་འོག་ད་ལྟའི་བར་གསེར་རྟ་བླ་རུང་སྒར་དང་ཡ་ཆེན་ཨོ་རྒྱན་བསམ་གཏན་གླིང་གཉིས་ནས་ཁྱོན་གྲྭ་བཙན་ ༢༠༠༠ ལྷག་ཕྱིར་ཕུད་བཏང་ཡོད་སྐོར་བརྗོད་འདུག རྒྱ་ནག་ནང་གི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལྟ་ཞིབ་ (HRW) ཚན་པའི་འགན་འཛིན་སོ་ཧྥེ་རེ་ཆར་ཌི་སོན་ (Sophie Richardson) ལགས་ཀྱིས། རྒྱ་ནག་གཞུང་གི་དབང་འཛིན་དཔོན་རིགས་ཚོས། བོད་མིའི་དགོན་སྡེ་ཁག་ཏུ་སྡོད་ཁང་འཛུགས་སྐྲུན་བྱེད་བཞིན་ཡོད་ལུགས་བཤད་པ་དེ་དངོས་ཡོད་གནས་ཚུལ་མིན་པར། ཁོ་ཚོས་བཙན་གྱིས་དགོན་སྡེ་ཁག་གི་གྲྭ་ཤག་གཏོར་བ་དང་། གྲྭ་བཙུན་དགོན་པ་ནས་ཕྱིར་ཕུད་གཏོང་བ། དེ་བཞིན། དགོན་སྡེ་ཁག་གི་ཆོས་ལུགས་བྱེད་སྒོའི་ཐོག་འཇིགས་སྐུལ་དང་དམ་དྲག་བྱེད་བཞིན་པ་དེས། བོད་ནང་དུ་ཆོས་ལུགས་རང་དབང་མེད་པ་ཧ་ཅང་གསལ་བོར་མཚོན་ཡོད་སྐོར་བརྗོད་འདུག གཞན། གནས་ཚུལ་དེའི་ནང་བོད་རང་སྐྱོང་ལྗོངས་ཟེར་བའི་དབང་འཛིན་དཔོན་རིགས་ཚོའི་བཀའ་རྒྱ་ཞིག་ལུང་འདྲེན་བྱས་པའི་ནང་། བོད་རང་སྐྱོང་ལྗོངས་ཀྱི་དཔོན་རིགས་ཚོས་ཡ་ཆེན་སྒར་གྱི་བཙུན་མའི་ནང་མི་ལ་བཙུན་མ་ཚོ་ཕྱིར་རང་ཁྱིམ་དུ་མ་ལོག་ཚེ། ཁོ་ཚོའི་ཐོབ་ཐང་ལག་འཁྱེར་གཞུང་བཞེས་བཏང་སྟེ་ཉེས་ཆད་ལྕི་མོ་གཅོད་རྒྱུ་ཡིན་སྐོར་དང་། ཡ་ཆེན་སྒར་གྱི་བཙུན་མ་ཚོ་ཕྱིར་རང་ཁྱིམ་དུ་ལོག་རྗེས་དགོན་སྡེ་གཞན་ནང་འཛུལ་ཞུགས་མི་ཆོག་པ་མ་ཟད། ཆོས་ཀྱི་བྱེད་སྒོ་སྣ་ཚོགས་སྤེལ་མི་ཆོག་པའི་བཙན་བཀའ་བཏང་ཡོད་སྐོར་སོགས་བརྗོད་འདུག ཕྱི་ལོ་ ༡༩༨༤ ལོར་ཨ་ཁྱིག་གྲུབ་དབང་འཇམ་དབྱངས་ལུང་རྟོགས་རྒྱལ་མཚན་མཆོག་གིས་ཕྱག་བཏབ་པའི་ཡ་ཆེན་ཨོ་རྒྱན་བསམ་གཏན་གླིང་ངམ་ཡ་ཆེན་སྒར་ནི་གསེར་རྟ་བླ་རུང་སྒར་དང་འདྲ་བར། བོད་མི་སེར་སྐྱ་དབྱེར་མེད་ཀྱིས་ནང་ཆོས་ལ་སློབ་གཉེར་དང་སྒྲུབ་པ་ཉམས་ལེན་གྱི་བསྟི་གནས་ཆེ་པོ་ཞིག་ཡིན་པ་དང་། དགོན་གནས་དེ་གར་གཏན་སྡོད་སློབ་གཉེར་བ་ཁྱོན་ ༡༠༠༠༠ ཡས་མས་ཤིག་ཡོད་འདུག   གསར་འགོད་པ། པདྨ་རྒྱལ།
