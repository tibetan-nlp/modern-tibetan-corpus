བཀྲ་ཤིས་སྐྱིད་ཀྱིས་རང་ལུས་མེར་བསྲེཊ།

མདང་དགོང་ཀན་ལྷོ་ཁུལ་བསང་ཆུ་རྫོང་བསང་ཁོག་ཡུལ་ཚོ་དངུལ་ར་སྡེ་བའི་བུད་མེད་བཀྲ་ཤིས་སྐྱིད་ཞུ་བ་ཞིག་གིས་རང་ལུས་མེར་བསྲེགས་གནང་ཡོད་འདུག འདི་གར་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། མདང་དགོང་ཀན་ལྷོ་ཁུལ་བསང་ཆུ་རྫོང་བསང་ཁོག་ཡུལ་ཚོའི་དངུལ་ར་སྡེ་བ་ནས་ཡིན་པ་བོད་རིགས་བུད་མེད་བཀྲ་ཤིས་སྐྱིད་ཞུ་བ་ཞིག་གིས། རྒྱ་གཞུང་གི་བཙན་གནོན་ལ་ངོ་རྒོལ་གྱིས་རང་ལུས་མེར་བསྲེགས་གནང་ཡོད་པ་དང་། ཅུང་ཙམ་ལས་མ་འགོར་བར་ཡུལ་མི་ཚོས་མཐོང་སྟེ་ཁོང་མོའི་ལུས་སྟེང་གི་མེ་བསད་ནས་སྐྱོབ་ཐབས་བྱས་ཡོད་ཀྱང་། མཚན་མོའི་ཆུ་ཚོད་ ༣ པ་ཡས་མས་སུ་སྐུ་གྲོངས་ཡོད་འདུག དེ་རིང་ཞོགས་པ་ནམ་ལངས་མ་ཐག་རྒྱ་གཞུང་ཉེན་རྟོག་པ་དང་དམག་མི་མང་པོ་ཡོང་སྟེ་ཁོང་མོའི་སྐུ་ཕུང་བཙན་འཁྱེར་བྱས་ཡོད་འདུག གནས་ཚུལ་ཐོན་རིམ་གཉིས་པ། ཁ་སང་ཕྱི་ཚེས་ ༢༧་ཉིན་བསང་ཁོག་ཡུལ་ཚོའི་ཐོག་རྒྱ་གཞུང་ཉེན་རྟོག་པ་དང་རྒྱ་མི་ལས་བྱེད་པ་བརྒྱ་དང་ལྔ་བཅུ་ཙམ་ཡོང་སྟེ། བསང་ཁོག་དངུལ་རྭའི་བོད་རིགས་ཁྱིམ་ཚང་ཞིག་གི་ཁང་པ་རྩ་གཏོར་གཏོང་སྟབས། ཁྱིམ་བདག་གིས་ཡ་ལན་སྤྲད་དེ་ལྕགས་མགོར་གྱིས་བརྒྱབ་ནས་ཉེན་རྟོག་པ་གཉིས་ཐང་ལ་འགྱེལ་དུ་བཅུག་ཅིང་། གཞན་ཚང་མ་ཁ་འཐོར་དུ་བཏང་ཡོད་ཀྱང་། མཐའ་མར་ཉེན་རྟོག་པས་ཁྱིམ་བདག་དེ་འཛིན་བཟུང་བྱས་ཡོད་འདུག གནས་ཚུལ་ཐོན་རིམ་གསུམ་པ། བཀྲ་ཤིས་སྐྱིད་ལགས་ནི་ཀན་ལྷོ་ཁུལ་བསང་ཆུ་རྫོང་བསང་ཁོག་ཡུལ་ཚོའི་དངུལ་ར་སྡེ་བ་ནས་ཡིན་ཞིང་། ད་ལྟ་རང་ལོ་ ༥༥ ཙམ་ཡིན་འདུག ཁོང་མོའི་ཟླ་བཟའ་སངས་རྒྱས་སྐྱབས་ཁོང་གཉིས་ལ་བུ་བཞི་དང་བུ་མོ་གཅིག་བཅས་བྱིས་པ་ལྔ་ཡོད་པ་མ་ཟད། དགུང་ལོ་ ༨༠ ལོན་པའི་པ་ལགས་ཞིག་ཀྱང་ཡོད་འདུག ཁོང་མོ་ནི་ཕྱི་ལོ་ ༢༠༠༩ ནས་ད་ལྟའི་བར་བོད་ནང་ཡོངས་གྲགས་སུ་རང་ལུས་མེར་སྲེག་གནང་མཁན་ ༡༤༢  པ་དང་། འདི་ལོར་རང་ལུས་མེར་སྲེག་གནང་མཁན་ ༧ པ་དེ་ཆགས་ཡོད་པ་རེད། གསར་འགོད་པ། པདྨ་མཚོ།
