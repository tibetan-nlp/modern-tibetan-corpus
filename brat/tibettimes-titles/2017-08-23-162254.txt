སངས་རྒྱས་ཀྱི་དབུ་ ༡༠ རྐུན་མར་ཤོར།

ལོ་ངོ ༥༠༠ ཙམ་གྱི་ལོ་རྒྱུས་ལྡན་ཞིང་རེ་རེར་ཨ་སྒོར་ ༤༥༠༠ ལྷག་གི་རིན་ཐང་ལྡན་པའི་སངས་རྒྱས་ཀྱི་འབུར་རྐོས་སྐུ་འདྲ་བཅུ་ཡི་དབུ་རྐུན་མས་བརྐུས་ཡོད་འདུག ཁ་སང་རྒྱ་ནག་ལྷོ་རྒྱུད་ཞོགས་པའི་ཆ་འཕྲིན་གྱིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༠ ཉིན་རྒྱ་ནག་གི་ལྷོ་ནུབ་སི་ཁྲོན་ཞིང་ཆེན་ནས་ལོ་ངོ་བརྒྱ་ཕྲག་ལྷག་སོང་བའི་སངས་རྒྱས་ཀྱི་སྐུ་འདྲ་བྲག་ངོས་སུ་འབུར་རྐོས་བྱས་པ་ ༥༣ ཡོད་པ་ལས་སྐུ་འདྲ་ ༡༠ ཡི་དབུ་རྐུན་མར་ཤོར་ཡོད་པ་དང་། ད་ལྟ་འབྲེལ་ཡོད་ཉེན་རྟོག་པས་གནད་དོན་དེའི་ཐད་ཞིབ་འཇུག་དང་རྐུན་མ་རྩད་གཅོད་གཏོང་བཞིན་ཡོད་འདུག སངས་རྒྱས་ཀྱི་སྐུ་འདྲ་དེ་དག་སི་ཁྲོན་ཞིང་ཆེན་གྱི་ཅཱ་ཅིང་རྫོང་ཁོངས་སུ་ཡོད་པ་དང་། རྒྱ་ནག་གི་མིང་རྒྱལ་རབས་(༡༣༦༨-༡༦༤༤)སྐབས་སུ་བྲག་ངོས་ལ་འབུར་རྐོས་བྱས་པའི་བཤད་སྲོལ་ཡོད་ཅིང་། དེ་ནི་ཧཱང་རྒྱལ་རྒྱུད་(༢༠༦ BC-AD ༢༢༠)ཀྱི་སྲིད་དབང་སྐབས་བྱུང་བའི་མཁས་དབང་གྲགས་ཅན་ཕང་དི་གོང་(Pang Degong)གི་བཞུགས་ཡུལ་གྱི་ཉེ་འགྲམ་དུ་ཡོད་འདུག བྲག་རྐོས་དེའི་ཉེ་འཁྲིས་སུ་གནས་སྡོད་བྱས་ཡོད་པའི་རང་ལོ་ ༤༠ ཡིན་པའི་ཡུལ་མི་ཀྲུའུ་མི་ཁྲིན་ཟེར་བ་ཞིག་གིས། ཕྱི་ཚེས་ ༢༠ ཉིན་ཁོང་གི་ཕ་རྒན་བྲག་ཕུག་དེའི་གཤམ་འོག་གི་རྫིང་བུའི་ནང་ཉ་འཛིན་དུ་ཕྱིན་ཏེ་ཉིན་གུང་ནང་དུ་ཟ་མ་བཟའ་རུ་ཡོང་སྐབས། བྲག་ངོས་ཀྱི་སྐུ་འདྲ་དག་གི་དབུ་མེད་པ་མཐོང་ཡོད་པ་དང་། དབུ་རྣམས་རྐུན་མས་བརྐུས་པ་ཤེས་ཏེ་འཕྲལ་དུ་ས་གནས་ཉེན་རྟོག་ཁང་ལ་ཞུ་གཏུག་བྱས་ཡོད་སྐོར་བརྗོད་འདུག ཡུལ་དེའི་གནའ་རྫས་སྲུང་སྐྱོབ་ཚོགས་པས། ཞིབ་འཇུག་བྱས་པ་བརྒྱུད་སངས་རྒྱས་ཀྱི་སྐུ་འདྲའི་དབུ་དེ་དག་ཕྱི་ཚེས་ ༡༩ ཉིན་གྱི་མཚན་མོ་ཡང་ན་། ཕྱི་ཚེས་ ༢༠ ཉིན་གྱི་ནམ་མ་ལངས་གོང་དུ་རྐུན་མར་ཤོར་ཡོད་སྐོར་བརྗོད་པ་ཙམ་ལས་དེ་འབྲེལ་གྱི་ཞིབ་ཕྲའི་གནས་ཚུལ་གང་ཡང་བརྗོད་མེད་པ་རེད། གནས་ཚུལ་དེ་བྱུང་རྗེས་ཡུལ་དེའི་གནའ་རྫས་ཉོ་ཚོང་བྱེད་མཁན་ཚོས། སངས་རྒྱས་ཀྱི་སྐུ་འདྲའི་དབུ་དེ་དག་ནག་ཚོང་རྒྱག་མཁན་ལ་ལྐོག་ཏུ་བཙོང་བ་ཡིན་ནའང་དབུ་རེ་རེ་ཨ་སྒོར་ ༤༥༠༠ ལྷག་ལ་བྲིན་ཐུབ་ངེས་ཡིན་ཚུལ་བརྗོད་ཡོད་འདུག འབུར་རྐོས་སྐུ་འདྲ་དེ་དག་ཅཱ་ཅིང་རྫོང་མུ་ཁྲིན་གྲོང་ཚོའི་རི་ངོས་ཤིག་ཏུ་ཡོད་པའི་ལུང་ཕོ་བྲག་ཕུག་ཏུ་ཡོད་པ་དང་། བྲག་ཕུག་གི་ཕྱོགས་བཞི་རི་བོས་བསྐོར་ཞིང་། ད་ལྟའི་བར་འབུར་རྐོས་སྐུ་འདྲ་དེ་དག་ལ་ལོ་ངོ་ ༥༠༠ ཙམ་གྱི་ལོ་རྒྱུས་ཡོད་སྟབས། རྒྱ་ནག་རྒྱལ་ཡོངས་ཀྱི་ནང་པ་དང་ལོ་རྒྱུས་དོ་གཉེར་ཅན་ཁོ་ན་ཙམ་མིན་པར། འཛམ་གླིང་གི་ཕྱོགས་གང་ས་ནས་ཡུལ་སྐོར་སྤྲོ་འཆམ་པ་དང་གནས་སྐོར་བ་སྟོང་ཕྲག་མང་པོ་ཡོང་གི་ཡོད་འདུག གསར་འགོད་པ། ཉི་མ་ཚེ་རིང་།
