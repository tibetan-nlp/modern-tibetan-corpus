གྲྭ་བ་གཉིས་ལ་ཁྲིམས་ཐག་བཅད་འདུག

ཉེ་ལམ་རྔ་པ་རང་སྐྱོང་ཁུལ་འབར་ཁམས་འབྲིང་རིམ་ཁྲིམས་ཁང་གིས་རྔ་པ་ཀིརྟི་དགོན་པའི་གྲྭ་བ་གཉིས་ལ་ལོ་གཉིས་དང་གསུམ་གྱི་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག གནས་ཚུལ་འབྱོར་གསལ་ལ། ཟླ་འདིའི་ཚེས་ ༧ ཉིན་གྱི་བོད་ནང་གི་སྔ་དྲོའི་ཆུ་ཚོད་ ༡༡ ཙམ་ལ་རྔ་པ་ཀིརྟི་དགོན་པའི་གྲྭ་བློ་བཟང་བསྟན་པ། རང་ལོ་༡༩ ཡིན་པ་ཁོང་ལ་རྔ་པ་རང་སྐྱོང་ཁུལ་གྱི་ལྟེ་གནས་འབར་ཁམས་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་ཟེར་བས་རྒྱལ་ཁབ་ལ་ངོ་ལོག་གི་བྱ་སྤྱོད་སྤེལ་བའི་ཉེས་མིང་འོག་ལོ་ ༢ ཀྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་ཅིང་། མི་འགྱང་བར་ས་གནས་གཞན་གྱི་བཙོན་ཁང་ཞིག་ཏུ་བསྐྱལ་རྒྱུ་ཡིན་འདུག ཁོང་ནི་ད་ལོའི་ཕྱི་ཟླ་ ༤  ཚེས་ ༢༦ ཉིན་རྔ་པ་རྫོང་མཁར་གྱི་གཞུང་ལམ་དབུས་མ་ནས་མགོ་ལ་བོད་ཀྱི་རྒྱལ་དར་བྲིས་མ་ཞིག་དཀྲིས་ཤིང་ལག་ཏུ་གོང་ས་མཆོག་གི་སྐུ་པར་ཆེན་པོ་ཞིག་བཟུང་ནས་ “བོད་ལ་རང་དབང་དགོས་པ་དང་། རྒྱལ་བ་རིན་པོ་ཆེ་བོད་ལ་ཕེབས་འཇུག་དགོས་”ཞེས་པའི་འབོད་ཚིག་སྒྲོག་བཞིན་གོམ་བགྲོད་གནང་མཁན་དེ་ཡིན་ལ། སྐབས་ཐོག་དེར་ཁྲོམ་ལམ་དུ་ཡོད་པའི་ཉེན་རྟོག་པ་རྣམས་ཀྱིས་འཛིན་བཟུང་གིས་མནར་གཅོད་མང་པོ་བཏང་ཡོད་ཅིང་། ཁོང་ལ་ཁྲིམས་ཐག་གཅོད་སྐབས་ནང་མིར་བརྡ་བཏང་སྟེ་ཁྲིམས་རྭར་ཡོང་བཅུགས་ཡོད་ཀྱང་ནང་མིར་ཁྲིམས་རྩོད་པ་གླ་རྒྱུའི་གོ་སྐབས་སྤྲད་མི་འདུག ཁོང་གི་ཕ་ཡུལ་བོད་མདོ་སྨད་རྔ་པ་རྫོང་རྨེའུ་རུ་མ་ཡུལ་ཚོའི་རུ་ཆེན་གཉིས་པ་ཡིན་ཞིང་། ཕ་ངག་ཆུང་དང་མ་སྒྲོལ་མ་ཆོས་ཁོ་གཉིས་ཀྱི་བུ་ཆུང་བ་ཡིན་པ་དང་། ཆུང་ཆུང་སྐབས་ནས་རྔ་པ་ཀིརྟི་དགོན་པར་གྲྭ་པ་བྱས་ཏེ་དེང་སྐབས་རྔ་པ་ཀིརྟི་དགོན་པའི་སྔོན་འགྲོའི་འཛིན་གྲྭ་དགུ་བ་ཡིན་འདུག ཁོང་ལ་མ་གཅིག་གི་སྤུན་རྒན་པ་གྲྭ་པ་གཉིས་ཡོད་པའི་གཅིག་དེང་སྐབས་རྒྱ་གར་ལྷོ་ཕྱོགས་འབྲས་སྤུངས་སུ་སློབ་གཉེར་གནང་གི་ཡོད་ལ། གཞན་དེ་རྔ་པ་ཀིརྟི་དགོན་པར་ཡོད་འདུག དེ་བཞིན་གོང་ཚེས་ཉིན་ཁོང་དང་མཉམ་དུ་རྔ་པ་ཀིརྟི་དགོན་པའི་གྲྭ་བློ་བཟང་རྒྱ་མཚོ། རང་ལོ་ ༢༠ ཡིན་པ་ཁོང་ལ་ཡང་རྒྱལ་ཁབ་ལ་ངོ་ལོག་གི་བྱ་སྤྱོད་སྤེལ་བའི་ཉེས་མིང་འོག་ལོ་ ༣ གྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་ཅིང་། མི་འགྱང་བར་ས་གནས་གཞན་གྱི་བཙོན་ཁང་ཞིག་དུ་བསྐྱལ་རྒྱུ་ཡིན་འདུག ཁོང་གིས་ཀྱང་ད་ལོའི་ཕྱི་ཟླ་ ༤ ཚེས་ ༢ ཉིན་རྔ་པ་རྫོང་མཁར་གྱི་གཞུང་ལམ་དུ་ཐོན་ནས་ལག་ཏུ་ཁོང་རང་ཉིད་ཀྱིས་བྲིས་པའི་བོད་ཀྱི་རྒྱལ་དར་བཟུང་ནས་“བོད་ལ་རང་དབང་དགོས་པ་དང་། རྒྱལ་བ་རིན་པོ་ཆེ་བོད་ལ་གདན་འདྲེན་ཞུ་དགོས་”ཞེས་པའི་འབོད་ཚིག་སྒྲོག་བཞིན་སྐད་འབོད་གནང་ཡོད་འདུག་ལ། སྐབས་ཐོིག་དེར་ཉེན་རྟོག་པའི་ལག་ཏུ་ཚུད་མེད་པར་ཡར་དགོན་ནང་དུ་འབྱོར་ཐུབ་ཙམ་བྱུང་ཡོད་ཀྱང་། ཕྱི་ཟླ་ ༤ ཚེས་ ༡༥ ཉིན་དགོན་པའི་ནང་ནས་ཉེན་རྟོག་པས་འཛིན་བཟུང་བྱས་ཏེ་ད་བར་བཀག་ཉར་ཁང་སོགས་བརྒྱུད་དེ་མནར་གཅོད་རིམ་པར་མྱངས་མཐར་ཐག་གཅོད་བྱས་པ་ཞིག་ཡིན་འདུག ཁོང་ལ་ཁྲིམས་ཐག་གཅོད་སྐབས་ནང་མིར་བརྡ་བཏང་སྟེ་ཁྲིམས་རྭར་ཡོང་བཅུགས་ཡོད་ཀྱང་ཁྲིམས་རྩོད་པ་གླ་རྒྱུའི་གོ་སྐབས་སྤྲད་མེད་པ་གཅིག་མཚུངས་རེད། ཁོང་གི་ཕ་ཡུལ་རྔ་པ་རྫོང་རྨེའུ་རུ་མ་ཡུལ་ཚོའི་རུ་ཆེན་གསུམ་པ་ཡིན་ཞིང་། ཕ་བྷེ་ཚེ་དང་། མ་ཞིས་ཆུང་གཉིས་ཀྱི་བུ་ཡིན་ཞིང་། སྲིང་མོ་ཆུང་ཆུང་ཞིག་ཀྱང་ཡོད་འདུག ཆུང་དུས་ནས་རྔ་པ་ཀིརྟི་དགོན་པར་གྲྭ་པ་བྱས་ཏེ་དེང་སྐབས་སྔོན་འགྲོའི་འཛིན་གྲྭ་དགུ་པ་ཡིན་འདུག གོང་གི་གནས་ཚུལ་འདི་ཀིརྟི་བྱེས་པ་གྲྭ་ཚང་གི་གྲྭ་ཀ་ཉག་ཚེ་རིང་ལགས་ཀྱིས་བཏང་འབྱོར་བྱུང་།
