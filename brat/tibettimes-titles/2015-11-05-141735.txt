ལིས་ཁྲུང་ཞིར་ལོ་ ༡༢ ཀྱི་ཁྲིམས་ཐག་བཅད།

ལྐོག་ཟ་རུལ་སུངས་ཚབས་ཆེན་བྱས་པའི་ནག་ཉེས་འོག་དེ་སྔོན་དཀར་མཛེས་ཁུལ་དང་རྔ་པ་ཁུལ་གྱི་དཔོན་པོ་བྱས་མྱོང་བའི་སི་ཁྲོན་ཞིང་ཆེན་སྲིད་གྲོས་སྐབས་ ༡༡ པའི་མགོ་ཁྲིད་ལིས་ཁྲུང་ཞི་ལ་ལོ་ ༡༢ ཀྱི་ཁྲིམས་ཐག་བཅད་འདུག   རྒྱ་ནག་ཤིན་ཧྭ་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཟླ་འདིའི་ཕྱི་ཚེས་ ༣ ཉིན་ཅང་ཞི་ཞིང་ཆེན་ནན་ཁྲང་གྲོང་ཁྱེར་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་གིས། ལྐོག་ཟ་རུལ་སུངས་དང་ལྐོག་རྔན་འབོར་ཆེན་བླངས་པའི་ནག་ཉེས་འོག སི་ཁྲོན་ཞིང་ཆེན་སྲིད་གྲོས་སྐབས་བཅུ་གཅིག་པའི་མགོ་ཁྲིད་ཟུར་པ་ལིས་ཁྲུང་ཞི་ཟེར་བ་དེར་ལོ་ ༡༢ ཀྱི་བཙོན་འཇུག་ཁྲིམས་ཆད་དང་། སྒེར་གྱི་རྒྱུ་ནོར་ལས་རྒྱ་སྒོར་ཁྲི་ ༡༠༠ གཞུང་བཞེས་བཏང་ཡོད་པ་མ་ཟད། ཁོས་ལྐོག་རྔན་འབོར་ཆེན་བླངས་བ་དེར་མུ་མཐུད་ཞིབ་བཤེར་བྱེད་བཞིན་ཡོད་སྟབས། ལྐོག་རྔན་རྣམས་རྩད་གཅོད་བྱས་ཏེ་རྒྱལ་ཁབ་ཀྱི་བང་མཛོད་དུ་བཅུག་རྒྱུ་ཡིན་པའི་ཁྲིམས་ཐག་བཅད་ཡོད་པ་ལྟར། ལིས་ཁྲུང་ཞི་ཡིས་དང་ལེན་བྱས་ཡོད་འདུག   ལིས་ཁྲུང་ཞི་ཡིས་ཕྱི་ལོ་ ༢༠༠༦ ཟླ ༥ པ་ནས་ཕྱི་ལོ་ ༢༠༡༢ ཟླ ༨ པའི་བར། ཁྲིན་ཏུའུ་འཛུགས་སྐྲུན་ལས་གཞི་སྤྱིའི་ཚོང་གཉེར་ཁང་དང་སི་ཁྲོན་ལྡུམ་རའི་ས་ཁང་བདག་ཁོངས་གསར་སྤེལ་ཚད་ཡོད་ཚོང་གཉེར་སོགས་ཁེ་ལས་ཁང་གང་མང་དང་ངན་འབྲེལ་བྱས་ཏེ། ས་ཁང་བདག་བཟུང་དང་སྒེར་གྱི་དབང་ཆ་རྩོད་ལེན་བྱས་ཡོད་པ་མ་ཟད། ཕྱི་ལོ་ ༢༠༠༦ ཟླ་ ༥ ་ནས་ ༢༠༡༢ ཟླ་ ༨ པ་བར་ཐད་ཀར་བླངས་བ་དང་ཡང་ན་ཁོ་རང་གི་བཟའ་ཟླ་ལིས་མུན་ཡུ་སོགས་བརྒྱུད་དེ། ཁྱོན་རྒྱ་སྒོར་དུང་ཕྱུར་གཅིག་ལྷག་( ༡.༡༠༩.༦༣༢.༡༣)གི་ལྐོག་རྔན་བླངས་ཡོད་འདུག   ཁོ་ཕྱི་ལོ་ ༡༩༧༢ ནས་སི་ཁྲོན་ཞིང་ཆེན་སྨན་རིགས་སྡེབ་སྦྱོར་བཟོ་གྲྭར་ལས་ཞུགས་བྱས་ཡོད་པ་དང་། ཕྱི་ལོ་ ༡༩༩༥ ནས་ ༡༩༩༦ བར་དཀར་མཛེས་ཁུལ་གྱི་མགོ་ཁྲིད་གཞོན་པ་དང་། ཕྱི་ལོ་ ༡༩༩༦ ནས་ ༢༠༠༠ བར་རྔ་པ་ཁུལ་གྱི་མགོ་ཁྲིད་ཀྱི་འགན་འཁུར་ཡོད་འདུག ཕྱི་ལོ་ ༢༠༠༠ ནས་བཟུང་སི་ཁྲོན་ཞིང་ཆེན་དམར་ཤོག་ཚོགས་པའི་དྲུང་ཆེ་དང་། རིམ་གྱིས་སི་ཁྲོན་ཞིང་ཆེན་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ཁང་གི་མགོ་ཁྲིད་དང་མང་ཆེན་གྱི་མགོ་ཁྲིད་གཞོན་པ་སོགས་བྱས་ཏེ། ཕྱི་ལོ་ ༢༠༡༣ ཟླ་དང་པོའི་ནང་སི་ཁྲོན་ཞིང་ཆེན་སྲིད་གྲོས་སྐབས་ ༡༡ པའི་མགོ་ཁྲིད་ཀྱི་འགན་འཁུར་ཡོད་འདུག ཡིན་ནའང་། ལོ་དེའི་ཕྱི་ཟླ་ ༡༢ པའི་ནང་རྒྱ་ནག་དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ཁང་གིས། ཁོ་བོར་ཞིབ་བཤེར་བྱེད་མགོ་བརྩམས་པ་ཕྱིར་བསྒྲགས་བྱས་ཡོད་པ་མ་ཟད། ཕྱི་ལོ་ ༢༠༡༤ ཟླ་དང་པོའི་ནང་སྐོང་ཚོགས་བྱས་པའི་སི་ཁྲོན་ཞིང་ཆེན་སྐབས་ ༡༡ པའི་རྒྱུན་ལས་ཚོགས་ཆེན་ཐེངས་གཉིས་པའི་ཐོག ཁོ་བོ་ལས་གནས་ནས་གནས་དབྱུང་བཏང་བ་ཡོངས་ཁྱབ་ངོས་འཛིན་བྱུང་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
