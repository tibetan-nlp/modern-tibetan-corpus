སྟོབས་རྒྱལ་ལགས་ཨང་གོ་གཅིག་པ་ཆགས།

བོད་པའི་ཕོ་གསར་སྟོབས་རྒྱལ་ལགས་བྷར་རྫིལ་ཨོ་རྩེད་འགྲན་ཚོགས་སྟེང་རྨིད་ ༥༠༠༠ ཅན་གྱི་སྐྱེས་པའི་རྒྱང་རྒྱུག་འགྲན་བསྡུར་ཁྲོད་གོ་གཅིག་པ་ཆགས། དེ་རིང་ཕྱི་དྲོའི་རྒྱ་གར་གྱི་ཆུ་ཚོད་ ༦།༠༠ སྟེང་། ལྷོ་ཨ་མི་རི་ཁའི་བྷར་རྫིལ་(Brazil)རྒྱལ་ཁབ་ནང་འཚོགས་བཞིན་པའི་རྒྱལ་སྤྱིའི་ཨོ་ལིམ་ཕིག་གི་རྩེད་འགྲན་སྟེང་། རྨིད་ ༥༠༠༠ ཅན་གྱི་སྐྱེས་པའི་རྒྱང་རྒྱུག་འགྲན་བསྡུར་ཁྲོད། རྒྱ་ནག་གི་རྩེད་འགྲན་རུ་ཁག་གི་མིང་ཐོག་ནས་མཉམ་ཞུགས་གནང་ཡོད་པའི་བོད་པའི་ཕོ་གསར་སྟོབས་རྒྱལ་ལགས་ཨང་ ༩༡ པ་ཆགས་ཡོད་པ་རེད། ཁོང་ད་ལྟ་རང་ལོ་ ༢༢ ཡིན་འདུག རྒྱང་རྒྱུག་ཁྲོད་ཨང་དང་པོ་ཁཱན་ཡ་རྒྱལ་ཁབ་ཀྱི་ལུས་རྩལ་པ་(Eliud Kipchoge)དང་། ཨང་གཉིས་པ་ཨེ་ཐེའོ་པི་ཡཱ་རྒྱལ་ཁབ་ཀྱི་ལུས་རྩལ་པ་(Feyisa Lilesa)། ཨང་གསུམ་པ་ཨ་རིའི་ལུས་རྩལ་པ་(Galen Rupp)བཅས་ཆགས་ཡོད་པ་རེད། ཐེངས་འདིར་ཁྱོན་འགྲན་ཞུགས་པ་ ༡༤༠ ཙམ་ཡོད་པའི་ཁྲོད། ཨང་དང་པོ་ཐོབ་མཁན་གྱི་མགྱོགས་ཚད་དུས་ཡུན་(2:08:44) དང་། ཨང་གཉིས་པ་ཐོབ་མཁན་དུས་ཡུན་(2:09:54)། ཨང་གསུམ་པ་ཐོབ་མཁན་དུས་ཡུན་(2:10:05)། བོད་པའི་ལུས་རྩལ་པ་སྟོབས་རྒྱལ་ལགས་དུས་ཡུན་(2:24:22) བཅས་ཡིན་སྟབས། ཨང་དང་པོ་དང་སྟོབས་རྒྱལ་གཉིས་བར་སྐར་མ་ ༡༥ ལྷག་གི་ཁྱད་པར་ཤོར་ཡོད། ཐེངས་འདིའི་རྒྱང་རྒྱུག་འགྲན་བསྡུར་ཁྲོད་རྒྱ་ནག་གི་ལུས་རྩལ་པ་ ༣ ཡོད་པ་ལས་ལུས་རྩལ་པ་(Dong Guojian)ཟེར་བ་དེ་ཨང་རིམ་ ༢༩ དང་། སྟོབས་རྒྱལ་ལགས་ཨང་རིམ་ ༩༡ ། གཞན་རྒྱ་རིགས་(Zhu Renxue)ཟེར་བ་དེ་ཨང་རིམ་ ༩༦ པ་བཅས་ཆགས་ཡོད་པ་རེད། ཐེངས་འདིའི་འགྲན་བསྡུར་ལ་བོད་ཕྱི་ནང་གཉིས་ཀྱི་བོད་མི་མང་པོ་ཞིག་གིས་རེ་ལྟོས་ཆེན་པོ་བཅངས་ཡོད་པ་དང་། མཇུག་ཏུ་འགྲན་བསྡུར་ཁྲོད་སྟོབས་རྒྱལ་ལགས་སུ་ཨང་གོ་གཅིག་པ་ལས་མ་ཐོབ་པ་དེར་སྤྱི་ཚོགས་བརྒྱུད་ལམ་ཁག་ཏུ་བློ་ཕམ་བྱེད་མཁན་རེ་ཟུང་བྱུང་ཡོད་ནའང་། བོད་མི་མང་པོ་ཞིག་གིས་རྒྱལ་སྤྱིའི་ཨོ་རྩེད་སྟེང་རྒྱལ་ཁའི་ཨང་རྟགས་ཐོབ་མིན་དེ་གཙོ་བོ་མིན་པར། བོད་པ་ལྟ་བུའི་མི་རིགས་གྲངས་ཉུང་དང་ད་ལྟའི་ཉམས་ཞན་གྱི་གནས་སྟངས་འོག་ཏུ། རྒྱལ་སྤྱིའི་ཨོ་རྩེད་འགྲན་ཚོགས་སྟེང་ཞུགས་ཐུབ་པ་ཙམ་གྱིས་རླབས་ཆེན་གྱི་རྒྱལ་ཁ་ཡིན་པ་མ་ཟད། བོད་མི་རིགས་ཀྱི་ལ་རྒྱ་དང་སྤོབས་པ་མཐོ་རུ་བཏེགས་ཡོད་པའི་བསྟོད་བསྔགས་ཞུ་བཞིན་འདུག སྟོབས་རྒྱལ་ལགས་ནི། ༡༩༩༤ ལོའི་ཟླ་ ༢ པའི་ཚེས་ ༡༦ ཉིན་ལྷ་ས་སྟག་རྩེ་རྫོང་དར་རྒྱས་ཤིང་དུ་སྐྱེས་ཤིང་། ༢༠༠༩ ལོ་ནས་བོད་རང་སྐྱོང་ལྗོངས་ལུས་རྩལ་སློབ་གྲྭའི་རྒྱུག་ཤར་སྦྱོང་བརྡར་ནང་ཞུགས་ཏེ་ ༢༠༡༡ ལོར་རྒྱལ་ཡོངས་གཞོན་ནུ་ལོ་ཆུང་གི་རྒྱུག་ཤར་འགྲན་བསྡུར་ཁྲོད་གྲུབ་འབྲས་ཆེན་པོ་བླངས་པ་དང་། ༢༠༡༤ ལོར་ལྷོ་ཀོ་རི་ཡར་ཚོགས་པའི་ཨེ་ཤ་ཡའི་རྩེད་འགྲན་སྟེང་རྨིད་ ༥༠༠༠ ཅན་གྱི་སྐྱེས་པའི་རྒྱང་རྒྱུག་འགྲན་བསྡུར་ཁྲོད་ཨང་ ༡༢ པ་དང་། ༢༠༡༥ ལོར་རྒྱ་ནག་ཨིའུ་ཧྭན་དུ་ཚོགས་པའི་ཨེ་ཤ་ཡའི་རྒྱང་རྒྱུག་འགྲན་བསྡུར་ཁྲོད་ཨང་ ༥ པ་བཅས་ཐོབ་སྟེ། ད་སྔ་ཁྱོན་རྒྱ་ནག་ནང་ཁུལ་དང་ཕྱི་ལོགས་བཅས་ནས་གསེར་གྱི་རྟགས་མ་ ༨ དང་། དངུལ་གྱི་རྟགས་མ་ ༢ ། ཟངས་ཀྱི་རྟགས་མ་ ༥ ཐོབ་ཡོད་འདུག རྒྱ་ནག་གི་གནས་ཚུལ་ནང་འཁོད་པ་ལྟར་ན། སྟོབས་རྒྱལ་ལགས་ནི་བོད་རང་སྐྱོང་ལྗོངས་ཁོངས་ནས་ ༡༩༨༨ ལོར་མདའ་འཕེན་ལུས་རྩལ་པ་རྡོ་རྗེ་ཆོས་དབྱངས་ལགས་ལྷོ་ཀོ་རི་ཡར་ཨོ་ལིམ་ཕིག་རྩེད་འགྲན་སྟེང་ཞུགས་པའི་རྗེས་ཀྱི་ཨོ་རྩེད་སྟེང་ཞུགས་མཁན་གཉིས་པ་དེ་ཆགས་ཡོད་འདུག
