རྒྱ་དཀར་ནག་གཉིས་སྤུན་ཟླ་རེད།

༧གོང་ས་མཆོག་གིས་རྒྱ་དཀར་ནག་གཉིས་མུ་མཐུད་ཁྱིམ་མཚེས་བྱས་ནས་འདུག་དགོས་པས་ད་ལྟའི་ས་མཚམས་རྩོད་རྙོག་གནས་སྟངས་དེ་ཛ་དྲག་པོ་རེད་བསམ་གྱི་མི་འདུག་ཅེས་གསུངས་འདུག དེ་རིང་རྒྱ་གར་ཧུན་རྡི་སི་ཐན་དུས་བབ་ཀྱིས་གནས་ཚུལ་སྤེལ་གསལ་ལྟར་ན། རྒྱ་དཀར་ནག་གཉིས་འབྲོག་ལམ་ཁུལ་དུ་ཕན་ཚུན་ས་མཚམས་རྩོད་རྙོག་གིས་གདོང་ཐུག་རྒྱག་གྲབས་བྱེད་བཞིན་པའི་གནས་སྟངས་སྐོར་ལ། ༧གོང་ས་མཆོག་གིས་གནས་སྟངས་དེ་ “ཧ་ཅང་ཛ་དྲག་པོ་ཞིག་མ་རེད།”ཅེས་གསུངས་ཡོད་འདུག ༧གོང་ས་ཏཱ་ལའི་བླ་མ་མཆོག་གིས་རྒྱ་དཀར་ནག་གཉིས་ཀྱིས་ཕན་ཚུན་ཚིག་རྩུབ་བརྗོད་རེས་བྱེད་བཞིན་ཡོད་ན་ཡང་། “རྒྱ་དཀར་ནག་གཉིས་སྤུན་ཟླ་ཡིན་” “Hindi Chini Bhai Bhai”ཞེས་པ་དེ་སྔར་བཞིན་མདུན་སྐྱོད་ཀྱི་ལམ་ཁ་ཁོ་ན་ཡིན་པ་རེད་ཅེས་གསུངས་ཡོད་འདུག ༧གོང་ས་མཆོག་གིས་ “རྒྱ་དཀར་ནག་གཉིས་མུ་མཐུད་ཁྱིམ་མཚེས་བྱས་ནས་འདུག་དགོས་པས། ངས་གནས་སྟངས་དེ་ཛ་དྲག་པོ་རེད་བསམ་གྱི་མི་འདུག”ཅེས་གསུངས་ཡོད་པ་དང་། ཁོང་གིས་མུ་མཐུད་ “རྒྱུད་ལམ་ཁག་གིས་གནས་སྟངས་དེ་རྙོག་ཟིང་བཟོ་ཡི་འདུག”ཅེས་གསུངས་ཡོད་འདུག དེ་རིང་གཟའ་ལྷག་པའི་ཉིན་༧གོང་ས་མཆོག་རྒྱ་གར་གྱི་(Editors Guild of India)ཞེས་པའི་བཀོད་སྒྲིག་གནང་བའི་སྐུ་ཞབས་(Rajendra Mathur)མཆོག་རྗེས་དྲན་ཚོགས་འདུར་ཆིབས་སྒྱུར་གནང་ཡོད་པ་དང་། སྐབས་དེར་ཁོང་གིས་ “༡༩༦༢ལོར་མཐའ་ན་རྒྱ་ནག་གི་དམག་རྒྱ་གར་གྱི་འབུམ་སྡེ་ལ་བར་དུ་སླེབས་ཡོད་ན་ཡང་། མཐའ་མ་དེར་ཕྱིར་འཐེན་བྱས་ཡོད་པ་རེད།”ཅེས་གསུངས་ཡོད་འདུག རྒྱ་དཀར་ནག་གཉིས་འབྲས་ལྗོངས་ཀྱི་འབྲོག་ལམ་ཁུལ་དུ་ཉིན་གྲངས་༥༠རིང་ལ་ཕན་ཚུན་གདོང་ཐུག་གིས་ས་མཚམས་བརྟན་སྲུང་བྱས་ནས་བསྡད་ཡོད་པ་མ་ཟད། རྒྱ་གར་དམག་གིས་རྒྱ་ནག་དམག་མི་ཚོར་ས་ཁུལ་དེར་གཞུང་ལམ་འཛུགས་སྐྲུན་བྱ་རྒྱུར་བཀག་འགོག་བྱེད་བཞིན་ཡོད་པ་རེད། རྒྱ་ནག་གིས་གཞུང་ལམ་འཛུགས་འཆར་གྱི་ས་ཁུལ་དེ་རྒྱ་ནག་གི་ས་ཆ་ཡིན་པས། འབྲོག་ལམ་མཐོ་སྒང་ཁུལ་གྱི་རྒྱ་གར་དམག་དཔུང་མྱུར་དུ་ཕྱིར་འཕེན་བྱ་དགོས་པ་བརྗོད་བཞིན་ཡོད། འབྲུག་གིས་འབྲོག་ལམ་ས་ཁུལ་ནི་འབྲུག་གི་མངའ་ཁོངས་ཡིན་པས་བདག་དབང་འབྲུག་ལ་ཡོད་སྐོར་བརྗོད་ཀྱི་ཡོད་ནའང་། རྒྱ་ནག་གཞུང་གིས་ས་ཁུལ་དེ་རྒྱ་ནག་ལ་དབང་བས་ཐིམ་ཕུ་ཡིས་པེ་ཅིང་མཉམ་དུ་བདག་དབང་རྩོད་གླེང་བྱེད་པའི་དབང་ཆ་མེད་སྐོར་བརྗོད་ཡོད་པ་རེད། ཉེ་སྔོན་རང་རྒྱ་གར་ཕྱི་སྲིད་བློན་ཆེན་(Sushma Swaraj)ཡིས་དང་པོ་ཕྱོགས་གཉིས་ཀྱི་དམག་དཔུང་ཕྱིར་འཐེན་བྱས་ཏེ། ས་མཚམས་རྩོད་རྙོག་གི་གནས་སྟངས་དེ་ཞི་བའི་ངང་ནས་སེལ་ཐུབ་རྒྱུར་རེ་བ་ཡོད་ཅེས་བརྗོད་ཡོད་པ་རེད། གསར་འགོད་པ། སངས་རྒྱས་དཀོན་མཆོག
