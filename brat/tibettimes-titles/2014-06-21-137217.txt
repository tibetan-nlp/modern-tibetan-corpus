ཕར་ཕྱིན་དབྱར་ཆོས་ཐེངས་གཉིས་པ།

མདོ་སྨད་ལྷོ་རྒྱུད་ཀིརྟིའི་དགོན་ཁག་གི་ཕར་ཕྱིན་དབྱར་ཆོས་ཆེན་མོ་ཐེངས་གཉིས་པ་ཟླ་འདིའི་ཚེས་ ༡༣ ཉིན་ནས་མགོ་ནས་བཟུང་རྔ་བ་ཀིརྟིའི་དགོན་དུ་འཚོགས་བཞིན་ཡོད་འདུག བཞུགས་སྒར་ཀིརྟིའི་བྱེས་པ་གྲྭ་གི་བོད་གནས་འབྲེལ་མཐུད་པ་ཀ་ཉག་ཚེ་རིང་དང་བློ་བཟང་ཡེ་ཤེས་གཉིས་འདི་གར་གནས་ཚུལ་བཏང་འབྱོར་ལྟར་ན། མདོ་སྨད་ལྷོ་རྒྱུད་ཀིརྟིའི་དགོན་ཁག་གི་ཕར་ཕྱིན་དབྱར་ཆོས་ཆེན་མོ་ཐེངས་གཉིས་པ་ཟླ་འདིའི་ཚེས་ ༡༣ ཉིན་ནས་མགོ་འཛུགས་ཀྱིས་རྔ་བ་ཀིརྟི་དགོན་དུ་འཚོགས་བཞིན་ཡོད་འདུག ཕར་ཕྱིན་དབྱར་ཆོས་ཆེན་མོ་ཞེས་པ་ནི་སྔ་མོ་ཡུལ་དབུས་ཀྱི་གདན་ས་གསང་ཕུར་ཕར་ཕྱིན་དབྱར་ཆོས་འཚོགས་སྐབས་སུ་སེ་འབྲས་དགའ་གསུམ་སོགས་ཀྱི་སློབ་གཉེར་བ་ཆེས་མང་བོ་རང་དབང་གིས་འདུ་འཛོམས་ཀྱིས་གཞུང་ཆེན་བཀའ་པོད་ལྔའི་ནང་ཚན་ཕར་ཕྱིན་ཐེག་པའི་གཞུང་ལུགས་རྒྱ་མཚོ་ལྟ་བུར་དམིགས་བསལ་ཐོས་བསམ་སློབ་གཉེར་གནང་རྒྱུའི་ལུགས་སྲོལ་ཡོད་པ་ཞིག་ཡིན་འདུག ད་རེབ་འཚོགས་བཞིན་པའི་ཕར་ཕྱིན་དབྱར་ཆོས་དེ་བཞིན་གསང་ཕུའི་ཕར་ཕྱིན་དབྱར་ཆོས་ཞེས་པའི་མིང་དང་སྦྲེལ་ནས་ཀིརྟིའི་དགོན་ཁག་ཁག་གི་ཐོས་བསམ་སློབ་གཉེར་ཡར་རྒྱས་ཀྱི་ཆེད་དམིགས་ནས་༸སྐྱབས་རྗེ་ཀིརྟི་རིན་པོ་ཆེ་མཆོག་གིས་ཐུགས་བཀོད་ལམ་སྟོན་བཞིན་གསར་འཛུགས་གནང་བ་ཞིག་ཡིན་ཞིང་། དེ་ཡང་ཀིརྟི་ཚང་གི་གདན་ས་ཆེ་ཁག་བཞིའི་ཕར་ཕྱིན་ལོ་རིམ་གཉིས་པ་ནས་ལྔ་བའི་བར་གྱི་འཛིན་གྲྭ་བཞིའི་སློབ་གཉེར་བ་གང་ཡོད་ལྷན་འཛོམས་ཀྱིས། གདན་ས་ཆེ་ཁག་བཞིར་རེས་མོས་བྱས་ནས་༸རྗེ་བདག་ཉིད་ཆེན་མོའི་གསུང་གི་དྲང་ངེས་ལེགས་བཤད་སྙིང་པོ་རྩོད་གཞུང་གི་གཙོ་བོར་བྱས་ནས་དེ་ལ་དམིགས་བཀར་གྱི་སློབ་གཉེར་གནང་དགོས་པའི་ཉི་ཤུའི་ཆོས་ཐོག་ཞིག་ཡིན། དེ་ལ་མཆོད་བརྗོད་འཛིན་གྲྭ། སྒྲོ་འདོགས་ངོས་འཛིན་གྱི་འཛིན་གྲྭ། དགག་བྱ་ངོས་འཛིན་གྱི་འཛིན་གྲྭ། རང་རྒྱུད་པའི་འཛིན་གྲྭ། བཅས་འཛིན་གྲྭ་བཞི་ཡོད་ལ། ཉིན་རེ་བཞིན་ཞོགས་ཆོས་དང་ཉིན་ཆོས་དགོང་མོའི་ཆོས་རྭ་བཅས་དང་། བདུན་རེའི་མཚམས་སུ་མཚན་ཕྱེད་དམ་བཅའ་རེ་དང་། ཆོས་ཐོག་གི་ཉིན་མཐའ་མའི་མཚན་མོར་དྲང་ངེས་མཚན་ཕུལ་དམ་བཅའ་ཆེན་མོ་གནང་དགོས་ཀྱི་ཡོད་འདུག ད་ལོ་ཀིརྟིའི་དགོན་ཁག་གི་ཚད་མའི་དགུན་ཆོས་གོ་སྒྲིག་ཚོགས་ཆུང་ནས་གོ་སྒྲིག་འོག སྟག་ཚང་ལྷ་མོ་ཀིརྟིའི་དགོན། རྔ་བ་ཀིརྟིའི་དགོན། ཚོ་བདུན་ཀིརྟིའི་དགོན། ཧོར་ཚང་ཀིརྟིའི་དགོན་བཅས་ནས་ཕེབས་པའི་སློབ་གཉེར་པ་བཞི་བརྒྱ་ལྷག་དང་། གཞན་ཡང་གདན་ས་ཁག་གི་ཁྲི་བ་དང་མཁན་པོ་དཔེ་ཁྲིད་དགེ་རྒན་བཅས་ཆེད་ཕེབས་ཀྱིས་རྔ་བ་ཀིརྟིའི་དགོན་དུ་འཚོགས་བཞིན་ཡོད་འདུག ཀིརྟིའི་དགོན་ཁག་གི་ཕར་ཕྱིན་དབྱར་ཆོས་ཐེངས་དང་པོ་ཕྱི་ལོ་ ༢༠༡༣ ལོར་གདན་ས་སྟག་ཚང་ལྷ་མོ་ཀིརྟིའི་དགོན་དུ་ཚོགས་ཡོད་འདུག གསར་འགོད་པ། དོན་གོ
