རྩོད་རྙོག་འདུམ་འགྲིག་བྱེད་མི་དགོས་ཟེར།

དེ་རིང་རྒྱ་ནག་ལྷོ་རྒྱུད་ནངས་མོའི་ཚགས་པར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། རྒྱལ་སྤྱིའི་རྩོད་རྙོག་འདུམ་འགྲིག་ཁང་གིས་རྒྱ་ནག་དང་ཧྥེ་ལེ་ཕེན་རྒྱལ་ཁབ་གཉིས་དབར་གྱི་རྒྱ་ནག་གི་ལྷོ་རྒྱུད་མཚོ་ཁུལ་རྩོད་གླེང་འདུམ་འགྲིག་བྱ་འཆར་ཡོད་པ་དེར་རྒྱ་ནག་གཞུང་གིས་མོས་མཐུན་བྱེད་ཀྱི་མེད་པ་མ་ཟད། ཁ་སང་རྒྱ་ནག་གི་ཕྱི་སྲིད་བློན་ཆེན་ཝང་ཡེས་ཁ་པར་སྐུད་ལམ་ནས་ཨ་རིའི་རྒྱལ་སྲིད་དྲུང་ཆེ་རྗོན་ཁི་རེ་ལ་ཝ་ཤིན་ཊོན་གཞུང་གིས་རྒྱ་ནག་གི་མངའ་ཁོངས་བདག་དབང་ལ་བརྩི་འཇོག་བྱེད་དགོས་ལུགས་བཤད་འདུག རྒྱ་ནག་གི་ཕྱི་སྲིད་བློན་ཆེན་ཝང་ཡེས་རྒྱལ་སྤྱིའི་རྩོད་གླེང་འདུམ་འགྲིག་ཁང་གིས་རྒྱ་ནག་དང་ཧྥེ་ལེ་ཕེན་རྒྱལ་ཁབ་གཉིས་དབར་གྱི་མཚོ་ཁུལ་རྩོད་གླེང་འདུམ་འགྲིག་བྱེད་འཆར་ཡོད་པ་དེ་ནི་དགོད་བྲོ་བའི་ཟློས་གར་ཞིག་ཡིན་ཞིང་། རྒྱ་ནག་གིས་དེ་ལ་ནམ་ཡང་ངོས་ལེན་མེད་སྟབས། ལས་རིམ་དེ་ངེས་པར་མཚམས་འཇོག་དགོས་པ་དང་། ཨ་རིའི་གཞུང་གིས་རྒྱ་ནག་གི་རྒྱལ་སྲུང་བདེ་འཇགས་དང་མངའ་ཁོངས་བདག་དབང་ལ་བསམ་ཤེས་དང་བརྩི་འཇོག་བྱེད་དགོས། རྒྱ་ནག་གིས་མཚོ་ཐོག་སྐྱེལ་འདྲེན་གྱི་ཁེ་དབང་དང་ཞི་བདེ། བརྟན་ལྷིང་བཅས་དང་འབྲེལ་བའི་རང་གི་མངའ་ཁོངས་སྲུང་སྐྱོབ་བྱེད་ཀྱི་ཡོད་སྟབས། རྒྱལ་སྤྱིའི་རྩོད་རྙོག་འདུམ་འགྲིག་ཁང་གིས་གནད་དོན་དེར་ཐེ་གཏོག་བྱེད་དབང་མེད་པ་དང་། ཨ་རྒྱ་གཉིས་ཀྱིས་ཕྱོགས་གཉིས་བར་གྱི་འབྲེལ་ལམ་རྒྱུན་འཁྱོངས་དང་འགལ་འདུ་ཞི་འཇགས་བཟོས་ཏེ་ཁེ་ལས་མཉམ་འབྲེལ་ལ་དོ་སྣང་བྱས་ན་ཡག་སྐོར་སོགས་བརྗོད་འདུག ཨ་རིའི་ཕྱི་སྲིད་ལྷན་ཁང་གི་མགྲིན་ཚབ་པ་ཕི་རེ་སི་ (Price) ཟེར་བ་ཞིག་གིས། ཝང་ཡེ་དང་རྗོན་ཁི་རེ་གཉིས་བར་བྱུང་བའི་འབྲེལ་མོལ་དེ་ནི་རྒྱལ་ཁབ་གཉིས་མཐུན་མོང་གི་ཁེ་ཕན་དང་འབྲེལ་བའི་རྩོད་རྙོག་སྐོར་ཡིན་ཚུལ་བརྗོད་འདུག གནས་ཚུལ་ལྟར་ན། ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༢ ཉིན་རྒྱལ་སྤྱིའི་རྩོད་རྙོག་འདུམ་འགྲིག་ཁང་གིས་རྒྱ་ནག་དང་ཧྥེ་ལེ་ཕེན་རྒྱལ་ཁབ་གཉིས་བར་གྱི་མཚོ་ཁུལ་བདག་དབང་རྩོད་གླེང་དེ་འདུམ་འགྲིག་བྱ་འཆར་ཡོད་པ་དང་། རྒྱ་ནག་གིས་དེ་ལ་ངོས་ལེན་བྱེད་ཀྱི་མེད་པ་དང་། པེ་ཅིན་གཞུང་གིས་རང་གི་མངའ་ཁོངས་སུ་གླེང་བཞིན་པའི་རྒྱ་ནག་གི་ལྷོ་རྒྱུད་མཚོ་ཁུལ་དེ་ནི་རྒྱལ་སྤྱིའི་མཚོ་ཐོག་ཚོང་ལམ་གཙོ་བོ་ཡིན་སྟབས། ཨ་རིའི་གཞུང་གིས་མཚོ་ཁུལ་དེའི་རང་དབང་གི་མཚོ་འགྲུལ་བདེ་འཇགས་ལ་རྟོགས་ཞིབ་དགོས་སྐོར་བརྗོད་ཡོད་ནའང་། ཁོ་ཚོས་རྒྱ་ནག་གིས་རྒྱལ་སྤྱིའི་རྩོད་རྙོག་འདུམ་འགྲིག་ཁང་གི་ཐག་གཅོད་ལ་ངོས་ལེན་བྱེད་མིན་ཐད་བློ་དོགས་བྱེད་བཞིན་ཡོད་པ་དང་། དེ་ཡང་ཕྱི་ལོ་ ༢༠༡༣ ལོར་རྒྱ་ནག་གཞུང་གིས་རྒྱ་ནག་གི་ཤར་རྒྱུད་མཚོ་ཁུལ་གྱི་བདག་དབང་རྩོད་གླེང་འདུམ་འགྲིག་ལ་ངོས་ལེན་མ་བྱས་པ་མ་ཟད། མཚོ་ཁུལ་དེའི་གླིང་ཕྲན་ཁག་ཅིག་འཛུགས་བསྐྲུན་དང་བདག་བཟུང་བྱས་ཡོད་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
