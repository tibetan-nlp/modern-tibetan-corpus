ངོ་རྒོལ་དང་དམ་དྲག་ཁྲོད་ཀྱི་འོས་བསྡུ།

ཁ་སང་གི་ཐེ་ལན་ཌིའི་རྒྱལ་ཡོངས་འོས་བསྡུ་ཆེན་མོ་དེ་ངོ་རྒོལ་དང་དམ་དྲག་ཁྲོད་ནས་མཇུག་བསྒྲིལ་ཡོད་ཀྱང་འོས་བསྡུའི་གྲུབ་འབྲས་གསལ་བསྒྲགས་བྱེད་སྟངས་སྐོར་ལ་ཆབ་སྲིད་པ་ཚོས་ཚོད་དཔག་སྣ་ཚོགས་བྱེད་བཞིན་ཡོད། ཐེ་ལན་ཌིའི་དམངས་གཙོ་ཚོགས་པ་དང་ཚོགས་པའི་རྒྱབ་སྐྱོར་བ་ཚོས་ངོ་རྒོལ་དང་འགོག་རྐྱེན་བཟོ་མིན་ལ་མ་ལྟོས་པར་ཐེ་ལན་ཌིའི་རྒྱལ་ཡོངས་འོས་བསྡུ་ཆེན་མོ་དེ་ཁ་སང་ཚེས་ ༢ ཉིན་གྱི་སྔ་དྲོའི་ཆུ་ཚོད་ ༨ ནས་ཕྱི་དྲོའི་ཆུ་ཚོད་ ༣ པའི་བར། ཐེ་ལན་ཌིའི་མངའ་སྡེ་ ༧༧ དང་། འོས་འདེམ་ས་ཁུལ་ ༣༧༥ ཙམ་དུ་ཉིན་བརྒྱ་དུས་གཅིག་ལ་འོས་བསྡུ་གནང་ཡོད་ཅིང་། ཐེ་ལན་ཌིའི་རྒྱལ་ཡོངས་སུ་མི་འབོར་ས་ཡ་ ༦༤ ཙམ་ཡོད་པ་ལས་ཐེངས་འདིའི་འོས་བསྡུའི་ཆ་ཤས་ལེན་མཁན་མི་མང་ས་ཡ་ ༤༩ བྱུང་ཡོད་འདུག ཐེངས་འདིའི་འོས་བསྡུ་ལམ་ལྷོང་ཡོང་ཆེད་སྲིད་གཞུང་ནས་རྒྱལ་ས་བྷེང་ཁོག་གི་འོས་བསྡུའི་ས་ཚིགས་ ༦༦༧༡ དང་། རྒྱལ་ཡོངས་འོས་བསྡུའི་ས་ཚིགས་ཁག་ཏུ་ཕྱོགས་འགལ་ཚོགས་པའི་ངོ་རྒོལ་བ་ཚོས་འགོག་རྐྱེན་མི་བཟོ་བའི་ཆེད་ཉེན་རྟོག་པ་ ༡༣༠༠༠༠ ཙམ་འགྲེམ་འཇོག་གིས་ལྟ་སྲུང་བྱེད་བཅུག་ཡོད་སྟབས། རྒྱལ་ཡོངས་འོས་བསྡུ་དེ་བརྒྱ་ཆ་ ༨༩ ཙམ་ཞིག་ལེགས་པོ་བྱུང་ཡོད་ཀྱང་། ཕྱོགས་འགལ་ཚོགས་པས་ངོ་རྒོལ་ཤུགས་ཆེན་བྱས་པར་བརྟེན་འོས་བསྡུའི་ས་ཚིགས་ ༥༠༠ ལྷག་གི་མི་མང་ས་ཡ་ ༦ ཙམ་གྱིས་འོས་བསྡུ་བྱེད་ཐུབ་མི་འདུག ཐེ་ལན་ཌིའི་ད་ལྟའི་སྲིད་བློན་ལྕམ་ཡིང་ལུག་ལགས་ཀྱིས་གཞུང་འཛིན་སྐྱོང་བྱེད་སྟངས་ཐད་མི་མང་པོ་ཞིག་འདོད་པ་མ་ཁེངས་པར། སྔར་ལོའི་ཕྱི་ཟླ་ ༡༡ པ་ཙམ་ནས་གཞུང་ལ་ངོ་རྒོལ་བྱེད་མགོ་ཚུགས་ཡོད་པ་དང་། མི་མང་ཁྲོད་ཕྱོགས་ཁག་གཉིས་སུ་གྱེས་ཏེ་རྒྱལ་ས་བྷེང་ཁོག་ཏུ་གཞུང་ལ་ངོ་རྒོལ་བསྟུད་མུར་བྱས་ཡོད་ཅིང་། ཤོག་ཁག་གཉིས་བར་གདོང་ཐུག་བྱུང་སྟེ་ད་སྔ་མི་བཅུ་གྲངས་ཤིག་འདས་གྲོངས་དང་མི་ ༦༠༠ ལྷག་ལ་རྨས་སྐྱོན་ཕོག་ཡོད་པ་མ་ཟད། ཉེ་ལམ་ཟླ་ ༡ ཚེས་ ༢༧ ཉིན་ངོ་རྒོལ་ཚོགས་པའི་མགོ་ཁྲིད་སུ་ཐིན་ (Suthin Tharatin) ལགས་གཞུང་འཛིན་ཚོགས་པའི་རྒྱབ་སྐྱོར་བ་ཚོས་མེ་མདའ་བརྒྱབ་སྟེ་བསད་ཡོད། ཐེ་ལན་ཌིའི་ཕྱོགས་འགལ་ཚོགས་པའི་མགོ་ཁྲིད་ཟུར་པ་དང་། ཐེངས་འདིའི་རྔམ་སྟོན་ཁྲོམ་སྐོར་སྣེ་ཁྲིད་བྱེད་མཁན་སོ་ཐེབ་ཀྱིས་བརྗོད་པར། ད་རེབ་ཐེ་ལན་ཌིའི་རྒྱལ་ཡོངས་ནང་འོས་བསྡུའི་ས་ཚིགས་མང་པོ་ཞིག་སྒོ་རྒྱག་དགོས་བྱུང་ཡོད་སྟབས། སྲིད་གཞུང་གིས་སྤྱི་དངུལ་ཆུད་ཟོས་སུ་བཏང་བ་ཙམ་ལས། ཐེངས་འདིའི་འོས་བསྡུའི་གྲུབ་འབྲས་གསལ་བསྒྲགས་བྱ་རྒྱུ་དཀའ་ལས་ཁག་པོ་ཡིན་སྐོར་བརྗོད་འདུག དེ་བཞིན། ཆབ་སྲིད་དབྱེ་ཞིབ་པ་ཚོས། ད་རེབ་ཐེ་ལན་ཌིའི་རྒྱལ་ཡོངས་འོས་བསྡུ་ཆེན་མོའི་ཁྲོད་ད་ལྟའི་གཞུང་འཛིན་ཚོགས་པར་རྒྱལ་ཁ་ཐོབ་སྲིད་ཀྱང་། ཕྱོགས་འགལ་ཚོགས་པས་མངའ་སྡེ་ཁག་ཅིག་དང་བྷེང་ཁོག་གི་འོས་བསྡུའི་ས་ཚིགས་མི་ཉུང་བ་ཞིག་ནང་འོས་ལྡུག་བཅུག་མེད་སྟབས། མ་འོངས་པར་ཁྲིམས་མཐུན་རང་བཞིན་ཐོག་གཞུང་འཛིན་སྐྱོང་བྱ་རྒྱུ་དཀའ་ལས་ཁག་པོ་ཆགས་ཡོད་སྐོར་བརྗོད་འདུག ཐེ་ལན་ཌིའི་དཔལ་འབྱོར་རིག་པའི་ཆེད་མཁས་པ་ (Kanittha Tambunlertchai) ལགས་ཀྱིས། ཐེ་ལན་ཌི་ནི་ཡར་རྒྱས་འགྲོ་བཞིན་པའི་རྒྱལ་ཁབ་ཅིག་ཡིན་སྟབས། རྒྱལ་ཁབ་ནང་ཟིང་འཁྲུག་དང་ངོ་རྒོལ་བྱས་ཏེ་ཡུལ་བདེ་ཞི་འཇགས་མི་ཡོང་བ་དེས་རྒྱལ་ཁབ་ཀྱི་དཔལ་འབྱོར་འཕེལ་རྒྱས་ཐོག་འགོག་རྐྱེན་བཟོས་པ་ཞིག་ཆགས་ཡོད་ཅིང་། ཆབ་སྲིད་པ་ཚོས་རང་ཉིད་ཀྱི་ཁེ་དབང་རྩོད་ལེད་བྱ་རྒྱུ་བཞག་སྟེ་མང་ཚོགས་ཀྱི་མངོན་འདོད་བསྒྲུབ་དགོས་སྐོར་དང་། གཞན། བྷེང་ཁཱོག་གཙུག་ལག་སློབ་གྲྭ་ཆེན་མོའི་ཆབ་སྲིད་ཚན་རིག་ཐོག་གི་སློབ་དཔོན་ཆེན་མོ (Thitinan Pongsudhirak) ལགས་ཀྱིས། ཐེ་ལན་ཌིའི་ནང་འཕྲད་བཞིན་པའི་ཆབ་སྲིད་ལས་འགུལ་དེ་ཕྱོགས་འགལ་ཚོགས་པ་དང་གཞུང་འཛིན་ཚོགས་པ་གཉིས་ཀྱི་གྲོས་མོལ་བརྒྱུད་སེལ་ཐབས་མ་བྱས་ཚེ། མ་འོངས་པར་ཐེ་ལན་ཌིའི་གཞུང་དེ་དམག་མིའི་སྟངས་འཛིན་འོག་འགྲོ་རྒྱུ་ཉག་ཅིག་ཡིན་ལུགས་སོགས་བརྗོད་འདུག
