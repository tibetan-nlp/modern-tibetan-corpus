ཨེ་ཊ་ལའི་ས་ཡོམ་ཁྲོད་ཉུང་མཐར་མི་ ༢༤༧
རྐྱེན་འདས་སུ་ཕྱིན།

ཨེ་ཊ་ལིའི་བྱང་ཤར་ཁུལ་གྱི་ས་ཡོམ་ཁྲོད་ད་ལྟའི་བར་ཉུང་མཐར་མི་གྲངས་ ༢༤༧ རྐྱེན་འདས་དང་མི་གྲངས་སྟོང་ཕྲག་བརྒལ་བར་རྨས་སྐྱོན་ཕོག་ཡོད་འདུག ཁ་སང་ཨེ་ཊ་ལིའི་རྒྱལ་ས་རོམ་གྱི་བྱང་ཤར་ཁུལ་དུ་བྱུང་བའི་ཚད་རིམ་ ༦།༢ ཅན་གྱི་ས་ཡོམ་ཁྲོད། ཁང་པ་དང་གཞུང་ལམ་སོགས་འཛུགས་སྐྲུན་ལ་གནོད་སྐྱོན་ཚབས་ཆེན་བཟོས་ཡོད་པ་མ་ཟད། གནོད་སྐྱོན་ཐེབས་ཡུལ་ཚབས་ཆེ་ཤོས་རི་ཊི་(Rieti)ཞིང་ཆེན་ནང་མི་གྲངས་ ༡༩༠ དང་ཨེ་སི་ཀོ་ལི་ཕེ་སི་ནོ་(Ascoli Piceno )ཞིང་ཆེན་ནང་མི་གྲངས་ ༥༧ བཅས། ད་ལྟའི་བར་ཉུང་མཐར་མི་གྲངས་ ༢༤༧ རྐྱེན་འདས་སུ་ཕྱིན་ཡོད་པ་དང་། མི་སྟོང་ཕྲག་བརྒལ་བར་རྨས་སྐྱོན་ཕོག་པ་དང་། གྱང་རོའི་འོག་ཏུ་མནན་ཏེ་མྱུར་སྐྱོབ་འཚོལ་ཞིབ་བྱེད་བཞིན་པའི་སྒང་ཡིན་འདུག མི་མང་བདེ་འཇགས་ལས་ཁུངས་ཀྱིས་རི་ཉུལ་བ་དང་བྲག་འཐོར་བ་སོགས་ཀྱིས་ད་ལྟའང་རི་ཁུལ་གྱི་གྲོང་སྡེ་ཁག་གི་ནང་ཛ་དྲག་མྱུར་སྐྱོབ་བྱེད་པར་དཀའ་ངལ་འཕྲད་ཀྱི་ཡོད་པ་དང་། མི་རྐྱེན་འདས་ཀྱི་གྲངས་འབོར་མུ་མཐུད་དུ་འཕར་རྒྱུ་ཡིན་ལུགས་བརྗོད་འདུ་ལ། གནོད་ཐེབས་ཡུལ་མི་ཞིག་གིས་ཁོ་རང་སྡོད་སའི་གྲོང་སྡེ་དེའི་ཁང་པ་བརྒྱ་ཆ་ ༩༥ འཐོར་ཞིག་ཏུ་གྱུར་ཡོད་སྐོར་བརྗོད་ཡོད་པ་རེད། ཨེ་ཊ་ལི་ནང་ཕྱི་ལོ་ ༢༠༠༩ ལོར་ཚད་རིམ་ ༥།༩ ཅན་གྱི་ས་ཡོམ་ཞིག་བརྒྱབ་སྟེ་མི་གྲངས་ ༣༠༠ ལྷག་རྐྱེན་འདས་སུ་ཕྱིན་ཡོད་པ་དང་། དེ་ནི་ཉེ་བའི་ལོའི་ནང་ས་ཡོམ་གནོད་སྐྱོན་ཚབས་ཆེ་ཤོས་སུ་རྩི་བཞིན་ཡོད་ཀྱང་། གནད་དོན་དཔྱད་ཞིབ་པ་ཚོས་ཐེངས་འདིའི་གནོད་སྐྱོན་དེ་ཚབས་ཆེ་ཤོས་ཆགས་རྒྱུ་ཡིན་སྐོར་རྗོད་ཀྱི་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
