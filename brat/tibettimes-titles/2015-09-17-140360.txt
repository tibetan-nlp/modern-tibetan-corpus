རུས་ཚད་ན་ཚ་ཁྱབ་བརྡལ་དུ་ཕྱིན།

  རྒྱ་གར་གྱི་རྒྱལ་ས་ལྡི་ལིར་རུས་ཚད་ན་ཚ་(Dengue)འགོས་ཁྱབ་ཆེན་པོ་འགྲོ་བཞིན་ཡོད་པ་ལྟར་བོད་མི་ ༧༥ ལྷག་ལ་ན་ཚ་དེ་ཕོག་ནས་སྨན་བཅོས་བྱས་དང་བྱེད་བཞིན་ཡོད་འདུག   ཉེ་སྔོན་ནས་རྒྱ་གར་གྱི་རྒྱལ་ས་ལྡི་ལིའི་གཙོས་གྲོང་ཁྱེར་ཁག་ཅིག་ཏུ་རུས་ཚད་ན་ཚ་འགོས་ཁྱབ་ཆེན་པོ་ཕྱིན་ཏེ། ནད་པ་མང་དྲགས་པའི་རྐྱེན་གྱིས་སྨན་ཁང་ཁག་ཏུ་ཉལ་ཁྲི་སོགས་ཀྱིས་འདང་གི་མེད་པའི་གནས་ཚུལ་ཐོན་གྱི་ཡོད་པ་ལྟར། དེ་རིང་འདི་གའི་གསར་འགོད་པས་ཁ་པར་བརྒྱུད་ལྡི་ལི་བསམ་ཡས་གླིང་གི་ས་གནས་གཞིས་མགོ་ལེགས་སྐྱེས་རྡོ་རྗེ་ལགས་སུ། བོད་མིའི་ནང་ན་ཚ་དེ་ཕོག་མཁན་ག་ཚོད་ཡོད་པ་དང་། ཁོང་ཚོའི་གནས་སྟངས་ག་འདྲ་ཡོད་པ། དེ་བཞིན་སྨན་བཅོས་དང་སྔོན་འགོག་གི་མཐུན་རྐྱེན་ག་འདྲ་བསྐྲུན་གྱི་ཡོད་མེད་སྐད་ཆ་དྲིས་པར་ཁོང་གིས། འདི་ཟླའི་ནང་བསམ་ཡས་གླིང་གཞིས་གྲོང་གི་བོད་མི་ ༧༥ ལྷག་ལ་རུས་ཚད་ན་ཚ་ཕོག་ཡོད་ནའང་། དུས་ཐོག་ཏུ་སྨན་ཁང་ལ་བསྟན་པ་དང་སྨན་བཅོས་ལེགས་པོ་བྱས་སྟབས་ཅུང་ཡག་པོ་དྲག་འདུག དེ་མིན་ད་སྔར་རུས་ཚད་ན་ཚ་ཚབས་ཆེན་པོ་དེ་འདྲ་བྱུང་མ་སོང་། ང་ཚོས་ཉེ་ལམ་མངའ་སྡེ་གཞུང་ལ་སྙན་སེང་ཞུས་པ་བརྒྱུད་ནང་དགོང་གཉིས་ལ་སྡོད་ཁང་གི་མཐའ་འཁོར་དུ་སྨན་རྫས་དང་ས་སྣུམ་གྱི་དུད་རླངས་གཏོང་བ་དང་། སྡོད་ཁང་ནང་གི་གྲང་ཆས་དང་ཁོར་ཡུག་གཙང་སྦྲ་བཟོ་བ་བཅས་ལ་བརྟེན་ནས་དུག་སྦྲང་གི་རུས་ཚད་ན་ཚ་སྔོན་འགོག་བྱེད་ཀྱི་ཡོད། ཅེས་བརྗོད་བྱུང་།   གསར་ལམ་ཁག་ཏུ་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། རྒྱ་གར་གྱི་རྒྱལ་ས་ལྡི་ལིའི་ནང་ད་ལྟའི་བར་ཉུང་མཐར་མི་གྲངས་ ༡༩༠༠ ལྷག་ལ་ན་ཚ་དེ་འགོས་ནས་གཞུང་འབྲེལ་དང་གཞུང་འབྲེལ་མ་ཡིན་པའི་སྨན་ཁང་ཁག་ནང་སྡོད་དགོས་བྱུང་ཡོད་པ་མ་ཟད། ཉུང་མཐར་མི་གྲངས་ ༡༤ ལྷག་རྐྱེན་འདས་སུ་ཕྱིན་ཡོད་འདུག   རྒྱ་གར་གཞུང་འབྲེལ་སྨན་ཁང་ཁག་ཡོངས་ནས་རུས་ཚད་ན་ཚ་ཕོག་པའི་ནད་པ་མང་པོར་བརྟག་དཔྱད་སྨན་བཅོས་བྱེད་ཀྱི་ཡོད་ནའང་། ཉིན་འདི་དག་རིང་རུས་ཚད་ན་ཚ་ཁྱབ་བརྡལ་ཆེན་པོ་ཕྱིན་པའི་རྐྱེན་གྱིས། སྨན་ཁང་ཁག་ཏུ་ནད་པ་བཞི་ལ་ཉལ་ཁྲི་གཅིག་ལས་ཐེམས་ཀྱི་མེད་པའི་དཀའ་ངལ་འཕྲད་ཀྱི་ཡོད་སྟབས། རྒྱ་གར་དབུས་འཕྲོད་བསྟེན་ལས་ཁུངས་ནས་སྒེར་བཙུགས་སྨན་ཁང་ཁག་ལ་ཉལ་ཁྲི་དང་བརྟག་དཔྱད་ཡོ་ཆས། སྨན་པ་དང་སྨན་ཞབས་སོགས་ཀྱི་རོགས་རམ་ཡོང་བའི་འབོད་སྐུལ་ཞུས་ཡོད་པ་བརྟེན། གཞུང་འབྲེལ་སྨན་ཁང་ཁག་ཏུ་ཉལ་ཁྲི་ ༡༠༠༠ ལྷག་ཇེ་མང་དུ་བཏང་ཡོད་པ་མ་ཟད། རུས་ཚད་ན་ཚར་བརྟག་དཔྱད་བྱེད་པར་ཧིན་སྒོར་ ༦༠༠ ཡན་ལེན་མི་ཆོག་པ་དང་། ནད་པའི་སྨན་རིན་དང་མྱུར་མཁོའི་ཡོ་ཆས་སོགས་གཞུང་ངོས་ནས་གཏོང་རྒྱུ། དེ་བཞིན་སྒེར་བཙུགས་སྨན་ཁང་ཁག་ནས་རུས་ཚད་ན་ཚ་ཅན་གྱི་ནད་པ་རྣམས་སྨན་ཁང་ནང་འདུག་ཏུ་བཅུག་སྟེ། བརྟག་དཔྱད་སྨན་བཅོས་ལེགས་པོ་བྱེད་དགོས་པ་ལས། ཉལ་ཁྲི་དང་སྨན་བཅོས་ཀྱི་དཀའ་ངལ་འོག་ནད་པ་རྣམས་སྨན་ཁང་ནས་ཕྱིར་ཕུད་བཏང་བ་ཡིན་ན། སྨན་པའི་ཐོབ་ཐང་ཕྱིར་བསྡུ་གནང་སྲིད་པའི་ཉེན་བརྡ་ཡང་བཏང་ཡོད་པ་རེད།   བདུན་ཕྲག་སྔོན་མའི་ནང་ལྡི་ལིར་གནས་སྡོད་ལོ་ན་ ༦ ནས་ ༧ བར་གྱི་རུས་ཚད་ན་ཚ་ཕོག་པའི་བྱིས་པ་གཉིས་ལ་སྒེར་བཙུགས་སྨན་ཁང་ཁག་གིས་བརྟག་དཔྱད་མ་བྱས་པར་ཕྱིར་ཕུད་བཏང་ཡོད་པ་དང་། ཕྱིས་སུ་བྱིས་པ་དེ་གཉིས་འདས་གྲོངས་སུ་གྱུར་ཡོད་པར་བརྟེན་བྱིས་པའི་ཁྱིམ་མི་དང་སྨན་ཁང་བར་རྙོག་གླེང་ཆེན་པོ་བྱུང་ཡོད་པ་དང་། དེ་བཞིན་རུས་ཚད་ན་ཚ་ཅན་གྱི་ནད་པ་རེ་ཟུང་ཨ་ཆད་འུ་ཐུག་གིས་རང་སྲོག་གཅོད་ཀྱི་ཡོད་པའི་ཛ་དྲག་གི་གནས་ཚུལ་དཀའ་ངལ་བྱུང་བཞིན་ཡོད་འདུག   སྤྱིར་བཏང་རྒྱ་གར་ནང་རུས་ཚད་ན་ཚ་འདི་ཆར་དུས་རྗེས་ཀྱི་ཕྱི་ཟླ་ ༡༠ ཡས་མས་ནང་བྱུང་གི་ཡོད་པ་དང་། རུས་ཚད་ན་ཚ་བྱུང་སྐབས་ལུས་ཡོངས་སུ་ཚ་བ་རྒྱས་པ་དང་མགོ་ན་བ། ཤ་རྒྱུས་དང་རུས་ཚིགས་གཟེར་བ། ཚ་འབྲུམ་འབུད་པ་དང་མགོ་ཡོམ་འཁོར་བ། སྣ་ཁྲག་བཞུར་བ་སོགས་ཀྱི་ནད་རྟགས་ཐོན་གྱི་ཡོད་པ་དང་། འགོས་ནད་དེ་སྔོན་འགོག་བྱེད་པར་ཉེ་འགྲམ་གྱི་ཁོར་ཡུག་དང་རང་ལུས་ལ་གཙང་སྦྲ་བྱེད་དགོས་པ་མ་ཟད། ཕྱི་ལོགས་སུ་འགྲོ་སྐབས་གྱོན་གོས་རིང་ཙམ་གྱོན་ཏེ་ལུས་སྨད་ཀྱི་ཤ་མི་མངོན་པ་བྱས་ན་ཕན་ཐོགས་ཡོད་པ་དང་། དེ་བཞིན་ཆུ་མང་པོ་འཐུང་བ་དང་འབུ་མི་འཁོར་བའི་བྱུག་རྫས་རིགས་བྱུག་ནའང་ཕན་ཐོགས་ཡོད་འདུག   གསར་འགོད་པ། པདྨ་རྒྱལ།
