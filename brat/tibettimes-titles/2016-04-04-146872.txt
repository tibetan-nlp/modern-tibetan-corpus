འགན་འཁྲི་དོན་འཁྱོལ་མ་བྱུང་བའི་མི་སྣ་ ༡༥༧
ལ་ཐག་གཅོད་བྱས།

འགན་འཁྲི་གཉིས་དོན་འཁྱོལ་མ་བྱུང་བའི་ཉེས་མིང་འོག་མཚོ་སྔོན་ཞིང་ཆེན་གྱི་མགོ་ཁྲིད་ལས་བྱེད་མི་སྣ་ ༡༥༧ ལ་འདྲི་གཅོད་ཐག་གཅོད་བྱས་ཡོད་འདུག དེ་རིང་རྒྱ་ནག་བོད་ལྗོངས་དྲ་བའི་ནང་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། འདས་ལོར་མཚོ་སྔོན་ཞིང་ཆེན་གྱིས་འགན་འཁྲི་གཉིས་དོན་འཁྱོལ་མ་བྱུང་བའི་མགོ་ཁྲིད་ལས་བྱེད་མི་སྣ་ ༡༥༧ ལ་འདྲི་གཅོད་ནན་མོ་བྱས་ཏེ་ཐག་གཅོད་བྱས་ཡོད་པ་མ་ཟད། མཚོ་སྔོན་ཞིང་ཆེན་སྲིད་གཞུང་དང་ཞིང་ཆེན་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ཁང་གཉིས་ཀྱིས་༼ཏང་སྲོལ་སྲིད་གཙང་འཛུགས་སྐྱོང་འགན་འཁྲི་རྩད་གཅོད་ཐད་ཀྱི་གནས་སྐབས་ལག་བསྟར་བྱེད་ཐབས༽དང་། ༼ཁྲིམས་བཤེར་ཞིབ་དཔྱོད་ལས་ཁུངས་ཀྱིས་གྱོད་གཅིག་ཟུང་བཤེར་གྱི་ཚོད་ལྟའི་བྱེད་ཐབས༽གཉིས་གཏན་ལ་ཕབ་སྟེ། ཞིབ་བཤེར་ཚན་ཆུང་ཆེད་འཛུགས་ཀྱིས་ཞིང་ཆེན་ཐད་གཏོགས་ལས་ཁུངས་དང་སྡེ་ཁག་གི་མགོ་ཁྲིད་ཚན་ཆུང་ ༦༣ ལ་ཞིབ་བཤེར་བྱས་རྗེས། རྗེས་ལུས་ཅན་གྱི་ལས་ཁུངས་ ༩ ལ་དུས་ཆད་གླེང་མོལ་བྱ་རྒྱུ་ཡིན་འདུག གནས་ཚུལ་བསྒྲགས་པའི་ནང་། གཏན་འབེབས་བྱས་པའི་བྱེད་ཐབས་དེ་གཉིས་ཀྱིས་འགན་འཁྲི་གཙོ་བོའི་ཐད་ཀྱི་གནས་ཚུལ་ ༡༤ དང་། ལྟ་སྐུལ་འགན་འཁྲིའི་གནས་ཚུལ་ ༡༢ དང་། གྱོད་གཅིག་ཟུང་བཤེར་གྱི་གནས་ཚུལ་ ༡༠ བཅས་ལ་རྩད་གཅོད་གསལ་བོ་བྱེད་པར་ཐག་བཅད་ཅིང་། ཐུན་མོང་གི་འགན་འཁྲི་དང་རང་སྒེར་གྱི་འགན་འཁྲི། མགོ་ཁྲིད་གཙོ་བོའི་འགན་འཁྲི་དང་མགོ་ཁྲིད་གཞོན་པའི་འགན་འཁྲི་བཅས་གསལ་བོར་ཕྱེ་ཡོད་པ་མ་ཟད། ཉེས་འཛོལ་དང་ཤུགས་རྐྱེན་ཆེ་ཆུང་ལ་གཞིགས་ཏེ་སྐྱོན་བརྗོད་བྱེད་པ་དང་། བསླབ་འདོམས་སྐུལ་ལྕག་གཏོང་བ། རྩ་འཛུགས་ལེགས་སྒྲིག་དང་གོ་གནས་དབྱུང་བ། ཏང་སྲིད་སྒྲིག་ཁྲིམས་ཀྱི་ཆད་པ་གཅོད་པ་སོགས་ཀྱི་བྱེད་ཐབས་བེད་སྤྱད་དེ། འགན་འཁྲི་རྩད་གཅོད་དངོས་སུ་ལག་བསྟར་བྱེད་རྒྱུ་རེད། ཅེས་བཀོད་འདུག ཡང་གནས་ཚུལ་དེའི་ནང་བཀོད་པ་ལྟར་ན། ༼༢༠༡༦ ལོའི་མགོ་ལོག་ཁུལ་གྱི་རིམ་པ་དང་པོའི་ཏང་སྲོལ་སྲིད་གཙང་འཛུགས་སྐྱོང་སྐོར་ཞིབ་བྱ་བ་ལག་བསྟར་བྱེད་ལུགས༽ལྟར་མགོ་ལོག་ཁུལ་གྱིས་སྐོར་ཞིབ་ཚན་ཆུང་གཉིས་བཙུགས་ཏེ། རྨ་སྟོད་རྫོང་དང་དགའ་བདེ་རྫོང་གཉིས་སུ་སྐོར་ཞིབ་བྱ་བ་མགོ་བརྩམས་ཡོད་པ་དང་། སྐོར་ཞིབ་བྱ་བའི་གཙོ་གནད་དེ་ཏང་གི་སྒྲིག་སྲོལ་དྲུག་བརྩི་སྲུང་བྱས་ཡོད་མེད་སྐོར་ཡིན་ལ། ཐེངས་འདིའི་སྐོར་ཞིབ་བྱ་བ་ཕྱི་ཟླ་ ༣ པའི་ཟླ་དཀྱིལ་ནས་ཟླ་ ༥ པའི་ཟླ་དཀྱིལ་དུ་མཇུག་རྫོགས་འཆར་ཡོད་འདུག གཞན་ཡང་། མཚོ་ནུབ་ཁུལ་སྲིད་གཞུང་ནས༼ཏང་སྲོལ་སྲིད་གཙང་འཛུགས་སྐྱོང་དང་རུལ་སུངས་ལ་ངོ་རྒོལ་བྱེད་པའི་ལས་དོན་སྐོར་གྱི་བརྡ་སྦྱོར༽དངོས་སུ་སྤེལ་ཏེ། ཏ་ཐུང་རྫོང་དང་གྲོ་ཚང་ས་ཁུལ་དུ་རྣམ་པ་བཞིའི་སྒོ་ནས་ཏང་ནན་གྱིས་སྐྱོང་བའི་ཚོད་ལྟའི་བྱ་བ་བསྒྲུབ་རྒྱུ་དང་། གཞི་རིམ་ཁག་ཏུ་འགན་འཁྲི་གཙོ་བོ་དོན་འཁྱོལ་ཡོང་བའི་ཐབས་ལམ་ཕན་འབྲས་ལྡན་པ་ཞིག་འཚོལ་རྒྱུ་ཡིན་སྐོར་བསྒྲགས་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
