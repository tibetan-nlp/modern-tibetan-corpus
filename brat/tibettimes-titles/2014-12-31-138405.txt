ཡུ་གུར་སྤྱི་ཁྱབ་ཀྱི་གོ་གནས་སྤར།

དེ་རིང་རྒྱ་ནག་གཞུང་གིས་ཤིན་ཅང་ཡུ་གུར་རང་སྐྱོང་ཁུལ་གྱི་སྤྱི་ཁྱབ་ནུར་བྷེ་ཀེ་རེ་རྒྱ་ནག་དབུས་ནུས་ཤུགས་ལས་དོན་ལྷན་ཁང་གི་འགན་འཛིན་དུ་གནས་རིམ་འཕར་ཡོད་འདུག དེ་རིང་རྒྱ་ནག་གི་གཞུང་འབྲེལ་གསར་ལམ་ཁག་དང་རོའུ་ཌར་དྲ་རྒྱ་ཐོག་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། རྒྱ་ནག་གཞུང་གིས་ཤིན་ཅང་ཡུ་གུར་རང་སྐྱོང་ཁུལ་གྱི་སྤྱི་ཁྱབ་ནུར་བྷེ་ཀེ་རེ་ (Nur Bekri)ཡི་ལས་གནས་སྤར་ཏེ། རྒྱ་ནག་དབུས་ནུས་ཁུངས་ལས་དོན་ལྷན་ཁང་གི་འགན་འཛིན་དང་། རྒྱལ་ཡོངས་འཕེལ་རྒྱས་དང་བཅོས་བསྒྱུར་ལྷན་ཁང་གི་ཚོགས་གཞོན་ལ་བསྐོ་བཞག་བྱས་ཡོད་ཀྱང་། ཁོང་གི་ཚབ་ཏུ་ཡུ་གུར་རང་སྐྱོང་ཁུལ་གྱི་སྤྱི་ཁྱབ་སུ་ཞིག་བསྐོ་རྒྱུ་ཡིན་མིན་ཕྱིར་བསྒྲགས་བྱས་མེད་སྐོར་བཀོད་འདུག འདིའི་ཐད་ཤིན་ཅང་དམར་ཤོག་ཚོགས་པའི་ཚོགས་གཙོ་ཀྲང་ (Zhang Chunxian) གིས་བརྗོད་དོན། “ནུར་བྷེ་ཀེ་རེ་ནི་ཡུ་གུར་མི་རིགས་ཁྲོད་ཀྱི་ཕུལ་བྱུང་ལས་བྱེད་ཅིག་ཡིན་པ་མ་ཟད། མགོ་ཁྲིད་ཅིག་གི་ཤེས་འཇོན་གཉིས་ལྡན་གྱི་མི་སྣ་ཞིག་ཀྱང་ཡིན། དེ་སྔོན་ཁོང་གིས་ཤིན་ཅང་གི་བརྟན་ལྷིང་སྲུང་སྐྱོབ་ལ་འགན་འཁུར་ནས་ཡུལ་བདེ་ཞི་འཇགས་བཟོས་ཡོད་སྟབས། ད་ལྟ་ཤིན་ཅང་སྔར་ལྷག་འཕེལ་རྒྱས་དང་བརྟན་ལྷིང་ལྡན་པ་ཞིག་ཆགས་ཡོད”ཅེས་བརྗོད་འདུག ཁ་སང་ཤིན་ཅང་གི་ཉིན་རེའི་ཚགས་ཤོག་གིས་གནས་ཚུལ་སྤེལ་བ་ཞིག་གི་ནང་། “ནུར་བྷེ་ཀེ་རེ་ཡི་ཤིན་ཅང་གི་དམར་ཤོག་ཚོགས་པའི་ཚོགས་གཞོན་གྱི་གོ་གནས་དེ། ཤིན་ཅང་གྱི་རྒྱལ་ས་ཨོ་རམ་ཁེའུ་ཡི་སྤྱི་ཁྱབ་ཟུར་བ་ཛཱ་ཀཱེར་(Shohrat Zakir)ལ་སྤྲོད་སྲིད་པ་ཞིག་རེད”ཅེས་བརྗོད་འདུག ༢༠༠༩ ལོའི་ཟླ་ ༧ ཚེས་༥ ཉིན་ཤིང་ཅང་གི་རྒྱལ་ས་ཝུ་རུ་མུ་ཆི་ནང་། ཡུ་གུར་བ་བརྒྱ་ལྷག་གིས་ཞི་བའི་རྔམ་སྟོན་ཁྲོམ་སྐོར་གྱིས་རྒྱ་གཞུང་ལ་ངོ་རྒོལ་བྱས་ཡོད་སྟབས། རྒྱ་གཞུང་གི་ཉེན་རྟོག་པ་དང་དམག་མི་འབོར་ཆེན་ཡོང་སྟེ། གང་བྱུང་མེ་མདའ་རྒྱག་པ་དང་དུག་རླངས་གཏོར་བ་སོགས་དྲག་ཆས་བེད་སྤྱོད་བཏང་ཡོད་སྟབས། ཕྱོགས་སོ་སོ་ནས་ཡུ་གུར་བ་ཆིག་ཁྲིར་ཉེ་བ་འདུ་འཛོམས་ཀྱིས་གཞུང་གི་ལས་ཁུངས་དང་རླངས་འཁོར། ཚོང་ཁང་དང་ཟ་ཁང་། སྤྱི་དམངས་ཀྱི་སྡོད་ཤག་བཅས་ལ་གཏོར་བཅོམ་དང་མེར་བསྲེགས་བཏང་ཡོད། སྐབས་དེར་རྒྱ་ནག་གི་དམག་མིས་གང་བྱུང་མེ་མདའ་བརྒྱབ་སྟེ་མི་ ༦༠༠ ལྷག་འདས་གྲོངས་སུ་ཕྱིན་ཡོད་པ་དང་། རྨས་སྐྱོན་ཕོག་པ་དང་བོར་བརླག་ཏུ་སོང་བ་ ༢༠༠༠ ཙམ་བྱུང་ཡོད། ཡུ་གུར་མི་རིགས་ཀྱིས་རྒྱ་གཞུང་ལ་ངོ་རྒོལ་མཚམས་མ་ཆད་པར་བྱས་པར་བརྟེན། འདས་པའི་ལོ་ཤས་རིང་ད་ལྟའི་ཡུ་གུར་རང་སྐྱོང་ཁུལ་གྱི་སྤྱི་ཁྱབ་ནུར་བྷེ་ཀེ་རེ་ཡིས་བདེ་འཇགས་བརྟན་བརླིང་ཆེད་ལས་ཀ་འདྲ་མིན་བྱས་ཡོད་པ་དང་། ཤིན་ཅང་ནི་རྒྱ་གར་དང་ཨབ་གྷན། པ་ཀི་སི་ཐན་དང་འདབས་འབྲེལ་ཆགས་ཡོད་ཅིང་། རྡོ་སྣུམ་དང་སོལ་རླངས། རྡོ་སོལ། ས་གཏེར་གྱིས་ཕྱུག་པའི་ཡུལ་གྲུ་ཞིག་ཡིན་སྟབས། རྒྱ་ནག་གཞུང་གིས་ནུར་བྷེ་ཀེ་རེ་རྒྱ་ནག་དབུས་ནུས་ཁུངས་ལས་དོན་ལྷན་ཁང་གི་འགན་འཛིན་ལ་བསྐོ་བཞག་བྱས་པ་ཞིག་ཡིན་འདུག རྒྱ་ནག་ཞི་མན་གཙུག་ལག་སློབ་ཆེན་གྱི་ནུས་ཁུངས་ཉམས་ཞིབ་ཁང་གི་འགན་འཛིན་ལིན་བྷོ་ (Lin Boqiang) ཡིས། ཞིན་ཅང་ནི་རྒྱ་ནག་རྒྱལ་ནང་གི་ནུས་ཁུངས་དང་རྡོ་སྣུམ་འབོར་ཆེན་ཡོད་པའི་ས་ཁུལ་ཞིག་ཡིན་སྟབས། ཁོང་ལས་གནས་དེར་བསྐོ་བཞག་བྱས་པ་དེ་ནི་འོས་ཤིང་འཚམ་པ་ཞིག་དང་དུས་ལ་བབ་པ་ཞིག་རེད་ཅེས་བརྗོད་འདུག ཆབ་སྲིད་དབྱེ་ཞིབ་པ་ཚོས། རྒྱ་ནག་གཞུང་གི་དཔོན་རིགས་ཆེ་གྲས་མང་ཆེ་བ་ཧན་མི་རིགས་ཆགས་ཡོད་ཀྱང་། ད་རེབ་ནུར་བྷེ་ཀེ་རེ་ལས་གནས་དེར་བསྐོ་བཞག་བྱས་པ་དེ་ནི། རྒྱ་གཞུང་གིས་ཡུ་གུར་གྱི་ཐོན་ཁུངས་རྩ་མེད་བཟོ་ཆེད་ཡིན་པ་ལས་གྲངས་ཉུང་མི་རིགས་ལ་ཆེ་མཐོང་བྱས་པ་ཞིག་མིན་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
