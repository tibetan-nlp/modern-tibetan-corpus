ཆ་འཕྲིན་དུས་རབས་ཀྱི་བོད་སྐྱེད་སྲིང་།

བཀྲས་མཐོང་ལྷ་སྒྲོན་ལགས་ནི་བོད་དོན་འཐབ་རྩོད་ལས་འགུལ་སྣེ་ཁྲིད་པ་གཙོ་གྲས་ཤིག་རེད། ཁོ་མོས་ ༡༩༩༨ ལོའི་བོད་ཀྱི་རང་དབང་རོལ་དབྱངས་འདུ་ཚོགས་ཐོག་ཏུ་གནང་བའི་གཏམ་བཤད་དེས་བོད་དོན་རྒྱབ་སྐྱོར་བའི་མི་རབས་གསར་བར་སེམས་འགུལ་ཐེབས་སུ་བཅུག་རྗེས། ཐོག་མར་བོད་དོན་གྱི་མགྲིན་ཚབ་པ་ཞིག་ཆགས་པ་རེད། ལྷ་སྒྲོན་ལགས་ཀྱིས་བོད་རང་བཙན་སློབ་ཕྲུག་ཚོགས་པའི་འགན་འཛིན་གྱི་འགན་ཁུར་ཏེ། ༢༠༠༨ ལོའི་ཨོ་ལིམ་ཕིག་རྩེད་འགྲན་གྱི་སྔོན་དང་། རྩེད་འགྲན་སྐབས་སུ་རྒྱ་ནག་གིས་བོད་ས་དབང་བསྒྱུར་བྱས་པར་ངོ་རྒོལ་གྱི་ལས་འགུལ་ཚད་མཐོན་པོ་སྤེལ་ཡོད། ༢༠༡༡ ལོར་རྒྱལ་སྤྱིའི་ཞི་བའི་འདུམ་འགྲིག་ལྟེ་གནས་ཁང་གིས་ཁོ་མོར་རྗམ་སི་ལོ་སོན་འཚེ་མེད་ཞི་བའི་གྲུབ་འབྲས་ཀྱི་ལོ་འཁོར་གཟེངས་རྟགས་དང་པོ་གནང་ཡོད། ཁོ་མོ་ནི་ཁ་ན་ཌར་སྐྱེས་བའི་བོད་པ་ཞིག་རེད་ལ། ཉེ་བའི་དུས་སུ། བོད་དོན་དང་འབྲེལ་བའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་དང་རང་དབང་གི་སྟོབས་ཤུགས་ཇེ་ཆེར་གཏོང་ཆེད་བརྡ་འཕྲིན་གྱི་མ་ལག་ལ་བསྟེན་ནས་ཐབས་བྱུས་ལྡན་པའི་འཚེ་མེད་ཞི་བའི་ལས་འགུལ་སྤེལ་ཆེད་བོད་དོན་ལས་འགུལ་བསྟི་གནས་ཁང་གསར་འཛུགས་གནང་། བཅར་འདྲི་འདིའི་ནང་དུ་རྭིག་རྣམ་ཁང་དབང་དང་མཉམ་དུ་ཁོ་མོའི་ཕྱག་ལས་དང་། རྒྱ་ནག་གི་དྲ་ལམ་ཞིབ་བཤེར། བོད་ནང་གི་དྲ་ལམ་རང་དབང་བཅས་ཀྱི་སྐོར་གླེང་ཡོད། དྲི་བ། ཉེ་ཆར་རང་དབང་ཁང་པས་བཏོན་པའི་ ༢༠༡༥ ལོའི་དྲ་ལམ་རང་དབང་གཞིར་བཞག་ན། རྒྱ་ནག་ནི་ལོ་འདིའི་དྲ་ལམ་རང་དབང་རྩིས་མེད་རྡོག་རོལ་གཏོང་མཁན་སྡུག་ཤོས་ཡིན་པ་བཀོད་འདུག བོད་ཀྱི་གནས་བབ་ཇི་འདྲ་རེད།     ལན། བོད་ནང་གི་དྲ་ལམ་རང་དབང་གི་ཐད་ནས། ང་ཚོས་དེ་ནི་སྐྱོ་ཤོས་ནང་གི་སྐྱོ་ཤོས་ལ་ངོས་འཛིན་བྱ་དགོས། རྒྱ་མི་ཞིག་གིས་བསམ་ཚུལ་བརྗོད་པའི་ནུས་པ་དེ་བོད་མི་ཞིག་ལ་བསྡུར་བ་ཡིན་ན། ནམ་རྒྱུན་བོད་མི་རྣམས་ཀྱིས་བློས་གཏོང་བྱེད་དགོས་པ་དང། བོད་མིའི་ཐོག་ལ་ཞིབ་བཤེར་བྱེད་པའི་ཚད་དེ་མཐོན་པོ་རེད། རྒྱ་གཞུང་གིས་བོད་མི་རྣམས་ཀྱིས་གཞུང་གི་དབང་ཐང་ལ་ཁ་གཏོད་གཅོག་མི་ཡོང་ངམ་སྙམ་པའི་དོགས་པ་ཆེན་པོ་བྱེད་ཀྱི་ཡོད་པ་རེད། དྲི་བ། ཞིབ་བཤེར་གྱི་སྤྱིའི་གནས་བབ་གང་འདྲ་འདུག   ལན། བོད་ནང་གི་བོད་མི་ཚོས་ཞིབ་བཤེར་གྱི་སྐོར་ཤེས་ཀྱི་འདུག དྲ་ལམ་དང་དེ་མིན་གཞན་ཅི་ཡིན་ཡང་། རྒྱ་ནག་གིས་བོད་རང་སྐྱོང་ལྗོངས་ཞེས་འབོད་པའི་དབུས་གཙང་ཁུལ་ཏུ། བོད་མི་ཚོ་ནི་བློ་ངོར་མི་ཤོང་བའི་ཚད་བཀག་ཅན་དང་ཁྱབ་ཆེ་བའི་ཞིབ་བཤེར་གྱི་འོག་ཏུ་གནས་ཡོད། བོད་རང་སྐྱོང་ལྗོངས་ཕྱི་རོལ་གྱི་རྒྱ་ནག་གི་ཞིང་ཆེན་མཚོ་སྔོན་དང་། སི་ཁྲོན། ཡུན་ནན་དང་ཀན་སུ་བཅས་བོད་ཀྱི་ལོ་རྒྱུས་ཐོག་གི་ས་ཁུལ་མང་ཆེ་བར་ཉེ་བའི་ལོ་ཤས་ཀྱི་རིང་ལ་ཞིབ་བཤེར་བྱེད་ཤུགས་ཆེ་རུ་བཏང་སོང་། ཁྱེད་ཀྱིས་དྲ་ལམ་དུ་མཆན་ཞིག་བཏབ་པ་ཙམ་གྱིས་འཇུ་ཡི་ཡོད་པ་ཙམ་མ་རེད། མི་མང་རྣམས་ལམ་མདོ་ནས་འགོག་གི་རེད་ལ། ལྷ་སར་བོད་མི་རྣམས་ཀྱི་ལག་ནས་ཁ་པར་འཕྲོག་ནས་ནང་དོན་སྔོག་གི་རེད། ཉིས་སྟོང་བརྒྱད་ལོའི་རྗེས་ཏེ། བོད་ཀྱི་མཐོ་སྒང་ཡོངས་རྫོགས་ཡར་ལངས་ཀྱིས་དཀྲོགས་སྐབས། རྒྱ་གཞུང་གིས་ཡར་ལངས་རྒྱབ་ཀྱི་ཇུས་འཐེན་པ་རེད་སྙམ་པའི་མི་རྣམས་ལུས་བཤེར་དང་། དྲ་ལམ་ནས་སྟངས་འཛིན་སོགས་རྩ་བ་བསྔོག་པའི་ལས་འགུལ་རྒྱ་ཆེ་རུ་བཏང་བ་རེད། ཁོང་ཚོས་དེ་ནི་བོད་མི་ཚོས་རྒྱའི་དབང་བསྒྱུར་འོག་གི་ལྟ་རྟོག་བྱེད་སྟངས་ལ་ཁོང་ཁྲོ་དང་སུན་སྣང་སྐྱེས་བ་ཕྱི་རུ་བཏོན་པའི་རྣམ་འགྱུར་ངོ་མ་དང་། དེ་ཡང་རང་འགུལ་གྱིས་བྱུང་བ་ཞིག་ཡིན་པ་ཁས་ལེན་བྱེད་ཐུབ་ཀྱི་ཡོད་པ་མ་རེད། དྲི་བ། དྲ་ལམ་གྱི་གནས་ཚུལ་རྣམས་གཙང་ཕྱག་དང་། བཀག་འགོག ཚགས་རྒྱག་པ་བཅས་ཀྱིས་བོད་ཕྱི་ནང་གི་དྲ་བའི་རང་དབང་ལ་ཤུགས་རྐྱེན་ཇི་འདྲ་ཐེབས་ཀྱི་འདུག   ལན། མཐུན་རྐྱེན་ཚད་མེད་འཛོམས་པའི་རྒྱ་ནག་གཞུང་ལྟ་བུར་མཚོན་ན། དྲ་ཚིགས་དང་དྲ་བའི་ནང་དོན་རྣམས་མེད་པ་བཟོ་རྒྱུ་ནི་ཧ་ཅང་ལས་སླ་བོ་ཞིག་རེད། དྲ་མཆན་ནམ་དྲ་ཚིགས་གང་ཡིན་ཡང་། ཉིན་གཅིག་ཡོད་ན། ཉིན་གཉིས་པ་དང་། སྐར་མ་གཉིས་པར་གར་སོང་ཆ་མེད་རེད།   ཐོམ་སི་ཁེ་ཕི་(Tom-Skype)་ནི་བོད་དང་རྒྱ་ནག་ཏུ་བེད་སྤྱོད་གཏོང་བཞིན་བའི་སི་ཁེ་ཕི་ཡི་ངོ་བོ་དམིགས་བསལ་བ་ཞིག་རེད། ང་ཚོས་གསལ་བོར་ཤེས་པ་ཞིག་ལ། གནད་ཚིག་ངེས་ཅན་ཁག་ཅིག་ཚགས་རྒྱག་གི་འདུག ཁྱེད་ཀྱིས་བསྐུར་བའི་འཕྲིན་ཐུང་དེ་ཕ་རོལ་པོར་སླེབས་མི་ཐུབ་པར། ལམ་བར་ནས་ཐོམ་སི་ཁེ་ཕི་བཟོ་ཚོང་ཁང་རང་གིས་བཀག་འགོག་དང་ཞིབ་བཤེར་བྱེད་ཀྱི་ཡོད།   ཕལ་ཆེར་བོད་མི་ཡོད་དོ་ཅོག་གིས་སྐད་འཕྲིན་བེད་སྤྱོད་གཏོང་གི་འདུག གནས་སྟངས་ཛ་དྲག་ཡིན་པའི་སྐབས་སུ་སྐད་འཕྲིན་སྟེང་དུ་གནས་ཚུལ་སྣ་ཚོགས་བཀོད་པའི་རྐྱེན་གྱིས་མི་མང་འཛིན་བཟུང་བྱེད་ཀྱི་ཡོད་པ་ཤེས་ཀྱི་ཡོད། བོད་ཤར་ཕྱོགས་མཚོ་སྔོན་གྱི་བོད་མི་ཞིག་གིས་༧གོང་ས་མཆོག་གི་གྱ་སྟོན་དང་འབྲེལ་བའི་གནས་ཚུལ་སྤེལ་སྟབས། ཁོང་འཇུ་བཟུང་བྱས་ཏེ་གར་སོང་ཆ་མེད་དུ་གྱུར་བ་རེད། ཁོང་ད་ལྟ་གང་དུ་ཡོད་མིན་གྱི་གནས་ཚུལ་སུ་གང་གིས་ཤེས་ཀྱི་ཡོད་པ་མ་རེད། ཐ་ན། ཆབ་སྲིད་དང་འབྲེལ་བའི་གནད་དོན་མི་སྤེལ་མཁན་རྣམས་ཀྱང་གཞུང་གི་སྟངས་འཛིན་འོག་བཅུག་ཡོད་ལ། ཁོང་ཚོའི་འབྲེལ་ལམ་གྱི་བརྒྱུད་ལམ་ཧ་གོ་ཡི་ཡོད་པ་རེད། བོད་མི་ཡོངས་རྫོགས་ཀྱིས་སྐད་འཕྲིན་ལྟ་བུའི་འབྲེལ་མཐུད་དྲ་ལམ་གཅིག་རང་གཅིག་བེད་སྤྱོད་བཏང་ན། རྒྱ་གཞུང་གིས་ཞིབ་བཤེར་གཅིག་བསྡུས་དང་བརྒྱུད་ལམ་སྟངས་འཛིན་བྱེད་རྒྱུ་ཧ་ཅང་ལས་སླ་བོ་ཆགས་ཀྱི་ཡོད་པ་མ་རེད། དྲི་བ། མི་ཁ་ཤས་ཀྱིས་བཤད་རྒྱུར། བོད་མི་ཚོས་ཞེད་སྣང་བྱེད་རྒྱུ་མི་འདུག མི་ཚང་མས་སོ་སོས་རང་ཐག་རང་གཅོད་ཀྱིས་གནད་དོན་རྣམས་དྲ་ཐོག་ཏུ་འགོད་དགོས་ཀྱི་འདུག ཨ་མདོ་དཔེ་མཚོན་ཏུ་བཞག་ན། ཁོང་ཚོས་གནས་ཚུལ་འདྲ་མིན་བཀོད་པས་རྒྱ་གཞུང་གིས་ཀྱང་གནས་ཚུལ་གང་འཚམ་ཞིག་འགོད་འཇུག་གི་འདུག གང་ཡིན་ཞེ་ན། ཁོང་ཚོས་མི་ཐམས་ཅད་བཙོན་ཏུ་འཇུག་ཐུབ་ཀྱི་མ་རེད། ལན། བོད་མི་ཚོས་རང་གི་མངོན་འདོད་བཤད་དགོས་ཀྱི་འདུག་ལ། ཁོང་ཚོས་དེ་ག་རང་སྒྲུབ་ཀྱི་འདུག སྐད་ཆ་ཁ་རུ་ཁ་ཐུག་བཤད་མཁན་ཚོས་དཔེ་མཚོན་གྱི་ལམ་ནས་སྐད་ཆ་བཤད་མཁན་དག་ལས་ལྷག་པའི་ཉེན་ཁར་གདོང་ལེན་བྱེད་ཀྱི་འདུག མི་གཞན་མང་པོ་ཞིག་གིས་སྨྲ་སྒོ་བཙུམས་ནས་བསྡད་ཡོད་པ་རེད། ང་ཨ་རི་ལྟ་བུའི་རང་དབང་དང་བདེ་སྐྱིད་ཀྱི་སར་བསྡད་དེ། བོད་ནང་གི་བོད་མི་ཚོས་འདི་བྱ་དགོས་པ་དང་དེ་བྱ་མི་ཉན་པའི་སྐད་ཆ་བཤད་ཐུབ་ཀྱི་མི་འདུག ཕྱི་སྡོད་བོད་མི་ཚོས་ཅི་ཞིག་ལེགས་ཤོས་ཡིན་མིན་གྱི་ཚོད་དཔག་མི་བྱེད་པར། བོད་ནང་གི་བོད་མི་ཚོས་ཅི་ཞིག་བཤད་ཀྱི་ཡོད་མེད་དང་། ཅི་ཞིག་སྒྲུབ་གྱི་ཡོད་མེད་ལ་ཉན་ནས་རྒྱབ་སྐྱོར་བྱེད་དགོས། དྲི་བ། ང་ཚོས་ཇི་ལྟར་རང་བཤེར་མ་བྱས་པར་གནད་འགག་བརྩད་གཅོད་ཅི་འདྲ་བྱེད་དགོས། ལན། ངས་བལྟས་ན། སྐད་འཕྲིན་བེད་སྤྱོད་ཀྱི་གནད་དོན་ཐོག་ནས། གཙོ་བོ་བརྒྱུད་ལམ་འདིས་བྱ་བ་ཇི་ལྟར་སྒྲུབ་ཀྱི་ཡོད་མེད་དང་། དེ་བེད་སྤྱོད་བྱས་ན་འགོག་རྐྱེན་དང་ཉེན་ཁ་ཅི་ཞིག་ཡོད་མེད་རྟོགས་རྒྱུ་དེ་རེད། ས་གནས་ས་ཐོག་གི་གནས་ཚུལ་ཡོང་ཁུངས་དང་བོད་ཀྱི་སྤྱི་ཚོགས་སྲུང་སྐྱོབ་ཀྱི་ཆེད་དུ། བོད་མི་ཚོས་སྤྱང་པོ་དང་བདེ་འཇགས་ཀྱི་སྒང་ནས་བྱ་བ་སྒྲུབ་དགོས། དྲི་བ། ཁྱེད་ཀྱི་ཉམས་མྱོང་ཐོག་ནས་བཤད་ན། རྒྱ་ནག་གཞུང་གིས་བེད་སྤྱོད་གཏོང་བཞིན་པའི་ཞིབ་བཤེར་གྱི་མ་ལག་རྣམས་ཀྱི་རྒྱ་ཆེ་ཆུང་དང་གནས་ཚད་གང་འདྲ་རེད། ལན། ཁོང་ཚོས་དྲ་ཐོག་སུམ་འཛུལ་གྱི་བྱ་བ་མང་པོ་ཞིག་དང་། བོད་མི་ཚོའི་དྲ་ལམ་ལ་སིམ་འཛུལ་བྱེད་སྟངས་ནི་ཚད་མཐོན་པོ་དེ་འདྲ་མ་རེད། བོད་མི་ཚོས་རྗེས་འདེད་གཏོང་མཁན་གྱི་རྒྱ་ནག་གི་དྲ་ལམ་སིམ་འཛུལ་བ་ཚོ་ནི་རང་ག་བ་ཞིག་རེད། མཚམས་རེར་བོད་མི་ཚོས་མཉེན་ཆས་གསར་སྣོན་བྱེད་ཀྱི་མེད་སྟབས། གསར་སྣོན་མ་བྱས་པའི་མཉེན་ཆས་ཤིག་བེད་སྤྱོད་བྱེད་ཀྱི་ཡོད་པ་རེད། གལ་ཏེ། ཁོང་ཚོས་གསར་སྣོན་མ་བྱས་པའི་མཉེན་ཆས་ཤིག་བེད་སྤྱོད་བྱེད་ཀྱི་ཡོད་ཚེ། དྲ་ལམ་གང་སར་བཏང་བའི་དྲ་འབུ་ཚོས་སིམ་འཛུལ་བྱེད་རྒྱུ་ལས་སླ་བ་ཡོད།   ང་ཚོས་རྒྱ་ནག་གི་དྲ་ལམ་སིམ་འཛུལ་བ་ཚོ་ནི་གནས་ཚད་མཐོན་པོ་ཡིན་སྙམ་གྱི་ཡོད་ཀྱང་། ཐ་ན། ཨ་རིའི་གཞུང་གི་ཆེས་མཐོའི་གནས་རིམ་ཏུ་སིམ་འཛུལ་བྱེད་སྐབས་སུའང་གཞུང་ཞབས་ཁག་ཅིག་ལ་ཁོང་ཚོའི་ཨ་མ་ཡིན་མདོག་གིས་འཕྲིན་ཐུང་གཏོང་གི་ཡོད། ཁོང་ཚོ་མགོ་སྐོར་ཐེབས་ཏེ་འཕྲིན་ཐུང་དེའི་ཁ་ཕྱེས་ཚེ་རྒྱ་མི་ཚོ་རང་གི་དྲ་ལམ་དུ་ཡོང་འཇུག་གི་ཡོད། འདི་ནི་མིས་བཟོས་བའི་ནོར་འཁྲུལ་ཞིག་ཡིན་སྟབས། གཞི་རྩའི་སློབ་སྦྱོང་དང་གོ་རྟོགས་ལ་བརྟེན་ནས་ནོར་འཁྲུལ་སེལ་ཐུབ་ཀྱི་ཡོད། ང་ཚོར་སོ་སོ་སྟངས་འཛིན་བྱེད་པའི་ནུས་པ་ནི་ང་ཚོས་བསམ་པ་ལས་ལྷག་ཅིག་ཡོད་སྟབས། དེ་ནི་དོན་དག་ལེགས་པོ་ཞིག་རེད།   ངས་བལྟས་ན། གནས་སྟངས་འདིར་ཀུན་འདུས་ཀྱི་ཐབས་ལམ་ཞིག་དགོས་ཀྱི་འདུག མི་དམངས་ལ་ཞེད་སྣང་སློང་བའམ་སྐྱེད་སྲིང་མ་བྱས་ན་ཐོབ་རྒྱུ་ཅི་ཡང་མེད། བོད་མི་ཚོ་ལས་སླ་བོར་ཞེད་སྣང་སྐྱེ་ཡི་མེད་པ་ནི་ཡག་པོ་དང་། དེ་ནི་ང་ཚོའི་འཐབ་རྩོད་ཀྱི་རེ་བ་ཆེ་ཤོས་དང་སེམས་ཤུགས་སློང་བའི་གཞི་རྩའི་དོན་དངོས་རེད། དྲི་བ། ད་བར་ཁྱེད་ཀྱིས་གནད་དོན་འདིའི་ཐད་ནས་བོད་མི་ཚོར་སློབ་སྦྱོང་ཇི་ལྟར་སྤྲོད་ཀྱི་ཡོད་མེད་དང་། སློབ་སྦྱོང་སྤྲོད་བྱེད་ཀྱི་ཐབས་ལམ་ཡག་ཤོས་དེ་གང་ཞིག་ཡིན་ནམ། དང་པོ། མི་དམངས་ལ་གོ་རྟོགས་དང་སློབ་སྦྱོང་སྤྲོད་པའི་ལས་འགུལ་དེ་གཞི་རྒྱ་ཆེ་བའི་ཐོག་ནས་གཙིགས་ཆེར་འཛིན་པ་དང་། དེ་ནི་མི་དམངས་ཀྱི་དྲ་ལམ་འཕྲོད་བསྟེན་གོ་རྟོགས་ལས་འགུལ་ཞིག་ལ་ངོས་འཛིན་བྱེད་ཀྱི་ཡོད། གལ་ཏེ། མི་དམངས་རྣམས་དྲ་ཐོག་ཏུ་འཕྲོད་བསྟེན་ལྡན་པའི་ཐོག་ནས་གསོན་གནས་བྱེད་པའི་ཐབས་ལམ་ཞིག་ཡོད་ན། དེ་ནི་ང་ཚོའི་ནུས་པ་འདོན་ས་དེ་ཡིན། མི་མང་ཁྲོད་དུ་དྲ་ལམ་འཕྲོད་བསྟེན་ལག་ལེན་བྱེད་པའི་གཞི་རྒྱ་ཆེ་བའི་གོ་རྟོགས་སྤེལ་དགོས་ཀྱི་འདུག གཞན་པ་ཞིག་ནི་དམིགས་ཚད་ཅན་གྱི་སྦྱོང་བརྡར་རེད། ང་ཚོས་ཁ་སང་དགོང་མོ་བོད་ནང་དུ་ངོ་རྒོལ་བྱས་པའི་གནས་ཚུལ་དེ་དེ་རིང་ཉིན་གུང་ཧ་གོ་ཐུབ་ཀྱི་ཡོད་པ་རེད་ལ། ངོ་རྒོལ་དེའི་བརྙན་ཐུང་ངམ་པར་མཐོང་ཐུབ་ཀྱི་རེད། འདི་ནི་ངོ་མཚར་ཅན་གྱི་ཡར་རྒྱས་ཤིག་རེད། ང་ཚོས་བོད་ཀྱི་ལས་འགུལ་བ་དང་། བོད་ནས་གནས་ཚུལ་རག་ཐབས་ལ་འབད་པ་བྱེད་མཁན། ཡང་ན། བོད་ནང་གི་བོད་མི་ཚོ་དང་སྐད་ཆ་བཤད་མཁན་ཚོར་དམིགས་ཚད་ལྡན་པའི་སྦྱོང་བརྡར་སྤྲད་དེ་ཆ་འཕྲིན་འཕྲོད་བསྟེན་གྱི་ལག་ལེན་ཡག་ཤོས་ཤིག་བསླབ་ཐུབ་དགོས། དྲི་བ། ལེན་ཁེ་སྒྲེན་རྒྱ་ནང་དུ་རྒྱ་བསྐྱེད་བྱེད་ཐབས་བྱེད་ཀྱི་འདུག གཞུང་གི་དབང་འཛིན་པ་ཚོས་གནོན་ཤུགས་འོག་རྒྱ་ནག་ནང་གི་བེད་སྤྱོད་བྱེད་མཁན་ཚོའི་ས་ནས་ཆབ་སྲིད་དང་འབྲེལ་བའི་ཚོར་བ་སྐྱེན་པའི་གནད་དོན་ཁག་ལག་ལེན་ཐོག་ནས་བཀག་འགོག་བྱེད་ཀྱི་འདུག གནད་དོན་འདི་ཅིའི་ཕྱིར་འབྱུང་གི་ཡོད་པ་རེད། ལན། ངས་བསམ་ཚུལ་ལ། འདི་ནི་ངོ་ཚ་ཆེན་པོ་ཞིག་རེད་ཅེས་བརྗོད་རྒྱུ་ལས་གཞན་མི་འདུག ལེན་ཁེ་སྒྲིན་དང་ངོ་དེབ། གྷོ་གྷིལ་ཡང་ན་ཌི་ཝེ་ཌིལ་ཡིན་ཡང་རུང་། ཁོང་ཚོར་རྒྱ་ནག་གཞུང་ཡང་དག་པའི་ཁ་ཕྱོགས་ཤིག་ཏུ་ཁྲིད་པའི་གོ་སྐབས་དང་ཤུགས་རྐྱེན་ཆེན་པོ་ཡོད་ཀྱང་། དེའི་ཚབ་ཏུ། ལམ་ཁ་གཞན་པ་བདམས་པ་རེད། མི་དམངས་རྣམས་རྒྱ་ནག་གི་ཁྲོམ་རའི་ནང་དུ་འཛུལ་རྒྱུར་འདོད་པ་ཆེན་པོ་འདུག ང་ཚོས་འབྱུང་འགྱུར་གྱི་ལོ་ཟླར་ཆ་འཕྲིན་བཟོ་གྲྭ་ཁག་ཅིག་གིས་རྒྱ་གཞུང་གིས་རེ་བ་ཅི་བཏོན་བསྒྲུབ་རྒྱུ་ཡིན་པ་མཐོང་གི་རེད། ཆ་འཕྲིན་བཟོ་ཚོང་ཁང་འདི་དག་ཆེན་པོ་ཆགས་པའི་རྒྱུ་མཚན་ནི། རང་དབང་དང་གུ་ཡངས་ལྡན་པའི་དྲ་ལམ་ནས་ཚོང་ལས་བྲིན་པའི་རྐྱེན་གྱིས་རེད། ད་ཆ་རྒྱ་ནག་གཞུང་ལ་རོགས་རམ་གྱིས་རྒྱ་ནག་གི་མི་སེར་དང་བོད་མི་ཚོའི་དྲ་ལམ་འགོག་པ་ནི་ཕུགས་རྒྱང་རིང་པོར་བལྟ་མ་ཐུབ་པའི་རྟགས་རེད། དྲི་བ། ཁྱེད་ཀྱིས་ནུབ་ཕྱོགས་རྒྱལ་ཁབ་ལྟར་ཁྲིམས་ལུགས་ཀྱི་འགྱུར་བ་ཐེབས་ཐུབ་པའི་ནུས་པ་གཅིག་མཚུངས་མེད་པའི་རྒྱ་ནག་དང་བོད་ནང་གི་དྲ་ལམ་རང་དབང་དང་ཞིབ་བཤེར་གྱི་ཐད་ལ་འགྱུར་བ་ཇི་ལྟར་གཏོང་ཐུབ་ཀྱི་འདུག ལན། ཐོག་མར་རང་དབང་དང་མང་གཙོའི་སྤྱི་ཚོགས་སུ་འཚོ་གནས་བྱེད་མཁན་ཚོས་མིག་དཔེ་སྟོན་བཞིན་སྣེ་ཁྲིད་ཐུབ། དབྱིན་ཇི་རྒྱལ་ཁབ་ལྟར། དྲ་ལམ་གསང་སྔོག་གི་མ་ལག་དག་ནི་འཇིགས་སྐུལ་རིང་ལུགས་པ་ཁོ་ནས་བེད་སྤྱོད་བྱེད་རྒྱུ་ཞིག་ངོས་འཛིན་བྱས་ན། དེས་འཛམ་གླིང་ཡོངས་ཀྱི་རང་དབང་ལ་དོ་ཕོག་བཟོ་གི་རེད། མ་ལག་འདི་ཚོས་ཇི་ལྟར་བྱ་བ་སྒྲུབ་ཀྱི་ཡོད་མེད་སྐོར་མི་དམངས་ལ་སློབ་སྦྱོང་སྤྲད་དེ། ཆ་འཕྲིན་གྱི་རྨོངས་པ་སེལ་བ་དང་། དྲ་ལམ་གྱི་བདེ་འཇགས་ཧ་གོ་རྒྱུ་ནི་ང་ཚོའི་གཞུང་གི་ཁེ་ཕན་རེད། མི་དམངས་ཀྱིས་མ་ལག་འདི་དང་ལག་རྩལ་དེ་དག་ཤེས་རྟོགས་བྱུང་ན། གསང་སྔོག་མ་ལག་དག་འགོག་པར་ལྟ་སྟངས་ཁྱད་མཚར་འཛིན་མི་སྲིད། ཁོང་ཚོས་མི་དམངས་ལ་བདེ་འཇགས་ཀྱི་དྲ་ལམ་ཞིག་དགོས་པའི་གལ་ཆེན་རང་བཞིན་དེ་རྟོགས་ཀྱི་རེད།   དེ་ནས་གཅིག་སྡུད་ལམ་ལུགས་ཀྱི་སྲིད་གཞུང་ལ་མི་དམངས་དཀྱུས་མ་ཚོར་ཞིབ་བཤེར་བྱེད་རོགས་བྱེད་མཁན་རྒྱལ་སྤྱིའི་སྒ་མ་(International Gamma)་དང་དྲ་ལམ་གཏོར་རྒོལ་རུ་ཁག་ལྟ་བུའི་ཨ་རི་དང་། དབྱིན་ཡུལ། ཁ་ན་ཌའི་བཟོ་ཚོང་རྣམས་ལ་འགན་ཡོད་པ་ཤེས་སུ་འཇུག་དགོས། བཟོ་ཚོང་ཁང་འདི་དག་ནི་ཁྲིམས་འགལ་ཡིན་པ་དང་། སྟངས་འཛིན་བྱེད་དགོས་པ། ཉེས་ཆད་གཅོད་དགོས་པ། ངོས་འཛིན་མེད་པ་བཟོ་དགོས། བྱ་སྤྱོད་འདི་རིགས་ལ་བཟོད་བསྲན་བྱེད་མི་ཉན་ཡང་། སྲིད་དོན་གྱི་ཐོག་ནས་ལག་པ་འཆང་ཐུབ་མིན་ནི་དྲི་བ་གཞན་པ་ཞིག་རེད། མཐའ་མཇུག མི་གཅིག་ཡིན་ཡང་རྒྱ་ནག་དང་འབྲེལ་ལམ་མི་འཛུགས་རྒྱུ་དེ་རེད། ཐོབ་ཐང་དང་རང་དབང་གི་ཆེད་དུ་འཐབ་རྩོད་བྱེད་མཁན་བོད་ནང་གི་བོད་མི་དང་། རྒྱ་ནག་གི་མི་སེར་རྣམས་ལ་ཕན་ཐོགས་པའི་མ་ལག་མང་པོ་ཞིག་མ་ལག་སྒམ་ཆུང་ནང་དུ་ཡོད་པ་རེད། ང་ཚོའི་གཞུང་རྣམས་ཀྱང་རྒྱ་ནག་དང་མཉམ་དུ་ཡོངས་གྲགས་ཀྱི་སྒོ་ནས་ཐད་ཀར་འབྲེལ་ལམ་འཛིན་རྒྱུ་དེ་ཕྱིར་འཐེན་བྱེད་དགོས། དྲི་བ། བསྟན་རྡོར་གྱི་བརྩམས་ཆོས་(བོད་ཀྱི་ཞི་བའི་འཐབ་རྩོད་ཀྱི་ལོ་རྒྱུས)་ནང་དུ། བོད་ཀྱི་འཐབ་རྩོད་ནི་འགྱུར་བའི་རང་བཞིན་ཅན་གྱི་ངོ་རྒོལ་ཞིག་ཏུ་འགྱུར་གྱི་འདུག་ཅེས་བྲིས་འདུག གནད་དོན་འདི་ཆ་འཕྲིན་ཁྱབ་ཁོངས་དང་འབྲེལ་བ་ཇི་འདྲ་འདུག དྲ་ལམ་རང་དབང་དང་། བདེ་འཇགས། ཆ་འཕྲིན་ལས་འགུལ་དང་འབྲེལ་བའི་ཐོག་ནས་བོད་ཀྱི་མ་འོངས་འགོག་རྒོལ་ལས་འགུལ་ཇི་འདྲ་ཆགས་ཀྱི་རེད། ལན། བརྡ་འཕྲིན་ལག་རྩལ་གྱི་ཡར་རྒྱས་བརྒྱུད་ནས་བོད་མི་ཚོར་བསམ་ཚུལ་བརྗེ་ལེན་དང་དགའ་སྐྱོ། རེ་བ་དང་རྨི་ལམ་སོགས་འཆད་རེས་བྱེད་པའི་ནུས་པ་དང་ལྡན་པར་གྱུར་བ་དེས་འཐབ་རྩོད་ལ་གསར་བརྗེ་བསླངས་བྱུང་། བོད་མི་མང་པོ་ཞིག་གིས་འཛམ་གླིང་གང་སར་ཁྱབ་པའི་རང་གི་ཕུ་ནུ་མིང་སྲིང་ཚོ་དང་འབྲེལ་འདྲིས་བྱེད་ཐུབ་ཀྱི་ཡོད་པ་ནི་ད་སྔ་བྱུང་མྱོང་མེད་པའི་དོན་དག་རེད། བོད་མི་ཚོས་འཐབ་རྩོད་ཁྲོད་དུ་རང་གི་རིག་གཞུང་དེ་ཆེས་ནུས་ཤུགས་ལྡན་པའི་མཚོན་ཆ་ཞིག་གི་ངོ་བོར་བེད་སྤྱོད་གཏོང་གི་ཡོད་པ་མཐོང་རྒྱུ་ཡོད། འདི་དག་ཡོད་ཚད་སྤྱི་ཚོགས་འབྲེལ་མཐུད་དྲ་བ་དང་ཁ་པར། དྲ་ལམ་བཅས་ནས་མཆེད་དེ་ཁྱབ་གདལ་ཕྱིན་ཡོད། ཕར་འབྲེལ་ཚུར་འབྲེལ་གྱི་རྣམ་པ་འདི་ནི་བོད་ནང་གི་བོད་མི་དང་། བཙན་བྱོལ་བོད་མི། ད་དུང་། རྒྱ་ནག་སྐམ་སར་སློབ་སྦྱོང་བྱེད་མཁན་གྱི་བོད་རིགས་གཞོན་སྐྱེས་བཅས་ཀྱི་དབར་ཏུ་ཁྱབ་ཡོད་སྟབས། རྒྱ་ནག་གིས་རྒྱུན་འཁྱོངས་བྱེད་བཞིན་པའི་བྱ་བ་གཅིག་གིས་ཀྱང་འགྱུར་བ་འདི་འགོག་ཐུབ་ཀྱི་མེད།   ༢༠༠༨ ལོའི་ཡར་ལངས་དེ་ནི་སློབ་འབྲིང་སློབ་མའི་ལོ་ཚད་དང་། བོད་མི་མཐོ་དམན་བར་གསུམ་གྱིས་སྤེལ་བའི་ཞི་བའི་ཡར་ལངས་ཞིག་རེད། གྲྭ་བཙུན་གཅིག་པུ་མིན་པར་སློབ་གྲྭའི་དགེ་རྒན་དང་། ཞིང་པ། འབྲོག་པ། སློབ་ཕྲུག་བཅས་ཀྱིས་ཆ་ཤས་བླངས་ཡོད། སྐབས་དེར་མགོ་བཙུགས་པ་དེ་ད་དུང་མཇུག་རྫོགས་མེད། སྐབས་དེའི་གནས་ཚུལ་གྱིས་མི་རབས་རྗེས་མ་རྣམས་ལ་ནུས་ཤུགས་བསྐྱེད་པ་དང་། བློ་སྤོབས་བསྐྱེད་པ། མངོན་འདོད་བསྐྲུན་ཡོད།   ངས་བལྟས་ན། ང་ཚོས་ལོངས་སྤྱོད་བྱེད་བཞིན་པའི་རང་དབང་གཅིག་མཚུངས་ཐོབ་འོས་པ་དང་། འཐོབ་འདོད་མཁན་གྱི་རྒྱ་མི་རིགས་ཚོའི་ལམ་དང་། ཁོང་ཚོའི་ལམ་ལ་སུ་དང་གང་གིས་ཀྱང་འགོག་རྐྱེན་བྱེད་ཐུབ་ཀྱི་མ་རེད། གནད་དོན་ནི། འགྱུར་བ་འདི་མགྱོགས་པོ་ཇི་ལྟར་ཡོང་མིན་དང་། ང་ཚོའི་སྲིད་གཞུང་ཡིན་ན་འདྲ། ང་ཚོ་སྒེར་བ་ཚོ་དང་ཚོགས་སྡེ་ཁག་ཡིན་ན་འདྲ། འགྱུར་བ་འདིར་རོགས་རམ་བྱེད་ཀྱི་ཡིན་ནམ། འགོག་རྐྱེན་བཟོ་གི་ཡིན་མིན་དེ་རེད། བསྒྱུར་མཆན། རྩོམ་པ་པོ་སྐུ་ཞབས་རིག་རྣམ་ཝང་ཁཱང་ལགས་ཀྱིས་བཀྲས་མཐོང་ལྷ་སྒྲོན་ལགས་སུ་བཅར་འདྲི་ཞུས་པ་ཞིག་ཆོས་རྒྱལ་སྐྱབས་ཀྱིས་འདིར་བོད་བསྒྱུར་ཞུས་ཡོད། རང་དབང་གི་རྩོམ་པ་པོ་རིག་རྣམ་ཝང་ཁཱང་(Rignam Wangkhang)ལགས་ནི། ཁེ་ན་ཌའི་སྨྲ་བརྗོད་རང་དབང་གསར་འགོད་ཚོགས་པ་(CJFE)ཞེས་པའི་ཞུ་གཏུགས་ལས་བྱེད་དང་བོད་རང་བཙན་སློབ་ཕྲུག་ཚོགས་པའི་འགན་འཛིན་ལྷན་ཚོགས་ཀྱི་ཚོགས་མི་བཅས་ཡིན།) ས་གནས་གསར་འགྱུར་མཁོ་གཏོང་བ་ཆོས་རྒྱལ་སྐྱབས།
