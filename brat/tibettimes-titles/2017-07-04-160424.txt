ཧོང་ཀོང་རྒྱ་ནག་གིས་བླངས་ཏེ་ལོ་ ༢༠
ཕྱིན་རྗེས་ཧོང་ཀོང་བ་མང་པོ་ཞིག་ལམ་ལ་ཆས།

སི་ཨིན་ཨིན་གསར་ཁང་གིས་ ༢༠༡༧།༦།༢༧ ཉིན་སྤེལ། སྐུ་ཞབས་ཊི་རན་སི་ཊམ་(Terence Tam)ལགས་ཀྱི་ཨ་ཕ་དང་ནང་མི་ཚོ་རྒྱ་ནག་གི་རིག་གནས་གསར་བརྗེའི་ཟིང་འཁྲུག་ལ་གཡོལ་ཏེ་རྒྱ་ནག་ནས་ཧོང་ཀོང་ལ་བྲོས་བྱོལ་དུ་ཡོང་ཡོད་པ་དང་། སྐབས་དེར་ཧོང་ཀོང་དབྱིན་ཇིའི་མངའ་འོག་ཏུ་ཡོད་པ་དང་། ཕྱི་ལོ་ ༡༩༦༠ དང་ ༡༩༧༠ ལོ་ཡར་མར་ལ་ཧོང་ཀོང་ནི་རྒྱ་ནག་དམར་ཤོག་ཚོགས་པའི་དབང་སྒྱུར་ལ་གཡོལ་མཁན་གྱི་སྐྱབས་བཅོལ་བ་མང་པོ་ཞིག་གི་ཁང་ཁྱིམ་ལྟ་བུར་གྱུར་ཡོད། ད་ལྟ་ལོ་ན་ ༣༩ ཡིན་པའི་སྐུ་ཞབས་ཊམ་ལགས་ཀྱིས་ཧོང་ཀོང་རྒྱ་ནག་ལ་རྩིས་སྤྲོད་བྱས་ཏེ་ལོ་ངོ་ ༢༠ འཁོར་སྐབས། ཁོ་རང་སླར་ཡང་ས་གཞན་ཞིག་ལ་བྲོས་བྱོལ་སྐྱབས་བཅོལ་དུ་འགྲོ་འདོད་ཡོད་ཟེར། ཧོང་ཀོང་འཛམ་གླིང་གི་གྲོང་ཁྱེར་ཕྱུག་ཤོས་གྲས་ཤིག་ཡིན་པའི་སྐབས་འདིར། སྐུ་ཞབས་ཊམ་ལགས་ཀྱིས་ཧོང་ཀོང་ཆབ་སྲིད་ཀྱི་རྙོག་དྲའི་ནང་ཚུད་ཡོད་སྐོར་བརྗོད་དོན། “ངས་རང་གི་ཨ་ཕ་ནང་བཞིན་རང་ཉིད་ཀྱི་ནང་མི་ཚོ་ཁྲིད་དེ་བྲོས་བྱོལ་དུ་འགྲོ་ཐུབ་མིན་ལྟ་རྒྱུ་ཡིན། རྒྱུ་མཚན་ནི། ང་རང་ཧོང་ཀོང་གི་གནས་བབ་འདིར་འདོད་བློ་ཁེངས་ཀྱི་མི་འདུག ཆབ་སྲིད་ཀྱི་གནས་སྟངས་དང་སྲིད་གཞུང་སོགས་ལ་ང་ཚོས་སྐྱོན་བརྗོད་བྱེད་དགོས་ས་མང་པོ་འདུག་” ཅེས་བརྗོད་ཡོད། ངས་ང་ཚོ་སྐྱབས་བཅོལ་བ་ལྟ་བུར་མཐོང་གི་འདུག སྐུ་ཞབས་ཊམ་ལགས་ཀྱིས་གནས་སྟངས་མང་པོའི་ཐད་གྲ་སྒྲིག་ཡག་པོ་བྱས་འདུག དཔེར་ན། སྐུ་ཞབས་ཊམ་ནི་ཧོང་ཀོང་གཙུག་ལག་སློབ་གཉེར་ཁང་ཞིག་གི་འཕྲུལ་ལས་རིག་པའི་འགན་འཛིན་གཞོན་པ་ཡིན་པ་དང་། ཁོ་རང་ལ་འཛམ་གླིང་ཐོག་ཆེས་རིན་གོང་མཐོ་བའི་ཧོང་ཀོང་གི་ཁང་བརྩེགས་དེ་འདྲ་མེད་ཀྱང་། ཁོ་རང་ཁོ་རང་ལ་འོས་ཤིང་འཚམ་པའི་ཁང་པ་ཡག་པོ་ཞིག་ཡོད། འོན་ཀྱང་། སྐུ་ཞབས་ཊམ་ལགས་ཀྱིས་ཧོང་ཀོང་ནང་རྒྱ་ནག་གི་སྟངས་འཛིན་ཤུགས་ཆེ་རུ་འགྲོ་བཞིན་ཡོད་པར་ངོ་རྒོལ་བྱེད་བཞིན་ཡོད། རྒྱལ་ཁབ་གཅིག་ལ་ལམ་ལུགས་གཉིས་ཀྱི་སྲིད་ཇུས་འོག ཧོང་ཀོང་གི་རྩ་ཁྲིམས་གཞིར་བཟུང་སྟེ་ཧོང་ཀོང་ནང་སྨྲ་བརྗོད་རང་དབང་དང་ཞི་བའི་ངོ་རྒོལ་བྱེད་པའི་ཐོབ་ཐང་སོགས་ཧོང་ཀོང་སྔོན་ཆད་དབྱིན་ཇིའི་འོག་ཡོད་སྐབས་དང་འདྲ་བར་ཡོད་དགོས་ངེས་ཡིན་པ་མ་ཟད། གནས་བབ་དེ་མ་མཐར་ཕྱི་ལོ་ ༢༠༤༧ ལོའི་བར་དུ་གནས་དགོས་ངེས་ཤིག་ཡིན་ཡང་། སྐུ་ཞབས་ཊམ་དང་གཞན་ཧོང་ཀོང་བ་མང་པོ་ཞིག་གིས་རྒྱ་ནག་གཞུང་ཁ་དན་ལ་གནས་ཀྱི་མེད་པར་མཐོང་གི་ཡོད་སྐོར་བརྗོད་དོན། “མི་ཁ་ཤས་ཀྱིས་ང་ཚོ་གནས་སྤོར་གྱི་ཡོད་ཟེར་ནས་སྐད་ཆ་སྟབས་བདེ་བོ་བཤད་ཀྱི་འདུག་མོད། འོན་ཀྱང་ང་རང་གི་མཐོང་སྣང་ལ་ང་ཚོ་ཧོང་ཀོང་ནས་བྲོལ་བྱོལ་བྱས་ཏེ་ས་གཞན་ལ་སྐྱབས་བཅོལ་དུ་འགྲོ་བར་མཐོང་གི་འདུག་”ཅེས་བརྗོད་ཡོད། སྐུ་ཞབས་ཊམ་ལགས་ཀྱི་བཙས་མ་ཐག་པའི་བུ་མོ་ལོ་གཉིས་ཀྱི་རྗེས་ལ་སྔོན་འགྲོའི་སློབ་གྲྭར་བསྐྱོད་ཐུབ་པ་དང་། སྐབས་དེར་ཁོས་རང་ཉིད་ཀྱི་ཁྱིམ་ཚང་ཡང་ཐེ་ཝན་ནང་གནས་སྤོ་བྱེད་འཆར་ཡོད་འདུག སྐུ་ཞབས་ཊམ་གྱི་སྐྱེས་མེད་ལ་ཐེ་ཝན་ནང་དཔལ་འབྱོར་མ་འཛུགས་ཀྱི་ཐོངས་མཆན་སྲིད་ཇུས་ཤིག་གི་འོག་ཐེ་ཝན་མི་སེར་ལག་ཁྱེར་ཐོབ་ཡོད་པ་དང་། སྐུ་ཞབས་ཊམ་ལགས་ཀྱིས་རང་གི་ཧོང་ཀོང་ནང་གི་ལས་ཡོལ་ཕོགས་ཐོབ་གཙང་བཟོ་བྱས་ཏེ་ཐེ་ཝན་ནང་ཧོང་ཀོང་གི་ཟས་རིགས་ཚོང་དོན་ཞིག་གཉེར་འཆར་ཡོད་པ་རེད། སྐུ་ཞབས་ཊམ་ནི་ཧོང་ཀོང་ནས་ཕྱིར་འཐོན་འདོད་ཡོད་མཁན་གྲས་ཀྱི་ཧོང་ཀོང་མི་སེར་ཞིག་ཡིན་པ་དང་། ལོ་སྔོན་མའི་ཧོང་ཀོང་ནང་ཡོད་པའི་རྒྱ་ནག་གི་གཙུག་ལག་སློབ་གཉེར་ཁང་ཞིག་གི་ཞིབ་འཇུག་ལྟར་ན། ཕྱི་ལོ་ ༢༠༡༤ ལོར་ཧོང་ཀོང་གི་མི་སེར་ ༥ ཁྲོད་ནས་ ༡ ཕྱིར་འཐོན་འདོད་ཡོད་པ་དང་། ད་ལྟ་ཧོང་ཀོང་མི་སེར་ ༥ ཁྲོད་ནས་ ༢ ཕྱིར་འཐོན་འདོད་བྱེད་ཀྱིན་ཡོད་སྟབས། དེ་དག་ཁྲོད་ཀྱི་བརྒྱ་ཆ་ ༡༠ ཡིས་སོ་སོའི་འདོད་པ་དངོས་སུ་ལག་བསྟར་བྱས་ཏེ། ཕྱི་རོལ་དུ་གནས་སྡོད་བྱེད་པའི་གཞི་རྟེན་གྲ་སྒྲིག་བྱས་ཡོད་འདུག ཧོང་ཀོང་ནང་འགྱུར་བ་ག་རེ་ཐེབས་ཡོད་དམ། ཧོང་ཀོང་ནང་ཆབ་སྲིད་དཀའ་རྙོག་བྱུང་སྟེ་མི་སེར་ཕྱི་ཕྱོགས་སུ་གནས་སྤོ་བྱས་བ་ཐོག་མ་མིན་པར། ཞིབ་འཇུག་བྱས་བ་ལྟར་ན། ཕྱི་ལོ་ ༡༩༨༩ ལོར་རྒྱ་ནག་གནམ་སྡེ་སྒོ་མོ་ཆེའི་དོན་རྐྱེན་བྱུང་རྗེས། ཧ་ལམ་ཧོང་ཀོང་གི་མི་སེར་ ༦༠༠༠༠ ཙམ་ཁེ་ན་ཌ་དང་ཨ་རི། ཨོ་སི་ཊོ་ལི་ཡ་སོགས་སུ་གནས་སྤོར་བྱས་ཡོད། འོན་ཀྱང་། དེའི་ཁྲོད་ཀྱི་མི་མང་པོ་ཞིག་ཕྱིའི་རྒྱལ་ཁབ་ཁག་ཏུ་སྔོན་ཆད་རང་ཉིད་ཀྱི་སེམས་ལ་དྲན་པ་ལྟ་བུའི་གནས་སྟངས་མེད་སྟབས། ཕྱིར་རང་ཡུལ་ཧོང་ཀོང་དུ་ལོག་ཡོད། སྐབས་དེར་པེ་ཅིན་གཞུང་གིས་ཧོང་ཀོང་འཛིན་བཟུང་བྱས་ཏེ་ལོ་ངོ་ ༡༥ རིང་ཧོང་ཀོང་གི་གྲོང་ཁྱེར་མང་ཆེ་ཤོས་སྟོང་བ་ཆགས་ཡོད། ཁེ་ན་ཌའི་གཞུང་གིས་ཞིབ་འཇུག་བྱས་པ་ལྟར་ན། ཕྱི་ལོ་ ༡༩༩༦ དང་ ༢༠༡༡ བར་ལ་ཧོང་ཀོང་གི་མི་སེར་ ༦༥༠༠༠ ཙམ་ཞིག་ཁེ་ན་ཌ་ནས་ཕྱིར་ཧོང་ཀོང་ནང་ལོག་ཡོད་ཀྱང་། ཕྱི་ལོ་ ༢༠༡༤ ལོར་རྒྱ་ནག་གཞུང་གིས་ཧོང་ཀོང་མི་མང་ལ་འོས་བསྡུའི་ཐོབ་ཐང་མེད་པ་བཟོས་ཤིང་། མ་རྩ་རིང་ལུགས་སུ་བསྒྱུར་བའི་རྩ་ཁྲིམས་ཀྱིས་ཧོང་ཀོང་སྔར་བཞིན་སྐྱོང་རྒྱུ་ཡིན་ཚུལ་བསྒྲགས་རྗེས། ཧོང་ཀོང་ནང་གི་རང་དབང་དོན་གཉེར་བ་དང་སློབ་ཕྲུག་ཚོས་ཟླ་གཉིས་ལྷག་རིང་ཧོང་ཀོང་གྲོང་ཁྱེར་གྱི་ལྟེ་བ་བཟུང་སྟེ་རྒྱ་ནག་གཞུང་དང་ཧོང་ཀོང་གཞུང་ལ་ངོ་རྒོལ་བྱས་ཡོད་པ་དང་། སྐབས་དེ་ནས་བཟུང་ཧོང་ཀོང་གི་མི་མང་པོ་ཞིག་ལ་རྒྱ་ནག་གིས་ཧོང་ཀོང་གི་རང་སྲིད་རང་སྐྱོང་ཟེར་བ་དེ་ཁ་ཙམ་ལས་དོན་གྱིས་ཧོང་ཀོང་མི་མང་གི་རང་དབང་ལ་རྡོག་རོལ་གཏོང་གི་ཡོད་པར་འཇིགས་སྣང་ཆེན་པོ་བསྐྲུན་ཡོད། ཕྱི་ལོ་ ༢༠༡༥ ལོར་ཧོང་ཀོང་གི་རྩོམ་པ་པོ་ ༥ ཡིས་རྒྱ་ནག་གིས་ཧོང་ཀོང་ལ་ཐེ་གཏོགས་ཤུགས་ཆེ་རུ་འགྲོ་བཞིན་ཡོད་སྐོར་གྱི་དཔེ་དེབ་ཅིག་བརྩམས་ཡོད། དེའི་ཁྲོད་ཀྱི་མི་གཉིས་ལ་ཡུ་རོབ་ཀྱི་མི་སེར་ལག་ཁྱེར་ཡོད་ཀྱང་གློ་བུར་དུ་གར་སོང་ཆ་མེད་ཆགས་ཏེ་བདུན་ཕྲག་ ༢ ཀྱི་རྗེས་སུ་རྒྱ་ནག་གི་བཙོན་ཁང་ཞིག་ཏུ་བཅུག་ཡོད་པའི་གནས་ཚུལ་ཐོན་ཡོད། ན་ནིང་ཁེ་ན་ཌའི་མི་སེར་ལག་ཁྱེར་ཡོད་པའི་རྒྱ་ནག་གི་ཕྱུག་བདག་ཞིག་ཀྱང་རང་ཉིད་ཀྱི་རྒྱས་སྤྲོས་ལྡན་པའི་(Four Season)མགྲོན་ཁང་ནས་བཙན་ཁྲིད་བྱས་རྗེས་ཤི་གསོན་གྱི་གནས་སྟངས་གསལ་བོ་མེད་པ་ཆགས་ཡོད། དེར་མ་ཟད་རྒྱ་ནག་གཞུང་གིས་ཧོང་ཀོང་གི་དམངས་གཙོའི་རང་དབང་ལ་རྡོག་རོལ་བཏང་སྟེ། ཧོང་ཀོང་གི་ཁྲིམས་བཟོ་བ་ཁག་ཅིག་ཀྱང་ལས་གནས་བཀག་སྡོམ་བྱས་ཡོད་པ་རེད། ད་ཆར་ཧོང་ཀོང་གི་རང་སྲིད་རང་སྐྱོང་གི་གནས་བབ་རིམ་གྱིས་ཉམས་ཆག་ཏུ་འགྲོ་བཞིན་ཡོད་པ་དེས་མི་ཁག་ཅིག་ལ་ཧོང་ཀོང་རང་བཙན་རྩོད་པར་སྐུལ་མ་བྱས་ཡོད། འོན་ཀྱང་ཁོ་ཚོ་གནས་སྐབས་རིང་ཨུ་ཚུགས་རྒྱག་མཁན་གྱི་སྡེ་ཞིག་ཆགས་ཡོད། ཚོགས་སྒེར་མང་པོ་ཞིག་གི་འཐུས་ཚབ་ ༡༢༠༠ ལས་གྲུབ་པའི་འོས་བསྡུ་ལྷན་ཚོགས་ཀྱིས་འདེམ་བསྐོ་བྱས་བའི་ཧོང་ཀོང་སྤྱི་ཁྱབ་གསར་བ་ལྕམ་ཀེར་ལཱམ་ལ་མི་མང་པོ་ཞིག་ཞེད་སྣང་སྐྱེ་བཞིན་ཡོད་པ་དང་། དེ་ནི་ལྕམ་ཀེར་ལཱམ་(Carrie Lam)གྱིས་ཟླ་ ༧ པའི་ཚེས་ ༡ ཉིན་ཧོང་ཀོང་སྤྱི་ཁྱབ་ཀྱི་ཡིག་ཚང་བཟུང་སྟེ་རྩོད་སྙོག་རང་བཞིན་ཅན་གྱི་ཁྲིམས་སྲིད་དོན་ཚན་ ༢༣ པ་ཞེས་བ་ཅིག་ལག་བསྟར་བྱས་ཏེ། དེས་ཧོང་ཀོང་ནང་རྒྱ་ནག་གཞུང་ལ་རྒྱབ་གཏོད་དང་ཁ་བྲལ། གྱེན་ལོག དེ་བཞིན། རྩ་གཏོར་བྱེད་འདོད་ཀྱི་ལས་འགུལ་གཅིག་ཀྱང་སྤེལ་མི་ཆོག་པ་བཟོ་སྲིད་པ་རེད་འདུག ཐེ་ཝན་ཕྱོགས་སུ་བལྟས་བ། ཧོང་ཀོང་ཨམ་ལེ་ཁི་སི་(Amlex)རུ་ཡོད་པའི་ཕྱི་མི་གནས་སྤོར་བའི་ལམ་སྟོན་པ་སྐུ་ཞབས་ཨན་ཌི་རའོ་ལའོ་(Andrew Lo)ལགས་ཀྱིས་བརྗོད་པ་ལྟར་ན།ཕྱི་ལོ་ ༢༠༡༤ ལོར་ཧོང་ཀོང་མི་མང་གིས་ཧོང་ཀོང་གྲོང་ཁྱེར་ལྟེ་བ་བཟུང་སྟེ་གཞུང་ལ་ངོ་རྒོལ་བྱས་སྟབས། ཧོང་ཀོང་གི་ཆབ་སྲིད་གནས་སྟངས་ཇེ་སྡུག་ཏུ་ཕྱིན་ཏེ་ཁོ་རང་གི་བཟོ་གྲྭའི་ཚོང་ཡང་ཉམ་མགོ་བཙུགས་སོང་པ་དང་། ཕྱོགས་གཞན་ཞིག་ནས་ཧོང་ཀོང་གི་སློབ་གསོའི་གོ་སྐབས་དམ་དྲག་ཇེ་ཆེར་ཕྱིན་པ་དང་། མི་མང་གི་འཚོ་བའི་འགྲོ་གྲོན་ཇེ་མཐོར་ཕྱིན་པ་སོགས་ཀྱི་རྐྱེན་པས་ཀྱང་ཁོ་རང་གི་བཟོ་གྲྭའི་ཚོང་ཁེ་མགོ་མཇུག་བསློགས་ཡོད་འདུག སྐུ་ཞབས་ལའོ་ལགས་ཀྱིས་ “ཧོང་ཀོང་མི་མང་གིས་ཧོང་ཀོང་རྒྱབ་ཏུ་བསྐྱུར་ནས་འགྲོ་དགོས་པའི་རྒྱུ་རྐྱེན་གཙོ་བོ་ནི། ཧོང་ཀོང་གི་ཆབ་སྲིད་གནས་བབ་བརྟན་པོ་མིན་པ་དང་། མི་མང་གི་མ་འོངས་པའི་འཚོ་བར་ངེས་གཏན་མེད་པ། དེ་བཞིན། སློབ་གསོའི་མཐུན་རྐྱེན་ཇེ་ཞན་དུ་ཕྱིན་པ་སོགས་ལ་བརྟེན་ཏེ་བྱུང་བ་རེད་”ཅེས་དང་། ཁོང་གིས་མུ་མཐུད་བརྗོད་དོན། ད་ལྟ་ཐེ་ཝན་དང་ལྷོ་རྒྱུད་ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་མཱ་ལེ་ཞི་ཡ། ཐེ་ལན་ཌི་སོགས་ནི་ཧོང་ཀོང་གི་ལས་བྱེད་རྒན་ཡོལ་ཞུས་ཟིན་པ་དང་། ཁྱིམ་ཚང་དག་གནས་སྤོ་བྱེད་འདོད་ཆེ་ཤོས་ཀྱི་ས་ཆ་དག་རེད། དེ་མིན། ཨོ་སི་ཊོ་ལི་ཡ་དང་ཁེ་ན་ཌར་གནས་སྤོ་བྱ་རྒྱུ་ནི་ད་ལྟ་ཕྱི་ལོ་ ༡༩༩༠ ལོའི་སྔ་རྗེས་ལྟ་བུའི་ལས་སླ་བོ་མ་རེད་ཅེས་བརྗོད་ཡོད། སྐུ་ཞབས་ཊམ་ལགས་ཀྱིས་ཐེ་ཝན་བདམས་ཡོད། རྒྱུ་མཚན་ནི། ཐེ་ཝན་གྱི་རྩ་བའི་རིག་གཞུང་ཧོང་ཀོང་དང་གཅིག་མཚུངས་ཡིན་སྟབས། ཁོ་རང་ནམ་རྒྱུན་ཐེ་ཝན་ནང་གུང་གསེང་དུ་སྐྱོད་བཞིན་ཡོད་ཟེར། སྐུ་ཞབས་ཊམ་ལགས་ཀྱིས་ཐེ་ཝན་ནང་དུའང་མི་ཚེ་སྐྱེལ་རྒྱུ་ལས་སླ་བོ་ཞིག་མིན་པ་ཚོར་ཀྱི་ཡོད། ཁོ་རང་གི་ཕ་སྐད་ཧོང་ཀོང་སྤྱི་སྐད་(Cantonese)ཡིན་ཡང་། ཁོས་ཐེ་ཝན་མི་མང་གིས་ཐེ་ཚོམ་ངང་བེད་སྤྱོད་གཏོང་བཞིན་པའི་རྒྱ་སྐད་བེད་སྤྱོད་གཏོང་གི་ཡོད་ཟེར་བ་དང་། ཐེ་ཝན་ནང་ཆ་སྙོམས་ཐོག་གླ་ཆ་དང་ཡོང་འབབ་སོགས་ཧོང་ཀོང་དང་བསྡུར་ན་ཉུང་ཉུང་རེད་ཟེར། སྐུ་ཞབས་ཊམ་ལགས་ཀྱིས། “ང་རང་ཐེ་ཝན་གྱི་རིག་གཞུང་ལ་དགའ་པོ་ཡོད། དེ་སྔོན་ངས་ལོ་གཅིག་རིང་དབྱིན་ཡུལ་དུ་སློབ་སྤྱོང་བྱས་པ་ཡིན། སྐབས་དེར་ང་ཐོག་མར་དབྱིན་ཇིའི་ནང་འབྱོར་བ་དང་། ངའི་སེམས་ནང་ང་རང་ས་ཆ་དེར་བདག་གི་མེད་པའི་ཚོར་སྣང་བྱུང་སོང་། འོན་ཀྱང་། ང་རང་ཐེ་ཝན་ལ་ཕྱིན་པའི་ཚོར་སྣང་དེ་མ་འདྲ་བ་ཞིག་རེད་”ཅེས་བརྗོད་ཡོད། ཐེ་ཝན་ནི་འཛམ་གླིང་ཐོག་རྒྱ་སྐད་བེད་སྤྱོད་གཏོང་མཁན་གྱི་དམངས་གཙོའི་རྒྱལ་ཁབ་ཁོ་ན་དེ་ཡིན་སྟབས་མི་མང་པོ་ཞིག་གི་ཡིད་དབང་འཕྲོག་ཡོད། ཐེ་ཝན་གཞུང་གི་ཞིབ་འཇུག་ལྟར་ན། ཐེ་ཝན་ནང་བརྟན་སྡོད་བྱེད་མཁན་གྱི་ཧོང་ཀོང་མི་སེར་ཕྱི་ལོ་ ༢༠༡༤ ལོ་དང་བསྡུར་ན་ད་ལྟ་ལྡབ་གཉིས་ཀྱིས་ཇེ་མང་དུ་སོང་ཡོད་ཀྱང་། རྩ་བའི་ ༢༠༡༦ ལོར་མི་གྲངས་ ༡༠༨༦ ཙམ་གནས་སྤོ་བྱས་ཡོད། འོན་ཀྱང་། ཐེ་ཝན་ལའང་ཆབ་སྲིད་ཀྱི་དཀའ་ངལ་དང་སྙོག་གྲ་མེད་པ་མིན། ཐེ་ཝན་ལོ་ངོ་ ༦༠ ལྷག་གི་རིང་རང་དབང་ཅན་གྱི་རྒྱལ་ཁབ་ཅིག་ཏུ་གནས་ཡོད་མེད་ལ་མ་བལྟས་བར། རྒྱ་ནག་གཞུང་གིས་ཐེ་ཝན་ནི་ཁ་བྲལ་ཐབས་མེད་པའི་རྒྱ་ནག་གི་ཆ་ཤས་ཤིག་ཡིན་ཟེར་བ་དང་། གལ་སྲིད་དེར་བཙན་ཤུགས་བཀོལ་དགོས་ཁེལ་ཡང་ཕྱིར་རྒྱ་ནག་གི་མངའ་འོག་ཏུ་ལེན་རྒྱུ་ཡིན་ཚུལ་སོགས་རྗོད་བཞིན་ཡོད་པ་དང་། ཕྱི་ལོ་ ༡༩༤༩ ལོར་རྒྱ་ནག་ནང་དམར་ཤོག་ཚོགས་པར་དམག་འཁྲུག་ལ་རྒྱལ་ཁ་ཐོབ་རྗེས་གུང་ཁྲན་ཏང་དང་གོ་མིན་ཏང་གཉིས་ཁ་བྲལ་དུ་ཕྱིན་ཡོད། སྐུ་ཞབས་ཊམ་ལགས་ཀྱིས། “ང་ལ་ལས་གནས་བརྟན་པོ་དང་ཡོང་འབབ། དེ་བཞིན། སྡོད་ཁང་སོགས་ཡོད་མོད། འོན་ཀྱང་ང་རང་ད་དུང་ངའི་ཁོར་ཡུག་འདིར་བློ་བདེ་བོ་དང་སྐྱིད་པོ་མི་འདུག སྤྱིར་བཏང་ང་ཚོ་ཕྱི་ལོ་ ༡༩༩༧ ལོའི་གོང་དུ་སྐྱེས་བ་ཡིན་སྟབས། སྐྱེས་ཚེས་ལག་ཁྱེར་དབྱིན་ཇིའི་གཞུང་གིས་ང་ཚོར་སྤྲད་ཡོད་པ་དང་། ང་ཚོས་ཕྱི་ལོ་ ༡༩༩༧ ལོར་ཧོང་ཀོང་རྒྱ་ནག་ལ་རྩིས་སྤྲོད་བྱས་པ་ཙམ་ནས་ཧོང་ཀོང་ནང་འགྱུར་བ་འགྲོ་བཞིན་ཡོད་པ་གང་ཡང་མཐོང་གི་མེད་ཀྱང་། ད་ལྟ་ང་ཚོས་འགྱུར་བ་ཤུགས་ཆེན་པོ་འགྲོ་བཞིན་ཡོད་པ་མཐོང་གི་འདུག་” ཅེས་སོགས་བརྗོད་ཡོད། ཟླ་བདུན་པའི་ཚེས་ ༡ ཉིན་མི་མང་སྟོང་ཕྲག་མང་པོས་ཧོང་ཀོང་གི་བཞའ་ཚན་ཆེ་བའི་དབྱར་ཁའི་ནམ་ཟླའི་ཁྲོད། ཧོང་ཀོང་གི་སྲང་ལམ་འཛིན་བཟུང་བྱས་ཏེ་ཧོང་ཀོང་རྒྱ་ནག་ལ་རྩིས་སྤྲོད་ཀྱི་ལོ་ངོ་ ༢༠ འཁོར་བའི་མཛད་སྒོའི་སྐབས་ཧོང་ཀོང་ལ་དམངས་གཙོ་དགོས་སྐོར་གྱི་སྐད་འབོད་རྒྱག་འཆར་ཡོད་པ་དང་། ད་ལོའི་ལོ་འཁོར་མཛད་སྒོ་དེར་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་ཧོང་ཀོང་ནང་ཐོག་མར་ཡོང་གི་ཡོད་སྟབས། དམིགས་བསལ་གྱི་མཛད་སྒོ་ཞིག་ཏུ་བརྩི་བཞིན་ཡོད། སྐུ་ཞབས་ཊམ་ལགས་འདས་བའི་སྐད་འབོད་ངོ་རྒོལ་གྱི་ལས་འགུལ་ཁག་གི་ནང་མཉམ་ཞུགས་བྱས་ཡོད་ཀྱང་། ད་ལོར་མཉམ་ཞུགས་བྱ་འཆར་མེད་པ་རེད། རྒྱུ་མཚན་ནི། ཁོས་ལས་འགུལ་དེ་དག་ལ་གྲུབ་འབྲས་ཐོན་གྱི་མེད་པར་ཚོར་བཞིན་ཡོད་སྟབས། ཁོས་ “ད་ལྟ་ང་ལ་འདམ་གཞན་གཅིག་རང་ལས་མེད་པ་དང་། དེ་ནི་ང་རང་ཧོང་ཀོང་གི་ཕྱི་ལོགས་སུ་འཐོན་རྒྱུ་དེ་རེད་ལ། འདམ་ག་དེར་གོམ་པ་རྒྱག་རྒྱུའི་དུས་ཚོད་ཀྱང་ད་ལྟ་དངོས་སུ་སླེབ་འདུག་”ཅེས་བརྗོད་ཡོད། གསར་འགོད་པ། སྐལ་བཟང་སྦྱིན་པ། འབྲེལ་ཐག http://edition.cnn.com/2017/06/26/asia/hong-kong-handover-exit/index.html
