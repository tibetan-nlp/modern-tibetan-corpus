བོད་མི་ ༡༠ གྲངས་ཤིག་བཙན་ཁྲིད།

ཁ་སང་སྔ་དྲོར་སི་ཁྲོན་ཞིང་ཆེན་གྱི་འཐུས་མིའི་ཚོཊ་ཆེན་དང་བསྟུན་ཏེ། ཚོགས་ཁང་ཕྱི་རོལ་དུ་བོད་མི་ ༡༠ གྲངས་ཤིག་གིས་ཞི་རྒོལ་གནང་བར་བརྟེན་ཉེན་རྟོག་པས་བཙན་ཁྲིད་བྱས་ཡོད་འདུག དེ་རིང་རྒྱ་ནག་དྲུག་བཞི་དྲ་བའི་ཐོག་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཁ་སང་སྔ་དྲོར་སི་ཁྲོན་ཞིང་ཆེན་གྱི་མི་མང་འཐུས་མི་སྐབས་ ༢ པའི་ཚོགས་ཆེན་ཐེངས་ ༣ པ་ཁྲིན་ཏུའུ་གྲོང་ཁྱེར་ཅིན་ཅང་མགྲོན་ཁང་དུ་ཚོགས་པའི་སྐབས་དང་བསྟུན་ཏེ། ས་གནས་ཀྱི་སྔ་དྲོའི་ཆུ་ཚོད་ ༡༡ ཙམ་ལ་མགྲོན་ཁང་ཕྱི་ལོགས་སུ་བོད་ཁུལ་ནས་ཡོང་བའི་འབྲོག་པ་ ༡༠ གྲངས་ཤིག་གིས། རང་སྲོག་གིས་འབྲོག་ས་སྲུང་སྐྱོབ་དང་། ཁྲིམས་འགལ་གྱིས་ས་ཆ་བསྡུ་ལེན་བྱེད་པར་མཐའ་གཅིག་ཏུ་ངོ་རྒོལ་བྱེད་ཀྱི་ཡོད། ང་ཚོའི་ས་ཆ་བཙན་བཟུང་བྱས་ཏེ་ང་ཚོ་ཁྱིམ་མེད་དུ་གྱུར་ཡོད། ཅེས་པ་སོགས་ཀྱི་འཕྲེད་དར་བཟུང་སྟེ་ཞི་རྒོལ་སྐད་འབོད་བྱས་ཡོད་པ་དང་། ངོ་རྒོལ་བྱས་ནས་སྐར་མ་ཁ་ཤས་ལས་མ་འགོར་བར་ཉེན་རྟོག་པ་ ༥༠ ཙམ་བསླེབས་ཏེ་བཙན་འཁྲིད་བྱས་ཡོད་འདུག སྐབས་དེར་ཡུལ་དངོས་སུ་ཡོད་པའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་དོན་གཉེར་བ་ཝུ་ཞན་ཅེ་ཞུ་བ་ཞིག་གིས། ཞོགས་པའི་ཆུ་ཚོད་ ༡༡ ལ། འབྲོག་པ་ཚོས་ང་ཚོའི་ཁང་ས་བཙན་བཟུང་བྱས་པ་ཕྱིར་སྤྲོད་རོགས། འབྲོག་སར་མཐའ་གཅིག་ཏུ་སྲུང་སྐྱོབ་བྱ་རྒྱུ་ཡིན་ཞེས་པའི་འཕྲེད་དར་བཟུང་སྟེ་མགྲོན་ཁང་མདུན་དུ་བསླེབས་སོང་། ཡིན་ནའང་། སྐར་མ་ཁ་ཤས་ལས་མ་འགོར་བར་ཉེན་རྟོག་པ་བཞི་བཅུ་ལྔ་བཅུ་ཞིག་ཡོང་སྟེ་ཁོང་ཚོའི་འཕྲེད་དར་བཙན་འཕྲོག་བྱས་པ་མ་ཟད། ཞི་རྒོལ་བ་ཚང་མ་རླངས་འཁོར་ཞིག་གི་ནང་བཅུག་སྟེ་ཁྲིད་སོང་ཡང་གང་དུ་འཁྲིད་ཡོད་མེད་ཤེས་མ་སོང་། དེ་རིང་མགྲོན་ཁང་གི་སྒོ་ཆེན་སོགས་སུ་ཉེན་རྟོག་ལྭ་བ་གྱོན་ཡོད་མེད་ཀྱི་ཉེན་རྟོག་པ་ཧ་ཅང་མང་པོ་འགྲེམས་འཇོག་བྱས་འདུག ཅེས་བརྗོད་ཡོད་པ་རེད། གཞན་ཡང་། ཕྱི་ཚེས་ ༢༧ ཉིན་མེན་ཡཱང་གྲོང་ཁྱེར་དུ་སྲིད་གཞུང་གིས་ཞིང་ས་བཙན་འཕྲོག་བྱས་པའི་ཞིང་པ་ ༡༠༠ ལྷག་འདུ་འཛོམས་བྱས་ཏེ། ཞིང་པ་མང་ཚོགས་ཀྱི་སྤྱི་ཚོགས་འགན་ལེན་དངུལ་ཇེ་མང་དུ་གཏོང་དགོས་པ་དང་འཚོ་དངུལ་དུས་ཐོག་ཏུ་སྤྲོད་དགོས་པའི་ཞུ་འབོད་ལས་འགུལ་སྤེལ་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
