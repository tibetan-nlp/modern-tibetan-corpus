
དྷ་རམ་ས་ལ་ཧི་མ་ཅལ་མངའ་སྡེའི་རྒྱལ་ས་གཉིས་པར་ངོས་འཛིན་བྱས།

ཁ་སང་ཕྱི་ཚེས་ ༡༩ ཉིན་ཧི་མ་ཅིལ་མངའ་སྡེའི་སྤྱི་ཁྱབ་བློན་ཆེན་སྐུ་ཞབས་ཝར་བྷ་དྷཱ་རཱ་སིང(Virbhadra Singh )་གིས་དྷ་རམ་སཱ་ལ་དེ་ཧི་མ་ཅལ་མངའ་སྡེའི་རྒྱལ་ས་གཉིས་པར་ངོས་འཛིན་གསལ་བསྒྲགས་གནང་འདུག དེ་ནི་ཉེ་ཆར་ཚོགས་པའི་མངའ་སྡེའི་གྲོས་ཚོགས་མཐའ་མའི་སྐབས་ཐག་གཅོད་བྱུང་བ་ཡིན་འདུག་ལ། ཁ་སང་སྤྱི་ཁྱབ་བློན་ཆེན་ཁོང་དྷ་ཤོད་སོགས་སུ་གཟིགས་སྐོར་གནང་བ་དང་བསྟུན་ཏེ་གསལ་བསྒྲགས་དེ་གནང་ཡོད་འདུག ཁོང་གིས་ “དྷ་རམ་ས་ལ་གནས་ཆགས་པའི་རྡོ་ལ་དྷར་རི་རྒྱུད་ནི་ལོ་རྒྱུས་ཀྱི་ངོས་འཛིན་གལ་ཆེན་ཞིག་ཡིན་སྟབས། ཧི་མ་ཅལ་མངའ་སྡེའི་རྒྱལ་ས་གཉིས་པར་ངོས་འཛིན་བྱས་ན་ཧ་ཅང་འོས་ཤིང་འཚམས་པ་ཞིག་རེད།” ཅེས་བརྗོད་འདུག དེ་ནི་ཧི་མ་ཅལ་མངའ་སྡེ་ལ་གྲོས་ཚོགས་རྐུབ་སྟེགས་ ༦༨ ཡོད་པ་ལས། ཀང་ར་རྫོང་(Kangra)དང་ཨོ་ནཱ་རྫོང་(Una)། ཧཱ་མོར་པུར་རྫོང (Hamirpur)སོགས་ཀྱིས་རྐུབ་སྟེགས་བརྒྱ་ཆ་ ༢༥ ཟིན་ཡོད་པ་མ་ཟད། དྷ་རམ་ས་ལ་ནི་ཧི་མ་ཅལ་མངའ་སྡེའི་ཀང་ར་རྫོང་གཙོས་པའི་ཅཱམ་བྷ་(Chamba)རྫོང་དང་། ཧཱ་མོར་པུར་(Hamirpur)རྫོང་། ཨོ་ནཱ(Una)རྫོང་སོགས་ལའང་གལ་གནད་ཆེན་པོ་ཆགས་ཡོད་པ་སོགས་ལ་བརྟེན་ཏེ་ད་རེབ་གྲོས་ཚོགས་ནང་དྷ་རམ་ས་ལ་ཧི་མ་ཅལ་མངའ་སྡེའི་རྒྱལ་ས་གཉིས་པར་ངོས་འཛིན་བྱུང་བ་ཞིག་ཡིན་འདུག སྤྱི་ཁྱབ་བློན་ཆེན་གྱིས་དྷ་རམ་ས་ལཱ་ཉེ་འཁོར་གྱི་ཡུལ་མི་ཚོས་ཐག་གཅོད་འདིའི་གལ་གནད་མི་རིང་བར་ངོས་ཟིན་ཡོང་གི་རེད་ལ། ཁོ་ཚོ་འདི་ནས་ལམ་ཐག་རིང་པའི་སིམ་ལཱ་སོགས་ལ་ལས་ཀ་བྱེད་དུ་འགྲོ་མི་དགོས་པ་ཆགས་ཡོང་གི་རེད། ཅེས་བརྗོད་ཡོད་འདུག དྷ་རམ་ས་ལ་དེ་སྲིད་བློན་མོ་ཌི་ཡི་དེང་རབས་ཅན་གྱི་གྲོང་ཁྱེར་དུ་བསྒྱུར་དགོས་པའི་རྒྱ་གར་གྱི་ཁྲོང་ཁྱེར་བརྒྱ་ཡི་ནང་དུ་ཚུད་ཡོད་པ་རེད་ལ། དྷ་ས་ནི་ཆོས་ལུགས་དང་རིག་གནས་སོགས་ཀྱི་ལྟེ་གནས་གལ་ཆེན་ཞིག་ཡིན་པ་དང་། ཡུལ་སྐོར་བ་སོགས་རྒྱུན་ཆད་མེད་པར་ཡོང་གི་ཡོད་པ་དེ་ནི་རྒྱ་གར་ཙམ་མ་ཡིན་པར་རྒྱལ་སྤྱིའི་ས་ཁྲའི་ངོས་སུ་གལ་གནད་ཆེ་བའི་ས་ཚིགས་ཤིག་ཆགས་ཡོད་པ་རེད། ཧི་མ་ཅལ་མངའ་སྡེའི་དགུན་དུས་ཀྱི་གྲོས་ཚོགས་འཚོགས་དུས་ ༡༢ པ་དེ་དྷ་རམ་ས་ལཱར་ཚོགས་ཡོད་འདུག ཧུན་རྡི་སི་ཐན་དྲ་བས་གནས་ཚུལ་སྤེལ་གསལ་ལྟར་ན། དྷ་ས་ཁུལ་གྱི་དབུ་ཁྲིད་དང་མི་མང་ཚོས་སྤྱི་ཁྱབ་བློན་ཆེན་ལགས་ལ་དྷ་ས་དེ་ཧི་མ་ཅལ་མངའ་སྡེའི་རྒྱལ་ས་གཉིས་པར་གསལ་བསྒྲགས་པར་ཐུགས་རྗེ་ཆེ་ཞུ་བཞིན་ཡོད་པ་རེད། ཅེས་བརྗོད་འདུག དྷ་རམ་ས་ལཱ་ནི་༧གོང་ས་མཆོག་གི་རྒྱུན་གཏན་བཞུགས་གནས་དང་བཙན་བྱོལ་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་ཆགས་ཡུལ་ལྟེ་བའང་ཡིན། གསར་འགོད་པ། སངས་རྒྱས་དཀོན་མཆོག
