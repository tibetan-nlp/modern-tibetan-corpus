དགེ་རྒན་ལ་རང་ཤི་ངོ་རྒོལ་བྱས།

རྒྱ་གར་གྱི་ལོ་ན་ ༡༢ ཡིན་པའི་སློབ་ཕྲུག་བུ་མོ་ཞིག་གིས་ང་ལ་དགེ་རྒན་གྱིས་མནར་གཅོད་བཏང་སོང་ཞེས་རང་ཤི་བརྒྱབ་ཡོད་འདུག བྷི་བྷི་སི་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན།  ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༨ ཉིན་རྒྱ་གར་ལྷོ་ཕྱོགས་མངའ་སྡེ་ཊ་མིལ་ནཱ་རྡོའི་གྲོང་ཁྱེར་(Tirunelveli)ནས་ཡིན་པའི་ལོ་ན་༡༢ ཅན་གྱི་སློབ་ཕྲུག་བུ་མོ་ཞིག་གི་འགྲིག་ཆས་གོས་ཐུང་ཕྱི་རུ་ཟླ་མཚན་ཐོན་ཡོད་པ་དགེ་རྒན་གྱིས་མཐོང་སྟེ། སློབ་ཕྲུག་ཚོའི་མདུན་ནས་བུ་མོ་དེ་ལ་དགེ་རྒན་གྱིས་སྐྱོན་བརྗོད་བྱས་ཏེ་འཛིན་ཁང་གི་ཕྱི་རུ་ཕུད་ནས་ལངས་བཅུག་པར་བརྟེན། བུ་མོ་དེས་ང་ལ་དགེ་རྒན་གྱིས་མནར་གཅོད་བཏང་སོང་ཞེས་རང་ཤི་བརྒྱབ་སྟེ་འདས་གྲོངས་སུ་ཕྱིན་ཡོད་འདུག བུ་མོ་དེའི་ཨ་མས་ངའི་བུ་མོར་འཛིན་གྲྭ་ཚོགས་སྐབས་ཟླ་མཚན་བབས་ཡོད་སྟབས།  བུ་མོས་དགེ་རྒན་ལ་དེའི་སྐོར་བཤད་སྐབས། དགེ་རྒན་དེས་མོ་གཙང་མ་མིན་པའི་ངོས་འཛིན་བྱས་ཏེ་འཛིན་གྲྭའི་ཕྱི་ལོགས་སུ་ཕུད་ནས་ལངས་ཏུ་བཅུག་སྟེ་བཞག་འདུག་པ་མ་ཟད།  དགེ་རྒན་དེས་དེ་སྔོན་ནང་ལས་མ་བྲིས་པ་བཤད་དེ་ངའི་བུ་མོར་བརྡུངས་མྱོང་ཡོད། ཅེས་བརྗོད་འདུག བུ་མོ་དེས་རང་ཤི་མ་བརྒྱབ་གོང་ཁ་ཆེམས་ཡི་གེ་ཞིག་བཞག་ཡོད་པའི་ནང་དུ། ངའི་བརྩེ་བའི་ཨ་མ་ལགས།  ང་ལ་བཟོད་བསྲན་གནང་རོགས་ཅེས་དང་།   ངའི་དགེ་རྒན་གྱིས་ང་ལ་ག་རེ་བྱས་ནས་སྐྱོན་བརྗོད་བྱས་པ་ཡིན་ནམ།  ད་དུང་ཁོ་ཚོས་ང་ལ་བརྙས་བཅོས་དང་མནན་གཅོད་འདི་འདྲ་ཞིག་གཏོང་དགོས་དོན་ཇི་ཡིན་ནམ་བསམ་བྱུང་།  ཞེས་སོགས་བྲིས་ཡོད་འདུག རྒྱ་གར་གྱི་རིག་གཞུང་གོམས་སྲོལ་ནང་དུ་བུད་མེད་ལ་ཟླ་མཚན་འབབ་སྐབས།  བུད་མེད་དེ་བཙོག་པ་ཡིན་པར་ངོས་འཛིན་བྱེད་ཀྱི་ཡོད་པ་དང་། ད་ལྟའང་ད་དུང་རྒྱ་གར་གྱི་གྲོང་གསེབ་ཁག་ཏུ་དེ་ལྟར་ངོས་འཛིན་བྱེད་མཁན་ཡོད་འདུག རྒྱ་གར་ཉེན་རྟོག་པས་བུ་མོས་རང་ཤི་བརྒྱབ་པའི་གྱོད་དོན་ཐོ་འགོད་བྱས་ཏེ་ཞིབ་འཇུག་བྱེད་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག་ལ། བུ་མོ་དེའི་སློབ་གྲྭས་ཀྱང་ཉེན་རྟོག་པ་དང་ལྷན་དུ་གྱོད་དོན་དེར་ཞིབ་འཇུག་བྱེད་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། སྐལ་བཟང་སྤྱིན་པ།
