སོག་པོའི་ས་མཚམས་སུ་ཁྲལ་གསར་པ་བཟོས།

རྒྱ་ནག་གཞུང་གིས་སོག་པོ་དང་རྒྱ་ནག་བར་གྱི་འགག་སྒོ་རུ་ལམ་ཁྲལ་དང་ཚོང་ཟོག་ལ་ཁྲལ་ལེན་པའི་སྲིད་བྱུས་གསར་པ་ཞིག་བཟོས་ཡོད་འདུག དེ་རིང་ནས་བཟུང་རྒྱ་ནག་གཞུང་གིས་སོག་པོ་དང་རྒྱ་ནག་བར་གྱི་འགག་སྒོ་གཙོ་བོ་གྷ་ཞོན་སུ་ཁེ་ཊི་(Gashuun Sukhait)ཡི་ནང་སོག་ཁོངས་ཀྱི་གཞུང་ལམ་དུ་འགྲོ་མཁན་རླངས་འཁོར་གཅིག་རེར་ལམ་ཁྲལ་རྒྱ་སྒོར་ ༡༠ རེ་དང་། སྐྱེལ་འདྲེན་བྱས་པའི་ཚོང་ཟོག་ཏུན་གཅིག་ལ་ཁྲལ་རྒྱ་སྒོར་ ༨ རེ་ལེན་པའི་སྲིད་བྱུས་གསར་པ་ཞིག་བཟོས་ཡོད་པ་དང་། འབྲེལ་ཡོད་ལས་བྱེད་ནས་ལམ་ཁྲལ་དང་ཚོང་ཟོག་གི་ཁྲལ་གསར་པ་དེ་དེ་རིང་ནས་ལག་བསྟར་བྱེད་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག སོག་རྒྱའི་འགག་སྒོ་དེ་ནས་གཙོ་བོ་སི་པིན་གྱི་(Rio Tinto Group)གཏེར་ལས་ཁང་གིས་ལྷོ་སོག་གི་(Oyu Tolgoi)ཟངས་ཀྱི་གཏེར་མཛོད་ནས་ཟངས་ཕྱིར་འདྲེན་བྱེད་བཞིན་ཡོད་པ་དང་། དེ་བཞིན་རྒྱ་ནག་ཧྲིན་ཧྭ་ཚོང་གཉེར་ཁང་གིས་འཛམ་གླིང་ཐོག་གི་རྡོ་སོལ་གཏེར་མཛོད་ཆེ་ཤོས་ལྷོ་སོག་གི་（Tavan Tolgoi）རྡོ་སོལ་ཡང་འགག་སྒོ་དེ་ནས་ཚོང་འདྲེན་བྱེད་ཀྱིན་ཡོད་པ་དང་། སོག་པོའི་རྒྱལ་ས་ཨོ་ལན་པཱ་ཐར་དུ་ཡོད་པའི་སོག་པོའི་གནད་དོན་དབྱེ་ཞིབ་པ་ཞིག་གིས། ད་ལྟའི་ཆར་ཉིན་རེར་སྐྱེལ་འདྲེན་རླངས་འཁོར་ ༩༠༠ ཙམ་གྱིས་འགོག་སྒོ་དེ་བརྒྱུད་ནས་འགྲོ་ཡི་ཡོད་པ་དང་། ཟླ་རེར་ཟངས་ཏུན་ཁྲི་ ༡༣.༣ རེ་རྒྱ་ནག་ནང་དབོར་འདྲེན་བྱེད་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག ཁ་སང་སོག་པོའི་གསར་ཁང་ཞིག་གིས་དེའི་སྐོར་ལ་བརྡ་ཁྱབ་ཅིག་སྤེལ་བའི་ནང་། ཟངས་གཏེར་ཏུན་གཅིག་རེར་རྒྱ་སྒོར་ཁྲི་གཅིག་ལས་བརྒལ་ཡོད་ལ། དེ་ཕྱིར་ཚོང་བྱེད་པར་སྤྱིའི་རིན་གོང་ལས་བརྒྱ་ཆ་ ༠.༢ ཁྲལ་དུ་འཇལ་དགོས་པ་དང་། ཁྲལ་བསྡུ་བའི་སྲིད་བྱུས་གསར་པ་དེ་ཕྱི་ཟླ་ ༡༢ ཚེས་ ༡ ཉིན་ནས་ལག་བསྟར་བྱ་རྒྱུ་ཡིན་སྐོར་བསྒྲགས་ཡོད་པ་རེད། བྷི་བྷི་སི་ཡིས། ཉེ་སྔོན་སོག་ཡུལ་མི་མང་གིས་ཏཱ་ལའི་བླ་མ་མཆོག་ཆོས་ཕྱོགས་ཀྱི་དབུ་འཁྲིད་ཅིག་ཏུ་ངོས་འཛིན་ཞུས་ཏེ་ཡུལ་དེའི་ནང་གདན་འདྲེན་ཞུས་ཡོད་ཀྱང་། རྒྱ་ནག་གཞུང་གིས་ཏཱ་ལའི་བླ་མ་མཆོག་ཉེན་ཁ་ཆེ་བའི་ཁ་བྲལ་བ་ཞིག་ཏུ་མཐོང་གི་ཡོད་པ་མ་ཟད། ཏཱ་ལའི་བླ་མ་མཆོག་སོག་ཡུལ་དུ་གདན་འདྲེན་ཞུ་སྐབས། རྒྱ་ནག་གཞུང་ཕྱོགས་ནས་ཏཱ་ལའི་བླ་མ་སོག་ཡུལ་དུ་གདན་འདྲེན་ཞུས་པ་དེས་རྒྱལ་ཁབ་གཉིས་བར་གྱི་འབྲེལ་བར་དོ་ཕོག་འགྲོ་རྒྱུ་ཡིན་སྟབས། འཕྲལ་དུ་རྩིས་མེད་གཏོང་དགོས་པའི་ཉེན་བརྡ་བཏང་ཡོད་པ་དེའི་སྐོར་བརྗོད་ཡོད་པ་མ་ཟད། ཕྱི་ཟླ་ ༡༡ ཚེས་ ༢༨ ཉིན་རྒྱ་ནག་དང་སོག་པོའི་བར་ཚོགས་འདུ་ཞིག་འཚོགས་འཆར་ཡོད་ཀྱང་རྒྱ་ནག་གཞུང་ཕྱོགས་ནས་རྩིས་མེད་བཏང་ཡོད་སྐོར་ཡང་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
