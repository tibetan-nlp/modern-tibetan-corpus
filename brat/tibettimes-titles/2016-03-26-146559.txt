གྲྭ་ཁྲི་བཙན་ལཊ་ཀློད་གྲོལ་བཏང་།

ཉེ་སྔོན་མགོ་ལོག་སྟོང་སྐྱབས་དགོན་པའི་དགེ་འདུན་པ་ཁྲི་བཙན་ལགས་བཙོན་འཇུག་དུས་ཡུན་རྫོགས་ནས་ཀློད་གྲོལ་བཏང་ཡོད་འདུག འབྲེལ་ཡོད་ཅིག་གིས་འདི་གར་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། མགོ་ལོག་སྟོང་སྐྱབས་དགོན་གྱི་དགེ་འདུན་པ་ཁྲི་བཙན་ནམ་སྨྱུག་མིང་དུ་ཁྲི་བོད་ཁྲག་ལགས། ཕྱི་ལོ་ ༢༠༡༣ ཟླ་ ༣ ཚེས་ ༡༡ ཉིན་རྒྱ་གཞུང་ཉེན་རྟོག་པས་མགོ་ལོག་ཁུལ་དགའ་བདེ་རྫོང་ཐོག་ནས་གློ་བུར་དུ་འཛིན་བཟུང་བཙན་ཁྲིད་བྱས་ཏེ། རྩོམ་ཡིག་བྲིས་ནས་སྤྱི་ཚོགས་ལ་འཁྲུག་རྐྱེན་བསླངས་བ་དང་རྒྱལ་ཁབ་ཁ་བྲལ་གཏོང་ཐབས་བྱས་པ་སོགས་ཀྱི་ཉེས་མིང་འོག་ཁོང་ལ་ཁྲིམས་ཐག་བཅད་ཡོད་པ་ལྟར། ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༩ ཉིན་བཙོན་འཇུག་དུས་ཡུན་རྫོགས་ཏེ་ཀློད་གྲོལ་བཏང་ཡོད་འདུག   རང་ལོ་ ༢༩ ཡིན་པའི་གྲྭ་ཁྲི་བཙན་ལགས་ནི། མཚོ་སྔོན་ཞིང་ཆེན་མགོ་ལོག་བོད་རིགས་རང་སྐྱོང་ཁུལ་དགའ་བདེ་རྫོང་རྡོ་ཁ་ཡུལ་ཚོའི་རྒྱ་སྐོར་སྨད་མ་སྡེ་བའི་མ་ཕག་མོ་ལགས་ཀྱི་བུ་ཡིན་པ་དང་། ཕྱི་ལོ་ ༢༠༡༣ ཟླ་ ༣ པའི་ཚེས་ ༨ ཉིན་ཁོང་གིས་བརྩམས་པའི་ <<བདེན་པའི་ཁ་རླངས>>ཞེས་པའི་དེབ་ཅིག་སྟོང་སྐྱབས་དགོན་པའི་ནང་དབུ་འབྱེད་འགྲེམས་སྤེལ་བྱས་ཡོད་པ་དང་།  དེབ་དེའི་ནང་སྔར་ལོའི་ཕྱི་ཟླ་ ༡ ཚེས་ ༨ ཉིན་མགོ་ལོག་དར་ལག་རྫོང་ཐོག་ཏུ། རྒྱ་གཞུང་གི་དྲག་གནོན་དབང་སྒྱུར་སྲིད་ཇུས་ལ་ཞི་བའི་ངོ་རྒོལ་གྱིས་རང་ལུས་མེར་བསྲེགས་གནང་སྟེ་སྐུ་གྲོངས་མཁན་སྟོང་སྐྱབས་དགོན་པའི་བླ་མ་སྤྲུལ་སྐུ་བསོད་བྷ་ལགས་ཀྱི་མཛད་རྣམ་དང་། དེ་བཞིན། བོད་མི་རིགས་ཀྱི་རང་དབང་བདེན་པའི་འཐབ་རྩོད་ཆེད་རང་ལུས་མེར་བསྲེགས་གནང་མཁན་རྒྱལ་གཅེས་དཔའ་བོ་དཔའ་མོ་རྣམས་ལ་རྗེས་དྲན་གདུང་འབོད་ཀྱི་རྩོམ་ཡིག་སོགས་བྲིས་ཡོད་འདུག   ཕྱི་ལོ་ ༢༠༡༣ ཟླ་ ༣ ཚེས་ ༡༥ ཉིན་དགའ་བདེ་རྫོང་གི་འབྲེལ་ཡོད་ལས་བྱེད་ཁག་ཅིག་སྟོང་སྐྱབས་དགོན་པའི་ནང་ཡོང་སྟེ། གྲྭ་ཁྲི་བཙན་ལགས་ཀྱིས་བརྩམས་པའི་༼བདེན་པའི་ཁ་རླངས༽ཞེས་པའི་དེབ་དེར་དཔལ་འབྱོར་གྱི་མཐུན་རྐྱེན་སྦྱོར་མཁན་གྲྭ་བ་རྣམས་རྩིས་སྤྲོད་དགོས་ལུགས་ཀྱི་ བཙན་བཀའ་བཏང་ཡོད་ཀྱང་། དགོན་པའི་གྲྭ་བཙུན་རྣམས་ཀྱིས་དང་ལེན་བྱས་མེད་སྟབས། ཕྱི་ཚེས་ ༡༧ ཉིན་དགའ་བདེ་རྫོང་གི་བདེ་སྲུང་མི་སྣ་དང་ལས་བྱེད་སོགས་མི་བརྒྱ་མ་ཟིན་ཙམ་དགོན་པའི་ནང་བསླེབས་ཏེ། ཁོ་ཚོ་དགོན་པ་གཞུང་གི་མགྲོན་ཁང་ནང་འདུག་འཇུག་དགོས་པ་དང་། དགོན་པའི་འདུ་ཁང་ནང་ཚོགས་འདུ་འཚོགས་འཇུག་དགོས་སྐོར་བརྗོད་ཡོད་ཀྱང་། དགོན་པའི་ངོས་ནས་འདུ་ཁང་ནང་ཚོགས་འདུ་བསྐང་རྒྱུ་དང་། དགོན་པ་གཞུང་གི་མགྲོན་ཁང་ནང་རྒྱ་གཞུང་གི་ལས་བྱེད་རུ་ཁག་སྡོད་འཇུག་རྒྱུ་དང་ལེན་བྱས་མེད་སྟབས། རྒྱ་གཞུང་གིས་དགོན་པའི་ར་སྐོར་ནང་ལས་བྱེད་རུ་ཁག་འདུག་སའི་སྡོད་གུར་མང་པོ་བརྒྱབ་པ་དང་། དགོན་པའི་འདུ་ཁང་གི་སྒོ་ཁར་དྲིལ་བསྒྲགས་སློབ་གསོའི་འཕྲེད་དར་འཕྱར་ཞིང་། འདུ་ཁང་གི་མདུན་གྱི་འཁྱམ་རའི་ཐང་དུ་གྲྭ་བ་རྣམས་ལ་རྒྱལ་གཅེས་ཆོས་གཅེས་ཀྱི་སློབ་གསོའི་ཚོགས་འདུ་བསྐངས་ཏེ། དགོན་པར་ཕྱོགས་ཡོངས་ནས་བཀག་སྡོམ་དམ་དྲག་ཆེན་པོ་བྱས་ཡོད་པ་རེད། ཕྱི་ལོ་ ༢༠༡༣ ལོར་ཤེས་རྟོགས་བྱུང་བ་ལྟར་ན། སྟོང་སྐྱབས་དགོན་པའི་ནང་རྒྱུན་བཞུགས་གྲྭ་པ་ ༢༠༠ ལྷག་དང་། རྒྱ་གཞུང་གི་ངོས་འཛིན་ལག་ཁྱེར་ཡོད་པའི་གྲྭ་པ་ ༡༠༠ ལྷག་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
