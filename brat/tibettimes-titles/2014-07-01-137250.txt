བདེ་སྲུང་དང་ཆོས་རིག་དྲུང་ཆེ་གསར་བསྐོ།

དེ་རིང་ཕྱི་ཟླ་ ༧ ཚེས་ ༡ ཉིན། དེ་སྔའི་བོད་ཀྱི་སྲིད་བྱུས་ཉམས་ཞིབ་ཁང་གི་ལས་བྱེད་འཕགས་པ་ཚེ་རིང་ལགས་བོད་མིའི་སྒྲིག་འཛུགས་བདེ་སྲུང་ལས་ཁུངས་ཀྱི་འཛིན་སྐྱོང་དྲུང་ཆེར་བསྐོ་བཞག་དང་། དེ་སྔའི་ནང་སྲིད་ལས་ཁུངས་ཀྱི་དྲང་འཕར་དོན་གྲུབ་རྡོ་རྗེ་ལགས་ཆོས་རིག་ལས་ཁུངས་ཀྱི་དྲུང་ཆེར་བསྐོ་བཞག་གནང་འདུག བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་གཞུང་འབྲེལ་དྲ་བའི་སྟེང་གོང་གསལ་དྲུང་ཆེ་རྣམ་གཉིས་ཀྱི་གནས་ཚུལ་འཁོད་པའི་ནང་། དྲུང་ཆེ་འཕགས་པ་ཚེ་རིང་ལགས་ནི། ཕྱི་ལོ་ ༡༩༩༣ ཟླ་ ༤ ཚེས་ ༩ ཉིན་ནས་ཚོད་ལྟའི་ཐོག་དྲུང་ལས་གནས་རིམ་དུ་བསྐོ་གཞག་གནང་། ཕྱི་ལོ་ ༡༩༩༣ ཟླ་ ༦ ཚེས་ ༡༥ ནས་བདེ་སྲུང་བོད་གནས་ཉམས་ཞིབ་ཁང་དུ་གནས་རིམ་རང་འཇགས་ཐོག་བསྐོ་གཞག་གནང་། ཕྱི་ལོ་ ༡༩༩༤ ཟླ་ ༤ ཚེས་ ༢༩ ནས་གཞུང་ཞབས་ངོ་མར་བསྐོ་གཞག་གནང་། ཕྱི་ལོ་ ༡༩༩༧ ཟླ་ ༥ ཚེས་ ༦ ནས་དྲུང་གཞོན་གནས་རིམ་དུ་གནས་སྤར། ཕྱི་ལོ་ ༡༩༩༨ ཟླ་ ༣ ཚེས་ ༡༧ ནས་བདེ་སྲུང་བོད་གནས་ཉམས་ཞིབ་ཁང་གི་དྲུང་གཞོན་ལས་འགན་རང་འཇགས། ཕྱི་ལོ་ ༡༩༩༨ ཟླ་ ༡༠ ཚེས་ ༢༦ ནས་ལོ་གཉིས་རིང་ཕོགས་མེད་སློབ་སྦྱོང་དུ་བསྐྱོད། ཕྱི་ལོ་ ༢༠༠༠ ཟླ་ ༨ ཚེས་ ༡༦ ནས་བདེ་སྲུང་བོད་གནས་ཉམས་ཞིབ་ཁང་གི་དྲུང་གཞོན། ཕྱི་ལོ་ ༢༠༠༢ ཟླ་ ༧ ཚེས་ ༤ ནས་ཟུང་དྲུང་དུ་གནས་སྤར་གྱིས་བདེ་སྲུང་བོད་གནས་ཉམས་ཞིབ་ཁང་གི་ཟུང་དྲུང་དུ་ བསྐོ་གཞག་གནང་། ཕྱི་ལོ་ ༢༠༡༠ ཟླ་ ༡ ཚེས་ ༡༨ ནས་ཞབས་ལོའི་བབ་ཀྱི་དྲུང་ཆེ་ལས་འཕར་གྱི་གནས་རིམ་དུ་གནས་སྤར་གནང་། ཕྱི་ལོ་ ༢༠༡༠ ཟླ་ ༡༠ ཚེས་ ༢༥ ནས་བདེ་སྲུང་ལས་ཁུངས་སུ་གནས་སྤོས་བཀོད་སྒྲིག་གནང་། ཕྱི་ལོ་ ༢༠༡༤ ཟླ་ ༦ ཚེས་ ༡༢ ནས་དྲུང་ཡིག་ཆེན་མོའི་གནས་རིམ་དུ་གནས་སྤར་གནང་། ཕྱི་ལོ་ ༢༠༡༤ ཟླ་ ༧ ཚེས་ ༡ ཉིན་ནས་བདེ་སྲུང་ལས་ཁུངས་ཀྱི་འཛིན་སྐྱོང་དྲུང་ཡིག་ཆེན་མོའི་ལས་གནས་ཐོག་ཕྱག་ལས་གནང་མགོ་བཙུགས་འདུག དྲུང་ཆེ་དོན་གྲུབ་རྡོ་རྗེ་ལགས་ནི། ཕྱི་ལོ་ ༡༩༨༣ ཟླ་ ༣ ཚེས་ ༡༥ ཉིན་ལས་བྱེད་དུ་ཞུགས་ཏེ་གནས་དགུ་པ་ཚོད་ལྟའི་ཐོག་ཏི་ཛུ་སྐད་དྲུང་དུ་བསྐོ་གཞག་གནང་། ༡༩༨༦ ཕྱི་ཟླ་ ༡ ཚེས་ ༢ ཉིན་ནས་ལས་འཁུར་རང་འཇགས་ཐོག་ལས་ལས་བྱེད་ངོ་མར་བསྐོ་གཞག་གནང་། ༡༩༨༨ ཟླ་ ༣ ཚེས་ ༡ ཉིན་ནས་གནས་རིམ་རང་འཇགས་ཐོག་བདེ་སླར་གཞིས་མགོའི་ལས་ཁུངས་སུ་གནས་སྤོས། ༡༩༩༡ ཟླ་ ༡ ཚེས་ ༡༥ ནས་གནས་བརྒྱད་པར་གནས་སྤར་གྱིས་ཀོ་ལི་གྷལ་མཉམ་དྲུང་དུ་བསྐོ་གཞག་གནང་། ༡༩༩༢ ཟླ་ ༨ ཚེས་ ༡ ནས་སྡེ་ཚན་འགན་འཛིན་དུ་བསྐོ་གཞག ༡༩༩༣ ཟླ་ ༧ ཚེས་ ༡ གནས་རང་འཇགས་ཐོག་མོན་གྷོ་མཉམ་དྲུང་དུ་བསྐོ་གཞག ༡༩༩༣ ཟླ་ ༡༢ ཚེས་ ༦ དྲུང་ལས་གནས་རིམ་དུ་གནས་སྤར། ༡༩༩༤ ཟླ་ ༡ ཚེས་ ༡༥ ལས་འགན་རང་འཇགས་བསྐོ་གཞག ཕྱི་ལོ་ ༡༩༩༥ ཟླ་ ༨ ཚེས་ ༡ ཨ་རིར་སློབ་བསྐྱོད། ༡༩༩༦ ཟླ་ ༡༠ ཚེས་ ༡༡ ཉིན་ནས་བོད་ཀྱི་ཟློ་གར་ཡིག་ཚང་གི་ལས་དྲུང་། ༡༩༩༧ ཟླ་ ༥ ཚེས་ ༡ ནས་མོན་གྷོ་མཉམ་དྲུང་དུ་བསྐོ་གཞག ༡༩༩༨ ཟླ་ ༢ ཚེས་ ༢༣ ནས་མེའོ་ཆོས་འཕེལ་གླིང་གི་གཞིས་མགོར་བསྐོ་གཞག ༡༩༩༩ ཟླ་ ༦ ཚེས་ ༢༥ ནས་གནས་རང་འཇགས་ཐོག་ལུགས་བསམ་ས་གནས་མགོ་འཛིན་ལས་ཁང་གི་ས་གནས་འགོ་འཛིན། ༢༠༠༡ ཟླ་ ༩ ཚེས་ ༥ ནས་གནས་རིམ་རང་འཇགས་ཐོག་ལྡི་ལི་དོན་ཁང་གི་དྲུང་ཆེར་བསྐོ་གཞག ༢༠༠༢ ཟླ་ ༡ ཚེས་ ༡༦ ནས་ལ་དྭགས་གཞིས་སྤྱི་རུ་བསྐོ་གཞག ༢༠༠༢ ཟླ་ ༧ ཚེས་ ༡ ནས་ཟུང་དྲུང་གནས་རིམ་དུ་གནས་སྤར་གྱིས་ལས་འགན་རང་འཇགས། ༢༠༠༣ ཟླ་ ༩ ཚེས་ ༡༨ ནས་མོན་གྷོ་འདོད་རྒུ་གླིང་ས་གནས་མགོ་འཛིན་དུ་གནས་སྤོས།  ༢༠༠༧ ཟླ་ ༦ ཚེས་ ༦ ནས་སྒང་ཏོག་ས་གནས་མགོ་འཛིན་དུ་བསྐོ་གཞག ༢༠༡༠ ཟླ་ ༡ ཚེས་ ༡༥ ནས་དྲུང་ཆེ་ལས་འཕར་དུ་གནས་སྤར། ༢༠༡༡ ཟླ་ ༡ ཚེས་ ༣ ནས་ནང་སྲིད་ལས་ཁུངས་སུ་གནས་སྤོས། ༢༠༡༤ ཟླ་ ༦ ཚེས་ ༡༡ ནས་དྲུང་ཡིག་ཆེན་མོའི་གནས་རིམ་དུ་གནས་སྤར། ༢༠༡༤ ཟླ་ ༧ ཚེས་ ༡ ནས་ཆོས་རིག་ལས་ཁུངས་ཀྱི་དྲུང་ཡིག་ཆེན་མོར་བསྐོ་གཞག་གནང་ཡོད་འདུག
