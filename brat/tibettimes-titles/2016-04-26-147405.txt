ཕྱི་རྒྱལ་སྐུ་ཚབ་བཀག་འགོག་བྱས།

པེ་ཅིན་གཞུང་གིས་མི་དཀྱུས་མའི་གྱོན་ཆས་སྤྲས་པའི་ཉེན་རྟོག་དམག་མི་མངགས་ཏེ་ཕྱིའི་རྒྱལ་ཁབ་ཁག་གི་སྐུ་ཚབ་ཚོ་རྒྱ་ནང་དུ་བཀག་བསྐྱིལ་བྱས་པའི་བཙོན་མ་ཚོར་ཐུག་འཕྲད་བྱེད་དུ་བཅུག་མི་འདུག ཁ་སང་ཧོང་ཀོང་རང་དབང་གསར་ཁང་ (hkfp) གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཧྥ་རན་སི་དང་འཇར་མན། ཁེ་ན་ཌ་དང་སུད་སི། ཡུ་རོབ་མཐུན་ཚོགས་བཅས་ཀྱི་སྐུ་ཚབ་ཚོགས་ཆུང་ཞིག་ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༣ ཉིན་རྒྱ་ནག་པེ་ཅིན་དུ་བསྐྱོད་དེ་ཕྱི་ལོ་ ༢༠༡༣ ལོའི་ཕྱི་ཟླ་ ༤ བ་ནས་པེ་ཅིན་གྲོང་ཁྱེར་གྱི་གླས་ཤག་ཏུ་བཀག་བསྐྱིལ་བྱས་ཡོད་པའི་རྒྱ་ནག་གི་སྤྱི་མང་ཐོབ་ཐང་རྩོད་ལེན་པ་རྒྱ་རིགས་སྐྱེས་མ་ནེ་ཡུ་ལན་ (Ni Yulan) ཟེར་བ་ཞིག་ལ་ལྟ་སྐོར་གནང་སྐབས། མི་སེར་དཀྱུས་མའི་ལྭ་བ་གྱོན་པའི་རྒྱ་ནག་གི་ཉེན་རྟོག་དམག་མིས་ཁོ་ཚོ་ལྕམ་ནེ་ཡ་ལན་ལ་ཐུག་འཕྲད་བྱེད་དུ་བཅུག་མེད་སྟབས། སྐུ་ཚབ་ཚོགས་ཆུང་དེས་ལྕམ་ནེ་ཡུ་ལན་ལ་ཁྱེར་བའི་ཟ་མ་སོགས་ཁོ་མོའི་བཟའ་ཟླའི་ལག་ཏུ་སྤྲད་ཡོད་འདུག ལྕམ་ནེ་ཡུ་ལན་ལགས་ཀྱིས་ཨེ་ཕེ་གསར་ཁང་གི་གསར་འགོད་པར་བརྗོད་པའི་གནས་ཚུལ་ནང་། ཁོང་མོ་ཕྱི་ལོ་ ༢༠༡༣ ལོའི་ཕྱི་ཟླ་ ༤ བའི་ནང་པེ་ཅིན་ཞི་ཁྲང་ (Xicheng) ཉེན་རྟོག་ཚན་པས་འཛིན་བཟུང་བཀག་བསྐྱིལ་བྱས་པ་དང་། དེ་རིང་ཕྱི་དྲོའི་ཆུ་ཚོད་ ༣ སྟེང་ཕྱིའི་རྒྱལ་ཁབ་ལྔ་ཡི་སྐུ་ཚབ་ཞི་ཁྲང་ས་ཁུལ་དུ་ང་ལ་ཐུག་འཕྲད་བྱེད་དུ་ཡོང་སྐབས། གྱོན་ཆས་དཀྱུས་མ་སྤྲས་པའི་ཉེན་རྟོག་དམག་མིས་ཁོ་ཚོ་བཀག་འགོག་བྱས་ཡོད་པ་རེད། དེ་བས་ངའི་བཟའ་ཟླ་ཕྱི་ལོགས་སུ་སྐུ་ཚབ་ཚོར་ཕེབས་བསུ་བྱེད་དུ་སོང་མོད། གྱོན་ཆས་དཀྱུས་མ་སྤྲས་པའི་ཉེན་རྟོག་པ་ཚོས་ཁོ་ཚོ་ངའི་ནང་དུ་ཡོང་བཅུག་མེད་པ་མ་ཟད། ད་དུང་ང་ཚོ་འདི་གར་ཁྲིམས་འགལ་གྱི་སྒོ་ནས་འཚོ་སྡོད་བྱས་ཡོད་ལུགས་བཤད་འདུག ང་ཚོས་འདི་གར་གླ་རིན་མཐོན་པོ་སྤྲད་ནས་སྡོད་ཁང་གླས་ཡོད། དེ་ཁྲིམས་འགལ་ཇི་ལྟར་ཡིན་ནམ། ཞེས་སོགས་བརྗོད་འདུག རྒྱལ་སྤྱིའི་ཉེ་ཡང་ཚོགས་པའི་ཞིབ་འཇུག་པ་ཕོན་ (Poon) ཟེར་བ་ཞིག་གིས། ཨ་རིའི་གཞུང་གིས་ལྕམ་ནེ་ལགས་ལ་གཟེངས་རྟགས་ཕུལ་རྗེས། རྒྱ་ནག་གཞུང་གིས་ལྕམ་ནེ་དང་ཁོང་མོའི་བཟའ་ཟླ་གཉིས་ལ་ཐེངས་མང་པོར་བརྙས་གཙེར་དང་བརྡབ་གསིག་གཏོང་བཞིན་ཡོད་འདུག རྒྱ་རིགས་འགྲོ་བ་མིའི་ཐོབ་ཐང་དང་ཁྲིམས་རྩོད་པ་མང་པོའི་གྲས་ནས་ལྕམ་ནེ་ལགས་ནི་རྒྱ་ནག་གཞུང་གི་བརྙས་གཙེར་དང་བརྡབ་གསིག་ལ་མ་བལྟས་པར་རྒྱ་ནང་གི་འགྲོ་བ་མིའི་ཐོབ་ཐང་རྩོད་ལེན་གནང་མཁན་ཞིག་རེད། ང་ཚོས་རྒྱལ་སྤྱིའི་སྤྱི་ཚོགས་དང་སྲིད་གཞུང་ཁག་ལ་ཞུ་རྒྱུར། རྒྱ་ནག་གཞུང་གིས་ལྕམ་ནེ་ལགས་དང་ཁོང་མིའི་ནང་མིར་བརྙས་གཙེར་དང་བརྡབ་གསིག་གཏོང་རྒྱུ་མཚམས་འཇོག་ཐབས་སུ་རྒྱ་ནག་གཞུང་ལ་གནོན་ཤུགས་སྤྲོད་དགོས་པའི་རེ་སྐུལ་ཞུ་ཡི་ཡོད། ཅེས་བརྗོད་འདུག གནས་ཚུལ་ལྟར་ན། རྒྱ་རིགས་སྐྱེས་མ་ནེ་ཡུ་ལན་ལགས་ནི་རྒྱ་ནང་གི་འགྲོ་བ་མིའི་ཐོབ་ཐང་རྩོད་ལེན་གནང་མཁན་ཞིག་ཡིན་པ་དང་། ཉེ་ཆར་ཨ་རིའི་གཞུང་གིས་ལྕམ་ནེ་ཡུ་ལན་ལགས་ལ་རྒྱལ་སྤྱིའི་སྙིང་སྟོབས་ཅན་གྱི་བུད་མེད་ (International Women of Courage awards) ཅེས་པའི་གཟེངས་རྟགས་ཤིག་ཕུལ་ཡོད་ནའང་། རྒྱ་ནག་གཞུང་གིས་ལྕམ་ནེ་ཡུ་ལན་ལགས་པེ་ཅིན་གྲོང་ཁྱེར་གྱི་ཁོང་མོའི་གླས་ཤག་ཏུ་བཀག་བསྐྱིལ་བྱས་ཏེ་ཨ་རི་ལ་འགྲོ་རུ་བཅུག་མེད་པ་དང་། གཟེངས་རྟགས་དེ་ནི་ཁོང་མོས་རྒྱ་ནག་གཞུང་གི་དྲག་གནོན་དང་དམ་དྲག་འོག་རྒྱ་ནང་གི་འགྲོ་བ་མིའི་ཐོབ་ཐང་རྩོད་ལེན་བྱེད་བཞིན་པར་ཕུལ་བ་ཞིག་ཡིན་ཞིང་། ཁོང་མོ་རྒྱལ་སྤྱིའི་སྙིང་སྟོབས་ཅན་གྱི་བུད་མེད་ཅེས་པའི་གཟེངས་རྟགས་དེ་བཞེས་མཁན་ཨང་ ༡༤ བ་ཡིན་པ་དང་། རྒྱལ་སྤྱིའི་སྙིང་སྟོབས་ཅན་གྱི་བུད་མེད་ཅེས་པའི་གཟེངས་རྟགས་དེ་ནི་འཛམ་གླིང་སྟེང་དྲང་བདེན་དང་འགྲོ་བ་མིའི་ཐོབ་ཐང་། དམངས་གཙོ་དང་ཕོ་མི་འདྲ་མཉམ་བཅས་ཀྱི་ཁེ་དབང་དང་ཐོབ་ཐང་རྩོད་ལེན་གནང་མཁན་སྐྱེས་མ་མཛད་རྗེས་ཅན་དག་ལ་སྩོལ་བཞིན་ཡོད་པ་ཞིག་ཡིན་པ་དང་། ལོ་ཤས་གོང་ལ་རྒྱལ་སྤྱིའི་སྙིང་སྟོབས་ཅན་གྱི་བུད་མེད་ཅེས་པའི་གཟེངས་རྟགས་དེ་བཞིན་བོད་དོན་འཐབ་རྩོད་པ་དང་རྩོམ་པ་པོ་ཚེ་རིང་འོད་ཟེར་ལགས་སུའང་ཕུལ་མྱོང་ཡོད་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
