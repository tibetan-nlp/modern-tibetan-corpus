
བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་ལས་བྱེད་རྩ་དགོངས་ཞུ་མཁན་ ༡༤ བྱུང་འདུག

བོད་མིའི་སྒྲིག་འཛུགས་དབུས་ས་གཉིས་སུ་གཞུང་ཞབས་ལས་བྱེད་ཁྱོན་ ༥༢༦ ཡོད་པའི་ཁྲོད་ནས་འདས་མ་ཐག་གི་ ༢༠༡༦ ལོར་ལས་བྱེད་རྩ་དགོངས་ཞུ་མཁན་ ༡༤ དང་བགྲེས་ཡོལ་དང་བགྲེས་ཡོལ་སྔ་བསྣུར་ཞུ་མཁན་ ༡༠ བྱུང་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༣༠ ཉིན་ནས་འཆར་ཟླ་ ༣ ཚེས་ ༢༤ ཉིན་བར། བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་ལས་བྱེད་འདེམས་བསྐོ་ལྷན་ཁང་གི་ངོས་ནས། བཞུགས་སྒར་ཉེ་འཁྲིས་ཀྱི་དབུས་སྣེ་ལེན་ཁང་གི་ཚོགས་ཁང་དུ། འདི་ལོའི་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་སྲི་ཞུའི་ལས་བྱེད་གསར་བ་ ༡༣ དང་ད་ལྟ་སྲི་ཞུ་སྒྲུབ་བཞིན་པའི་ལས་བྱེད་ ༡༩ བཅས་ཁྱོན་ལས་བྱེད་ ༣༢ ཙམ་ལ་ལས་དོན་བྱེད་ཕྱོགས་སྐོར་ལ་ཟབ་སྦྱོང་ཞིག་སྤྲོད་བཞིན་ཡོད་པ་ལྟར། འདི་གའི་གསར་འགོད་པས་ལས་བྱེད་འདེམས་ལྷན་གྱི་དྲུང་ཚབ་འཇིགས་མེད་ཚེ་རིང་ལགས་སུ་བཅར་འདྲི་ཞུས་པར། ཁོང་གིས་ཐེངས་འདིའི་ཟབ་སྦྱོང་ཁྲོད་ཉེ་ཆར་གསར་འདེམས་བྱུང་བའི་ལས་བྱེད་ ༡༣ ཡོད་པའི་ཁྲོད། ཆོད་གན་ལས་བྱེད་གཅིག་འཕུད་པའི་གཞན་ཚང་མ་མཐོ་སློབ་མཐར་ཕྱིན་ཏེ་ཡོང་བའི་སློབ་མ་རྐྱང་རྐྱང་ཡིན་སྐོར་བརྗོད་སོང་། འཇིགས་མེད་ཚེ་རིང་ལགས་སུ། འདས་མ་ཐག་གི་ ༢༠༡༦ ལོར་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་ལས་བྱེད་རྩ་དགོངས་སུ་འགྲོ་མཁན་ག་ཚོད་བྱུང་ཡོད་པ་དང་། རྩ་དགོངས་སུ་འགྲོ་མཁན་དེ་དག་ཁྲོད་རྒན་ཡོལ་རྩ་དགོངས་ཀྱི་གནས་སྟངས་གང་འདྲ་ཆགས་ཡོད་མེད་དྲིས་པར་ཁོང་གིས། ཕྱི་ལོ་ ༢༠༡༦ ཟླ་ ༤ ཚེས་ ༡ ནས་ཕྱི་ལོ་ ༢༠༡༧ ཟླ་ ༢ ཚེས་ ༡ བར་སྲི་ཞུའི་ལས་བྱེད་རྩ་དགོངས་ཞུ་མཁན་ ༡༤ དང་། ལས་བྱེད་བགྲེས་ཡོལ་དང་བགྲེས་ཡོལ་སྔ་བསྣུར་ཞུ་མཁན་ ༡༠ བྱུང་ཡོད་པ་རེད། ཅེས་བརྗོད་སོང་། ད་ལྟའི་ཆར་བོད་མིའི་སྒྲིག་འཛུགས་ལ་ལས་བྱེད་ས་མིག་སྟོང་པ་ག་ཚོད་ཡོད་མེད་སྐོར་ལ་ཁོང་གིས། བོད་མིའི་སྒྲིག་འཛུགས་ལ་ལས་བྱེད་ས་མིག་ཁྱོན་ ༦༧༤ ཡོད་པ་དང་། ད་ལྟ་ས་མིག་སྟོང་པ་ ༦༠ ནས་ ༧༠ བར་ཡོད་སྐོར་བརྗོད་བྱུང་། བོད་མིའི་སྒྲིག་འཛུགས་དབུས་ས་གཉིས་སུ་གཞུང་ཞབས་འདེམས་ལྷན་བརྒྱུད་ནས་བསྐོ་བཞག་གནང་བའི་ལས་བྱེད་ཁྱོན་ག་ཚོད་ཡོད་མེད་སྐོར་ལ། འདེམས་བསྐོ་ལྷན་ཁང་བརྒྱུད་དེ་བསྐོ་བཞག་བྱས་པའི་སྲི་ཞུ་བ་ཁྱོན་བསྡོམས་ ༥༢༦ ཙམ་ཡོད་དང་། གངས་ཅན་སྐྱིད་གཞུང་རང་དུ་རྩིས་ཞིབ་ལས་བྱེད་ཆ་ཚང་ཚུད་པའི་ལས་བྱེད་གྲངས་ ༣༧༢ ཡོད་ཅེས་བརྗོད་བྱུང་། ཆོད་གན་ལས་བྱེད་ ༣༢ ཡོད་པ་དང་། ཆོད་གན་དེ་དག་འབྲེལ་ཡོད་ལས་ཁུངས་ཁག་ནས་གསལ་བསྒྲགས་གནང་སྟེ་ཡིག་རྒྱུགས་བཏང་རྗེས། བཀའ་དྲུང་ཡིག་ཚང་བརྒྱུད་དེ་ལས་བྱེད་འདེམས་བསྐོ་ལྷན་ཁང་གིས་མཐའ་མའི་བསྐོ་བཞག་བཀའ་འཁྲོལ་གནང་གི་ཡོད་པ་རེད། དེ་མ་གཏོགས་གཞན་ཐད་བསྐོ་ལས་བྱེད་དང་རང་ཁོངས་ལས་བྱེད། དེ་བཞིན། དྭང་བླངས་ལས་བྱེད་སོགས་ལས་བྱེད་འདེམས་ལྷན་བརྒྱུད་ནས་བསྐོ་བཞག་གནང་གི་ཡོད་པ་མ་རེད། ཅེས་བརྗོད་སོང་། མཇུག་ཏུ་གསར་འགོད་པས། དེ་སྔ་བཀའ་ཤག་སྐབས་བཅུ་བཞི་པའི་གསར་གཏོད་གནང་བའི་གཞུང་ཞབས་ནང་ཚུད་རྒྱུགས་གཏོང་མཁན་རྣམས་ལ་ས་ར་མཐོ་སློབ་ཏུ་ཟླ་བ་ལྔ་དྲུག་རིང་ཟབ་སྦྱོང་སྤྲད་དེ། མཐར་ཡིག་རྒྱུགས་ཀྱི་ལམ་ནས་གཞུང་ཞབས་ནང་བསྡུ་བཞིན་ཡོད་པའི་ལས་འཆར་དེ་འགྲོ་ཡི་ཡོད་མེད་དྲིས་པར། ཁོང་གིས་ས་ར་མཐོ་སློབ་ཏུ་ཟླ་དྲུག་རིང་ཟབ་སྦྱོང་སྤྲོད་པ་དེ་འཛིན་སྐྱོང་བཀའ་ཤག་ནས་གོ་སྒྲིག་གནང་གི་ཡོད་སྟབས། དེའི་སྐོར་གསལ་བོ་ཞུ་ཐུབ་མ་སོང་ཞེས་བརྗོད་སོང་། གསར་འགོད་པ། པདྨ་རྒྱལ།
