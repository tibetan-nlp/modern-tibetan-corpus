ཆོས་དབྱངས་སྐྱིད་ཨང་ལྔ་པ་ཆགས།

བོད་པའི་བུད་མོ་ཆོས་དབྱངས་སྐྱིད་ལགས་བྷར་རྫིལ་ཨོ་རྩེད་འགྲན་ཚོགས་སྟེང་སྤྱི་ལེ་ ༢༠ ཅན་གྱི་སྐྱེས་མའི་འདུར་རྒྱུག་འགྲན་བསྡུར་ཁྲོད་ཨང་ལྔ་པ་ཐོབ། དོ་ནུབ་རྒྱ་གར་གྱི་མཚན་མོའི་ཆུ་ཚོད་ ༡༡།༠༠ པའི་ཐོག ལྷོ་ཨ་མི་རི་ཁའི་བྷར་རྫིལ་(Brazil)རྒྱལ་ཁབ་ཏུ་འཚོགས་བཞིན་པའི་རྒྱལ་སྤྱིའི་ཨོ་ལིམ་ཕིག་གི་རྩེད་འགྲན་སྟེང་། སྤྱི་ལེ་༢༠ ཅན་གྱི་བུད་མེད་འདུར་རྒྱུག་འགྲན་བསྡུར་ཁྲོད། རྒྱ་ནག་གི་རྩེད་འགྲན་རུ་ཁག་གི་མིང་ཐོག་ནས་མཉམ་ཞུགས་གནང་ཡོད་པའི་བོད་པའི་བུ་མོ་ཆོས་དབྱངས་སྐྱིད་ལགས་ཨང་ལྔ་པ་ཆགས་ཡོད་པ་རེད། ཐེངས་འདིར་ཁྱོན་འགྲན་ཞུགས་གནང་མཁན་ ༦༣ ཡོད་པའི་ཁྲོད་ནས་ཨང་དང་པོ་རྒྱ་ནག་གི་བུ་མོ་ལེའུ་ཧོང་(Liu Hong)ཞུ་བ་དུས་ཡུན་(1:28:35)དང་། ཨང་གཉིས་པ་མེག་སི་ཁོའོ་བུ་མོ་ (Maria Guadalupe Gonzalez)ཞུ་བ་དུས་ཡུན་(1:28:37)། ཨང་གསུམ་པ་རྒྱ་ནག་གི་བུ་མོ་(Lu Xiuzhi)་ཞུ་བ་དུས་ཡུན་(1:28:42)། ཨང་བཞི་པ་ཨི་ཊ་ལིའི་བུ་མོ་(Antonella Palmisano)ཞུ་བ་དུས་ཡུན་(1:29:03)། ཨང་ལྔ་པ་བོད་པའི་བུ་མོ་ཆོས་དབྱངས་སྐྱིད་ལགས་དུས་ཡུན་(1:29:04)བཅས་ཆགས་ཏེ། ཨང་དང་པོ་ལེའུ་དང་ཨང་ལྔ་པ་ཆོས་དབྱངས་སྐྱིད་ཀྱི་བར་སྐར་ཆ་ ༢༩ ཙམ་གྱི་ཁྱད་པར་ཤོར་ཡོད་པ་རེད། དེ་སྔོན་ ༢༠༡༢ ལོའི་དབྱིན་ཇུ་ལོན་ཌོན་ཨོ་རྩེད་འདུར་རྒྱུག་འགྲན་བསྡུར་ཁྲོད་ཐོག་མ་བོད་པའི་བུ་མོ་ཆོས་དབྱངས་སྐྱིད་ལགས་སུ་ཨང་གསུམ་པ་ཐོབ་པར་ངོས་འཛིན་གྱིས་ཟངས་ཀྱི་རྟགས་མ་སྤྲད་ཡོད་ཀྱང་། སྐབས་དེར་ཨང་གཉིས་པ་འཐོབ་མཁན་ར་ཤི་ཡའི་བུ་མོ་(Kaniskina)ཞུ་བ་དེས་སྨན་རྫས་བེད་སྤྱོད་བཏང་ཡོད་པའི་རྙོག་གླེང་བྱུང་སྟེ། རྗེས་ལ་ཆོས་དབྱངས་སྐྱིད་བསྐྱར་དུ་ཨང་གཉིས་པར་ངོས་འཛིན་བྱུང་ཡོད་པ་རེད། སྐབས་དེར་ཨང་གསུམ་པ་ཐོབ་མཁན་རྒྱ་མོ་(Hong Liu)ཞུ་བ་དེར་ད་ལོར་ཨང་དང་པོ་ཐོབ་ཡོད་པ་རེད། བོད་རིགས་བུ་མོ་ཆོས་དབྱངས་སྐྱིད་ལགས་ནི། ཕྱི་ལོ་ ༡༩༩༠ ཟླ་ ༡༡ ཚེས་ ༡༡ ཉིན་ཨ་མདོ་མཚོ་སྔོན་མཚོ་བྱང་ཁུལ་འདབ་བཞི་རྫོང་ཝ་མགྱོགས་སྡེ་བ་རུ་སྐྱེས་ཤིང༌། ཁོང་མོ་ནི་རྒྱལ་སྤྱིའི་ཨོ་ལིམ་ཕིག་རྩེད་འགྲན་སྟེང་ཞུགས་མཁན་བོད་རིགས་བུ་མོ་ཐོག་མ་དེ་ཆགས་ཡོད་པ་རེད། ཟླ་འདིའི་ཚེས་ ༢༡ ཉིན་བོད་པའི་ཕོ་གསར་སྟོབས་རྒྱལ་ལགས་ཨོ་ལིམ་ཕིག་རྩེད་འགྲན་སྟེང་རྨིད་ ༥༠༠༠ ཅན་གྱི་སྐྱེས་པའི་རྒྱང་རྒྱུག་འགྲན་བསྡུར་ཁྲོད་ཞུགས་རྒྱུ་ཡིན་འདུག སྟོབས་རྒྱལ་ལགས་ནི་ ༡༩༩༤ ལོའི་ཟླ་ ༢ པའི་ཚེས་ ༡༦ ཉིན་ལྷ་ས་སྟག་རྩེ་རྫོང་དར་རྒྱས་ཤིང་དུ་སྐྱེས་ཤིང་། ༢༠༡༤ ལོར་ལྷོ་ཀོ་རི་ཡར་ཚོགས་པའི་ཨེ་ཤ་ཡའི་རྩེད་འགྲན་སྟེང་རྨིད་ ༥༠༠༠ ཅན་གྱི་སྐྱེས་པའི་རྒྱང་རྒྱུག་འགྲན་བསྡུར་ཁྲོད་ཨང་ ༡༢ པ་དང་། ༢༠༡༥ ལོར་རྒྱ་ནག་ཨིའུ་ཧྭན་དུ་ཚོགས་པའི་ཨེ་ཤ་ཡའི་རྒྱང་རྒྱུག་འགྲན་བསྡུར་ཁྲོད་ཨང་ ༥ པ་བཅས་ཐོབ་མྱོང་ཡོད་པ་རེད།
