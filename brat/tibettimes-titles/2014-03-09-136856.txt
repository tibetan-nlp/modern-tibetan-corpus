གྲྭ་པ་དམར་གསོད་དང་འཛིན་བཟུང་།

http://tibettimes.net/admin/articles.php?&page=edit&id=8618 རྒྱ་གཞུང་གིས་ཆབ་མདོ་ཝོན་པ་དགོན་པའི་གྲྭ་བ་ཞིག་ལ་གཅར་རྡུང་བཏང་སྟེ་དམར་གསོད་བཏང་ཡོད་པ་མ་ཟད། གྲོང་གསར་དགོན་པའི་དབུ་མཛད་རྒན་པ་འཇུ་བཟུང་བཙན་འཁྲིད་བྱས་ཡོད་འདུག ཟླ་སྔོན་མའི་ཕྱི་ཚེས་ ༢༨ ཉིན་གྱི་ཕྱི་དྲོའི་ཆུ་ཚོད་ ༣།༠༠ ཡས་མས་སུ་རྒྱ་གཞུང་གིས་ཆབ་མདོ་ཁུལ་ཝོན་པ་དགོན་པའི་གྲྭ་བཀྲ་ཤིས་དཔལ་འབྱོར་ལགས། ཉེན་རྟོག་པས་ས་གནས་སུ་སྔོག་གཤེར་བྱེད་སྐབས་ཁོང་གི་སྡོད་གནས་སུ་སྲིད་སྐྱོང་བློ་བཟང་སེང་གེ་མཆོག་གི་གཏམ་བཤད་འོད་སྡེར་དང་། ༧གོང་ས་མཆོག་གི་བཀའ་སློབ་ཀྱི་དཔེ་དེབ་བཅས་རྙེད་སོན་བྱུང་ཡོད་ཚུལ་བཤད་དེ་བཙན་འཁྲིད་བྱས་ཡོད་འདུག ཕྱི་ཟླ་ ༣ ཚེས་ ༡ ཉིན་ཉེན་རྟོག་པས་ཁོང་ལ་གཅར་རྡུང་ཚད་མེད་བཏང་སྟེ་ལུས་པོ་ཡོངས་སུ་རྨ་བཟོས་པ་དང་ཁ་གྲགས་མི་ཐུབ་པའི་གནས་སུ་བསྐྱལ་རྗེས་ཕྱིར་ནང་མིར་རྩིས་སྤྲོད་བྱས་ཡོད་པ་མ་ཟད། ཉེན་རྟོག་པས་ནང་མིར་ཁོང་གི་རང་སྲོག་གཅོད་རྩིས་བྱས་ཏེ་གནས་སྟངས་དེ་འདྲ་ཆགས་ཡོད་སྐོར་བརྗོད་ཡོད་འདུག རྗེས་སུ་ནང་མི་ཚོས་མགྱོགས་མྱུར་ངང་སྨན་ཁང་དུ་འཁྲིད་ཡོད་ཀྱང་། ཉིན་དེའི་ཕྱི་དྲོའི་ཆུ་ཚོད་ ༣ ཡས་མས་སུ་སྨན་ཁང་ལ་མ་བསླེབ་པའི་ལམ་བར་དུ་སྐུ་གྲོངས་ཡོད་འདུག གྲྭ་བཀྲ་ཤིས་དཔལ་འབྱོར་ལགས་ནི། ཆབ་མདོ་ཁུལ་ཝོན་པ་སྡེ་བ་ནས་ཡིན་པ་དང་། ད་ལྟ་རང་ལོ་ ༣༤ ཡིན་འདུག ཁོང་གི་ཨ་མ་དེ་སྔོན་ནས་གྲོངས་ཡོད་པ་དང་ཕ་མིང་ལ་དངོས་གྲུབ་ཞུ་བཞིན་ཡོད་འདུག ཁོང་གིས་དེ་སྔོན་སྡེ་དགེ་རྫོང་སར་དགོན་དུ་ལོ་ ༡༣ རིང་བོད་ཀྱི་ཐུན་མོང་མང་ཡིན་པའི་ཆོས་དང་རིག་གཞུང་ལ་སློབ་གཉེར་གནང་སྟེ་སློབ་དཔོན་གྱི་མཚན་གནས་བཞེས་ཡོད་པ་དང་། ཕྱིས་སུ་ཝོན་པ་དགོན་པའི་ནང་བཞུགས་ཡོད་འདུག གནས་ཚུལ་བྱུང་རྗེས་དགོན་པའི་གྲྭ་མང་གིས་ཁོང་གི་ཆེད་དུ་བསྔོ་སྨོན་མཆོད་འབུལ་གནང་གི་ཡོད་འདུག དེ་བཞིན། ཀུན་བཟང་བསྟན་འཛིན་ལགས་ཀྱིས་འདི་གར་གནས་ཚུལ་ངོ་སྤྲོད་གནང་དོན་ལྟར་ན།  ཟླ་འདིའི་ཚེས་ ༤ ཉིན་གྱི་མཚན་མོར་ཆབ་མདོ་དཔའ་ཤོད་རྫོང་གི་ཉེན་རྟོག་པ་ཁག་ཅིག་གློ་བུར་གྲོང་གསར་དགོན་པའི་ནང་ཡོང་སྟེ། དགོན་པའི་དབུ་མཛད་རྒན་པ་བློ་བཟང་ཆོས་འབྱོར་ལགས་ཀྱི་གྲྭ་ཤག་སྔོག་གཤེར་དང་ལག་འཁྱེར་ཁ་པར་ཞིབ་བཤེར་བྱེད་དགོས་ཚུལ་བཤད་ཡོད་འདུག སྐབས་དེར་ཁོང་གིས་ཁ་པར་རྩིས་སྤྲོད་བྱས་མེད་ཀྱང་ཉེན་རྟོག་པས་ཁོང་གིས་སྐད་འཕྲིན་བརྒྱུད་མི་གཞན་ལ་༧གོང་ས་མཆོག་གི་སྐུ་པར་དང་བཀའ་སློབ་བསྐུར་ཡོད་ཚུལ་དང་། ཁ་པར་ནང་སྐུ་པར་ཉར་ཚགས་བྱས་ཡོད་སྐོར་སོགས་བཤད་དེ་འཛིན་བཟུང་བཙན་འཁྲིད་བྱས་ཡོད་འདུག གྲྭ་བློ་བཟང་ཆོས་འབྱོར་ལགས་ནི་ཆབ་མདོ་ཚ་ཕུག་སྡེ་བ་ཡིན་པ་དང་། ད་ལྟ་རང་ལོ་ ༢༦ ཙམ་ཡིན་འདུག ཁོང་གི་ཕ་མིང་ལ་ཆོས་འཕེལ་དང་མ་མིང་ལ་བག་ཆུང་ཞུ་བཞིན་ཡོད་འདུག གསར་འགོད་པ། དོན་གོ
