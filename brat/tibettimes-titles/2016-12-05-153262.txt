སྒྲོལ་མ་མཚོ་བཙོན་གྲོལ་ཐོབ་པ།

དེ་ཡང་ཕྱི་ལོ་ ༢༠༡༦ ཟླ་ ༡༢ ཚེས་ ༤ ཉིན་གྱི་མཚན་མོའི་ཆུ་ཚོད་༩ ཡས་མས་སུ་རྔ་པ་རྫོང་རྨེའུ་རུ་མའི་རུ་ཆེན་གསུམ་པའི་དགེ་རིག་ཚང་གི་སྒྲོལ་མ་མཚོར་རྒྱ་གཞུང་གིས་ལོ་ངོ་གསུམ་ལ་ཁྲིམས་ཐག་བཅད་པའི་བཙོན་འཇུག་དུས་ཡུན་ཚང་ནས་ཕྱིར་རང་ཁྱིམ་དུ་བསྐྱལ་འདུག སྐབས་དེར་སྒྲོལ་མ་མཚོ་ལ་གཉེན་ཉེ་དང་ཡུལ་མི། བོད་མི་ཉིས་བརྒྱ་ལྷག་གིས་ཁ་བཏགས་གཡོགས་ནས་ཕེབས་བསུ་བྱས་འདུག ཕྱི་ལོ་ ༢༠༡༣ ཟླ་ ༡༢ ཚེས་ ༣ ཉིན་རྔ་བ་རྫོང་རྨེའུ་རུ་མའི་ཡུལ་ཚོ་གསེར་སྡེ་ནས་དཀོན་མཆོག་ཚེ་བརྟན་ལགས་ཀྱིས་རང་ལུས་མེར་བསྲེགས་གནང་སྐབས། སྒྲོལ་མ་མཚོས་ཁོང་གི་སྐུ་ཕུང་རླངས་འཁོར་ནང་དུ་འཇོག་རོགས་བྱས་པ་སོགས་ཀྱི་རྐྱེན་གྱིས་བཙན་བཟུང་བྱས་ཏེ་ཟླ་བ་ ༡༡ རིང་ལ་བཀག་ཉར་ཁང་ནས་གཅར་རྡུང་དང་མནར་གཅོད་སྣ་ཚོགས་མྱངས་དགོས་བྱུང་ཞིང་། ཕྱི་ལོ་ ༢༠༡༤ ཟླ་ ༡༡ ཚེས་ ༣ ཉིན་སྒྲོལ་མ་མཚོར་རྒྱ་གཞུང་གིས་མི་བསད་པའི་ནག་ཉེས་བཙུགས་ནས་ལོ་གསུམ་ལ་ཁྲིམས་ཐག་བཅད་ནས་སི་ཁྲོན་མེ་ཡན་བཙོན་ཁང་ནང་བཅུགས་པ་རེད། ཕྱི་ལོ་ ༢༠༡༥ ཟླ་ ༡༢ ནང་ལ་ཁྲིམས་ཁང་རང་ནས་ཡི་གེ་སྔ་ཕྱི་གཉིས་ཁྱིམ་བདག་ཚོར་སྤྲད་དེ། སྒྲོལ་མ་མཚོར་ཛ་དྲག་གི་གཤག་བཅོས་གཏོང་དགོས་སྐོར་འཁོད་ཡོད་པ་དང་ཁྱིམ་བདག་ཚོས་མིང་རྟགས་འགོད་དགོས་ཚུལ་བཙན་བཀའ་གཏོང་སྐབས། ཁྱིམ་བདག་ཚོས་ཁྲིམས་ཁང་ལ་ཁ་པར་བཏང་སྟེ་སྒྲོལ་མ་མཚོ་ལ་དངོས་སུ་ཐུག་བཅུག་ན་སྒྲོལ་མ་མཚོའི་གནས་བབ་གང་ཞིག་ཡིན་པ་ཤེས་རྟོགས་ཡོང་བས། དངོས་སུ་ཐུག་འཇུག་པའི་རེ་འདུན་བཏོན་ཡང་ཐུག་བཅུག་མེད་པ་མ་ཟད། སྒྲོལ་མ་མཚོས་ཁྲིམས་ཁང་ནས་ཁྱིམ་བདག་ཚོར་སྐར་མ་ལྔའི་རིང་ཁ་པར་བཏང་སྟེ་བཙོན་སྲུང་བ་ཚོས་ང་ལ་ན་ཚ་ཞིག་ཡོད་པས་གཤག་བཅོས་གཏོང་དགོས་པའི་བཙན་སྐུལ་ཡང་ཡང་བྱས་ཀྱང་། ངའི་ངོས་ནས་ཁས་ལེན་རྩ་བ་ནས་བྱས་མེད། ཁྱོད་ཚོས་ཀྱང་ཁས་ལེན་རྩ་བ་ནས་མ་བྱེད་ཅེས་བརྗོད་འདུག ད་སྐབས་སྒྲོལ་མ་མཚོའི་གནས་སྟངས་དེ་མགོ་ཡུ་འཁོར་བ་དང་གཉིད་མི་ཡོང་། གཟུགས་པོ་སྐྱོ་བོ་ཆགས་པ། སྔར་གྱི་གཤིས་ཀར་འགྱུར་བ་ཐེབས་པ། སྙིང་ཁམས་མི་བདེ་བ་སོགས་ཀྱི་གནས་སྟངས་ཆགས་ཡོད་སྟབས། མ་འོངས་པའི་མོ་རང་གི་འཕྲོད་བསྟེན་ཐད་ནང་མི་ཚོས་སེམས་སེམས་འཚབ་ཆེན་པོ་བྱེད་བཞིན་ཡོད་འདུག སྒྲོལ་མ་མཚོ་ནི། རྔ་བ་རྫོང་རྨེའུ་རུ་མའི་དགེ་ཚང་གི་ཡིན་ཞིང་། ཕ་དགེ་རིག་དང་མ་དོན་ཁོ་གཉིས་ཡིན་འདུག ཁོང་མོ་ལ་ལོ་ ༡༤ ཅན་གྱི་བུ་མོ་ཚུལ་ཁྲིམས་སྒྲོལ་མ་ཟེར་བ་ཞིག་ཡོད་འདུག ཅེས་ཨོ་སི་ཊོ་ལི་ཡར་གནས་སྡོད་རྨེའུ་དཀོན་རྒྱམ་ལགས་ཀྱིས་གནས་ཚུལ་བཏང་འབྱོར་བྱུང་ཞིང་། ཁོང་མོ་དང་རྨེའུ་དཀོན་རྒྱམ་གཉེན་འབྲེལ་ཡིན་འདུག
