འབྲོག་ཁྱི་ལྡོམ་པོའི་ལས་དབང་།

འབྲོག་ཁྱིའི་རིན་གོང་ཉིན་རེ་བཞིན་མར་ཟགས་པར་བརྟེན་སྔོན་ཆད་རིན་གོང་ཆེན་པོ་སྤྲད་དེ་ཉོས་པའི་འབྲོག་ཁྱི་རྣམས་བདག་མེད་དུ་གྱུར་ཏེ་མི་ཕྱུག་སེམས་ཅན་ལ་ཉེན་ཁ་བཟོ་ཡི་ཡོད་འདུག ཁ་སང་(NextShark)ཞེས་པའི་གསར་ཁང་གིས། དེ་སྔོན་འཛམ་གླིང་ཐོག་རིན་གོང་ཆེ་ཤོས་སུ་གྱུར་བའི་བོད་ཀྱི་འབྲོག་ཁྱིར་ཁྲོམ་རྭའི་ནང་རིན་གོང་མར་ཆགས་པའི་རྐྱེན་གྱིས།  ཁ་སང་དེ་རིང་བོད་རང་སྐྱོང་ལྗོངས་དང་མཚོ་ལྷོ་ས་ཁུལ་མང་པོར་འབྲོག་ཁྱི་སྟོང་ཕྲག་མང་པོ་བདག་མེད་དུ་གྱུར་ཏེ།  ཁྱི་ན་ཚ་ཅན་དང་ཁྱི་སྨྱོན་གྱི་གྲངས་འབོར་མང་དུ་ཕྱིན་ཏེ་ས་གནས་ཁག་མང་པོའི་སྒོ་ཟོག་སེམས་ཅན་ཙམ་མ་ཟད། ས་གནས་མི་མང་ལ་ཁྱི་སྨྱོན་གྱིས་རྨུག་སྟེ་འདས་གྲོངས་སུ་ཕྱིན་པའི་དཀའ་ངལ་སོགས་འཕྲད་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག ཧྲང་ཧེ་གསར་ཁང་གི་གསར་འགྱུར་ཞིག་ཏུ།  ཕྱི་ལོ་ ༢༠༡༥ ལོར་བོད་རང་སྐྱོང་ལྗོངས་ཁུལ་གྱི་འབྲོག་ཁྱིའི་ཚོང་ལས་ཁང་གསུམ་ཆ་གཅིག་གིས་ཚོང་སྒོ་རྒྱག་དགོས་ཁེལ་ཏེ། ལོ་དེར་མཚོ་ལྷོ་ཁུལ་དུ་འབྲོག་ཁྱིའི་ཐོག་ཚོང་ལས་རྒྱ་སྒོར་ས་ཡ་ ༢༠༠ ནས་ས་ཡ་ ༥༠ བར་མར་ཟགས་ཡོད་སྟབས། ཁྱི་ཚོང་རྒྱག་མཁན་མང་པོ་ཞིག་གིས་འབྲོག་ཁྱི་བདག་མེད་དུ་གཡུགས་ཡོད་པ་དང་། ད་ལྟ་མཚོ་ལྷོ་ཁུལ་དུ་ཁྱི་སྡོམ་གྱི་གྲངས་འབོར་མང་ནས་མང་དུ་འགྲོ་བཞིན་ཡོད་པ་མ་ཟད།  གྲོང་ཚོ་རེ་ཟུང་ལ་ཆེས་ཉུང་མཐའ་ཡང་ཁྱི་ལྡོམ་ ༥༠༠ ཡོད་པའི་གནས་སྟངས་ཆགས་ཡོད་སྟབས།  ཁྱི་ལྡོམ་དེ་དག་གིས་མི་མང་ལ་རྨུག་པ་སོགས་ཚེ་སྲོག་ལའང་ཉེན་ཁ་བཟོ་ཡི་ཡོད་འདུག ཤི་ཨན་སེམས་ཅན་བདག་སྐྱོང་ཁང་གི་འགན་འཛིན་ཅཱང་ཧྥོང་(Jiang Hong)གིས། ས་གནས་སྲིད་གཞུང་ཁག་ཅིག་གིས་ཁྱི་ལྡོམ་རྣམས་བསྡུས་ཏེ་ཁྱི་ར་བརྒྱབ་ཡོད་ཀྱང་། འབྲོག་ཁྱིའི་རིན་གོང་ཉིན་རེ་ནས་ཉིན་རེ་བཞིན་མར་ཟགས་ཀྱིན་ཡོད་སྟབས། ཁྱི་ལྡོམ་གྱི་གྲངས་འབོར་མང་དུ་འགྲོ་ཡི་ཡོད་པ་མ་ཟད། ཁྱི་སྨྱོན་གྱི་ཁ་གྲངས་ཀྱང་ཇེ་མང་དུ་འགྲོ་བཞིན་ཡོད།  ཅེས་བརྗོད་འདུག་ལ། ན་ནིང་ཕྱི་ཟླ་ ༡༡ པའི་ནང་མཚོ་ལྷོ་ཁུལ་གྱི་ལོ་ན་ ༨ ཅན་གྱི་བུ་མོ་ཞིག་ལ་ཁྱིས་རྨུགས་ཏེེ་འདས་གྲོངས་སུ་ཕྱིན་ཡོད་པ་རེད། ཡང་ཅཱང་ཧྥོང་གིས་བརྗོད་ན། མཚོ་ལྷོ་ཁུལ་གྱི་གྲོང་ཁྱེར་ཁག་ཅིག་གིས་ཁྱི་ལྡོམ་ཐད་ཀར་གསོད་པའི་ལས་འགུལ་སྤེལ་རྩིས་བྱས་ཡོད་ཀྱང་།  ས་གནས་དེ་དག་ནང་བོད་རིགས་མང་པོ་ཡོད་པ་དང་། ལྷག་པར་དུ་ནང་པའི་དགོན་སྡེ་ཁག་གིས་སེམས་ཅན་གསོད་པ་ནི་སྡིག་ལས་ཤིག་ཡིན་པར་ངོས་འཛིན་བྱས་ཏེ།  ད་ལྟ་ཁྱི་ལྡོམ་མང་པོ་ཞིག་ལ་བོད་ཀྱི་དགོན་པ་ཁག་ཅིག་གིས་ཁྱི་ལྟོ་སྤྲད་དེ་གསོ་སྐྱོང་བྱེད་བཞིན་ཡོད་ཀྱང་། ད་ལྟ་བོད་ཀྱི་འབྲོག་ཁྱིའི་ལས་དབང་བཤད་དཀའ་བའི་གནས་ཤིག་གི་ནང་ལྷུང་ཡོད་འདུག འབྲོག་ཁྱི་ནི་བོད་དང་སོག་པོ་སོགས་འབྲོག་ཁུལ་དུ་སྐྱེས་འཚར་བྱུང་བའི་ཁྱི་རིགས་སྟོབས་ཆེན་པོ་ཞིག་ཡིན་ཞིང་། བོད་ནང་ཁྱིའི་རིན་གོང་ཤུགས་ཆེར་འཕར་བ་དང་བསྟུན་འབྲོག་ཁྱི་རྒྱུད་སྤེལ་གྱི་ལས་གཞི་མང་པོ་སྤེལ་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༡༡ ལོར་བོད་ཀྱི་འབྲོག་ཁྱི་ཞིག་ལ་ཨ་སྒོར་ས་ཡ་ ༡།༩ བྱས་ནས་ཕྱིར་ཚོང་བྱས་ཡོད་པ་རེད། གསར་འགོད་པ། སྐལ་བཟང་སྦྱིན་པ།
