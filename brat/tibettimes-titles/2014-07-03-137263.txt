ཞི་རྒོལ་བ་ ༥༡༡ འཛིན་བཟུང་བྱས།

ཧོང་ཀོང་ནང་མི་གྲངས་འབུམ་ ༥ ལྷག་གིས་རྒྱ་གཞུང་གི་དབང་བསྒྱུར་ལ་ངོ་རྒོལ་བྱེད་སྐབས། ཉེན་རྟོག་པས་ཞི་རྒོལ་བ་ ༥༡༡ འཛིན་བཟུང་བྱས་ཡོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༡ ཧོང་ཀོང་རྒྱ་ནག་ལ་རྩིས་སྤྲོད་བྱས་ནས་ལོ་ངོ་ ༡༧ འཁོར་བའི་དུས་དྲན་ཉིན། སྔ་དྲོའི་ཆུ་ཚོད་ ༡༠ ཙམ་ནས་དགུང་མོའི་ཆུ་ཚོད་ ༡༢ བར་ཧོང་ཀོང་གི་ཆབ་སྲིད་དང་དཔལ་འབྱོར་གྱི་ལྟེ་གནས་གྲོང་ཁྱེར་ཀྲུང་ཧན་དུ། ཧོང་ཀོང་དམངས་ཁྲོད་འགྲོ་བ་མིའི་ཐོབ་ཐང་ཚོགས་པ་དང་ཐའེ་ཝན་འགྲོ་བ་མིའི་ཐོབ་ཐང་ཡར་སྐུལ་ཚོགས་པ། སའི་གོ་ལའི་སྤྱི་མང་ཐེབས་རྩ་ཁང་སོགས་ཀྱི་སྣེ་ཁྲིད་དེ། མི་མང་འབུམ་ ༥ ལྷག་གིས་རྒྱ་གཞུང་གི་དབང་བསྒྱུར་ལ་ངོ་རྒོལ་དང་། ཧོང་ཀོང་དམིགས་བསལ་སྲིད་འཛིན་ཉིང་ཀྲིན་ཡེན་ལས་གནས་ནས་ཕྱིར་འབུད་གཏོང་དགོས་པ་སོགས་ཀྱི་འབོད་སྒྲ་བསྒྲགས་ཏེ་ཁྲོམ་སྐོར་བྱས་ཡོད་པ་དང་། སྐབས་དེར་ཉེན་རྟོག་པ་ ༤༠༠༠ ལྷག་གིས་བཀག་འགོག་བྱས་ཡོད་པ་མ་ཟད། ཁ་སང་ཕྱི་ཚེས་ ༢ ཉིན་ཉེན་རྟོག་ཕྱོགས་ནས་ཁྲིམས་འགལ་གྱིས་འདུ་འཛོམས་རྩ་འཛུགས་དང་ལས་འགུལ་ནང་མཉམ་ཞུགས་བྱས་པའི་ནག་ཉེས་འོག ཞི་རྒོལ་བ་ ༥༡༡ འཛིན་བཟུང་བྱས་ཡོད་པ་དང་། ཧོང་ཀོང་གི་ལོ་རྒྱུས་ནང་མི་གྲངས་དེ་འདྲ་མང་པོ་འཛིན་བཟུང་བྱས་པ་ཐོག་མ་དེ་ཆགས་ཡོད་ལུགས་བསྒྲགས་འདུག ཐེངས་འདིའི་ལས་འགུལ་ནང་མཉམ་ཞུགས་གནང་བའི་ཨོ་མིན་དམངས་གཙོ་ཚོགས་པའི་ཚོགས་མི་སུའུ་ཅཱ་ཧོ་ཡིས། ཨོ་མིན་མི་མང་གིས་རྒྱ་ནག་གུང་ཁྲན་ཏང་དང་གཞུང་གི་ཚོང་ལས་ངན་འབྲེལ་སྡེ་ཁག་གཉིས་ལ་ཁ་གཏད་གཅོག་བཞིན་ཡོད་པ་དང་། མ་འོངས་པའི་མི་རབས་ཚོ་འདྲ་མཉམ་གྱི་སྤྱི་ཚོགས་ཤིག་ཏུ་འཚོ་གནས་ཆེད། ང་ཚོས་རྒྱ་གཞུང་གི་ཆབ་སྲིད་ཀྱི་དམིགས་བསལ་དབང་ཆ་མེད་པར་བཟོས་ཏེ། མི་མང་གི་འོས་འདེམས་དབང་ཆ་རྩོད་ལེན་བྱེད་དགོས་སྐོར་བརྗོད་ཡོད་པ་དང་། ཧོང་ཀོང་གཞུང་གིས་ཞི་རྒོལ་བ་ ༥༡༡ འཛིན་བཟུང་བྱས་པར་ཧོང་ཀོང་ཆབ་སྲིད་པ་དང་ཚོགས་པ་ཁག་གིས་དོ་སྣང་ཆེན་པོ་བྱེད་ཀྱི་ཡོད་པ་མ་ཟད། ཉེན་རྟོག་ཕྱོགས་ནས་དྲག་པོའི་སྲིད་བྱུས་བཟུང་པར་སྐྱོན་བརྗོད་བྱས་འདུག ཧོང་ཀོང་གི་གནད་དོན་དབྱེ་ཞིབ་པ་ཚོས། རྒྱ་གཞུང་གིས་ཧོང་ཀོང་གི་ཉེན་ཁ་ཐག་གཅོད་བྱེད་དགོས་ན། ངེས་པར་དུ་ཧོང་ཀོང་མི་མང་གི་མངོན་འདོད་བསྒྲུབ་སྟེ། མི་མང་གི་བདག་དབང་དང་དམངས་གཙོའི་ལམ་ལུགས་རྒྱ་སྐྱེད་གཏོང་དགོས། གལ་སྲིད་རྒྱ་གཞུང་གིས་དྲག་ཤུགས་ཀྱི་ཐོག་ནས་གནད་དོན་ཐག་གཅོད་བྱས་པ་ཡིན་ན། བསམ་ཡུལ་ལས་འདས་པའི་མཇུག་འབྲས་ངན་པ་འབྱུང་རྒྱུ་ཡིན་སྐོར་དང་། ཧོང་ཀོང་གི་དམངས་གཙོ་མེད་པར་བཟོས་ན་རྒྱ་ནག་སྐམ་ས་ཆེན་མོའི་དམངས་གཙོ་ཡང་བར་ཐག་ཇེ་རིང་ནས་ཇེ་རིང་དུ་འགྲོ་རྒྱུ་ཡིན་སྟབས། ཧོང་ཀོང་གི་དམངས་གཙོའི་འཐབ་རྩོད་དེ་རྒྱ་ནག་སྤྱི་ཡོངས་ཀྱི་འཐབ་རྩོད་ཅིག་ཏུ་རྩིས་ན་འགྲིག་སྟབས། ཧོང་ཀོང་དང་ཐའེ་ཝན། རྒྱ་ནག་སྐམ་ས་ཆེན་མོ་བཅས་ཀྱི་མི་མང་གིས་མཉམ་གཅིག་ཏུ་དམངས་གཙོའི་ཁེ་དབང་སྲུང་སྐྱོབ་དགོས་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
