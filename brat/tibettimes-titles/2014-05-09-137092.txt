གཉིས་ཕན་མཉམ་རྒྱལ་གྱི་གོམ་ཞེང་།

རྒྱ་ནག་གི་སྲིད་བློན་ལི་ཁེ་ཆང་ཨ་ཧྥི་རི་ཀའི་ཨན་གྷོ་ལ་རྒྱལ་ཁབ་ནང་སྐོར་སྐྱོད་བྱས་ཏེ་ཚོང་དོན་གྲོས་མཐུན་ཁག་ཅིག་ལ་མིང་རྟགས་བཀོད་ཡོད་འདུག ཟླ་འདིའི་ཚེས་ ༤ ཉིན་ནས་བཟུང་རྒྱ་ནག་གི་སྲིད་བློན་ལི་ཁེ་ཆང་གིས་ཨ་ཧྥི་རི་ཀའི་མཐུན་གྲོགས་རྒྱལ་ཁབ་ཁག་ཅིག་ནང་སྐོར་སྐྱོད་བྱེད་བཞིན་ཡོད་ཅིང་། ཁ་སང་ཨན་གྷོ་ལའི་སྲིད་འཛིན་ཇོ་སེ་ཨེ་ཌར་ཌོ་ (Jose Eduardo Dos Santos) ་དང་སྲིད་འཛིན་གཞོན་པ་མཱ་ཎུལ་ (Manuel Vicente)སོགས་ལ་ཐུག་འཕྲད་བྱས་ཏེ། རྒྱ་གཞུང་གིས་ཨན་གྷོ་ལའི་ཞིང་ལས་འཕེལ་རྒྱས་ལ་རོགས་སྐྱོར་དང་། གཞུང་ལམ་དང་གློག་ཁང་གསར་རྒྱག ཨན་གྷོ་ལའི་རྒྱལ་སའི་རྒྱལ་སྤྱིའི་གནམ་གྲུའི་བབ་ཐང་གསར་སྐྲུན། ཨན་གྷོ་ལ་མེ་འཁོར་བབ་ཚུགས་ནས་ཁོང་གྷོ་དང་ཨེཊ་ལཱན་ཊིག་མངའ་ཁུལ་རྒྱལ་ཁབ་ནང་ཤར་གཏོང་ཐུབ་པའི་མགྱོགས་མྱུར་མེ་འཁོར་ལྕགས་ལམ་བཟོ་སྐྲུན། སྨན་བཅོས་ལྟ་སྐྱོང་དང་ས་ཁང་འཛུགས་སྐྲུན། ཨན་གྷོ་ལས་རྒྱ་ནག་ལ་ལོ་ལྟར་རྡོ་སྣུམ་ཌོན་ (metric tons) ས་ཡ་༨༩།༤༥ ཚོང་རྒྱུ་སོགས་ཀྱི་གྲོས་མཐུན་ལ་མིང་རྟགས་བཀོད་ཡོད་པ་མ་ཟད། ལི་ཁེ་ཆང་གིས་ཕྱི་སྡོད་རྒྱ་རིགས་དང་ཁེ་ལས་བདག་པོ་ཚོར་ཐུག་སྟེ་བགྲོ་གླེང་ཚོགས་འདུ་ཞིག་ཀྱང་བསྐོངས་ཡོད་འདུག འདིའི་ཐད་ལི་ཁེ་ཆང་གིས། “རྒྱ་ནག་དང་ཨན་གྷོ་ལའི་འབྲེལ་ལམ་ནི་གཉིས་ཕན་མཉམ་ལས་ཤིག་ཡིན་སྟབས། མ་འོངས་པར་རྒྱལ་ཁབ་གཉིས་ཀྱི་འབྲེལ་ལམ་ལ་སྔར་མེད་ནུས་པ་ཞིག་ཐོན་ངེས་རེད།” ཅེས་དང་། ཨན་གྷོ་ལའི་སྲིད་འཛིན་ཇོ་སེ་ཨེ་ཌར་ཌོ་ཡིས། “ རྒྱ་ནག་ནི་རྒྱལ་སྤྱིའི་ཐོག་སྟོབས་འབྱོར་མངའ་དབང་ཆེ་ཤོས་གྲས་ཀྱི་རྒྱལ་ཁབ་ཅིག་ཡིན་སྟབས། རྒྱ་ནག་དང་འབྲེལ་ལམ་སྔར་ལྷག་འཕེལ་རྒྱས་གཏོང་ཐུབ་པ་བྱས་ཏེ། ཨན་གྷོ་ལའི་མཐུན་གྲོགས་རྒྱལ་ཁབ་ཡག་ཤོས་ཤིག་ཏུ་འགྱུར་ཐུབ་པ་བྱ་རྒྱུ་ཡིན།” ཅེས་སོགས་བརྗོད་འདུག རྒྱ་ནག་གི་ཚོང་དོན་བློན་ཆེན་གྷོ་ཧོ་ཁྲན་ (Gao Hucheng) གྱིས། རྒྱ་ནག་དང་ཨན་གྷོ་ལ་ནི་འཕེལ་རྒྱས་འགྲོ་བཞིན་པའི་རྒྱལ་ཁབ་ཅིག་ཡིན་སྟབས། གློག་ཁང་དང་མེ་འཁོར་ལྕགས་ལམ། སོ་ནམ་ཞིང་ལས། གཏེར་རྫས་ཉོ་ཚོང་། བཟོ་ལས་ཚན་རྩལ་གཙོས་རྒྱལ་ཁབ་ཀྱི་འཛུགས་སྐྲུན་བྱ་གཞག་ཐད་སྒོ་ཀུན་མཉམ་ལས་བྱས་ཏེ། རྒྱལ་ཁབ་འཕེལ་རྒྱས་གཏོང་ཐབས་ལ་ཤུགས་སྣོན་རྒྱག་རྒྱུ་ཡིན་སྐོར་བརྗོད་འདུག ༡༩༧༨ ལོར་ཨན་གྷོ་ལ་དང་ཨ་ཧྥི་རི་ཀའི་རྒྱལ་ཁབ་གཙོས་ཡར་རྒྱས་འགྲོ་བཞིན་པའི་རྒྱལ་ཁབ་ཁག་གིས། རྒྱ་ནག་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་གཏན་འཇགས་ཚོགས་མི་ལྔའི་ཁོངས་སུ་ཞུགས་པར་རྒྱབ་སྐྱོར་བྱས་པ་ནས་བཟུང་རྒྱལ་ཁབ་གཉིས་ཀྱི་འབྲེལ་ལམ་མགོ་འཛུགས་ཙམ་བྱུང་ཡོད་ཀྱང་། ཨན་གྷོ་ལར་ནང་འཁྲུག་བསྟུད་མར་བྱུང་སྟབས་ལམ་མདུན་སྐྱོད་ཆེན་པོ་བྱུང་མེད། ༢༠༠༠ ལོར་རྒྱ་ནག་དང་ཨ་ཧྥི་རི་ཀའི་གླེང་སྟེགས་ཚོགས་འདུ་ཐོག་མ་ཚོགས་སྐབས། རྒྱ་གཞུང་དང་ཨན་གྷོ་ལར་བར་ཨ་སྒོར་ས་ཡ་ ༤༦༥ དང་། ༢༠༠༦ ལོར་སྲིད་བློན་ཝུན་ཅ་པའོ་ལོར་ཨན་གྷོ་ལར་སྐོར་སྐྱོད་བྱས་ཏེ་ཨ་སྒོར་ཐེར་འབུམ་ ༡༠།༢༣ གྱི་ཚོང་དོན་གྲོས་མཐུན་དང་། ཨ་སྒོར་ཐེར་འབུམ་ ༩ ཡི་སྐྱེད་མེད་བུ་ལོན་གཡར་ཡོད་འདུག ༢༠༠༧ ལོར་ཨན་གྷོ་ལ་དང་རྒྱ་ནག་བར་ཨ་སྒོར་ཐེར་འབུམ་ ༢༤།༨ དང་། ༢༠༠༩ ལོ་ནས་ ༢༠༡༢ ལོ་བར་ཨ་སྒོར་ཐེར་འབུམ་ ༢༨། ༢༠༡༣ ལོར་ཨ་སྒོར་ཐེར་འབུམ་ ༣༦ གི་ཚོང་འབྲེལ་བྱས་ཡོད་སྟབས། ༢༠༡༢ ལོ་དང་བསྡུར་ན་ཨ་སྒོར་ཐེར་འབུམ་ ༨ མང་དུ་ཕྱིན་ཡོད་འདུག ༢༠༡༣ ལོའི་རྒྱལ་སྤྱིའི་དངུལ་ཁང་གི་སྙན་ཐོའི་ནང་། རྒྱ་གཞུང་གིས་ཨ་ཧྥི་རི་ཀའི་མཐུན་གྲོགས་རྒྱལ་ཁབ་ཁག་འཕེལ་རྒྱས་གཏོང་ཐབས་ལ་བརྒྱ་ཆ་ ༢༨ ཙམ་ཞིག་ལེགས་སྐྱེས་ཕུལ་ཡོད་པ་མ་ཟད། བུ་ལོན་ཨ་སྒོར་ཐེར་འབུམ་ ༣།༢ ཆག་ཡང་བཏང་ཡོད་སྐོར་འཁོད་འདུག ༢༠༠༨ ལོ་ནས་བཟུང་རྒྱ་ནག་ནི་རྒྱལ་སྤྱིའི་ཐོག་ཨ་ཧྥི་རི་ཀར་ཚོང་འབྲེལ་མང་ཤོས་བྱེད་མཁན་གྱི་རྒྱལ་ཁབ་དེ་ཆགས་ཡོད་པ་མ་ཟད། ༢༠༡༣ ལོར་རྒྱ་ནག་དང་ཨ་ཧྥི་རི་ཀའི་མཐུན་གྲོགས་རྒྱལ་ཁབ་བར་ཨ་སྒོར་ཐེར་འབུམ་ ༢༡༠ ཡི་ཚོང་འབྲེལ་བྱས་ཡོད་འདུག ཨ་རིའི་རྒྱལ་དོན་དྲུང་ཆེ་ཇོན་ཁེ་རེ་ཡང་ཉིན་འདི་འགར་ཨ་ཧྥི་རི་ཀའི་རྒྱལ་ཁབ་ཁག་ཅིག་ནང་སྐོར་སྐྱོད་བྱེད་བཞིན་ཡོད་ཅིང་། ཚེས་ ༤ དང་ ༥ བཅས་ཉིན་གཉིས་རིང་ཨན་གྷོ་ནང་བསྐྱོད་དེ་ཨན་གྷོ་ལའི་སྲིད་འཛིན་དང་ཐུག་འཕྲད་གྱིས་གློག་འཕྲུལ་ལས། སོ་ནམ་ཐོན་ལས་འཕེལ་རྒྱས་གཏོང་ཕྱོགས་སོགས་ཀྱི་སྐོར་ལ་གྲོས་ཆོད་བཞག་ཡོད་པ་མ་ཟད། ཨན་གྷོ་ལར་དམག་རྒྱག་གནམ་གྲུ་ (Boeing 777) རྟགས་ཅན་ཉོ་རྒྱུར་ཨ་སྒོར་ས་ཡ་ ༣༠༠ ཡི་བུ་ལོན་གཡར་ཡོད་འདུག ༢༠༡༣ ལོར་ཨ་རི་དང་ཨན་གྷོ་ལ་བར་ཨ་སྒོར་ཐེར་འབུམ་ ༡༠།༢ ཀྱི་ཚོང་འབྲེལ་དང་། དེ་དག་ལས་ཨ་སྒོར་ས་ཡ་ ༢༥ རྡོ་སྣུམ་ཚོང་འབྲེལ་བྱས་ཡོད་འདུག ཨན་གྷོ་ལ་ཡི་ཤུའི་མཐོ་སློབ་གྱི་དཔལ་འབྱོར་རིག་པའི་ཆེད་མཁས་པ་ཝ་སན་ཌི་ (Vicente Pinto de Andrade) ལགས་ཀྱིས། “རྒྱ་གཞུང་གི་ཐབས་ཇུས་ལྟར་ཨོ་བྷ་མས་ཨན་གྷོ་ལ་དང་ཨ་ཧྥི་རི་ཀའི་རྒྱལ་ཁབ་ཁག་དང་མཉམ་འབྲེལ་བྱས་ཏེ། ཨ་རིའི་སྟོབས་ཤུགས་ཕྱོགས་ཡོངས་ནས་འཕེལ་རྒྱས་གཏོང་ཐབས་ལ་འབད་བརྩོན་བྱེད་བཞིན་ཡོད།” ཅེས་བརྗོད་འདུག ཨན་གྷོ་ལ་སྲིད་འཛིན་གྱི་དཔལ་འབྱོར་རིག་པའི་སློབ་སྟོན་པ་ཁཱ་སེ་མཱོ་ (Basilio Cassomo) ལགས་ཀྱིས། “ རྒྱ་ནག་དང་ཨན་གྷོ་ལའི་འབྲེལ་ལམ་ནི་གཉིས་ཕན་འཕེལ་རྒྱས་ཡོང་ཐབས་ཀྱི་འབྲེལ་ལམ་ཞིག་ཡིན་པ་མ་ཟད། རྒྱལ་ཁབ་གཉིས་བར་ཆ་རྐྱེན་གང་ཡང་མེད་པར་ཚོང་འབྲེལ་གྲོས་ཆོད་འཇོག་བཞིན་ཡོད་སྟབས། རྒྱལ་སྤྱིའི་སྒྲིག་འཛུགས་ཁག་གིས་ཨན་གྷོ་ལའི་བདག་དབང་ལ་རྒྱ་གཞུང་གིས་ཐེ་ཇུས་བྱེད་བཞིན་ཡོད་སྐོར་སྙོན་འཛུགས་བྱེད་བཞིན་པ་དག་ནི་གཞི་རྩ་གང་ཡང་མེད།” ཅེས་བརྗོད་འདུག སྲིད་དོན་དབྱེ་ཞིབ་པ་ཚོས། ཉེ་ཆར་ཨ་རིའི་རྒྱལ་དོན་དྲུང་ཆེ་ཇོན་ཁེ་རེ་ཨ་ཧྥི་རི་ཀའི་རྒྱལ་ཁབ་ཁག་ཏུ་སྐོར་སྐྱོད་ཕྱིན་པ་དང་བསྟུན་ཏེ། ལི་ཁེ་ཆང་ཨ་ཧྥི་རི་ཀའི་མཐུན་གྲོགས་རྒྱལ་ཁབ་ཁག་བར་ཚོང་དོན་གྲོས་མཐུན་བཞག་པ་དེ་ནི། ཨ་རིས་ཁ་གཏད་གཅོག་ཐབས་ཤིག་ཡིན་སྐོར་བརྗོད་འདུག སང་ཉིན་ལི་ཁེ་ཆང་ཁེན་ཡར་སྐོར་སྐྱོད་དང་། ཚེས་ ༡༡ ཉིན་རྒྱ་ནག་ལ་ཕྱིར་ལོག་བྱ་རྒྱུ་ཡིན་འདུག གསར་འགོད་པ། དོན་གྲུབ།
