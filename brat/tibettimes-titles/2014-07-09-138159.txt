རབ་བྱུང་ཕོ་མོ་ཚོས་ཆང་མ་འཐུང་ཨཱ།

དེ་རིང་༸གོང་ས་མཆོག་གིས་ལ་དྭགས་ཞུ་བའི་ཚལ་དུ་སྐྱ་སེར་ཚང་མས་ཆང་འཐུང་རྒྱུ་སྤོང་དགོས་པ་དང་། ལྷག་པར་དུ་རབ་བྱུང་ཕོ་མོ་ཚོས་ཆང་གི་དྲི་ཙམ་ཡང་མ་འཕོག་པ་གནང་དགོས་ཞེས་གསུངས་སོང་། དེ་རིང་ཞོགས་པའི་ཆུ་ཚོད་ ༨ །༠༠ ནས་ ༡༡།༠༠ བར་༸གོང་ས་མཆོག་གིས་ལ་དྭགས་ཞི་བའི་ཚལ་གསུང་ཆོས་རྭ་བར་མི་མང་ལ། དཔལ་མགོན་འཕགས་པ་ཀླུ་སྒྲུབ་ཀྱིས་མཛད་པའི་དབུ་མ་རིན་ཆེན་ཕྲེང་བའི་གསུང་ཆོས་བསྩལ་ཡོད་ཅིང་། གསུང་ཆོས་དང་འབྲེལ་ནས་ཁོང་གིས་སྐྱ་སེར་སུ་ཞིག་ཡིན་ན་ཆང་རག་སོགས་འཐུང་ན་མི་འགྲིག་པའི་སྐོར་ལ་བཀའ་སློབ་གནང་དོན། ཆང་འཐུང་རྒྱུ་གཏན་དུ་སྤོང་ཐུབ་ན་ཡག་པོ་ཞེ་དྲག་རེད། ཆང་སྤོང་མ་ཐུབ་ཀྱང་ཉུང་འཐུང་བྱེད་དགོས། ཡུལ་དུས་ཀྱིས་མ་འགྲིག་པ་ཡིན་ནམ་ལ་དྭགས་པ་ཚོ་ཆང་འཐུང་རྒྱུར་དགའ་བོ་ཡོད་མདོག་ཁ་བོ་རེད། ཕྱི་རྒྱལ་བ་ཚོ་ཡིན་ནའང་སྤྲོ་ཚོགས་སྐབས་སོགས་སུ་ཟ་མ་མ་ཟོས་གོང་ཆང་ཉུང་ཙམ་འཐུང་བའི་གོམས་གཤིས་འཇགས་ཡོད་སྟབས་ཉུང་འཐུང་བྱེད་དགོས། ཁྱིམ་པ་ཚོས་ཆང་འཐུང་བ་དང་ར་བཟི་ནས་འཁྲུག་པ་རྒྱག་པ་དང་ཁྱིམ་ནང་མ་འཆམ་པ་བྱུང་ན་ཡག་པོ་ཡོད་པ་མ་རེད། འབྲུ་མང་པོས་ཆང་བཟོ་པ་ལས་འབྲུའི་རྩམ་པ་ཟ་བ་གང་རུང་བྱས་ན་ཡག་གི་རེད། རབ་བྱུང་ཕོ་མོ་ཚོར་མཚོན་ན་ཆང་གི་དྲི་མ་ཙམ་ཡང་མ་ཕོག་པ་བྱེད་རོགས། བོད་ནང་དེ་རིགས་ཁ་ཤས་ཁ་ཤས་ཡོད་ས་རེད། ཧི་མ་ལ་ཡའི་རི་རྒྱུད་ནང་རབ་བྱུང་གཙང་མ་བྱེད་མཁན་ཡིན་མིན་མི་ཤེས་ཀྱང་ཆང་འཐུང་མཁན་ཡོད་པ་འདྲ་བོ་འདུག དེ་ལ་གཟབ་གཟབ་བྱེད་དགོས་རེད་མ་གཏོགས་འགྲིག་གི་མ་རེད། རབ་བྱུང་བའི་དགོན་པའི་གྲྭ་ཤག་ཁ་ཤས་ཀྱི་འཁྲིས་སུ་ཆང་དྲི་ཐིལ་ན་ཡག་པོ་ཡོད་པ་མ་རེད། དེ་གཅིག་ཞུ་འདོད་བྱུང་སོང་། ལོ་ ༣༠ དང་ ༤༠ གོང་ང་པཱི་ཏི་ལ་ཕྱིན་ནས་དགོན་པ་ཞིག་གི་ནང་བསྡད་པ་ཡིན། དགོན་པའི་དཀོན་གཉེར་དགེ་སློང་ཡག་པོ་ཞིག་འདུག ཁོང་དེ་སྔ་བཀྲ་ཤིས་ལྷུན་པོ་དགོན་པར་ཕྱིན་ཏེ་དཔེ་ཆ་བལྟས་ཡོད་འདུག ངེད་གཉིས་སྐད་ཆ་འདྲ་བྱུང་སྐབས། ངས་ཕྱེད་ཀ་རྩེད་མོ་རྩེ་ནས་ཁོང་ལ་གཞིས་ཀ་རྩེའི་ཆང་ཞིམ་པ་འདུག་གམ། པཱི་ཏིའི་ཆང་ཞིམ་པ་འདུག་དྲིས་པར་ཁོང་གིས་ཐེ་ཚོམ་མེད་པར་གཞིས་ཀ་རྩའི་ཆང་ཞེ་དྲག་ཞིམ་པ་ཡོད་ཟེར། དེར་བརྟེན། ཁོང་གཞིས་ཀ་རྩེར་ཡོད་དུས་ཆང་བཞེས་ཀྱི་ཡོད་པ་འདྲ། ཡང་པཱི་ཏིའི་དགོན་པ་ཞིག་གིས་དབྱར་གནས་ཚོགས་ཏེ་གླིང་ག་གཏོང་གི་ཡོད་སར་མང་པོ་ཞིག་ར་བཟི་ནས་འཁྱར་འཁྱོར་འགྲོ་བའི་སྐོར་གླིང་རིན་པོ་ཆེས་ང་ལ་གསུངས་བྱུང་། དེ་རིགས་རྩ་བ་ཉིད་ནས་བྱ་རྒྱུ་ཡོད་པ་མ་རེད། ང་ཚོ་སྔ་མོ་ནས་ད་ལྟའི་བར་དུ་འགྲོ་སྟངས་དེ་ག་རང་ཡིན། ད་དུང་ཡང་དེ་ག་རང་འགྲོ་རྒྱུ་ཡིན་ལབ་ན་འགྲིག་གི་ཡོད་པ་མ་རེད། དེ་སྔ་ང་ཚོའི་འགྲོ་སྟངས་མང་པོ་ཞིག་ནོར་བསྡད་ཡོད་པ་རེད། ཆོས་ལུགས་ཕན་ཚུན་འཐེན་འཁྱེར་དང་ཕྲག་དོག་བྱེད་པ་དང་། དེ་བཞིན་འདུལ་ཁྲིམས་ཞེ་དྲག་གནང་མ་ཤེས་པར་བསྡད་པ་དེ་ཚོ་ནོར་བ་རེད། མུ་མཐུད་འདི་ལྟར་ཕྱིན་ན་སངས་རྒྱས་བསྟན་པ་གནས་མི་ཐུབ་པ་ངས་ཐག་བཅད་ནས་བཤད་ཐུབ། སངས་རྒྱས་ཀྱི་བསྟན་པ་ཟེར་བ་འདི་བྱེད་རྒྱུས་ཕོངས་ནས་སངས་རྒྱས་ཀྱི་བསྟན་པ་བྱེད་པ་ཞིག་མ་རེད། སྔ་མོའི་སྤྱི་ཚོགས་རྗེས་ལུས་ཡིན་སྟབས་བྱེད་རྒྱུས་ཕོངས་ནས་ཁྱིམ་ཚང་ནང་གི་བུ་ལྐུག་ལྐུག་ཚུལ་གྱིས་ཁྱིམ་གཞི་འཛིན་ཐུབ་ཀྱི་མེད་ཟེར་ནས་དགོན་པར་བཅུག་པ་དེ་འདྲ་ད་ལྟ་ཡོང་གི་མ་རེད། དེང་སང་འཛམ་གླིང་གི་དངོས་པོ་ཡར་རྒྱས་ཕྱིན་ཏེ་སློབ་གྲྭར་འཛུལ་ནས་སློབ་སྦྱོང་བྱས་ཏེ་སོ་སོ་ལས་རིགས་འདྲ་མིན་ནང་ཞུགས་དུས། དགོན་པར་འཛུལ་བ་ལས་སོ་སོའི་ལྟོ་གོས་ཀྱི་འགྲོ་སྟངས་མང་པོ་ཞིག་ཤེས་པ་དང་ལས་ཀ་བྱེད་པ་ཡག་པོ་ཡོད། དགོན་པའི་དགེ་འདུན་པ་ཚོས་སངས་རྒྱས་ཀྱི་ཆོས་ལ་གོ་བ་ལོན་པ་དང་སངས་རྒྱས་ཆོས་ལ་དངོས་གནས་རྒྱུ་མཚན་མཐོང་པའི་བློ་ཞིག་བྱུང་ཚེ། རབ་བྱུང་གི་མཐོང་ཆེ་རུ་འགྲོ་བ་དང་རབ་བྱུང་མུ་མཐུད་ནས་ཡོང་གི་རེད། གཞན། ཁྱེད་རང་ཚོས་རྨོངས་པའི་ཆོས་ཀྱི་འགྲོ་སྟངས་ལྟར་ཕྱིན་ན་བསྟན་པ་ཡུན་རིང་གནས་རྒྱུ་ཁག་པོ་རེད། ཅེས་གསུངས་སོང་། མཇུག་ཏུ་ ༸གོང་ས་མཆོག་གིས། ངས་ཡག་པོ་ཡོང་བར་ཤ་ཚས་བཤད་པ་ཡིན་མ་གཏོགས་ང་ལ་ཁྱད་པར་གང་ཡང་མི་འདུག ལྷག་པར་དུ་ཁྱེད་རང་ལ་དྭགས་ཚོ་མགོ་རྟིང་བསློགས་ནས་ཕྱིན་ཡང་ང་ལ་ཁྱད་པར་མེད། ཅེས་ནན་གསུངས་གནང་སོང་། གསར་འགོད་པ་ལྷ་སྐྱབས་རྒྱལ་གྱིས་ལ་དྭགས་རྒྱས་ས་སླེ་ནས།
