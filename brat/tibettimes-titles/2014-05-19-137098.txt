གད་སྙིཊ་མེར་སྲེག་གི་ལས་གཞིར་ངོ་རྒོལ།

ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༠ ཉིན་རྒྱ་ནག་ཀྲི་ཅང་ཞིང་ཆེན་ཧང་ཀྲུའུ་གྲོང་ཁྱེར་དུ་ཡ་གླིང་གི་གད་སྙིགས་མེར་སྲེག་ལྟེ་གནས་ཆེ་ཤོས་འཛུགས་པར་ངོ་རྒོལ་བྱས་འདུག ཟླ་སྔོན་མའི་ཕྱི་ཚེས་ ༢༦ ནས་བཟུང་རྒྱ་ནག་ཀྲི་ཅང་ཞིང་ཆེན་གྱི་ལྟེ་གནས་ཧང་ཀྲུའུ་གྲོང་ཁྱེར་དུ། རྒྱ་གཞུང་གིས་ཨེ་ཤ་ཡའི་གད་སྙིགས་མེར་སྲེག་ལྟེ་གནས་ཆེ་ཤོས་རྒྱག་པའི་ལས་གཞི་དེར། ས་གནས་མི་མང་གིས་ངོ་རྒོལ་བསྟུན་མར་བྱས་ཏེ། ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༠ མི་གྲངས་ཁྲི་ ༣ ཡས་མས་འདུ་འཛོམས་ཀྱིས་ངོ་རྒོལ་ཤུགས་ཆེན་བྱས་ཡོད་སྟབས། དེ་མ་ཐག་རྒྱ་གཞུང་གིས་ཉེ་འགྲམ་གྲོང་ཁྱེར་སོགས་ནས་ཉེན་རྟོག་པ་དང་དྲག་ཆས་ཉེན་རྟོག་པ་ ༥༠༠༠ ལྷག་བཏང་ནས་དྲག་གནོན་བྱས་ཡོད་པ་དང་། ཐེངས་འདིའི་ངོ་རྒོལ་ཁྲོད་མི་བརྒྱ་གྲངས་ལ་ཤི་རྨས་དང་། ངོ་རྒོལ་བ་མང་པོ་འཛིན་བཟུང་བྱས་པའི་གནས་ཚུལ་ཐོན་ཡོད་ཀྱང་། ད་ལྟའི་ཆར་རྒྱ་གཞུང་གིས་གསར་འགྱུར་བརྒྱུད་ལམ་ཁག་ལ་དམ་དྲག་ཆེན་པོ་བྱེད་བཞིན་ཡོད་སྟབས། ཤི་རྨས་ཀྱི་གྲངས་འབོར་ཁ་གསལ་མི་འདུག གོང་གི་གནད་དོན་ཐད་རྒྱ་ནག་གཞུང་འབྲེལ་ཞིན་ཧྭ་གསར་ཁང་གིས། ངོ་རྒོལ་བ་ ༡༠ ལྷག་དང་ཉེན་རྟོག་པ་ ༢༩ ལ་རྨས་སྐྱོན་ཕོག་ཡོད་པ་དང་། རླངས་འཁོར་ ༣༠ ལྷག་ལ་ཆག་སྐྱོན་བཟོས་ཡོད་པའི་ནང་ཉེན་རྟོག་རླངས་འཁོར་གཉིས་མེར་བསྲེགས་བཏང་ཡོད་བ་སོགས་ཟིང་འཁྲུག་ཆེན་པོ་བསླངས་ཡོད་ཀྱང་། མིའི་ཚེ་སྲོག་ཤོར་བའི་གནས་ཚུལ་བྱུང་མེད་སྐོར་དང་། ངོ་རྒོལ་དེ་སྤྱི་ཚོགས་བདེ་འཇགས་ལ་གཏོར་བཤིག་བཏང་བའི་ཁྲིམས་འགལ་གྱི་ནག་ཉེས་ཚབས་ཆེན་ཞིག་ཡིན་སྟབས། ངོ་རྒོལ་ངན་སྐུལ་བ་ཚོ་རིམ་གྱིས་རྩད་གཅོད་བྱེད་བཞིན་ཡོད་ལུགས་བསྒྲགས་འདུག ཁ་སང་ཕྱི་ཚེས་ ༡༡ ཉིན་ཧང་ཀྲུའུ་གྲོང་ཁྱེར་སྲིད་གཞུང་གིས་བརྡ་ཁྱབ་སྤེལ་ཏེ། མི་མང་གི་མོས་མཐུན་དང་ཁྲིམས་ཐོག་གཏན་འབེབས་མ་བྱུང་བར་ལས་གཞི་མཚམས་འཇོག་བྱ་རྒྱུ་ཡིན་སྟབས། མི་མང་གིས་བསྐྱར་དུ་ངོ་རྒོལ་མི་བྱེད་པར་སྤྱི་ཚོགས་ཀྱི་བདེ་འཇགས་སྲུང་སྐྱོང་བྱེད་དགོས་ལུགས་བརྗོད་ཡོད་པ་མ་ཟད། ཧང་ཀྲུའུ་མི་མང་ཁྲིམས་ཁང་དང་མི་མང་ཞིབ་བཤེར་ཁང་། ཉེན་རྟོག་ཁང་། ཁྲིམས་འཛིན་ལྷན་ཁང་བཅས་ཀྱིས། ངོ་རྒོལ་བ་ ༡༥ མིང་དང་འདྲ་པར་ཕྱིར་བསྒྲགས་བྱས་ཏེ། རང་འགུལ་གྱིས་སྤྱི་ཚོགས་བདེ་འཇགས་ལ་གཏོར་བཤིག་བཏང་བའི་ནག་ཉེས་ཁས་ལེན་བྱེད་དུ་ཡོང་དགོས་པའི་བཀའ་རྒྱ་བཏང་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
