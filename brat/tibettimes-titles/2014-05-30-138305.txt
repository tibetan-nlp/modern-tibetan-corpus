འབར་གད་ཁྲོད་མི་ ༣༡ རྐྱེན་འདས།

དེ་རིང་ཤིན་ཅང་གི་ལྟེ་གནས་ཨོ་རོམ་ཆིའི་ཁྲོམ་སྲང་ཞིག་ཏུ་འབར་གད་དོན་རྐྱེན་བྱུང་སྟེ་མི་གྲངས་ ༣༡ རྐྱེན་ལམ་དུ་བཏང་བ་དང་། མི་ ༩༤ ལ་རྨས་སྐྱོན་ཕོག་ཡོད་འདུག དེ་རིང་སྔ་དྲོ་ཆུ་ཚོད་ ༧།༤༨ ཙམ་ལ། ཤིན་ཅང་ཡུ་གུར་མི་རིགས་རང་སྐྱོང་ཁུལ་གྱི་ལྟེ་གནས་ཨོ་རོམ་ཆི་སཱ་ཡཱས་པཱ་ཁེ་ས་ཁུལ་གྱི་ཁྲོམ་སྲང་ཞིག་ཏུ། འབར་གད་ཀྱི་དོན་རྐྱེན་ཞིག་བྱུང་སྟེ་མི་གྲངས་ ༣༡ ལྷག་རྐྱེན་འདས་སུ་གྱུར་བ་དང་། མི་ ༩༤ ལ་རྨས་སྐྱོན་ཕོག་སྟེ་སྨན་ཁང་དུ་བསྐྱལ་ཡོད་འདུག ཡུལ་དངོས་ཀྱི་མི་ཡིས་བརྗོད་པ་ལྟར་ན། སྐབས་དེར་ཁྲོམ་སྲང་དུ་ཞོགས་པའི་བཟའ་ཆས་ཉོ་ཚོང་གི་འཚང་ཁ་ཧ་ཅང་ཆེ་བ་དང་། གློ་བུར་དུ་རླངས་འཁོར་གཅིག་ཁྲོམ་གཞུང་གི་བྱང་ཕྱོགས་ནས་རྒྱུགས་ཡོང་བ་དང་། རླངས་འཁོར་གཞན་ཞིག་ལྷོ་ངོས་ནས་རྒྱུགས་ཡོང་སྟེ་གང་བྱུང་གིས་མི་མང་གཅིར་རྡུང་བཏང་ཡོད་པ་མ་ཟད། འབར་རྫས་བཅུ་གྲངས་ཤིག་གཡུགས་ཏེ་ཁྲོམ་སྲང་གང་སར་ཁྲག་གིས་ཁྱབ་སོང་སྐོར་དང་། འབར་གད་དོན་རྐྱེན་བྱུང་རྗེས། ཉེན་རྟོག་པས་ཉེ་འཁྲིས་ཀྱི་ཞོགས་པའི་ཁྲོམ་ར་ཡང་སྒོ་རྒྱག་ཏུ་བཅུག་པར་བརྟེན། གྲོང་ཁྱེར་ཡོངས་ཀྱི་སྔོ་ཚལ་དང་ཤིང་ཏོག་སོགས་ཀྱི་རིན་གོང་ལྡབ་ལོག་གིས་འཕར་ཡོད་ལུགས་བརྗོད་འདུག ཉེ་སྔོན་ཕྱི་ཟླ་ ༤ ཚེས་ ༣༠ ཉིན་ཨོ་རོ་མོ་ཆིའི་མེ་འཁོར་ས་ཚིགས་སུ་འབར་གད་དོན་རྐྱེན་བྱུང་ཏེ། མི་གྲངས་ ༧༩ ལ་རྨས་སྐྱོན་ཕོག་པ་དང་། མི་ ༣ རྐྱེན་འདས་སུ་གྱུར་ཏེ་ཉིན་གྲངས་ ༢༢ ལས་མ་འགོར་བར་ཡང་བསྐྱར་ཤི་རྨས་ཀྱི་དོན་རྐྱེན་ཚབས་ཆེན་བྱུང་ཡོད་འདུག གཞན་ཡང་། ཁ་སང་ཤིན་ཅང་ཡུ་གུར་མི་རིགས་རང་སྐྱོང་ཁུལ་མཐོ་རིམ་ཁྲིམས་ཁང་གིས་གསར་འགོད་ལྷན་ཚོགས་ཤིག་བསྡུས་ཏེ། ཉེ་སྔོན་ཤིན་ཅང་གི་ས་ཁུལ་ཁག་ ༦ གི་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་ཚོས་རྟོག་ཞིབ་བྱས་པ་བརྒྱུད་དེ། འཇིགས་སྐུལ་དྲག་སྤྱོད་དང་འབྲེལ་བའི་བརྙན་རིས་བརྒྱུད་བསྒྲགས་བྱས་པའི་ཉེས་དོན་ ༡༦ རྩད་ཆོད་ཡོད་པ་མ་ཟད། ཉེས་ཅན་ ༣༩ ལ་ཁྲིམས་ཆད་བཅད་ཡོད་པ་དང་། དེ་དག་གི་ནང་བཙོན་འཇུག་དུས་ཡུན་རིང་ཤོས་ལོ་ ༡༥ ཡིན་ལུགས་བསྒྲགས་ཡོད་ཀྱང་། བརྒྱུད་བསྒྲགས་བྱས་པའི་བརྙན་རིས་ནང་མི་རིགས་བར་འཁོན་འཛིང་དང་དམའ་འབེབས་བྱས་ཡོད་སྐོར་ཙམ་ལས་ནང་དོན་གཙོ་བོ་ག་རེ་ཡིན་མིན་གསལ་བོ་བརྗོད་མེད་སྟབས། འཛམ་གླིང་ཡུ་གུར་མི་རིགས་མཐུན་ཚོགས་ཀྱི་མགྲིན་ཚབ་པ་དེས་ལི་ཤཱ་ལགས་ཀྱིས། འདས་པའི་ལོ་འགའི་རིང་ཤིན་ཅང་ནང་ཁ་པར་སོགས་ཀྱི་བརྒྱུད་ལམ་ལ་དམ་དྲག་ཆེན་པོ་བྱས་ཡོད་པས། འཇིགས་སྐུལ་དྲག་སྤྱོད་དང་འབྲེལ་བའི་བརྙན་རིས་བརྒྱུད་བསྒྲགས་བྱས་ཚུལ་བརྗོད་པ་དེར་དོགས་པ་ཆེན་པོ་ཡོད་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
