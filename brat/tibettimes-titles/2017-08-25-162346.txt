བླ་རུང་སྒར་དུམ་བུ་གཉིས་སུ་བཅད།

རྒྱ་ནག་གཞུང་གིས་བླ་རུང་སྒར་གླིང་དགོན་གཉིས་སུ་བཅད་དེ་དམར་ཤོག་ཚོགས་པའི་མི་སྣ་ ༦ དོ་དམ་པར་གསར་དུ་བསྐོ་བཞག་གིས་སྔར་ལས་སྟངས་འཛིན་དམ་དྲག་ཤུགས་ཆེ་རུ་གཏོང་བཞིན་ཡོད་འདུག རྒྱ་ནག་གཞུང་གིས་བོད་ཀྱི་དགོན་སྡེ་ཆེ་ཤོས་གསེར་རྟ་བླ་རུང་དགོན་ལ་ལོ་  ༡ ལྷག་གི་རིང་གཏོར་བཤིག་གི་ལས་འགུལ་སྤེལ་རྗེས།  ཉེ་ཆར་རྒྱ་ནག་གཞུང་གིས་གསེར་རྟ་བླ་རུང་དགོན་དང་ལྔ་རིག་སློབ་གླིང་གཉིས་བར་ལྕགས་ར་བརྒྱབ་སྟེ་ལོགས་ཁ་བཟོ་ཡི་ཡོད་པ་མ་ཟད། དགོན་པ་དང་སློབ་གླིང་གཉིས་ལ་དོ་དམ་ལས་ཁུངས་བཙུགས་ཏེ། དེའི་འགན་འཛིན་དང་དྲུང་ཡིག་བཅས་ལ་དམར་ཤོག་ཚོགས་མི་ ༦ བསྐོ་བཞག་བྱས་ཏེ་བླ་རུང་སྒར་གྱི་འཛིན་སྐྱོང་དང་ཉིན་རེའི་བྱེད་སྒོ་ཡོངས་ལ་སྟངས་འཛིན་ནན་མོ་བྱེད་པའི་ཐབས་ཇུས་འཐེན་གྱི་ཡོད་འདུག རྒྱ་ནག་གཞུང་གིས་རྒྱ་ནག་ཉིན་རེའི་ཚགས་པར་གྱི་དྲ་བའི་སྟེང་།  བླ་རུང་དགོན་ལྔ་རིག་ནང་བསྟན་སློབ་གླིང་གི་གླིང་དང་དགོན་ལོགས་བཀར་བྱེད་སྟངས་ཀྱི་རྟོགས་སླའི་ཤོག་བྱང་ཞེས་པའི་བསྒྲགས་ཡིག་ཅིག་སྤེལ་བའི་ནང་དུ།  བླ་རུང་དགོན་པ་དང་ལྔ་རིག་སློབ་གླིང་གཉིས་གྲྭ་བཙུན་རྣམས་ཀྱི་ཁེ་ཕན་ལ་དམིགས་ཏེ་ལོགས་ཁ་བྱས་ཡོད་སྐོར་འཁོད་ཡོད་པ་མ་ཟད།  དགོན་པ་དང་སློབ་གླིང་གཉིས་ཀྱི་མི་འབོར་ལ་ཚད་གཞི་བཟོས་ཏེ།  བླ་རུང་དགོན་ལ་དོ་དམ་ལས་བྱེད་གཙོས་བའི་ཁྱོན་བསྡོམས་གྲྭ་བཙུན་ ༣༥༠༠ ལས་བརྒལ་མི་ཆོག་པ་དང་།  ལྔ་རིག་ནང་བསྟན་སློབ་གླིང་ལ་དགེ་རྒན་དང་དོ་དམ་ལས་བྱེད་སོགས་ཁྱོན་བསྡོམས་སློབ་མ་གྲངས་  ༡༥༠༠ ལས་བརྒལ་མི་ཆོག་པར་བཟོས་ཡོད་འདུག་ལ། གླིང་དགོན་དེ་གཉིས་ནང་སི་ཁྲོན་བོད་རིགས་ཁུལ་གྱི་སློབ་མ་ལས་ས་གནས་གཞན་གྱི་སློབ་མ་བསྡུ་སྐོང་བྱེད་མི་ཆོག་པའི་སྒྲིག་ཁྲིམས་སོགས་བཟོས་ཡོད་པ་རེད། བསྒྲགས་ཡིག་དེའི་ནང་དུ།  རྒྱ་ནག་གཞུང་གིས་གསེར་རྟ་གླིང་དགོན་གཉིས་ཀྱི་བྱེད་སྒོ་མི་འདྲ་སྟབས།  བླ་རུང་དགོན་པས་ཁྲིམས་མཐུན་ཐོག་ཆོས་ལུགས་ཀྱི་བྱེད་སྒོ་སྤེལ་རྒྱུ་དང་།  དེ་ཡང་རྒྱ་ནག་གི་ཆོས་ལུགས་བྱེད་སྒོའི་སྒྲིག་ཁྲིམས་གཞིར་བཞག་ཐོག་ཆོས་ལུགས་ཀྱི་བྱེད་སྒོ་སྤེལ་དགོས་པ་དང་།  དེ་བཞིན།  ལྔ་རིག་ནང་བསྟན་སློབ་གླིང་གིས་ཁྲིམས་མཐུན་ཐོག་ནང་བསྟན་རིག་གཞུང་ལ་སློབ་སྤྱོང་བྱེད་དགོས་པ་སོགས་གླིང་དགོན་གཉིས་ལོགས་བཀར་བྱེད་པའི་རྒྱུ་མཚན་དུ་དྲངས་ཡོད་པ་དང་།  དེ་བཞིན། གླིང་དགོན་གཉིས་ཀྱི་བར་འབྱེད་བྱེད་ལྕགས་ར་དང་གྱང་ར་བརྩིགས་རྒྱུ་ཡིན་འདུག་ཅིང་།  རྒྱ་ནག་གིས་ནང་བསྟན་སློབ་གླིང་ལ་གླིང་ཚོམ་ ༦ དང་དགོན་པར་གླིང་ཚོམ་ ༡༠ རུ་དབྱེ་སྟེ།  གྲྭ་བའི་ལག་ཁྱེར་གྱི་ཁ་དོག་དམར་པོ་དང་། བཙུན་མའི་ལག་ཁྱེར་སེར་པོ།  དེ་བཞིན་གླིང་དགོན་གཉིས་ཀྱི་ནང་དུ་ཡོད་པའི་མི་མང་ཚོར་ལག་ཁྱེར་ལྗང་ཁུ་བཟོས་ཏེ་ཕར་ཚུར་འགྲོ་འོང་གི་སྒྲིག་ལམ་རྒྱས་པ་བཟོ་རྒྱུ་ཡིན་འདུག རྒྱ་ནག་གིས་བླ་རུང་སྒར་གཏོར་བཤིག་གཏོང་ཆེད་དུ་དེ་སྔོན་ནས་གྲྭ་བཙུན་གྱི་ཁ་གྲངས་ཇེ་ཉུང་དུ་གཏོང་གི་ཡོད་པ་དང་། དེའི་རྗེས་རིམ་བཞིན་བླ་རུང་སྒར་ལ་འགྲོ་ལམ་གསར་རྒྱག་བྱས་ཏེ་ཡུལ་སྐོར་བ་ཇེ་མང་དུ་བཏང་ཡོད་པ་དང་། ད་ལྟ་བླ་རུང་སྒར་ཁག་གཉིས་སུ་དབྱེ་སྟེ་ཆོས་ལུགས་བྱེད་སྒོ་དང་ནང་བསྟན་སློབ་གླིང་གི་སློབ་གསོའི་ཁ་ཕྱོགས་སོགས་བླ་རུང་སྒར་སྤྱིའི་འཛིན་སྐྱོང་སྤེལ་གསུམ་ལ་དོ་དམ་ནན་པོའི་ཐོག་སྟངས་འཛིན་བྱེད་བཞིན་ཡོད་པ་རེད་འདུག གསར་འགོད་པ། སྐལ་བཟང་སྦྱིན་པ།
