རང་ཤི་ངོ་རྒོལ་དང་རྨས་མ་འདས་གྲོངས།

ཁ་སང་སེར་ཤུལ་རྫོང་འདན་མ་ཤུག་པ་སྡེ་བའི་བློ་དཔལ་བཟང་ཞུ་བ་ཞིག་གིས་རྒྱ་གཞུང་གི་བཙན་གནོན་ལ་ངོ་རྒོལ་བྱས་ཏེ་བཀག་ཉར་ཁང་དུ་རང་ཤི་བརྒྱབ་ཡོད་པ་སོགས་འདན་མའི་གནས་སྟངས་ཛ་དྲག་ཆགས་ཡོད་འདུག སྡེ་དགེ་རྒྱལ་ཡོངས་ལྷན་ཚོགས་ཀྱི་ཚོགས་གཙོ་འཆམ་ར་དྲི་མེད་རྒྱལ་མཚན་ལགས་ཀྱིས་འདི་གར་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། ཕྱི་ཚེས་ ༡༦ ཉིན་རྒྱ་གཞུང་གིས་ཤུག་པ་སྡེ་པའི་སྐྱེས་པ་ལོ་ན་ ༡༢ ཡན་དང་རྒན་འཁོགས་མན་གྱི་སྐྱ་སེར་ཚང་མ་འཛིན་བཟུང་བྱས་ཏེ། ལོ་ཕྱུག་བཙོན་ཁང་དང་ས་གནས་སྨན་ཁང་གསར་པའི་ནང་བཀག་ཉར་བྱས་ཡོད་པ་མ་ཟད། བཙན་གྱིས་སྐྲ་བྲེགས་ནས་མགོ་རིལ་བཟོས་ཡོད་པ་དང་། ཉེ་སྔོན་རྨས་སྐྱོན་ཕོག་པ་ཚོའང་བཀག་ཉར་ཁང་དུ་བཙན་ཁྲིད་ཀྱིས་སྨན་བཅོས་ཀྱི་མཐུན་རྐྱེན་མེད་པར་བཟོས་ཡོད་སྟབས། ཁ་སང་ཕྱི་ཚེས་ ༡༧ ཉིན། ཉེ་སྔོན་རྨས་སྐྱོན་ཕོག་པའི་བློ་དཔལ་བཟང་ཞུ་བ་ཞིག་གིས། རྒྱ་གཞུང་གི་བཙན་གནོན་ལ་ངོ་རྒོལ་བྱས་ཏེ་བཀག་ཉར་ཁང་དུ་རང་ཤི་བརྒྱབ་ཡོད་པ་དང་། ཉིན་དེར་ཉེ་སྔོན་རྨས་སྐྱོན་ཕོག་པའི་གསར་བུ་རང་ལོ་ ༢༢ ཡིན་པ་ཞིག་ཀྱང་བཀག་ཉར་ཁང་དུ་འདས་གྲོངས་སུ་གྱུར་ཡོད་ལ། ཕྱི་ཚེས་ ༡༢ ཉིན་མེ་མདའ་ཕོག་པའི་དབང་གྲགས་ལགས་ཀྱི་བུ་ཀུན་དགའ་ཤེས་རབ་སོགས་རྨས་མ་ཁག་ཅིག་ལའང་སྨན་ཁང་གིས་མདེལ་མགོ་ཕྱིར་བཏོན་རོགས་བྱས་མེད་སྟབས། ད་ལྟ་ཛ་དྲག་གི་གནས་སུ་ལྷུང་ཡོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༢ ཉིན་ཤུག་པ་སྡེ་བའི་བོད་མི་ཚོས་སྡེ་དཔོན་དབང་གྲགས་ལགས་འཕྲལ་དུ་ཀློད་གྲོལ་གཏོང་དགོས་པའི་ཞུ་འབོད་སྐབས། རྒྱ་གཞུང་གིས་ཉེན་རྟོག་པ་དང་དམག་མི་མང་པོ་ཆེད་གཏོང་གིས་དྲག་གནོན་བྱས་ཏེ་མི་ ༡༠ ཙམ་ལ་རྨས་སྐྱོན་བཟོས་ཡོད་པ་མ་ཟད། ཡུལ་མི་མང་པོ་འཛིན་བཟུང་བྱས་པའི་ཁྲོད་ནས་མི་ ༢༥ ཡི་མིང་ཐོ་ཕྱི་ཕྱོགས་སུ་འབྱོར་བ་རྣམས་ནི། སྡེ་དཔོན་གཞོན་པ་དབང་ཆེན་ལགས་དང་ཤུག་པ་མེ་རུ་དགོན་པའི་སྒྲུབ་དཔོན་ཀརྨ་རིན་ཆེན་ལགས། རང་ཤི་བརྒྱབ་པའི་བློ་དཔལ་བཟང་། པདྨ་རྒྱལ་ལོ། ཡོན་ཏན་སྐྱབས། བཀྲ་དགའ། ཀུན་དགའ་བསྟན་འཛིན། ཟླ་བ་བཀྲ་ཤིས། ཀརྨ་བཀྲ་ཤིས། བཀྲ་ཤིས་མགོན་པོ། ཀུན་དགའ་ཤེས་རབ། ཡེ་ཤེས། འཇམ་དབྱངས་བསོད་ནམས། བཀྲ་ཤིས། ཚེ་རིང་། རྒྱ་ལུག ཐུབ་བསྟན་ཆོས་འཕེལ། བླ་སྐྱབས། བུ་མོ། མཚོ་རྒྱ། སྦུ་རྒྱལ། དབྱངས་ཅན་ལྷ་མོ། ཚེ་སྐྱབས་དབང་མོ། ཡེ་ཤེས་སྒྲོལ་མ། དབྱངས་ཅན་དཔལ་མོ་བཅས་ཡིན་པ་དང་། དྲག་གནོན་ཁྲོད་ཁོང་ཚོར་རྨས་སྐྱོན་ཕོག་སྟེ་སྐྱེ་རྒུ་སོགས་སུ་སྨན་བཅོས་བྱེད་དུ་ཕྱིན་ཡོད་ཀྱང་། རྒྱ་གཞུང་གིས་མུ་མཐུད་སྨན་བཅོས་བྱེད་དུ་མ་བཅུག་པར་བཀག་ཉར་ཁང་དུ་བཙན་ཁྲིད་བྱས་ཡོད་པ་མ་ཟད། ཕྱི་ཕྱོགས་ནས་འབྲེལ་གཏུག་བྱེད་པར་བཀག་སྡོམ་གྱིས་ནད་གཡོག་དང་སྨན་བཅོས་ཀྱི་མཐུན་རྐྱེན་མེད་པར་བཟོས་ཡོད་འདུག གནས་ཚུལ་མཁོ་སྤྲོད་པས་བརྗོད་པ་ལྟར་ན། དང་ཐོག་སྡེ་བའི་སྐྱེས་པ་ཚང་མ་ལོ་ཕྱུག་གི་བཙོན་ཁང་དུ་བཅུག་ཡོད་ཀྱང་ཕྱིས་སུ་མ་ཤོང་པར་ལྷག་མ་རྣམས་སྨན་ཁང་གསར་པའི་ནང་བཀག་ཉར་བྱེད་དགོས་བྱུང་ཡོད་ལ། སྤྱིར་བཏང་ཤུག་པ་སྡེ་བའི་ནང་མི་གྲངས་ ༧༠༠ ཙམ་ཡོད་ཀྱང་། ད་ལྟའི་ཆར་ཤུག་པ་སྡེ་བའི་ནང་རྒན་རྒོན་དང་བྱིས་པ། བུད་མེད་བཅས་ལས་ལྷག་མེད་པར་སྡེ་བའང་སྟོང་སེང་སེང་ལྟ་བུ་ཞིག་ཏུ་གྱུར་ཡོད་འདུག དེ་རིང་ཕྱི་ཚེས་ ༡༨ ཉིན་ནས་བཟུང་ཉིན་གྲངས་ ༣ རིང་ཨ་རིའི་ཉིག་ཡོག་ཏུ་ཡོད་པའི་འཛམ་གླིང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་དབུས་ལས་ཁུངས་མདུན་དུ། ས་གནས་བོད་རིགས་ཚོས་ཕྱི་དྲོའི་ཆུ་ཚོད་ ༤ པ་ནས་ ༧ པ་བར་གདུང་སེམས་མཉམ་སྐྱེད་ཆེད་ངོ་རྒོལ་ཞུ་འབོད་ཀྱི་ལས་འགུལ་ཞིག་མགོ་འཛུགས་བྱ་རྒྱུ་ཡིན་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
