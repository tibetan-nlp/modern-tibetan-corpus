སྐྱབས་བཅོལ་པ་ས་ཡ་ ༥༡།༢ ལྷག་འདུག

ཁ་སང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་སྐྱབས་བཅོལ་ལྷན་ཁང་གིས་འཛམ་གླིང་སྐྱབས་བཅོལ་བའི་སྐོར་གྱི་གསལ་བསྒྲགས་ཤིག་སྤེལ་བའི་ནང་བོད་མི་སྐྱབས་བཅོལ་བ་ཚོར་རོཊ་སྐྱོར་བྱ་རྒྱུ་འཁོད་འདུག ཁ་སང་ཕྱི་ཚེས་ ༢༠ ཉིན་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་སྐྱབས་བཅོལ་ལྷན་ཁང་གིས་ཐེངས་ ༡༤ པའི་འཛམ་གླིང་གི་སྐྱབས་བཅོལ་ཉིན་མོ་སྲུང་གཙི་དང་བསྟུན་ཏེ་ ༢༠༡༣ ལོའི་སྐྱབས་བཅོལ་པའི་སྙན་ཐོ་ཞིག་བཏོན་ཡོད་པའི་ནང་། སེ་རི་ཡ་དང་ཨབ་གྷན། སོ་མ་ལི་ཡ་ (Somalia)། སུ་ཌན་(Sudan)། ཀོན་གྷོ་ (Congo)། འབར་མ། ཨི་རག ཨེ་རི་ཊི་ཡ་ (Eritrea) བཅས་གཙོས་རྒྱལ་ཁབ་ཁག་ནང་ཆབ་སྲིད་དང་ཆོས་ལུགས། མི་རིགས་བཅས་ཀྱི་གནད་དོན་ཐོག་ནས་ཟིང་འཁྲུག་བྱུང་སྟེ། ༢༠༡༣ ལོའི་ནང་འཛམ་གླིང་ནང་གི་སྐྱབས་བཅོལ་པའི་གྲངས་འབོར་ས་ཡ་ ༥༡།༢ ལྷག་ལ་འཕར་ཡོད་པ་དང་། ལྷག་པར་དུ་སེ་རི་ཡའི་མང་ཚོགས་ཀྱི་འཚོ་བའི་གནས་སྟངས་ཇེ་སྡུག་ཏུ་འགྲོ་བཞིན་ཡོད་སྟབས། རྒྱལ་སྤྱིའི་ཚོགས་པ་ཁག་གིས་འགྲོ་བ་མིའི་བདེ་རྩའི་ངོ་བོའི་ཐོག་ནས་རོགས་སྐྱོར་ཡོང་བར་འབོད་སྐུལ་ཞུས་ཡོད་འདུག རྒྱལ་ཚོགས་ཆེས་མཐོའི་སྐྱབས་བཅོལ་རྟོགས་ཞིབ་ཚོགས་པ(The UN High Commissioner for Refugees)ཞེས་པས་གསར་འགོད་ལྷན་ཚོགས་ཤིག་སྐོང་ཚོགས་གནང་སྟེ་བོད་ནས་བལ་ཡུལ་བརྒྱུད་རྒྱ་གར་དུ་སྐྱབས་བཅོལ་ཡོང་མཁན་བོད་མི་རྣམས་བལ་ཡུལ་དུ་སྡོད་རིང་ཟས་གོས་སོགས་གང་སྤྱིའི་མཐུན་རྐྱེན་སྦྱར་རྒྱུ་མ་ཟད། རྒྱ་གར་དུ་བདེ་འབྱོར་ཐུབ་རྒྱུར་རོགས་རམ་གནང་རྒྱུའི་ཁས་ལེན་ཡང་གནང་ཡོད་འདུག གསར་འགོད་གསལ་བསྒྲགས་ནང་། འཛམ་གླིང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་ཆེས་མཐོའི་སྐྱབས་བཅོལ་རྟོགས་ཞིབ་ཚོགས་པས་ བོད་ནས་བལ་ཡུལ་བརྒྱུད་རྒྱ་གར་དུ་རང་གི་རྩ་བའི་བླ་མ་མཇལ་ཆེད་བྲོས་བྱོལ་དུ་ ཕེབས་མཁན་བོད་མི་རྣམས་རྒྱ་གར་དུ་བདེ་ཐང་ངང་འབྱོར་ཐུབ་རྒྱུར་མཐུན་འགྱུར་ རོགས་རམ་གནང་དང་གནང་བཞིན་པ་མ་ཟད། བོད་མི་སྐྱབས་བཅོལ་བ་ཁོང་རྣམས་བལ་ཡུལ་དུ་སྡོད་རིང་ཆེས་མཐོའི་སྐྱབས་བཅོལ་རྟོགས་ཞིབ་ཚོགས་པས་སྨན་བཅོས་སོགས་ཀྱི་མཐུན་འགྱུར་ཡང་གནང་བཞིན་ཡོད། ཅེས་སོགས་འཁོད་འདུག མཉམ་འབྲེལ་རྒྱལ་ཚོགས་སྐྱབས་བཅོལ་ལྷན་ཁང་གི་འགན་འཛིན་སྐུ་ཞབས་གྷོ་ཊིར་རེ་སི་ (Antonio Guterres)ཡིས་བྷེ་བྷེ་སེའི་གསར་འགོད་པར། དེང་སང་འཛམ་གླིང་ནང་རྙོག་འཛིང་མང་པོ་བྱུང་ནས་སྐྱབས་བཅོལ་པའི་གྲངས་འབོར་འཕེལ་ཏེ་དཀའ་ངལ་སྣ་མང་འཕྲད་ཀྱི་ཡོད་པ་དེ་དག་སུས་ཀྱང་སེལ་ཐུབ་ཀྱི་མེད་པ་མ་ཟད། རྒྱལ་ཚོགས་བདེ་འཇགས་ལྷན་ཚོགས་ཀྱང་བེད་མེད་ལྟ་བུར་ཆགས་ཡོད་སྟབས། རྒྱལ་ཁབ་ཚང་མས་བསམ་ཤེས་དང་མཉམ་འབྲེལ་ཐོག་ནས་གདོང་ལེན་བྱེད་དགོས་ཀྱི་འདུག་ཅེས་བརྗོད་འདུག འདས་པའི་ལོ་ངོ་གསུམ་ལྷག་རིང་སེ་རི་ཡའི་ནང་ཟིང་འཁྲུག་བྱུང་སྟེ་ཁྱོན་ས་ཡ་ ༢ ལྷག་ཁྱིམ་མཚེས་རྒྱལ་ཁབ་ཨི་རག་དང་ལེ་བྷེ་ནོན་ (Lebanon)། ཇོར་ཌན་ (Jordan)། ཏུར་ཀི་ (Turkey) བཅས་ནང་སྐྱབས་བཅོལ་དུ་ཕྱིན་ཡོད་ཅིང་། ད་སྔ་སེ་རི་ཡའི་ནང་མི་མང་བརྒྱ་ཆ་ ༡༠།༨ ཙམ་ལ་འཐུང་ཆུ་དང་ཟ་མ། འཕྲོད་བསྟེན། སྡོད་གནས་སོགས་ཀྱི་དཀའ་ངལ་ཛ་དྲག་ཆགས་ཡོད་སྟབས། མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱིས་རྒྱལ་སྤྱིའི་ཚོགས་སྡེ་ཁག་གིས་ཁ་ཡོད་ལག་ཡོད་ཀྱི་རོགས་སྐྱོར་ཡོང་བར་འབོད་སྐུལ་བྱས་ཡོད། འཛམ་གླིང་གི་སྐྱབས་བཅོལ་ཉིན་མོ་འདི་ནི། ༢༠༠༠ ལོའི་ཟླ་ ༡༢ པའི་ཚེས་ ༤ ཉིན་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱིས་གཏན་འབེབས་བྱས་ཤིང་། ༢༠༠༡ ལོའི་ཟླ་ ༦ ཚེས་ ༢༠ ཉིན་ནས་བཟུང་ལོ་ལྟར་ཉིན་མོ་འདི་སྲུང་གཙི་གནང་གི་ཡོད་འདུག གསར་འགོད་པ། ལྷ་སྐྱབས་རྒྱལ།
