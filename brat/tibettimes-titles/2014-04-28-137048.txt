ཨོ་བྷ་མ་ལྷོ་ཀོ་རི་ཡར་སྐོར་སྐྱོད།

དེ་རིང་ཨོ་བྷ་མ་ལྷོ་ཀོ་རི་ཡར་གཟིགས་སྐོར་གནང་བ་དང་འབྲེལ་བྱང་ཀོ་རི་ཡས་རྡུལ་ཕྲན་མཚོན་ཆ་ཚོད་ལྟ་བྱེད་འཆར་ལ་ངོ་རྒོལ་གྱི་རྣམ་འགྱུར་བསྟན་ཡོད་འདུག ཨོ་བྷ་མའི་ཉིན་བརྒྱད་རིང་གི་ཨེ་ཤ་ཡའི་མཐུན་གྲོགས་རྒྱལ་ཁབ་བཞིའི་ནང་སྐོར་སྐྱོད་གནང་གི་ཡོད་པ་ལྟར། དེ་རིང་ལྷོ་ཀོ་རི་ཡའི་ནང་སྐོར་སྐྱོད་མགོ་ཚུགས་གནང་ཡོད་ཅིང་། ཨོ་བྷ་མས་ལྷོ་ཀོ་རི་ཡའི་སྲིད་འཛིན་ལྕམ་སྐུ་པག་གྷེན་ཧེ་ (Park Geun-hye)ལགས་དང་ཐུག་འཕྲད་གནང་སྟེ། བྱང་ཀོ་རི་ཡས་རྡུལ་ཕྲན་མཚོན་ཆ་ཐོན་བཞི་པ་ཚོད་ལྟ་བྱ་འཆར་ཡོད་པ་དེར་གདོང་ལེན་དང་། ཨ་རིའི་དམག་དཔུང་མུ་མཐུད་ལྷོ་ཀོ་རི་ཡའི་ནང་གཏན་སྡོད་བྱེད་ཕྱོགས། ཞི་བདེ་རྒྱ་མཚོའི་མངའ་ཁུལ་གྱི་བདག་དབང་རྩོད་ལེན་ལ་ཁ་གཏད་གཅོག་ཐབས། ཨ་རི་དང་ཚོང་འབྲེལ་བྱེད་ཤུགས་ཆེ་རུ་གཏོང་ཐབས་སོགས་ཀྱི་སྐོར་ལ་གྲོས་བསྡུར་གནང་ཡོད་པ་མ་ཟད། ཉེ་ཆར་ལྷོ་ཀོ་རི་ཡའི་མཚོ་གྲུ་འགྱེལ་ཏེ་མི་མང་པོ་རྐྱེན་ལམ་དུ་ཕྱིན་པར་ཐུགས་གསོའི་འཚམས་འདྲི་ཡང་ཞུས་ཡོད་འདུག ཨོ་བྷ་མས་བརྗོད་དོན། “ ལྷོ་ཀོ་རི་ཡ་ནི་ཨ་རིའི་མཐུན་གྲོགས་རྒྱལ་ཁབ་ཅིག་དང་གྲོགས་པོ་ཞིག་ཡིན་སྟབས། ངས་ནང་མི་དང་མི་མང་ཚོའི་ཚབ་ཞུས་ཏེ་གྲུ་གཟིངས་འགྱེལ་ཏེ་འདས་གྲོངས་སུ་གྱུར་བའི་བུ་ཕྲུག་གི་ཕ་མ་དག་ལ་གདུང་སེམས་མཉམ་སྐྱེད་ཞུ་བཞིན་ཡོད།” ཅེས་དང་། ཞི་བདེ་རྒྱ་མཚོའི་བདག་དབང་གི་དཀའ་རྙོག་དེ་རྒྱ་ནག་དང་འབྲེལ་བ་ཆགས་ཡོད་སྟབས། རྒྱ་ནག་གིས་ཞི་མོལ་བརྒྱུད་ནས་འགྲིག་འཇགས་ཡོང་ཐབས་བྱ་དགོས་པ་དང་། བྱང་ལྷོ་ཀོ་རི་ཡས་ལོ་མང་རིང་རྒྱལ་སྤྱིའི་ཁྲིམས་ལུགས་དང་འགལ་བའི་བྱ་སྤྱོད་བརྩམས་དང་རྩོམ་བཞིན་ཡོད་པར། ཨ་རིས་རྒྱལ་སྤྱིའི་ཁྲིམས་ལྟར་ཐག་གཅོད་བྱ་རྒྱུ་ཡིན་སྐོར་སོགས་བརྗོད་འདུག བྱང་ཀོ་རི་ཡའི་ཕྱི་སྲིད་ལྷན་ཁང་གིས། ཨོ་བྷ་མའི་སྐོར་སྐྱོད་དེ་ཨེ་ཤ་ཡའི་མངའ་ཁུལ་ནང་ཨ་རིའི་དམག་དོན་སྟོབས་ཤུགས་རྒྱ་སྐྱེད་གཏོང་ཐབས་ཤིག་ཡིན་སྟབས། ཨེ་ཤི་ཡའི་རྒྱལ་ཁབ་ཁག་གིས་ཨ་རིའི་ངོ་སྲུང་སྲིད་ཇུས་ལ་མགོ་བསྒུར་དགོས་དོན་མེད་སྐོར་བརྗོད་འདུག   ལྷོ་ཀོ་རི་ཡའི་སྲིད་འཛིན་ལྕམ་སྐུ་པག་གྷེན་ཧེ་ཡིས། སྲིད་འཛིན་ཨོ་བྷ་མའི་གདུང་སེམས་མཉམ་སྐྱེད་ཀྱི་འཚམས་འདྲི་འདིས། ལྷོ་ཀོ་རི་ཡའི་མང་ཚོགས་ཚོར་རྩ་ཆེའི་བློ་སྟོབས་ཤིག་སྦྱིན་ཡོད་སྟབས། རྒྱལ་ཁབ་གཉིས་ཀྱི་འབྲེལ་ལམ་སྔར་ལྷག་འཕེལ་རྒྱས་འགྲོ་རྒྱུ་མ་ཟད། ཞི་བདེ་རྒྱ་མཚོའི་བདག་དབང་རྩོད་ལེན་ཐད་འཕྱར་གྲུ་མཉམ་འདེགས་བྱས་ཏེ་བཙན་རྒྱལ་རིང་ལུགས་དག་ལ་འགོག་རྒོལ་བྱ་རྒྱུ་ཡིན་སྐོར་བརྗོད་འདུག བྱང་ཀོ་རི་ཡས་ ༢༠༠༦ དང་ ༢༠༠༩། ༢༠༡༣ བཅས་ལོ་གསུམ་ནང་རྡུལ་ཕྲན་མཚོན་ཆ་ཚོད་ལྟ་བྱས་ཏེ། ཨེ་ཤ་ཡ་དང་འཛམ་གླིང་ཞི་བདེ་སྲུང་སྐྱོབ་ལ་འཇིགས་སྣང་སྐུལ་ཡོད་སྟབས། ཨ་རི་གཙོས་རྒྱལ་སྤྱིའི་སྒྲིག་འཛུགས་ཁག་གིས་སྐྱོན་བརྗོད་བྱས་ཡོད་པ་མ་ཟད། མ་འོངས་པར་རྡུལ་ཕྲན་མཚོན་ཆ་ཚོད་ལྟ་བྱས་ཚེ་རྒྱལ་ཁབ་ཁག་གིས་ཁ་གཏད་གཅོག་རྒྱུ་ཡིན་སྐོར་བསྒྲགས་ཡོད། ཨོ་སི་ཊོ་ལི་ཡའི་རྒྱལ་ཡོངས་གཙུག་ལག་སློབ་ཆེན་གྱི་ཀོ་རི་ཡའི་གནད་དོན་ཆེད་མཁས་པ་ལེའུ་ནི་ཌི་པེ་ཌོར་(Leonid Petrov) ལགས་ཀྱིས། “ངའི་བསམ་པར། བྱང་ཀོ་རི་ཡའི་ངོས་ནས་ཨ་རིའི་སྲིད་འཛིན་ཨོ་བྷ་མ་ལྷོ་ཀོ་རི་ཡའི་ནང་སྐོར་སྐྱོད་གནང་བ་དང་བསྟུན་ཏེ་རྡུལ་ཕྲན་མཚོན་ཆ་ཚོད་ལྟ་བྱེད་མིན་གང་ལྟར་ཡང་། བྱང་ཀོ་རི་ཡའི་རྡུལ་ཕྲན་མཚོན་ཆའི་ཚོད་ལྟ་རྦད་དེ་ཕྱིར་འཐེན་བྱེད་མི་སྲིད།” ཅེས་བརྗོད་འདུག ཆབ་སྲིད་དབྱེ་ཞིབ་པ་ཚོས། སྲིད་འཛིན་ཨོ་བྷ་མའི་ཨེ་ཤ་ཡའི་སྐོར་སྐྱོད་དེ་ནི། རྒྱ་ནག་དང་བྱང་ཀོ་རི་ཡར་ཆབ་སྲིད་ཀྱི་རྣམ་འགྱུར་ཞིག་ཡིན་ཡང་། ཨེ་ཤ་ཡའི་མངའ་ཁུལ་ལ་ཨ་རིའི་སྟོབས་ཤུགས་བསླེབ་ཆེད་ཡིན་སྟབས། མ་འོངས་པར་ཨ་རིའི་མཐུན་ཕྱོགས་རྒྱལ་ཁབ་ཁག་དང་། རྒྱ་ནག་གི་མཐུན་ཕྱོགས་རྒྱལ་ཁབ་ཁག་གི་བར་དབང་ཆའི་འཐབ་རྩོད་ཤུགས་ཆེ་རུ་འགྲོ་སྲིད་སྐོར་བརྗོད་འདུག ༢༠༡༣ ལོར་ལྕམ་སྐུ་པག་གྷེན་ཧེ་ལགས་ལྷོ་ཀོ་རི་ཡའི་སྲིད་འཛིན་ཆགས་རྗེས། ཁོ་མོས་ལྷོ་བྱང་ཀོ་རིའི་འབྲེལ་ལམ་ཡར་རྒྱས་གཏོང་རྒྱུ་ནི་གལ་ཆེན་ཞིག་ཏུ་རྩིས་ཏེ། བྱང་ཀོ་རི་ཡ་ལྷོ་ཀོ་རི་ཡའི་མཐུན་གྲོགས་རྒྱལ་ཁབ་ཡག་པོ་ཞིག་ཆགས་རྒྱུར་འབད་བརྩོན་བྱ་རྒྱུ་ཡིན་སྐོར་བསྒྲགས་ཀྱང་། ད་ལྟ་ལྷོ་བྱང་གཉིས་ཀྱི་འབྲེལ་ལམ་ཇེ་སྡུག་ཏུ་འགྲོ་བཞིན་ཡོད་འདུག སང་ཉིན་སྲིད་འཛིན་ཨོ་བྷ་མ་ལྷོ་ཀོ་རི་ཡའི་མངའ་ཁུལ་ནང་ཡོད་པའི་ཨ་རིའི་དམག་མི་ ༢༨༥༠༠ གི་ས་ཚིགས་ཁག་ཅིག་ནང་འཚམས་གཟིགས་དང་འབྲེལ། དེ་རྗེས་མཱ་ལེ་ཤི་ཡ་དང་ཧྥི་ལེ་ཕིན་ནང་ཕེབས་འཆར་ཡོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
