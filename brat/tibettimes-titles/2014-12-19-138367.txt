ལ་དྭཊ་མི་མང་གིས་ཞི་རྒོལ་བྱས།༼པར༽

ཁ་སང་ཕྱི་ཚེས་ ༡༨ ཉིན་ལ་དྭགས་ཀྱི་རྒྱལ་ས་སླེ་རུ། ལ་དྭགས་ནང་པའི་ལྷན་ཚོགས་ཀྱི་གོ་སྒྲིག་འོག ལ་དྭགས་དགོན་པ་ཚོགས་པ་དང་སླེ་ཡི་ཤུ་ཚོགས་པ་སོགས་ཀྱིས་སྣེ་ཁྲིད་དེ། མི་མང་བརྒྱ་ཕྲག་མང་པོས་བྷོ་ཊི་སྐད་ཡིག་ནི་ང་ཚོའི་ངོ་རྟགས་ཡིན་པས། ལ་དྭགས་པ་དང་ཧི་མ་ལ་ཡའི་རི་རྒྱུད་ཀྱི་མི་མང་གིས་དུས་ཡུན་རིང་པོ་ནས་བྷོ་ཊིའི་སྐད་ཡིག་རྒྱ་གར་གཞུང་གི་རྩ་ཁྲིམས་ལེའུ་བརྒྱད་པའི་ནང་བཅུག་པའི་དགོས་འདུན་བཏོན་ཡོད་ཀྱང་། ཅིའི་ཕྱིར་ད་ལྟའི་བར་མ་ཚུད་པ་རེད་ཅེས་པ་སོགས་ཀྱི་འཕྲེད་དར་བཟུང་སྟེ་ཞི་བའི་རྔམ་སྟོན་ཁྲོམ་སྐོར་གཞི་རྒྱ་ཆེན་པོ་ཞིག་བྱས་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
