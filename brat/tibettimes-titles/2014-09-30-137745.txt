གངས་སྐོར་ལམ་ཁའི་རྙོག་གླེང་།

རྒྱ་གར་སྲིད་བློན་མུ་དྷི་སྲིད་གཞུང་གིས་ཧིན་རྡུའི་གངས་རིན་པོ་ཆེ་དང་མ་ཕམ་གཡུ་མཚོར་གནས་སྐོར་དུ་འགྲོ་མཁན་ཚོར་ལ་དྭགས་བདེན་མཆོག་དང་ཀུ་མོན། ཀིན་ནཱོར་བཅས་ཀྱི་ལམ་ཁ་མ་ཕྱེས་པར་སྣ་ཐོད་ལ་ནས་འགྲོ་རྒྱུ་བྱས་པ་དེར་ས་གནས་མང་ཚོགས་ཚོས་སྐྱོན་བརྗོད་བྱེད་བཞིན་ཡོད་འདུག ཉེ་ལམ་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་རྒྱ་གར་ནང་ཉིན་གསུམ་རིང་གི་གཟིགས་སྐོར་སྐབས། རྒྱ་དཀར་ནག་བར་ཚོང་འབྲེལ་གཙོས་པའི་ལས་དོན་ཁག་ ༡༢ ལ་མིང་རྟགས་བཀོད་ཡོད་པའི་ཁྲོད། རྒྱ་གར་གཞུང་གིས་རྒྱ་གར་གྱི་ཧིན་རྡུའི་གནས་མཇལ་བ་ཚོ་གངས་རིན་པོ་ཆེ་དང་མ་ཕམ་གཡུ་མཚོར་གནས་སྐོར་དུ་འགྲོ་བར་འགྲོ་གྲོན། འགྲིམ་བཞུད་བདེ་འཇགས་སོགས་ཀྱི་ཆེད་དུ། ལ་དྭགས་ས་གནས་བདེན་མཆོག་དང་ཨོ་ཊིར་ར་ཀན་མངའ་སྡེའི་ས་གནས་ཀུ་མོན་ (Kumaon)། ཧི་མ་ཅལ་མངའ་སྡེའི་ས་གནས་ཀིན་ནཱོར་ (Kinnaur) བཅས་ནང་གི་ས་མཚམས་ཁག་སྒོ་ཕྱེ་རྒྱུ་མིན་པར། འབྲས་ལྗོངས་ས་མཚམས་སྣ་ཐོད་ལ་ནས་གཞིས་ཀ་རྩེ་བརྒྱུད་གནས་མཇལ་དུ་འགྲོ་རྒྱུའི་ལམ་ཁ་ཕྱེ་རྒྱུའི་གྲོས་མཐུན་བྱུང་ཡོད་སྟབས། ཨོ་ཊིར་ར་ཀན་མངའ་སྡེའི་སྤྱི་ཁྱབ་བློན་ཆེན་ཧ་རེ་ཞི་ (Harish Rawat) ལགས་དང་ལ་དྭགས་ས་གནས་ཀྱི་ཀོང་རེ་སི་ཚོགས་པ་གཙོས་གོང་གི་ས་གནས་མང་ཚོགས་ཀྱིས་མུ་དྷི་གཞུང་གི་ཐག་གཅོད་དེར་སྐྱོན་བརྗོད་བྱེད་བཞིན་ཡོད་འདུག ཨོ་ཊིར་ར་ཀན་མངའ་སྡེའི་སྤྱི་ཁྱབ་བློན་ཆེན་ཧ་རེ་ཞི་ལགས་ཀྱིས་མངའ་སྡེ་ནང་ཡོད་པའི་བྷེ་ཇེ་པེ་གྲོས་ཚོགས་འཐུས་མི་ཚོས་སྲིད་བློན་མུ་དྷི་མཆོག་གི་མདུན་ཅོག་སྟེང་གནད་དོན་འདི་འཁྱེར་ཏེ་ཡང་བསྐྱར་བསམ་ཞིབ་བྱེད་དགོས་པའི་རེ་སྐུལ་ཡི་གེ་ཞིག་ཕུལ་ཡོད་པ་དང་། ཁོང་གིས་ “ཐག་གཅོད་དེས་རྒྱ་གར་རྒྱལ་ཡོངས་དད་ལྡན་མང་ཚོགས་ཀྱི་རྩ་བའི་དད་སེམས་ལ་ཕོག་ཐུག་ཚབས་ཆེན་བཟོས་ཡོད་”ཅེས་བརྗོད་འདུག ལ་དྭགས་ས་གནས་ཀྱི་ཀོང་རེ་སི་ཚོགས་པས། ཐེངས་འདིར་བྷེ་ཇེ་པེ་གཞུང་གིས་ལ་དྭགས་མི་མང་ལ་མགོ་སྐོར་བཏང་བ་མ་ཟད། བྷེ་ཇེ་པེ་གྲོས་ཚོགས་འཐུས་མི་ཐུབ་བསྟན་ཆོས་དབང་གིས་དབུས་གཞུང་ལ་གནས་ཚུལ་མཁོ་སྤྲོད་བྱེད་པར་འཐུས་ཤོར་སོང་བས་ཁོང་ལ་འགན་འཁུར་ཆ་ཚང་ཡོད་སྟབས། སྲིད་གཞུང་གིས་ལ་དྭགས་མི་མང་ལ་ཕྱོགས་ལྷུང་བྱས་པའི་ཐག་གཅོད་ལ་ས་གནས་ཀོང་རེ་སི་ཚོགས་པས་མཐའ་གཅིག་ཏུ་ངོ་རྒོལ་ཡོད་ཚུལ་སོགས་བརྗོད་ཡོད་འདུག དེ་ནི་ཟླ་སྔོན་མའི་སྲིད་བློན་མུ་དྷི་ལ་དྭགས་དང་ཀར་གྷེལ་ནང་གཟིགས་སྐོར་སྐབས་ལ་དྭགས་རང་སྐྱོང་རི་ལྗོངས་ཡར་རྒྱས་ལྷན་ཚོགས་ཀྱིས་བདེན་མཆོག་ས་མཚམས་ལ་གཞུང་ལམ་འཛུགས་སྐྲུན་དང་ཚོང་ལས་བྱེད་སྒོ་མགོ་འཛུགས་བྱེད་དགོས་པའི་དགོས་འདུན་ཁག་ཅིག་བཏོན་ཡོད་འདུག ལོ་འགའི་གོང་ནས་རྒྱ་གར་གྱི་ཧིན་རྡུའི་གནས་མཇལ་བ་ཚོ་ལྡི་ལི་ནས་བལ་ཡུལ་རྒྱལ་ས་ཀ་ཐ་མན་ཊྲོ་རུ་སོང་སྟེ། དེ་ནས་བལ་བོད་ས་མཚམས་གྲོ་མོ་བརྒྱུད་དེ་ཉིན་ ༡༠ ཙམ་རིང་གངས་རིན་པོ་ཆེ་དང་མ་ཕམ་གཡུ་མཚོ་གཉིས་ལ་གནས་མཇལ་བྱེད་ཐུབ་ཀྱི་ཡོད་པ་དང་། གཞན། ཨོ་ཊིར་ར་ཀན་མངའ་སྡེའི་ས་གནས་ཀུ་མོན་ (Kumaon) གྱི་ལེ་པོ་ལེ་ཁུ་ (Lipulekh) ལ་ནས་སྤུ་ཧྲང་བརྒྱུད་ནས་ཉིན་ ༢༧ ཙམ་རིང་རྐང་ཐང་ལ་སོང་ན་གནས་མཇལ་ཐུབ་ཀྱི་ཡོད་ཅིང་། ད་ལྟ་སྣ་ཐོད་ལ་བརྒྱུད་ཉིན་ ༨ ཙམ་ནང་འགོར་འདུག   ལོ་རེའི་ནང་རྒྱ་གར་ནང་གི་ཧིན་རྡུའི་ཆོས་ལུགས་རྗེས་འབྲང་པ་ཚོ་གངས་རིན་པོ་ཆེ་དང་མ་ཕམ་གཡུ་མཚོ་གཉིས་ལ་འགྲོ་མཁན་ཇེ་མང་དུ་འགྲོ་བཞིན་ཡོད་ལ། རྒྱ་ནག་ཕྱི་སྲིད་ལས་ཁུངས་ཀྱིས་བརྗོད་པ་ལྟར་ན། ༢༠༡༣ ལོའི་ནང་རྒྱ་གར་ནས་གནས་མཇལ་དུ་ཡོང་མཁན་ ༡༤༠༨༤ ལྷག་ཡོད་འདུག ལྷག་པར་དུ་གྷོ་ཇི་ར་ཐའེ་མངའ་སྡེ་ནས་གནས་མཇལ་བ་ ༦༠༠༠ ལྷག་རེ་ལོ་ལྟར་ཡོང་གི་ཡོད་ཚུལ་སོགས་བརྗོད་ཡོད་འདུག ཧིན་རྡུའི་ཆོས་ལུགས་ལ་མཚོན་ན། གངས་རིན་པོ་ཆེ་ནི་ལྷ་དབང་ཕྱུག་གི་སྒོམ་སྒྲུབ་ཉམས་ལེན་གནང་བའི་གནས་མཆོག་ཅིག་ཡིན་པར་ངོས་འཛིན་བྱེད་ཀྱི་ཡོད་པ་དང་། ཧིན་རྡི་ཆོས་ལུགས་པ་ཚོས་མ་ཕམ་གཡུ་མཚོའི་ཆུ་འཐུང་ན་ལོ་བརྒྱ་ལྷག་གི་སྡིག་ཡོངས་རྫོགས་སེལ་ཐུབ་པ་མ་ཟད། ལྷ་དབང་ཕྱུག་གི་གནས་སུ་འགྲོ་ཐུབ་པའི་བཤད་ཚུལ་ཞིག་ཡོད་འདུག དེ་མིན། བན་དང་བོན། གཅེར་བུ་བ་སོགས་ཀྱིས་ཀྱང་རང་རང་གི་ངོས་འཛིན་དང་བཤད་རྒྱུན་ལྟར་གངས་རིན་པོ་ཆེ་དང་མ་ཕམ་གཡུ་མཚོ་གཉིས་གནས་ཧ་ཅང་རྩ་ཆེན་པོར་བརྩིས་ཏེ་གནས་སྐོར་དུ་འགྲོ་སྲོལ་ཞིག་ཡོད་འདུག གསར་འགོད་པ། ལྷ་སྐྱབས་རྒྱལ།
