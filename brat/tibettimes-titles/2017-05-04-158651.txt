སྤྱི་ཚོགས་བརྒྱུད་ལམ་ཁག་ཏུ་ད་ལོའི་ཕྱི་ཟླ་ ༡༡ པའི་ནང་༧གོང་ས་མཆོག་གིས་རྒྱ་གར་འབྲས་ལྗོངས་མངའ་སྡེ་ར་ཝང་ལར་དུས་འཁོར་དབང་ཆེན་གནང་རྒྱུ་ཡིན་སྐོར་ཁྱབ་པ་དེ་བདེན་པ་མིན་ཡང་། དེ་རིང་འདི་གས་དགའ་ལྡན་ཕོ་བྲང་ལ་བཀའ་འདྲི་ཞུས་པར་༧གོང་ས་མཆོག་གིས་སི་ཀེམ་ནང་གསུང་ཆོས་བསྩལ་རྒྱུའི་ཞལ་བཞེས་གནང་ཡོད་པ་དངོས་ཡོད་ཡིན་སྐོར་གསུངས་བྱུང་། གཟའ་མིག་དམར་ཉིན་ངོ་དེབ་བརྒྱུད་དེ་འདི་ལོའི་ཕྱི་ཟླ་ ༡༡ པའི་ནང་༧གོང་ས་མཆོག་གིས་རྒྱ་གར་བྱང་ཕྱོགས་སི་ཀེམ་མངའ་སྡེ་ནང་དུས་འཁོར་དབང་ཆེན་སྩོལ་རྒྱུ་ཡིན་པའི་གནས་ཚུལ་ཞིག་ཁྱབ་པ་དང་བསྟུན་ཏེ། བོད་ཕྱི་ནང་གཉིས་ཀྱི་སྤྱི་ཚོགས་བརྒྱུད་ལམ་ཁག་ཏུ་གླེང་སློང་ཆེན་པོ་བྱེད་བཞིན་ཡོད་ཀྱང་། གཞུང་འབྲེལ་གསར་ལམ་ཁག་ཏུ་གནས་ཚུལ་དེའི་སྐོར་གང་ཡང་བཀོད་མེད། ཟླ་སྔོན་མར་༸གོང་ས་མཆོག་རྒྱལ་ས་ལྡི་ལིར་མཁས་དབང་ཨེམ་ཨལ་སེན་ལྡེའི་མཆོག་གི་གཟེངས་རྟགས་བཞེས་བར་ཕེབས་པ་དང་བསྟུན་ཏེ། ཕྱི་ཚེས་ ༢༧ ཉིན་རྒྱ་གར་འབྲས་ལྗོངས་མངའ་སྡེའི་བློན་ཆེན་ཁག་ཅིག་༸གོང་ས་མཆོག་ལ་མཇལ་ཁར་བཅར་ཏེ། མ་འོངས་པར་འབྲས་ལྗོངས་ཀྱི་ས་གནས་ར་ཝང་ལར་དུས་འཁོར་དབང་ཆེན་ཞིག་སྩོལ་གནང་ཡོང་བའི་རེ་འདུན་ཞུས་ཡོད་འདུག ཕྱི་ཚེས་ ༢༧ གྱི་དགོང་དྲོའི་ཆུ་ཚོད་ ༩།༥༨ ཐོག་ (སི་ཀེམ་གྱི་ང་རོ། The voice of Sikkim)ཞེས་པའི་གཞུང་འབྲེལ་ངོ་དེབ་སྟེང་། སི་ཀེམ་མངའ་སྡེའི་གཞུང་གིས་༸གོང་ས་ཏཱ་ལའི་བླ་མ་མཆོག་ལ་མ་འོངས་པ་ར་ཝང་ལར་དུས་འཁོར་དབང་ཆེན་སྩོལ་གནང་ཡོང་བའི་རེ་འདུན་ཞུས་ཡོད་པ་དང་། ད་དུང་ “མཇལ་ཁ་ལེགས་འགྲུབ་བྱུང་སོང་། གནས་ཚུལ་ཞིབ་ཕྲ་ཡོང་བར་སྒུག་ཡོད།”ཅེས་འཁོད་ཡོད་པ་ལས་༧གོང་ས་མཆོག་གིས་དུས་འཁོར་དབང་ཆེན་ཞལ་བཞེས་གནང་ཡོད་མེད་གསལ་ཁ་བཀོད་མེད། གནས་ཚུལ་ཅུང་ཁ་གསལ་ཤོས་ཤིག་ནི་ (སི་ཀེམ་སྲིད་གཞུང་། Government of Sikkim)ཞེས་པའི་སྟེང་། “༸གོང་ས་ཏཱ་ལའི་བླ་མ་མཆོག་གིས་སི་ཀེམ་ནང་དུས་འཁོར་དབང་ཆེན་གནང་རྒྱུ་མིན་ཡང་། དེ་དང་འབྲེལ་ཡོད་ཀྱི་གསུང་ཆོས་གནང་སྟེ་སི་ཀེམ་མི་མང་ལ་ཆོས་འབྲེལ་འཇོག་རྒྱུ་ཐུགས་གཏན་འཁེལ་ཡོད།” ཅེས་གསལ་བསྒྲགས་སྤེལ་ཡོད་པ་དང་། གསུང་ཆོས་དེ་ག་དུས་གནང་རྒྱུ་ཡིན་མིན་གྱི་ཟླ་ཚེས་ད་དུང་གཏན་འཁེལ་ཐུབ་མེད་ཀྱང་། སྐུ་སྒེར་ཡིག་ཚང་ནས་ཆིབས་སྒྱུར་ཟླ་ཚེས་གང་མགྱོགས་འབྱོར་འཆར་ཡོད་སྟབས། རྗེས་སུ་སླར་ཡང་གསལ་བསྒྲགས་སྤེལ་འཆར་ཡོད་སྐོར་སོགས་བཀོད་ཡོད། དེ་རིང་འདི་གའི་གསར་འགོད་པས་དགའ་ལྡན་ཕོ་བྲང་གི་དྲུང་ཆེ་འཆི་མེད་རིག་འཛིན་མཆོག་ལ་བཀའ་འདྲི་ཞུས་པར། ཁོང་གིས་ ཉེ་ཆར་༧གོང་ས་མཆོག་ལྡི་ལིར་བཞུགས་སྐབས་སི་ཀེམ་མངའ་སྡེའི་གཞུང་གི་བློན་ཆེན་ཁ་ཤས་ཕེབས་ཏེ། སི་ཀེམ་ནང་དུས་འཁོར་དབང་ཆེན་ཞིག་གནང་ཐུབ་ན་ཞེས་ཞུས་ཡོད་པ་དངོས་ཡོད་རེད། ཡིན་ནའང་སྐབས་དེར་༸གོང་ས་མཆོག་གིས་ཁྱེད་རང་ཚོས་དུས་འཁོར་དབང་ཆེན་ཞུ་བ་ལས་ཉིན་ཤས་རིང་གསུང་ཆོས་ཞུ་ཐུབ་ན་ལེགས་པའི་དགོངས་ཚུལ་གསུངས་བ་ལྟར། ཕྱི་ཟླ་ ༡༠ དང་ ༡༡ ལ་སི་ཀེམ་མངའ་སྡེར་གནམ་གཤིས་ཡག་པོ་ཡོད་སྟབས། བྱུང་ན་ད་ལོའི་ཕྱི་ཟླ་ ༡༠ དང་ ༡༡ དང་། མ་བྱུང་ན་ལོ་རྗེས་མའི་ནང་ཕེབས་རྒྱུ་ཡིན་ཞེས་བཀའ་གནང་ཡོད་པ་རེད། ཅེས་གསུངས་བྱུང་། གསར་འགོད་པ།
