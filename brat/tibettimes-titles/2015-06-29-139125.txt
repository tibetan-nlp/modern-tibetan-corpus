རྒྱ་ནག་ནང་དུག་རྫས་སྤྱོད་མཁན་ཁྲི་ ༡༤༠༠
འདུག

ཉེ་ལམ་རྒྱ་ནག་དབུས་གཞུང་དུག་རྫས་གཏན་འགོག་ལྷན་ཁང་གི་འགན་འཛིན་ལིའུ་ཡའེ་ཅིང་གིས་ ༢༠༡༤ ལོའི་རྒྱ་ནག་ནང་གི་དུག་རྫས་སྐོར་གྱི་སྙན་ཐོ་ཞིག་ཁྱབ་བསྒྲགས་བྱས་པའི་ནང་། ཕྱི་ལོ་ ༢༠༡༤ ལོ་གཅིག་པུར་རྒྱ་ནག་ནང་དུག་རྫས་སྤྱོད་མཁན་བཞི་འབུམ་བརྒྱད་ཁྲི་ ༤༨༠༠༠༠ ཙམ་རྩད་ཆོད་ཡོད་པ་དང་། ཁྱོན་བསྡོམས་ད་ལྟ་རྒྱ་ནག་ནང་དུག་རྫས་སྤྱོད་མཁན་བྱེ་བ་གཅིག་ས་ཡ་བཞི་ ༡༤༠༠༠༠༠༠ ལས་བརྒལ་ཡོད་ཚུལ་བཀོད་འདུག རྒྱ་ནག་གི་གསལ་བསྒྲགས་ལྟར་ན། དུག་རྫས་སྤྱོད་མཁན་གྱི་གྲངས་འབོར་དེ་རྒྱ་ནག་གི་མི་གྲངས་དང་བསྡུར་ན་མི་ ༡༠༠ ནང་དུ་དུག་རྫས་སྤྱོད་མཁན་གཅིག་རེ་བྱུང་ཡོད་པ་དང་། རྫོང་དང་གྲོང་ཁྱེར་ ༣༠༤༨ ནང་ཁྱབ་ཡོད་འདུག རྒྱལ་སྤྱིའི་འཕྲོད་བསྟེན་སྒྲིག་འཛུགས་ཀྱིས་རྒྱ་ནག་གཞུང་གི་གསལ་བསྒྲགས་གྲངས་ཐོ་དེར་ཡིད་ཆེས་བྱེད་ཀྱི་མེད། དེ་ནི་རྒྱལ་སྤྱིའི་ཐོག་ད་ལྟ་དུག་རྫས་སྤྱོད་མཁན་དུང་ཕྱུར་ ༢ ལྷག་ཡོད་སྟབས། རྒྱ་ནག་གི་གྲངས་འབོར་དེས་རྒྱལ་སྤྱིའི་དུག་རྫས་ངན་གོམ་པ་སྤྱིའི་བརྒྱ་ཆ་ ༧ ལས་ཟིན་མེད། ཚེས་ ༢༦ ཉིན་སླར་ཡང་རྒྱ་ནག་གིས་གནས་ཚུལ་སྤེལ་བའི་ནང་། རྒྱ་ནག་གཞུང་གིས་ད་ལོའི་ཕྱི་ཟླ་ ༡ ཚེས་ ༡༤ ཉིན་བུད་མེད་རང་ལོ་ ༤༣ ཅན་ཞིག་འཛིན་བཟུང་གིས་ཞིབ་བཤེར་བྱས་པ་བརྒྱུད་དེ། མོ་རང་གི་གཟུགས་པོའི་ནང་དུག་རྫས་སྐྱེལ་འདྲེན་བྱས་ཡོད་པ་ཤེས་རྟོགས་བྱུང་ཡོད་པ་དང་། མཐར་མོ་རང་ལ་ལོ་ ༡༥ ཡི་ཁྲིམས་ཐག་བཅད་ཡོད་པ་མ་ཟད། རྒྱ་སྒོར་ཁྲི་ ༡༥ ཡི་ཉེས་ཆད་བཏང་སྟེ་གྱོད་དོན་དེ་དང་འབྲེལ་བའི་མི་སྣ་མང་པོ་ཞིག་ལ་ཞིབ་བཤེར་བྱེད་བཞིན་ཡོད་འདུག  དེ་ནི་སྔར་ལོའི་ཕྱི་ཟླ་ ༩ ཚེས་ ༡༡ ཉིན་དུག་རྫས་ཚོང་པ་ཞིག་གི་མངག་བཀོད་ལྟར། མོ་རང་གི་ཁོག་པའི་ནང་དུག་རྫས་ཁུག་མ་ ༦༤ ཙམ་བཞག་ནས་ཞེས་ཁྲང་ནས་ཁྲུན་ཏུའུ་རུ་སོང་སྟེ་ཚོང་བསྒྱུར་བྱས་ཡོད་པའི་བདེན་དཔང་བྱུང་ཡོད་འདུག ཟླ་འདིའི་ཚེས་ ༢༥ ཉིན། རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་དང་སྲིད་བློན་ལི་ཁེ་ཆང་སོགས་པེ་ཅིན་མི་དམངས་ཚོགས་ཁང་ཆེན་མོའི་ནང་། རྒྱ་ནག་གི་དུག་རྫས་སྔོན་འགོག་གི་མི་སྣ་ཚོར་ཐུག་འཕྲད་བྱས་ཏེ། དུག་འགོག་འཐབ་རྩོད་གཏིང་ཟབ་བྱས་ཏེ་དུག་རྫས་མགྱོགས་མྱུར་ངང་ཁྱབ་བརྡལ་དུ་འགྲོ་བཞིན་པར་འགོག་རྒོལ་བྱས་ཏེ། མི་དམངས་ཀྱི་འཕྲོད་བསྟེན་ལ་འགན་ལེན་ཡག་པོ་བྱ་དགོས་ཚུལ་བརྗོད་ཡོད་འདུག རྒྱ་ནག་གཞུང་དུག་འགོག་ལྷན་ཁང་གི་སྙན་ཐོ་དེ་བཏོན་པ་དང་བསྟུ་ཏེ། རྒྱ་ནག་གཞུང་འབྲེལ་གསར་ལམ་ཁག་ཏུ་བློ་ཚབ་རང་བཞིན་གྱི་ཐོག་ནས་རྒྱ་ནག་ནང་གི་དུག་རྫས་ངན་གོམས་སྐོར་གྱི་གནས་ཚུལ་མང་པོ་ཞིག་རིམ་བསྟུད་དུ་ཐོན་དང་འཐོན་བཞིན་འདུག
