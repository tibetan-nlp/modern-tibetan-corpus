ལྡན་མ་བློ་ཆོས་རིན་པོ་ཆེ་སྐུ་གཤེཊ།

ཉེ་ལམ་རྣམ་རྒྱལ་གྲ་ཚང་གི་མཁན་ཟུར་ལྡན་མ་བློ་ཆོས་རིན་པོ་ཆེ་མཆོག་སྐུ་གཟུགས་བདེ་མིན་གྱི་དབང་གིས་སྙུང་གཞི་བཞེས་ཏེ། ཁ་སང་ཉིན་རྒྱབ་ཆུ་ཚོད་ ༣།༣༠ ཡོལ་ཙམ་ལ་ཁོང་གི་གཟིམ་ཤག་ཏུ་སྐུ་གཤེགས་ཡོད་པ་དང་ད་ལྟ་རིན་པོ་ཆེ་མཆོག་ཐུགས་དམ་ལ་བཞུགས་ཡོད་པ་མ་ཟད། ཟླ་འདིའི་ཚེས་ ༡༥ ཉིན་༧གོང་ས་མཆོག་ལ་མཇལ་ཁ་ཞུས་པ་སོགས་ངོ་མཚར་བའི་རྟགས་དང་བཅས་སྐུ་གཤེགས་ཡོད་འདུག ལྡན་མ་རྗེ་བཙུན་བློ་བཟང་འོད་ཟེར་ཆོས་དབྱིངས་རྒྱ་མཚོ་དཔལ་བཟང་བོ་མཆོག་ནི། ༡༩༢༨ ལོའི་ཟླ་ ༨ ཚེས་ ༣ ཉིན་ཁམས་སྒ་བ་འགུལ་གཡས་གྲོང་གི་ཁྱིམ་ཚང་ཞིག་ཏུ་ཡབ་དངོས་གྲུབ་རིག་འཛིན་དང་ཡུམ་བདེ་མཚོ་བཟང་མོ་གཉིས་ཀྱི་སྲས་སུ་སྐུ་འཁྲུངས་ཤིང་། དགུང་ལོ་དྲུག་ལ་སོན་སྐབས་པཎ་ཆེན་སྐུ་ཕྲེང་དགུ་པ་མཆོག་གིས་ཟིལ་དཀར་དགོན་པའི་མཁས་པའི་དབང་པོ་ཆོས་དབྱིངས་རྒྱ་མཚོ་མཆོག་གི་ཡང་སྲིད་དུ་ངོས་འཛིན་གནང་ཡོད་པ་དང་། དགུང་ལོ་ ༡༠ ལ་ཕེབས་སྐབས་སྤུག་བླ་མཆོག་སྤྲུལ་བློ་བཟང་ཐུབ་བསྟན་གྱི་དྲུང་ནས་དགེ་ཚུལ་གྱི་སྡོམ་པ་བཞེས། ༡༩༣༨ ལོར་ལྷ་སའི་འབྲས་སྤུངས་བློ་གསལ་གླིང་དུ་ཆོས་ཞུགས་མཛད་དེ་དགེ་བའི་བཤེས་གཉེན་སྟོང་དཔོན་རིན་པོ་ཆེ་མཆོག་དང་། བློ་གསལ་གླིང་གི་ཕ་ར་ཤག་སྐོར་མཁན་རིན་པོ་ཆེ་རྗེ་བཙུན་ཉི་མ་རྒྱལ་མཚན་མཆོག དེ་བཞིན་ཧྲེ་ཧོར་དགེ་བཤེས་བསོད་ནམས་མགོན་པོ་གཙོས་མཁས་གྲུབ་གཉིས་ལྡན་གྱི་དགེ་བའི་བཤེས་གཉེན་དུ་མའི་དྲུང་ནས་གཞུང་ཆེན་བཀའ་པོད་ལྔ་ལ་ཐོས་བསམ་མཛད། ༡༩༤༦ ལོར་སྐྱབས་རྗེ་ཡོངས་འཛིན་གླིང་རིན་པོ་མཆོག་གིས་དྲུང་ནས་བསྙེན་རྫོགས་ཀྱི་སྡོམ་པ་བཞེས་ཤིང་། དགུང་ལོ་ ༢༥ ལ་ཕེབས་སྐབས་ཏེ་༡༩༥༣ ལོར་ལྷ་ལྡན་ཆོས་འཕྲུལ་སྨོན་ལམ་སྐབས་དགེ་བཤེས་ལྷ་རམས་པའི་ཡོན་ཚད་ཨང་དང་པོ་བཞེས་རྗེས། དཔལ་ལྡན་རྒྱུད་སྨད་གྲྭ་ཚང་དུ་སྔགས་ཕྱོགས་ཀྱི་ཐོས་བསམ་སློབ་གཉེར་མཇུག་སྐྱོང་གནང་སྟེ་ ༡༩༥༨ ལོར་བསླབ་པ་མཐར་སོན་གནང་ཡོད་འདུག ༡༩༥༩ ལོར་རྒྱ་དམག་གིས་བོད་བཙན་བཟུང་བྱས་རྗེས། རིན་པོ་ཆེ་མཆོག་རྒྱ་གར་ལ་བཙན་བྱོལ་དུ་ཕེབས་ཏེ་ཝཱ་ར་ཎ་སིར་ལོ་གཉིས་དང་། ཀ་ལི་ཀ་ཏ་གཙུག་ལག་སློབ་ཆེན་ནང་ལོ་གཉིས་རིང་ཞིབ་འཇུག་སློབ་གཉེར་གནང་ཞིང་། དེ་བཞིན་ལ་དྭགས་མཚན་ཉིད་སློབ་གཉེར་ཁང་དུ་ལོ་ ༦ རིང་སློབ་སྤྱི་གནང་ཡོད་པ་དང་། ༡༩༦༧ ལོར་མ་ནཱ་ལི་དགོན་པ་ཞིག་གི་མཁན་པོ་གནང་བ་སོགས་ཀྱི་མཛད་འཕྲིན་བསྐྱངས་ཡོད་པ་མ་ཟད། ༡༩༨༦ ནས་ ༡༩༩༡ ལོའི་བར་རྣམ་རྒྱལ་གྲྭ་ཚང་གི་མཁན་པོ་གནང་སྟེ། གྲྭ་ཚང་གི་སྔགས་ཀྱི་འཛིན་གྲྭ་གསར་འཛུགས་སོགས་གནང་ཡོད་འདུག དེ་བཞིན། རིན་པོ་ཆེ་མཆོག་གིས་བོད་རྒྱུད་ནང་བསྟན་མི་ཉམས་རྒྱུན་འཛིན་ཐུབ་ཆེད་དྷ་ས་ཐེག་ཆེན་ཆོས་གླིང་གཙུག་ལག་ཁང་དང་གདན་ས་ཁག་གཙོས། རྒྱུད་གྲྭ་སྟོད་སྨད་ཀྱི་དགེ་འདུན་པ་ཚོར་མདོ་སྔགས་ཀྱི་གཞུང་ཆེན་ཁག་འཆད་ཁྲིད་མཛད་ཡོད་པ་མ་ཟད། རྒྱ་གར་དང་བོད་གཙོས་པའི་དོན་གཉེར་ཅན་མང་བོར་དབང་ལུང་མན་ངག་གནང་ཡོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
