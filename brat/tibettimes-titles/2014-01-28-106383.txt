མིང་ཚིག་བེད་སྤྱོད་གཏོང་ཕྱོཊ་ལ་གཟབ་རོཊ།

 ཁ་སང་གཟའ་མིག་དམར་ཉིན་དགའ་ལྡན་ཕོ་བྲང་ཡིག་ཚང་གིས་གནས་འཕྲིན་ཞིག་བསྐུར་ཏེ། བོད་སྐད་དང་བོད་ཡིག་གི་གསར་ལམ་ཁག་གི་མིང་ཚིག་བེད་སྤྱོད་གནང་ཕྱོཊ་ཐོག་དགོངས་བཞེས་གནང་དགོས་ཚུལ་གསུངས་འདུག ཟླ་འདིའི་ཚེས་ ༢༧ ཉིན་དགའ་ལྡན་ཕོ་བྲང་ཡིག་ཚང་གིས། བོད་སྐད་དང་བོད་ཡིག་གི་གསར་འགྱུར་བརྒྱུད་ལམ་ཁག་གི་ལས་འཛིན་རྣམ་པར་མིང་ཚིག་བེད་སྤྱོད་གཏོང་ཕྱོགས་ཐོག་དགོངས་བཞེས་གནང་དགོས་པའི་གནས་འཕྲིན་ཞིག་བསྐུར་བའི་ནང་། དགོངས་གསལ་རྒྱ་ཕྱོགས་ནས་ཡིག་ངག་གཉིས་ཐོག་ནས་ཡང་ནས་ཡང་དུ་བོད་སྐད་ཀྱི་སྐད་ཡིག་ཐོག་བཙན་འཛུལ་ལྟ་བུས། བོད་སྐད་ལ་གཟའ་ཉི་མ་ཞེས་མིང་ཡོད་མུས་ལ་”ཤིན་ཅི་”ཞེས་འབྲི་འབོད་བྱེད་པ་དང་། དེ་བཞིན་ལས་ཁུངས་ལ་”པུའི”དང་། “ ཐིང་” “ཁྲུའུ་” སོགས་དང་། ལས་ཁུངས་ཀྱི་གཙོ་འཛིན་ལ་ “པུའུ་ཀྲང་” སོགས་དང་། འགོ་ཁྲིད་ལ་ “ཀྲུའུ་ཞི་” དང་། “ཀྲུའུ་རིན་” དེ་བཞིན་ས་གནས་ཀྱི་རྫོང་ལ་ “ཤན་” དང་། དེའི་འོག་ལ་ “ཤང་” ཟེར་བ་ལྟ་བུ་ཡང་ཡང་བེད་སྤྱོད་བྱས་རྐྱེན་དེང་བོད་ནས་ཕེབས་མཁན་མང་པོས་ཀྱང་དེ་ལྟའི་བརྗོད་ཚིག་བེད་སྤྱོད་བྱེད་པ་དང་། འདི་གར་གནས་ཚུལ་ཕུལ་འབྱོར་ཁག་གི་ནང་དུའང་དེ་ལྟར་བྲིས་པ་ཡོང་གི་འདུག དེ་བཞིན་འབྲི་ཆུའི་སྟོད་རྒྱུད་ཀྱི་གྲོང་རྡལ་བོད་མིང་ལ་དགའ་སྐྱེ་རྒུ་མདོ་ཞེས་ལོ་ངོ་བརྒྱ་ཕྲག་མང་པོའི་གོང་ནས་ཡོད་དང་ཡོད་བཞིན་པར་རྒྱ་ཕྱོགས་ནས་དགོས་དབང་ལྟ་བུས་ “ ཡུས་ཧྲུའུ་” ཞེས་འབོད་འབྲི་བྱེད་བཞིན་ཡོད་པར། རང་ཕྱོགས་ནས་ཐུགས་སྣང་ཆུང་སྐྱོན་ལྟ་བུས་དེ་ལྟར་ལད་ཟློས་གནང་བ་དེས་རིམ་པས་བོད་ཀྱི་མིང་ཚིག་ངོ་མ་བརླག་ཉེན་དུ་མཐོང་པས། སླད་མིང་ཚིག་བེད་སྤྱོད་གནང་ཕྱོགས་ཐོག་དགོངས་བཞེས་ངེས་པར་གནང་འོས་དཔྱད་གཞིའི་ཚུལ་དུ། ཞེས་འཁོད་འདུག    གསར་འགོད་པ། དོན་གོ
