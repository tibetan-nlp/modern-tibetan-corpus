དྲག་ཤུཊ་ཀྱིས་ངོ་རྒོལ་བ་ཕྱིར་འབུད་བཏང་།

ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༤ ཉིན་གྱི་ཞོགས་པར་ཐའེ་ཝན་གཞུང་གིས་དྲག་ཤུགས་ལ་བརྟེན་ནས་ཞི་རྒོལ་བ་ཚོ་སྲིད་འཛིན་ལྷན་ཁང་ནས་ཕྱིར་འབུད་བཏང་ཡོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༣ ཉིན་གྱི་སྔ་དྲོར་གསར་འགོད་ལྷན་ཚོགས་ཤིག་གི་ཐོག ཐའེ་ཝན་སྲིད་འཛིན་མཱ་ཡེན་ཅིག་གིས། རྒྱ་ནག་དང་ཐའེ་ཝན་བར་གྱི་ཚོང་འབྲེལ་གན་རྒྱ་དེ་མུ་མཐུད་རྒྱུན་འཁྱོངས་བྱ་དགོས་ལུགས་བརྗོད་ཡོད་པར་བརྟེན། ཉིན་དེའི་མཚན་གུང་ཆུ་ཚོད་ ༡༢ ཙམ་ལ། ངོ་རྒོལ་བྱེད་མཁན་གྱི་སློབ་མ་དང་མི་མང་རྣམས་ཐའེ་ཝན་སྲིད་འཛིན་ལྷན་ཁང་གི་ཚོགས་རའི་ནང་འཛུལ་ཏེ་ངོ་རྒོལ་བྱས་ཡོད་པ་མ་ཟད། མི་བཅུ་གྲངས་ཤིག་ལས་ཁུངས་ནང་འཛུལ་ཡོད་སྟབས། སྲིད་འཛིན་ལྷན་ཁང་གི་མགོ་ཁྲིད་ཅཱང་ཡས་ཧྭ་ཡིས་བཀའ་རྒྱ་བཏང་སྟེ། ཉེན་རྟོག་པས་ངོ་རྒོལ་བ་ཚོར་གཅར་རྡུང་བཏང་ནས་ཕྱིར་འབུད་གཏོང་རྩིས་བྱས་ཡོད་ཀྱང་། ངོ་རྒོལ་བ་ཚོ་ཕན་ཚུན་འཐམས་ཏེ་ཕྱིར་འབུད་བཏང་ཐུབ་མེད་པ་དང་། ཕྱི་ཚེས་ ༢༤ ཉིན་གྱི་ཞོགས་པའི་ཆུ་ཚོད་ ༤ པ་ཙམ་ལ་ཉེན་རྟོག་པས་ཆུ་གཏོར་འཕྲུལ་ཆས་ཀྱིས་ངོ་རྒོལ་བ་ཚོ་སྲིད་འཛིན་ལྷན་ཁང་ནས་བཙན་གྱིས་ཕྱིར་འབུད་བཏང་ཡོད་པ་མ་ཟད། ཕྱི་ལོགས་སུ་ཡོད་པའི་ངོ་རྒོལ་བ་ཚོའང་ཁ་འཐོར་དུ་བཏང་ཡོད་འདུག ཕྱི་ཚེས་ ༢༤ ཉིན་གསར་འགོད་ལྷན་ཚོགས་ཤིག་གི་ཐོག སྲིད་འཛིན་ལྷན་ཁང་གི་མགོ་ཁྲིད་ཅཱང་ཡས་ཧྭ་ཡིས། ཐེངས་འདིའི་ངོ་རྒོལ་ལས་འགུལ་ཁྲོད་ཁྱོན་མི་གྲངས་ ༦༡ སྲིད་འཛིན་ལྷན་ཁང་གི་ལས་ཁུངས་སུ་འཛུལ་ཏེ་གཞུང་རྒྱུར་གཏོར་བརླག་བཏང་བ་མ་ཟད། སྲིད་འཛིན་དཔོན་གྱི་ལས་ཁུངས་ནང་འཛུལ་རྩིས་བྱས་ཡོད་སྟབས། ཁོ་ཚོ་ཚང་མ་འཛིན་བཟུང་བྱས་ཡོད་པ་དང་། དེ་དག་ལས་མི་ ༣༥ ལ་མུ་མཐུད་འདྲི་རྩོད་བྱེད་བཞིན་ཡོད་ལ། སྲིད་འཛིན་ལྷན་ཁང་དུ་ནང་སྲིད་དང་ཕྱི་འབྲེལ། དཔལ་འབྱོར། ལྷག་པར་དུ་རྒྱལ་ཁབ་བདེ་འཇགས་ཀྱི་ཡིག་ཆ་གལ་ཆེན་མང་པོ་གསོག་ཉར་བྱས་ཡོད་པས། ད་རེབ་ངོ་རྒོལ་བ་ཚོས་ནག་ཉེས་ཚབས་ཆེན་བཟོས་ཡོད་སྐོར་བརྗོད་འདུག་ཀྱང་། སྲིད་འཛིན་ལྷན་ཁང་ནས་ཐེངས་འདིའི་ངོ་རྒོལ་སྐབས་གཞུང་ཡིག་གལ་ཆེན་སོགས་བོར་བརླག་ཏུ་གྱུར་མེད་ལུགས་བསྒྲགས་འདུག གཞན་ཡང་། གསར་འགོད་ལྷན་ཚོགས་ཐོག་ཅཱ་ཡས་ཧྭ་ཡིས། ཉེན་རྟོག་པས་སྲིད་འཛིན་ལྷན་ཁང་སྲུང་སྐྱོབ་ཆེད་སྤེལ་བའི་བྱ་སྤྱོད་ལ་གཞི་མེད་སྐྱོན་འཛུགས་བྱས་ཏེ། གསར་ཁང་དང་མི་འགའ་རེས་རྨས་སྐྱོན་ཕོག་པའི་ངོ་རྒོལ་བ་ཚོའི་འདྲ་པར་བརྒྱུད་ལམ་དུ་བཏང་ནས་གནམ་བདེ་སྒོ་མོ་ཆེའི་དྲུག་བཞིའི་དོན་རྐྱེན་དང་འདྲ་བའི་དྲག་གནོན་གྱི་སྲིད་བྱུས་སྤེལ་གྱི་ཡོད་སྐོར་བརྗོད་པ་དེ། སྲིད་གཞུང་གིས་འཁྱོག་བརྗོད་ཚབས་ཆེན་ཞིག་ལ་ངོས་འཛིན་བྱེད་ཀྱི་ཡོད་ལུགས་སོགས་བརྗོད་ཡོད་འདུག ངོ་རྒོལ་བ་ཕྱིར་འབུད་བཏང་བའི་གནད་དོན་དེའི་ཐད། ཁ་སང་ཐའེ་ཝན་གྱི་སློབ་ཆེན་ཁག་གི་དགེ་རྒན་ཚོས་གྲུབ་པའི་༼ཐའེ་ཝན་དམངས་གཙོ་སྲུང་སྐྱོབ་ཚོགས་པ༽ཟེར་བས་གསལ་བསྒྲགས་ཤིག་སྤེལ་ཏེ། ཐའེ་ཝན་སྲིད་འཛིན་མཱ་ཡེན་ཅིག་དང་སྲིད་འཛིན་ལྷན་ཁང་གི་མགོ་ཁྲིད་ཅཱང་ཡས་ཧྭ་གཉིས་ཀྱིས། རྒྱ་ཆེའི་མི་མང་གི་རེ་འདོད་ལ་མཐོང་ཆུང་བྱས་པ་མ་ཟད། དྲག་ཤུགས་ལ་བརྟེན་ནས་མི་མང་མཐར་སྐྲོད་བཏང་བ་དེ་སྲིད་དབང་སྒེར་གཅོད་པའི་བྱ་སྤྱོད་ཅིག་ཡིན་སྟབས། ཅཱང་ཡས་ཧྭ་འཕྲལ་དུ་གནས་དབྱུང་གཏོང་དགོས་པ་མ་ཟད། ཐའེ་ཝན་གྱིས་སོ་སོའི་རྒྱལ་ཁབ་དེ་དམངས་གཙོའི་རྒྱལ་ཁབ་ཅིག་ཆགས་པའི་རེ་བ་ཡོད་ན། ད་ལྟ་སྤྱི་ཚོགས་ལས་འགུལ་དང་ངོ་རྒོལ་ལས་འགུལ་ཙམ་མ་ཡིན་པར་དམངས་གཙོའི་ཆབ་སྲིད་ལམ་ལུགས་ལ་བསྐྱར་ཞིབ་བྱེད་དགོས་སྟབས། སློབ་མ་ཚོས་སྤྱི་དམངས་སྲིད་ཁྲིམས་ཚོགས་འདུ་བསྡུ་བར་ནུས་ཤུགས་འདོན་དགོས་སྐོར་སོགས་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
