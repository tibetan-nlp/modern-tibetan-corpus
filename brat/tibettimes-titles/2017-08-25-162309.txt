དྷ་སར་ས་རུད་བྱུང་སྟེ་གཞུང་ལམ་འགག

ཆར་པ་རྒྱུན་རིང་བབས་པའི་རྐྱེན་གྱིས་བཞུགས་སྒར་དྷ་རམ་ས་ལའི་དྷ་སྟེང་བོད་ཁྱིམ་དུ་འགྲོ་སའི་ལམ་ཀ་ཞིག་ཏུ་ས་རུད་བྱུང་སྟེ་འོག་གི་གཞུང་ལམ་ཐོག་རླངས་འཁོར་འགྲོ་འོང་བྱེད་མི་ཐུབ་པར་གྱུར། དྷ་རམ་ས་ལར་ཆར་བ་ཤུགས་ཆེན་བབས་པའི་རྐྱེན་གྱིས་དེ་རིང་ཞོགས་པ་ཆུ་ཚོད་༨་༣༠ སྟེང་དྷ་ས་མེག་ལོར་གྷན་ཇི་ཡི་རླངས་འཁོར་འབབ་ཚིགས་ཀྱི་ཉེ་འཁྲིས་སུ་ས་རུད་བྱུང་སྟེ། དྷ་སྟེང་བོད་ཁྱིམ་སློབ་གྲྭར་འགྲོ་སའི་ལམ་ཀ་ཉེན་ཁ་ཅན་དུ་གྱུར་ཡོད་པ་མ་ཟད། གཤམ་འོག་ཏུ་ཡོད་པའི་དྷ་ཤོད་དུ་འགྲོ་སའི་གཞུང་ལམ་ནས་རླངས་འཁོར་འགྲོ་འོང་མི་ཐུབ་པར་གྱུར་འདུག འདི་གའི་གསར་འགོད་པ་ཡུལ་དངོས་སུ་འབྱོར་སྐབས་ས་བརྐོ་འཕྲུལ་འཁོར་ཞིག་གིས་གཞུང་ལམ་སྟེང་གི་ས་རྡོ་རྣམས་གཙང་བཟོ་བྱེད་བཞིན་འདུག་ལ། དྷ་ཤོད་དང་མེག་ལོར་བར་གྱི་གླ་འཁོར་ཁ་ལོ་བ་སན་ཇི་ཟེར་བ་ཞིག་གིས། ངས་ནམ་རྒྱུན་མེག་ལོར་ནས་དྷ་ཤོད་བར་གླ་གཏོང་རླངས་འཁོར་གཏོང་གི་ཡོད་པ་དང་། འོན་ཀྱང་།  ད་སྐབས་ཆར་དུས་ཡིན་སྟབས།  དྷ་ཤོད་ནས་མེག་ལོར་བར་གྱི་གཞུང་ལམ་ལ་ཆར་བས་གཏོར་སྐྱོན་མང་པོ་ཐེབས་ཀྱི་ཡོད་པ་ལྟར།  དེ་རིང་སྔ་དྲོར་ས་རུད་རྒྱུགས་ཏེ་གཞུང་ལམ་འགག་སོང་། ཉེ་ཆར་ཡང་དྷ་སྟེང་དང་དྷ་ཤོད་བར་ཡོད་པའི་རྒྱ་གར་དམག་སྒར་གྱི་ཉེ་འཁྲིས་སུ་ས་རུད་རྒྱུགས་ཏེ། ནམ་རྒྱུན་མེག་ལོར་ནས་མགོ་བཙུགས་ཏེ་ལྡི་ལིར་འགྲོ་བའི་འགྲུལ་སྐྱེལ་རླངས་འཁོར་ཆེ་གྲས་ཁག་དྷ་ཤོད་ནས་དྷ་སྟེང་ལ་ཡོང་མི་ཐུབ་པ་ཆགས་འདུག ཅེས་བརྗོད་སོང་། ད་རིང་གི་ས་རུད་ཀྱིས་གཞུང་ལམ་འགག་པ་དང་། དྷ་ཤོད་དང་མེག་ལོར་བར་འགྲོ་འོང་བྱེད་བཞིན་པའི་རླངས་འཁོར་དང་སྦག་སྦག རྐང་གསུམ་གླ་འཁོར་སོགས་འགྲོ་མི་ཐུབ་པར་གྱུར་བ་ལས། མི་དང་སེམས་ཅན། དེ་བཞིན་ཁང་པ་སོགས་ལ་གནོད་སྐྱོན་གང་ཡང་ཐེབས་མི་འདུག བརྒྱུད་ལམ་དུ་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། འདི་ལོའི་ཕྱི་ཟླ་ ༧ ནང་ཙམ་ནས་ཆར་དུས་མགོ་བཙུགས་ཏེ་ད་ལྟའི་བར་དུ་ཨེ་ཤ་ཡའི་ལྷོ་ཁུལ་རྒྱ་གར་དང་བལ་ཡུལ། རྒྱ་ནག བྷང་ལ་དེ་ཞི་སོགས་ཀྱི་ནང་ཉུང་མཐར་མི་གྲངས་ ༢༥༠ རྐྱེན་ལམ་དུ་ཤོར་ཡོད་པ་དང་། མི་གྲངས་འབུམ་གཉིས་ལྷག་གི་རྒྱུ་ནོར་ལ་གནོད་པ་ཐེབས་ཏེ་གནས་སྐབས་རིང་ས་གནས་གཞན་དུ་གནས་སྤོས་བྱེད་དགོས་བྱུང་ཡོད་འདུག གསར་འགོད་པ།  སྐལ་བཟང་སྤྱིན་པ།
