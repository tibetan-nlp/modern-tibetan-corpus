ཀྲུའུ་པིན་ཧྲན་ལ་ཞིབ་བཤེར་བྱས།

རྒྱ་ནག་ལྷོ་བརྒྱུད་ཞ་མིན་གྲོང་ཁྱེར་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་གིས་ཧུའུ་པེ་ཞིང་ཆེན་དམར་ཤོག་དྲུང་ཆེ་ཀྲུའུ་པིན་ཧྲན་གྱིས་སྤྱི་མང་གི་རོགས་དངུལ་དང་དངོས་ཟོན་ཁྱོན་རྒྱ་སྒོར་ས་ཡ་༤༠ ལྐོག་ཟ་བྱས་པའི་ནག་ཉེས་འོག་ཞིབ་བཤེར་བྱས་ཡོད་འདུག དེ་རིང་རོའུ་ཊར་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཁ་སང་རེས་གཟའ་ལྷག་པའི་ཉིན་རྒྱ་ནག་ལྷོ་བརྒྱུད་ཞ་མིན་ (Xiamen) གྲོང་ཁྱེར་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་གིས་ཧུའུ་པེ་ཞིང་ཆེན་དམར་ཤོག་དྲུང་ཆེ་ཀྲུའུ་པིན་ཧྲན་ (Zhou Benshun) གྱིས་ཧུའུ་པེ་ཞིང་ཆེན་དང་ཧུ་ནན་ཞིང་ཆེན་གཉིས་ནང་གོ་གནས་འཕོ་འགྱུར་དང་ཚོང་གཉེར་མཉམ་འབྲེལ། དེ་བཞིན། མི་སྒེར་བཅས་ཀྱི་མཐུན་འགྱུར་རོགས་དངུལ་ལྐོག་ཟ་དང་དངོས་ཟོག་སྒེར་སྤྱོད་བྱས་པའི་ནག་ཉེས་འོག་ཞིབ་བཤེར་བྱས་ཡོད་འདུག ཁྲིམས་ཁང་དེས། ཕྱི་ལོ་ ༢༠༠༣ ནས་ ༢༠༡༥ བར་ཀྲུའུ་པིན་ཧྲན་དང་ཁོའི་ནང་མི་ཚོས་གཞུང་སྒེར་གཉིས་ཀྱི་ཐེབས་རྩའི་མ་དངུལ་དང་དངོས་ཟོན་ཁྱོན་རྒྱ་སྒོར་ས་ཡ་ ༤༠ ལྐོག་ཟ་བྱས་ཡོད་པ་དང་། ཁྲིམས་ཁང་ནས་ཀྲུའུ་པིན་ཧྲན་གྱིས་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་ནག་ཉེས་ཁས་ལེན་བྱས་པ་མ་ཟད། དེ་ལ་འགྱོད་བཤགས་ཀྱང་བྱས་ཡོད་ལུགས་བཤད་འདུག ཀྲུའུ་པིན་ཧྲན་ནི་ད་ལྟ་ལོ་ ༦༦ ཡིན་ཞིང་ཁོ་སྔོན་ཆད་རྒྱ་ནག་དབུས་དམར་ཤོག་དྲུང་ཆེ་ཟུར་བ་ཀྲུའུ་ཡཱོང་ཁང་གི་ལས་གྲོགས་ཡིན་པ་དང་། སྐབས་དེར་ཁོས་ལོ་ངོ་ ༥ རིང་རྒྱ་ནག་དབུས་ཆབ་སྲིད་ལྷན་ཁང་དང་འཐུས་མི་ལྷན་ཚོགས་བཅས་ཀྱི་སྤྱི་ཁྱབ་དྲུང་ཆེའི་འགན་ཁུར་མྱོང་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༡༣ ལོར་ཧུའུ་པེ་ཞིང་ཆེན་དམར་ཤོག་དྲུང་ཆེར་བསྐོ་བཞག་བྱས་ཡོད་འདུག གནས་ཚུལ་ལྟར་ན། ཕྱི་ལོ་ ༢༠༡༢ ལོའི་རྒྱལ་ཡོངས་དམངས་ཆེན་སྐབས་ ༡༨ པ་ནས་ཞི་ཅིན་ཕིན་གྱི་ལྐོག་ཟ་རུལ་སུངས་ལ་རྒོལ་རྡུང་དང་གཙང་བཤེར་ལས་འགུལ་གཏན་འབེབས་དང་ལག་བསྟར་བྱས་པ་ནས་ད་ལྟའི་བར་དམར་ཤོག་ཚོགས་པའི་མཐོ་རིམ་དཔོན་རིགས་ ༡༢༠ ཙམ་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་ཉེས་མིང་འོག་འཛིན་བཟུང་བཀག་བསྐྱིལ་བྱས་ཡོད་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
