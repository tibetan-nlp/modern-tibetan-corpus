ཁྲིན་ལུང་གིས་དགོངས་སེལ་ཞུས་འདུག

ཁྲིན་ལུང་གི་བུ་ཧྥང་ཙུ་མེན་སྨྱོ་རྫས་ལྐོག་འཐེན་གྱི་དོགས་གཞིའི་འོག་འཛིན་བཟུང་བྱས་ཡོད་པར་ཁྲིན་ལུང་གིས་བུའི་ཚབ་བྱས་ཏེ་མི་མང་ཡོངས་ལ་དགོངས་སེལ་ཞུས་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༤ ཉིན་རྒྱ་ནག་པེ་ཅིན་ཉེན་རྟོག་ཁང་གིས། རྒྱ་ནག་གི་གློག་བརྙན་འཁྲབ་སྟོན་པ་གྲགས་ཅན་ཁྲིན་ལུང་གི་བུ་ཧྥང་ཙུ་མེན་དང་། ཐའེ་ཝན་གྱི་གློག་བརྙན་འཁྲབ་སྟོན་པ་ཁུ་ཀྲིན་ཏུང་གཉིས་སྨྱོ་རྫས་ལྐོག་འཐེན་གྱི་དོགས་གཞིའི་འོག་པེ་ཅིན་ནས་འཛིན་བཟུང་བྱས་ཡོད་པ་མ་ཟད། ཉེན་རྟོག་ཕྱོགས་ནས་ཧྥང་ཙུ་མེན་གྱི་ཁྱིམ་དུ་སྔོག་བཤེར་བྱས་ཏེ་སྨྱོ་རྫས་སྲང་ ༡༠༠ ལྷག་རྙེད་སོན་བྱུང་ཡོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༠ ཉིན་གྱི་མཚན་མོར་ཁྲིན་ལུང་གིས་ཝེ་པའོ་གནས་འཕྲིན་ཐོག ཁོ་རང་གི་བུའི་ཚབ་བྱས་ཏེ་མི་མང་ལ་དགོངས་སེལ་འཕྲིན་ཐུང་ཞིག་བཏང་བའི་ནང་། སྤྱི་མང་གི་མི་སྣ་ཞིག་ཡིན་པའི་ཆ་ནས་ང་རང་ཧ་ཅང་ངོ་ཚ་ཡི་འདུག་ལ། ཨ་ཕ་ཡིན་པའི་ཆ་ནས་ང་ལ་སེམས་ཁྲལ་ཞེ་དྲགས་འདུག ལྷག་པར་དུ་ངའི་བཟའ་ཟླ་ལ་སེམས་འཚབ་འདུག གཞོན་སྐྱེས་ཚོས་འདི་ནི་ཉེན་བརྡ་ཞིག་ཏུ་ངོས་འཛིན་གྱིས་སྨྱོ་རྫས་དང་རྒྱང་ཐག་རིང་པོ་བྱེད་པའི་རེ་བ་ཡོད། ནོར་འཁྲུལ་ཤོར་ན་མཇུག་འབྲས་ལ་རང་ཉིད་ཀྱིས་འགན་འཁུར་དགོས་པ་རེད། ང་ཁོ་བོའི་ཨ་ཕ་ཡིན་པས་མ་འོངས་པར་ཁོ་དང་མཉམ་དུ་གནད་དོན་ལ་གདོང་གཏོད་པར་མོས་མཐུན་ཡོད། ངེད་ཕ་བུ་གཉིས་ལ་ཐབས་ལམ་མེད་ཀྱང་འགན་འཁུར་རྒྱུ་ཡིན་པ་མ་ཟད། ངས་བུ་ཡི་ཚབ་བྱས་ཏེ་རྒྱ་ཆེའི་མི་མང་ལ་བསམ་པ་རྣམ་དག་གིས་དགོངས་སེལ་ཞུ། ཞེས་བྲིས་འདུག བརྒྱུད་ལམ་ཁག་ནས་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། ཁྲིན་ལུང་གིས་གནས་ཚུལ་ཤེས་རྟོགས་བྱུང་མ་ཐག་པེ་ཅིན་དུ་ཕྱིན་ཏེ། རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་ལའང་ཧྥང་ཙུ་མེན་སྐྱོབ་པའི་རེ་སྐུལ་བྱས་ཡོད་ཀྱང་། ཁྲིན་ལུང་རྒྱ་ནག་སྲིད་གྲོས་ཀྱི་རྒྱུན་ལས་དང་། སྲིད་གཞུང་གི་དཔོན་རིགས་དང་ཚོང་པ་ཚོར་འབྲེལ་བ་ཧ་ཅང་ཡག་པོ་ཡོད་ལ། ལྷག་པར་དུ་ཁྲིན་གྱི་འབྲེལ་འདྲིས་བྱེད་ས་མང་ཆེ་བ་རྒྱ་ནག་གི་སྲིད་འཛིན་ཟུར་པ་ཅང་ཙེ་མེན་གྱི་ཕྱོགས་གཏོགས་ཀྱི་མི་སྣ་ཡིན་སྟབས། ཞི་ཅིན་ཕིན་གྱིས་ཏག་ཏག་ཅང་ཙེ་མེན་སོགས་ཁོ་བོའི་ཕྱོགས་གཏོགས་ཚོ་ཕམ་པར་རྒྱག་པའི་ལྐོག་ཇུས་འཐེན་གྱི་ཡོད་པས། ཞི་ཡིས་ཁྲིན་ལུང་གི་རེ་སྐུལ་དང་ལེན་བྱས་མེད་ལུགས་བརྗོད་འདུག ཕྱི་ཚེས་ ༡༤ ཉིན་ཧྥང་ཙུ་མེན་དང་ཁུ་ཀྲིན་ཏུང་གཉིས་འཛིན་བཟུང་བྱས་རྗེས། རྒྱ་ནག་གཞུང་ཕྱོགས་ནས་ཉིན་དེར་ཁོ་གཉིས་ཀྱི་གཙོས་སྨྱོ་རྫས་ལྐོག་འཐེན་པ་འགའ་འཛིན་བཟུང་བྱས་ཡོད་ལུགས་ཙམ་ལས་གསལ་བོ་བརྗོད་མེད་ཀྱང་། ཀུ་ཤུ་ཉིན་རེའི་ཚགས་པར་ཁང་གིས་སྐབས་དེར་མི་ ༤ འཛིན་བཟུང་བྱས་ཡོད་པ་དང་། དེའི་ནང་གི་གཅིག་རྒྱ་ནག་གི་གླུ་མ་གྲགས་ཅན་དང་གློག་བརྙན་འཁྲབ་སྟོན་པ་ཁྲོ་ཞེ་ཡིན་པ་དང་། གཞན་ཞིག་ནི་པེ་ཅིན་མཁར་གྱི་སྐྱེས་གཟུགས་ལེགས་ཤིང་འབྱོར་ཕྱུག་ལྡན་པའི་ཕོ་གསར་བཞི་རུ་འབོད་པའི་ནང་གི་ཝང་ལི་ཟེར་བ་དེ་ཡིན་ལུགས་བརྗོད་ཀྱི་འདུག པེ་ཅིན་གྱི་ཁྲིམས་རྩོད་པ་ཁྲིན་ལུང་སན་གྱིས། ད་ལྟའི་གནས་བབ་ལ་བལྟས་ན། པེ་ཅིན་ཉེན་རྟོག་ཁང་གིས་ཧྥང་ཙུ་མེན་གྱི་ཁྱིམ་ནས་རྙེད་པའི་སྨྱོ་རྫས་ཉེས་དོན་དེ་ནག་ཉེས་རིམ་པ་གཉིས་པར་བརྩིས་འདུག་སྟབས། སྨྱོ་རྫས་གནོད་སྐྱོན་སྔོན་འགོག་བཅའ་ཁྲིམས་ཀྱི་དོན་ཚན་ལྔ་པའི་ནང་གསེས་ ༢ པ་གཞིར་བཟུང་ན། ཧྥང་ཙུ་མེན་ལ་ལོ་ ༥ ཡན་གྱི་བཙོན་འཇུག་ཁྲིམས་ཆད་དང་། རྒྱ་སྒོར་ཁྲི་ ༡༢༥ ཡི་ཉེས་ཆད་ཕོག་སྲིད་སྐོར་བརྗོད་འདུག་ཀྱང་། བརྒྱུད་ལམ་ཁ་ཤས་ན་ཧྥང་ཙུ་མེན་གྱིས་ལོ་ངོ་ ༨ ཡར་སྔོན་ནས་སྨྱོ་རྫས་སྤྱོད་ཀྱི་ཡོད་པས་སྲོག་ཁྲིམས་གཅོད་སྲིད་སྐོར་ཡང་བརྗོད་ཀྱི་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
