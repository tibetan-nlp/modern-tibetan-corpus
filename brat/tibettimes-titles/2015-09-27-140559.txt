དྷ་ས་ས་འཐུས་ཀྱི་ཚོཊ་དུས་བཞི་པ་ཚོཊ།

  ཁ་སང་ཕྱི་ཚེས་ ༢༦ ཉིན་དྷ་ས་མེག་གྷན་ས་གནས་ཁོར་ཡུག་ཚོགས་ཁང་དུ། སྐབས་བརྒྱད་པའི་དྷ་ས་ས་གནས་འཐུས་ཚོགས་ཀྱི་ཚོགས་དུས་བཞི་པ་ཚོགས་ཏེ། ས་གནས་འགོ་འཛིན་ལས་ཁུངས་ཀྱི་ལས་བསྡོམས་ཐོག་དྲི་བ་དྲིས་ལན་དང་ཁོར་ཡུག་གཙང་བཟོ་སོགས་ཀྱི་ཐོག་བགྲོ་གླེང་གནང་ཡོད་འདུག   ཚོགས་འདུའི་ཐོག་མར་དྷ་ས་ས་གནས་འཐུས་ཚོགས་ཀྱི་ཚོགས་གཙོ་ངག་དབང་ཐར་པ་ལགས་ཀྱིས། ནམ་རྒྱུན་ས་འཐུས་རྣམ་པ་ཁག་གཅིག་གིས་གཅིག་བྱེད་དགོས་འདུག གཉིས་བྱེད་དགོས་འདུག་གསུང་མཁན་མང་པོ་ཡོད་པ་རེད། དངོས་སུ་ལས་ཀ་བྱེད་ཡོང་དུས་དུས་ཚོད་གཏོང་ཐུབ་མཁན་ཁག་པོ་རེད་འདུག སྤྱི་པའི་དོན་དག་དེ་སྣང་མེད་གཏོང་བ་དེ་འདྲ་བྱེད་ཉན་གྱི་མ་རེད།  དྷ་སའི་རྒྱ་ཆེའི་མང་ཚོགས་རྣམ་པས་ང་ཚོས་བྱེད་ཀྱི་རེད། འཇོན་གྱི་རེད་བསམ་ནས་ཡར་བཞག་པ་རེད། མིང་དང་རྐུབ་སྟེགས་གཅིག་པུ་སྲུང་མཁན་ཞིག་ལ་ང་ཚོ་བཞག་མེད་པ་གསལ་པོ་རེད། ང་ཚོས་མི་མང་གི་ཆེད་དུ་ལས་ཀ་བྱེད་ཀྱི་ཡིན་ལབ་ནས་དགོངས་པ་མ་ཞུས་པར་བསྡད་བཞག་པ་ནང་བཞིན། སྤྱི་པའི་ལས་དོན་ཐོག་འགན་ཁུར་དགོས་པ་གལ་ཆེན་པོ་རེད། ཞེས་དང་།   ཁོང་གིས་མུ་མཐུད། ང་ཚོས་བསམ་བློ་རྒྱ་ཆེ་ཙམ་གཏོང་དགོས་པ་གལ་ཆེན་པོ་ཆགས་ཡོད་རེད། བོད་ས་ཤོར་སྐབས་རང་དབང་དང་མི་རིགས་ཀྱི་ཆེད་དུ་བོད་མི་འབུམ་བཅུ་གཉིས་ཀྱི་ཚེ་སྲོག་ཤོར་བ་རེད། དེའི་རྗེས་བཙོན་ཁང་ནང་དང་། གང་བྱུང་དམར་གསོད་བྱས་པ་དེ་ཚོ་གྲངས་ཀ་དེའི་ནང་ཚུད་ཡོད་པ་མ་རེད། ཕྱི་ལོ་༢༠༠༨ ལོའི་ས་བྱི་ཞི་རྒོལ་ཆེན་མོའི་སྐབས་བོད་མི་སྟོང་ཕྲག་མང་པོའི་སྲོག་ཤོར་ཡོད་པ་རེད། དེ་བཞིན་ཕྱི་ལོ་༢༠༠༩ ནས་ད་བར་རང་ལུས་མེར་སྲེག་གནང་མཁན་གྲངས་༡༤༢ ཆགས་ཡོད་པ་དང་། བཙན་བྱོལ་ནང་དུ་ཕྱི་ལོ་༡༩༩༨ ནས་ད་བར་རང་ལུས་མེར་སྲེག་གནང་མཁན་མི་གྲངས་༧ ཆགས་ཡོད། ཁོང་རྣམ་པས་ལོངས་སྤྱོད་ཕར་ཞོག་སྲོག་ཀྱང་ང་ཚོ་བོད་གངས་ཅན་ལྗོངས་ཀྱི་མི་རིགས་ དང་རིག་གཞུང་གི་ཆེད་དུ་བཏང་གནང་བ་དེ་སྣང་མེད་དུ་མི་འཇོག་པར། དེའི་ཆེད་དུ་བསམ་གཞིག་ནན་ཏན་གྱིས་ཕྱག་ལས་གནང་དགོས་པ་གལ་པོ་རེད། འདི་ཚོར་བསམ་ བློ་གཏང་ན་ཕྱོགས་རིས་དང་འཐེན་འཁྱེར་གྱི་བསམ་བློ་ཞིག་སྐྱེ་སའི་གཞི་མེད་པ་འདྲ་བོ་རེད། རྩ་བའི་ཆ་ནས་ང་ཚོ་ས་འཐུས་ནང་ཡག་པོ་མཐོང་གི་ཡོད། ཕན་ཚུན་བར་དགའ་མི་དགའི་དཀའ་ངལ་ཆེན་པོ་ཡོད་པ་མཐོང་གི་མི་འདུག སྤྱི་ཚོགས་ནང་དཀའ་ངལ་ཆུང་ཚག་རིགས་ཡོང་སྲིད་པ་རེད། གང་ཟག་སྒེར་ངོས་ནས་གང་ཐུབ་ ཐུབ་ཀྱིས་སྤྱི་ལ་ཕན་པའི་བསམ་བློ་གཏོང་ཁུལ་བྱེད་ཀྱི་ཡོད། གང་ལྟར་ང་ཚོས་ཕན་མ་ཐོགས་ཀྱང་གནོད་མ་སྐྱེལ་ཟེར་བའི་ཚིག་འདི་ང་ཚོས་ཡིད་ལ་བཟུང་ནས་ལས་དོན་སྒྲུབ་དགོས་པ་གནད་འགག་རེད། ཅེས་བརྗོད་འདུག   དྷ་ས་ས་གནས་མགོ་འཛིན་ལགས་ཀྱིས་ལས་འཆར་འགའ་འགྲུབ་མེད་པའི་སྐོར་ལ་ཁོང་གིས། ས་གནས་འགོ་འཛིན་ལགས་ཀྱིས་གནང་གི་ཡིན་གསུངས་ནས་ལས་དོན་འགོ་འཛུགས་མ་ཐུབ་པ་ཁ་ཤས་ཡོད་པ་རེད། དང་པོ་ཁོར་ཡུག་ཚོགས་ཁང་། ལས་ཁུངས་མ་ལག་དང་བཅས་པ་གསར་རྒྱག་གི་ལས་གཞི་འགོ་འཛུགས་བྱེད་འཆར་དེ་དང་། དྷ་སའི་ཐང་ཁྲོམ་ཀྱི་ཁྲོམ་ར་གསར་རྒྱག་གི་ལས་འཆར་བཅས་པ་འགྲུབ་ཡོད་མ་རེད། ལས་གཞི་འདི་ཚོ་ས་གནས་ལ་མཚོན་ན་ལས་གཞི་ཆེན་པོ་རེད་ལ་ཁ་ཚ་དགོས་གཏུགས་ཀྱི་ལས་ གཞི་ཞིག་ཀྱང་ཆགས་ཡོད་རེད། འདི་ཚོ་ས་འགོ་མཆོག་གི་ཐུགས་སྣང་གཙོ་བོ་གནང་དགོས་འདུག་ཟེར་ནས་འབོད་སྐུལ་ཞུ་ ཡི་ཡོད་ཅེས་སོགས་བརྗོད་ཡོད་པ་རེད།   གཞན་ཡང་། ཉིན་དེར་གཞུང་འབྲེལ་མྱ་ངན་གུས་འདུད་ཀྱི་གྲོས་ཆོད་དང་། དྷ་ས་ས་གནས་འགོ་འཛིན་ལས་ཁུངས་ཀྱི་ལས་བསྡོམས་སྐོར་ལ་དྲི་བ་དྲི་ལན་དང་།  བཞུགས་སྒར་ཁུལ་གྱི་ཁོར་ཡུག་གཙང་བཟོ། ས་གནས་ཡུལ་མིའི་མཐུན་ལམ། དེ་བཞིན་ས་གནས་མི་མང་གི་བདེ་སྡུག་སོགས་ཀྱི་ཐོག་ལ་བགྲོ་གླེང་གནང་ཡོད་པ་རེད།།   ངག་དབང་ཐར་པ་ལགས་ཀྱིས་གནས་ཚུལ་མཁོ་གཏོང་གནང་། ་
