སྒྲིག་གཞི་དོན་ཚན་བརྒྱད་གཏན་འབེབས་བྱས།

ཏང་གི་ཚོགས་ཆེན་ཐེངས་ ༡༩ བ་བདེ་ལེགས་ངང་འཚོགས་ཆེད་དམར་ཤོག་ཚོགས་པས་གསར་དུ་སྒྲིག་གཞི་དོན་ཚན་བརྒྱད་གཏན་འབེབས་བྱས་འདུག དེ་རིང་རྒྱ་ནག་རྒྱལ་ཡོངས་སུ་རྒྱལ་སྟོན་དུས་ཆེན་དང་ཟླ་མཆོད་དུས་ཆེན་གཉིས་སྲུང་གཙི་བྱེད་བཞིན་པའི་སྐབས་དང་བསྟུན་ཏེ། རྒྱ་ནག་གཞུང་གིས་རྒྱལ་ཡོངས་འཐུས་མི་ཚོགས་ཆེན་ཐེངས་ ༡༩ བ་བདེ་ལེགས་ངང་འཚོགས་ཆེད་གསར་དུ་སྒྲིག་གཞི་དོན་ཚན་བརྒྱད་གཏན་འབེབས་ཀྱིས་ཡོངས་ཁྱབ་གསལ་བསྒྲགས་སྤེལ་འདུག སྒྲིག་གཞི་དོན་ཚན་༨ ཀྱི་ནང་དོན་གཙོ་བོ་ནི། དུས་ཆེན་ལ་ཁ་གཡར་ཏེ་གཞུང་གི་ལས་བྱེད་ཚོས་ཕན་ཚུན་བར་ལྐོག་རྔན་སྤྲོད་ལེན་བྱས་མི་ཆོག་པ་དང་གཞུང་རྒྱུ་སྒེར་ཟ་བྱས་མི་ཆོག་པ། གཞུང་དོན་དང་འབྲེལ་མེད་ཀྱི་ལས་འགུལ་གོ་སྒྲིག་བྱས་མི་ཆོག་པ། དེ་བཞིན། གཉེན་སྟོན་དང་འདས་མཆོད། དགའ་སྟོན་སོགས་ཀྱི་བྱེད་སྒོ་ཁག་གཟབ་རྒྱས་ངང་བསྡུ་སྐོང་བྱས་མི་ཆོག་པ་སོགས་ཡིན་འདུག གསལ་བསྒྲགས་ཡིག་ཆའི་ནང་། སྒྲིག་གཞི་་དོན་ཚན་ ༨ དེ་ཞིང་ཆེན་དང་ཁུལ་རིམ་པའི་འབྲེལ་ཡོད་ལས་ཁུངས་སོ་སོས་འགན་འཁུར་ཏེ་དངོས་སུ་ལག་བསྟར་བྱེད་དགོས་ཚུལ་བཤད་འདུག གཞན། རྒྱ་ནག་གཞུང་གི་ལྐོག་ཟ་རུལ་སུངས་ལ་ཁ་གཡར་ཏེ་ཕྱི་ལོ་ ༢༠༡༣ ནས་ ༢༠༡༧ བར་མངའ་རིས་ས་ཁུལ་སྤུ་ཧྲེང་རྫོང་གི་རྫོང་དཔོན་ཟུར་བ་ཚེ་རིང་རྡོ་རྗེ་དང་། ཉིང་ཁྲི་གྲོང་ཁྱེར་སྨན་གླིང་རྫོང་ཁོར་ཡུག་སྲུང་སྐྱོང་ལས་ཁུངས་ཀྱི་འགན་འཛིན་ཚེ་རྡོར། དེ་བཞིན་ལྷོ་ཁ་གྲོང་ཁྱེར་ལྷུན་རྩེ་གྲོང་རྡལ་གྱི་རྟེའུ་ར་བསམ་གཏན་གླིང་དགོན་པའི་དོ་དམ་ལས་ཁུངས་ཀྱི་འགན་འཛིན་ཀུན་བཟང་ནོར་རྒྱས་སོགས་བོད་རིགས་ལས་བྱེད་ཁག་ཅིག་གི་གོ་གནས་མེད་པར་བཟོས་པ་མ་ཟད། སྒེར་དངུལ་གཞུང་བཞེས་བཏང་ཡོད་པ་རེད། གསར་འགོད་པ། སྐལ་བཟང་སྤྱིན་པ།
