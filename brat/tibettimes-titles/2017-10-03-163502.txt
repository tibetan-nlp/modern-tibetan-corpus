སྦེལ་ཀོབ་ལུཊ་བསམ་ལ་རྒྱལ་ཁ་ཐོབ།

རྒྱལ་ཡོངས་བོད་མིའི་སིལ་སྤོལ་འགྲན་བསྡུར་ཁྲོད་བདེ་སྐྱིད་སླར་གསོ་རུ་ཁག་གི་རྩེད་འགྲན་པ་རྡོ་རྗེ་རྣམ་རྒྱལ་ལགས་སུ་རྩེད་འགྲན་པ་རྩེ་ཕུད་ཀྱི་བྱ་དགའ་ཐོབ་སོང་། ཁ་སང་ཉིན་རྒྱབ་ཏུ་བཞུགས་སྒར་དྷ་རམ་ས་ལའི་གངས་སྐྱིད་སིལ་སྤོལ་རྩེད་ཐང་དུ། རྒྱ་གར་རྒྱལ་ཡོངས་བོད་མིའི་སིལ་སྤོལ་གྱི་རྟིང་རྩོད་འགྲན་བསྡུར་ཁྲོད་རྒྱ་གར་ལྷོ་ཕྱོགས་སྦེལ་ཀོབ་བདེ་སྐྱིད་སླར་གསོའི་རུ་ཁག་དང་ལུགས་བསམ་རུ་ཁག་གཉིས་ནས་འགྲན་བསྡུར་བྱས་པ་བརྒྱུད། བདེ་སླར་རུ་ཁག་གིས་སྤོ་ལོ་གྲངས་ ༣༥ དང་། ལུགས་བསམ་རུ་ཁག་གིས་སྤོ་ལོ་གྲངས་ ༤༧ བླུགས་ཏེ། ཨང་དང་པོའི་གསེར་གྱི་བུམ་པ་དང་བྱ་དགའ་གསོལ་རས་ཧིན་སྒོར་ཁྲི་ ༤ ཐོབ་སོང་། དེ་བཞིན། ཐེངས་འདིའི་རྒྱལ་ཡོངས་རྩེད་འགྲན་ཁྲོད་ས་རཱ་བོད་ཀྱི་མཐོ་སློབ་རུ་ཁག་ལ་སྤྱོད་ལམ་ལེགས་ཤོས་ཀྱི་དགའ་བྱ་དང་། བདེ་སྐྱིད་སླར་གསོ་རུ་ཁག་གི་རྩེད་འགྲན་པ་རྡོ་རྗེ་རྣམ་རྒྱལ་ལགས་སུ་རྩེད་འགྲན་པ་རྩེ་ཕུད་ཀྱི་བྱ་དགའ་ཐོབ་སོང་། བོད་མིའི་སྒྲིག་འཛུགས་འཕྲོད་བསྟེན་ལས་ཁུངས་ནས་གཙོ་གཉེར་གནང་ཞིང་། དབུས་བོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་དང་བོད་རྒྱལ་ཡོངས་རྩེད་རིགས་ཚོགས་པ་གཉིས་ནས་ཕྱག་རོགས་གནང་བའི་རྒྱལ་ཡོངས་རྩེད་འགྲན་ནང་། རྒྱ་གར་ལྷོ་ཕྱོགས་ནས་རུ་ཁག་ ༣། རྒྱ་གར་བྱང་ཤར་ནས་རུ་ཁག་ ༡། བལ་ཡུལ་ནས་རུ་ཁག་ ༡། བཞུགས་སྒར་དྷ་ས་ནས་རུ་ཁག་ ༣། སྦིར་གཞིས་ཆགས་ཁུལ་ནས་རུ་ཁག་ ༡། དྷེ་ར་ལྡུན་ཁུལ་ནས་རུ་ཁག་ ༢ བཅས་ཁྱོན་རུ་ཁག་ ༡༢ མཉམ་ཞུགས་གནང་འདུག ཐེངས་འདིའི་རྩེད་འགྲན་གོ་སྒྲིག་གི་མིགས་ཡུལ་གཙོ་བོ་ནི་བོད་མིའི་སྤྱི་ཚོགས་ནང་གློ་ནད་ཀྲི་བིའི་ནད་གཞི་སྐོར་གོ་རྟོགས་དང་སྔོན་འགོག་ཡོང་ཕྱིར་ཡིན་འདུག མཇུག་བསྡོམས་མཛད་སྒོའི་ཐོག་སྐུ་མགྲོན་གཙོ་བོ་སྲིད་སྐྱོང་བློ་བཟང་སེང་གེ་མཆོག་གིས།  བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་ངོས་ནས་བོད་མིའི་འཕྲོད་བསྟེན་དེ་གལ་ཆེན་པོ་རྩིས་བའི་ཐོག ད་རེས་ཀྱི་འདི་འདྲ་བའི་རྩེད་རིགས་ཀྱིས་མཚོན་པའི་མཐུན་རྐྱེན་འདྲ་མིན་སྦྱོར་བཞིན་ཡོད་པ་དང་། མ་འོངས་པར་བཙན་བྱོལ་བོད་མིའི་སྤྱི་ཚོགས་དེ་སྤྱི་ཚོགས་གཞན་དང་མ་འདྲ་བར་ཐང་ཤོས་ཤིག་ཆགས་པའི་རེ་བ་ཡོད་སྐོར་གསུངས་ཡོད་པ་རེད། གསར་འགོད་པ། ཉི་མ་ཚེ་རིང་།
