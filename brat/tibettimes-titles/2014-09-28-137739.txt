རྒྱ་དམག་ལ་དྭགས་ནས་ཕྱིར་འཐེན།

ཁ་སང་རྒྱ་དཀར་ནག་གི་ཕྱི་སྲིད་བློན་ཆེན་བར་གྲོས་མོལ་བརྒྱུད་ལ་དྭགས་ནང་བཙན་འཛུལ་བྱེད་བཞིན་པའི་རྒྱ་དམག་ཕྱིར་འཐེན་བྱ་རྒྱུ་བྱས་ཡོད་འདུག ཁ་སང་ཕྱི་ཚེས་ ༢༥ ཉིན་ཨ་རིའི་གྲོང་ཁྱེར་ནེའུ་ཡོག་ཏུ་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་ཚོགས་ཆེན་ཐེངས་ ༦༩ པ་འཚོགས་བཞིན་པའི་སྐབས་དང་བསྟུན་ཏེ། རྒྱ་གར་ཕྱི་སྲིད་བློན་ཆེན་ལྕམ་སོ་ཞི་མཱ་ (Sushama Swaraj)རྒྱ་ནག་ཕྱི་སྲིད་བློན་ཆེན་ཝང་ཡེ་ (Wang Yi) ལ་ཐུག་འཕྲད་བྱས་ཏེ། ཟླ་འདིའི་ ༢༡ ཉིན་ནས་བཟུང་རྒྱ་དམག་ ༥༠ ལྷག་རྒྱ་གར་བྱང་ཕྱོགས་ལ་དྭགས་ས་གནས་ཆུ་དམར་ནང་འཛུལ་ཏེ། སྡོད་གུར་བརྒྱབ་པ་མ་ཟད་དང་སྤྱི་ལེ་ལྔ་ཙམ་གྱི་སར་ལམ་ཁ་བཟོས་པར་བརྟེན། ཕྱོགས་གཉིས་ཀྱི་དམག་མི་ཆིག་སྟོང་ཙམ་ཕན་ཚུན་ཁ་གཏད་དུ་ལངས་པའི་ཛ་དྲག་གནས་སྟངས་སྐོར་ལ་གྲོས་མོལ་བྱས་མཐར། དེ་རིང་ནས་བཟུང་ཟླ་འདིའི་ཚེས་ ༣༠ ཉིན་བར་ཕྱོགས་གཉིས་ཀྱི་དམག་མི་ཁོངས་སོ་སོའི་རྒྱལ་སྲུང་ས་ཚིགས་སུ་ཕྱིར་ལོག་བྱ་རྒྱུའི་གྲོས་མཐུན་བྱུང་ཡོད་འདུག ཕྱི་སྲིད་བློན་ཆེན་ལྕམ་སོ་ཞི་མཱ་ཡིས་ནེའུ་ཡོག་ཏུ་ཡོད་པའི་རྒྱ་གར་གསར་འགོད་པར། ཟླ་འདིའི་ནང་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་ཐེངས་དང་པོར་ལྡི་ལིར་གཟིགས་སྐོར་སྐབས། རྒྱ་དམག་གིས་ལ་དྭགས་བཙན་འཛུལ་གྱིས་མི་མང་ལ་མིག་དཔེ་ངན་པ་བཞག་ཡོད་ཀྱང་། རྒྱ་གར་ལ་མཚོན་ན་ཁོ་རང་གི་གཟིགས་སྐོར་ལ་གྲུབ་འབྲས་ཆེན་པོ་བྱུང་ཡོད་པའི་ངོས་འཛིན་བྱེད་བཞིན་ཡོད་ཅེས་སོགས་བརྗོད་འདུག   ས་མཚམས་རྩོད་རྙོག་ཐད་རྒྱ་ནག་ཧྲེང་ཧེ་སྤྱི་ཚོགས་ཚན་རིག་ཁང་གི་རྒྱལ་སྤྱིའི་སྲིད་དོན་འབྲེལ་ལམ་གྱི་ཉམས་ཞིབ་པ་ཧོ་ཏྲི་ཡུང་ཟེར་བས། མུ་དྷི་གཞུང་གིས་རྒྱ་གར་བྱང་ཕྱོགས་ཀྱི་མཐའ་མཚམས་ས་ཐིག་ (LAC) ཁུལ་དུ། གཞུང་ལམ་འཛུགས་སྐྲུན་དང་ལག་ཆ་དབོར་འདྲེན་བྱས་ཏེ་གཏན་སྲུང་ས་ཚིགས་རྒྱག་པར་གྲ་སྒྲིག་བྱེད་ཀྱིན་ཡོད། དེ་ནི་ཁྲིམས་འགལ་གྱི་དམག་དོན་ཐབས་ཇུས་ཤིག་ཡིན་སྟབས་རྒྱ་ནག་གིས་བརྡ་ལན་མི་སྤྲོད་ག་མེད་བྱུང་བ་ཞིག་རེད་ཅེས་བརྗོད་འདུག རྒྱ་གར་བདེ་སྲུང་ལས་ཁུངས་ཀྱིས། རྒྱལ་ཁབ་གཉིས་ཀྱིས་མཐའ་མཚམས་ས་ཐིག་ (LAC) ལ་ངོས་འཛིན་བྱེད་སྟངས་མ་འདྲ་བའི་དབང་གིས་ས་མཚམས་རྩོད་རྙོག་ལྷག་གི་ཡོད་ཀྱང་། གང་མགྱོགས་པེ་ཅིན་དང་ལྡི་ལི་གཉིས་ཀྱིས་དཀའ་རྙོག་སེལ་ཐབས་བྱེད་རྒྱུ་ཡིན་སྐོར་བརྗོད་འདུག འདིའི་ཐད་ལ་དྭགས་རང་སྐྱོང་རི་ལྗོངས་ཡར་རྒྱས་ལྷན་ཚོགས་ཀྱི་ཚོགས་གཙོ་རིག་འཛིན་བཟོད་པ་ལགས་ཀྱིས་བརྗོད་པར། ཉེ་བའི་ལོ་བཅུ་ཕྲག་གཉིས་གསུམ་རིང་རྒྱ་གར་གཞུང་གིས་ལ་དྭགས་ཁུལ་གྱི་ས་མཚམས་སྣང་མེད་བཞག་པར་བརྟེན། ལ་དྭགས་ཁུལ་གྱི་རྩྭ་ས་མང་པོ་ཞིག་རྒྱ་ནག་གི་ལག་ཏུ་ཤོར་ཏེ་ད་ལྟ་རྒྱ་ནག་གི་བདག་དབང་ཡོད་ཚུལ་གྱིས་རྒྱ་གར་གཞུང་ལ་སྐྱོན་བརྗོད་བྱས་ཡོད་འདུག  རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་རྒྱ་གར་ནང་གཟིགས་སྐོར་སྐབས། རྒྱ་གར་སྲིད་བློན་མུ་དྷི་ཡིས་རྒྱ་དཀར་ནག་གཉིས་བར་སྔ་བོ་ནས་ཡིད་ཆེས་བློས་ཐུབ་ཀྱི་འབྲེལ་ལམ་ཡོད་ན། རྒྱལ་ཁབ་གཉིས་ཀྱི་ས་མཚམས་དང་སྲིད་དོན་འབྲེལ་ལམ་ཞི་བདེ་ཡོང་ངེས་རེད་ཅེས་བརྗོད་པ་དང་། དེ་མཚུངས་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱིས་ཀྱང་། ཁོས་རྒྱ་གར་དང་མཉམ་དུ་ས་མཚམས་སུ་ཞི་བདེ་བརྟན་བརླིང་ཡོང་བར་མཉམ་ལས་བྱེད་པར་ཁས་ལེན་བྱེད་ཀྱི་ཡོད་ཚུལ་བརྗོད་འདུག  སྲིད་དོན་དབྱེ་ཞིབ་པ་ཚོས། མུ་དྷི་ཡིས་ཉི་ཧོང་དང་ལྷོ་ཤར་ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་ཚང་མར་སྲིད་དོན་དང་དམག་དོན་ གཙོས་ཚོང་ལས་དང་རིག་གཞུང་ལ་སོགས་ཀྱི་འབྲེལ་ལམ་དམ་ཟབ་གཏོང་གི་ཡོད་སྟབས། ཞི་ཅིན་ཕིན་གྱིས་མུ་དྷི་ལ་སྣོན་ཤུགས་དང་མི་དགའ་བའི་རྣམ་འགྱུར་སྟོན་ཆེད། རྒྱ་གར་ས་ཁོངས་སུ་ཡང་ཡང་བཙན་འཛུལ་བྱས་པ་དང་། རྒྱ་ནག་གི་དམག་དཔོན་ཚོས་ས་ཁུལ་རང་བཞིན་གྱི་དམག་འཁྲུག་རྒྱག་པར་གྲ་སྒྲིག་བྱེད་དགོས་པའི་གཏམ་བཤད་སྤེལ་ཡོད་ཚུལ་བརྗོད་འདུག  རྒྱ་གར་གཞུང་གིས་ད་ལོའི་ཟླ་བ་བརྒྱད་རིང་རྒྱ་དམག་གིས་རྒྱ་གར་གྱི་ས་མཚམས་ནང་ཐེངས་ ༣༣༤ ཙམ་ཚུར་བརྒལ་བྱས་ཡོད་ཚུལ་བརྗོད་པ་ལྟར། རྒྱ་ནག་གིས་ཀྱང་རྒྱ་གར་དམག་མིས་ཕར་ཕྱོགས་ས་མཚམས་ནང་ཐེངས་ ༤༡༠ ཙམ་བརྒལ་ཡོད་པའི་སྐྱོན་འཛུགས་བྱས་ཡོད་འདུག   ༡༩༦༢ ལོར་རྒྱ་དཀར་ནག་བར་དམག་འཁྲུག་བྱུང་བའི་རྗེས་ནས་ད་ལྟའི་བར་ས་མཚམས་སྐོར་ལ་གྲོས་མོལ་ཐེངས་ ༡༧ ཙམ་བྱས་ཡོད་ཀྱང་དོན་འབྲས་སོན་མེད། གསར་འགོད་པ། ལྷ་སྐྱབས་རྒྱལ།
