
རྒྱ་གར་གཞུང་གིས་དྷ་རམ་ས་ལའི་སྐུ་མགྲོན་གྱི་ཐོངས་མཆན་ཕྱིར་བསྡུ་བྱས།

ལྡི་ལི་གཞུང་གིས་༧གོང་ས་མཆོག་ལ་མཇལ་བཅར་ཞུ་མཁན་རྒྱ་རིགས་བུད་མེད་ལུ་ཅིན་ཧྭ་ཟེར་བ་ཞིག་གི་ཐོངས་མཆན་ཕྱིར་འཐེན་བྱས་ཡོད་འདུག དེ་རིང་ཧོང་ཀོང་རང་དབང་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཁ་སང་རེས་གཟའ་ཕུར་བའི་ཉིན་ལྡི་ལི་སྲིད་གཞུང་གིས་རྒྱ་རིགས་བུད་མེད་ལུ་ཅིན་ཧྭ་ (Lu Jinghua) ཟེར་བ་ཞིག་རྒྱ་གར་བྱང་ཕྱོགས་དྷ་རམ་ས་ལར་བོད་མིའི་སྐྱབས་མགོན་༧གོང་ས་རྒྱལ་བ་རིན་པོ་ཆེ་མཆོག་ལ་མཇལ་འཕྲད་ཞུ་བར་བཅར་རྒྱུའི་ཐོངས་མཆན་ཕྱིར་འཐེན་བྱས་ཡོད་འདུག ལུ་ཅིན་ཧྭ་ནི་ ༡༩༨༩ ལོར་གནམ་བདེ་སྒོ་མོ་ཆེའི་ཐང་ཆེན་དུ་དམངས་གཙོའི་འོས་ལངས་ལ་རྒྱ་ནག་གཞུང་གིས་དྲག་གནོན་བྱས་རྗེས་ཁོང་མོ་ཨ་རིར་སྐྱབས་བཅོལ་དུ་ཕྱིན་ཡོད་འདུག ཕྱི་སྡོད་རྒྱ་རིགས་བུད་མེད་ལུ་ཅིན་ཧྭ་ཡིས་ཁ་སང་ང་༧གོང་ས་སྐྱབས་མགོན་ཆེན་པོ་མཆོག་གི་བཞུགས་གནས་དྷ་རམ་ས་ལར་འཚོགས་ཆར་ཡོད་པའི་བགྲོ་གླེང་ཚོགས་འདུའི་སྟེང་མཉམ་ཞུགས་བྱེད་དུ་ཡོང་སྐབས། ནེའུ་ཡོག་ (JFK) གནམ་ཐང་དུ་རྒྱ་གར་མཁའ་འགྲུལ་ལས་བྱེད་ཚོས་ང་བཀག་འགོག་བྱས་སོང་། སྤྱིར་རྒྱ་གར་གཞུང་གིས་ང་ལ་ཐོངས་མཆན་སྤྲད་ཡོད་ཀྱང་། ནེའུ་ཡོག་ (JFK) གནམ་ཐང་གི་རྒྱ་གར་མཁའ་འགྲུལ་ལས་བྱེད་ཚོས་ང་ལ་དགོངས་དག ཁྱེད་རང་སོང་ཆོག་གི་མ་རེད་ཟེར། ང་རྒྱ་གར་དྷ་རམ་ས་ལར་༧གོང་ས་ཏཱ་ལའི་བླ་མ་མཆོག་ལ་མཇལ་དུ་འགྲོ་གི་ཡོད་སྟབས། ཁོ་ཚོས་ང་བཀག་འགོག་བྱས་པ་རེད། ཅེས་བརྗོད་འདུག གནས་ཚུལ་ལྟར་ན། ལྡི་ལི་གཞུང་གིས་རྒྱ་རིགས་བུད་མེད་ལུ་ཅིན་ཧྭ་ཡི་ཐོངས་མཆན་ཕྱིར་འཐེན་བྱེད་དོན་གཙོ་བོ་རྒྱ་ནག་གིས་གནོན་ཤུགས་སྤྲད་པ་ཞིག་ཡིན་པ་དང་། པེ་ཅིན་གཞུང་གི་གནོན་ཤུགས་དང་ཛ་དྲག་གི་འབོད་སྐུལ་ལ་བརྟེན་ནས་ཕྱི་ཚེས་ ༢༥ ཉིན་རྒྱ་གར་གཞུང་གིས“འཛམ་གླིང་ཡུ་གུར་ལྷན་ཚོགས” (WUC) ཀྱི་ཚོགས་གཙོ་ཌོལ་ཀུན་ཨེ་ས་ (Dolkun Isa) ལགས་ཀྱི་ཐོངས་མཆན་ཡང་ཕྱིར་འཐེན་བྱེད་དགོས་བྱུང་ཡོད་པ་རེད། རྒྱ་གར་ཕྱི་སྲིད་ལྷན་ཁང་གི་མགྲིན་ཚབ་པ་བེ་ཀ་སི་ཝ་རུ་ཕི་ (Vikas Swarup) ཟེར་བ་ཞིག་གིས། ང་ཚོས་སྐུ་ཞབས་ཨེ་ས་ལགས་ལ་ཐོངས་མཆན་སྤྲད་རྗེས། ཁོང་མོས་རང་ཉིད་རྒྱ་གར་ཧི་མ་ཅལ་གྲོང་སྡེ་དྷ་རམ་ས་ལར་འཚོགས་འཆར་ཡོད་པའི་བགྲོ་གླེང་ཚོགས་འདུའི་སྟེང་མཉམ་ཞུགས་བྱེད་དུ་འགྲོ་གྲབས་ཡོད་སྐོར་ཕྱི་བསྒྲགས་བྱས་ཡོད་སྟབས། འཇིགས་སྐུལ་དྲག་སྤྱོད་པ་ཞིག་ལ་ཐོངས་མཆན་སྤྲོད་པའི་མཆོག་མཆན་མེད་སྐོར་དང་། ང་ཚོ་དམངས་གཙོའི་སྤྱི་ཚོགས་ཤིག་ལ་མཚོན་ན་སྤྱི་ཚོགས་བདེ་འཇགས་ལ་དོ་འཁུར་ཡོད་སྐོར་སོགས་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
