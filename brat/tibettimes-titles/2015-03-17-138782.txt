སོག་ཙན་དན་དགོན་པའི་གྲྭ་པ་ ༧
འཛིན་བཟུང་།

ཕྱི་ཟླ་ ༣ ཚེས་ ༡༤ ཉིན་གྱི་དགོང་མོར་སོག་རྫོང་ཙན་དན་དགོན་པའི་གྲྭ་པ་ ༧ རྒྱ་གཞུང་གི་ཉེན་རྟོག་པས་འཛིན་བཟུང་བཀག་ཉར་བྱས་ཡོད་པ་དང་། ཁོང་རྣམ་པ་འཛིན་བཟུང་བྱེད་པའི་རྒྱུ་མཚན་ཉེན་རྟོག་པས་གསལ་བཤད་བྱས་མི་འདུག ས་གནས་ཡུལ་མི་དག་གིས་ཚོད་དཔག་ལྟར་ན། བོད་དོན་དང་འབྲེལ་བའི་པར་རིས་དང་གནས་ཚུལ་ཕན་ཚུན་གཏོང་རེས་བྱས་པས་ཡིན་ཚུལ་བརྗོད་འདུག འཛིན་བཟུང་བྱས་པའི་སོག་རྫོང་ཙན་དན་དགོན་པའི་གྲྭ་རྣམ་རྒྱལ་ཚུལ་ཁྲིམས་དང་། གྲྭ་བློ་གྲོས་བསྟན་འཛིན།  གྲྭ་ཚུལ་ཁྲིམས་སྒོ་འབྱེད།  གྲྭ་ཚུལ་ཁྲིམས་རྣམ་རྒྱལ།  གྲྭ་ཐབས་མཁས་ལྷུན་འགྲུབ།   གྲྭ་འཇིག་མེད་ཚུལ་ཁྲིམས། གྲྭ་འཇིག་མེད་གྲགས་པ་བཅས་ཡིན་འདུག་ལ། ཁོང་རྣམས་འཛིན་བཟུང་བྱེད་པའི་རྒྱུ་མཚན་ཉེན་རྟོག་པས་གསལ་བཤད་བྱས་མེད་པ་མ་ཟད། ས་ཕྱོགས་གང་དུ་བཙོན་འཇུག་བྱས་མིན་ཡང་ཤེས་རྟོགས་བྱུང་མི་འདུག སྤྱིར་ཙན་དན་དགོན་པའི་ཕྱི་ནང་བར་གསུམ་ལ་གསང་བའི་པར་ཆས་གྲངས་ལྔ་བཅུ་ཙམ་བཙུགས་ཡོད་པ་དང་། དེང་སྐབས་དགོན་པའི་ཕྱི་ནང་བར་གསུམ་དུ་ཉེན་རྟོག་མི་སྣ་མང་པོ་འགྲེམས་འཇོག་གིས་ཆུ་ཚོད་ཉེར་བཞིའི་རིང་ལྟ་སྲུང་བྱས་ཡོད་པ་རེད། མཚན་མོར་གྲྭ་པ་རྣམས་ཕྱི་ལ་ཆབ་གསང་དུ་འགྲོ་དུས་ཉེན་རྟོག་མི་སྣས་སུན་པོ་བཟོ་བ་དང་། འཇིགས་སྐུལ་བྱེད་པ་སོགས་དེང་སྐབས་དགོན་པའི་བླ་གྲྭ་རྣམས་འཇིག་སྣང་དང་དངངས་སྐྲག་འོག་གནས་ཡོད་འདུག  ལོ་ན་བཅོ་བརྒྱད་མ་ཟིན་པར་གྲྭ་པ་བྱེད་མི་ཆོག་པའི་བཙན་བཀའ་བཏང་འདུག གོང་གི་གྲྭ་རྣམ་རྒྱལ་ཚུལ་ཁྲིམས་ནི་ ༢༠༡༡ ལོར་ཕྱི་ཕྱོགས་ལ་འབྲེལ་བ་ཡོད་པར་བསྙད་ནས་ཉིན་༧རིང་བཀག་ཉར་བྱེད་མྱོང་འདུག་ལ། ༢༠༡༢ ལོའི་ཟླ་ ༡༠ ཚེས་ ༦ ཉིན་འཛིན་བཟུང་བྱས་ཏེ། ཐོག་མར་ནག་ཆུའི་བཙོན་ཁང་དང་བཀག་ཉར་ཁང་གཉིས་ཀྱི་ནང་རེས་མོས་ཀྱིས་ཟླ་ ༤ དང་ཉིན་གྲངས་ ༢༨ ཙམ་བཙོན་འཇུག་དང་། དེའི་རྗེས་ལྷ་སའི་སྟོད་ལུང་བཙོན་ཁང་དུ་ཟླ་ ༣ དང་ཉིན་ ༡༡ རིང་བཙོན་འཇུག་བྱས་པ་བཅས་ཁྱོན་ཟླ་ ༨ ལྷག་བཙོན་འཇུག་མནར་གཅོད་བཏང་མྱོང་ཡོད་འདུག འདི་ལོའི་ཕྱི་ཟླ་ ༣ པའི་ཚེས་ཤར་ནས་བཟུང་བོད་ས་ཁྱོན་ཡོངས་ལ་སྤྱི་ཚོགས་དྲ་འཕྲིན་ཁག་གིས་གཙོས་བརྒྱུད་ལམ་ཡོངས་རྫོགས་ལ་དམ་དྲག་ནན་པོ་བྱེད་བཞིན་ཡོད་པ་མ་ཟད། ལྷག་པར་ནག་ཆུའི་ཤར་ཕྱོགས་རྫོང་གསུམ་སྟེ། སྦྲ་སོག་འབྲི་གསུམ་གྱི་ཁྱོན་ལ་དམ་དྲག་ཆེན་པོ་བྱེད་བཞིན་འདུག དེ་སྔོན་ཆབ་སྲིད་ཀྱི་གནད་དོན་འཕྲད་མྱོང་ཡོད་པ་ཚང་མ་ཡུལ་ཚོའི་གཞུང་ལས་ཁང་དུ་འབོད་འགུག་བྱེད་བཞིན་འདུག་ཅེས་བོད་ནས་འབྲེལ་ཡོད་ཅིག་གིས་གནས་ཚུལ་ངོ་སྤྲོད་གནང་སོང་། ཞེས་ངག་དབང་ཐར་པ་ལགས་ཀྱིས་གནས་ཚུལ་བཏང་འབྱོར་བྱུང་།
