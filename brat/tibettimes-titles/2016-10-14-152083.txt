ཐའེ་ལན་ཌིའི་རྒྱལ་པོ་སྐུ་གྲོངས།

ཁ་སང་ཐའེ་ལན་ཌིའི་རྒྱལ་པོ་བྷུ་མི་པོལ་ཨ་ཌོལ་ཡེ་ཌི་ཇི་ལགས་སྐུ་གྲོངས་ཏེ་རྒྱལ་སྲས་མཱ་ཧཱ་ཝེ་ཇི་ར་ལོང་ཀོན་གྱིས་རྒྱལ་སྲིད་འཛིན་དགོས་པ་ཁྱབ་བསྒྲགས་བྱས་ཡོད་འདུག ཁ་སང་ཐའེ་ལན་ཌིའི་ཉིན་རྒྱབ་ཆུ་ཚོད་ ༢།༢༢ ཐོག ཡུལ་དེའི་རྒྱལ་པོ་བྷུ་མི་པོལ་ཨ་ཌོལ་ཡེ་ཌི་ཇི་ (Bhumibol Adulyadej)ལགས་སྙུང་གཞིའི་རྐྱེན་གྱིས། རྒྱལ་ས་བྷན་ཁོག་ཏུ་ཡོད་པའི་རྒྱལ་ཡོངས་ཀྱི་སྨན་ཁང་ཆེ་ཤོས་(Siriraj Hospital)ནང་དགུང་གྲངས་ ༨༨ ཐོག་སྐུ་གྲོངས་ཡོད་པ་དང་། ཁོང་ལ་སྨན་བཅོས་བྱེད་སྐབས་གཞུང་འབྲེལ་དང་གཞུང་འབྲེལ་མིན་པའི་ཐོག་ནས་ཐའེ་ལན་ཌིའི་མི་མང་སྟོང་ཕྲག་མང་པོ་ཁོང་བཞུགས་སའི་སྨན་ཁང་དང་། སྲིད་གཞུང་གི་ལས་ཁུངས་མདུན་སོགས་སུ་འདུ་འཛོམས་བྱས་ཏེ། རྒྱལ་པོའི་སྐུ་པར་བཟུང་སྟེ་ཁོང་མྱུར་ཏུ་དྲག་སྐྱེད་ཡོང་བའི་སྨོན་ལམ་བརྒྱབ་ཡོད་ཀྱང་། ཡུན་རིང་གི་སྙུང་གཞི་དྲག་སྐྱེད་བྱུང་མ་ཐུབ་པར་སྐུ་གྲོངས་ཡོད་འདུག ཁ་སང་ཐའེ་ལན་ཌིའི་སྲིད་བློན་པར་ཡོ་ཊི་ཅན་ཨོ་ཅཱ་(Prayut Chan-o-cha)ཡིས། ཐའེ་ལན་ཌིའི་རྒྱལ་སྲས་མཱ་ཧཱ་ཝེ་ཇི་ར་ལོང་ཀོན་(Maha Vajiralongkorn)གྱིས་རྒྱལ་སྲིད་འཛིན་པའི་དུས་ལ་བབས་ཡོད་སྐོར་ཁྱབ་བསྒྲགས་བྱས་ཡོད་སྟབས། ཆབ་སྲིད་དཔྱད་ཞིབ་པ་མང་པོས་རྒྱལ་སྲས་ཆབ་སྲིད་ལ་དོ་སྣང་ཆུང་ཞིང་། འཆལ་པོ་རྩུབ་སྤྱོད་ཆེ་བ། ལྷག་པར་དུ་ཕྱི་རྒྱལ་དུ་བྲོས་བྱོལ་ཕྱིན་པའི་ཐའེ་ལན་ཌིའི་སྲིད་བློན་ཐི་སིན་ཞི་ནཱ་ཝཱ་ཊིར་(Thaksin Shinawatra )དང་སྒེར་འབྲེལ་ཟབ་མོ་ཡོད་པ་སོགས་ཀྱི་གླེང་གཞི་ཆེན་པོ་ཡོད་སྟབས། ཕྱི་ལོ་ ༡༩༧༠ ཙམ་ནས་བཟུང་ཐའེ་ལན་ཌིའི་མི་མང་གིས་རྒྱལ་སྲས་ཁོང་ལ་ལྟ་སྟངས་ལེགས་པོ་མེད་པར་བརྟེན། ཁོང་གིས་རྒྱལ་སྲིད་བཟུང་ན་ཡུལ་དེའི་ཆབ་སྲིད་བརྟན་ལྷིང་ཐད་དཀའ་རྙོག་ཡོང་བའི་ཉེན་ཁ་ཞིག་ཏུ་ངོས་འཛིན་བྱེད་བཞིན་ཡོད་འདུག ཐའེ་ལན་ཌིའི་མི་མང་གིས་དགུང་ལོ་ ༦༡ ཡིན་པའི་རྒྱལ་པོའི་སྲས་མོ་ཆུང་བ་མཱ་ཧཱ་ཅག་རི་སི་རིན་ཌོན་(Maha Chakri Sirindhorn)གྱིས་རྒྱལ་སྲིད་བཟུང་ན་འོས་འཚམ་ཡིན་པའི་ངོས་འཛིན་བྱེད་བཞིན་ཡོད་ཀྱང་། ཐའེ་ལན་ཌིའི་རྩ་ཁྲིམས་ནང་བུད་མེད་ཀྱིས་རྒྱལ་སྲིད་འཛིན་ཆོག་གི་མེད་པ་རེད། རྒྱལ་པོ་བྷུ་མི་པོལ་ཨ་ཌོལ་ཡེ་ཌི་ཇི་ལགས་ནི། འཛམ་གླིང་ཐོག་གི་རྒྱལ་པོའི་སྲིད་དབང་འཛིན་མཁན་རྒྱུན་རིང་ཤོས་དེ་ཆགས་ཡོད་པ་དང་། ཁོང་གིས་རྒྱལ་སྲིད་བཟུང་བའི་ལོ་ངོ་ ༧༠ ནང་ཐའེ་ལན་ཌིའི་སྲིད་བློན་ཐེངས་ ༣༥ ལ་འཕོ་འགྱུར་བྱུང་ཡོད་ཅིང་། ཁོང་གིས་སྲིད་བློན་རིམ་ཕེབས་དང་མཉམ་དུ་དམག་དོན་སྲིད་བསྒྱུར་ཐེངས་ ༢༠ ལྷག་གནང་ཡོད་པ་དང་། ཡུལ་དེའི་ནང་ཆབ་སྲིད་དང་དཔལ་འབྱོར་སོགས་ཀྱི་དཀའ་རྙོག་བྱུང་ཐེངས་རེ་རེར་ཁོང་དངོས་སུ་ཕེབས་ཏེ་ཐུགས་འཁུར་ཆེན་པོ་བཞེས་ཡོད་པ་མ་ཟད། རྒྱུན་དུ་དབུལ་ཕོངས་ས་ཁུལ་དུ་འཚམས་གཟིགས་སུ་ཕེབས་ཏེ་འབངས་ལ་བྱམས་ཤིང་འབངས་ཀྱི་འཚོ་བར་ཐུགས་འཁུར་ཆེན་པོ་གནང་ཡོད་པར་བརྟེན། མི་མང་གིས་ཁོང་ལ་གུས་བརྩི་དང་དགའ་མོས་ཤུགས་ཆེན་ཞུ་ཡི་ཡོད་པ་དང་། ཁོང་སྐུ་གྲོངས་རྗེས་རྒྱལ་སྲིད་འཛིན་སྐྱོང་གི་གནད་དོན་ཐད་རྒྱལ་ཡོངས་ནང་དོ་སྣང་དང་སེམས་འཚབ་ཆེན་པོ་བྱེད་བཞིན་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
