རྒྱ་དཀར་ནག་བར་གྱི་ཉི་ཧོང་གི་ཁེ་བཟང་།

རྒྱ་གར་གྱི་སྲིད་བློན་མུ་དྷེ་ཉི་ཧོང་ལ་གཟིགས་སྐོར་དུ་ཕེབས་ཏེ་ཉི་ཧོང་སྲིད་བློན་ཤིང་ཛོ་ཨ་བྷེ་དང་རྒྱ་གར་དང་ཉི་ཧོང་བར་ཚོང་ལས་དང་དམག་དོན་བར་གྲོས་མཐུན་མང་པོ་ཞིག་བཞག་ཡོད་འདུག གཟའ་སྤེན་པའི་ཉིན་ནས་བཟུང་རྒྱ་གར་སྲིད་བློན་ནཱ་རེན་ཌ་མུ་དྷེ་ཉི་ཧོང་ནང་གཟིགས་སྐོར་མགོ་འཛུགས་གནང་ཡོད་ཅིང་། ཁ་སང་ཚེས་ ༣༡ ཉིན་ཁེའུ་ཀྲོ་ (Kyoto )གྲོང་ཁྱེར་གྱི་ནང་པའི་དགོན་རྙིང་ཌོ་ཇི་ (Toji ) དང་ཁེན་ཁཱ་ཁུ་ཇི་(Kinkakuji) རུ་གཟིགས་སྐོར་སྐབས། ཉི་ཧོང་གི་སྲིད་བློན་ཤིང་ཛོ་ཨ་བྷེ་ཆེད་མངགས་ཁེའུ་ཀྲོ་གྲོང་ཁྱེར་དུ་རྒྱ་གར་སྲིད་བློན་ལ་ཕེབས་བསུར་བཅར་ཡོད། ཐེངས་འདིར་རྒྱ་གར་སྲིད་བློན་ཉི་ཧོང་ནང་ཉིན་ ༥ རིང་གི་ལས་རིམ་ཡོད་པ་དང་། རྒྱལ་ཁབ་གཉིས་ཀྱི་སྲིད་བློན་བར་རྒྱ་གར་ཝ་ར་ཎཱ་སི་ཡར་ཐོན་ཅན་གྱི་གྲོང་ཁྱེར་ཞིག་ཏུ་འགྱུར་ཐབས་ཀྱི་གྲོས་མཐུན་ལ་མིང་རྟགས་བཀོད་ཡོད་པ་མ་ཟད། ལོ་རེར་རྒྱ་གར་གྱིས་ཉི་ཧོང་གི་བཟོ་གྲྭ་ཁག་ལ་མཁོ་བའི་རྒྱུ་ཆ་ (rare earths material) ཏུན་ ༢༠༠༠ ནས་ ༢༣༠༠ བར་ཚོང་འབྲེལ་བྱ་རྒྱུ་དང་། ཉི་ཧོང་གིས་རྒྱ་གར་ནང་བཟོ་གྲྭ་གསར་འཛུགས་ཀྱིས་ལོ་རེར་ཨ་སྒོར་ཐེར་འབུམ་ ༢ ཀྱི་མ་འཛུགས་བྱ་རྒྱུ། དེ་བཞིན། རྒྱ་གར་ལ་རྡུལ་ཕྲན་གློག་ཤུགས་ཀྱི་རྒྱུ་ཆ་རིགས་ཉི་ཧོང་ནས་ནང་འདྲེན་བྱ་རྒྱུ། ཉི་ཧོང་གིས་རྒྱ་གར་གྱི་མཚོ་དམག་ལ་མཁོ་བའི་འཚོལ་ཞིབ་དང་དམག་རྒྱག་གནམ་རྒྱུ་ཕྱིར་ཚོང་བྱ་རྒྱུ། ལོ་རེའི་ནང་རྒྱལ་ཁབ་གཉིས་ཀྱི་མཚོ་དམག་བར་དམག་སྦྱོང་དང་། ཉི་ཧོང་གི་རྒྱབ་སྐྱོར་འོག་རྒྱ་གར་ནང་མྱུར་མགྱོགས་མེ་འཁོར་ལྕགས་ལམ་འཛུགས་སྐྲུན་བྱ་རྒྱུ་བཅས་ལ་གྲོས་འཆམ་བྱུང་ཡོད་འདུག རྒྱ་གར་སྲིད་བློན་ནཱ་རེན་ཌ་མུ་དྷེ་ཡིས། “ད་རེབ་ཉི་ཧོང་ནང་སྐོར་སྐྱོད་འདིས་ཨེ་ཤ་ཡའི་དམངས་གཙོའི་རྒྱལ་ཁབ་རྙིང་ཤོས་གཉིས་ཀྱི་འབྲེལ་ལམ་ཤོག་ལྷེ་གསར་པ་ཞིག་ཕྱེ་མགོ་བརྩམས་ཡོད་སྟབས། རྒྱལ་སྤྱིའི་ཐོག་རྒྱལ་ཁབ་གཉིས་ཀྱི་ཐབས་ཇུས་དང་ཚོང་དོན་འབྲེལ་ལམ་སྔར་ལྷག་འཕེལ་རྒྱས་འགྲོ་རྒྱུར་ཕན་ཐོགས་ཆེན་པོ་ཡོད” ཅེས་བརྗོད་འདུག ཉི་ཧོང་སྲིད་བློན་ཤིང་ཛོ་ཨ་བྷེ་ཡིས། “ངས་རྒྱ་གར་གྱི་སྲིད་བློན་ནཱ་རེན་ཌ་མུ་དྷེ་དང་མཉམ་དུ་ཌོ་ཇི་དགོན་པར་ལྟ་སྐོར་གྱིས་སངས་རྒྱས་བཅོམ་ལྡན་འདས་ཀྱི་སྐུ་བརྙན་ལ་ལྟ་སྐབས། ལོ་རྒྱུས་ཐོག་རྒྱ་གར་དང་ཉི་ཧོང་གི་འབྲེལ་ལམ་དེ་གཏིང་ཟབ་པོ་ཡོད་པ་དྲན་སྐུལ་བྱས་ཡོད་སྟབས། ད་རེབ་འབྲེལ་མོལ་འདི་བརྒྱུད་རྒྱལ་ཁབ་གཉིས་བར་གྱི་འབྲེལ་ལམ་ཡར་རྒྱས་དང་རྒྱལ་སྤྱིའི་དཀའ་རྙོག་སེལ་ཐབས། ལྷག་དོན་དུ་ཨེ་ཤ་ཡའི་བརྟན་ལྷིང་སྲུང་སྐྱོབ་ཐོག་ནུས་པ་གལ་ཆེན་ཐོན་ཡོད” ཅེས་བརྗོད་འདུག རྒྱ་གར་གྱི་ཐབས་ཇུས་ཆེད་མཁས་པ་བྷ་ར་ཋ་ཁར་ནཱ་ཌེ་ (Bharat Karnad) ལགས་ཀྱིས། “རྒྱ་གར་དང་ཉི་ཧོང་གི་འབྲེལ་ལམ་ཡར་རྒྱས་གཏོང་དོན་གཙོ་བོ་ནི། ཉེ་ལམ་རྒྱ་ནག་གཞུང་གིས་ཁྱིམ་མཚེས་རྒྱལ་ཁབ་ཁག་ལ་དྲག་རྩུབ་བཀོལ་ཏེ། ཨེ་ཤ་ཡའི་བརྟན་ལྷིང་ཐོག་དཀའ་ངལ་བཟོ་བཞིན་ཡོད་སྟབས། རྒྱལ་ཁབ་གཉིས་མཉམ་འབྲེལ་བྱས་ཏེ་རྒྱ་ནག་ལ་གདོང་ལེན་བྱེད་འཆར་ཡོད་པ་དང་། རྒྱ་ནག་གི་ཕྱོགས་གཅིག་ལ་ཉི་ཧོང་དང་ཕྱོགས་གཞན་དེར་རྒྱ་གར་ཡོད་སྟབས་རྒྱལ་ཁབ་འདི་གཉིས་རྒྱ་ནག་དང་དཔལ་འབྱོར་མཉམ་ལས་རྒྱུན་འཁྱོངས་བྱ་དགོས་ཀྱང་། རྒྱ་ནག་གི་དྲག་རྩུབ་དང་རྒྱ་སྐྱེད་རིང་ལུགས་ལ་རྒྱ་གར་དང་ཉི་ཧོང་གཉིས་ཀྱིས་མཉམ་འབྲེལ་ཐོག་ཁ་གཏད་གཅོག་དགོས་པ་ཆགས་ཡོད” ཅེས་བརྗོད་འདུག རྒྱ་གཞུང་དང་ཉི་ཧོང་གི་འབྲེལ་ལམ་ཡག་པོ་ཡོད་སྐབས། ཉི་ཧོང་གི་བཟོ་གྲྭ་ཁག་གི་རྒྱུ་ཆ་(rare earths material) རྒྱ་ནག་ནས་ནང་འདྲེན་བྱེད་བཞིན་ཡོད་ཀྱང་། ཉེ་ལམ་རྒྱལ་སྤྱིའི་ཐོག་དཔལ་འབྱོར་ཡར་རྒྱས་ཆེ་ཤོས་ཀྱི་རྒྱལ་ཁབ་གཉིས་པ་དང་གསུམ་པའི་བར་གྱི་འབྲེལ་ལམ་ཇེ་སྡུག་ཏུ་ཕྱིན་ཡོད་སྟབས། དེ་རིང་ཉི་ཧོང་སྲིད་བློན་གྱིས་རྒྱ་གར་གྱི་སྲིད་བློན་དང་ (2+2 dialogue) ཞེས་པའི་གྲོས་མོལ་གནང་སྟེ། མ་འོངས་པར་ཉི་ཧོང་གི་བཟོ་གྲྭ་ཁག་ལ་མཁོ་བའི་རྒྱུ་ཆ་དག་རྒྱ་གར་ནས་ནང་འདྲེན་བྱ་རྒྱུ་བྱས་ཡོད་འདུག རྒྱ་གར་སྲིད་བློན་ནཱ་རེན་ཌ་མུ་དྷེ་རྒྱ་གར་གྷུ་ཅུ་ར་ཌེ་མངའ་སྡེའི་བློན་ཆེན་གནང་སྐབས། ཉི་ཧོང་ལ་ཐེངས་གཉིས་ཙམ་ཕེབས་ཏེ། སྲིད་བློན་ཤིང་ཛོ་ཨ་བྷེ་དང་ཐུག་འཕྲད་གནང་མྱོང་ཡོད་པ་དང་། ཤིང་ཛོ་ཨ་བྷེ་ཡིས་སྤྱི་ཚོགས་དྲ་རྒྱ་བརྒྱུད་ཁོང་གིས་རྒྱལ་སྤྱིའི་ཐོག་མོས་གུས་བྱེད་ཡུལ་གྱི་མི་སྣ་གསུམ་ཡོད་པའི་ཁྲོད་ནས་གཅིག་ནཱ་རེན་ཌ་མུ་དྷེ་ཡིན་སྐོར་བརྗོད་ཡོད་འདུག ཉི་ཧོང་གི་སྲིད་ཇུས་ཉམས་ཞིབ་པ་ཨཱ་ལིན་ཛཱ་ཌེ་རི་ལགས་ཀྱིས། “ཉི་ཧོང་གིས་རྒྱ་གར་དང་མཉམ་ལས་བྱེད་ཤུགས་ཆེ་རུ་བཏང་བ་དེས། ཉི་ཧོང་གིས་རྒྱ་ནག་གཞུང་དང་ཚོང་འབྲེལ་བྱེད་འདོད་མེད་པའི་རྣམ་འགྱུར་བསྟན་པ་ཞིག་ཡིན་སྟབས། མ་འོངས་པར་རྒྱལ་ཁབ་གཉིས་བར་འབྲེལ་ལམ་ཇེ་ཞན་དུ་འགྲོ་རྒྱུ་ཡིན་སྟབས། གོ་སྐབས་འདི་བརྒྱུད་ནས་རྒྱ་གར་ལ་ཁེ་ཕན་ཆེན་པོ་རག་རྒྱུར་རེ་བ་ཡོད” ཅེས་བརྗོད་འདུག ལོ་སྔ་ཕྱིར་རྒྱ་ནག་གི་མཐའ་སྲུང་དམག་མིས་རྒྱ་གར་གྱི་ས་མཚམས་ཁག་ལ་འཇབ་རྒོལ་ཡང་ཡང་བྱེད་བཞིན་ཡོད་པ་དང་། སྔ་ལོར་རྒྱ་ནག་གིས་ཉི་ཧོང་དང་རྩོད་རྙོག་ཡོད་པའི་སིང་ཁཱ་ཁཱོ་ (Senkaku) མཚོ་གླིང་དང་། ལྷོ་རྒྱ་ནག་རྒྱ་མཚོའི་བདག་དབང་རྩོད་གླེང་ཤུགས་ཆེ་རུ་བཏང་བ། ཉི་ཧོང་གི་སྲིད་བློན་གྱིས་འཛམ་གླིང་དམག་ཆེན་གཉིས་པའི་སྐབས་རྒྱ་ནག་དང་ཀོ་རི་ཡའི་མི་སེར་དམར་གསོད་གཏོང་མཁན་གྱི་དམག་མིའི་དྲན་རྟེན་ལྷ་ཁང་ལ་མཆོད་འབུལ་བྱས་ཏེ་རྒྱ་ནག་ལ་ཁོང་ཁྲོ་ཆེན་པོ་བསླངས་པ་བཅས་ལ་བརྟེན་ཏེ། ཉི་ཧོང་དང་རྒྱ་ནག་བར་གྱི་འབྲེལ་ལམ་མངོན་གསལ་དུ་ཇེ་སྡུག་ཏུ་ཕྱིན་པ་དང་བསྟུན་ཏེ། རྒྱ་གར་དང་ཉི་ཧོང་གཉིས་ཀྱིས་འབྲེལ་ལམ་ཤུགས་ཆེ་རུ་བཏང་སྟེ། དཔལ་འབྱོར་ཚོང་འབྲེལ་དང་ཆབ་སྲིད་ཐབས་ཇུས་དང་དམག་ཆས་སོགས་ཀྱི་ཐོག་འབྲེལ་ལམ་གསར་པ་ཞིག་མགོ་འཛུགས་བྱས་ཡོད། གསར་འགོད་པ། དོན་གྲུབ།
