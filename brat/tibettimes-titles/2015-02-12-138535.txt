མི་ ༡༠༢༡༦༨ ཞིབ་བཤེར་ཐག་གཅོད་བྱས།

རྒྱ་ནག་གཞུང་འབྲེལ་ཕྱོགས་ནས་བསྒྲགས་པའི་གནས་ཚུལ་ནང་ལྐོག་ཟ་རུལ་སུངས་ལ་རྒོལ་རྡུང་ལས་འགུལ་ཁྲོད་དཔོན་རིགས་འབུམ་གཅིག་ལྷག་ལ་ཞིབ་བཤེར་ཐག་གཅོད་བྱས་ཡོད་ཚུལ་འཁོད་འདུག ཟླ་འདིའི་ཚེས་ ༨ ཉིན་ཀན་སུའུ་ཞིང་ཆེན་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གིས་གསལ་བསྒྲགས་སྤེལ་བ་ལྟར་ན། ༢༠༡༤ ལོར་ཞིང་ཆེན་ཡོངས་ཀྱི་དཔོན་རིགས་ ༢༩༡༠ ཙམ་ལ་ཞིབ་བཤེར་ཐག་གཅོད་བྱས་ཡོད་པ་དང་། དེ་དག་ཁྲོད་མི་ ༧༩༤ ཙམ་ཏང་ནས་ཕྱིར་འཕུད་བཏང་ཡོད་འདུག ཟླ་སྔོན་མའི་ཚེས་ ༢༧ ཉིན་བོད་རང་སྐྱོང་ལྗོངས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གིས་གསལ་བསྒྲགས་སྤེལ་བའི་ནང་། ༢༠༡༤ ལོར་དཔོན་ལྗོངས་ཡོངས་ཀྱི་དཔོན་རིགས་ ༢༤༠ ལ་ཞིབ་བཤེར་ཐག་གཅོད་བྱས་ཡོད་དང་། དེ་དག་ཁྲོད་མི་ ༡༣༨ ཙམ་ཏང་ནས་ཕྱིར་འཕུད་བཏང་ཡོད་འདུག ཟླ་སྔོན་མའི་ཚེས་ ༥ ཉིན། རྒྱ་ནག་དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ཁང་གིས་གསལ་བསྒྲགས་བྱས་པ་ལྟར་ན། ༢༠༡༤ ལོར་རྒྱ་ནག་རྒྱལ་ཡོངས་ཀྱི་དཔོན་རིགས་ ༥༣༤༠ ཙམ་ལ་ཞིབ་བཤེར་ཐག་གཅོད་བྱས་ཡོད་པ་དང་། དེ་དག་ལས་མི་ ༢༢༨༠ ཙམ་ཏང་ནས་ཕྱིར་འཕུད་བཏང་ཡོད་འདུག རྒྱ་ནག་གཞུང་གིས་གཏན་འབེབས་དོན་ཚན་ ༨ ཞེས་པ་ལག་བསྟར་མགོ་འཛུགས་བྱས་པ་ནས་བཟུང་ ༢༠༡༤ ཟླ་ ༡ ཚེས་ ༡༢ ཉིན་བར། རྒྱ་ནག་ཁྱོན་ཡོངས་སུ་ཏང་ཡོན་ལས་བྱེད་ ༧༡༧༤༨ ཙམ་ལ་ཞིབ་བཤེར་ཐག་གཅོད་བྱས་ཡོད་པ་དང་། དེ་དག་ལས་མི་ ༢༣༦༤༦ ཙམ་ཏང་ནས་ཕྱིར་འཕུད་བཏང་ཡོད་འདུག རྒྱ་ནག་དབུས་གཞུང་གིས་གཏན་འབེབས་བྱས་པའི་ ‘དོན་ཚན་བརྒྱད’ ཅེས་པ་ལག་བསྟར་མགོ་འཛུགས་བྱས་པ་ནས་ ༢༠༡༤ ཟླ་ ༡༢ ཚེས་ ༣༡ ཉིན་བར། རྒྱ་ནག་རྒྱལ་ཡོངས་མི་ ༡༠༢༡༦༨ ཙམ་ལ་ཞིབ་བཤེར་ཐག་གཅོད་བྱས་ཡོད་པ་དང་། དེ་དག་ལས་ ༣༡༣༣༨ ཙམ་ཏང་ནས་ཕྱིར་འཕུད་བཏང་ཡོད་འདུག
