
དབྱངས་ཅན་སྒྲོལ་མ་དམག་ཁོངས་ནས་ཕྱིར་བུད་འདུག

  རྒྱ་ནག་གི་མཁའ་དམག་གཞས་མ་གྲགས་ཆེན་ཧཱན་ཧོན་གཟའ་ཟླ་བའི་ཉིན་རྒྱུ་མཚན་ཁ་གསལ་མེད་པའི་ཐོག་ནས་དམག་ཁོངས་ནས་ཕྱིར་འཐོན་དགོས་བྱུང་ཡོད་འདུག རྒྱ་ནག་གི་སྤྱི་ཚོགས་བརྒྱུད་ལམ་ཁག་ཅིག་ནང་གནས་ཚུལ་ཐོན་པར། རྒྱ་ནག་གི་མཁའ་དམག་ཁོངས་སུ་ཡོད་པའི་གཞས་མ་གྲགས་ཆེན་ཧན་ཧོན་ (Han Hong)། བོད་མིང་ལ་དབྱངས་ཅན་སྒྲོལ་མ་ཟེར་བ་ཁོང་ཟླ་འདིའི་ཚེས་ ༧ ཉིན་དམག་ཁོངས་ནས་ཕྱིར་འཐོན་དགོས་བྱུང་ཡོད་པའི་གནས་ཚུལ་འདུག ཐེངས་འདིར་ཁོ་མོ་རང་འགུལ་གྱིས་ཐོན་པ་ཡིན་ནམ། ཡང་ན་སྲིད་གཞུང་གིས་འཐོན་དགོས་པའི་ཐག་གཅོད་བྱས་པ་གང་ཡིན་ཁ་གསལ་མེད་ཀྱང་། ཁོ་མོས་ཐེངས་འདིར་ཁོ་མོ་ཕྱིར་འཐོན་འདོད་མེད་ཀྱང་། རྒྱུ་རྐྱེན་ཞིག་གི་དབང་གིས་འཐོན་དགོས་བྱུང་ཡོད་པའི་རྣམ་འགྱུར་བསྟན་ཡོད།   རེས་གཟའ་ཟླ་བའི་ཉིན་ཧན་ཧོན་མོ་རང་གི་ཝེ་པའོ་དྲ་བའི་སྟེང་། ཁོ་མོའི་དམག་ལྭའི་པར་ཞིག་དང་དེའི་འོག་ཏུ། “ཁྱེད་ལ་ཁ་འབྲེལ་དགོས་ཀྱང་ཚེ་གང་པོར་ཁྱེད་ལ་དགའ” ཞེས་དང་། དེའི་རྗེས་ལ་ “ཚེ་འདིར་ང་ཤི་རུང་གསོན་རུང་མཁའ་དམག་གི་མི་ཡིན” ཞེས་བཀོད་ཡོད་པ་མ་ཟད། མིག་ཆུ་འདོན་བཞིན་པའི་རི་མོ་སོགས་བཀོད་དེ་རང་ཉིད་དམག་ནང་ནས་ཕྱིར་ཐོན་པ་ཁྱབ་བསྒྲགས་བྱས་ཡོད་འདུག   རྒྱ་ནག་གི་ཀྲུང་ཞིན་དྲ་རྒྱ་ (Zhongxin Wang) ཞེས་པའི་གསར་འགོད་པས་ཧན་ཧོན་ལས་རོགས་ཤིག་ལ་གནས་ཚུལ་དྲིས་པར། ཧན་ཧོན་དམག་ཁུལ་ནས་ཕྱིར་ཐོན་པ་བདེན་པ་ཡིན་ཚུལ་བརྗོད་ཡོད་འདུག   དེ་རིང་ཞི་ཐང་རིན་ཞེས་པའི་དྲ་བས་གནས་ཚུལ་སྤེལ་བའི་ནང་། ཧན་ཧོན་གྱི་གོ་གནས་ནི་རྒྱ་ནག་གི་མཁའ་སྐྱོད་དམག་དཔུང་ཆབ་སྲིད་སྡེ་ཚན་གྱི་ཟློས་གར་ཚོགས་པའི་ཚོགས་གཙོ་གཞོན་པ་ཡིན་ཞིང་། ཁོང་མོར་དམག་དཔོན་གཞོན་པའི་ཐོབ་ཐང་ཡོད་སྐོར་བཀོད་ཡོད།   ཧོང་ཀོང་མིན་པའོ་བརྒྱུད་ལམ་གྱིས། ཧན་ཧོན་ལོ་ ༡༥ ནས་རྒྱ་ནག་གི་དམག་མིའི་ནང་འཛུལ་བཞུགས་བྱས་ཏེ་ད་བར་ལོ་ ༢༩ རིང་དམག་ནང་དུ་བསྡད་ཡོད་ཅིང་། སྲིད་གཞུང་གིས་གློ་བུར་དུ་ཁོ་མོ་དམག་ཁུལ་ནས་ཕྱིར་འཕུད་བྱས་པ་དེ་ནི། དམག་མི་ཉུང་འཕྲི་གཏོང་པའི་སྲིད་བྱུས་ལ་འབྲེལ་བ་ཡོད་ཚུལ་བཀོད་འདུག   རྒྱ་ནག་དུས་དོན་དཔྱད་ཞིབ་པ་ཀྲའོ་ཨེར་ཅོན་ (Zhao Er Jun) གྱིས། “ཧན་ཧོན་དམག་ཁོངས་ནས་ཕྱིར་ཐོན་པའི་གནས་ཚུལ་བསྒྲགས་པ་དེ་ཕལ་ཆེར་རྒྱ་ནག་གི་བརྡ་ལན་ཞིག་རེད་བསམ་གྱི་འདུག ཞི་ཅིན་ཕིན་གྱིས་འགན་འཁུར་བའི་དམག་མི་ཉུང་འཕྲི་གཏོང་དགོས་ན། ཐོག་མར་དམག་ཁུལ་གྱི་ཟློས་གར་ཚོགས་པ་ནས་ཉུང་འཕྲི་གཏང་སྲིད་པ་རེད།” ཅེས་བརྗོད་འདུག   གླུ་མ་ཧན་ཧོན་ནམ་དབྱངས་ཅན་སྒྲོལ་མ་ནི་ཕྱི་ལོ་ ༡༩༧༡ ལོའི་ཟླ་ ༩ ཚེས་ ༢༦ ཉིན་བོད་ལྗོངས་ཆབ་མདོར་སྐྱེས་ཡོད་པ་དང་། རྒྱ་ནག་མི་དམངས་བཅངས་གྲོལ་དམག་དཔུང་ཟེར་བའི་ཟློས་གར་སློབ་གླིང་ནས་མཐར་ཕྱིན་ཡོད་འདུག ཁོ་མོའི་ཨ་ཕའི་མིང་ཧན་ཏི་ཅཱང་ (Han De Jiang) ཟེར་ཞིང་ཁོ་བོ་ནི་ཁྲིན་ཏུའུ་དམག་ཁུལ་གྱི་ཟློས་གར་ཚན་པའི་བཞད་གར་འཁྲབ་སྟོན་པ་ཞིག་ཡིན་འདུག ཕྱི་ལོ་ ༡༩༧༦ ལོར་འདས་གྲོངས་སུ་ཕྱིན་ཡོད་འདུག ཁོ་མོའི་ཨ་མའི་མིང་ལ་ཚེ་བརྟན་སྒྲོལ་མ་ཟེར་ཞིང་། དྲ་ཚུགས་འགའ་རེའི་སྟེང་ཡོན་ཞི་ (Yong Xi) ཞེས་རྒྱ་མིང་ཞིག་ཀྱང་བཀོད་འདུག རྒྱ་ནག་དྲ་རྒྱ་ཆེ་ཤོས་པེ་ཏའུ་ནང་དུ་ཧན་ཧོན་གྱི་མི་ཚེ་ལོ་རྒྱུས་བཀོད་པའི་ནང་ཁོ་མོ་བོད་པ་ཡིན་སྐོར་བྲིས་འདུག   ཟླ་སྔོན་མའི་ཟླ་མཇུག་ལ་རྒྱ་གཞུང་གིས་དམག་དཔུང་བཅོས་བསྒྱུར་གྱི་སྒྲོམ་གཞི་ཞིག་ཁྱབ་བསྒྲགས་བྱས་ཡོད་པ་རེད།   གསར་འགོད་པ། འཇིགས་མེད་རྒྱ་མཚོ།
