
བོད་ཀྱི་སྨན་རྩིས་ཁང་གི་མིང་ཐོག་ནས་གང་བྱུང་བྱེད་མཁན་ཚོ་ཁྲིམས་ཐོག་ལ་སྐྱོལ།

དེ་རིང་སྔ་དྲོར་སྨན་རྩིས་ཁང་མཐོ་སློབ་ཚོགས་ཁང་དུ་བོད་ཀྱི་ཆེས་མཐོའི་སྨན་པའི་ལྷན་ཚོགས་དབུ་བརྙེས་པའི་དུས་དྲན་ཐོག ད་ལྟའི་ཆར་ལྷན་ཚོགས་ཀྱི་ཁོངས་སུ་ཁྱོན་སྨན་པ་ ༤༥༠ དང་སྨན་སྦྱོར་ཁང་ ༥ ཡིས་དེབ་སྐྱེལ་བྱས་ཡོད་སྐོར་བརྗོད་སོང་།   དེ་རིང་སྔ་དྲོའི་ཆུ་ཚོད་ ༡༠།༣༠ ཐོག་ངྷ་བོད་ཀྱི་སྨན་རྩིས་ཁང་མཐོ་སློབ་ཚོགས་ཁང་དུ། བོད་ཀྱི་ཆེས་མཐོའི་སྨན་པའི་ལྷན་ཚོགས་དབུ་བརྙེས་ནས་ལོ་ངོ་ ༡༡ འཁོར་བའི་དུས་དྲན་དང་བསྟུན་ཏེ། ཉིན་གྲངས་ ༢ རིང་གི་དམིགས་བསལ་ཚོགས་འདུ་ཐེངས་གཉིས་པ་དབུ་འཛུགས་གནང་སོང་། ཚོགས་འདུ་དབུ་འབྱེད་མཛད་སྒོའི་ཐོག་བོད་མིའི་སྒྲིག་འཛུགས་འཕྲོད་བསྟེན་ལས་ཁུངས་ཀྱི་དྲུང་ཆེ་སྦྲ་གཉིས་ཚེ་རྒྱལ་ལགས་དང་སྤྱི་འཐུས་རྒྱུན་ལས། དེ་བཞིན་བོད་ཀྱི་ཆེས་མཐོའི་སྨན་པའི་ལྷན་ཚོགས་ཀྱི་རྒྱུན་ལས་དང་འབྲེལ་ཡོད་སྨན་པ་བགྲེས་གཞོན་བཅས་འཛོམས་འདུག སྐབས་དེར་སྨན་པའི་ལྷན་ཚོགས་ཀྱི་ཚོགས་གཙོ་ངག་དབང་བཟོད་པ་ལགས་ཀྱིས་ལྷན་ཚོགས་ཀྱི་དགོས་དམིགས་སྐོར་ལ་གཏམ་བཤད་གནང་དོན། སློབ་གྲྭ་དེབ་སྐྱེལ་དང་སྨན་སྦྱོར་དེབ་སྐྱེལ། སྨན་པ་དེབ་སྐྱེལ་བྱེད་པ་བཅས་ནི་སྨན་པའི་ལྷན་ཚོགས་ཀྱི་ལས་འགན་གཙོ་བོ་གསུམ་ཡིན་པ་དང་། དེ་ཚོ་འཐུས་ཤོར་ཕྱིན་པ་ཡིན་ན་རིག་གཞུང་ཉམས་པའི་རྒྱུ་གཙོ་བོ་ཞིག་ཆགས་ཀྱི་ཡོད། བོད་མིའི་སྒྲིག་འཛུགས་ནས་རྒྱབ་སྐྱོར་ཤུགས་ཆེན་པོ་དགོས་པ་གལ་ཆེན་པོ་རེད། དཔེར་ན། གསོ་བ་རིག་པ་ལག་ལེན་བསྟར་མཁན་དྷ་རམ་ས་ལ་གཅིག་པུ་མ་རེད། དེང་སང་གཞན་དབང་སྟབས་བསྟུན་གྱི་ཐོག་ནས་ལུང་པ་མང་པོ་ཞིག་ལ་ཁྱབ་ཡོད་སྟབས། དེ་ཚོ་ཚང་མར་ང་ཚོ་སྨན་པའི་ལྷན་ཚོགས་ཚོགས་པ་གཅིག་གིས་སྙོབ་ཐུབ་རྒྱུ་ཁག་པོ་ཆགས་ཡོད་སྟབས། དེ་ཚོར་གཞིས་མགོ་དང་སྐུ་ཚབ་དོན་གཅོད་ཚང་མས་འཕྱར་གྲུ་མཉམ་འདེགས་གནང་བའི་འབོད་སྐུལ་ཞུ་རྒྱུ་ཡིན། སྨན་པའི་ལྷན་ཚོགས་ཀྱི་ངོས་ནས་སྨན་པ་ཚད་ལྡན་ཐོན་མེད་པ་ཚོ་བཀག་སྡོམ་བྱེད་དགོས་པའི་རྒྱུ་མཚན་ནི་སྨན་རང་དགར་སྦྱོང་མཁན་ཚོར་བཀག་སྡོམ་བྱེད་པའི་དམིགས་ཡུལ་མིན་པར། དུས་རབས་ཉེར་གཅིག་པ་དང་མཐུན་པའི་ལམ་ཁ་ཞིག་གི་ཐོག་འགྲོ་དགོས་ན་ང་རང་ཚོའི་སྤུས་ཚད་སྟངས་འཛིན་དང་རྒྱལ་སྤྱིའི་ཁྲིམས་འགྲོ་སྟངས་ལྟར་འགྲོ་དགོས་ཀྱི་ཡོད་སྟབས། དམིགས་ཡུལ་དེ་ཚོ་གཞིར་བཞག་པའི་ཐོག་ནས་སྟངས་འཛིན་བྱེད་བཞིན་ཡོད་སྐོར་སོགས་བརྗོད་སོང་། སྨན་རྩིས་ཁང་གི་སྤྱི་ཁྱབ་འགན་འཛིན་བཀྲ་ཤིས་ཚེ་རིང་ལགས་ཀྱིས་ཡིག་ཐོག་ནས་ཚོགས་འདུ་དེར་གསུང་བཤད་ཅིག་བསྐུར་ཡོད་པའི་ནང་། སྨན་པའི་ལྷན་ཚོགས་དང་བོད་ཀྱི་སྨན་རྩིས་ཁང་གཉིས་ཀྱི་འབྲེལ་བ་ནི་འཛམ་གླིང་རྒྱལ་སྤྱི་དང་ཨ་རི་གཉིས་ཀྱི་འབྲེལ་བ་ལྟ་བུ་ཡིན་པ་དང་། དེང་སང་རྒྱལ་སྤྱིའི་ནང་བོད་ཀྱི་གསོ་བ་རིག་པའི་ཐོག་དོ་སྣང་དང་དགའ་ཞེན་ཇེ་ཆེར་འགྲོ་བཞིན་པའི་སྐབས་སུ་གང་བྱུང་བྱེད་མཁན་ཡང་མང་དུ་འགྲོ་བཞིན་པར་བརྟེན། སྨན་པའི་ལྷན་ཚོགས་ནས་བྱ་ར་ལྟ་བུ་གནང་རོགས་ཞུ་རྒྱུ་མ་ཟད། ལས་དོན་འདིའི་ཐོག་འགྲོ་སོང་ཆེན་པོ་ཡོན་སྟབས། བོད་ཀྱི་སྨན་རྩིས་ཁང་ནས་དཔལ་འབྱོར་མཐུན་རྐྱེན་རྒྱ་མི་ཆུང་བ་ཞུ་ཆོག་ཅེས་སྔ་རྗེས་སུ་ཞུས་ཡོད་ལ། བོད་ཀྱི་སྨན་རྩིས་ཁང་གི་མཚན་བྱང་བེད་སྤྱོད་གཏོང་མཁན་ཚོར་སྨན་པའི་ལྷན་ཚོགས་ཀྱིས་སློབ་གསོ་གོ་བཀོན་གནང་སྟེ། དགེ་ཕན་མ་བྱུང་ཚེ་ཁྲིམས་གཏུགས་བྱས་ཏེ་རྩ་ཆེའི་དུས་ཚོད་དང་དཔལ་འབྱོར་གྲོན་གཏོང་རྒྱུར་ཐེ་ཚོམ་མེད་སྐོར་སོགས་འཁོད་འདུག དེ་རྗེས་ཨ་ཡུ་ཤི་བོད་ཀྱི་གསོ་བ་རིག་པའི་སྨན་པའི་སྐུ་ཚབ་འཕྲིན་ལས་དབྱངས་འབྱོར་ལགས་ཀྱིས། སྨན་པའི་ལྷན་ཚོགས་ཀྱི་ངོས་ནས་ལ་དྭགས་དང་ཧི་མ་ཡ་ལ་ཁུལ་གྱི་སྨན་པ་དང་སྨན་སྦྱོར་ཁང་ཚོར། ལྷན་ཚོགས་ཀྱི་ཁོངས་སུ་ཁྲིམས་མཐུན་དེབ་སྐྱེལ་དགོས་པའི་རེ་སྐུལ་རིམ་པར་ཞུས་ཀྱང་དོན་འབྲས་དེ་ཙམ་ཐོན་གྱི་མེད་ལུགས་བརྗོད་སོང་། དེའི་རྗེས་སྤྱི་འཐུས་སྐུ་ཚབ་རྒྱ་རིགས་ཐར་ལགས་ཀྱིས། བོད་ཀྱི་ནང་ཆོས་དང་གསོ་བ་རིག་པ་འདི་གཉིས་ནི། སྲོལ་རྒྱུན་རང་བཞིན་གྱི་ཐོག་ནས་མི་རིགས་ཀྱི་ངོ་བོ་མཚོན་བྱེད་དང་མི་རིགས་ཀྱི་འདུ་ཤེས་གྲུབ་བྱེད་ཀྱི་རིག་གཞུང་མ་འདྲ་ཞིག་ཆགས་ཡོད་ཀྱང་། ལོ་རབས་ ༥༠ པ་ནས་མགོ་བཙུགས་ཏེ་རྒྱ་ནག་དམར་ཤོག་གིས་བོད་ཀྱི་རྩ་ཆེ་བའི་རིག་གཞུང་དེ་འཕྲོ་བརླག་དང་བརྡབ་གསིག་ཆེན་པོ་གཏོང་བཞིན་ཡོད་པ་རེད། གལ་སྲིད་བོད་ཀྱི་རིག་གཞུང་འདི་མེད་པར་བཏང་བ་ཡིན་ན་བོད་མི་རིགས་འདི་རྩ་བརླག་ཏུ་འགྲོ་རྒྱུ་དང་། བོད་མི་རིགས་འདི་རྩ་བརླག་ཏུ་ཕྱིན་པ་ཡིན་ན། བོད་ཟེར་བའི་ག་དུས་ཡིན་ནའང་རྒྱ་ནག་གཞུང་གི་སེམས་ན་མ་བདེ་བ་ཞིག་ཡོད་པ་དེ་གྲགས་མེད་ཚོར་མེད་གཏོང་འཆར་ཡོད་པ་དང་། འཆར་གཞི་དེ་འདྲ་ཞིག་གི་ཐོག་ནས་བོད་ཀྱི་བཅུ་ཕྲག་རིག་གནས་ཀྱི་ནང་ནས་བོད་མི་རིགས་ལ་ཕན་ཐོགས་ཤིང་རྩ་ཆེ་བའི་རིག་གཞུང་ནང་ཆོས་དང་གསོ་བ་རིག་པ་རྒྱ་གཞུང་གིས་འཕྲོ་བརླག་ཏུ་བཏང་ཚར་བ་རེད། ཡིན་ནའང་། སྤྱི་ཚོགས་ཀྱི་གནས་བབ་དེ་འདྲ་ཞིག་གི་འོག་ནས་༧གོང་ས་མཆོག་གི་བཀའ་དྲིན་ཉག་གཅིག་ལ་བརྟེན་ཏེ། ང་ཚོས་སྐྱབས་བཅོལ་གྱི་ལུང་པ་ནས་སོ་སོའི་རིག་གཞུང་གི་ནང་ནས་དམིགས་བསལ་གྱིས་བོད་ཀྱི་གསོ་བ་རིག་པ་འདི་ཉམས་པ་སོར་ཆུད་དང་མ་ཉམས་གོང་འཕེལ་དུ་བཏང་སྟེ། ད་ལྟ་ང་ཚོས་བལྟ་ཡོང་བའི་སྐབས་སུ་རྒྱལ་ཁའི་ཕྱོགས་ཀྱི་གྲུབ་འབྲས་ཤིག་མཐོང་ཡོང་གི་འདུག བཙན་བྱོལ་སྤྱི་ཚོགས་ནང་ནས་བོད་ཀྱི་གསོ་བ་རིག་པར་སླར་ཡང་གསོན་ཤུགས་རྒྱས་ཏེ་རིམ་གྱིས་འཕེལ་རྒྱས་འགྲོ་བཞིན་པའི་རིག་གཞུང་ཞིག་ཆགས་ཡོད་སྟབས། མ་འོངས་པ་ལའང་འདི་ལ་བདག་པོ་རྒྱག་རྒྱུ་དང་། ཉམས་པ་སོར་ཆུད་དང་མ་ཉམས་གོང་འཕེལ་གཏོང་རྒྱུ་དེ་ཧ་ཅང་གལ་ཆེན་པོ་ཞིག་ཏུ་མཐོང་གི་འདུག ཅེས་བརྗོད་སོང་། མཇུག་ཏུ་སྐུ་མགྲོན་གཙོ་བོ་སྦྲ་གཉིས་ཚེ་རྒྱལ་ལགས་ཀྱིས། ལྷན་ཚོགས་འདི་ནི་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་ཁྲིམས་བཟོ་ལྷན་ཁང་ནས་ཁྲིམས་ཡིག་གཏན་ལ་ཕབ་ཏེ་ཟུར་འཛུགས་གནང་ཞིང་། ༧གོང་ས་མཆོག་གི་བཀའ་འཁྲོལ་ཐོབ་པའི་ལྷན་ཚོགས་གལ་ཆེན་པོ་ཞིག་ཡིན་པ་དང་། འདིའི་དམིགས་ཡུལ་གཙོ་བོ་ཐུན་མོང་མ་ཡིན་པའི་བོད་ཀྱི་གསོ་བ་རིག་པ་སློབ་གཉེར་གནང་སྟངས་དང་། གསོ་བ་རིག་པའི་སྨན་པ་དང་སྨན་སྦྱོར་ཚད་དང་ལྡན་པ་ཞིག་ཡོང་བར་སྟངས་འཛིན་བྱེད་མཁན་ཞིག་དགོས་པ་ལྟར་ཆེད་འཛུགས་གནང་ཡོད་པ་དང་། ད་ལྟའི་དུས་སུ་འཛམ་གླིང་ཐོག་བོད་ཀྱི་གསོ་བ་རིག་པའི་སྨན་སྦྱོར་སྙན་གྲགས་དང་སྦྲགས་ཕན་ཡོན་ག་ཚོད་ཡོད་མེད་གསལ་བོ་ཆགས་ཡོང་དུས། སྨན་སྦྱོར་ཚད་ལྡན་མིན་པ་དང་སྨན་རྫུན་མ་རིགས་བཟོས་ཏེ་ཚོང་གི་ཁེ་བཟང་ལེན་གྱི་སྐབས་འཚོལ་བ་བཀག་འགོག་དང་། སྨན་པ་ཡིན་ནའང་འཁྱར་རེ་འཁྱོར་རེ་མ་ཡིན་པར་ཚད་དང་ལྡན་པ་ནད་པར་སྨན་བཅོས་ཏག་ཏག་བྱེད་ཐུབ་མཁན་ཞིག་ཡོང་བར་སྟངས་འཛིན་བྱེད་མཁན་དེ་སྨན་པའི་ལྷན་ཚོགས་ཆགས་ཡོད་པ་ལྟར། འདས་པའི་ལོ་ ༡༡ ནང་གྲུབ་འབྲས་གང་ལ་གང་འཚམ་ཐོན་ཡོད་པ་དང་། ད་ལྟའི་གནས་སྟངས་ག་འདྲ་ཆགས་ཡོད་པ་དང་མ་འོངས་པར་དུས་ཀྱི་འགྱུར་འགྲོས་དང་བསྟུན་ཏེ་ག་བར་སླེབ་ཀྱི་ཡོད་མེད་ཐོག་བསམ་བློ་གཏོང་རྒྱུ་གནད་འགག་ཆེ་ཤོས་ཆགས་ཡོད་སྐོར་དང་། སྨན་པའི་ལྷན་ཚོགས་ཀྱི་ཚོགས་མི་འོས་འདེམས་སྐབས་ནས་བཟུང་བོད་ཀྱི་གསོ་བ་རིག་པའི་གལ་གནད་རང་བཞིན་ལ་བསམ་བློ་བཏང་སྟེ། ཚོགས་མི་འོས་འདེམས་བྱས་པ་ཡིན་ན་མ་འོངས་པར་ཚོགས་པའི་ལས་ཀ་ལེགས་པོ་ཡོང་བར་ཕན་ཐོགས་ཡོད་སྐོར་སོགས་གསུངས་སོང་། ཚོགས་ཐོག་ཏུ་འདས་ལོའི་ལས་བསྡོམས་སྙན་སྒྲོན་ཞུས་པའི་ནང། ད་ལྟའི་ཆར་ཆེས་མཐོའི་སྨན་པའི་ལྷན་ཚོགས་སུ་ཐད་བསྐོས་ཚོགས་མི་ ༢ དང་འོས་འདེམས་གནང་བའི་ཚོགས་མི་ ༨ བཅས་ཁྱོན་འཛིན་སྐྱོང་ཚོགས་མི་བཅུ་དང་། ལས་ཁུངས་སུ་གཏན་འཇགས་ལས་བྱེད་ཆེད་ལས་གཅིག་དང་། རྩིས་པ་གཅིག ལས་དྲུང་དང་དངུལ་གཉེར་གཅིག་ལྕོགས་གཅིག ལས་བྱ་ཡིག་གད་ཅིག་ལྕོགས་གཅིག་བཅས་ཁྱོན་ལས་བྱེད་ ༤ ཡོད་པ་དང་། ལྷན་ཚོགས་འོག་ཕྱི་རྒྱལ་ཁག་ཏུ་ས་གནས་འབྲེལ་མཐུད་པ་ ༡༧ དང་། དེབ་སྐྱེལ་བྱས་ཟིན་པའི་སྨན་པ་ ༤༥༠ དང་མི་རིགས་འདྲ་མིན་གྱི་བོད་སྨན་མཐུན་གྲོགས་ཚོགས་མི་ ༥༠ བཅས་ཡོད་པ་དང་། གསོ་རིག་སྨན་སྦྱོར་ཁང་ངམ་སྨན་སྦྱོར་སྡེ་ཚན་ཁྱོན་ ༥ ཡིས་དེབ་སྐྱེལ་བྱས་ཟིན་ཡོད་སྐོར་དང་། ཕྱི་ལོ་ ༢༠༡༤ ཟླ་ ༤ ཚེས་ ༡ ནས་ཕྱི་ཟླ་ ༡༡ ཚེས་ ༢༠ བར་བྱུང་ཧིན་སྒོར་(༧༡༠༡༦༩)དང་། སོང་ཧིན་སྒོར་(༡༠༠༣༧༨༤)ཡིན་པ་དང་། ད་ལྟ་ཡོད་བཞིན་པ་ཧིན་སྒོར་(༢༠༨༨༩༩༦)ཡོད་སྐོར་བཅས་བཀོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
