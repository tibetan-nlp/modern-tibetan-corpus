ནག་ཆུ་སོག་རྫོང་གི་གྲྭ་པ་ ༦ འཇུ་བཟུང་།

རྒྱ་གཞུང་གིས་ནག་ཆུ་སོག་རྫོང་དྲིལ་མདའ་དགོན་པའི་གྲྭ་བ་ ༤ དང་། སོག་ཙན་དན་དགོན་པའི་གྲྭ་བ་ ༢ བཅས་སྔ་ཕྱི་འཛིན་བཟུང་བྱས་ཏེ་ད་ཆར་གར་ཡོད་མ་ཤེས་པར་ལྷག་འདུག ངག་དབང་ཐར་པ་ལགས་ཀྱིས་འདིར་གར་གནས་ཚུལ་བཏང་འབྱོར་ལྟར་ན། རྒྱ་གཞུང་གིས་ཟླ་འདིའི་ཚེས་ ༡༧ གྱི་མཚན་མོར་ནག་ཆུ་སོག་རྫོང་ཁྲི་རྡོ་ཡུལ་ཚོ་དྲིལ་མདའ་དགོན་པའི་དབུ་མཛད་ལས་ཐོག་གྲྭ་ཚངས་དབྱང་རྒྱ་མཚོ་ལགས་དང་། ཚེ་དབང་། ཨ་ཚེ། རྒྱལ་མཚན་བཅས་གྲྭ་བ་བཞི་འཛིན་བཟུང་བྱས་ཏེ་ད་ཆར་གར་ཡོད་ཅི་བྱུང་གི་གནས་ཚུལ་ཅི་ཡང་མ་ཤེས་པར་ལྷག་འདུག་ལ། ད་ཆར་རྒྱ་གཞུང་གིས་ཁྲི་རྡོ་ཡུལ་ཚོའི་བརྒྱུད་ལམ་ཡོངས་རྫོགས་བཀག་ཡོད་པ་མ་ཟད། ས་གནས་སུ་དམག་སྒར་བརྒྱབ་ཏེ་མི་མང་གི་འགྲོ་སྡོད་ལ་དམ་དྲག་བྱེད་ཀྱི་ཡོད་འདུག དེ་བཞིན།  ཟླ་འདིའི་ཚེས་ ༡༤ ཉིན་ནག་ཆུ་སོག་རྫོང་ཙན་དན་དགོན་པའི་ནང་རྒྱ་ནག་གི་རྒྱ་དར་མེར་བསྲེཊ་དང་། དགོན་པར་བཅའ་སྡོད་ལས་བྱེད་ཀྱི་སྒོ་ངོས་སུ་ “བོད་རང་བཙན་ཡིན”ཞེས་པའི་ཡི་གེ་བྲིས་པའི་ཉེས་མིང་འོག གཡག་ལ་ཡུལ་ཚོའི་གཡང་ངེ་གྲོང་གཡང་གཟིགས་ཚང་གི་རང་ལོ་ ༢༠ ཡིན་པའི་གྲྭ་དགེ་འདུན་གྲགས་པ་འཛིན་བཟུང་བྱས་ཡོད་པ་མ་ཟད། ཚེས་ ༡༦ ཉིན་འཕྲིན་ཐུང་བརྒྱུད་རྒྱ་གཞུང་ལ་ངོ་རྒོལ་གྱི་རྩོམ་ཡིག་ཅིག་གྲོགས་པོ་དང་དགོན་པའི་ལས་དོན་མགོ་ཁྲིད་ཚོའི་ཁ་པར་ནང་དུ་བསྐུར་བའི་ཉེས་མིང་འོག སོག་རྫོང་གཡག་ལ་ཡུལ་ཚོའི་གཡེག་གུ་གྲོང་གི་ལྷ་འཁྱིལ་ཚང་གི་རང་ལོ་ ༢༠ ཡིན་པའི་ཙན་དན་དགོན་པའི་གྲྭ་ཆོས་དབྱིངས་སྐལ་ལྡན་འཛིན་བཟུང་བྱས་ཏེ། ད་ཆར་ཁོང་གཉིས་གང་དུ་བཀག་ཉར་བྱས་ཡོད་མེད་ཤེས་རྟོགས་མ་ཐུབ་པར་ལྷག་འདུག ད་ཆར་སོག་ཙན་དན་དགོན་པའི་གྲྭ་པ་མང་པོ་ཞིག་ཉིན་ལྟར་རྒྱ་གཞུང་གི་ཉེན་རྟོག་ཁང་ནང་དུ་འབོད་འགུག་གིས་འདྲི་བརྩད་བྱེད་བཞིན་ཡོད་པ་མ་ཟད། དགོན་པའི་གྲྭ་རྣམ་རྒྱལ་ཚུལ་ཁྲིམས་ཀྱི་ཨ་མ་ན་ཚའི་དབང་གིས་ཁྲིན་འདུར་སྨན་བཅོས་གནང་བར་ཕྱིན་ཡོད་ནའང་། ས་གནས་ཉེན་རྟོག་པ་པ་སངས་དགྲ་འདུལ་གཙོས་པའི་ཉེན་རྟོག་པ་ཁག་ཅིག་གིས་སྨན་བཅོས་མ་ཚར་གོང་བཙན་དབང་གིས་དགོན་པར་འབོད་འགུག་བྱས་ཡོད་པ་མ་ཟད། ས་གནས་གང་དུ་སོང་ཡང་རྗེས་འདེད་བྱེད་བཞིན་ཡོད་འདུག་ལ། ས་གནས་བོད་མི་ཚོའི་ཁ་པར་ནང་པར་དང་རྩོམ་ཡིག་སོགས་བསྙད་འདོག་ས་རག་པ་ཙམ་གྱི་འཛིན་བཟུང་བཙོན་འཇུག་དང་མནར་གཅོད་གཏོང་གི་ཡོད་འདུག གསར་འགོད་པ། དོན་གོ
