གྲྭ་པདྨ་རྡོ་རྗེ་ལཊ་འཛིན་བཟུང༌།

ཁ་སང་སྡེ་དགེ་རྫ་ཁོག་རྫ་དགོན་གསར་དགོན་པའི་གྲྭ་པདྨ་རྡོ་རྗེ་ཞུ་བ་ཞིག་གིས་རྒྱ་གཞུང་གི་བཙན་གནོན་ལ་ཞི་རྒོལ་གནང་སྟེ་ཉེན་རྟོག་པས་བཙན་འཁྲིད་བྱས་ཡོད་འདུག འདི་གར་འབྲེལ་ཡོད་ནས་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། ཁ་སང་ཕྱི་ཚེས་ ༦ ཉིན་གྱི་བོད་ནང་གི་ཉིན་རྒྱབ་ཆུ་ཚོད་ ༢ པ་ཡས་མས་སུ། དཀར་མཛེས་ཁུལ་དཀར་མཛེས་རྫོང་ཐོག་ཏུ་གྲྭ་པདྨ་རྡོ་རྗེ་ཞུ་བ་ཞིག་གིས། ༼ྋརྒྱལ་བ་རིན་པོ་ཆེ་བོད་དུ་གདན་འདྲེན་ཞུ་དགོས། ཁོང་སྐུ་ཚེ་ཁྲི་ལོར་བརྟན་པ་ཤོག ༽ཅེས་ཞི་བའི་སྐད་འབོད་གནང་ཡོད་པ་དང༌། ཅུང་ཙམ་ལས་མ་འགོར་ས་གནས་ཉེན་རྟོག་པ་བདུན་བརྒྱད་ཅིག་བསླེབས་ཏེ། གང་བྱུང་གཅར་རྡུང་བཏང་ནས་ཁོང་གི་ཁ་སྣ་སོགས་ནས་ཁྲག་དམར་འཛར་འཛར་དུ་བཙན་འཁྲིད་བྱས་ཡོད་འདུག ད་ལྟའི་ཆར་ཁོང་གང་དུ་བཀག་ཉར་བྱས་ཡོད་པ་དང་གནས་སྟངས་ག་འདྲ་ཡིན་སོགས་ཤེས་རྟོགས་ཐུབ་ཀྱི་མེད་སྟབས། ཁོང་གི་དགོན་པ་དང་ཁྱིམ་མི་གཉེན་འབྲེལ་ཚོས་ཐུགས་ཁྲལ་ཆེན་པོ་གནང་བཞིན་ཡོད་འདུག གྲྭ་པདྨ་རྡོ་རྗེ་ལགས་ནི་སྡེ་དགེ་རྫ་ཁོག་དོ་རྡ་གྲོང་ཚོའི་རྫ་དགོན་གསར་དགོན་ནས་ཡིན་པ་དང༌། ད་ལྟ་རང་ལོ་ ༡༧ ཡིན་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
