རྨ་ལྷོ་རྩེ་ཁོག་ཏུ་མེར་བསྲེཊ།

ཁ་སང་གཟའ་ཉི་མའི་སྔ་དྲོ་རྨ་ལྷོ་རྩེ་ཁོག་ཏུ་གྲྭ་བ་ཞིག་གིས་རང་ལུས་མེར་བསྲེཊ་ཀྱིས་རྒྱ་གཞུང་གི་དྲག་གནོན་དབང་བསྒྱུར་སྲིད་ཇུས་ལ་ངོ་རྒོལ་གནང་ཡོད་འདུག ཁ་སང་ཕྱི་ཚེས་ ༡༦ ཉིན་གྱི་སྔ་དྲོ་རྨ་ལྷོ་རྩེ་ཁོག་རྫོང་བྱ་དར་ཞང་བྱ་དར་དགོན་པའི་གྲྭ་བ་ཞིག་གིས། དགོན་པའི་གླིང་གི་ཕྱི་རོལ་དུ་རང་ལུས་མེར་བསྲེཊ་ཀྱིས་རྒྱ་གཞུང་གི་དྲག་གནོན་དབང་བསྒྱུར་སྲིད་ཇུས་ལ་ངོ་རྒོལ་གནང་ཡོད་འདུག གནས་ཚུལ་བྱུང་རྗེས་རྒྱ་གཞུང་གིས་ས་གནས་ཀྱི་བརྒྱུད་ལམ་ལ་བཀག་སྡོམ་ནན་མོ་བྱས་ཡོད་སྟབས། མེར་བསྲེཊ་གནང་མཁན་གྱི་མཚན་དང་འཆི་གསོན། ས་གནས་ཀྱི་དམ་དྲག་གནས་སྟངས་སོགས་གསལ་བོ་མ་ཤེས་པར་ལྷག་ཡོད། གནས་ཚུལ་འདིའི་ཐད་ཁ་སང་རྒྱ་ནག་གཞུང་འབྲེལ་ཞིན་ཧྭ་གསར་ཁང་གིས། ཕྱི་ཚེས་ ༡༦ ཉིན་རྨ་ལྷོ་ཁུལ་རྩེ་ཁོག་རྫོང་དུ་དགེ་འདུན་པ་ཞིག་གིས་རང་ལུས་མེར་བསྲེགས་བཏང་ཡོད་པ་དང། གནས་ཚུལ་དེ་བྱུང་མ་ཐག་ཉེན་རྟོག་པ་ས་གནས་སུ་ཕྱིན་ཡོད་པའི་གནས་ཚུལ་དྲ་ལམ་དུ་བཀོད་འདུག་ལ། ཡང་ཨེ་ཕེ་གསར་ལས་ཁང་ (AP) གིས་དྲ་རྒྱུའི་ཐོག རྨ་ལྷོ་རྩེ་ཁོག་རྫོང་ས་གནས་བདེ་འཇགས་ལྷན་ཁང་གི་ལས་བྱེད་པ་ཞིག་གིས། ཁོང་གི་མིང་དང་ཞིབ་ཕྲའི་གནས་ཚུལ་སོགས་གསལ་བོ་བཤད་མེད་ཀྱང་། ཁ་སང་ས་གནས་སུ་རང་ལུས་མེར་བསྲེགས་ཀྱི་དོན་རྐྱེན་བྱུང་བ་དངོས་ཡོད་ཡིན་ཚུལ་དང་། མེར་བསྲེཊ་གནང་ཡུལ་དགོན་པའི་གླིང་གི་ཕྱི་རོལ་དུ་བྱུང་ཡོད་ཚུལ་ངོ་སྤྲོད་བྱས་ཡོད་སྐོར་བཀོད་འདུག གསར་འགོད་པ། དོན་གོ
