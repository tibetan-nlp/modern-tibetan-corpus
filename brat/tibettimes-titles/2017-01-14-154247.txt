དུས་འཁོར་དབང་ཆེན་གྱི་བྱུང་སོང་རྩིས་ཁྲ།

དེ་རིང་ཉིན་གུང་གནས་མཆོག་རྡོ་རྗེ་གདན་གྱི་གསུང་ཆོས་ར་བར། ༧གོང་ས་མཆོག་གི་དཔལ་དུས་ཀྱི་འཁོར་ལོའི་དབང་ཆེན་ཐེངས་ ༣༤ པ་མཇུག་བསྒྲིལ་བའི་མཛད་སྒོ་ཚོགས་པའི་སྟེང་། དུས་འཁོར་དབང་ཆེན་གོ་སྒྲིག་ཚོགས་ཆུང་གི་ཚོགས་གཙོ་ཆོས་རིག་བཀའ་བློན་གཡུ་ཐོག་ཀརྨ་དགེ་ལེགས་མཆོག་གིས། ཐེངས་འདིའི་དུས་འཁོར་དབང་ཆེན་གྱི་རྩིས་ཁྲ་དབྱིན་བོད་གཉིས་ནང་གསལ་བསྒྲགས་གནང་དོན་ལྟར་ན། ཕྱི་ལོ་ ༢༠༡༧ ཟླ་ ༡ ཚེས་ ༢ ནས་ ༢༠༡༧ ཟླ་ ༡ ཚེས་ ༡༤ ཉིན་བར་བྱུང་སོང་བསྡད་གསུམ་གྱི་གནས་སྟངས་གཤམ་གསལ་ལྟར་ཆགས་ཡོད་འདུག བྱུང་དངུལ། དུས་དབང་ཡོན་བདག་རྣམས་ཀྱིས་ཕུལ་བ་ཧིན་སྒོར་འབུམ་ ༨༠༥ དང་ ༣༤༢༥༣ ། ཞལ་འདེབས་ཁག་ལྔ་ཡོད་པ་ལས་དུས་འཁོར་དབང་ཆེན་ལ་ཞལ་འདེབས་སྒོར་འབུམ་ ༢༩༥ དང་ ༣༨༥༤༡ ། དགེ་འདུན་གྱི་མང་ཇ་དང་ཞལ་བག་ལ་ཞལ་འདེབས་སྒོར་འབུམ་ ༦༣༣ དང་ ༩༦༦༦ ། དགེ་འདུན་པར་སྐུ་འགྱེད་ལ་ཞལ་འདེབས་སྒོར་འབུམ་ ༥༢༠ དང་ ༡༣༥༤༤ ། བརྟན་བཞུགས་ཀྱི་ཞལ་འདེབས་ལ་སྒོར་འབུམ་ ༡༠༨ དང་ ༨༥༦༧༥ ། ཆོས་ཞུ་བའི་གུར་གྱི་ཆེད་དུ་བྱུང་བ་སྒོར་འབུམ་ ༩༣ དང་ ༤༨༦༣༧ ། ཕྱི་དངུལ་ཧིན་སྒོར་དུ་འཛའ་བསྒྱུར་བྱས་པར་སྒོར་འབུམ་ ༥༨ དང་ ༦༩༠༦༢ ། སོང་དངུལ། དུས་འགོར་དབང་ཆེན་གྱི་འགལ་རྐྱེན་བར་ཆད་སེལ་ཆེད་ཞབས་རིམ་ལ་སྒོར་འབུམ་ ༡༨ ། གསུང་ཆོས་ར་བ་གསར་སྐྲུན་དང་དབུ་གུར། སྡིངས་ཆ། རླུང་འཁོར་འཕྲུལ་ཆས་སོགས་ལ་སྒོར་འབུམ་ ༢༢༥ ། དུས་དབང་དང་འབྲེལ་བའི་ཕྱག་དཔེ་ཁག་ ༨ དཔར་སྐྲུན་ལ་སྒོར་འབུམ་ ༤༥༠ །(དབྱིན་སྐད་ནང་འབུམ་ ༤༥ ཞེས་གསུངས་སོང་།) དགེ་འདུན་པར་སྐུ་འགྱེད་ལ་སྒོར་འབུམ་ ༡༠༠༠༨ །(དབྱིན་སྐད་ནང་འབུམ་ ༡༠༠༨ ཞེས་གསུངས་སོང་།) བརྟན་བཞུགས་ལ་འགྲོ་གྲོན་སྒོར་འབུམ་ ༦༠ ། བརྟན་བཞུགས་ལ་དངུལ་སྨར་ཕུལ་བ་སྒོར་འབུམ་ ༥༥ དང་ ༥༥༥༥༥ ། མང་ཇ་དང་ཞལ་བག་འབྲས་སིལ་སོགས་ལ་སྒོར་འབུམ་ ༢༥༠ ། ཆོས་ཞུ་བའི་སྡོད་གུར་ལ་སྒོར་འབུམ་ ༡༩༩ ། བོད་ནས་ཕེབས་པའི་ཆོས་ཞུ་བར་སྣེ་ལེན་ལ་སྒོར་འབུམ་ ༨༦ དང་ ༥༠༠༠༠ ། ཉམ་ཐག་གི་སྣེ་ལེན་དང་ལམ་གྲོན་ལ་སྒོར་འབུམ་ ༡༦༠ ། མི་མང་བདེ་འཇགས་དང་ཆོས་ཞུ་བ་དེབ་འགོད་སོགས་ལ་སྒོར་འབུམ་ ༦༥ ། (འདིའི་བར་དབྱིན་སྐད་ནང་སྒྲ་སྒམ་དང་བརྙན་སོགས་ལ་སྒོར་འབུམ་ ༡༨༥ ཞེས་གསུངས་སོང་།) མི་མང་ལ་དྲིལ་བསྒྲགས་ལས་དོན་ཆེད་སྒོར་འབུམ་ ༡༥ ། མི་མང་འཕྲོད་བསྟེན་ལ་སྒོར་འབུམ་ ༦༥ ། ཆུ་གློག་གཙང་སྦྲ་སོགས་ལ་སྒོར་འབུམ་ ༢༣༠ ། མགྲོན་ཁང་དང་སྣུམ་འཁོར་གླ་རིན་ལ་སྒོར་འབུམ་ ༡༠༠ ། ལས་ཁུངས་ཀྱི་འཛིན་ཆས་དང་དཔར་སྐྲུན་སོགས་ལ་སྒོར་འབུམ་ ༧ ། ལས་བྱེད་པའི་གསོལ་ཐབ་ཁང་ལ་སྒོར་འབུམ་ ༥༠ ། འགྲིམ་འགྲུལ་ཟུར་ཕོགས་དང་དམིགས་བསལ་ལམ་གྲོན་བཅས་ལ་སྒོར་འབུམ་ ༩༥ ། སྡེ་ཚན་ཁག་ཅིག་ལ་ཐུགས་རྗེ་ལེགས་འབུལ་ཆེད་ཟུར་བཅད་བྱས་པ་སྒོར་འབུམ་ ༧༥ ། ལྷག་དངུལ། ངེས་མེད་འགྲོ་གྲོན་གྱི་དམིགས་སྒོའི་ཆེད་དུ་ལྷག་པ་སྒོར་འབུམ་ ༤༩ དང་ ༢༡༨༢༣ ། ཞེས་གསལ་བསྒྲགས་གནང་བ་དང་འབྲེལ་ཏེ། གོང་གི་རྩིས་ཁྲ་དེ་རོབ་རྩིས་ཡིན་པ་དང་། ལོ་གཅིག་རིང་སྒོ་ཕྱེས་ཏེ་འཇོག་རྒྱུ་ཡིན་སྟབས། དོགས་འདྲི་དང་ནོར་འཁྲུལ་རིགས་ཡོད་ཚེས་འབྲེལ་བ་གནང་རོགས་གསུངས་སོང་། རྩིས་ཁྲ་གསལ་བསྒྲགས་ཀྱི་ནང་དོན་དེ་ཙམ་ཁ་གསལ་མིན་པ་མ་ཟད། རྩིས་ནོར་རེ་ཟུང་མེད་དམ་སྙམ་པའི་དོགས་གཞི་འདུག་པ་དང་། བཀའ་བློན་མཆོག་གིས་དབྱིན་སྐད་དང་བོད་སྐད་གཉིས་ནང་གསལ་བསྒྲགས་གནང་བའི་བྱུང་སོང་གཅིག་པ་མིན་པ་བཅས་ལ་བརྟེན། འདི་གའི་གསར་འགོད་པའི་ལག་ཏུ་ཡིག་ཐོག་གི་རྩིས་ཁྲ་མ་རག་སྟབས།  ཞིབ་ཏུ་གསན་འདོད་ཡོད་རིགས་ཚོས་དྲ་ཐག་འདི་བརྒྱུད་ནས་གསུངས་བཤད་དངོས་ལ་གསན་རོགས།
