རྩོམ་པ་པོ་ཞིག་ལ་ཁྲིམས་ཆད་ལོ་ ༧
ལྷག་བཅད།

བོད་ཀྱི་རྩོམ་པ་པོ་བློ་མིག་ལགས་ལ་རྒྱལ་ཁབ་ཀྱི་གསང་བ་ཕྱིར་བསྒྲགས་པ་དང་ཁ་བྲལ་གྱི་བྱ་སྤྱོད་སྤེལ་བའི་ཉེས་མིང་འོག་ལོ་ ༧ དང་ཕྱེད་ཀྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག ཨོ་སི་ཊོ་ལི་ཡར་གནས་སྡོད་རྨེའུ་དཀོན་རྒྱམ་ལགས་ཀྱིས་འདི་གར་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། ཕྱི་ཟླ་ ༥ ཚེས་ ༩ ཉིན་སི་ཁྲོན་ཝིན་ཁྲན་མི་དམངས་ཁྲིམས་ཁང་ཟེར་བ་ནས་རྒྱལ་ཁབ་ཀྱི་གསང་བ་ཕྱིར་བསྒྲགས་པ་དང་མེས་རྒྱལ་ཁ་བྲལ་གྱི་བྱ་སྤྱོད་སྤེལ་བའི་ཉེས་མིང་འོག བོད་ཀྱི་རྩོམ་པ་པོ་དངོས་མིང་ལ་ཇོ་བློ་བཟང་འཇམ་དབྱངས་དང་སྨྱུག་མིང་བློ་མིག་ལགས་ལ་ལོ་ ༧ དང་ཕྱེད་ཀ་ཡི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག འདས་ལོའི་ཕྱི་ཟླ་ ༤ ཚེས་ ༡༧ ཉིན་གྱི་མཚན་མོའི་དུས་ཚོད་ ༡༡།༣༠ ཡས་མས་སུ། རྔ་བ་རྫོང་ཐོག་ནས་རྒྱ་གཞུང་ཉེན་རྟོག་པས་ཁཽང་གློ་བུར་དུ་འཛིན་བཟུང་བཙན་འཁྲིད་བྱས་ཏེ། ཝིན་ཁྲན་ལྟ་སྲུང་ཁང་ཟེར་བ་སོགས་ནས་ལོ་ངོ་ ༡ དང་ཟླ་གཅིག་ལྷག་ལ་གཅར་རྡུང་མནར་གཅོད་ཁྲོད་བཀག་ཉར་བྱས་རྗེས། ནང་མི་ཉེ་འབྲེལ་དང་ཁྲིམས་རྩོད་པ་སོགས་སུ་གཅིག་ཀྱང་ཁྲིམས་རར་ཡོང་དུ་བཅུག་མེད་པ་མ་ཟད། ནང་མིར་ག་དུས་ཁྲིམས་ཐག་གཅོད་མིན་གྱི་བརྡ་ལན་ཙམ་ཡང་སྤྲད་མེད་པར་གསང་བའི་ཐོག་ནས་ཁྲིམས་ཐག་བཅད་ཡོད་པ་རེད། དེ་རིང་གཞི་ནས་ཆུ་ཚོད་ཕྱེད་ཀ་རིང་ཁོང་གི་ཁྱིམ་མི་ཚོར་ཐུག་ཏུ་བཅུག་ཡོད་ཀྱང་། ཁྲིམས་ཁང་ནས་ཁྱིམ་མི་ཚོར་ཁོང་ལ་ཁྲིམས་ཐག་ག་དུས་བཅད་མིན་དང་ནག་ཉེས་ག་རེ་ཡོད་སོགས་གསལ་བཤད་གང་ཡང་བྱས་མེད་ཀྱང་། ཐུག་འཕྲད་སྐབས་བློ་མིག་ལགས་ཀྱིས་ནང་མི་ཚོར་ང་ལ་ཕྱི་ལོ་ ༢༠༠༩ ནས་བཟུང་སྟེ་ ༢༠༡༥ བར་རྒྱལ་ཁབ་ཀྱི་གསང་བ་ཕྱིར་བསྒྲགས་པ་དང་རྒྱལ་ཁབ་ཁ་བྲལ་གྱི་བྱ་སྤྱོད་སྤེལ་བའི་ནག་ཉེས་བཙན་འགེལ་བྱས་རུང་ཁས་ལེན་རྩ་བ་ནས་བྱས་མྱོང་མེད་ཅེས་བརྗོད་ཡོད་འདུག ད་ལྟ་རང་ལོ་ ༢༨ ཡིན་པའི་རྩོམ་པ་པོ་བློ་མིག་ལགས་རྔ་བ་རྫོང་རྨེའུ་རུ་མའི་རུ་ཆེན་གསུམ་པའི་ཇོ་ཚང་ནས་ཡིན་ཞིང་། ཕ་མིང་ལ་རྡོར་ཁོ་དང་མ་མིང་ལ་འཛམ་དཀར་ཟེར་གྱི་ཡོད་འདུག ཁོང་རང་ལོ་ ༦ གི་སྟེང་རྔ་བ་ཀིརྟི་དགོན་པར་གྲྭ་པ་བྱས་ཤིང་། ཀིརྟིའི་ནང་བསྟན་གཞོན་ནུའི་སློབ་གྲྭར་ཞུགས་ཏེ་སློབ་སྦྱོང་རྒྱུན་སྐྱོང་བྱེད་སྐབས། རྒྱ་གཞུང་གིས་གྲྭ་ཕྲུག་བདུན་བརྒྱ་ལྷག་ཡོད་པའི་ནང་བསྟན་གཞོན་ནུའི་སློབ་གྲྭ་བཙན་གྱིས་སྒོ་བརྒྱབ་ཡོད་སྟབས། དེ་ནས་ཁོང་གིས་རྔ་བ་ཀིརྟི་དགོན་གྱི་གཙོས་བྲག་མགོ་དགོན་དང་རེབ་གོང་། བླ་རུང་སྒར་སོགས་ནས་ནང་ཆོས་དང་རྩོམ་རིག་སོགས་ལ་སློབ་གཉེར་གནང་ཡོད་འདུག ཕྱི་ལོ་༢༠༡༠ ལོར་ཁོང་གིས་༼ན་བུན་སེར་པོའི་འཕྱུར་སྟངས།༽ཟེར་བའི་དེབ་ཅིག་བརྩམས་ནས་ཟི་ལིང་ནས་དཔར་བསྐྲུན་བྱས་ཡོད་པ་མ་ཟད། འདས་ལོར་ཁོང་གིས་༼ལྕགས་སྒྲོག་གིས་བཀྱིགས་པའི་གཏམ་རྒྱུད།༽ དང་༼རྔ་བའི་ཡི་གེ་བ་དག་ལ་མཚོན་གྱིས་བསྣུན་གྱིན་ཡོད།༽། ༼ང་མ་ཤི་བར་དུ་ལྟ་ཚུལ་བཤད་ཁོ་ཐག་ཡིན།༽ ཟེར་བ་སོགས་བྲིས་ཏེ། རྒྱ་གཞུང་གིས་བོད་ནང་གི་རྩོམ་འབྲི་རང་དབང་དང་སྨྲ་བརྗོད་རང་དབང་སོགས་ལ་བཙན་གནོན་ཤུགས་ཆེན་བྱེད་བཞིན་ཡོད་པ་འཛེམས་ཟོན་མེད་པའི་ཐོག་ནས་ཐེར་འདོན་གནང་ཡོད་པ་རེད། གཞན་ཡང་། དེ་སྔོན་ཁོང་གིས་བོད་ནང་གི་རྩོམ་པ་པོ་མང་པོ་དང་ལྷན་དུ་༼ཟི་ལིང་གཟའ་ཉི་མའི་ཁ་བརྡ་༽ ཟེར་བ་སོགས་ཀྱི་བགྲོ་གླེང་མང་པོ་བྱས་མྱོང་ཡོད་པ་མ་ཟད།  ཐོག་དང་པོར་རྔ་བ་རྫོང་ནས་༼རང་གྲོལ་བསྐྱར་ཞིབ་༽དང་༼གཟའ་ཉི་མའི་ཁ་ བརྡ།༽ཟེར་བའི་བགྲོ་གླེང་གསར་འཛུགས་བྱས་ཏེ་གཞོན་སྐྱེས་མང་པོར་ཤུགས་རྐྱེན་ཆེན་པོ་ཐེབས་ཡོད་པ་དང་། དེ་བཞིན་རྨེའུ་རུ་མ་ཡུལ་ཚོ་ནས་༼སྨྲ་བརྗོད་མེད་པའི་འུ་ཅག་༽ཟེར་བའི་དམངས་ཚོགས་དང་མཉམ་དུ་བགྲོ་གླེང་ཞིག་ཀྱང་གོ་སྒྲིག་གནང་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
