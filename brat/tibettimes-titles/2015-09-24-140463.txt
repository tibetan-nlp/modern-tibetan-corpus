ཞི་ཅིན་ཕིན་གྱི་སྐོར་སྐྱོད་ལ་ངོ་རྒོལ།

  རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱིས་ཨ་རིའི་ནང་རྒྱལ་དོན་སྐོར་སྐྱོད་བྱེད་བཞིན་ཡོད་པའི་སྐབས་དང་བསྟུན་ཏེ་བོད་པ་དང་ཡུ་གུར་བ། ཆོས་འཁོར་རྩ་རླུང་སྒོམ་སྒྲུབ་པ་སོགས་ནས་ངོ་རྒོལ་སྐད་འབོད་བྱས་ཡོད་འདུག   དེ་རིང་ཕྱི་ཚེས་ ༢༤ ཉིན་ནས་བཟུང་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱིས་ཨ་རིའི་སྲིད་འཛིན་ཨོ་བྷ་མ་མཆོག་གིས་གཙོས་གཞུང་འབྲེལ་མི་སྣ་ཁག་ཅིག་དང་ལྷན་དུ་རྒྱལ་ཁབ་གཉིས་བར་གྱི་འབྲེལ་ལམ་དང་དམག་དོན། ཆབ་སྲིད། དཔལ་འབྱོར། འགྲོ་བ་མིའི་ཐོབ་ཐང་སོགས་ཀྱི་དཀའ་རྙོག་ཐད་ལ་གྲོས་གླེང་བྱེད་བཞིན་ཡོད་འདུག   ཨ་རིའི་གཞུང་འབྲེལ་དང་ཁེ་ལས་སྡེ་ཁག་གི་དཔོན་རིགས་ཚོས་བརྗོད་པ་ལྟར་ན། ཨ་རིའི་གཞུང་གིས་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱི་རྒྱལ་དོན་སྐོར་སྐྱོད་དང་བསྟུན་ཏེ། ཉེ་ལམ་ཁོ་ཚོར་རྒྱ་ནག་ནང་ཕྲད་པའི་དྲ་ཐོག་གསང་ཉུལ་གྱི་གནད་དོན་དང་། ལྷག་པར་དུ་གཞུང་ཕྱོགས་ཀྱི་ཚོང་དོན་དྲ་ལམ་ཁག་ཏུ་ལྐོག་འཛུལ་དང་གསང་བ་རྐུ་འཁྱེར་བྱས་པའི་གནད་དོན་དང་འབྲེལ་བའི་དཀའ་རྙོག་ཡོད་ཚད་ལ་ཞིབ་ཕྲ་གླེང་མོལ་དགོས་པའི་ཛ་དྲག་གི་འབོད་སྐུལ་ཞུས་ཡོད་ཀྱང་། ། ཞི་ཅིན་ཕིན་གྱི་རྒྱལ་དོན་སྐོར་སྐྱོད་དང་བསྟུན་རྒྱ་ནག་གིས་སླར་ཡང་བཟང་ཕྱོགས་ཀྱི་རྣམ་འགྱུར་བསྟན་ནས་རྒྱལ་ཁབ་གཉིས་ཀྱི་ཉོ་འཚོང་སྡེ་ཁག་བར་དཔལ་འབྱོར་མཉམ་ལས་ཀྱི་ཁེ་སྤོགས་གཉེར་བའི་གོ་སྐབས་འཚོལ་བཞིན་ཡོད་པ་རེད།   ཡང་ཨ་རིའི་བདེ་སྲུང་ལྷན་ཁང་གི་སློབ་སྟོན་པ་གཞོན་པ་བྷན་རོ་ཌི་ (Ben Rhodes) ལགས་ཀྱིས། ཨ་རིའི་སྲིད་འཛིན་ཨོ་བྷ་མ་མཆོག་གིས་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་དང་ལྷན་དུ་སྒེར་གྱི་དགོང་ཟས་གསོལ་སྟོན་ཞིག་གི་ཐོག རྒྱལ་ཁབ་གཉིས་བར་གྱི་དམག་དོན་དང་བྱང་ཀོ་རི་ཡའི་རྡུལ་ཕྲན་མཚོན་ཆ། སའི་གོ་ལའི་ཚ་དྲོད་དང་དཔལ་འབྱོར་འཛུགས་སྐྲུན། འགྲོ་བ་མིའི་ཐོབ་ཐང་གི་གནད་དོན་སོགས་ཀྱི་དཀའ་རྙོག་ཐད་ལ་གྲོས་མོལ་བྱ་རྒྱུ་ཡིན་པ་དང་། སང་ཉིན་སྲིད་འཛིན་རྣམ་གཉིས་ཀྱི་གཙོས། ཕྱོགས་གཉི་གའི་གཞུང་འབྲེལ་སྐུ་ཚབ་རྣམས་རྒྱལ་ཁབ་ཀྱི་གསོན་སྟོན་ཐོག་མཉམ་ཞུགས་བྱ་རྒྱུ་ཡིན་པ་མ་ཟད། གསར་འགོད་གསལ་བསྒྲགས་ཤིག་ཀྱང་སྤེལ་འཆར་ཡོད་སྐོར་བརྗོད་འདུག   ཕྱི་ཚེས་ ༢༢ ཉིན་ཞི་ཅིན་ཕིན་ཨ་རིའི་རྒྱལ་ས་ཝ་ཤིན་ཊོན་མངའ་སྡེའི་གྲོང་ཁྱེར་སེ་ཊལ་ (Seattle) ནང་ཉིན་གསུམ་རིང་ལ་བཟོ་ལས་རིག་རྩལ་དང་ཉོ་འཚོང་ལས་རིགས་ཁག་གི་མགོ་ཁྲིད་ཁག་ཅིག་དང་ལྷན་དུ། འབྱུང་འགྱུར་རྒྱལ་ཁབ་གཉིས་ཀྱི་ཚོང་འབྲེལ་རྒྱ་བསྐྱེད་གཏོང་ཕྱོགས་ལ་གྲོས་མོལ་བྱས་ཡོད་པ་དང་། སྐབས་དེར་ཞི་ཅིན་ཕིན་གྱི་སྡོད་གནས་ནུབ་གླིང་མགྲོན་ཁང་ (Western Hotel) གི་གཞུང་ལམ་འགྲམ་དུ། བོད་རིགས་སྤྱི་མཐུན་ཚོགས་པ་དང་ས་གནས་བོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས། དེ་བཞིན་ཡུ་གུར་དང་ཐའེ་ཝན། ཆོས་འཁོར་རྩ་རླུང་སྒོམ་སྒྲུབ་ཚོགས་པ་བཅས་མི་མང་སྟོང་ཕྲག་མང་པོས་སོ་སོའི་དགོས་འདུན་གྱི་འཕྲེད་དར་འཕྱར་ནས་ཞི་བའི་སྐད་འབོད་བྱས་ཡོད་པ་དང་། སྐབས་དེར་བོད་དོན་ལས་འགུལ་བ་ཚོས། རྒྱ་ནག་གི་དྲག་གནོན་སྲིད་ཇུས་མཚམས་འཇོག་དང་ཆབ་སྲིད་བཙོན་མ་རྣམས་ཀློད་གྲོལ་གཏོང་དགོས། སྲིད་འཛིན་ཨོ་བྷ་མ་མཆོག་གིས་ཞི་ཅིན་ཕིན་དང་ལྷན་དུ་བོད་དོན་དང་བོད་ནང་གི་འགྲོ་བ་མིའི་ཐོབ་ཐང་གི་གནས་སྟངས་སྔར་བཞིན་ཇེ་སྡུག་ཏུ་འགྲོ་བཞིན་ཡོད་པ་ལས། ཡར་རྒྱས་གུ་ཡངས་སུ་འགྲོ་གི་མེད་པའི་དངོས་ཡོད་གནས་ཚུལ་གླེང་སློང་བྱེད་དགོས། ཞེས་འབོད་སྐུལ་ཞུས་ཡོད་འདུག   གསར་འགོད་པ། པདྨ་རྒྱལ།
