ལྐོག་ཟ་རུལ་སུངས་ལ་ཁྲིམས་བཤེར།

རྒྱ་ནག་སི་ཁྲོན་ཞིང་ཆེན་གྱི་སྤྱི་ཁྱབ་གཞོན་པ་ཟུར་པ་ཞིག་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་དོགས་གཞིའི་འོག་འཛིན་བཟུང་བཀག་ཉར་བྱས་ཡོད་འདུག ཟླ་འདིའི་ཚེས་ ༩ ཉིན་རྒྱ་ནག་ཞིབ་དཔྱོད་ལས་ཁུངས་ཀྱིས་སི་ཁྲོན་ཞིང་ཆེན་གྱི་སྤྱི་ཁྱབ་གཞོན་པ་ཟུར་པ་ཀོ་ཡཱོང་ཞང་ (Guo Yongxiang ) གིས། དབང་ཆ་བེད་སྤྱོད་ལོག་པར་བཏང་བ་དང་། རང་ཉིད་ཀྱི་བུ་དང་ངན་འབྲེལ་གྱིས་ལྐོག་རྔན་བླངས་ཡོད་པ་ཤེས་རྟོགས་བྱུང་སྟེ། དམར་ཤོག་ཚོགས་མི་དང་ལས་གནས་ནས་གནས་དབྱུང་བཏང་ཡོད་པ་མ་ཟད། ད་ཆར་འཇུ་བཟུང་བཀག་ཉར་བྱས་ཡོད་འདུག རྒྱ་ནག་ཞིབ་དཔྱོད་ལས་ཁུངས་ཀྱིས། “ ཀོ་ཡཱོང་ཞང་གིས་དབང་ཆ་བེད་སྤྱོད་ལོག་པར་བཏང་བ་དང་རང་ཉིད་ཀྱི་བུ་དང་ངན་འབྲེལ་གྱིས་རྒྱུ་ནོར་ལྐོག་ཟ་བྱས་པ། ཀྲུའུ་ཡཱོང་ཁང་ལ་རོགས་བྱས་ཏེ་དམར་ཤོག་ཚོགས་པའི་ཁྲིམས་སྒྲིག་དང་འགལ་བའི་བྱ་བ་ལས་ཡོད་པ་དང་། སྲིད་དབང་མགོ་རྟིང་སློག་ཐབས་བཅས་ནག་ཉེས་ཁག་བསགས་ཡོད་སྟབས་འཇུ་བཟུང་བྱས་པ་ཡིན། ” ཞེས་བརྗོད་འདུག རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱིས་ལྐོག་ཟ་རུལ་སུངས་ལ་རྡུང་རྡེག་ཚ་ནན་གཏོང་བའི་སྲིད་ཇུས་ལག་བསྟར་བྱས་ཏེ། ༢༠༡༣ ཟླ་ ༡༢ ཚེས་ ༢ ཉིན་རྒྱ་ནག་དབུས་ཆབ་སྲིད་སྡེ་ཚན་གྱི་རྒྱུན་ལས་ཟུར་པ་ཀྲུའུ་ཡཱོང་ཁང་འཇུ་བཟུང་བྱས་ཡོད་པ་མ་ཟད། ཀྲུའུ་དང་མཉམ་འབྲེལ་བྱེད་མཁན་གྱི་མི་སྣ་མང་པོ་ཞིག་ལ་རྒྱ་ནག་ཁྲིམས་སྒྲིག་ཞིབ་བཤེར་ལྷན་ཁང་གིས་ཞིབ་བཤེར་བརྒྱུད་འཇུ་བཟུང་བྱས་ཡོད། དེ་དག་ཁྲོད་ཧུའུ་ནན་ཞིང་ཆེན་གྱི་སྤྱི་ཁྱབ་གཞོན་པ་ཅེ་ཝུན་ལིན་ (Ji Wenlin) དང་། རྒྱ་ནག་རྡོ་སྣུམ་བཙོ་སྦྱང་བཟོ་གྲྭའི་འགན་འཛིན་ལེ་ཧྰ་ལིན་ (Li Hualin) བཅས་གཙོས་མི་སྣ་ཁག་ཅིག་འཛིན་བཟུང་བྱས་ཡོད་འདུག  ཟླ་འདིའི་ཚེས་ ༧ ཉིན་ཀྲུའུ་ཡི་ཉེ་འབྲེལ་གཙོས་པའི་འབྲེལ་ཡོད་མི་སྣ་ ༣༠༠ ཙམ་ལ་འདྲི་གཅོད་བྱས་ཏེ། ཨ་སྒོར་ཐེར་འབུམ་ ༡༤།༥ ཡི་རྒྱུ་ནོར་དང་དངུལ་གཞུང་བཞེས་བཏང་ཡོད་འདུག་ལ། ཀྲུའུ་ཡཱོང་ཁང་གི་བུའི་ཚོང་རོགས་སི་ཁྲོན་གྱི་ཕྱུག་བདག་ལུའུ་ཧེ་ཕིན་དང། རྒྱ་ནག་བཅིངས་གྲོལ་དམག་གི་སྤྱི་ཁྱབ་དམག་འཁྲུག་ཐབས་ཇུས་ལྷན་ཁང་གི་དམག་སྤྱི་གཞོན་པ་ཟུར་པ་ཀོའུ་ཅུན་ཧྲེ་བཅས་འཛིན་བཟུང་བྱས་ཡོད་འདུག ཆབ་སྲིད་དབྱེ་ཞིབ་པ་ཚོས། རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱིས་ལྐོག་ཟ་རུལ་སུངས་རྩ་མེད་གཏོང་བའི་སྲིད་ཇུས་ལག་བསྟར་བྱས་ཏེ་གནད་ཡོད་མི་སྣ་མང་པོ་འཛིན་བཟུང་བྱས་པ་ལ་བརྟེན་ནས། རྒྱ་ནག་ནང་ལྐོག་ཟ་རུལ་སུངས་བྱེད་མཁན་ཇེ་ཉུང་དུ་འགྲོ་སྲིད་ཀྱང་། ཞི་ཅིན་ཕིན་དང་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་དོགས་གཞི་ཡོད་པའི་མགོ་ཁྲིད་རྒན་པ་ཝུན་ཇ་པའོ་དང་། སྲིད་འཛིན་ཟུར་པ་ཅང་ཙེ་མིན་ཤོག་ཁག་བར་འགལ་ཟླ་ཆགས་སྲིད་སྐོར་བརྗོད་འདུག ཐེངས་འདིར་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་ཉེས་མིང་འོག་འཛིན་བཟུང་བྱས་ཡོད་པའི་ཀོ་ཡྰོང་ཞང་ནི། ཀྲུའུ་ཡཱོང་ཁང་སི་ཁྲོན་ཞིང་ཆེན་དམར་ཤོག་ཚོགས་པའི་དྲུང་ཆེའི་ལས་འགན་འཁུར་སྐབས། སི་ཁྲོན་ཞིང་ཆེན་གྱི་སྤྱི་ཁྱབ་གཞོན་པ་ཡིན་འདུག་ལ། ༢༠༠༢ ལོར་ཀོ་ཡྰོང་ཞང་གིས་སི་ཁྲོན་དམར་ཤོག་ཚོགས་པའི་དབུས་རྒྱུན་ལས་དང་རྗེས་སུ་དྲུང་ཆེའི་འགན་འཁུར་ཡོད་འདུག ༢༠༠༦ ལོར་སི་ཁྲོན་ཞིང་ཆེན་གྱི་སྤྱི་ཁྱབ་པ་གཞོན་པ་དང་། ༢༠༠༨ ལོར་སི་ཁྲོན་མི་དམངས་འཐུས་ཚོགས་ཀྱི་ཚོགས་གཞོན་གྱི་ལས་འགན་འཁུར་ཡོད་འདུག  ༢༠༠༩ ལོ་ནས་ ༢༠༡༣ ལོའི་ཟླ་ ༦ ཚེས་ ༢༣ བར་སི་ཁྲོན་ཞིང་ཆེན་ཤེས་ཡོན་དང་སྒྱུ་རྩལ་ལྷན་ཚོགས་ཀྱི་ཚོགས་གཙོ་བཅས་འཁུར་ཡོད་འདུག་ལ། མི་རིང་བར་རྒྱ་ནག་དབུས་ཁྲིམས་ཁང་གིས་ཁོང་ལ་ཁྲིམས་ཆད་གཏོང་རྒྱུ་ཡིན་ཚུལ་བརྗོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
