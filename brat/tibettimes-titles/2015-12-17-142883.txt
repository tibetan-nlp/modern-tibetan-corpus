ས་གནས་གཞོན་ནུ་སྒོ་རྒྱག་རྒྱུ་ཡིན།

  ནེའུ་ཡོག་དང་ནེའུ་ཇིར་ས་གནས་གཞོན་ནུས་བོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་ཀྱི་བཞེངས་ཕྱོགས་ལ་འགྱུར་བ་མ་བཏང་བར་དབུས་ལ་འབྲེལ་བ་ཞུ་རྒྱུ་མིན་གསུངས་པ་མ་ཟད། ལོ་རྗེས་མའི་གཞོན་ནུའི་ཚོགས་ཆེན་སྟེང་གི་ཐག་གཅོད་ལའང་བརྩི་བཀུར་བྱེད་འཆར་མི་འདུག་སྟབས་ཐག་གཅོད་འདི་བྱས་པ་ཡིན་ཟེར།   ཁ་སང་དབུས་བོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་སྤྱི་ཁྱབ་རྒྱུན་ལས་ཁང་གིས་གསལ་བསྒྲགས་ཤིག་སྤེལ་ཏེ། ནེའུ་ཡོག་དང་ནེའུ་ཇར་སི་ས་གནས་གཞོན་ནུ་གནས་སྐབས་རིང་སྒོ་རྒྱག་རྒྱུ་ཡིན་སྐོར་བསྒྲགས་འདུག གཞོན་ནུའི་གསལ་བསྒྲགས་ནང་བཀོད་དོན། “དེ་ཡང་ད་ལྟའི་སྐབས་ ༡༣ པའི་ནིའུ་ཡོག་ས་གནས་བོད་ཀྱི་གཞོན་ནུའི་རྒྱུན་ལས་ཁང་དེ་བཞིན་བློ་ཕམ་ཆེན་པོའི་ངང་གནས་སྐབས་སྒོ་རྒྱག་རྒྱུ་དང་། ས་གནས་སུ་བར་བརྒལ་འབྲེལ་མཐུད་པ་བཞི་ (interim committee) ས་རྒྱུན་ཟུར་པའི་མང་མོས་འོས་བསྡུ་བརྒྱུད་ནས་བསྐོ་བཞག་བྱས་ཡོད་པར། སྐབས་ ༡༣ པའི་ནིའུ་ཡོག་དང་ནིའུ་ཇར་སིའི་ས་གཞོན་རྒྱུན་ལས་ནས་སྤྱི་ལོ་ ༢༠༡༦ ཟླ་བ་ ༡ ཚེས་ ༢༠ གོང་དུ་ནིའུ་ཡོག་དང་ནིའུ་ཇར་སིའི་ས་གནས་གཞོན་ལྷན་གྱི་རྒྱུ་དབང་ཡོངས་རྫོགས་ཆད་ལུས་མེད་པར་བར་བརྒལ་འབྲེལ་མཐུད་པར་རྩིས་སྤྲོད་གཙང་མ་དགོས་པ་དང་། དེ་རིང་སྤྱི་ལོ་ ༢༠༡༥ ཟླ་བ་ ༡༢ ཚེས་ ༡༦ ནས་བཟུང་མ་འོངས་པར་བོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་ཀྱི་ངོས་ལེན་ཡོད་པའི་ས་གཞོན་རྒྱུན་ལས་གསར་འདེམས་མ་བྱུང་བར། ནེའུ་ཡོག་དང་ནེའུ་ཇར་སིའི་ནང་བོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་ཀྱི་ངོ་ཚབ་ནུས་པའི་ས་གནས་གཞོན་ལྷན་ཞིག་མེད་པས། ཚོགས་སྒེར་སུ་རུང་ནས་བོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་ཀྱི་མིང་དང་། ཚོགས་དར། ལས་རྟགས། ལས་དམ། ལས་ཤོག་སོགས་བེད་སྤྱོད་བྱ་མི་ཆོག་པ་ནན་བརྗོད་བྱ་རྒྱུ་དང་། གལ་སྲིད་དེ་ལྟར་བྱེད་མཁན་བྱུང་ཚེ་དེ་ལས་བྱུང་བའི་གནད་དོན་གྱི་འགན་ཁུར་ཡོངས་རྫོགས་འཁུར་དགོས།” ཞེས་སོགས་བཀོད་འདུག   འདས་པའི་ལོ་འགའི་གོང་ནས་བཟུང་བོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་ནང་དེ་སྔའི་གཞོན་ནུའི་རང་བཙན་ལངས་ཕྱོགས་ལ་འགྱུར་གཏོང་དགོས་པའི་དགོངས་ཚུལ་བྱུང་བ་ལྟར། སྔར་ལོ་རྒྱ་གར་ལྷོ་ཕྱོགས་ཀྱི་གཞོན་ནུའི་ས་གནས་ཁག་བརྒྱད་ཀྱིས་ལངས་ཕྱོགས་དབུ་མའི་ལམ་ལ་འགྱུར་བ་བཏང་ཡོད་པ་དང་། ད་ལོའི་ལོ་མགོ་ནས་བཟུང་ཨ་རིའི་ནེའུ་ཡོག་དང་ནེའུ་ཇར་སི་ས་གནས་གཞོན་ནུས་ཀྱང་། བོད་མིའི་སྒྲིག་འཛུགས་དང་༧གོང་ས་མཆོག་གི་བཞེངས་ཕྱོགས་ཐོག་རྡོག་བསྒྲིལ་ན་རྩ་དོན་ལ་ནུས་པའི་འཐོན་པའི་ངོས་འཛིན་བྱུང་སྟབས། དབུས་བོད་ཀྱི་གཞོན་ནུའི་རྒྱུན་ལས་ཚོ་དང་དགོངས་ཚུལ་མི་མཐུན་པ་བྱུང་ཡོད།
