རྩོམ་སྒྲིག་པ་སོཊ་འཛིན་བཟུང་།

ཁ་སང་རྒྱ་ནག་གི་ཚོང་ལས་ཚགས་པར་གྲགས་ཅན་ཞིག་གི་རྩོམ་སྒྲིག་པ་སོགས་མི་ ༨ གནས་ཚུལ་རྫུན་མ་བསྒྲགས་པའི་ཉེས་མིང་འོག་འཛིན་བཟུང་བྱས་ཡོད་འདུག ཁ་སང་རྒྱ་ནག་ཀོང་ཀྲུའུ་རུ་ཡོད་པའི་ཚོང་ལས་ཚགས་པར་གྲགས་ཅན་༼དུས་རབས་ཉེར་གཅིག་པའི་དཔལ་འབྱོར་གནས་འཕྲིན།༽ཞེས་པའི་རྩོམ་སྒྲིག་པ་གཙོ་བོ་ལུའུ་ཏུང་དང་། རྩོམ་སྒྲིག་པ་གཞོན་པ་ཀྲུའུ་པེན་སོགས་མི་ ༨ ཧྲང་ཧེ་ཉེན་རྟོག་པས་འཛིན་བཟུང་བྱས་ཡོད་པ་དང་། འཛིན་བཟུང་བྱས་པའི་རྒྱུ་རྐྱེན་ཐད་རྒྱ་ནག་ཞིན་ཧྭ་གསར་ཁང་གིས། ཁོ་ཚོས་ཧྲང་ཧེ་རིན་ཡཱན་དང་ཧྲིན་ཀྲིན་ཞིན་ཅི་ལིན་ཚོང་གཉེར་ཁང་གཉིས་ལ་ངན་འབྲེལ་བྱས་ཏེ། དྲིལ་བསྒྲགས་གླ་ཆ་མཐོ་བོ་བླངས་ཏེ་གནས་ཚུལ་རྫུན་མ་སྤེལ་བའི་དོགས་གཞི་ཡོད་པ་མ་ཟད། ད་རེས་ཀྱི་ཉེས་དོན་དེ་པེ་ཅིན་དང་ཀང་ཏུང་། ཧྲང་ཧེ་སོགས་ཀྱི་ཁེ་ལས་ཁང་བཅུ་གྲངས་ཤིག་དང་ངན་འབྲེལ་ཡོད་པའི་དོགས་གཞི་ཡོད་སྟབས། ད་ལྟ་ཞིབ་བཤེར་བྱེད་བཞིན་པའི་སྒང་ཡིན་ལུགས་བརྗོད་འདུག དུས་རབས་ཉེར་གཅིག་པའི་དཔལ་འབྱོར་གནས་འཕྲིན་ཟེར་བའི་ཚགས་པར་དེ། ཀོང་ཀྲུའུ་ལྷོ་ཕྱོགས་གསར་འགྱུར་ལས་རིགས་ཆེད་གཉེར་ཚོགས་པས་ཕྱི་ལོ་ ༢༠༠༡ ལོར་ཐོག་མར་པར་སྐྲུན་བྱེད་མགོ་བརྩམས་ཤིང་། ཚོགས་པ་དེས་རྒྱ་གཞུང་གི་བརྒྱུད་ལམ་ཞིབ་བཤེར་དང་བཀག་སྡོམ་གྱི་སྲིད་བྱུས་འོག་གསར་འགྱུར་མཁོ་འདོན་གྱི་ལས་ཀར་བརྩོན་ལེན་བྱེད་བཞིན་ཡོད་སྟབས། སྤྱི་ཚོགས་ཐོག་ནས་མཐོང་ཆེན་དང་གདེང་འཇོག་ལེགས་པོ་ཐོབ་ཡོད་འདུག ཚགས་པར་དེ་ཉིན་ ༣ ་ནང་ཐེངས་གཅིག་འདོན་གྱི་ཡོད་པ་དང་། འདོན་ཐེངས་རེར་ཚགས་པར་གྲངས་ཁྲི་ ༧༥ ལྷག་འདོན་བཞིན་ཡོད་འདུག ཚོགས་པ་དེས་ཚགས་པར་དེ་གཅིག་པུ་མ་ཡིན་པར་༼དངུལ་དོན་དོ་དམ་བདུན་རེའི་ཚགས་པར༽དང་༼དུས་རབས་ཉེར་གཅིག་པའི་ཚོང་ལས་དཔྱད་གླེང་།༽། ༼ཚོང་དོན་འགྲུལ་བཞུད།༽། དེ་བཞིན་༼དུས་རབས་ཉེར་གཅིག་པའི་དྲ་བ༽སོགས་ཁེ་ལས་དང་འབྲེལ་བའི་ཚགས་པར་དང་དྲ་བ་མང་པོ་གཉེར་བཞིན་ཡོད་འདུག འདི་ལོའི་ཕྱི་ཟླ་ ༣ པའི་ནང་རྒྱ་ནག་དབུས་དྲིལ་བསྒྲགས་ལྷན་ཁང་གིས་བརྒྱུད་ལམ་དུ་བསླུ་བྲིད་རང་བཞིན་གྱི་གནས་ཚུལ་དང་གསར་འགྱུར་རྫུན་སྒྲོག་ལ་རྒོལ་རྡུང་གཏོང་དགོས་པའི་བཀའ་རྒྱ་བཏང་སྟེ། རྒྱ་ནག་གི་སྒེར་གཉེར་གསར་ཁང་ཁག་གི་གཙོས་གསར་ཁང་མང་པོའི་རྩོམ་སྒྲིག་པ་དང་གསར་འགོད་པ་མང་པོ་གནས་དབྱུང་བཏང་བ་དང་། འཛིན་བཟུང་བྱས་ཡོད་པ་མ་ཟད། རྩོམ་སྒྲིག་པས་རང་ཤི་བརྒྱབ་པ་སོགས་ཀྱི་གནས་ཚུལ་རིམ་པར་ཐོན་ཡོད་པ་ལྟར། ཕྱི་ལོའི་ཟླ་ ༧ ནང་རྒྱ་ནག་དབུས་བརྙན་འཕྲིན་ཁང་གི་མགོ་འཛིན་པ་གྲགས་ཅན་རེ་ཁྲིན་ཀང་དང་། ནོར་སྲིད་གསར་འགྱུར་སྡེ་ཚན་གྱི་འགན་འཛིན་གཞོན་པ་ལི་ཡཱོང་གཉིས་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་དོགས་གཞིའི་འོག་འཛིན་བཟུང་བྱས་ཡོད་པ་དང་། དེ་བཞིན་ཟླ་སྔོན་མའི་ནང་དབུས་བརྙན་འཕྲིན་ཁང་ཁྱབ་བསྒྲགས་སྡེ་ཚན་གྱི་འགན་འཛིན་ཀོ་ཀྲིན་ཞི་དང་། བརྙན་འཕྲིན་སྒྲིག་སྦྱོར་བ་ཐཱན་ལི་ཝུ་གཉིས་ཀྱང་ལྐོག་ངན་བླངས་བའི་དོགས་གཞིའི་འོག་འཛིན་བཟུང་བྱས་ཏེ་ཞིབ་བཤེར་བྱེད་བཞིན་ཡོད་འདུག ཉེ་སྔོན་རྒྱལ་མཚམས་མེད་པའི་གསར་འགོད་མཐུན་ཚོགས་ཀྱིས་འདི་ལོའི་གསར་འགྱུར་རང་དབང་གི་ཞིབ་འཇུག་ཐོ་གཞུང་ཞིག་ཕྱིར་བསྒྲགས་བྱས་པའི་ནང་། འཛམ་གླིང་གི་རྒྱལ་ཁབ་ ༡༨༠ ནང་ནས་རྒྱ་ནག་ནི་ཨང་རིམ་ ༡༧༥ པ་སྟེ་གནས་བབ་སྡུག་ཤོས་ཀྱི་གྲས་སུ་ཚུད་ཡོད་པ་དང་། མཐུན་ཚོགས་དེའི་བསྡོམས་རྩིས་ལྟར་ན། གཏེར་ཁ་སྔོག་འདོན་བྱེད་མཁན་དང་ཚོང་གཉེར་ཁང་སོགས་ལ་ངན་འབྲེལ་བྱས་ཏེ། ལྐོག་རྔན་བླངས་བ་དང་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་བྱ་སྤྱོད་སྤེལ་བ་སོགས་ཀྱི་ཉེས་མིང་འོག རྒྱ་ནག་གི་བཙོན་ཁང་ཁག་ནང་ཁྱོན་གསར་འགོད་པ་ ༢༨ ཙམ་བཙོན་འཇུག་བྱས་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
