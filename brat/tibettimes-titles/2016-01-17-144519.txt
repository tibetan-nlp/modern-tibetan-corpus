འདས་ལོར་བོད་ནས་རྒྱ་གར་དུ་ཡོང་མཁན་ ༨༧
བྱུང་འདུག

  ཕྱི་ལོ་ ༢༠༡༥ ལོར་བོད་ནས་སྣེ་ལེན་ཁང་དུ་ཁྱོན་མི་གྲངས་ ༨༧ ཙམ་འབྱོར་ཡོད་ཅིང་། མང་ཆེ་བ་མདོ་སྟོད་སྨད་གཉིས་ནས་ཡིན་འདུག དབུས་སྣེ་ལེན་ཁང་ནས་ལམ་གྲོན་དང་གྱོན་གོས་བླངས་ཏེ་ཕྱིར་བོད་དུ་ལོག་རྒྱུ་ཡིན་ཟེར་མཁན་ ༡༣ བྱུང་ཡོད་འདུག   འདི་གའི་གསར་འགོད་པས་དྷ་རམ་ས་ལའི་དབུས་སྣེ་ལེན་ཁང་ལ་རྒྱུས་ལོན་བྱས་པ་ལྟར་ན། ༢༠༡༥ ལོར་བོད་ནས་བལ་ཡུལ་སྣེ་ལེན་འབྲེལ་མཐུད་ཁང་བརྒྱུད་རྒྱ་གར་ནང་སྐྱབས་བཅོལ་དུ་ཡོང་མཁན་ ༨༧ ཙམ་འབྱོར་ཡོད་པ་དང་། དེའི་ནང་ནས་དགོན་སྡེ་ཁག་ཏུ་འགྲོ་མཁན་གྲྭ་བཙུན་ ༣༨ དང་། སློབ་གྲྭ་ཁག་ཏུ་འགྲོ་མཁན་ ༢༨། དྷ་ས་སོགས་སུ་རང་དགར་སྡོད་མཁན་ ༦ ཙམ་བྱུང་ཡོད་འདུག   སྣེ་ལེན་ཁང་གི་མཐུན་འགྱུར་འོག་བོད་ནས་ཡོང་མཁན་ཚང་མར་སྨན་ཁང་ནས་ཁྲག་དང་གཅིན་པ་སོགས་གཟུགས་པོ་ཡོངས་སུ་བརྟག་དཔྱད་གནང་བཞིན་ཡོད་པ་ལྟར། ཕྱི་ལོ་ ༢༠༡༤ ལོར་མཆིན་པའི་གཉན་ཚད་ཅན་གྱི་ནད་པ་ ༡༥ བྱུང་ཡོད་ཀྱང་ ༢༠༡༥  ལོར་ན་ཚ་ཚབས་ཆེན་དེ་རིགས་ཡོད་མཁན་གཅིག་ཀྱང་བྱུང་མི་འདུག   དབུས་སྣེ་ལེན་ཁང་གི་འབྱོར་ཐོའི་རོབ་རྩིས་ལྟར་ན། ༢༠༡༠ ལོར་བོད་ནས་སེར་སྐྱ་ཁྱོན་ ༧༧༠ བསླེབས་ཡོད་པ་དང་། ༢༠༡༡ ལོར་ ༧༥༣། ༢༠༡༢ ལོར་ ༧༥༦། ༢༠༡༣ ལོར་ ༡༥༠། ༢༠༡༤ ལོར་ ༡༢༠ སྐྱབས་བཅོལ་དུ་འབྱོར་ཡོད་འདུག   ༢༠༠༣ ལོའི་ཚུན་དུ་བོད་ནས་ཡོང་མཁན་མང་ཤོས་དེ་ ༢༠༠༥ ལོ་ཆགས་ཡོད་པ་དང་། ལོ་དེར་བོད་མི་ ༣༡༢༩ ཙམ་བལ་ཡུལ་སྣེ་ལེན་འབྲེལ་མཐུད་ཁང་བརྒྱུད་རྒྱ་གར་ནང་འབྱོར་ཡོད་འདུག    གསར་འགོད་པ། པདྨ་མཚོ།
