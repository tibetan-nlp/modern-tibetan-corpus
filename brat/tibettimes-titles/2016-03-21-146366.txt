ཆབ་སྲིད་བཙོན་པ་ཞིག་འདས་གྲོངས།

བོད་ཀྱི་ཆབ་སྲིད་བཙོན་པ་རྒྱ་ཡེ་ཕུན་ཚོགས་ལགས་ལ་རྒྱ་ནག་གིས་ཡུན་རིང་བཙོན་འཇུག་མནར་གཅོད་ཚད་མེད་བཏང་རྐྱེན་མདང་ནུབ་རང་ཤག་ནས་སྐུ་གྲོངས་ཡོད་འདུག   ཁོང་དང་གྲོང་ཚོ་གཅིག་པ་དྷ་སར་གནས་སྡོད་ཡེ་ཤེས་བསྟན་འཛིན་ལགས་ཀྱིས་འདི་གའི་གསར་ཁང་དུ་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། ཁོང་སྔོན་ཆད་རྒྱ་གར་དུ་ཡོང་ནས་༧གོང་ས་མཆོག་ལ་མཇལ་ཁ་ཞུས་ཏེ་ཁོང་གི་སྐུ་མདུན་ནས་དགེ་སློང་གི་སྡོམ་པ་བླངས་རྗེས། ཕྱིར་བོད་ལ་ལོགས་ནས་མཚོ་སྔོན་པོ་མཚོ་སྙིང་དུ་ལྷ་ཁང་ཞིག་བཞེངས་གྲུབ་ནས་ལོ་གཅིག་གི་རྗེས་སུ། དེ་སྔོན་༧གོང་ས་མཆོག་གིས་ཁོང་ལ་གནང་བའི་བཀའ་ཡིག་དང་ཁོས་རྒྱ་གར་ནས་ཁྱེར་ཡོང་བའི་དཔེ་ཆ་འགའ་བཅས་ས་གནས་ཉེན་རྟོག་པའི་ལག་ཏུ་ཤོར་བའི་རྐྱེན་གྱིས། ཁོང་འཛིན་བཟུང་བཙོན་འཇུག་དང་གཅར་རྡུང་བྱས་ནས་པུས་མོ་སོགས་ནད་པ་བཟོས་ཏེ་ཡར་ལངས་མི་ཐུབ་པར་བཏང་ཡོད་སྟབས། ཧ་ལམ་ལོ་ངོ་བཅུ་ཕྲག་ཅིག་རིང་ཁོང་ནད་པ་ཆགས་ཡོད་པ་རེད། བདུན་ཕྲག་གཉིས་ཀྱི་སྔོན་ལ་ཁོང་མཚོ་སྔོན་ཆབ་ཆ་བོད་སྨན་ཁང་དུ་བརྟག་དཔྱད་བྱེད་དུ་ཕྱིན་ཡོད་ནའང་། སྨན་ཁང་གིས་ཁོང་ལ་དྲག་སྐྱེད་ཡོང་ཐབས་མེད་ལུགས་བཤད་ནས་ཕྱིར་རང་ཁྱིམ་དུ་ལོགས་དགོས་བྱུང་ཡོད་པ་དང་། མདང་དགོང་བོད་ནང་ཆུ་ཚོད་ ༡ པའི་སྟེང་སྐུ་གྲོངས་ཡོད་པ་དང་ད་ལྟ་དགུང་ལོ་ ༨༥ ཡིན་པ་རེད། ཁོང་ནི་བོད་ལ་ཤ་ཞེན་ཆེ་མཁན་ཞིག་ཡིན་སྟབས་ཁོང་གི་མི་ཚེ་ཧྲིལ་བོ་བོད་བསྟན་པ་ཆབ་སྲིད་ཀྱི་དོན་དུ་རྫོགས་ཡོད་པ་མ་ཟད། ཁོང་ཉིད་བོད་བསྟན་པ་ཆབ་སྲིད་ཀྱི་དོན་དུ་ལུས་སྲོག་བློས་གཏོང་མཁན་གནད་ཡོད་མི་སྣ་ཞིག་དང་བོད་བསྟན་པ་ཆབ་སྲིད་ཐད་མཛད་རྗེས་བླ་མེད་འཇོག་མྱོང་མཁན་ཞིག་ཀྱང་ཡིན་སྐོར་སོགས་བརྗོད་སོང་།   ཁོང་གི་མི་ཚེ་ལོ་རྒྱུས་ལྟ་ན། རྒྱ་ཡེ་ཕུན་ཚོགས་ལགས་ནི་ཕྱི་ལོ་ ༡༩༣༡ ལོར་མཚོ་ལྷོ་ཁུལ་གུང་ཧོ་རྫོང་རིག་སྨོན་ཞིང་རྒྱ་ཡེ་སྡེ་བའི་འབྲོག་ཁྱིམ་ཕལ་བ་ཞིག་ནས་སྐྱེས་ཤིང་། ཆུང་དུས་ནས་མདོ་སྨད་དྷི་ཚ་དགོན་དུ་བོད་ཀྱི་ཚན་ཉིད་རིག་པ་དང་སུམ་རྟག་དག་གསུམ། དེ་བཞིན་སྒྲ་རིག་རྩིས་གསུམ་སོགས་ལ་སློབ་གཉེར་གནང་བ་དང་། ཕྱིས་སུ་རིག་སྨོན་ཞིང་རྒྱ་ཡེ་བ་ནས་སློབ་ཆུང་སློབ་གྲྭ་ཞིག་གསར་འཛུགས་བྱས་ཏེ་རང་ཉིད་ཀྱིས་སློབ་གྲྭའི་དགེ་རྒན་དང་སློབ་སྤྱི་གཅིག་ལྕོགས་ཀྱི་འགུན་ཁུར་ནས་བོད་རྒྱ་དབྱིན་གསུམ་གྱི་སྐད་ཡིག་དང་རྩིས་རིག་སོགས་སློབ་འཁྲིད་བྱས་མྱོང་བ་དང་། སློབ་གྲྭ་དེའི་ནང་བོད་རིགས་སློབ་མ་ ༦༠ ལྷག་ཡོད་འདུག   ཕྱོ་ལོ་ ༡༩༨༠ ལོར་རྒྱ་ནག་དམར་ཤོག་ཚོགས་པའི་སྤྱི་ཁྱབ་དྲུང་ཆེ་ཟུར་བ་ཧུའུ་ཡོ་པང་མཚོ་ལྷོ་ ཁུལ་དུ་གཟིགས་སྐོར་སྐབས་སུ། ཁོང་གིས་ས་གནས་བོད་རིགས་མི་མང་གི་དཀའ་སྡུག་སོགས་དངོས་ཡོད་གནས་ཚུལ་སྙན་སེང་ ཞུས་ཡོད་པ་རེད།   ཕྱི་ལོ་ ༡༩༩༢ ལོའི་ཕྱི་ཟླ་ ༣ པར་རྒྱ་གར་དུ་ཡོང་ནས་༧གོང་ས་མཆོག་ལ་མཇལ་ཁ་ཞུས་ཡོད་པ་མ་ཟད། སྐབས་དེར་སྐུ་མདུན་ནས་དགེ་སློང་གི་སྡོམ་པའང་བླངས་ཡོད་ཅིང་། ཕྱི་ལོ་ ༡༩༩༨ ལོའི་ཕྱི་ཟླ་ ༨ པར་ས་གནས་ཉེན་རྟོག་དཔོན་རིགས་ཚོའི་བཀའ་ཁྱབ་འོག་ཁོང་འཛིན་བཟུང་བཙོན་ཇུས་བྱས་ཡོད་པ་དང་། ཁོང་འཛིན་བཟུང་བྱེད་དོན་གཙོ་བོ་དེ་སྔོན་༧གོངས་ས་མཆོག་གིས་ཁོང་ལ་གནང་བའི་བཀའ་ཡིག་ཅིག་དང་ཁོང་གིས་རྒྱ་གར་ནས་ཁྱེར་ཡོང་བའི་དཔེ་ཆ་ཁ་ཤས་བཅས་ས་གནས་ཉེན་རྟོག་པའི་ལག་ཏུ་ཤོར་བའི་རྐྱེན་གྱིས་ཡིན་ཞིང་། སྐབས་དེར་ཁོང་དགུང་ལོ་ ༦༨ ཡིན་འདུག མ་ཟད་ཕྱི་ལོ་ ༡༩༦༠ ལོར་རིག་གནས་གསར་བརྗེ་སྐབས་ཀྱི་ངལ་རྩོལ་སྒྱུར་བཀོད་ལས་འགུལ་ཁྲོད་ཁོང་ལ་འཐབ་རྩོད་ཀྱིས་སྡུག་སྦྱོང་སུན་གཙེར་བཏང་ཡོད་པ་དང་། ༡༩༥༩ ལོར་རྒྱ་ནག་གིས་བོད་ལ་བཙན་འཛུལ་སྐབས་སུ་ཁོང་ཡང་བསྐྱར་འཛིན་བཟུང་བཙོན་འཇུག་བྱས་ཡོད་འདུག   གསར་འགོད་པ། པདྨ་རྒྱལ།
