རྡོ་རྗེ་གླིང་དུ་ངོ་རྒོལ་ལས་མཚམས་བཞག

དུས་ཚོད་ངེས་མེད་བར་ངོ་རྒོལ་ལས་མཚམས་བཞག་པ་དེས་རྒྱ་གར་གྱི་ཇ་ཡི་ཐོན་ཁུངས་ཆེ་ཤོས་ཀྱི་ས་ཁུལ་རྡོ་རྗེ་གླིང་(Darjeeling)གི་འཚོ་བ་སྙིང་བོས་སྟོང་བར་བྱས་ཡོད། དེ་ནི་བལ་ཡུལ་གྱི་སྐད་སྨྲ་བའི་ས་གནས་གོར་ཁ་ཚོགས་སྡེ་ཞིག་གིས་ཁོ་ཚོར་མངའ་སྡེ་ལོགས་སུ་དགོས་པའི་རེ་འདུན་འདོན་བཞིན་ཡོད། བྷི་བྷི་སི་གསར་ཁང་གིས་ ༢༠༡༧།༧།༦ ཉིན་སྤེལ། ངོ་རྒོལ་དེ་ནི་ཉེ་སྔོན་རང་ནུབ་བྷང་གྷལ་མངའ་སྡེའི་སློབ་གྲྭ་ཡོངས་ཀྱི་སློབ་ཚན་བྷང་གྷ་ལེ་སྐད་ཡིག་ནང་དུ་ཁྲིད་སྤེལ་བྱེད་དགོས་པའི་གཏན་འབེབས་བྱས་པར་རྐྱེན་བྱས་ཏེ། མི་མང་གིས་ལས་མཚམས་འཇོག་པའི་ངོ་རྒོལ་ཞིག་སྤེལ་ཡོད་པ་རེད། ལས་འགུལ་དེ་རྡོ་རྗེ་གླིང་གཙོ་བྱས་ཏེ་གོར་ཁ་མང་བའི་ས་གནས་མང་པོ་ཞིག་ནང་ངོ་རྒོལ་གྱི་མེ་ལྕེ་དེ་མཆེད་མགོ་བཙུགས་ཡོད། ངོ་རྒོལ་དེར་གོར་ཁ་མི་མང་བཅིངས་གྲོལ་མཐུན་ཚོགས་(Gorkha People’s Liberation Front)ཀྱིས་སྣེ་ཁྲིད་བྱེད་བཞིན་ཡོད་པ་དང་། ཁོ་ཚོས་ས་གནས་གཞུང་གི་ལས་ཁུངས་ཁག་ལ་འཇིགས་སྐུལ་དང་། སྲང་ལམ་གྱི་ཚོང་ཁང་ཟང་ཁང་། རླངས་འཁོར་སོགས་རྡུང་གཅོག་བྱས་ཡོད་པ་མ་ཟད། གོར་ཁའི་ཆེད་དུ་ “ཐག་གཅོད་མ་བྱུང་བར་དུ་འཐབ་འཛིང་བྱ་དགོས།” ཞེས་རྗོད་བཞིན་ཡོད་པ་རེད། ཁོ་ཚོས་ལོགས་སུ་དབྱེ་དགོས་པར་བརྗོད་བཞིན་པའི་མངའ་སྡེ་ནི་ནུབ་བྷང་གྷལ་གྱི་མཐའ་མཚམས་སུ་གཏོགས་པའི་བྱང་ཕྱོགས་རི་ཁུལ་དག་ཡིན། ངོ་རྒོལ་ལས་མཚམས་འཇོག་པ་དེ་ཟླ་སྔོན་མའི་ནང་ཡིན་ཡང་། ཁ་ཆེའི་རཱ་མཱ་དྷཱན་དུས་ཆེན་སྐབས་ཚོང་ཁང་སོགས་སྒོ་ཕྱེ་ཡོད་པ་རེད། ངོ་རྒོལ་ལས་མཚམས་བཞག་པ་དེས་ས་གནས་མང་ཚོགས་སྟོང་ཕྲག་མང་པོར་ཅ་དངོས་བསྐྱེལ་འདྲེན་སོགས་ལ་དཀའ་ངལ་ཆེན་པོ་འཕྲད་བཞིན་ཡོད་པ་རེད། རྡོ་རྗེ་གླིང་ནི་དབྱར་ཀ་ཡུལ་སྐོར་སྤྲོ་འཆམ་པ་སྟོང་ཕྲག་མང་པོ་འདུ་སའི་ས་ཚིགས་ཤིག་ཡིན་ན་ཡང་། ཕྱི་ཟླ་ ༦ པའི་ནང་ངོ་རྒོལ་དྲག་སྤྱོད་སྤེལ་བ་དེས་ཡུལ་སྐོར་སྤྲོ་འཆམ་པ་ཚོ་ཕྱིར་འཐོན་མགོ་བཙུགས་ཡོད་པ་རེད། ངོ་རྒོལ་ལས་འགུལ་གྱི་སྣེ་ཁྲིད་མཁན་བྷེ་མཱལ་རྒུ་རོང་(Bimal Gurung)གིས་ངོ་རྒོལ་ལས་མཚམས་འཇོག་དགོས་པའི་གསལ་བསྒྲགས་བཏང་བ་ནས་བཟུང་ངེས་མེད་དུ་ཡིབ་ནས་ཡོད་པ་དང་། ཁོའི་རྒྱབ་སྐྱོར་བ་ཚོས་གཞུང་གི་ཉེན་རྟོག་པ་ཚོར་བཅར་རྡུང་དང་ཚོང་ཁང་སོགས་གཏོར་བཅོམ་བྱས་ཏེ་སྒོ་འབྱེད་འཇུག་གི་མེད། བྷང་གྷལ་ཉེན་རྟོག་དམག་མིས་ཁོ་རང་ལས་ཁུངས་དང་ས་གནས་རི་ཁུལ་སོགས་ཀྱི་གབ་སྦས་བྱ་ཡུལ་ཡོངས་སུ་བཙལ་ཡོད་པ་དང་། ཁོའི་རྒྱབ་སྐྱོར་བ་མང་པོས་གཞུང་གི་ལས་ཁུངས་དང་རླངས་འཁོར་སོགས་མེར་བསྲེགས་བཏང་བས་འཛིན་བཟུང་བྱས་ཡོད་པ་རེད། ངོ་རྒོལ་བ་ཚོས་མངའ་སྡེ་བློན་ཆེན་མཱ་མཱ་ཊཱ་བྷ་ནཱར་རྗེ་ཡི་འདྲ་གཟུགས་མེར་བསྲེགས་བཏང་ཡོད། ཁོ་མོ་གོང་རི་སི་ཚོགས་པའི་མི་ཡིན་པ་དང་། ད་ལྟའི་མངའ་སྡེའི་གཞུང་འཛིན་བློན་ཆེན་ཡིན། ཟླ་དྲུག་པའི་ནང་ས་གནས་ཉེན་རྟོག་པས་གཞུང་གི་དམག་མིར་རོགས་སྐྱོར་དགོས་པའི་རེ་སྐུལ་ཞུས་ཡོད་པ་དང་། ངོ་རྒོལ་དྲག་སྤྱོད་སྤེལ་བའི་ནང་དུ་མ་མཐར་ཡང་མི་ལྔ་བསད་ཡོད་པ་དང་། ཉེན་རྟོག་པ་ ༣༠ ཚུད་པའི་མི་བརྒྱ་ལྷག་ལ་རྨས་སྐྱོན་ཕོག་ཡོད་པ་རེད། རྡོ་རྗེ་གླིང་དུ་ ༡༩༨༠ ལོར་མངའ་སྡེ་ལོགས་སུ་དགོས་པའི་ངོ་རྒོལ་བྱས་སྐབས་མི་ ༡༢༠༠ ལྷག་ཤི་ཡོད་པ་དང་། མཐའ་མར་གོར་ཁ་ས་ཁུལ་ལ་རང་སྐྱོང་ལྷན་ཚོགས་ཤིག་བཙུགས་ཏེ་གོར་ཁ་རི་ཁུལ་དུ་རང་སྐྱོང་གི་ཐོབ་དབང་ཡོད་པ་ཐག་གཅོད་བྱས་ཡོད། ཡིན་ན་ཡང་ ༡༩༨༠ ལོའི་ངོ་རྒོལ་སྣེ་ཁྲིད་པ་དམག་ཟུར་བ་སོ་བྷཱ་སི་གྷེ་ཤཱན(Subhas Ghishing)ཡིན་པ་དང་། ད་ལྟ་ཁོང་གི་ཚབ་ཏུ་དེ་སྔའི་མགོ་ཁྲིད་གཞོན་པ་རྒུ་རོང་བསླེབས་ཡོད་སྟབས། ཁོ་ནི་དེ་ལས་ཀྱང་གོར་ཁ་མངའ་སྡེ་ལོགས་སུ་དགོས་པར་འཁྲེགས་འཛིན་མཁན་ཞིག་ཡིན་པ་རེད། དུས་ཡུན་རིང་པོར་རྡོ་རྗེ་གླིང་ས་གནས་ནི་འཇིགས་སྣང་ཆེ་ཞིང་གཏན་འཇགས་མེད་པའི་ས་གནས་ཤིག་ཆགས་ཡོད་པ་དང་། གལ་ཏེ་མ་འོངས་པར་མུ་མཐུད་གནས་སྟངས་དེ་མུར་ལྷག་ཚེ་རྒྱ་གར་གྱི་ཇ་ཕོར་ལ་གནོད་པ་ཐེབས་རྒྱུ་ཡིན་པ་གདོན་མི་ཟའོ།། གསར་འགོད་པ། སངས་རྒྱས་དཀོན་མཆོག
