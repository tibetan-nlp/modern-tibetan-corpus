སྲིད་འཛིན་ཟུར་བར་སྐྱོན་འཛུགས།

  ཧོང་ཀོང་གྲོང་ཁྱེར་གྱི་སྲིད་འཛིན་ཟུར་བ་ཙིན་ཡེན་ཆན་གྱིས་སྤྱི་མང་གི་རྒྱུ་ནོར་ལྐོག་ཟ་བྱས་པའི་ཉེས་དོན་གཉིས་ཀྱི་ཐོག་ནས་སྐྱོན་འཛུགས་ཁྲིམས་བཤེར་བྱས་ཡོད་འདུག     ཧོང་ཀོང་རང་དབང་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། དེ་རིང་སྔ་དྲོར་ཧོང་ཀོང་གྲོང་ཁྱེར་ལྐོག་ཟ་རུལ་སུངས་ཞིབ་བཤེར་རང་དབང་ལྷན་ཁང་ (ICAC) གིས། ཧོང་ཀོང་གྲོང་ཁྱེར་གྱི་སྤྱི་ཁྱབ་ཟུར་བ་ཙིན་ཡེན་ཆན་དབུས་ལས་ཁུངས་སུ་ཁྲིད་དེ། ཁོས་སྔོན་ཆད་ཧོང་ཀོང་སྤྱི་མང་གཞུང་སྒྲུབ་ཁང་ལོག་སྤྱོད་བྱས་པའི་ཉེས་དོན་གཉིས་ཀྱི་ཐོག་ནས་སྐྱོན་འཛུགས་བྱས་ཡོད་པ་མ་ཟད། དེ་རིང་ཉིན་རྒྱབ་གྲོང་ཁྱེར་ཤར་རྒྱུད་ཁྲིམས་ཁང་ཞེས་པའི་ནང་ཁྲིམས་བཤེར་བྱེད་མགོ་བརྩམས་ཡོད་པ་དང་། ཧོང་ཀོང་སྤྱི་མང་གྱོན་དོན་གཏུག་བཤེར་ཁང་གི་འགན་འཛིན་ཟུར་བ་ཁུ་ཅ་ཞོང་གིས་ཙིན་ཡེན་ཆན་གྱི་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་གནད་དོན་དེ་འབྱུང་འགྱུར་ཟླ་བ་ ༥ ནང་ཐག་གཅོད་རྒྱུ་ཡིན་ཚུལ་བརྗོད་འདུག     ཧོང་ཀོང་གྲོང་ཁྱེར་གྱི་དབུས་བརྙན་འཕྲིན་དང་རླུང་འཕྲིན་ཁང་ (RTHK) གཉིས་ནས་གནས་ཚུལ་རྒྱང་སྲིང་བྱས་པའི་ནང་། སྲིད་འཛིན་ཟུར་བ་ཙིན་ཡེན་ཆན་གྱིས་སྤྱི་མང་ལ་སྲི་ཞུ་བསྒྲུབས་པའི་དུས་ཡུན་ལོ་ངོ་ ༤༥ རིང་ཁོས་རྒྱལ་ནང་གི་ལྐོག་ཟ་རུལ་སུངས་ཞིབ་བཤེར་རང་དབང་ལྷན་ཁང་དང་མཉམ་ལས་ཐོག དབང་སྒྱུར་སྲིད་ཇུས་ལག་བསྟར་བྱས་ཡོད་ཀྱང་། ཕྱི་ལོ་ ༢༠༡༢ ལོར་རྒྱལ་ནང་གི་ལྐོག་ཟ་རུལ་སུངས་ཞིབ་བཤེར་ལྷན་ཁང་གིས་ཙིན་ཡེན་ཆན་གྱིས་ཕྱི་ཕྱོགས་ཀྱི་ཚོང་པ་གྲགས་ཅན་ཁག་ཅིག་གི་ལྐོག་རྔན་བླངས་ཏེ། ཁོ་ཚོ་དང་མཉམ་འབྲེལ་གྱིས་མཚོ་ཐོག་ཁེ་ལས་གཉེར་ཡོད་པ་མ་ཟད། ཧྲིན་ཀྲིན་གྲོང་ཁྱེར་དུ་ཁོ་ལ་རྒྱས་སྤྲོས་ཆེ་བའི་སྡོད་ཁང་ཆེན་པོ་ཞིག་ཀྱང་ཡོད་པ་སོགས་ཀྱི་ཐོག་འདས་པའི་ལོ་ངོ་གསུམ་དང་ཕྱེད་ལྷག་རིང་ཞིབ་བཤེར་མཐིལ་ཕྱིན་བྱས་ཡོད་སྐོར་བརྗོད་འདུག     ཙིན་ཡེན་ཆན་གྱིས་ཕྱི་ལོ་ ༢༠༠༥ ནས་ཕྱི་ལོ་ ༢༠༡༢ ལོའི་བར་དབྱིན་ཇིའི་མི་སེར་སྤེལ་ཡུལ་ཧོང་ཀོང་གྲོང་ཁྱེར་སྲིད་གཞུང་གི་སྤྱི་ཁྱབ་དང་ནོར་སྲིད་ལས་ཁུངས་ཀྱི་དྲུང་ཆེ་བཅས་ཀྱི་འགན་ཁུར་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༡༢ ལོར་ཁོ་བོ་རྒན་གཡོལ་དུ་འགྲོ་སྐབས་རང་ལོ་ ༧༠ ཡིན་པ་མ་ཟད། ལོ་དེའི་ཕྱི་ཟླ་ ༩ ནང་ཧོང་ཀོང་གྲོང་ཁྱེར་སྲིད་གཞུང་གི་དྲུང་ཡིག་ཆེན་མོ་ཞི་ལོ་ཡོས་ཧོང་ཀོང་གྲོང་ཁྱེར་གྱི་སིན་ཧོང་རྨང་གཞི་འཕེལ་རྒྱས་ཚད་ཡོད་ཚོང་གཉེར་ཁང་གི་རྒྱུ་ནོར་ལྐོག་ཟ་བྱས་པའི་ནག་ཉེས་འོག་ལོ་ ༧ དང་ཕྱེད་ཀའི་ཁྲིམས་ཐག་བཅད་ཡོད་པ་རེད།     གསར་འགོད་པ། པདྨ་རྒྱལ།
