
ངས་ཀྲུང་ཧྭ་མིན་གོ་ཡི་རྩ་ཁྲིམས་གླེང་པ་ཡིན།

  གཟའ་སྤེན་པའི་ཉིན་རྒྱ་ནག་དང་ཐའེ་ཝན་གྱི་སྲིད་འཛིན་རྣམ་གཉིས་ལོ་ངོ་ ༦༠ རྗེས་ཀྱི་ཐུག་འཕྲད་དང་པོ་གནང་བའི་ཐད་རྒྱལ་སྤྱིའི་ཐོག་གླེང་སློང་ཤུགས་ཆེན་པོ་ཡོང་བཞིན་ཡོད་པ་མ་ཟད་རྒྱ་ནག་དང་ཐའེ་ཝན་ནང་དུའང་ངོས་འཛིན་དང་ཚོད་དཔག་སྣ་ཚོགས་བྱེད་བཞིན་འདུག   ཕྱི་ཚེས་ ༧ ཉིན་སེང་གྷ་པུར་རྒྱལ་ཁབ་ནང་གི་ཤམ་བྷ་ལའི་གསོལ་མགྲོན་ཁང་དུ། རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་དང་ཐའེ་ཝན་གྱི་སྲིད་འཛིན་མཱ་ཡོན་ཅའུ་གཉིས་ཐུག་འཕྲད་བྱས་ཡོད་ཅིང་། དེ་རིང་པའོ་ཞིན་དྲ་བས་གནས་ཚུལ་སྤེལ་བའི་ནང་། མཱ་ཞཱི་གཉིས་ཀྱི་ཐུག་འཕྲད་རྗེས་ཐའེ་ཝན་སྲིད་འཛིན་ཕྱིར་ལོག་ཁའི་གནམ་ཐང་དུ་གསར་འགོད་པར། ངས་ག་རེ་གླེང་ཡོད་པ་རེད་ཅེ་ན། ༡༩༩༢ ལོའི་མཐུན་མོང་གི་ངོས་འཛིན་དང་། ཀྲུང་གོ་གཅིག་ལ་དོན་འགྲེལ་མ་འདྲ་བ། རྒྱ་ནག་དང་ཐའེ་ཝན་གྱི་རང་བཙན་གླེང་གི་ཡོད་པ་མ་རེད། དེ་ནི་ང་ཚོའི་ཀྲུང་ཧྭ་མིན་གོའི་རྩ་ཁྲིམས་ཀྱིས་ཆོག་མཆན་ཐོབ་ཀྱི་མེད་པས་རེད། ངས་ཀྲུང་ཧྭ་མིན་གོའི་རྩ་ཁྲིམས་གླེང་པ་ཡིན། གཙོ་བོ་རྩ་ཁྲིམས་ཀྱི་དོན་ཚན་ལེགས་བཅོས་གཅིག་པུ་གླེང་བ་ལས་ཀྲུང་ཧྭ་མིན་གོ་གླེང་མེད། རྩ་ཁྲིམས་ཀྱི་དོན་ཚན་བཅོས་བསྒྱུར་གླེང་པ་ཡིན་སྟབས། རྩ་ཁྲིམས་ཐེངས་གཉིས་ལ་གླེང་ཡོད་པ་དང་ཀྲུང་ཧྭ་མིན་གོ་ཐེངས་གཅིག་བཤད་པ་ཡིན། རྒྱ་ནག་གཅིག་ལ་དོན་འགྲེལ་མ་འདྲ་བ་ཡོད་པ་ཐེངས་གཅིག་གླེང་པ་ཡོད་སྟབས། བཤད་དགོས་པ་ཆ་ཚང་ང་ཚོས་བཤད་ཡོད། ཅེས་བརྗོད་ཡོད་སྐོར་བཀོད་འདུག   མཱ་ཡོན་ཅའུ་ཡི་གསར་འགོད་གསལ་བསྒྲགས་སྐབས། ༡༩༩༢ ལོའི་མཐུན་མོང་གི་ངོས་འཛིན་ཞེས་ཐེངས་ ༡༢ གླེང་ཡོད་པ་དེ་ཕྱོགས་གཉིས་ཀྱི་ཞི་བདེ་ཡོང་པའི་བརྗོད་གཞི་གལ་ཆེན་ཆགས་ཡོད་པར་ཐའེ་ཝན་གྱི་གསར་འགོད་པས་དེའི་སྐོར་གཏིང་འདྲི་བྱས་སྐབས། མཱ་ཡོན་ཅའུ་ཡིས་ལན་དུ། རྒྱ་ནག་གཅིག་ལ་དོན་འགྲེལ་མ་འདྲ་བ་ཡོད་པ་དེ་ཐའེ་ཝན་དང་རྒྱ་ནག་སྐམ་ས་གཉིས་ཀྱིས་རྒྱ་ནག་གཅིག་གི་གནད་དོན་ལ་ལྟ་ཚུལ་མ་འདྲ་བ་དང་། སྔོན་ཆད་ ༡༩༩༢ ལོར་ཕྱོགས་གཉིས་ཀྱིས་ཐུན་མོང་གི་ངོས་འཛིན་མ་འདྲ་བའི་ཐོག་ནས་རྒྱ་ནག་གི་གོ་དོན་ལ་འགྲེལ་བཤད་བརྒྱབ་པ་རེད། དེ་ཕྱོགས་གཉིས་ལ་དགོས་པའི་བར་སྟོང་ཡང་རེད། ཅེས་བརྗོད་འདུག   ཁོང་གིས་མུ་མཐུད་དུ། གལ་སྲིད་ཁྱེད་ཀྱིས་ད་རེབ་ཀྱི་ཐུག་འཕྲད་ལ་མཇུག་འབྲས་ག་རེ་བྱུང་ཡོད་པ་དྲིས་ན། ཆེས་གལ་ཆེ་ཤོས་ཏེ་མཚོ་འགག་གཉིས་ཀྱི་མགོ་འཁྲིད་ཐོག་མ་ཐུག་འཕྲད་བྱས་པ་མ་ཟད། འབྲེལ་ཡོད་ཀྱི་གནད་དོན་གླེང་མོས་བྱུང་པ་རེད། ང་ཚོས་ཐག་གཅོད་བྱ་དགོས་པའི་གནད་དོན་མང་པོ་ཡོད་པ་དང་། འདས་པའི་ལོ་མང་རིང་ཕྱོགས་གཉིས་ཀྱིས་གནད་དོན་གང་མང་ཞིག་ཐག་གཅོད་བྱས་ཡོད་ཀྱང་། ད་དུང་གནད་དོན་མང་པོ་ལྷག་ཡོད་པ་མ་ཟད། གནད་དོན་གསར་བ་བྱུང་ན་དེ་ལས་མཐོ་བའི་སྟེགས་བུའི་སྟེང་ཐག་གཅོད་བྱ་དགོས་ཀྱི་འདུག ཅེས་བརྗོད་འདུག   ཡང་ཁོང་གིས་གསལ་བཤད་ནང་། ང་ཚོས་སྐུ་ཞབས་ཞི་ཡི་མདུན་དུ། རྒྱ་ནག་གཅིག་གི་གནད་དོན་ལ་ལྟ་ཚུལ་མ་འདྲ་བ་དེར་ང་ཚོས་ཀྲུང་ཧྭ་མིན་གོ་ཡི་རྩ་ཁྲིམས་དང་དམངས་གཙོའི་ཆབ་སྲིད་ཀྱི་སྒྲོམ་གཞིའི་ནང་ལ་རྒྱ་ནག་གཉིས་ཁས་ལེན་མེད་པ་དེར། མི་ཁ་ཤས་ཀྱིས་ང་ཚོས་ཀྲུང་ཧྭ་མི་དམངས་སྤྱི་མཐུན་རྒྱལ་ཁབ་ལ་ངོ་དགའ་བྱེད་བཞིན་པའི་གོ་བ་ལོག་གི་འདུག དེ་མ་རེད། དེར་ང་ཚོའི་རྩ་ཁྲིམས་ཀྱིས་ཀྱང་ངོས་ལེན་པའི་གོ་དོན་རེད། ཅེས་བརྗོད་འདུག   ཁ་སང་ཐའེ་ཝན་དམངས་གཙོ་ཡར་ཐོན་ཚོགས་པའི་ཚོགས་གཙོ་སོན་ཁྲའོ་ཡའུ་ (Song Chuyu) ལ་གསར་འགོད་པས་མཱ་ཞི་གཉིས་ཀྱི་ཐུག་འཕྲད་སྐོར་ལ་བཅའ་འདྲི་བྱས་སྐབས་ཁོས། ཕྱོགས་གཉིས་ཀྱི་འབྲེལ་བར་བསྔགས་བརྗོད་ཡོད། ཡིན་ཡང་འདྲ་མཉམ་དང་བརྩི་འཇོག་གི་རྩ་དོན་འོག་ཕྱོགས་གཉིས་ཀྱི་ཚོགས་འདུའི་མགོ་མཇུག་དང་གསར་འགོད་གསལ་བསྒྲགས་སོགས་ལ་གནས་བབ་འདྲ་མཉམ་མེད་པ་མཐོང་པར་ཁོས་ཐོག་མར་ག་རེ་བྱས་ནས་དམའ་རིམ་མི་ཡིས་གསར་འགོད་གསལ་བསྒྲགས་སྤེལ་བ་རེད། དེའི་རྗེས་ང་ཚོ་དེ་ནས་སྒུག་དགོས་པ་རེད། དེ་འདྲ་མཉམ་རེད་དམ། དེ་ལ་རྒྱ་རེད་དམ། ཞེས་བརྗོད་འདུག   ཡང་སོན་ཁྲའོ་ཡའུ་ཡིས། མཱ་ཡོན་ཅའུ་ཡིས་རྒྱ་ནག་གཅིག་དང་ ༡༩༩༢ ལོའི་ཐུན་མོང་གི་ངོས་འཛིན་ཐེངས་མང་གླེང་ཡང་། ཀྲུང་ཧྭ་མིན་གོ་ཐའེ་ཝན་ལ་ཡོད་པ་གླེང་དགོས་པའི་གོ་སྐབས་དམ་འཛིན་བྱས་མེད་པ་དང་། མཱ་ཞི་གཉིས་ཀྱི་བཞུགས་མོས་ཚོགས་འདུ་གྲོལ་རྗེས་ཐོག་མར་མཱ་ཡོན་ཅའུ་རང་ཉིད་ཀྱིས་གསར་འགོད་གསར་བསྒྲགས་སྤེལ་ཡོད་ཅིང་། རྒྱ་ནག་ཕྱོགས་ནས་ཐའེ་ཝན་ལས་དོན་གཞུང་སྒྲུབ་ཁང་གི་འགན་འཛིན་ཀྲང་ཀྲི་ཅོན་ (Zhang zhijun) གྱིས་གསར་འགོད་གསལ་བསྒྲགས་བྱས་པ་གཉིས་ལ་ཁྱད་པར་ཆེན་པོ་ཡོད་པ་རེད། ཅེས་ངོས་འཛིན་བྱེད་ཀྱི་འདུག   ཐའེ་ཝན་དམངས་གཙོ་ཡར་ཐོན་ཚོགས་པའི་ཚོགས་གཙོ་ཚད་ཡེ་ཝིན་ (Tsai, Ing-Wen) གྱིས། ཁ་སང་ཚོགས་འདུའི་སྔོན་ལ་ང་ཚོས་ངེས་པར་དུ་ཐའེ་ཝན་མི་མང་གི་ལ་རྒྱ་དང་ཐའེ་ཝན་མི་མང་གི་རེ་འདུན་ཕྱི་གསལ་ནང་གསལ་གྱིས་ཁྱབ་བསྒྲགས་དང་། གལ་ཆེ་ཤོས་དེ་ཆབ་སྲིད་ཀྱི་གཞི་རྩའི་ཆ་རྐྱེན་གང་ཡང་མེད་དགོས་ཞེས་མཱ་ཡོན་ཅའུ་ལ་ལབ་ཡོད་ཀྱང་། ཁ་སང་བལྟས་ཡོང་དུས་ཁོ་ཚོས་རྒྱལ་སྤྱིའི་སྟེགས་བུ་བེད་སྤྱོད་བཏང་སྟེ། ཆབ་སྲིད་དང་གཞི་རྩའི་ཆ་རྐྱེན་གཅིག་གི་སྒྲོམ་གཞིའི་འོག་ཐའེ་ཝན་མི་མང་ལ་མ་འོངས་པར་འདེམས་ཀའི་རང་དབང་ཆུང་དུ་བཏང་སོང་། ཞེས་དང་། རྒྱལ་ཁབ་གཅིག་ལ་ཆེས་མཁོ་བའི་རང་དབང་དང་ཁེ་དབང་ཐོག་སྲིད་འཛིན་མཱ་ཡོན་ཅའུ་ཡིས་རྒྱལ་ཁབ་གཅིག་གིས་ཁུར་དགོས་པའི་འོས་འགན་དེ་ཁུར་མ་སོང་། ཞེས་བརྗོད་འདུག   གསར་འགོད་པ། འཇིགས་མེད་རྒྱ་མཚོ།
