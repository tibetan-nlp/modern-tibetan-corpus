དེ་སྔའི་ཧིན་སྒོར་ ༥༠༠ ཅན་དང་ ༡༠༠༠
ཅན་གཉིས་བེད་སྤྱོད་གཏོང་མི་ཆོག་ཟེར།

རྒྱ་གར་སྲིད་བློན་མུ་ཌེ་མཆོག་གིས་ཛ་དྲག་གི་གསལ་བསྒྲགས་སྤེལ་ཏེ་དོ་ནུབ་ནས་བཟུང་ཧིན་སྒོར་ལྔ་བརྒྱ་ལོ་དང་སྟོང་ལོ་བེད་སྤྱོད་བཏང་མི་ཆོག་པ་དང་འབྱུང་འགྱུར་ཉིན་ལྔ་བཅུའི་ནང་ཚུན་ལ་དངུལ་དེ་དག་ཡོངས་རྫོགས་བརྗེ་བོ་བསློག་དགོས་སྐོར་བསྒྲགས་འདུག ད་ལྟ་རང་རྒྱ་གར་གྱི་སྲིད་བློན་ནཱ་རེན་ཌིར་མུ་ཌེ་མཆོག་གིས་ཛ་དྲག་གི་གསལ་བསྒྲགས་སྤེལ་ཏེ། དོ་ནུབ་མཚན་གུང་ཆུ་ཚོད་ ༡༢ ནས་བཟུང་ཧིན་སྒོར་ ༥༠༠ ཅན་དང་ ༡༠༠༠ ཅན་གཉིས་ཕར་སྤྲོད་ཚུར་ལེན་གྱི་བེད་སྤྱོད་གཏོང་མི་ཆོག་པ་མ་ཟད། མི་སོ་སོའི་ལག་ཏུ་ཉར་ཡོད་པའི་ཧིན་སྒོར་ལྔ་བརྒྱ་ལོ་དང་སྟོང་ལོ་ཅན་ཡོངས་རྫོགས་ཕྱི་ཟླ་ ༡༡ པའི་ཚེས་ ༡༠ ནས། ཕྱི་ཟླ་ ༡༢ པའི་ཚེས་ ༣༠ ཉིན་བར་གྱི་ཉིན་གྲངས་ ༥༠ ནང་ཚུན་ལ་དངུལ་ཁང་དང་སྦྲགས་ཁང་ཁག་ནང་འཇུག་ཐུབ་བྱེད་རྒྱུ་དང་། ཡང་ན་བརྗེ་བོ་རྒྱག་དགོས་པ་ལས་དུས་ཚོད་དེ་ལས་བརྒལ་ཚེ་དངུལ་དེ་དག་ཡོངས་རྫོགས་རྩིས་མེད་གཏོང་རྒྱུ་ཡིན་སྐོར་བརྗོད་འདུག ཕྱི་ཟླ་ ༡༡ ཚེས་ ༩ དང་ ༡༠ ཉིན་གཉིས་རིང་རྒྱ་གར་གྱི་དངུལ་ཁང་ཡོངས་རྫོགས་སྒོ་རྒྱག་རྒྱུ་ཡིན་པ་མ་ཟད། དངུལ་འདོན་བྱང་བུ་(ATM)བརྒྱུད་དེ་དངུལ་འདོན་ཐུབ་ཀྱི་མེད་པ་དང་། ཚེས་ ༡༡ ཉིན་ནས་དངུལ་ཁང་ཁག་སྒོ་འབྱེད་བྱ་རྒྱུ་ཡིན་འདུག ཛ་དྲག་གི་ལས་འགུལ་དེ་ནི་ཁྲིམས་འགལ་གྱི་དངུལ་ཉར་བ་དང་དངུལ་རྫུན་མ་བཀྲམ་པ། དེ་བཞིན། ནག་ཚོང་དང་ལྐོག་ཟ་རུལ་སུངས་སོགས་བཀག་སྡོམ་བྱེད་ཆེད་རྒྱལ་ཁབ་ཀྱི་ཁྲིམས་ཐོག་ནས་གཏན་འབེབས་བྱས་པའི་ལས་དོན་གལ་ཆེན་ཡིན་སྐོར་སོགས་བརྗོད་འདུག་ལ། ཐེངས་འདིའི་ལས་འགུལ་དེ་དང་མཉམ་དུ་ཧིན་སྒོར་ ༥༠༠ ཅན་དང་ ༢༠༠༠ ཅན་གྱི་དངུལ་ལོ་གསར་པའང་འགྲེམ་སྤེལ་བྱས་ཡོད་འདུག
