ད་ལོ་རྒྱ་གར་ལ་སྐྱབས་བཅོལ་དུ་ཡོང་མཁན་ ༩༨
འདུག

དེ་རིང་ཕྱི་དྲོར་བདེ་སྲུང་བཀའ་བློན་འཕགས་པ་ཚེ་རིང་མཆོག་གིས་གྲོས་ཚོགས་ནང་། བདེ་སྲུང་ལས་ཁུངས་ཀྱི་ལས་བསྡོམས་སྙིང་བསྡུས་སྙན་སྒྲོན་ཞུས་ཡོད་པ་དང་། དེའི་རྗེས་ལ་སྤྱི་འཐུས་རྒྱུན་ལས་ཀྱིས་བདེ་སྲུང་ལས་ཁུངས་ཀྱི་ལས་བསྡོམས་རྒྱས་པ་ཤོག་ངོས་ ༡༥ ཡོད་པའི་ཐོག་ཞིབ་འཇུག་བྱས་པའི་སྙན་ཐོ་སྙན་སེང་ཞུ་རྗེས། སྤྱི་འཐུས་ཁག་ཅིག་གིས་བདེ་སྲུང་བཀའ་བློན་མཆོག་ལ་དྲི་བ་དང་སོ་སོའི་དགོངས་འཆར་སོགས་བཏོན་ཡོད། བོད་མིའི་སྒྲིག་འཛུགས་བདེ་སྲུང་ལས་ཁུངས་ནི། བོད་མིའི་སྒྲིག་འཛུགས་ལ་ལྷན་ཁང་བདུན་ཡོད་པའི་ཁྲོད་ཀྱི་གཅིག་ཡིན་པ་དང་། བདེ་སྲུང་ལས་ཁུངས་ལ་གངས་སྐྱིད་ཀྱི་ལས་ཁུངས་ཨ་མ་དང་། མེག་ལོར་གྱི་བདེ་སྲུང་ཡན་ལག དེ་བཞིན། སྣེ་ལེན་ཁང་བཅས་སྡེ་ཚན་གསུམ་ཡོད། བདེ་སྲུང་ལས་ཁུངས་ཀྱི་ལས་འགན་གཙོ་བོ་༧གོང་ས་མཆོག་གི་སྐུའི་སྲུང་སྐྱོབ་དང་། བོད་ནས་གསར་དུ་འབྱོར་མཁན་དང་ཕྱིར་བོད་དུ་ལོག་མཁན། དེ་བཞིན། ཕྱི་རྒྱལ་ཁག་ཏུ་སྐྱོད་མཁན། སྐྱབས་བཅོལ་ལག་འཁྱེར་དང་ཕྱི་སྐྱོད་ལག་འཁྱེར། བཙན་བྱོལ་བོད་མིའི་སྤྱི་ཚོགས་ནང་གི་བདེ་འཇགས་དང་འབྲེལ་བའི་ལས་དོན་རིགས་སོགས་ཡིན། བདེ་སྲུང་ལས་ཁོངས་ཀྱི་ལས་བསྡོམས་ནང་བཀོད་པ་ལྟར་ན། ད་ལོར་བོད་མི་ཕྱི་སྐྱོད་ལག་འཁྱེར་(IC)ཞུ་མཁན་མི་སྐྱ་ཕོ་མོ་ ༢༧༢༣ དང་། གྲྭ་བཙུན་ ༣༤༠ བཅས་ཁྱོན་བསྡོམས་ ༣༠༦༣ བྱུང་ཡོད་འདུག་ལ། ནང་མི་ཟླ་སྒྲིལ་ (Travel Document)ཡིག་ཆ་བཟོ་མཁན་ ༢༢ བྱུང་ཡོད་འདུག ད་ལོར་བོད་ནས་ཕེབས་མཁན་ཁྱོན་བསྡོམས་ ༡༢༨༢ བྱུང་ཡོད་པའི་ཁྲོད་ཕྱིར་བོད་དུ་ལོག་མཁན་ ༡༡༩༤ དང་། བལ་ཡུལ་རྒྱ་གར་གཞུང་ཚབ་ནས་སྐྱབས་བཅོལ་དུ་སྡོད་ཆོག་པའི་ཡིག་ཆ་ཞུ་མཁན་ ༩༨ བྱུང་ཡོད་འདུག གཞན། རྒྱ་གར་ནས་བོད་ནང་གཉེན་འཕྲད་དུ་སྐྱོད་མཁན་ ༢༨༢ བྱུང་ཡོད་འདུག རྒྱ་གར་ནང་སྐྱེས་པའི་ཕྲུ་གུ་ ༡༠༤ དང་། བོད་ནས་ཕེབས་མཁན་ ༡༡༠ བཅས་ཁྱོན་བསྡོམས་ ༢༡༤ ལ་སྐྱབས་བཅོལ་ལག་དེབ་(RC)གསར་བཟོ་བྱས་ཡོད་འདུག དེ་བཞིན། སྐྱབས་བཅོལ་ལག་དེབ་དུས་འགྱངས་བྱེད་མཁན་ཁྱོན་བསྡོམས་ ༥༣༤ བྱུང་ཡོད་འདུག ད་ལོར་ཕྱི་རྒྱལ་ནས་ཕྱིར་ལོག་ཐོངས་ཆན་(Return VISA)རྒྱབ་གཉེར་ཞུ་མཁན་ ༥༡༠ དང་། ཕྱི་རྒྱལ་འགྲོ་བའི་ཡིག་ཆ་(Exit Permit)ཞུ་མཁན་ ༩༦༦ བྱུང་ཡོད་འདུག བདེ་སྲུང་ལས་ཁུངས་ཀྱི་ལས་བསྡོམས་ཐོག་སྤྱི་འཐུས་ཚོས་བགྲོ་གླེང་གནང་སྐབས། སྤྱི་འཐུས་ཁག་ཅིག་གིས་༧གོང་ས་མཆོག་གི་སྐུ་དཔྱྭའི་སྐོར་ལ་དམ་ཡིད་གཟབ་སྔར་ལས་མཐོར་འདེགས་གནང་དགོས་སྐོར་དང་། ཕྱི་རྒྱལ་ཁག་ཏུ་ཆིབས་བསྒྱུར་སྐབས་དོལ་རྒྱལ་བ་དང་རྒྱ་ནག་གཞུང་གི་ཆེད་མངགས་ངོ་རྒོལ་བ་ཚོར་དམ་དོན་འཐུས་ཤོར་མེད་པ་གནང་གལ་ཆེ་སྐོར་བརྗོད་སོང་།
