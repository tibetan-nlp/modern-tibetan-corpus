བོད་མི་ ༩༡༤ ཁེ་ན་ཌར་འབྱོར་ཟིན།

ཨ་རུ་ན་ཅལ་མངའ་སྡེ་ནས་བོད་མི་ ༩༡༤ ཁེ་ན་ཌར་འབྱོར་ཟིན་པ་དང་མི་གྲངས་ ༥༥ ལ་གཞུང་ཚབ་ནས་དྲི་བ་དྲིས་ལན་དང་གཟུགས་པོར་བརྟག་དཔྱད་བཅས་གནང་ཚར་ཡོད་འདུག འདི་གའི་གསར་འགོད་པས་བོད་མིའི་སྒྲིག་འཛུགས་ནང་སྲིད་ལས་ཁུངས་ཀྱི་འབྲེལ་ཡོད་ལས་བྱེད་དངོས་གྲུབ་འོད་ཟེར་ལགས་སུ། ཨ་རུ་ན་ཅལ་མངའ་སྡེ་ནས་བོད་མི་ ༡༠༠༠ ཁ་ན་ཌར་གཞིས་སྤོར་གྱི་ལས་གཞིའི་གནས་སྟངས་ད་ལྟ་ག་འདྲ་ཆགས་ཡོད་པའི་སྐོར་བཅར་འདྲི་ཞུས་པ་ལྟར་ན། ཕྱི་ལོ་ ༢༠༡༡ ལོ་ནས་བཟུང་དབུས་བོད་མིའི་སྒྲིག་འཛུགས་དང་ཁ་ན་ཌ་གཞུང་མཉམ་འབྲེལ་ཐོག རྒྱ་གར་བྱང་ཤར་གྱི་ཨ་རུ་ན་ཅལ་མངའ་སྡེའི་ཁོངས་སུ་ཡོད་པའི་བོད་མིའི་གཞིས་གྲོང་མེའོ་ཆོས་འཕེལ་གླིང་དང་ཏེ་ཛུ་དར་རྒྱས་གླིང་། ཀྲུ་གྲིང་དང་འབུམ་ལ་བསྟན་འཛིན་གླིང་བཅས་ནས་མི་གྲས་ ༡༠༠༠ མ་བརྒལ་བ་ཞིག་ཁ་ན་ཌར་གཞིས་སྤོར་གཏོང་བའི་ལས་གཞི་དབུ་འཛུགས་གནང་ཡོད་པ་ལྟར། ད་ལྟའི་བར་གྱི་གཏོོང་རིམ་ ༡༠༥ ཡི་ནང་ཁྱོན་མི་གྲངས་ ༩༡༤ བཏང་ཟིན་པ་དང་། མི་གྲངས་ ༥༥ ལ་གཞུང་ཚབ་ཁང་ནས་གཟུགས་པོར་བརྟག་དཔྱད་དང་དྲི་བ་དྲིས་ཅན་བཅས་གནང་ཚར་ཡོད་སྟབས། འདི་ལོའི་ཕྱི་ཟླ་ ༤ པའི་ནང་བོད་མི་ ༥ ཕར་གཏོང་རྒྱུ་ཡིན་འདུག དངོས་གྲུབ་འོད་ཟེར་ལགས་ཀྱིས། ལས་གཞིའི་ཆོད་གན་ནང་ཕྱི་ལོ་ ༢༠༡༦ ཟླ་ ༥ ཚེས་ ༡༡ ཉིན་ནས་མི་འགྲོའི་སྙན་ཞུ་མཚམས་གཅོད་རྒྱུ་ཡིན་ཞེས་ཞུས་ཡོད་པ་ལྟར། དུས་ཡུན་དེའི་ནང་ཚུད་དུ་ཁ་ན་ཌའི་གཞུང་ཚབ་ཁང་ལ་ཁྱོན་མི་གྲངས་ ༡༠༥༠ ཡི་སྙན་ཞུ་ཕུལ་ཟིན་ཡོད་པ་དང་། སྤྱིར་མི་གྲངས་ ༡༠༠༠ ལས་མ་བརྒལ་བ་ལེན་རྒྱུའི་འཆར་གཞི་ཡིན་ཡང་སྐབས་རེར་ཁ་ན་ཌ་ཕྱོགས་ནས་དགག་བྱ་བྱུང་བ་སོགས་ལ་ཁ་གསབ་དང་། ལྷག་པར་དུ་ལས་གཞི་མགོ་བརྩམས་པ་ནས་ད་ལྟའི་བར་དུ་ཕྲུ་གུ་ ༢༠ གསར་དུ་སྐྱེས་ཡོད་པར་བརྟེན་སྙན་ཞུའི་གྲངས་འབོར་གང་འཚམ་ཡར་འཕར་དགོས་བྱུང་ཡོད་པ་བརྗོད་བྱུང་། དུས་ཚོད་འགྱངས་དགོས་བྱུང་བའི་རྒྱུ་རྐྱེན་གཅིག་ནི་གཟུགས་པོའི་འཕྲོད་བསྟེན་ཐོག་དཀའ་ངལ་ཡོད་པ་དཔེར་ན། གློ་ནད་ལྟ་བུ་ཡོད་པ་ཡིན་ན་དུས་ཡུན་ཟླ་ ༦ དང་ཡང་ན་ལོ་གཅིག་དེ་འདྲ་ཕྱིར་འགྱངས་ཀྱིས་སྨན་བཅོས་བྱེད་དགོས་པ་དང་། རྒྱུ་རྐྱེན་གཞན་ཞིག་ནི་དམག་སྒར་ནས་ཐོན་པ་རྣམས་ལ་ཐད་ཀར་ཁ་ན་ཌ་གཞུང་རང་ནས་ཁོ་ཚོའི་ལོ་རྒྱུས་རྒྱབ་ལྗོངས་ལ་ཞིབ་འཇུག་བྱས་རྗེས་འཐུས་པ་བྱུང་ན་གཞི་ནས་ཕར་འགྲོ་ཆོག་གི་ཡོད་སྐོར་བརྗོད་བྱུང་། ཁ་ན་ཌ་གཞུང་གི་མི་གྲངས་ཚད་འཛིན་ལྟར་བྱས་ན། ལས་གཞི་འདིའི་ཁྲོད་ད་དུང་མི་གྲངས་ ༨༦ ཙམ་ཕར་འབྱོར་དགོས་ཀྱི་ཡོད་པ་རེད།   གསར་འགོད་པ། པདྨ་མཚོ།
