པདྨ་རིག་འཛིན་ཀློད་གྲོལ་བཏང་།

རྔ་བ་རྫོང་ནས་ཡིན་པའི་སྒྲ་བརྙན་རྡུང་ལེན་པ་པདྨ་རིག་འཛིན་ལགས་བཙོན་ཁང་ནས་ཀློད་གྲོལ་བཏང་ཡོད་འདུག   དེ་རིང་བོད་ནང་ནས་གནས་ཚུལ་བཏང་བ་ལྟར་ན། སྒྲ་བརྙན་རྡུང་ལེན་པ་པདྨ་རིག་འཛིན་ལགས་བཙོན་ཁང་ནས་ཀློད་གྲོལ་བཏང་ཡོད་པ་དང་། ཁོང་གི་ཁྱིམ་མི་གཉེན་ཉེ་དང་གྲོགས་པོ་སོགས་སེར་སྐྱ་མང་པོ་བསུ་བར་ཕྱིན་ཡོད་འདུག   པདྨ་རིག་འཛིན་ལགས་ཕྱི་ལོ་ ༢༠༡༣ ཟླ་ ༥ ཚེས་ ༧ ཉིན་ཉེན་རྟོག་པས་ཁྲིན་ཏུའུ་གྲོང་ཁྱེར་ནས་འཛིན་བཟུང་བྱས་ཏེ། ལོ་གཅིག་ལྷག་རིང་འདྲི་གཅོད་བཀག་ཉར་བྱས་རྗེས། ཁོང་གི་རྡུང་ལེན་ཆེད་སྒྲིག་ནང་ཆབ་སྲིད་ཀྱི་གནད་དོན་ཡོད་པའི་ཉེས་མིང་འོག འདས་ལོའི་ཕྱི་ཟླ་ ༡༡ ཚེས་ ༢༧ ཉིན་ཁྲིན་ཏུའུ་གྲོང་ཁྱེར་ཅེན་ཉོས་ཆས་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་ཟེར་བ་ཞིག་གིས། ཁོང་གི་ཁྱིམ་མི་ཚོར་ཁྲིམས་རྩོད་པ་གླ་རུ་མ་བཅུག་པར་ཉེས་ཆད་རྒྱ་སྒོར་ཁྲི་ ༥ དང་། ལོ་ ༢ ཟླ་ ༦ གི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་ཀྱང་དེ་རིང་ཀློད་གྲོལ་བཏང་ཡོད་འདུག ད་ལྟའི་ཆར་རྒྱ་གཞུང་གིས་རྔ་བ་ཕྱོགས་ཀྱི་སྐད་འཕྲིན་སོགས་བརྒྱུད་ལམ་ཁག་ལ་དམ་དྲག་ཆེན་པོ་བྱེད་བཞིན་ཡོད་སྟབས། ཁོང་གི་སྐུ་གཟུགས་གནས་སྟངས་སོགས་ཅི་ཡང་ཤེས་རྟོགས་ཐུབ་ཀྱི་མི་འདུག   རྡུང་ལེན་པ་པདྨ་རིག་འཛིན་ལཊ་རྔ་བ་རྫོང་རྨེ་རུ་མའི་རུ་ཁག་ ༤ པ་ནས་ཡིན་ཞིང་། ཆུང་དུས་གྲྭ་པ་བྱས་ཏེ་རྔ་བ་རྫོང་གནམ་མཚོ་དགོན་པར་ཞུགས་ཡོད་པ་དང་། རིམ་གྱིས་གསེར་རྟ་བླ་རུང་ལྔ་རིག་ནང་བསྟན་སློབ་གླིང་དུའང་སློབ་གཉེར་གནང་མྱོང་ཡོད་ཀྱང་། ཕྱིས་སུ་སྐྱ་སར་བབས་ཡོད་འདུག ཕྱི་ལོ་ ༡༩༨༢ ལོ་ནས་བཟུང་ཁོང་གིས་༼གངས་ལྗོངས་དྲན་གླུ༽་དང་༼མིག་ཆུ༽་སོགས་རྒྱ་གཞུང་གི་བཙན་གནོན་འོག་གི་བོད་ཀྱི་གནས་སྟངས་སྤྱི་དང་བོད་མིའི་དཀའ་སྡུག་མཚོན་པའི་རྡུང་ལེན་གྱི་འོད་སྡེར་ཁ་ཤས་འགྲེམ་སྤེལ་བྱས་ཡོད་ཅིང་། རྒྱ་ཆེའི་བོད་མི་མང་པོས་དགའ་བསུ་ཐོབ་ཡོད་འདུག ཁོང་བཙོན་འཇུག་མ་བྱས་གོང་ཁྲིན་ཏུའུ་གྲོང་ཁྱེར་དུ་གློག་བརྙན་བཟོ་སྐྲུན་ཁང་ཞིག་གཉེར་ཡོད་ཀྱང་། རྒྱ་གཞུང་གིས་ཐེངས་མང་ཉེན་བརྡ་བཏང་རྗེས་སྒོ་བརྒྱབ་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
