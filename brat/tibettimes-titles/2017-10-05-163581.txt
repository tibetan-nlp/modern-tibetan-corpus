འཐུས་མི་ ༡༣ གྱི་མིང་ཐོ་རྩིས་མེད་བཏང་།

རྒྱ་ནག་གཞུང་གིས་ཐའེ་ཝན་དང་ཧོང་ཀོང་། ཨོ་མིན་བཅས་ཀྱི་འཐུས་མི་ཚུད་པའི་ཚོགས་ཆེན་ ༡༩ པའི་འཐུས་མི་ ༡༣ གྱི་མིང་ཐོ་རྩིས་མེད་བཏང་ཡོད་འདུག དེ་རིང་རྒྱ་ནག་ལྷོ་རྒྱུད་ཞོགས་པའི་ཆ་འཕྲིན་གྱི་ཞིབ་འཇུག་ལྟར་ན། ཕྱི་ཟླ་ ༩ ཚེས་ ༢༩ ཉིན་རྒྱ་ནག་གཞུང་གིས་ཏང་གི་ཚོགས་ཆེན་ ༡༩ པའི་ཚོགས་ཞུགས་འཐུས་མི་ ༢༢༨༧ གྱི་མིང་ཐོ་གསལ་བསྒྲགས་བྱས་ཡོད་པ་དང་།  སྤྱིར་བཏང་ཚོགས་ཞུགས་པ་ཁྱོན་ ༢༣༠༠ ཡོད་ཀྱང་བྱ་སྤྱོད་ཚད་ལྡན་མིན་པར་ཁ་གཡར་ཏེ་འཐུས་མི་ ༡༣ གྱི་མིང་ཐོ་རྩིས་མེད་བཏང་ཡོད་འདུག རྩིས་མེད་བཏང་བའི་འཐུས་མིའི་ཁོངས་སུ་ཧོང་ཀོང་དང་ཨོ་མིན་ལས་སྒྲུབ་ཁང་གི་འགན་འཛིན་འཐུས་མི་ཀྲུང་ཞོ་མིན་ (Zhang Xiaoming) དང་། ཧོང་ཀོང་ནང་ཡོད་པའི་རྒྱ་ནག་དངུལ་ཁང་གི་འགན་འཛིན་ཡུ་ཡེ་ (Yue Yi) སོགས་ཧོང་ཀོང་དང་ཨོ་མིན། ཐའེ་ཝན་བཅས་ཀྱི་འཐུས་མིའང་ཚུད་ཡོད་འདུག ཨོ་མིན་དང་ཧོང་ཀོང་ཞིབ་འཇུག་ལྷན་ཚོགས་ (CAHKMS) ཀྱི་ཚོགས་གཞོན་ལུའུ་སུའུ་ཀེ་ (Lau Siu-kai) ཞུ་བ་ཞིག་གིས། འཐུས་མི་བཅུ་གསུམ་གྱི་མིང་ཐོ་རྩིས་མེད་བཏང་བ་དེ་ནི། འཐུས་མིའི་སྤྱོད་ལམ་དང་རྣམ་འགྱུར་མ་ལེགས་པའི་གནད་དོན་ཞིག་མིན་པར། དེ་ནི་སྲིད་འཛིན་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱིས་ཏང་གི་སྒྲིག་སྲོལ་ལ་སྔར་ལས་ལྷག་པར་དམ་དྲག་བྱ་ཐབས་ཤིག་ཡིན་པས། འཐུས་མི་དེ་ཚོ་ཚོགས་ཆེན་ཐོག་མཉམ་ཞུགས་ཆོག་པའི་གསལ་བསྒྲགས་སྤེལ་སྲིད་པ་མ་རེད། ཅེས་བརྗོད་འདུག རྒྱ་ནག་དམར་ཤོག་ཚོགས་པའི་ལོ་རྒྱུས་ཞིབ་འཇུག་པ་ཀྲང་ལེ་ཧྥན་ (Zhang Lifan) ཟེར་བ་ཞིག་གིས། འཐུས་མི་དེ་ཚོའི་མིང་ཐོ་གསལ་བསྒྲགས་མ་བྱས་པ་ནི། ཨོ་མིན་དང་ཧོང་ཀོང་སྲིད་འཛིན་ས་ཁུལ་གཉིས་སྔོན་ཆད་དབྱིན་ཇིའི་མི་སེར་སྤེལ་གནས་ཆགས་རྗེས། ཏང་གི་ངོས་ནས་སྲིད་འཛིན་ས་ཁུལ་དེ་གཉིས་ཀྱི་འཐུས་མིའི་གནས་ཚུལ་ཧ་ཅང་གསང་རྒྱ་བྱེད་ཀྱི་ཡོད་པ་དང་། དབྱིན་ཇིའི་གཞུང་གིས་ཧོང་ཀོང་དང་ཨོ་མིན་གྱི་བདག་དབང་རྒྱ་ནག་གཞུང་ལ་རྩིས་སྤྲད་སྐབས། པེ་ཅིན་གཞུང་གིས་ཨོ་མིན་དང་ཧོང་ཀོང་གཞུང་མང་གཉིས་ལ་ཁས་ལེན་ཁག་ཐག་ག་རེ་བྱས་ཡོད་པ་དེ་ལ་འདས་པའི་ལོ་ངོ་ ༥༠ རིང་འགྱུར་བ་གང་ཡང་ཕྱིན་མི་འདུག ཅེས་བཤད་འདུག ལོ་ ༥ རེའི་ནང་ཐེངས་རེར་འཚོགས་བཞིན་པའི་རྒྱ་ནག་མི་མང་འཐུས་མི་ཚོགས་ཆེན་ནི་དམར་ཤོག་ཚོགས་པའི་སྲིད་དབང་ལ་འཕོ་འགྱུར་གཏོང་བའི་སྲིད་དོན་གལ་ཆེ་ཤོས་ཆགས་ཀྱི་ཡོད་སྟབས། འབྲེལ་ཡོད་རྒྱལ་ཁབ་དང་སྲིད་དོན་མི་སྣ་ཚོས་དོ་སྣང་ཤུགས་ཆེན་བྱེད་ཀྱི་ཡོད་པ་དང་། ལྷག་པར་དུ་དམར་ཤོག་ཚོགས་པ་ནང་ཁུལ་གྱི་གནད་ཡོད་དཔོན་རིགས་ཚོས་ཕན་ཚུན་བར་དོགས་ཟོན་དང་ཞིབ་བཤེར་ནན་མོ་བྱེད་ཀྱིན་ཡོད་པ་དང་། ད་ལོའི་ཚོགས་ཆེན་དེ་ཕྱི་ཟླ་ ༡༠ ཚེས་ ༡༨ ཉིན་པེ་ཅིན་གྲོང་ཁྱེར་གྱི་མི་མང་ཚོགས་ཁང་ཆེན་མོའི་ནང་འཚོགས་འཆར་ཡོད་པ་རེད། གསར་འགོད་པ། པདྨ་རྒྱལ།
