ལིན་ཅི་ཧྭ་ལ་ཚེ་བཙོན་གྱི་ཁྲིམས་ཐག་བཅད།

ཉེས་དོན་གསུམ་གྱི་ཐོག་ནས་ཐེན་ཅེན་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་གིས་རྒྱ་ནག་དམར་ཤོག་འཐབ་ཕྱོགས་གཅིག་གྱུར་ཁང་གི་གཙོ་འཛིན་ཟུར་བ་ལིན་ཅི་ཧྭ་ལ་ཚེ་བཙོན་གྱི་ཁྲིམས་ཐག་བཅད་འདུག དེ་རིང་རྒྱ་ནག་ལྷོ་རྒྱུད་ཞོགས་པའི་ཚགས་པར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཁ་སང་རེས་གཟའ་ཟླ་བའི་ཉིན་ཐེན་ཅེན་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་དང་པོས་རྒྱ་ནག་གི་སྲིད་འཛིན་ཟུར་བ་ཧུའུ་ཅིན་ཐའོ་ཡི་ཆབ་སྲིད་སློབ་སྟོན་པ་དང་རྒྱ་ནག་དམར་ཤོག་འཐབ་ཕྱོགས་གཅིག་གྱུར་ཁང་གི་གཙོ་འཛིན་ལིན་ཅི་ཧྭ་ཡིས་དམངས་སྒོར་ས་ཡ་ ༧༧ ལྐོག་ཟ་བྱས་པ་དང་། རྒྱལ་ཁབ་ཀྱི་གསང་བ་རྐུ་ཁྱེར་དང་སྲིད་དབང་ལོག་སྤྱོད་བྱས་པའི་ཉེས་དོན་ཐོག་ཚེ་བཙོན་གྱི་ཁྲིམས་ཐག་བཅད་སྐོར་བསྒྲགས་འདུག ཐེན་ཅེན་འབྲིང་རིམ་མི་མང་ཁྲིམས་དང་པོས་ལིན་ཅི་ཧྭས་ལྐོག་རྔན་སྤྲོད་ལེན་བྱས་པ་མ་ཟད། འདས་པའི་མི་ལོ་ ༥ རིང་རྒྱལ་ཁབ་ཀྱི་གསང་བ་རྐུ་འཁྱེར་དང་འདས་པའི་མི་ལོ་ ༤ རིང་སྲིད་དབང་ལོག་སྤྱོད་བྱས་པའི་ནག་ཉེས་ར་སྤྲོད་ཐོག་ཁོ་ལ་ཚེ་བཙོན་གྱི་ཁྲིམས་ཐག་བཅད་ལུགས་བཤད་ཅིང་། རྒྱ་ནག་གཞུང་འབྲེལ་ཞིན་ཧྭ་གསར་ཁང་གིས་ལིན་ཅི་ཧྭ་དང་ཁོའི་བཟའ་ཟླ་ ཀུའུ་ལི་ཕིན་གཉིས་ལ་ཁྲིམས་བཤེར་བྱེད་སྐབས་ཁོང་གཉིས་ཀྱི་ནག་ཉེས་ར་སྤྲོད་བྱེད་མཁན་དཔང་མི་གཉིས་ཡོད་ཚུལ་བཤད་འདུག ལིན་ཅི་ཧྭ་ནི་ད་སྐབས་ལོ་ ༥༩ ཡིན་པ་དང་། སྔོན་ཆད་ཧུའུ་ཅིན་ཐའོ་ཡི་སྲིད་དབང་སྐབས་སུ་དམར་ཤོག་ཚོགས་པའི་ཆབ་སྲིད་སློབ་སྟོན་པ་ཡིན་ཞིང་། ཕྱི་ལོ་ ༢༠༡༢ ལོ་ནས་ ༢༠༡༤ ལོའི་བར་རྒྱ་ནག་རྒྱལ་ཡོངས་སྲོད་གྲོས་སློབ་སྟོན་ཁང་གི་གཙོ་འཛིན་གཞོན་པ་དང་རྒྱ་ནག་དམར་ཤོག་འཐབ་ཕྱོགས་གཅིག་གྱུར་ཁང་གི་གཙོ་འཛིན་བཅས་ཀྱི་གོ་དབང་བཟུང་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༡༤ ལོའི་ཕྱི་ཟླ་ ༡༢ བའི་ནང་རྒྱ་ནག་དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ཁང་གིས་ལིན་ཅི་ཧྭ་ཡི་རྒྱ་ནག་དམར་ཤོག་འཐབ་ཕྱོགས་གཅིག་གྱུར་ཁང་གི་གོ་དབང་རྩིས་ལེན་དང་དོག་གཞིའི་ཞིབ་བཤེར་བྱེད་འགོ་བཙུགས་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༡༥ ལོའི་ཕྱི་ཟླ་ ༧ ཚེས་ ༢༠ ཉིན་ལིན་ཅི་ཧྭ་རྒྱ་ནག་དམར་ཤོག་ཚོགས་ཁོངས་ནས་ཕྱིར་ཕུད་བཏང་སྟེ། ནག་ཉེས་ཆེན་པོ་གསུམ་གྱི་འོག་ཆེས་མཐོའི་མི་དམངས་ཞིབ་དཔྱོད་ཁང་གིས་ཁྲིམས་བཤེར་བྱས་པ་དང་། ཕྱི་ལོ་ ༢༠༡༦ ཕྱི་ཟླ་ ༥ བར་ཐེན་ཅེན་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་གིས་ཁྲིམས་བཤེར་བྱས་ཏེ་ཁ་སང་ཕྱི་ཚེས་ ༤ ཉིན་ཚེ་བཙོན་གྱི་ཁྲིམས་ཐག་བཅད་ཡོད་པ་རེད། གསར་འགོད་པ། པདྨ་རྒྱལ།
