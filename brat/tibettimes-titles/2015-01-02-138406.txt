མི་ ༣༦ རྡོག་བརྫིས་བཏང་ནས་བསད།

ཕྱི་ལོ་གསར་ཚེས་ཀྱི་སྔ་ནུབ་ཏུ་རྒྱ་ནག་ཧྲེང་ཧེ་གྲོང་ཁྱེར་གྱི་ལོ་གསར་བསུ་བའི་མི་ཚོགས་ཁྲོད་ཉུང་མཐར་མི་ ༣༦ ལ་རྡོག་བརྫིས་བཏང་ནས་བསད་ཡོད་འདུག ཕྱི་ལོའི་ལོ་གསར་གྱི་སྔ་ནུབ་སྟེ་ཕྱི་ཚེས་ ༣༡ ཉིན་གྱི་དགོང་མོ་ཆུ་ཚོད་ ༡༡།༣༠ ཡས་མས་སུ། རྒྱ་ནག་ཧྲེང་ཧེ་གྲོང་ཁྱེར་བད་ཐན་ཁྲིན་ཡས་ཐང་ཆེན་དུ་མི་ཚོགས་ཆེན་པོ་འདུས་ཏེ་ལོ་གསར་བསུ་བའི་དགའ་སྟོན་ལ་རོལ་སྐབས། འཚང་ཁ་བརྒྱབ་སྟེ་ཉུང་མཐར་མི་གྲངས་ ༣༦ ལ་རྡོག་བརྫིས་བཏང་ནས་རྐྱེན་འདས་དང་། མི་ ༤༧ ལ་རྨས་སྐྱོན་ཕོག་སྟེ་མྱུར་སྐྱོབ་བྱེད་བཞིན་ཡོད་ཀྱང་། འདས་པོའི་གྲངས་འབོར་མུ་མཐུད་འཕར་བཞིན་ཡོད་འདུག རྒྱ་ནག་གཞུང་ཕྱོགས་ནས་བསྒྲགས་པ་ལྟར་ན།  འདས་པོ་དང་རྨས་སྐྱོན་ཕོག་མཁན་མང་ཆེ་བ་སློབ་ཆེན་སློབ་མ་ཚུད་པའི་བུད་མེད་དང་བྱིས་པ་ཡིན་ཞིང་། རྐྱེན་འདས་སུ་ཕྱིན་པའི་ནང་སྦྲུམ་མ་གཅིག་དང་མ་ལེ་ཞེ་ཡའི་སློབ་མ་ཞིག་ཡོད་ལ། ད་ལྟའི་ཆར་དོན་རྐྱེན་གྱི་འབྱུང་རྐྱེན་ཐད་ཞིབ་བཤེར་བྱེད་བཞིན་ཡོད་སྐོར་དང་། ཁ་སང་ས་གནས་དེར་འཁྲབ་སྟོན་འདུ་ཚོགས་ཤིག་གོ་སྒྲིག་བྱེད་འཆར་ཡོད་ཀྱང་མེད་པར་བཟོས་ཡོད་འདུག དོན་རྐྱེན་བྱུང་སྐབས་ཡུལ་དངོས་སུ་ཡོད་པའི་གྲོང་མི་ཞིག་གིས། ཉི་ཧོང་ལྟ་བུ་ཞིག་གི་ནང་དུ་གནད་དོན་འདི་འདྲ་ཞིག་བྱུང་བ་ཡིན་ན། མགོ་ཁྲིད་མང་པོས་གཞུང་མང་ལ་དགོངས་དག་ཞུ་བ་དང་སོ་སོའི་བྲང་ཁ་གཤགས་ཏེ་རང་ཤི་རྒྱག་པ་རེད། ཧྲེང་ཧེ་གྲོང་ཁྱེར་གྱི་མགོ་ཁྲིད་ཚོས་དེ་ལྟར་བྱེད་མི་ཐུབ་ཀྱང་ཆེས་མ་མཐའ་ཡང་སྒོ་ཕྱིར་ཐོན་ཏེ་རང་འགན་དང་དུ་ལེན་དགོས་པ་རེད། སྲིད་གཞུང་གིས་བསྒྲགས་པའི་འདས་གྲོངས་ཕྱིན་པའི་གྲངས་འབོར་དེ་བདེན་པ་ཡིན་མིན་མི་ཤེས། སྲིད་གཞུང་གིས་གསང་བ་བྱེད་ཀྱིན་ཡོད་པས་མི་མང་ལ་དོན་དངོས་ཤེས་རྟོགས་བྱེད་པའི་དབང་ཆ་ཡོད་པ་མ་རེད། ཅེས་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
