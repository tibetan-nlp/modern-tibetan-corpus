བོད་མི་མང་གི་སེམས་ལ་རྨ་ཁ་མ་བཟོ།

སུད་སིའི་ནང་བོད་ཀྱི་ཚོགས་སྡེ་ཁག་གི་མཛད་གཙོ་རྣམ་པ་གདན་ཞུའི་ཐོག་དོལ་རྒྱལ་རྗེས་འབྲང་བ་དང་༧གོང་ས་མཆོག་གི་སྐུ་ཕྱྭ་བཅས་ཀྱི་སྐོར་ལ་བགྲོ་གླེང་གནང་ཡོད་འདུག   འཆར་ལོའི་ཟླ་གཉིས་པའི་ནང་༧གོང་ས་མཆོག་སུད་སིར་ཆིབས་སྒྱུར་སྐྱོང་འཆར་ཡོད་པ་དང་བསྟུན་ཏེ། ཡུ་རོབ་སུད་སིའི་ཕྱི་ཚེས་ ༦ ཉིན། བོད་ཀྱི་མི་རབས་གསར་བའི་མཐུན་ཚོགས་ཀྱི་གོ་སྒྲིག་འོག སུད་སི་ས་གནས་ཨོར་ལེ་ཁོན་དུ། དོལ་ཚོགས་ཀྱི་བྱ་སྤྱོད་དང་སྐུའི་སྲུང་སྐྱོབ་སྐོར་ཞེས་པའི་བརྗོད་གཞིའི་ཐོག རི་ཁོན་མཁན་རིན་པོ་ཆེ་མཆོག་དང་ཡུ་རོབ་སྤྱི་འཐུས་བཀྲེས་མཐོང་དབང་པོ་ལགས། བོད་རིགས་མི་མང་གི་ཚོགས་གཙོ་གངས་གཞོན་ཚང་བློ་བཟང་དང་། སྤྱི་ཟུར་གསང་གླིང་ཚེ་རིང་རྡོ་རྗེ་གཙོས་པའི་སྤྱི་ཟུར་རྣམ། སུད་སི་ས་གནས་བོད་ཀྱི་བུད་མེད་ལྷན་ཚོགས་ཀྱི་སྐུ་ཚབ་དང་། ཆོལ་ཀ་གསུམ་གྱི་ཚོགས་གཙོ་རྣམ་པ། འཛམ་གླིང་བོད་མིའི་བདེན་དོན་དང་བླངས་ལས་འགུལ་ཁང་གི་སྐུ་ཚབ་བཅས་ཆེད་དུ་གདན་ཞུ་དང་། གཞན་མི་མང་མཉམ་ཞུགས་ཐོག་གོང་འཁོད་བརྗོད་གཞིའི་ཐོག་བགྲོ་གླེང་གནང་ཡོད་འདུག བགྲོ་གླེང་ཐོག་དོལ་ཚོགས་ཀྱི་ད་ལྟའི་བྱ་སྤྱོད་ཀྱིས་བོད་པའི་འཐབ་རྩོད་དང་ཆིག་སྒྲིལ། ༧གོང་ས་མཆོག་གི་སྐུ་ཕྱྭ་བཅས་ལ་ཉེན་ཁ་བཟོ་བཞིན་པའི་ཐད་ངོས་འཛིན་གདོང་ལེན་ཇི་ལྟར་དགོས་མིན་ཐད་དགོངས་ཚུལ་སྣ་ཚོགས་བརྗོད་འདུག་ཅིང་། དོལ་ཚོགས་ཀྱིས་བྱ་སྤྱོད་འདིས་བོད་མི་རིགས་ཀྱི་རྩ་དོན་ལ་རྡོག་རོལ་གཏོང་གི་ཡོད་པ་མ་ཟད། ཁོ་ཚོས་ཆོས་དད་རང་དབང་ལ་ཁ་གཡར་ནས་རྒྱ་གཞུང་གི་བྱེད་བབ་ལས་སྡུག་པའི་བོད་པའི་ཆོས་དང་རིག་གཞུང་། ཆོལ་གསུམ་མཐུན་སྒྲིལ། བོད་མིའི་རྩ་དོན་བཅས་ལ་གནོད་པ་སྐྱེལ་བཞིན་པ་མ་ཟད། བོད་ཕྱི་ནང་གཉིས་ཀྱི་མི་མང་གི་སེམས་ལ་རྨ་ཁ་ཚད་མེད་བཟོ་བཞིན་ཡོད་སྟབས། དོལ་ཚོགས་ཀྱིས་༧གོང་ས་མཆོག་ལ་གཞི་མེད་སྐྱོན་འཛུགས་ཀྱི་ལས་འགུལ་གང་མགྱོགས་མཚམས་འཇོག་དགོས་སྐོར་མི་མང་ཚོས་བསམ་ཚུལ་ཤུགས་ཆེན་པོ་བྱུང་ཡོད་འདུག གཞན། འཆར་ལོའི་ནང་༧གོང་ས་མཆོག་སུད་སིར་ཆིབས་སྒྱུར་སྐབས་དོལ་ཚོགས་ཀྱིས་ངོ་རྒོལ་ལས་འགུལ་རྩོམ་འཆར་ལ་ང་ཚོས་གདོང་ལེན་ཇི་ལྟར་དགོས་མིན་དང་། སུད་སི་མི་མང་སྒེར་སོ་སོའི་སྟེང་ནས་འགན་ལེན་ཇི་ལྟར་གནང་ཕྱོགས་བཅས་ཀྱི་ཐད་སྒུ་མགྲོན་དང་མི་མང་རྣམས་ཀྱིས་བསམ་ཚུལ་ལྷུག་བརྗོད་གནང་ཡོད་འདུག མཇུག་ཏུ་བཙན་བྱོལ་ནང་བཞུགས་པའི་བོད་པ་ཚོས་ནམ་རྒྱུན་སྐད་འབོད་ཙམ་དང་། ཁྲོམ་སྐོར་ལས་འགུལ་ཙམ་ཁོ་ན་མ་ཡིན་པར་ལག་ལེན་ལ་འཁྱོལ་ཐུབ་ཅིང་དོན་ཕན་ལྡན་པའི་ལས་འགུལ་གཞན་ཇི་ཞིག་སྤེལ་དགོས་མིན་ཐད་སྐུ་མགྲོན་དང་མི་མང་ཚོའི་བར་བསམ་ཚུལ་བརྗེ་རེས་བྱུང་ཡོད་འདུག ཙ་ཡུས་ཕུན་ཚོགས་ལགས་ཀྱིས་ས་གནས་ནས་གནས་ཚུལ་བཏང་འབྱོར་བྱུང་།
