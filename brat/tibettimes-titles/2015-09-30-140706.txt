འབར་གས་དོན་རྐྱེན་ཐེངས་ ༡༣ བྱུང་།

  རྒྱ་ནག་ཀོང་ཞི་ཞིང་ཆེན་ལེའུ་ཀྲུའུ་གྲོང་ཁྱེར་གྱི་ལྟེ་གནས་ཤིག་ཏུ་འབར་གས་དོན་རྐྱེན་ཐེངས་ ༡༣ བྱུང་སྟེ་མི་མང་པོར་ཤི་རྨས་བྱུང་ཡོད་འདུག   དེ་རིང་ཕྱི་དྲོར་ཀོང་ཞི་ཞིང་ཆེན་ལེའུ་ཁྲིན་རྫོང་དང་དེའི་ཉེ་འཁོར་དུ་བསྟུད་མར་འབར་གས་དོན་རྐྱེན་ཐེངས་ ༡༣ བྱུང་ཡོད་པ་དང་། རྒྱ་ནག་གཞུང་འབྲེལ་ཤིན་ཧྭ་གསར་ཁང་གིས།  དོན་རྐྱེན་དེའི་ཁྲོད་དུ་ད་ལྟའི་བར་མི་ ༣ འདས་གྲོངས་སུ་ཕྱིན་པ་དང་མི་ ༡༣ ལ་རྨས་སྐྱོན་ཚབས་ཆེན་ཐེབས་ཡོད་པ་དང་། ད་ལྟའི་ཆར་འབར་གས་ཀྱི་བྱུང་རྐྱེན་ཐད་རྟོག་ཞིབ་བྱེད་བཞིན་ཡོད་སྐོར་བསྒྲགས་འདུག   ཡང་དེ་རིང་ཕྱི་དྲོར་ནུབ་རྒྱལ་གསར་ཤོག་ཅེས་པའི་སྟེང་གནས་ཚུལ་སྤེལ་བའི་ནང་། འབར་གས་ཀྱི་དོན་རྐྱེན་བྱུང་སའི་ས་ཆ་ཁག་་ནི་ལེའུ་ཁྲིན་རྫོང་གི་མཐའ་འཁོར་གྱི་ཚོང་གཉེར་ཁང་དང་། བཙོན་ཁང་། ཏཱ་པོའི་གྲོང་རྡལ་སྲིད་གཞུང་། སྒོ་རྩིས་ཚོང་ཁང་། རླངས་འཁོར་ས་ཚུགས། སྨན་ཁང་། སྲོག་ཆགས་གསོ་སྐྱོང་ཁང་གི་སྡོད་གནས། སྔོ་ཚལ་ཚོང་རྭ། དེ་བཞིན་ནད་ཡམས་སྔོན་འགོག་ལྟེ་གནས་སོགས་ཡིན་སྐོར་བཀོད་འདུག   ལེའུ་ཀྲུའོ་གྲོང་ཁྱེར་ནི་ཀོང་ཞི་ཞིང་ཆེན་གྱི་དཀྱིལ་ཤར་རམ་འབྲུག་གྲོང་དུ་འབོད་པའི་ས་ཁུལ་ཞིག་ཡིན་ཞིང་། དེ་ནི་ཀོང་ཞི་ཞིང་ཆེན་བཟོ་ལས་ཐོན་སྐྱེད་ཀྱི་ལྟེ་གནས་ཆེ་ཤོས་ཤིག་ཡིན་པ་མ་ཟད། ལོ་རྒྱུས་རང་བཞིན་ལྡན་པའི་རིག་གནས་ཀྱི་བསྟི་གནས་ཤིག་ཀྱང་ཡིན་འདུག ད་ལྟའི་བར་ལོ་ངོ་ ༢༡༠༠ ལྷག་གི་ལོ་རྒྱུས་ལྡན་པའི་ལེའུ་ཀྲུའུ་གྲོང་ཁྱེར་ནས་ལེའུ་ཁྲིན་རྫོང་བར་སྤྱི་ལེ་ ༤༠ ལྷག་ཡོད་ཅིང་། ལེའུ་ཁྲིན་རྫོང་གི་ཏཱ་པོའི་གྲོང་རྡལ་ནང་ཡུལ་མི་ཁྲི་ ༤ ལྷག་འཚོ་སྡོད་བྱེད་ཀྱི་ཡོད་འདུག གསར་འགོད་པ། འཇིགས་མེད་རྒྱ་མཚོ།
