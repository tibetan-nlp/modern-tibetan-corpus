བོད་མིའི་སྒྲིག་འཛུཊ་ལ་ཨ་སྒོར་ས་ཡ་ ༡༧
སྤྲོད་པར་བཀའ་འཁྲོལ་གནང་།

ཨ་རིའི་གྲོས་ཚོགས་གོང་འོག་གཉིས་ཀྱི་ (AC) ཞེས་པའི་གྲོས་ཚོགས་ཀྱི་རོགས་དངུལ་གཏོང་ལེན་ཚོགས་ཆུང་ནས་ཕྱི་ལོ་ ༢༠༡༨ ལོའི་ལོ་འཁོར་སྔོན་རྩིས་ནང་བོད་མིའི་སྒྲིག་འཛུགས་ལ་ཨ་སྒོར་ས་ཡ་ ༡༧ སྤྲོད་པར་བཀའ་འཁྲོལ་གནང་ཡོད་འདུག ཁ་སང་རྒྱ་གར་གསར་ཁང་ (The Quint) ཞེས་པས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། བདུན་ཕྲག་སྔོན་མའི་ནང་དུ་ཨ་རིའི་གྲོས་ཚོགས་གོང་འོག་གཉིས་ཀྱི་ (AC) ཞེས་པའི་གྲོས་ཚོགས་ཀྱི་རོགས་དངུལ་གཏོང་ལེན་ཚོགས་ཆུང་གིས། ཨ་རིའི་རོགས་ཚོགས་ (Aid) བརྒྱུད་ནས་ཕྱི་ལོ་ ༢༠༡༨ ལོའི་ལོ་འཁོར་སྔོན་རྩིས་ནང་བོད་མིའི་སྒྲིག་འཛུགས་ལ་རོགས་དངུལ་ཨ་སྒོར་ས་ཡ་ ༡༧ སྤྲོད་པར་བཀའ་འཁྲོལ་གནང་ཡོད་པ་དང་། མཐའ་མའི་ཐག་གཅོད་དེ་ཨ་རིའི་གྲོས་ཚོགས་ནང་བགྲོ་གླེང་གནང་བ་བརྒྱུད་གཏན་འབེབས་གནང་རྒྱུ་ཡིན་འདུག དེ་རིང་སྔ་དྲོར་འདི་གའི་གསར་འགོད་པ་བོད་མིའི་སྒྲིག་འཛུགས་དཔལ་འབྱོར་ལས་ཁུངས་སུ་བསྐྱོད་དེ་དཔལ་འབྱོར་བཀའ་བློན་མཆོག་ལ་གནས་ཚུལ་དེའི་སྐོར་བཀའ་འདི་ཞུས་སྐབས། བཀའ་བློན་མཆོག་གིས་གནས་ཚུལ་དེ་ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༩ ནས་དབུ་འཛུགས་གནང་རྒྱུ་ཡིན་པའི་གྲོས་ཚོགས་ཚོགས་དུས་བཞི་པའི་ཐོག་གསལ་བཤད་གནང་རྒྱུ་ཡིན་སྐོར་གསུངས་བྱུང་། (The Quint) གསར་ཁང་གི་གནས་ཚུལ་ནང་། ཕྱི་ལོ་ ༢༠༡༨ ལོའི་ལོ་འཁོར་སྔོན་རྩིས་ནང་བཀའ་འཁྲོལ་གནང་བའི་ཨ་སྒོར་ས་ཡ་ ༡༧ གྱི་ཁོངས་ནས་ཨ་སྒོར་ས་ཡ་ ༨ བོད་རང་སྐྱོང་ལྗོངས་དང་བོད་རང་སྐྱོང་ས་ཁུལ་ཁག་གི་ཁོར་ཡུག་སྲུང་སྐྱོབ་དང་། བོད་མིའི་ཆོས་དང་རིག་གཞུང་སྲུང་སྐྱོབ་ཀྱི་ཆེད་དུ་ཡིན་པ་དང་། བོད་མིའི་སྒྲིག་འཛུགས་སྲ་བརྟན་ཡོང་ཆེད་ཨ་སྒོར་ས་ཡ་ ༣།  རྒྱ་གར་དང་བལ་ཡུལ་གཙོས་པའི་ལྷོ་ཨེ་ཤ་ཡའི་ནང་གི་གནས་སྡོད་སྐྱབས་བཅོལ་བོད་མིའི་ཤེས་ཡོན་དང་། ཚོང་ལས་དང་རིག་རྩལ་གོང་འཕེལ་གཏོང་ཆེད་ཨ་སྒོར་ས་ཡ་ ༦ གཏོང་འཆར་ཡོད་སྐོར་འཁོད་འདུག གཞན་ཡང་བོད་དོན་འབྲེལ་མཐུད་ཁང་ལ་རོགས་སྐྱོར་ཨ་སྒོར་ས་ཡ་ ༡ དང་། བོད་དོན་དྲིལ་བསྒྲགས་ཆེད་ཨ་སྒོར་ས་ཡ་ ༡༥། དེ་བཞིན་ཨ་རིའི་རླུང་འཕྲིན་ཁང་དང་ཨེ་ཤ་ཡ་རང་དབང་རླུང་འཕྲིན་ཁང་གཉིས་ལ་མུ་མཐུད་རོགས་དངུལ་སྤྲོད་འཆར་ཡོད་སྐོར་དང་། ཕྱི་ལོ་ ༢༠༡༨ ལོའི་རོགས་དངུལ་ཨ་སྒོར་ས་ཡ་ ༡༧ དེ་ཕྱི་ལོ་ ༢༠༡༧ ལོའི་རོགས་དངུལ་གྱི་གྲངས་འབོར་དང་གཅིག་པ་ཆགས་ཡོད་སྐོར་སོགས་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
