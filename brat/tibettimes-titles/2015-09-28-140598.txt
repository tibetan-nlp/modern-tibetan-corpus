བོད་ཁུལ་གྱི་ཆུའི་འཇིག་ཉེན་གླེང་བ།

  རྒྱ་ནག་ནི་ཨེ་ཤ་ཡའི་ཆུ་མཛོད་མང་ཤོས་ཀྱི་རྒྱལ་ཁབ་སྟེ་ཁྱོན་ཆུ་མཛོད་ ༢༥༠༠༠ ལྷག་ཡོད་པས་ལྷོ་ཤར་ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་ ༡༡ ལྷག་ལ་གནོད་སྐྱོན་ཐེབས་ཀྱི་ཡོད་འདུག   ཁ་སང་ཉིན་རྒྱབ་ཆུ་ཚོད་ ༤ སྟེང་བོད་ཀྱི་གསོལ་མགྲོན་ཁང་དུ། བུད་མེད་ལྷན་ཚོགས་དང་བོད་རང་བཙན་སློབ་ཕྲུག་ཚོགས་པ། འགྲོ་བ་མིའི་ཐོབ་ཐང་དང་མང་གཙོ་འཕེལ་རྒྱས་ལྟེ་གནས་ཁང་བཅས་མཉམ་འབྲེལ་གྱི་གོ་སྒྲིག་འོག འཛམ་གླིང་ཆུའི་ཉིན་མོ་ (WWD) སྲུང་གཙི་བྱས་ཏེ། ཨེ་ཤ་ཡའི་ཆུའི་འཇིག་ཉེན་ཞེས་པའི་བརྗོད་གཞིའི་ཐོག་བགྲོ་གླེང་ཚོགས་འདུ་ཞིག་ཚོགས་ཡོད་པ་དང་། སྐུ་མགྲོན་གཙོ་བོ་འགྲོ་བ་མིའི་ཐོབ་ཐང་དང་མང་གཙོ་འཕེལ་རྒྱས་ལྟེ་གནས་ཁང་གི་འགན་འཛིན་ཚེ་རིང་མཚོ་མོ་ལགས་དང་། བོད་མི་མང་སྤྱི་འཐུས་ལས་ཐོག་ཤར་གླིང་ཟླ་སྒྲོན་ལགས། བོད་རང་དབང་སློབ་ཕྲུག་ཚོགས་པའི་ལས་འཆར་འགན་འཛིན་ལྕམ་རྗོར་རྗེ་བཅས་ཀྱི་གཙོས། དྷ་ས་ཁུལ་གྱི་མི་མང་དང་ཕྱི་རྒྱལ་བ་སོགས་མི་གྲངས་ ༥༠ ལྷག་མཉམ་འཛོམས་འདུག   བུད་མེད་ལྷན་ཚོགས་ཀྱི་ཁོངས་གཏོགས་ཁོར་ཡུག་ཡར་རྒྱས་ཚོགས་པའི་ལས་བྱེད་ནོར་འཛིན་སྒྲོལ་མ་ལགས་ཀྱིས་ཐེངས་འདིའི་བགྲོ་གླེང་གི་དགོས་དམིགས་ངོ་སྤྲོད་གནང་སྐབས། བོད་ནི་ཨེ་ཤ་ཡའི་གཙང་ཆུ་དང་གཏེར་རྫས། འཁྱག་རོམ་སོགས་ཀྱི་ཐོན་ཁུངས་གཙོ་བོ་ཡིན་སྟབས། རྒྱ་ནག་གཞུང་གིས་མུ་མཐུད་བོད་ནང་གི་རང་བྱུང་ཐོན་ཁུངས་ཁག་སྔོག་འདོན་དང་། ལྷག་པར་དུ་བོད་ཁུལ་གྱི་ཆུ་བོ་དང་གཙང་བོ་བཀག་བསྐྱིལ། ཆུ་རགས་བརྩིགས་པ་དང་གློག་འདོན་ས་ཚིགས་བསྐྲུན་པ། དེ་ལ་བརྟེན་ནས་བོད་ཁུལ་གྱི་འབྲོག་སྡེ་དང་གྲོང་སྡེ་མང་པོ་བཙན་གྱིས་གནས་སྤོ་རུ་འཇུག་པ་སོགས་རང་བྱུང་ཁོར་ཡུག་ལ་གནོད་སྐྱོན་བཟོ་བ་དང་། བོད་མི་ཚོའི་འཚོ་གནས་ཉམ་ཉེས་སུ་གཏོང་བཞིན་པར་གཞིས་ལུས་རྒྱ་ཆེའི་མི་མང་གིས་འགོག་རྒོལ་གནང་མུས་ཡིན་པ་ལྟར། ང་ཚོས་བོད་ནང་བོད་མི་ཚོར་འཕྲད་བཞིན་པའི་ཛ་དྲག་གི་གནས་སྟངས་འདིའི་ཐད་ལ་བགྲོ་གླེང་དང་གྲོས་སྡུར་བྱེད་དགོས་ཡོད་སྟབས། དེ་རིང་གོང་ཞུས་ཚོགས་པ་གསུམ་མཉམ་འབྲེལ་གྱིས་བགྲོ་གླེང་འཚོགས་འདུ་འདི་གོ་སྒྲིག་ཞུས་པ་ཡིན་ལ། སྤྱིར་བཏང་གཙང་ཆུ་བཀག་བསྐྱིལ་དང་ཆུ་རག་འཛུགས་སྐྲུན་སོགས་ཀྱི་བཀའ་རྙོག་འདི་དག་བོད་ཙམ་མ་ཟད་ལྷོ་ཤར་ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་ ༡༡ ལྷག་ལ་འཕྲད་ཀྱི་ཡོད་པ་རེད། ཞེས་བརྗོད་བྱུང་།   དེའི་རྗེས་འགྲོ་བ་མིའི་ཐོབ་ཐང་དང་མང་གཙོ་འཕེལ་རྒྱས་ལྟེ་གནས་ཁང་གི་འགན་འཛིན་ཚེ་རིང་མཚོ་མོ་ལགས་ཀྱིས། དེང་སྐབས་འཛམ་གླིང་ཤར་ནུབ་གང་སར་འབྱུང་བཞིན་པའི་གཙང་ཆུའི་འཇིག་ཉེན་གྱི་དཀའ་རྙོག་འདི་དག་བོད་ནང་གི་ཁོར་ཡུག་གནས་སྟངས་དང་འབྲེལ་བ་ཆེན་པོ་ཡོད། རྒྱ་གཞུང་གིས་བོད་ཁུལ་དུ་ཆུ་མཛོད་དང་ཆུ་རགས་མང་པོ་བརྩིགས་ཏེ། འདབས་འབྲེལ་རྒྱལ་ཁབ་ཁག་ནང་ནུས་ཤུགས་མཁོ་འདྲེན་གྱི་ཁེ་ལས་གཉེར་བཞིན་ཡོད་པ་ཚང་མས་མཁྱེན་གསལ་ལྟར། ད་ལྟའི་བར་རྒྱ་ནག་གཞུང་གིས་བོད་དང་རྒྱ་ནག དེ་བཞིན་ལྷོ་ཤར་ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་ཁག་ནང་མི་མང་གི་འདོད་མོས་དང་བདེ་སྡུག ཁོར་ཡུག་གི་གནས་སྟངས་སོགས་ལ་བསམ་བློ་མི་གཏོང་བར་འཛུགས་སྐྲུན་གྱི་ཁེ་ལས་ཁོ་ན་གཉེར་གྱི་ཡོད་པར་བརྟེན། དེང་སྐབས་བོད་ནང་གི་ཁོར་ཡུག་ཏུ་ལོ་མང་པོར་ཐན་སྐྱོན་བསྟུད་མུར་འབྱུང་བ་དང་སེམས་ཅན་མང་པོར་ཤི་རྨས་ཚབས་ཆེན་ཐེབས་ཀྱི་ཡིན་པ་རེད། དེར་བརྟེན་ད་ལྟའི་ཁོར་ཡུག་གི་གནས་སྟངས་དང་མི་མང་གི་བདེ་སྡུག་གཙོར་བཟུང་ནས། བོད་ནང་གི་མཚེའུ་དང་གཙང་ཆུ་སོགས་བེད་སྤྱོད་ལེགས་པོར་གཏང་བ་ཡིན་ན། འཚོ་གནས་ཀྱི་ཉམ་ཉེས་འདི་དག་སྔོན་འགོག་ཐུབ་ཀྱི་ཡོད། སྤྱིར་བཏང་ལྷོ་ཤར་ཨེ་ཤ་ཡའི་གཙང་ཆུའི་འབྱུང་ཁུངས་གཙོ་བོ་བོད་ཁུལ་ཆགས་ཀྱི་ཡོད་པ་དང་། དེ་ལ་རྒྱ་ནག་གི་ས་གཤིས་ཞིབ་འཇུག་པ་ཧྥན་ཞའོ་ཟེར་བ་ཞིག་གིས“ལྷོ་ཤར་ཨེ་ཤ་ཡའི་གཙང་ཆུའི་འབྱུང་ཁུངས”ཞེས་མིང་བཏགས་ཡོད་པ་རེད། རྒྱ་གཞུང་གིས་བོད་ཁུལ་གྱི་གཙང་ཆུ་མང་པོ་བཀག་བསྐྱིལ་དང་ཆུ་རགས་འབོར་ཆེན་འཛུགས་བསྐྲུན་བྱེད་ཆེད། རྨ་ཆུ་དང་འབྲི་ཆུ། གཙང་ཆུ་སོགས་ཀྱི་འགྲམ་རྒྱུད་དུ་འཚོ་བའི་འབྲོག་སྡེ་དང་གྲོང་སྡེ་སྟོང་ཕྲག་མང་པོའི་རྩ་ས་འཕྲོག་ནས་བཙན་གྱིས་གནས་སྦོ་རུ་འཇུག་གི་ཡོད་པ་དང་། ཉེ་ལམ་ནས་ཡུལ་ཤུལ་ཁུལ་རྫོང་ཁག་ ༡༦ ལྷག་ཏུ་བཅད་བགོས་བྱས་པའི་ཁྲོད་ཀྱི་ཡུལ་མི་བརྒྱ་ཆའི་ ༩༠ བོད་མི་ཡིན་པ་རེད། དེ་ལྟར་བོད་ཁུལ་གྱི་འབྲོག་སྡེ་དང་གྲོང་སྡེ་མང་པོ་གནས་སྤོ་རུ་བཅུག་སྟེ། བོད་མི་ཚོའི་འཚོ་གནས་ཀྱི་གནས་སྟངས་ཇེ་སྡུག་ཏུ་ཕྱིན་ནས་དཀའ་སྡུག་ཟློག་མེད་ལ་འཕྲད་ཀྱི་ཡོད་པ་རེད་ཅེས་དང་།   ཡང་ཁོང་མོས་རྒྱ་ནག་མི་མང་སྤྱི་མཐུན་རྒྱལ་ཁབ་ནང་ལོགས་སུ་གངས་དང་འཁྱག་རོམ་སྟོང་ཕྲག་ ༤༦ ཡོད་པའི་གྲས་ཀྱི་བརྒྱ་ཆའི་ ༥༠ བོད་ཁུལ་དུ་ཡོད་པ་རེད། དེ་བས་ཉེན་ཁ་ཆེས་ཤོས་དེ་བོད་ཁུལ་གྱི་གཙང་ཆུའི་འབྱུང་ཁུངས་གཙོ་བོ་གངས་དང་འཁྱག་རོམ་ཡིན་སྟབས། དེང་སྐབས་བོད་ཁུལ་གྱི་གངས་དང་འཁྱག་རོམ་མང་པོ་ཞིག་ལ་དྲོད་གཤེར་རྒྱས་ཀྱི་ཡོད་པ་དང་། བོད་ཁུལ་གྱི་གཙང་ཆུ་མང་པོ་རྒྱ་གར་དང་པ་ཀི་སི་ཐན། བྷང་ལ་དྷེ་ཤི་སོགས་ལྷོ་ཤར་ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་ཁག་ནང་བཞུར་གྱི་ཡོད་སྟབས། ལྷོ་ཤར་ཨེ་ཤ་ཡའི་མི་མང་ས་ཡ་ ༡༠ ལྷག་གི་འཐུང་ཆུའི་ཐོན་ཁུངས་གཙོ་བོ་དེ་བོད་མཐོ་སྒང་གི་གངས་དང་འཁྱག་རོམ་ཡིན་པ་རེད། དེར་མ་ཟད་ལྷོ་ཤར་ཨེ་ཤ་ཡའི་ནང་ཆར་ཆུའི་མང་ཉུང་དེ་ཡང་བོད་ཁུལ་གྱི་གངས་རིའི་མང་ཉུང་ལ་རག་ལས་ཡོད། བོད་ཁུལ་གྱི་གངས་དང་འཁྱག་རོམ་ཉུང་དུ་ཕྱིན་པ་ཡིན་ན། ལྷོ་ཤར་ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་ཁག་ཅིག་ནང་ཆར་ཆུ་ཉུང་དུ་འགྲོ་གི་ཡོད་པ་རེད། སྤྱིར་བཏང་གཙང་ཆུ་བྲི་བཞིན་པའི་གནད་དོན་གཙོ་བོ་དེ་རང་བྱུང་ཁམས་ལ་གནོད་སྐྱོན་ཐེབས་པའི་རྐྱེན་གྱིས་ཡིན་ཞིང་། བོད་ཁུལ་གྱི་གཙང་ཆུ་མང་པོ་བཀག་བསྐྱིལ་བྱས་པའི་རྐྱེན་གྱིས། ཆུ་མང་ཤོས་ས་འོག་ཏུ་ཐིམ་པ་དང་ས་སྟེང་སྐམ་ནས་བྱེ་ཐང་དུ་འགྱུར་གྱི་ཡོད་ལ། ཆུ་ནང་གི་སྲོག་ཆགས་མང་པོར་ཤི་རྨས་ཚབས་ཆེན་ཐེབས་ཀྱི་ཡོད་པ་རེད། ཅེས་བརྗོད་སོང་།   ཡང་ཁོང་མོས། རྒྱ་ནག་གི་སྤྱི་ཚོགས་ཚན་རིག་ཁང་ནས་བཏོན་པའི་སྙན་ཐོ་ཞིག་གི་ནང་། ལོ་རེར་བོད་ནང་གི་གངས་འཁྱགས་ཀྱི་ཞུ་ཚད་བརྒྱ་ཆའི་ ༧ ཟིན་གྱི་ཡོད་སྐོར་བཀོད་འདུག གལ་ཏེ་ལོ་རེར་གངས་འཁྱགས་ཀྱི་ཞུ་ཚད་བརྒྱ་ཆའི་ ༧ ཟིན་པ་ཡིན་ན། ཕྱི་ལོ་ ༢༠༥༠ ལོའི་བར་བོད་ཁུལ་གྱི་གངས་དང་འཁྱག་རོམ་གྱི་གསུམ་ཆའི་ ༢ ལྷག་མར་བྲི་བའི་ཉེན་ཁ་འབྱུང་གི་ཡོད་པ་དང་། ཡང་ཕྱི་ལོ་ ༢༠༡༠ ཨ་རིའི་གཞུང་གིས་བཏོན་པའི་སྙན་ཐོའི་ནང་། འདས་པའི་མི་ལོ་ ༤༠ རིང་ལ་བོད་ནང་གི་གངས་དང་འཁྱག་རོམ་ཞུ་ཚད་བརྒྱ་ཆའི་ ༢༠ ཟིན་ཀྱི་ཡོད་སྐོར་བརྗོད་འདུག དེས་ན་ཕྱིས་འབྱུང་གི་ལོ་ངོ་ ༤༠ ནང་ལ་བོད་ཁུལ་གྱི་གངས་དང་འཁྱག་རོམ་ཞུ་ཚད་བརྒྱ་ཆའི་ ༤༠ ནས་ ༦༠ ཟིན་གྱི་ཡོད་པ་དང་། ལྷག་པར་དུ་རྒྱ་ནག་གཞུང་གི་ལོ་གཉིས་ཀྱི་འཆར་གཞི་སྟེ། ཕྱི་ལོ་ ༢༠༡༡ ལོ་ནས་ཕྱི་ལོ་ ༢༠༡༥ བར་གཙང་ཆུ་བཀག་བསྐྱིལ་དང་ཆུ་རགས་འཛུགས་སྐྲུན་གྱི་ལས་འཆར་མུ་མཐུད་ལག་བསྟར་བྱེད་བཞིན་པའི་ནང་། བོད་ཁུལ་གྱི་ཡར་ལུང་གཙང་བོ་དང་རྒྱལ་མོ་རྔུལ་ཆུ། རྨ་ཆུ་དང་འབྲི་ཆུ། རྫ་ཆུ། བ་ཡན་ཧ་རི་སོགས་ཡོད་ཅིང་། ཕྱི་ལོ་ ༢༠༠༥ ལོར་རྒྱལ་མོ་རྔུལ་ཆུའི་ཁར་ཆུ་རགས་ཆེན་པོ་ ༡༣ བརྩིག་པའི་འཆར་གཞིའི་ནང་ནས་ཆུ་རགས་ཆེན་པོ་ ༥ འདི་ལོར་འགྲུབ་ཏུ་འཇུག་འཆར་ཡོད་པ་མ་ཟད། རྒྱ་ནག་གཞུང་གིས་ ༢༠༢༠ ལོའི་བར་དུ་ཆུ་རགས་འཛུགས་སྐྲུན་གྱི་ལས་འཆར་ཡོངས་སུ་འགྲུབ་ཐུབ་བྱ་འཆར་ཡོད་པ་དང་། རྒྱ་ནག་གཞུང་གིས་ཟམ་པ་དང་གཞུང་ལམ། ལྷག་དོན་རྫས་དུད་མང་པའི་བཟོ་གྲྭ་འབོར་ཆེན་རྒྱག་གི་ཡོད་སྟབས། ཁོར་ཡུག་དང་མཁའ་དབུགས་སྦག་བཙོག་ཆགས་པའི་ཛ་དྲག་གི་དཀའ་ངལ་མང་པོ་ཡོང་གི་ཡོད་པ་རེད། ཅེས་བརྗོད་བྱུང་།   བོད་རང་བཙན་སློབ་ཕྲུག་ཚོགས་པའི་ལས་འཆར་འགན་འཛིན་ལྕམ་རྗོར་རྗེ་ལགས་ཀྱིས། ཕྱི་ལོ་ ༢༠༠༠ ལོ་ནས་ངས་བོད་ཁུལ་གྱི་གཙང་ཆུའི་སྐོར་ལ་ཞིབ་འཇུག་བྱེད་འགོ་བརྩམས་པའི་ནང་། རྒྱ་ནག་གཞུང་གིས་ཡར་ལུང་གཙང་བོ་སོགས་བོད་ཁུལ་གྱི་གཙང་ཆུ་གཙོ་བོ་ཁག་བཀག་བསྐྱིལ་བྱས་ཡོད་པ་རེད། སྤྱིར་བཏང་ཨེ་ཤ་ཡའི་མི་མང་ས་ཡ་ ༢ ལྷག་གི་ཆུའི་འཚོ་རྟེན་གཙོ་བོ་བོད་ཁུལ་ནས་བཞུར་ཡོང་བའི་གཙང་ཆུ་ལ་བརྟེན་ཡོད་སྟབས། མ་འོངས་པར་ཁོང་ཚོའི་འཐུང་ཆུའི་འཚོ་རྟེན་ཉེན་ཁར་ཐུག་ཅིང་། རྒྱ་ནག་གཞུང་གིས་འདོད་རྔམ་ཆེན་པོས་བོད་ཁུལ་གྱི་གཙང་ཆུ་བཀག་བསྐྱིལ་དང་ཆུ་རགས་འབོར་ཆེན་བརྩིགས་ཏེ། བོད་ནས་བཞུར་ཡོང་བའི་གཙང་ཆུར་བརྟེན་པའི་རྒྱ་གར་དང་པ་ཀི་སི་ཐན། བྷང་ལ་དྷེ་ཤི་སོགས་ལྷོ་ཤར་ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་ཁག་གི་འཚོ་གནས་ལ་འཇིག་ཉེན་བཟོ་བཞིན་ཡོད་པ་རེད། ཅེས་ཟེར།   མཇུག་ཏུ་བོད་མི་མང་སྤྱི་འཐུས་ལས་ཐོག་ཤར་གླིང་ཟླ་སྒྲོན་ལགས་ཀྱིས། བོད་ཁུལ་གྱི་གཙང་ཆུ་དག་ལྷོ་ཤར་ཨེ་ཤ་ཡའི་སྲོག་ལྟ་བུ་ཆགས་ཡོད། རྒྱ་ནག་ཚུད་པའི་ཨེ་ཤ་ཡའི་མི་འབོར་མང་ཤོས་ཀྱི་རྒྱལ་ཁབ་ ༡༡ ལྷག་བོད་ཁུལ་ནས་ཐོན་པའི་གཙང་ཆུ་ལ་བརྟེན་ཡོད་ཀྱང་། ཆུའི་དཀའ་ངལ་དང་བཙོག་ཆུ་ལས་བྱུང་བའི་ནད་ཡམས་བཅས་ཀྱི་དཀའ་རྙོག་ཆེ་ཤོས་དེ་ཡང་ང་ཚོ་ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་ཁག་ཆགས་ཡོད་པ་རེད། དེར་བརྟེན་མ་འོངས་པར་དམག་ཆེན་གསུམ་པའམ་བཞི་བ་ཞིག་འབྱུང་བའི་དབང་དུ་བཏང་ན། དེ་ཨེ་ཤ་ཡའི་ཆུའི་རྩོད་རྙོག་གི་སྟེང་ནས་འབྱུང་གི་རེད་ལ། སྤྱིར་བཏང་ཆུ་རགས་དང་ཆུ་མཛོད་རྒྱག་མཁན་རྒྱ་ནག་གཅིག་པུ་མ་ཡིན་པར། རྒྱ་གར་ནང་ལོགས་སུའང་རྒྱག་གི་ཡོད་པ་རེད། གང་ལྟར་མ་འོངས་པར་ཆུ་རགས་དང་ཆུ་མཛོད་ག་ཚོད་ཅིག་བརྒྱབ་པ་ཡིན་ན། གཙང་ཆུར་ཚ་དྲོད་རྒྱས་པ་དང་མིའི་འཐུང་ཆུ་གཙང་མ་ཉུང་དུ་འགྲོ་གི་ཡོད་པ་རེད། རྒྱ་ནག་ལ་ཆ་མཚོན་ན། ཕྱི་ལོ་ ༡༩༥༩ ལོར་ཆུ་མཛོད་ཁྱོན་ ༤༠༠༠ ཙམ་མ་གཏོགས་མེད་ཀྱང་། ད་ཆར་རྒྱ་ནག་ནང་ཆུ་མཛོད་སྟོང་ཕྲག་ ༢༥ ལྷག་ཡོད་དུས། ལོ་རེར་ཆུ་མཛོད་ ༤༠༠༠ རྒྱག་གི་ཡོད་པ་དང་། ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་ཁག་ནང་ཆུ་མཛོད་མང་ཤོས་དེ་རྒྱ་ནག་ཆགས་བསྡད་ཡོད། དཔེར་ན་ཡར་ལུང་གཙང་བོ་གཅིག་པུར་ཆུ་མཛོད་ ༢༤ བརྒྱབ་ཡོད་སྟབས། དེའི་ཉེན་ཁ་ང་ཚོ་ཚང་མས་ཤེས་གསལ་རེད། ཅེས་བརྗོད་སོང་།   གསར་འགོད་པ། པདྨ་རྒྱལ།
