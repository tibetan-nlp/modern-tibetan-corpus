གྲྭ་ཤག་ ༦༠༠ ཙམ་གཏོར་ཏེ་གྲྭ་བཙུན་ ༡༠༠
ལྷག་ཕྱིར་ཕུད་བཏང་།

གསེར་རྟ་བླ་རུང་སྒར་གྱི་གྲྭ་ཤག་ ༦༠༠ མ་ཟིན་ཙམ་གཏོར་ཏེ་བོད་རང་སྐྱོང་ལྗོངས་ནས་ཡིན་པའི་གྲྭ་བཙུན་ ༡༡༠ ཉིན་ཤས་སྔོན་བཙན་གྱིས་ཕྱིར་ལོག་བཏང་ཡོད་འདུག   འདི་གར་བོད་མི་ཞིག་གིས་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། ཕྱི་ཟླ་ ༩ ཚེས་ ༢༩ ཉིན་གྱི་ཕྱི་དྲོར། རྒྱ་ནག་གཞུང་གིས་གསེར་རྟ་བླ་རུང་སྒར་དུ་སློབ་གཉེར་གནང་བཞིན་པའི་བོད་རང་སྐྱོང་ལྗོངས་ཁོངས་ནས་ཡིན་པའི་དགེ་འདུན་པ་ ༤༠ དང་བཙུན་མ་ ༧༠ བཙན་གྱིས་ཕྱིར་ལོག་བཏང་ཡོད་པ་མ་ཟད། དེ་བཞིན་ཞིང་ཆེན་དང་ཁུལ་གཞན་ནས་ཡོང་བའི་གྲྭ་བཙུན་མང་པོ་ཞིག་ཀྱང་མི་འགྱང་བར་བླ་རུང་སྒར་ནས་བཙན་གྱིས་ཕྱིར་ལོག་གཏང་འཆར་ཡོད་པའི་གནས་ཚུལ་ཐོན་ཡོད་འདུག ས་གནས་རང་ནས་ཡིན་པའི་བོད་མི་ཞིག་གིས། བོད་རང་སྐྱོང་ལྗོངས་ནས་ཡིན་པའི་གྲྭ་བཙུན་ཚོ་བཙན་གྱིས་ཕྱིར་ཕུད་གཏོང་སྐབས། ཤུལ་དུ་ལྷག་པའི་གྲྭ་བཙུན་ཚོས་ཕྱིར་ཕུད་བཏང་བའི་འཛིན་གྲོགས་ཚོའི་བསླབ་པ་སློབ་གཉེར་ལ་བར་ཆད་མི་འབྱུང་བའི་སྨོན་འདུན་དང་། དོན་མེད་ཀྱི་རྡུང་རྡེག་དང་མནར་གཅོད་ཉུང་བའི་གསོལ་འདེབས་བཅས་སྐྱོ་གདུང་གི་མིག་ཆུའི་ཁྲོད་ནས་སྐྱེལ་མ་བྱས་ཡོད་སྐོར་བརྗོད་འདུག ཡང་གནས་ཚུལ་མཁོ་སྤྲོད་པས། འདི་ལོ་ཕྱི་ཟླ་ ༧ པའི་ཚེས་ ༢༠ ནས་བཟུང་སྟེ་རྒྱ་ནག་གཞུང་གིས་གསེར་རྟ་བླ་རུང་སྒར་གྱི་གྲྭ་ཤག་མང་པོ་བཙན་གྱིས་གཏོར་མགོ་བཙུགས་ཡོད་པ་ལྟར། ད་ལྟའི་བར་ཁྱོན་གྲྭ་ཤག་ ༦༠༠ མ་ཟིན་ཞིག་གཏོར་ཚར་ཡོད་པ་དང་། མུ་མཐུད་ལས་མི་མང་པོ་ཆེད་གཏོང་གིས་གཏོར་བཤིག་གཏོང་བཞིན་ཡོད་ཀྱང་། གྲྭ་བཙུན་རྣམས་པས་བླ་མའི་ཞལ་གདམས་སེམས་ལ་བཟུང་སྟེ། མིག་ཆུ་བཙན་གྱིས་མིད་དེ་མིག་རྒྱང་རིང་སར་བལྟས་ནས་དབང་ཡོད་བཙན་གནོན་ལ་བཟོད་སྒོམ་གནང་བཞིན་ཡོད་པ་དང་། ཕྱིར་ཕུད་བཏང་བའི་གྲྭ་བཙུན་རྣམས་ཕྱིར་རང་ཡུལ་དུ་ལོག་རྗེས་སོ་སོའི་དགོན་པ་ནང་མུ་མཐུད་སློབ་གཉེར་གནང་ཆོག་མིན་ད་ལྟ་ཁ་གསལ་མེད་སྐོར་བརྗོད་བྱུང་། གསེར་རྟ་བླ་རུང་སྒར་གཏོར་བཤིག་གཏོང་བའི་ལས་གཞི་དེ་སྐབས་ ༦ པའི་བོད་ཁུལ་ལས་སྒྲུབ་གྲོས་མོལ་ཚོགས་འདུ་དང་རྒྱལ་ཡོངས་ཆོས་ལུགས་ལས་དོན་ཚོགས་འདུ་ཟེར་བ་གཉིས་ཀྱི་ཐོག་གཏན་འབེབས་བྱུང་སྟེ་འདི་ལོའི་ཕྱི་ཟླ་ ༧ ཚེས་ ༢༠ ཉིན་ནས་བཟུང་དངོས་སུ་ལག་བསྟར་བྱས་ཡོད་པ་དང་། སྐབས་དེར་དོན་ཚན་ ༨ ཅན་གྱི་བཀའ་ཁྱབ་ཡི་གེ་ཞིག་བཏང་བའི་ནང་། དགོན་པའི་གྲྭ་པ་དང་བཙུན་མ། དགེ་བསྙེན་བཅས་གྲངས་འབོར་ཉུང་འཕྲིད་བྱ་རྒྱུ་ཡིན་པ་དང་། ཕྱི་ལོ་ ༢༠༡༦ ལོར་གྲྭ་བཙུན་གྲངས་འབོར་ ༢༢༠༠ ཕྱིར་འབུད་གཏོང་རྒྱུ་ཡིན་འདུག དེའི་ནང་དགེ་བསྙེན་གྲངས་ ༡༠༢༩ དགོན་པ་ལས་ལོགས་སུ་བགར་རྒྱུ། གྲྭ་པ་དང་བཙུན་མ་གྲངས་ ༡༢༠༠ དགོན་པ་ནས་ཕྱིར་འབུད་གཏོང་རྒྱུ་དང་། དེའི་ཁྲོད་དུ་ཞིང་ཆེན་གཞན་ནས་ཡིན་པའི་གྲྭ་བཙུན་གྲངས་ ༦༠༠ ཚུད་ཡོད་པ་དང་། ད་དུང་ཕྱི་ལོ་ ༢༠༡༧ ལོར་དགོན་པའི་གྲྭ་བཙུན་སོགས་ཀྱི་ཁྱོན་བསྡོམས་མི་གྲངས་ ༥༠༠༠ ལ་ཚོད་འཛིན་བྱ་དགོས་ཀྱི་ཡོད་འདུག གལ་སྲིད། ཕྱི་ལོ་ ༢༠༡༧ ཟླ་ ༩ ཚེས་ ༣༠ སྔོན་ལ་མི་གྲངས་ ༥༠༠༠ དུ་ཚོད་འཛིན་མ་ཐུབ་པ་ཡིན་ན། ༢༠༡༧ ལོར་དགོན་པའི་མི་གྲངས་ཀྱི་མིང་ལྟར་ཕྱིར་འབུད་གྲྭ་བཙུན་གྱི་མི་གྲངས་ཇེ་མང་དུ་གཏོང་རྒྱུ་ཡིན་པ་མ་ཟད། འདི་ལོར་ཕྱིར་འབུད་བྱས་པའི་གྲྭ་པ་དང་བཙུན་མའི་མིང་ཐོ་རྣམས། ཕྱི་ལོ་ ༢༠༡༦ ཟླ་ ༦ ཚེས་ ༡༥ སྔོན་ལ་གོང་རིམ་སྲིད་གཞུང་ལ་རྩིས་སྤྲོད་ཐོག ཕྱི་ཟླ་ ༡༠ ཚེས་ ༣༠ སྔོན་ལ་ཕྱིར་འབུད་བཏང་ཚར་དགོས་པ་དང་། དགོན་པར་གཏན་སྡོད་ཚད་གཞིའི་མི་གྲངས་ ༥༠༠༠ ཡི་མིང་ཐོ་གོང་རིམ་ལ་སྤྲོད་དགོས་པ་དང་། དེའི་ནང་ཞིང་ཆེན་གཞན་གྱི་གྲྭ་བཙུན་གྲངས་ ༡༠༠༠ ལས་བརྒལ་མི་ཆོག་པ་རེད། དེ་ལས་གཞན་པའི་དགོན་པར་གནས་སྡོད་མི་སྣ་རྣམས་ཀྱི་མིང་ཐོ་ཞིབ་བཤེར་གྱིས་ཕྱི་ ལོ་ ༢༠༡༦ ཟླ་ ༨ ཚེས་ ༣༠ སྔོན་ལ་གོང་རིམ་ལ་སྤྲོད་དགོས་ཀྱི་ཡོད་སྐོར་བཀོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
