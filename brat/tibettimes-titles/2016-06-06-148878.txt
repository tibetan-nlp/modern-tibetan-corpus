འདི་ལོར་བླ་རུང་སྒར་ནས་གྲྭ་བཙུན་ ༡༢༠༠
འཕུད་རྒྱུ་ཡིན།

རྒྱ་གཞུང་གིས་གསེར་རྟ་བླ་རུང་སྒར་གྱི་སྤྱིའི་མི་འབོར་ལ་ཚོད་འཛིན་བྱས་ཏེ་འདི་ལོ་ཁོ་ནར་ཞིང་ཆེན་གཞན་ནས་ཡིན་པ་ ༦༠༠ བཅས་ཁྱོན་གྲྭ་བཙུན་ ༡༢༠༠ ཕྱིར་འཕུད་གཏོང་རྒྱུ་དང་ཁང་པ་ ༡༥༠༠ གཏོར་རྒྱུ་ཡིན་འདུག དེ་རིང་སྤྱི་ཚོགས་དྲ་ལམ་དུ་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། སྐབས་ ༦ པའི་བོད་ཁུལ་ལས་སྒྲུབ་གྲོས་མོལ་ཚོགས་འདུ་དང་རྒྱལ་ཡོངས་ཆོས་ལུགས་ལས་དོན་ཚོགས་འདུ་གཉིས་ཀྱི་ཐོག་གཏན་འབེབས་བྱུང་བ་ལྟར། གསེར་རྟ་བླ་རུང་སྒར་དག་སྒྲིག་བཅོས་སྐྱོང་ཟེར་བའི་བྱ་འགུལ་འོག དཀར་མཛེས་ཁུལ་འཐབ་ཕྱོགས་གཅིག་གྱུར་ཁང་དང་ཁུལ་མི་རིགས་ཆོས་ལུགས་ལས་དོན་ཁང་། ཁུལ་སློབ་གསོ་ཁང་། ཁུལ་ཉེན་རྟོག་ལས་ཁུངས།  ཁུལ་གྲོང་སྡེ་འཛུགས་སྐྲུན་ལས་ཁུངས། གསེར་རྟ་རྫོང་སྲིད་གཞུང་དང་སྒར་གྱི་འབྲེལ་ཡོད་མི་སྣ་བཅས་ཀྱི་མཉམ་འབྲེལ་འགན་ལེན་བྱས་ཏེ། ཕྱི་ལོ་ ༢༠༡༧ ཟླ་ ༩ ཚེས་ ༣༠ ཉིན་ནང་ཚུན་དུ་བླ་རུང་སྒར་གྱི་གྲྭ་བཙུན་སྤྱིའི་གྲངས་འབོར་ ༥༠༠༠ ལ་ཚོད་འཛིན་བྱས་ཏེ། འདི་ལོ་ཁོ་ནར་གྲྭ་བཙུན་ ༡༢༠༠ ཕྱིར་འཕུད་གཏོང་རྒྱུ་དང་། དེའི་ནང་ཉུང་མཐར་ཞིང་ཆེན་གཞན་ནས་ཡིན་པའི་མི་ ༦༠༠ ཡོད་དགོས་པ། དེ་བཞིན། ཕྱི་ལོ་ ༢༠༡༣ ནས་ད་ལྟའི་བར་ཕྱིར་འཕུད་བཏང་བའི་གྲྭ་བཙུན་གྱི་ཁང་པ་དང་སྒྲིག་འགལ་གྱིས་གསར་རྒྱག་བྱས་པའི་ཁང་པ་སོགས་ཁྱོན་ཁང་པ་ ༡༥༠༠ འདི་ལོའི་ཕྱི་ཟླ་ ༡༠ ཚེས་ ༣༠ ཉིན་གྱི་ཡར་སྔོན་དུ་གཏོར་བཤིག་ལེགས་འགྲུབ་ཡོང་དགོས་པ། གཞན་ཡང་། སྒར་གྱི་འབྲེལ་ཡོད་མི་སྣའི་སྣེ་འཁྲིད་འོག་སྒར་གྱི་གྲྭ་བཙུན་ཚོར་རྒྱལ་ཁབ་ཀྱི་ཁྲིམས་ལུགས་ལ་སྲུང་བརྩི་བྱེད་པའི་དམ་བཅའི་ཡི་གེ་འབྲི་རུ་འཇུག་དགོས་པ་མ་ཟད། གསང་བའི་བརྙན་ལེན་འཕྲུལ་ཆས་བཙུགས་ཏེ་ནང་འོང་བཀག་རྒྱ་འཛུགས་སྐྲུན་དགོས་སྐོར་སོགས་ཀྱི་བཀའ་རྒྱ་བཀྲམས་ཡོད་འདུག དེ་ཡང་བཀའ་རྒྱའི་ནང་། ལས་འགན་གཙོ་བོ་ནི་ཕྱི་ལོ་ ༢༠༡༦ ལོར་མི་ ༢༢༠༠ ཕྱིར་འགྲོ་དགོས་པ་དང་། དེའི་ནང་ནས་དགེ་བསྙེན་ ༡༠༢༩ ལོགས་སུ་དགར་རྒྱུ་དང་། དོན་དངོས་ཐོག་གྲྭ་ཇོ་ ༡༢༠༠ ཕྱིར་འཕུད་དགོས་ལ། དེའི་ནང་དུ་མ་མཐའ་ཡང་ཞིང་ཆེན་གཞན་པའི་གྲྭ་བཙུན་ ༦༠༠ ཡོད་དགོས། ཕྱི་ལོ་ ༢༠༡༧ ལོར་ཁྱོན་མི་གྲངས་ ༥༠༠༠ ལ་ཚོད་འཛིན་ཐུབ་དགོས་པ་དང་། གལ་ཏེ་ ༢༠༡༧ ཟླ་ ༩ ཚེས་ ༣༠ ཉིན་མི་གྲངས་ ༥༠༠༠ ལ་ཚོད་འཛིན་མ་ཐུབ་པ་ཤེས་ཚེ། དངོས་ཡོད་མི་གྲངས་ཀྱི་མིང་ཐོ་དང་བསྟུན་ནས་ ༢༠༡༧ ལོར་ཕྱིར་འཕུད་མི་གྲངས་ཇེ་མང་དུ་གཏོང་དགོས། སྲིད་གཞུང་ནས་ཕྱིར་འཕུད་གཏོང་དགོས་པའི་གྲྭ་བཙུན་རྣམས་ཀྱི་མིང་དང་ཁང་པའི་གྲངས་ཀར་དངོས་བཤེར་ནན་མོ་བྱེད་རྒྱུ་དང་། འདི་ལོའི་ཕྱི་ཟླ་ ༦ ཚེས་ ༡༥ ཡི་སྔོན་དུ་འདི་ལོར་ཕྱིར་འཕུད་གཏོང་དགོས་པའི་གྲྭ་བཙུན་རྣམས་ཀྱི་མཚན་ཐོ་གོང་རིམ་ལ་སྤྲོད་དགོས་པ་མ་ཟད། ཚད་ལྡན་གཅིག་དང་དོན་གནས་གསུམ་གྱི་ལས་འགན་ལེགས་འགྲུབ་བྱུང་རྗེས། རྒྱུན་སྡོད་པ་ ༥༠༠༠ དང་དེ་མིན་རྣམས་ཀྱི་མིང་ཐོར་ཞིབ་བཤེར་བྱས་ནས་གོང་རིམ་ལ་སྤྲོད་དགོས་པ་དང་། འདི་ལོའི་ཕྱི་ཟླ་ ༡༠ ཚེས་ ༣༠ ཉིན་གྱི་སྔོན་ལ་གྲྭ་བཙུན་ ༡༢༠༠ ཕྱིར་འཕུད་ཀྱི་ལས་གཞི་ལེགས་འགྲུབ་བྱུང་དགོས་པ་དང་། སྒར་སྤྱིའི་མི་གྲངས་འབོར་ ༥༠༠༠ ནང་ཞིང་ཆེན་གཞན་གྱི་མི་ ༡༠༠༠ ལས་ཡོད་མི་ཆོག་སྐོར་བརྗོད་འདུག ཡང་བཀའ་རྒྱའི་ནང་ལས་འགན་གཙོ་བོ་ནི་ཕྱི་ལོ་ ༢༠༡༣ ལོ་ནས་ད་བར་ཕྱིར་འཕུད་བྱས་པའི་གྲྭ་བཙུན་གྱི་ཁང་པ་དང་། སྒྲིག་འགལ་གྲྭ་ཁང་གསར་པ། རྒན་གསོ་ཁང་དུ་སོང་བ་རྣམས་ཀྱི་ཁང་པ། བཙུན་མོའི་འགྲུལ་ཁང་དུ་སོང་བ་རྣམས་ཀྱི་ཁང་པ་བཅས་ཁྱོན་ཁང་པ་ ༡༥༠༠ འདི་ལོའི་ཕྱི་ཟླ་ ༡༠ ཚེས་ ༣༠ ཉིན་གྱི་སྔོན་དུ་གཏོར་བཤིག་ལེགས་འགྲུབ་བྱུང་དགོས་སྐོར་དང་། དགེ་བསྙེན་གླིང་ལོགས་སུ་བཀར་ཏེ་སྒར་དང་དགེ་བསྙེན་རྣམས་བསྲེ་མཚམས་ནས་རྨིད་ ༣༠ བར་གྱི་འཛུགས་སྐྲུན་རྣམས་བཤིག་ནས་བར་ཐག་ཕྱེ་རྒྱུ་ཡིན་སྐོར་སོགས་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
