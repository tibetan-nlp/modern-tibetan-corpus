མི་མང་ས་ཡ་ ༢༡ བྲན་གཡོག་ཏུ་གྱུར།

ཁ་སང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལྷན་ཁང་གིས་གསར་འགོད་གསལ་བསྒྲགས་སྤེལ་ཏེ་དེང་སྐབས་འཛམ་གླིང་གྲུ་གང་སར་མི་མང་ས་ཡ་ ༢༡ ཙམ་དཔལ་འབྱོར་འཚོལ་སྒྲུབ་ཀྱི་བྲན་གཡོག་ཆགས་ཀྱི་ཡོད་པའི་གྲས་སུ་བྱིས་པ་ས་ཡ་ ༥ ཟིན་གྱི་ཡོད་སྐོར་བརྗོད་འདུག གསར་འགོད་གསལ་བསྒྲགས་དེའི་ནང་། འདིར་བྲན་གཡོག་ཅེས་འཁོད་ཡོད་པ་དེ་གནའ་བོའི་བྲན་གཡོག་ལ་མི་གོ་བར་དེང་གི་ང་ཚོས་དུས་རབས་འདིར་མངོན་གསལ་དུ་འགྱུར་བཞིན་པའི་བྲན་གཡོག་གི་གནས་སྟངས་ལ་གོ་དགོས། ཕྱི་ལོ་ ༢༠༡༦ གཅིག་པུར་རྒྱལ་ཚོགས་ཀྱི་དང་བླངས་ཐེབས་རྩ་ཁང་ (UNVFCFS) གིས་འཛམ་གླིང་རྒྱལ་ཁབ་ ༣༤ ནང་མི་མང་ ༢༩༠༠༠ ལྷག་གི་འཚོ་གནས་ལ་མཐུན་འགྱུར་རོགས་སྐྱོར་བྱས་ཡོད་པ་དང་། དེའི་གྲས་ཀྱི་སུམ་ཆ་གཅིག་ལོ་ཆུང་བྱིས་པ་རེད། ཅེས་འཁོད་འདུག གནས་ཚུལ་ལྟར་ན། འཆར་ཟླ་ ༡༢ པའི་ཚེས་ ༢ ཉིན་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལྷན་ཚོགས་ཀྱིས་“དེང་དུས་ཀྱི་བྱིས་པ་མང་པོ་བྲན་གཡོག་ཏུ་འགྱུར་བཞིན་པའི་གནད་དོན་ཐེར་འདོན” (RCFMS) ཞེས་པའི་བརྗོད་གཞིའི་ཐོག་ནས་རྒྱལ་སྤྱིའི་མི་མང་བྲན་གཡོག་གི་གནས་བབ་ལས་ཐར་བའི་ཉིན་མོ་ (IDAS) སྲུང་གཙི་གནང་རྒྱུ་ཡིན་པ་མ་ཟད། ད་དུང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་དང་བླངས་ཐེབས་རྩ་ (UNVFCFS) ཁང་ཚུགས་ནས་ལོ་འཁོར་ ༢༥ ལོན་པའི་ཉིན་མོ་ཡང་སྲུང་གཙི་གནང་རྒྱུ་ཡིན་འདུག དང་བླངས་ཐེབས་རྩ་ཁང་དེ་ནི་རྒྱལ་སྤྱིའི་སྟེང་གི་བྱིས་པ་ངལ་རྩོལ་བ་གཙོས་པའི་བུད་མེད་ཚོ་བཙན་གྱིས་གཉེན་སྒྲིག་ཏུ་འཇུག་པ་དང་། དངུལ་དངོས་ཀྱི་སྟོབས་འབྱོར་ལ་བརྟེན་ནས་བག་མ་ཉོ་བ་དང་མེད་མུག་ལ་བརྟེན་ནས་བཟའ་ཟླ་འཚོང་བ། འཚོ་གནས་ཀྱི་བུ་ལོན་དང་དམག་ཁྲལ། བཙན་དབང་གིས་སྨད་འཚོང་གི་ལས་སུ་འཇུག་པ་དང་མི་མང་བྲན་གཡོག་ཏུ་འཇུག་པ། དེ་བཞིན། ཉིན་གླར་བརྟེན་ནས་འཚོ་གནས་འཛིན་པ་དང་བུན་དངུལ་གྱི་རྐྱེན་པས་མི་མང་ཚེ་གཡོག་བྱེད་པ་སོགས་ཀྱི་དཀའ་སྡུག་ལ་གཞིགས་ཏེ་ཕྱི་ལོ་ ༡༩༩༡ ལོར་གསར་འཛུགས་བྱས་ཡོད་པ་དང་། དང་བླངས་ཐེབས་རྩ་ཁང་དེས་ད་ལྟའི་བར་རྒྱལ་སྤྱིའི་རང་དབང་སྒྲིག་འཛུགས་ཁག་ ༤༠༠ ལྷག་ལ་འབྲེལ་འཛིན་བྱས་པ་བརྒྱུད་ནས་དེང་སྐབས་དཔལ་འབྱོར་འཚོལ་སྒྲུབ་ཀྱི་བྲན་གཡོག་ཏུ་འགྱུར་བཞིན་པའི་སྐྱེས་པ་དང་བུད་མེད། ལྷག་པར་དུ་ལོ་ཆུང་བྱིས་པ་སྟོང་ཕྲག་མང་པོའི་འཚོ་གནས་ཐད་མཐུན་འགྱུར་རོགས་སྐྱོར་བྱས་ཡོད་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
