
འབྱོར་ལྡན་རུལ་སུངས་པ་ལས་དཔོན་རིགས་རུལ་སུངས་པ་མང་བ་འདུག

  ཁ་སང་འབྱོར་ལྡན་ (Hú Rùn) ཞེས་པའི་རྒྱ་ནག་གི་འབྱོར་ལྡན་ལོ་འཁོར་ཞིབ་འཇུག་སྙན་ཐོ་ཞིག་བཏོན་པའི་ནང་། འདས་པའི་ལོ་ངོ་ ༡༧ རིང་རྒྱ་ནག་གི་འབྱོར་ལྡན་པ་བརྒྱ་ཆ་གཅིག་གིས་ལྐོག་ཟ་རུལ་སུངས་དང་འབྲེལ་བའི་ནག་ཉེས་བསགས་ཡོད་སྐོར་བཀོད་འདུག   རྒྱ་ནག་གི་འབྱོར་ལྡན་ཞིབ་འཇུག་སྙན་ཐོ་གསར་འཛུགས་བྱེད་མཁན་རུ་ཕིར་ཊི་ཧོ་གྷི་ཝར་ཧྥི་ (Rupert Hoogewerf)ལགས་ཀྱིས་བརྗོད་པ་ལྟར་ན། ལྐོག་ཟ་རུལ་སུངས་བྱས་པའི་ནག་ཉེས་འོག་འཛིན་བཟུང་བྱས་ཟིན་པ་དང་ལྟ་ཞིབ་བཤེར་འོག་ཡོད་པའི་རྒྱ་ནག་གི་མཐོ་རིམ་དཔོན་རིགས་དང་ཁེ་ལས་མ་དངུལ་དོ་དམ་པ་ཚོ་དང་། རྒྱ་ནག་ནང་གི་འབྱོར་ལྡན་ལྐོག་ཟ་རུལ་སུངས་བྱེད་མཁན་གཉིས་བསྡུར་ན། འབྱོར་ལྡན་ལྐོག་ཟ་རུལ་སུངས་བྱེད་མཁན་གྱི་གྲངས་འབོར་ཅུང་ཉུང་ཉུང་ཡིན་སྐོར་བརྗོད་འདུག   གནས་ཚུལ་ལྟར་ན། རྒྱ་ནག་གི་སྤྱི་ཚོགས་བརྒྱུད་ལམ་ཁག་ཏུ་ཕྱི་ལོ་ ༡༩༩༩ ལོར་རྒྱ་ནག་གི་འབྱོར་ལྡན་ཞིབ་འཇུག་སྙན་ཐོ་བཏོན་པ་ནས་ད་ལྟའི་བར་རྒྱ་ནག་ནང་འབྱོར་ལྡན་པ་ ༣༠༠༠ གྲས་ཀྱི་ ༣༥ ལྐོག་ཟ་རུལ་སུངས་དང་འབྲེལ་བའི་ནག་ཉེས་འོག་ཁྲིམས་བཤེར་བྱེད་མུས་ཡིན་སྟབས། རྒྱ་ནག་གི་ལོ་འཁོར་འབྱོར་ལྡན་ཞིབ་འཇུག་སྙན་ཐོའི་ནང་བཏོན་པའི་འབྱོར་ལྡན་ཐོ་འགོད་རེའུ་མིག་དེ་ནི་དོན་དམ་ཉེས་མེད་མི་སེར་དམར་གསོད་གཏོང་བའི་ལེའུ་ཞིག་ཡིན་པར་ཚོད་དཔག་བྱེད་ཀྱི་ཡོད་པ་དང་། འདི་ལོར་རྒྱ་ནག་གི་འབྱོར་ལྡན་པ་ ༡༡ ཙམ་ལ་ལྐོག་རྔན་སྤྲོད་ལེན་གྱི་ནག་ཉེས་འོག་ཚུད་པའི་གྲས་ཀྱི་འབྱོར་ལྡན་པ་ ༩ རྒྱལ་ཁབ་ཀྱི་ཁེ་ལས་མ་དངུལ་དོ་དམ་པ་ཡིན་ཞིང་། ཁོ་ཚོའི་གྲས་ཀྱི་འབྱོར་ལྡན་པ་མང་ཆེ་བ་ཧྲང་ཧེ་དང་པེ་ཅིན་གྲོང་ཁྱེར་ནས་ཡིན་པ་དང་། ཆ་སྙོམས་ཀྱིས་ཁོ་ཚོ་ལོ་ ༤༧ ཡས་མས་ཡིན་པ་དང་ཁོ་ཚོར་ལོ་ ༡༠ རེའི་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག   ན་ནིང་ནས་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱིས་ལྐོག་ཟ་རུལ་སུངས་གཙང་བཤེར་སྲིད་ཇུས་ལག་བསྟར་བྱས་ཏེ། རྒྱ་རིགས་འབྱོར་ལྡན་པ་མང་པོར་ཁྲིམས་བཤེར་བྱེད་མུས་ཡིན་པ་དང་། ཟླ་འདིའི་ནང་རྒྱ་ནག་གི་འབྱོར་ལྡན་པ་ཞུ་མེན་ (Xu Ming) ཟེར་བ་ཞིག་དང་ཌ་ལེན་ཧྲེ་ཌ་ (Dalian Shide Group) ཚོགས་པ་གཉིས་ལ་ལྐོག་རྔན་སྤྲོད་ལེན་གྱི་ནག་ཉེས་ཡོད་པའི་ཁྲིམས་ཆད་ཕོག་ཡོད་པ་མ་ཟད། གཟའ་འཁོར་སྔོན་མར་རྒྱ་ནག་ནང་སྙན་གྲགས་ཡོད་པའི་འབྱོར་ལྡན་པ་གོ་གུང་ཁྲང་ (Guo Guangchang) དང་ཧྥོ་སུན་ (Fosun Group)ཚོགས་པའི་གཙོ་འཛིན་གཉིས་ཀྱིས་ཀྱང་རྒྱལ་ཁབ་ཀྱི་ཁེ་ལས་མ་དངུལ་ལྐོག་ཟ་བྱས་པའི་ནག་ཉེས་འོག་འཛིན་བཟུང་བྱས་ཡོད་འདུག   གསར་འགོད་པ། པདྨ་རྒྱལ།
