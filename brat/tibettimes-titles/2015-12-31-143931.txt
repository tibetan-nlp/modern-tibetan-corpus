སྐུ་ཕྱྭར་སྔར་ལས་དོ་ནན་བྱ་རྒྱུ་ཡིན།

  ཧི་མ་ཅལ་མངའ་སྡེའི་ཉེན་རྟོག་ཁང་གིས་༧གོང་ས་མཆོག་གི་བཞུགས་གནས་ཀྱི་སྲུང་སྐྱོབ་ཆེད་མི་རྒྱུ་གཉིས་ཀྱི་ཐོག་སྔར་ལས་ཤུགས་གནོན་གནང་རྒྱུ་ཡིན་སྐོར་བརྗོད་འདུག   དེ་རིང་ཧི་མ་ཅལ་མངའ་སྡེའི་ཉེན་རྟོག་ཁང་གིས་རྒྱ་གར་གྱི་ཁྲོལ་བྷོན་(The Tribune)དྲ་བའི་ཐོག རྒྱ་གར་གསང་བའི་ལས་ཁུངས་(CID)དང་ (Intelligence Bureau)། དམག་དོན་གསང་བའི་ལས་ཁུངས་སོགས་གསང་བའི་ལས་ཁུངས་མང་པོ་ཞིག་གིས་མཉམ་ལས་ཞིབ་འཇུག་བྱས་པ་བརྒྱུད། ༧གོང་ས་ཏཱ་ལའི་བླ་མ་མཆོག་གི་སྐུ་ཕྱྭའི་བདེ་སྲུང་ཡར་རྒྱས་གཏོང་ཆེད། ཁོང་གི་བཞུགས་གནས་ཀྱི་བདེ་སྲུང་མི་སྣ་ཇེ་མང་དུ་གཏོང་རྒྱུ་ཡིན་པ་དང་། སྤུས་ཚད་ལེགས་པའི་བརྙན་ལེན་འཕྲུལ་ཆས་དང་། ཅ་དངོས་ཞིབ་བཤེར་འཕྲུལ་ཆས། དེ་བཞིན། ཉེན་སྲུང་འཕྲུལ་ཆས་རིགས་གསར་འཛུགས་བྱ་རྒྱུ་ཡིན་སྐོར་བསྒྲགས་འདུག དེ་བཞིན། “ནང་སྲིད་ལས་ཁུངས་ཀྱིས་ང་ཚོའི་ལས་འཆར་དེར་རྒྱབ་སྐྱོར་ཐོབ་པ་ཡིན་ན། ང་ཚོས་བདེ་སྲུང་མི་སྣ་ ༥༠ ནས་ ༡༠༠ ལྷག་བར་ཕར་གནོན་བྱ་རྒྱུ་ཡིན།” ཞེས་སོགས་བརྗོད་འདུག   ཧི་མ་ཅལ་མངའ་སྡེའི་ཉེན་རྟོག་ཁང་གིས་གོང་གནས་ཚུལ་གྱི་ཞིབ་ཕྲའི་ལས་འཆར་མངའ་སྡེའི་ནང་སྲིད་ལས་ཁུངས་སུ་ཕུལ་ཡོད་སྟབས། མ་འོངས་པར་མངའ་སྡེའི་ནང་སྲིད་ལས་ཁུངས་ཀྱིས་དབུས་ནང་སྲིད་ལས་ཁུངས་ལ་སྙན་གསེང་ཞུས་ཏེ། ༧གོང་ས་མཆོག་གི་སྐུ་ཕྱྭའི་བདེ་སྲུང་ཆེད་དཔལ་འབྱོར་གྱི་འགྲོ་གྲོན་སྔར་ལས་མང་པོ་གཏོང་བར་བཀའ་འཁྲོལ་ཞུ་རྒྱུ་ཡིན་འདུག   ཡང་གནས་ཚུལ་དེའི་ནང་། ༧གོང་ས་མཆོག་གི་སྐུ་ཕྱྭའི་བདེ་སྲུང་ཤུགས་གནོན་བྱེད་དགོས་དགོས་གཙོ་བོ་ནི། ཕྱི་ཕྱོགས་ནས་སྐུ་ཕྱྭར་ཉེན་ཁ་བཟོ་སྲིད་པའི་དོགས་ཟོན་བྱེད་ཀྱི་ཡོད་པ་མ་ཟད། དེ་སྔོན་༧གོང་ས་མཆོག་གི་བཞུགས་གནས་ཉེ་འགྲམ་ནས་རྒྱ་ནག་གི་གསང་ཉུལ་བ་ཁག་ཅིག་རྒྱ་གར་གསང་བའི་ལས་ཁུངས་ཀྱིས་འཛིན་ཟུང་བྱེད་སྐབས། གསང་ཉུལ་བ་དེ་ཚོས་༧གོང་ས་མཆོག་གི་ཕོ་བྲང་གི་ས་ཁྲ་དང་གསང་བའི་ཡིག་ཆ་ཁག་ཅིག་ཉར་ཡོད་པ་ལག་སོན་བྱུང་ཡོད་ལུགས་སོགས་བརྗོད་འདུག   གཞན། ཟླ་འདིའི་ཚེས་ ༡༤ ཉིན། རྒྱ་གར་དབུས་གཞུང་གིས་རྒྱ་གར་མཁའ་འགྲུལ་ལས་ཁུངས་ལ། དེ་སྔོན་ནས་རྒྱ་གར་དབུས་གཞུང་གིས་བོད་མིའི་སྐྱབས་མགོན་༧ཏཱ་ལའི་བླ་མ་མཆོག་ལ་རྒྱ་གར་རྒྱལ་ཡོངས་ཀྱི་གནམ་ཐང་ཁག་ཏུ་ལུས་བཤེར་བྱེད་མི་དགོས་པའི་དམིགས་བསལ་ཆེ་མཐོང་ཞིག་སྤྲད་ཡོད་པ་དེ་མུ་མཐུད་ལག་བསྟར་བྱེད་དགོས་པའི་བཀའ་བཏང་ཡོད་འདུག   གསར་འགོད་པ། བསྟན་འཛིན་ཆོས་གྲགས།
