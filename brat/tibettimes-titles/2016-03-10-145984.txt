བོད་མི་ཞི་རྒོལ་བ་མང་པོ་བཀག་ཉར་བྱས།

དེ་རིང་གསུམ་བཅུའི་དུས་དྲན་ཐེངས་ ༥༧ པ་སྲུང་གཙི་གནང་སྐབས་རྒྱ་གར་གྱི་རྒྱལ་ས་ལྡི་ལི་ནས་བོད་མི་ཞི་རྒོལ་བ་ ༩༠ ལྷག་བཀག་ཉར་བྱས་ཡོད་འདུག   དེ་རིང་བོད་ཀྱི་རྒྱལ་ས་ལྷ་སར་བོད་མིའི་ཞི་བའི་སྒེར་ལངས་གསུམ་བཅུའི་དུས་དྲན་གྱི་ཉིན་མོ་དང་བསྟུན། སྔ་དྲོའི་ཆུ་ཚོད་ ༩ ཐོག་རྒྱ་གར་གྱི་རྒྱལ་ས་ལྡི་ལིར་གཞུང་འབྲེལ་གྱི་དུས་དྲན་སྲུང་གཙི་གནང་རྗེས། དབུས་གཞོན་ནུ་ལྷན་ཚོགས་ཀྱི་གོ་སྒྲིག་འོག བསམ་ཡས་གླིང་ས་གནས་གཞོན་ནུ་དང་རོ་ཧི་ནི་ས་གནས་གཞོན་ནུས་མཉམ་རུབ་བྱས་ཏེ། ལྡི་ལི་ཇན་ཊར་མན་ཊར་དུ་མི་མང་ ༢༠༠ ལྷག་གིས་ཞི་རྒོལ་ཁྲོམ་སྐོར་བྱས་ཡོད་པ་མ་ཟད། ལྡི་ལིར་ཡོད་པའི་རྒྱ་ནག་གཞུང་ཚབ་ཁང་གི་མདུན་དུ་ཞི་བའི་སྐད་འབོད་གནང་ཡོད་པར་བརྟེན། རྒྱ་གར་ཉེན་རྟོག་པ་མང་པོ་ཡོང་སྟེ་ལས་འགུལ་བ་བུ་མོ་ ༨༠ བཅས་ཁྱོན་མི་གྲངས་ ༩༠ ལྷག་བཙན་འཁྲིད་ཀྱིས་བཀག་ཉར་འོག་ཡོད་འདུག   འདི་གའི་གསར་འགོད་པས་ཁ་པར་བརྒྱུད་ས་གནས་གཞོན་ནུའི་རིག་གཞུང་འགན་འཛིན་བཀྲ་ཤིས་ལགས་སུ་སྐད་ཆ་འདྲི་སྐབས། ཐེངས་འདིའི་དུས་དྲན་ཐོག་ཉིན་གྲངས་ ༣ རིང་ཞི་རྒོལ་ལས་འགུལ་སྤེལ་འཆར་ཡོད་ལ། དེ་རིང་ཆུ་ཚོད་ ༢ རེའི་ནང་ལས་འགུལ་བ་རྣམས་ཆ་བགོས་ཀྱིས། ས་གནས་ཁག་མི་འདྲ་བ་ལྔ་ནས་རྒྱ་ནག་གཞུང་ཚབ་ཁང་ལ་བཙན་འཛུལ་གྱི་ལས་འགུལ་སྤེལ་གྱི་ཡོད་པ་དང་། དེ་བཞིན་ད་ལྟའི་བར་འཛམ་གླིང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་དང་ཡུ་རོབ་མཐུན་ཚོགས་སོགས་ལ་བོད་དོན་ཐོག་དོ་ཁུར་གནང་དགོས་པའི་འབོད་སྐུལ་སྙན་ཞུ་རབས་དང་རིམ་པར་ཕུལ་ཡོད་ཀྱང་། ལག་བསྟར་དོན་འཁྱོལ་ཅི་ཡང་བྱུང་མེད་སྟབས། སང་ཉིན་ལས་འགུལ་བ་ཚོས་འཛམ་གླིང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་དང་ཡུ་རོབ་མཐུན་ཚོགས་སོགས་ལ་སྙན་ཞུ་རྣམས་དོན་འཁྱོལ་མ་ཡོང་བའི་སྐོར་ནས་དྲི་བ་འབུལ་རྒྱུ་ཡིན་ལ། དྲི་བ་འབུལ་སྟངས་དེའང་གཉིད་སད་ཤོག་ཟེར་བའི་བརྡ་ལན་སྤྲོད་ཆེད། རྔ་རྡུང་བ་དང་སི་རིལ་(Whistle)རྒྱག་པ་སོགས་བྱས་ནས་འབོད་སྐུལ་གྱི་ལས་འགུལ་སྤེལ་འཆར་ཡོད་པ་དང་། གནངས་ཉིན་བུད་མེད་སྒེར་ལངས་ཉིན་ལྡི་ལིའི་ཀོ་ནཱ་ཊི་ཁུལ་(Cannaught Place)གྱི་པ་ལི་ཀ་ཁྲོམ་གཞུང་དུ་བུད་མེད་ ༡༠༠ ལྷག་འདུ་འཛོམས་ཀྱིས། སྤྱིར་འགྲོ་བ་མིའི་ཐོབ་ཐང་དང་བྱེ་བྲག་བུད་མེད་ཀྱི་ཐོབ་ཐང་སྐོར་ནས་ཞི་བའི་ལས་འགུལ་སྤེལ་རྩིས་ཡོད་སྐོར་བརྗོད་བྱུང་།   དེ་རིང་ཕྱི་ལོ་ ༡༩༥༩ ལོར་བོད་མིའི་ཞི་བའི་སྒེར་ལངས་ཆེན་མོའི་ཁྲོད་བོད་བསྟན་པ་ཆབ་སྲིད་ཆེད་གཅེས་པའི་ཚེ་སྲོག་ཤོར་བའི་རྒྱལ་གཅེས་དཔའ་བོ་དཔའ་མོ་རྣམས་ལ་རྗེས་དྲན་གྱི་ཉིན་མོ་འདིར།  འཛམ་གླིང་གི་ཡུལ་ལུང་གང་སར་ཡོད་པའི་བོད་མི་ཚོས་རྒྱལ་གཅེས་དཔའ་བོ་དཔའ་མོ་རྣམ་ལ་རྗེས་དྲན་མྱ་ངན་དང་འབྲེལ། ས་གནས་སོ་སོའི་རྒྱ་ནག་གཞུང་ཚབ་ཁག་གི་མདུན་དུ་ཞི་བའི་སྐད་འབོད་ཀྱི་ལས་འགུལ་སྣ་མང་སྤེལ་བཞིན་ཡོད་པ་ལྟར། སྤྱི་ཚོགས་འཕྲིན་ལམ་ཁག་ཏུ་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། དེ་རིང་སྔ་དྲོར་བོད་ཀྱི་ཆབ་སྲིད་བཙོན་ཟུར་མང་པོ་ཡོད་པའི་རྒྱལ་ཁབ་ཨོ་སི་ཊོ་ལི་ཡའི་མེལ་བྷོན་(Melbourne)དང་སིག་ནེ་(Sydney)གྲོང་ཁྱེར་གྱི་རྒྱ་ནག་གཞུང་ཚབ་མདུན་དུ། ས་གནས་བོད་མི་མང་པོ་འདུ་འཛོམས་ཀྱིས་ཞི་རྒོལ་ལས་འགུལ་སྤེལ་ཡོད་པ་མ་ཟད། མེལ་བྷོན་གྱི་རྒྱ་ནག་གཞུང་ཚབ་ཁང་རེ་ཞིག་སྒོ་རྒྱག་དགོས་བྱུང་ཡོད་འདུག གཞན་ཡང། ཉེ་བའི་ལོ་ཤས་ནས་བཟུང་རྒྱ་ནག་གཞུང་གི་དཔལ་འབྱོར་དང་ཆབ་སྲིད་ཀྱི་གནོན་ཤུགས་འོག བལ་ཡུལ་གཞུང་ནས་ཁུལ་དེའི་བོད་མི་ཚོས་གསུམ་བཅུའི་དུས་དྲན་སྲུང་གཙི་ཞུ་བར་དམ་དྲག་བཀག་སྡོམ་ཤུགས་ཆེན་བྱེད་བཞིན་ཡོད་པ་ལྟར། དེ་རིང་ཡང་མཆོད་རྟེན་ཉེ་འཁོར་གྱི་གཙོས་བོད་མི་འདུས་སྡོད་ཁུལ་དུ་དམག་མི་མང་པོ་འགྲེམས་འཇོག་བྱས་ཏེ་དམ་དྲག་ཆེན་པོ་བྱེད་བཞིན་ཡོད་འདུག   གསར་འགོད་པ། པདྨ་མཚོ།
