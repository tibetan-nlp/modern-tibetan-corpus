ཆབ་སྲིད་བཙོན་པ་ཞིག་ཀློད་གྲོལ།

ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༢ ཉིན། དཀར་མཛེས་ཁུལ་གསེར་རྟ་ཡར་ལུང་ཡུལ་ཚོ་ཆོས་སྐོར་སྨད་མའི་ཚེ་སྐྱབས་ལགས་བཙོན་འཇུག་དུས་ཡུན་ཚང་སྟེ་ཀློད་གྲོལ་བཏང་ཡོད་འདུག ཚེ་སྐྱབས་ལགས་ཀྱིས་༼བོད་ན་གཞོན་དམར་ཤོག་རྒོལ་ལན་ཚོགས་པ།༽ཞེས་པའི་ཚོགས་མི་བྱས་ཏེ། ཆབ་སྲིད་དང་འབྲེལ་བའི་ཡིག་ཆ་རིགས་འགྲེམ་སྤེལ་བྱས་པའི་ཉེས་མིང་འོག ཕྱི་ལོ་ ༢༠༡༢ ཟླ་ ༨ པའི་ཚེས་ ༡༣ ཉིན་རྒྱ་གཞུང་གིས་ཁོང་འཛིན་བཟུང་བྱས་ཏེ། ལོ་ ༢ ཀྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་པ་ལྟར། ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༢ ཉིན་བཙོན་འཇུག་དུས་ཡུན་ཚང་སྟེ་རང་ཁྱིམ་དུ་བསྐྱལ་ཡོད་སྟབས། ཕྱི་ཚེས་ ༡༣ ཉིན་ཁོང་གི་ཁྱིམ་མི་གཉེན་ཉེ་དང་ཡུལ་མི་ཚོས་ཁོང་ཚང་གི་ར་སྐོར་ནང་གུར་ཕབ་སྟེ། ༼བོད་མིའི་བདེན་དོན་འཐབ་རྩོད་པ་གངས་བུ་ཚེ་སྐྱབས་ལགས་ཕེབས་པར་དགའ་བསུ་ཞུ།༽ཞེས་པའི་སྦྱར་ཡིག་ཆེན་པོ་རླངས་འཁོར་སོགས་ཀྱི་ངོས་སུ་སྦྱར་ཏེ། ཁོང་ལ་ཁ་བཏགས་བསྒྲོན་པ་དང་གླུ་གར་འཁྲབ་སྟོན་སོགས་བྱས་ཏེ་དགའ་བསུ་བྱས་ཡོད་འདུག བསོད་ནམས་ལགས་ཀྱིས་འདི་གར་གནས་ཚུལ་བཏང་བ་ལྟར་ན། ཁོང་བཙོན་ཁང་དུ་ཡོད་སྐབས་ཟ་མ་མ་འཕྲོད་པ་དང་སྨན་བཅོས་ཀྱི་ཆ་རྐྱེན་ཞན་པ་སོགས་ཀྱིས། མཁལ་མའི་ནང་རྡོ་སྲན་ཆགས་པ་དང་ཕོ་ཁ་མི་བདེ་བ་སོགས་ཀྱི་ནད་གཞི་ཕོགས་ཡོད་པ་མ་ཟད། མ་འོངས་ལོ་ ༢ རིང་ཁོང་གསེར་རྟ་རྫོང་མཁར་ལས་ས་གནས་གཞན་དུ་འགྲོ་མི་ཆོག་པ་དང་། ལོ་ ༢ ཀྱི་ཆབ་སྲིད་ཀྱི་ཁེ་དབང་ཕྱིར་བསྡུ་དང་རྒྱུན་དུ་རྗེས་འདེད་པ་གཏོང་རྒྱུ་ཡིན་ལུགས་སོགས་བརྗོད་ཡོད་སྟབས། བཙོན་འཇུག་གི་དུས་ཡུན་རྫོགས་ཏེ་ཀློད་གྲོལ་ཐོབ་ཡོད་ཀྱང་། ཁོང་དང་ཁོང་གི་ཁྱིམ་མི་ཚོ་སེམས་ཁྲལ་དང་དངངས་སྐྲག་འོག་གནས་ཡོད་འདུག ཚེ་སྐྱབས་ལགས་ད་ལྟ་རང་ལོ་ ༣༦ ཡིན་ལ། ཁྱིམ་དུ་ཨ་མ་ལོ་ ༦༩ ཡས་མས་ཅན་ཞིག་བཅས་བཟའ་མི་ ༥ ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
