གསེར་རྟ་ནས་གྲྭ་བ་ཞིག་འཛིན་བཟུང་།

གསེར་རྟ་ཕུ་གུ་དགོན་གྱི་གྲྭ་ཆོས་ཀྱིས་ལགས་རྒྱ་གཞུང་གིས་བཙན་ཁྲིད་བྱས་ཡོད་པ་དང་ཉིན་དེའི་རྗེས་ལ་ཁོང་གི་ཕ་མ་གཅིག་པའི་གཅེན་མོ་སྐྱིད་འཛོམས་དང་མོའི་བུ་གྲགས་པ་གཉིས་ཀྱང་འཛིན་བཟུང་བྱས་ཡོད་འདུག དེ་རིང་འདི་གའི་གསར་ཁང་དུ་མིང་བརྗོད་འདོད་མེད་པའི་འབྲེལ་ཡོད་མི་སྣ་ཞིག་གིས་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། ད་ལོའི་སྐུའི་གྱ་སྟོན་ཡས་མས་ནས་རྒྱ་གཞུང་གིས་གསེར་རྟ་ཕྱོགས་ཀྱི་ཁ་པར་དང་དྲ་རྒྱའི་བརྒྱུད་ལམ་བཀག་འགོག་བྱས་ཡོད་ཅིང་། ད་ལྟ་གཞི་ནས་ཁ་པར་དང་དྲ་རྒྱ་སྒོ་ཅུང་གུ་ཡངས་ཆགས་བཞིན་པ་དང་བསྟུན་ཏེ་གནས་ཚུལ་འབྱོར་བར། ཕྱི་ཟླ་ ༦ ཚེས་ ༡༩ ཉིན་གྱི་ཕྱི་དྲོར་མདོ་སྨད་དབལ་ཤུལ་གསེར་རྟ་རྫོང་སྙི་སྟོད་ཡུལ་ཚོ་ཕུ་གུ་དགོན་གྱི་གྲྭ་ཆོས་ཀྱིས་ཟེར་བ་ཞིག་རྒྱ་གཞུང་གིས་འཛིན་བཟུང་བཙན་ཁྲིད་བྱས་ཡོད་པ་དང་། ཕྱི་ཉིན་ཚེས་ ༢༠ ཉིན་ཁོང་གི་ཕ་མ་གཅིག་པའི་གཅེན་མོ་སྐྱིད་འཛོམས་དང་མོའི་བུ་གྲགས་པ་གཉིས་ཀྱང་འཛིན་བཟུང་བྱས་ཏེ། ཉིན་ ༡༥ རིང་ལ་བཀག་ཉར་གྱིས་གཅར་རྡུང་མནར་གཅོད་བཏང་མཐར་ཀློད་གྲོལ་བཏང་ཡོད་འདུག ཁོང་གིས་མུ་མཐུད་དུ་བརྗོད་རྒྱུར། གྲྭ་ཆོས་ཀྱིས་ལགས་ལ་མཆིན་པའི་ན་ཚ་དང་མཚེར་བའི་ན་ཚ་ལྕི་མོ་ཡོད་སྟབས། རྒྱུན་པར་རྒྱ་སྨན་དང་བོད་སྨན་གཉིས་ཀ་བསྟེན་བཞིན་པ་དང་། ཞིང་ཆེན་ཕྱི་ནང་གི་སྨན་ཁང་ཆེ་ཁག་ཏུ་བརྟག་དཔྱད་བྱས་པའི་ཡིག་ཆ་གསལ་བོ་ཡོད་པ་སོགས་གོ་མྱོང་། ཡིན་ན་ཡང་། ད་རེབ་རྒྱ་གཞུང་གིས་ཁོང་ལ་ཉེས་འཛུགས་གང་ཞིག་བྱས་ཡོད་མེད་གསལ་ཁ་མི་འདུག ཁོང་བཙན་ཁྲིད་བྱས་ནས་ཟླ་གཅིག་ལྷག་འགོར་ནའང་ད་དུང་གར་ཡོད་གང་ཁྲིད་ཤེས་རྟོག་ཐུབ་ཀྱི་མི་འདུག ཅེས་བརྗོད་སོང་། ཕ་མིང་ལ་ཞོ་སྐྱ་ཉོ་དང་། མ་མིང་ལ་ཚེ་རིང་སྐྱིད་ཟེར་བའི་རང་ལོ་ ༣༨ ཡིན་པའི་གྲྭ་ཆོས་ཀྱིས་ནི་ཆུང་དུས་ནས་སྙི་སྟོད་ཡུལ་འཚོ་ཕུ་གུ་དགོན་དུ་རབ་བྱུང་བསྒྲུབས་ཏེ། ནང་ཆོས་ལ་བསླབ་པ་སློབ་གཉེར་གནང་བཞིན་པའི་གྲྭ་བ་དཀྱུས་མ་ཞིག་ཡིན་འདུག   གསར་འགོད་པ། འཇིགས་མེད་རྒྱ་མཚོ།
