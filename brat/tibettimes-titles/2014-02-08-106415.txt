འབྲི་རུར་དམར་གསོད་དང་ཁྲིམས་ཆད།

རྒྱ་གཞུང་གིས་འབྲི་རུ་རྫོང་གི་བོད་མི་གཉིས་དམར་གསོད་དང་། བོད་མི་གཉིས་ལ་ལོ་ ༡༠ ནས་ ༡༡ བར་གྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག བོད་ཀྱི་འགྲོ་བ་མིའི་ཐོབ་ཐང་དང་མང་གཙོ་འཕེལ་རྒྱས་ལྟེ་གནས་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། རྒྱ་ནག་གཞུང་གིས་ ༢༠༡༣ ལོའི་ཟླ་ ༡༡ ཚེས་ ༢༣ ཉིན་ལྷ་སར་འཛིན་བཟུང་བྱས་པའི་རྒན་ངག་དབང་འཇམ་དབྱངས་ལགས། བཀག་ཉར་སྐབས་འདྲི་གཅོད་སྐབས་གཅར་རྡུང་བྱས་ཏེ་དམར་གསོད་བཏང་ཡོད་པ་མ་ཟད། སྔར་ལོའི་ཕྱི་ཟླ་ ༡༢ ནང་འབྲི་རུ་རྫོང་འབྲི་རུ་ཞང་ལྕམ་རམ་གྲོང་གི་ན་གཞོན་དཀོན་མཆོག་གྲགས་པ་ལགས་འཛིན་བཟུང་བཙན་འཁྲིད་བྱས་ཏེ། ༢༠༡༤ ལོའི་ཟླ་ ༡ ཚེས་ ༢༠ ཡས་མས་སུ་རྒྱ་གཞུང་ཉེན་རྟོག་པས་བཙོན་ཁང་ནང་ཉེས་རྡུང་ཚད་མེད་བཏང་སྟེ་དམར་གསོད་བཏང་རྗེས། ཕུང་པོ་ནང་མིར་རྩིས་སྤྲོད་བྱས་ཡོད་པ་དང་། ད་དུང་ཕུང་པོ་རྩིས་སྤྲོད་སྐབས་གནས་ཚུལ་དེའི་སྐོར་ཕྱིར་སྒྲོག་མི་ཆོག་པའི་བཙན་བཀའ་བཏང་ཡོད་འདུག དེ་བཞིན། སྔར་ལོའི་ཕྱི་ཟླ་ ༡༡ ཚེས་ ༢༣ ཉིན་འབྲི་རུ་རྫོང་ཚྭ་ཆུ་ཞང་ཨུ་ཐང་གྲོང་ནས་ཡིན་པའི་གྲྭ་སྐལ་བཟང་ཆོས་གླང་ལགས་ལྷ་སར་འཛིན་བཟུང་བྱས་ཡོད་པ་དང་། འདི་ལོའི་ཕྱི་ཟླ་ ༡ ནང་ཁོང་ལ་མི་མང་འདུ་འཛོམས་བྱེད་བཅུག་པའི་ཉེས་མིང་འོག་ལོ་ ༡༠ བཙོན་འཇུག་གི་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག་ལ། སྔར་ལོའི་ཕྱི་ཟླ་ ༡༠ ཚེས་ ༣ ཉིན་འཛིན་བཟུང་བྱས་པའི་འབྲི་རུ་རྫོང་ཟླ་ཐང་ཞང་གཡང་ཐང་གྲོང་ཚོ་བཞི་པའི་རྡོ་རྗེ་གྲགས་རྩལ་ལགས་སུ། དེ་སྔོན་གནས་རི་ནགས་ལྷའི་གཏེར་ཁ་སྔོག་འདོན་ལ་བཀག་འགོག་བྱེད་མཁན་གཙོ་གྲས་ཡིན་པའི་ཐད་ནས་ལོ་ ༧ དང་། མི་གཞན་པར་དངུལ་མང་པོ་བསྐྱི་གཡར་བྱས་པར་ལོ་ ༣ ། རྒྱ་གཞུང་གི་བཅའ་སྡོད་ལས་བྱེད་མི་སྣ་བཀག་འགོག་བྱས་པར་ལོ་ ༡ བཅས། ཁྱོན་བསྡོམས་ལོ་ ༡༡ ཁྲིམས་ཐག་བཅད་ཡོད་འདུག  གཞན། སྔར་ལོའི་ཕྱི་ཟླ་ ༡༢ ནང་རྒྱ་གཞུང་གིས་ལྷ་སར་འཛིན་བཟུང་བྱས་པའི་འབྲི་རུ་རྫོང་ཚ་ཆུ་ཞང་རབ་བརྟན་དགོན་པའི་གྲྭ་ཚུལ་ཁྲིམས་སྙན་གྲགས་ལགས། ད་ཆར་ཡང་གར་ཡོད་ཅི་བྱུང་གི་གནས་ཚུལ་ཅི་ཡང་མ་ཤེས་པར་ལྷག་ཡོད་འདུག་ལ། རྒྱ་གཞུང་གིས་ད་ཆར་ཡང་འབྲི་རུ་རྫོང་དུ་དམག་མི་འགྲེམས་འཇོག་གིས་དམ་དྲག་ལྟ་རྟོག་བྱེད་ཀྱི་ཡོད་འདུག གྲྭ་སྐལ་བཟང་ཆོས་གླང་ལགས། གྲྭ་ཚུལ་ཁྲིམས་སྙན་གྲགས་ལགས།   གསར་འགོད་པ། དོན་གོ
