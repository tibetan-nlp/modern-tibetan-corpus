ཆོས་དཔོན་ལྐོག་གསོད་པར་ཁྲིམས་ཆད།

ཟླ་འདིའི་ཕྱི་ཚེས་ ༥ ཉིན་ཤིན་ཅང་ཁཱ་ཧྲིན་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་གིས་ཆོས་དཔོན་ཞིག་ལྐོག་གསོད་བྱས་པའི་ཉེས་མིང་འོག མི་ ༢ ལ་སྲོག་ཁྲིམས་དང་གཅིག་ལ་ཚེ་བཙོན་གྱི་ཁྲིམས་ཐག་བཅད་འདུག ཁ་སང་རྒྱ་གཞུང་གིས་ཕྱི་བསྒྲགས་བྱས་པ་ལྟར་ན། འདི་ལོའི་ཕྱི་ཟླ་ ༧ ཚེས་ ༣༠ ཉིན་ཞོགས་པར་ཤིན་ཅང་ཡུ་གུར་མི་རིགས་རང་སྐྱོང་ཁུལ་ཁཱ་ཧྲིན་གྲོང་ཁྱེར་དུ་ཡོད་པའི་རྒྱ་ནག་གི་ཁ་ཆེའི་ལྷ་ཁང་ཆེ་ཤོས་ཧེ་གྷ་མེ་སུ་ཆི་ཊི་(Hëytgah Meschiti)ཡི་ཆོས་དཔོན་ཇོམ་ཊཱ་ཧིར་(jume tahir)ཟེར་བ་ལྐོག་གསོད་བྱས་པའི་ཉེས་མིང་འོག ཟླ་འདིའི་ཕྱི་ཚེས་ ༥ ཉིན་ཤིན་ཅང་ཁཱ་ཧྲིན་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་གིས་མི་རིགས་གང་ཡིན་མ་བཀོད་པའི་ལོ་ན་ ༡༨ ཡིན་པ་གྷི་ནི་ཧཱ་སན་(Gheni Hasan)དང་། ལོ་ ༡༩ ཡིན་པའི་ནཱར་མེ་མེ་ཊི་ཨེ་བྷི་ཌི་མེ་མེ་ཊི་(Nurmemet Abidilimit)ཟེར་བ་གཉིས་ལ་སྲོག་ཁྲིམས་དང་། ཨེ་ཊཱ་ཝུ་ལཱ་ཊོར་སན་ཟེར་བ་ཞིག་ལ་ཚེ་བཙོན་བཅས་ཀྱི་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག ཆོས་དཔོན་ལྐོག་གསོད་ཀྱི་དོན་རྐྱེན་བྱུང་རྗེས། ཁྰ་ཧྲིན་ཉེན་རྟོག་ཁང་གིས་དོགས་ཡོད་ཉེས་ཅན་གཉིས་ཀྱང་འཛིན་བཟུང་བྱེད་སྐབས་མེ་མདའ་བརྒྱབ་ནས་དམར་གསོད་བཏང་ཡོད་འདུག རྒྱ་ནག་ཉིན་རེའི་ཚགས་པར་ཐོག ལྐོག་གསོད་བྱེད་མཁན་གྱི་ཉེས་ཅན་ཚོར་ཕྱི་ཕྱོགས་ནས་བརྒྱུད་སྐུལ་བྱས་པར་བརྟེན། ལྐོག་གསོད་ཀྱི་ཉེས་དོན་བཟོས་ཡོད་སྟབས། རྗེས་ཕྱོགས་དེ་རིགས་ཀྱི་གནད་དོན་མི་འབྱུང་ཆེད་ལྐོག་གསོད་བྱེད་མཁན་ལ་ཁྲིམས་ཆད་ལྗིད་མོ་བཅད་ཡོད་ལུགས་བསྒྲགས་ཡོད་ཀྱང་། འཇར་མན་དུ་ཡོད་པའི་འཛམ་གླིང་ཡུ་གུར་མི་རིགས་མཐུན་ཚོགས་ཀྱི་མགྲིན་ཚབ་པ་ཌི་ལི་ཞ་ཡིས། རྒྱ་གཞུང་གིས་ཡུ་གུར་བ་ཚོའི་ཐོག་བཟོད་བསྲན་བྲལ་བའི་དྲག་གནོན་གྱི་སྲིད་བྱུས་སྤེལ་བའི་རྐྱེན་གྱིས་གནད་དོན་འདི་རིགས་འབྱུང་བཞིན་ཡོད་པ་དང་། ཆོས་དཔོན་ལྐོག་གསོད་བྱས་པའི་ཉེས་མིང་གཡོགས་ཏེ་ཁྲིམས་ཆད་ལྗིད་མོ་བཅད་པ་དེས། ཡུ་གུར་མི་རིགས་ཀྱི་འཁོན་སེམས་མཚམས་འཇོག་ཡོང་བར་ཕན་ཐོགས་མེད་པར་སྔར་ལས་ཤུགས་ཆེ་རུ་འགྲོ་རྒྱུ་ཡིན་སྟབས། རྒྱ་གཞུང་གིས་ཡུ་གུར་མི་རིགས་ཀྱི་ཆོས་ལུགས་དང་འཚོ་བའི་གོམས་སྲོལ་ལ་བརྩི་བཀུར་བྱས་ཏེ། གནད་དོན་གསར་པ་མི་སློང་བའི་འབོད་སྐུལ་བྱས་འདུག ཆོས་དཔོན་དེ་ལོ་ན་ ༧༤ ཡིན་ཞིང་། ཡུ་གུར་མི་རིགས་རང་སྐྱོང་ཁུལ་ཁ་ཆེ་ཆོས་ལུགས་མཐུན་ཚོགས་ཀྱི་ཚོགས་གཙོ་གཞོན་པ་ ཡིན་ལ། ཁོ་བོ་རྒྱ་ནག་གི་ཡུ་གུར་བ་འབུམ་བརྒྱ་ ༡ ཙམ་གྱི་ནང་ནས་རྩོད་གླེང་ཆེན་པོ་ཡོད་པའི་མི་སྣ་ཞིག་ཡིན་འདུག རྒྱ་གཞུང་གི་བརྒྱུད་ལམ་ཁག་ནས་ཁོ་བོ་སྲིད་གཞུང་གི་ཁྲིམས་འགལ་གྱི་ཆོས་ལུགས་ བྱ་འགུལ་བཀག་སྡོམ་ལ་མཐའ་གཅིག་ཏུ་རྒྱབ་སྐྱོར་བྱེད་མཁན་གྱི་མི་སྣ་གལ་ཆེན་ཡིན་ སྐོར་བསྒྲགས་ཡོད་པ་དང་། ཁ་ཆེ་ཆོས་ལུགས་བྱ་འགུལ་བཀག་སྡོམ་ཟེར་བའི་ནང་གཞོན་སྐྱེས་ཚོ་ཁ་ཆེའི་ལྷ་ཁང་དུ་ འཛུལ་བར་བཀག་འགོག་དང་། ཁ་ཆེའི་གསུང་རབ་སློབ་གྲྭ་བཀག་འགོག བུད་མེད་ཚོས་གདོང་རས་གྱོན་པ་དང་སྐྱེས་པ་ཚོས་ཨ་ར་རིང་པོ་འཇོག་པར་བཀག་འགོག་བྱེད་པ་སོགས་ཚུད་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
