རྒྱ་ནག་གིས་བཙན་འཛུལ་བྱེད་ཀྱི་འདུག

ཁ་སང་རྒྱ་ནག་གི་མཚོ་དམག་གྲུ་གཟིངས་གསུམ་གྱིས་མེ་ནཱ་མེ་ཁོ་ཇེ་མཱ་ (Minamikojima)ཞེས་པའི་སཱིན་ཁཱ་ཁའོ་གླིང་ཕྲན་གྱི་མཚོ་ཁུལ་དུ་བཙན་འཛུལ་བྱས་འདུག ཁ་སང་སྔ་དྲོའི་ཆུ་ཚོད་ ༡༠ པའི་ཐོག ཉི་ཧོང་དང་རྒྱ་ནག་གིས་བདག་དབང་རྩོད་ལེན་བྱེད་བཞིན་པའི་ཞི་བདེ་རྒྱ་མཚོའི་མངའ་ཁུལ་སཱིན་ཁཱ་ཁའོ་གླིང་ཕྲན་གྱི་(Minamikojima)ཞེས་པའི་སྤྱི་གཡར་མཚོ་ཁུལ་དུ། རྒྱ་ནག་གི་མཚོ་དམག་གྲུ་གཟིངས་ ༢༡༥༡ དང་ ༢༡༡༣། ༢༡༠༢ རྟགས་ཅན་གསུམ་གྱིས་མེ་ནཱ་མེ་ཁོ་ཇེ་མཱ་དུ་བཙན་འཛུལ་གྱིས་ཆུ་ཚོད་ ༢ རིང་སྐོར་གཡེང་བྱས་ཏེ་ཕྱིར་ལོག་ཡོད་པ་དང་། ཉི་ཧོང་གཞུང་གིས་རྒྱ་ནག་གི་བྱ་སྤྱོད་དེར་སྐྱོན་བརྗོད་ཤུགས་ཆེན་བྱས་འདུག  ཉི་ཧོང་སྲིད་བློན་ཤིང་ཛོ་ཨ་བྷེ་ལགས་ཀྱིས། “སཱིན་ཁཱ་ཁའོ་གླིང་ཕྲན་ནི་རྒྱལ་སྤྱིའི་ཁྲིམས་ལུགས་དང་ལོ་རྒྱུས་གང་ནས་བལྟས་ཀྱང་ཉི་ཧོང་གི་ཁ་བྲལ་ཐབས་མེད་པའི་ཆ་ཤས་ཤིག་ཡིན་ཡང་། ཉེ་ལམ་རྒྱ་གཞུང་གིས་ཞི་བདེ་རྒྱ་མཚོའི་མངའ་ཁུལ་ཐོག་མཁའ་དབྱིངས་དབྱེ་མཚམས་ས་ཐིག་གསར་བཟོ་དང་། སཱིན་ཁཱ་ཁའོ་མཚོ་གླིང་གི་བདག་དབང་རྒྱ་ནག་ལ་ཡོད་ཅེས་ཕྱིར་བསྒྲགས་ཀྱིས་མཚོ་དམག་གྲུ་གཟིངས་བཏང་སྟེ་འཇིགས་སྐུལ་བྱེད་བཞིན་ཡོད་སྟབས། རྒྱལ་སྤྱིའི་སྒྲིག་འཛུགས་ཁག་གིས་རྒྱ་གཞུང་ལ་གནོན་ཤུགས་སྤྲད་དེ་གྲོས་མོལ་ལམ་ནས་གནད་དོན་དེ་སེལ་དགོས་པའི་སྐུལ་མ་བྱ་དགོས”ཞེས་བརྗོད་འདུག རྒྱ་ནག་གི་ཕྱི་སྲིད་བློན་ཆེན་ཝང་ཡེས། རྒྱ་ནག་གིས་ཉི་ཧོང་དང་མཉམ་དུ་སཱིན་ཁཱ་ཁའོ་གླིང་ཕྲན་གྱི་བདག་དབང་རྩོད་ལེན་ཐད་ཀྱི་གནད་དོན་དེ་ཞི་བའི་གྲོས་མོལ་བརྒྱུད་སེལ་འདོད་ཡོད་ཀྱང་། གྲོས་མོལ་དེ་མགོ་འཛུགས་མ་བྱས་པའི་སྔོན་ལ་ཉི་ཧོང་གིས་མཚོ་གླིང་དེ་རྒྱ་ནག་གིས་ཁ་བྲལ་ཐབས་མེད་པའི་ཆ་ཤས་ཡིན་པ་ངོས་འཛིན་བྱེད་དགོས་སྐོར་བརྗོད་འདུག ཨ་རིའི་རྒྱལ་དོན་དྲུང་ཆེ་ཇོན་ཀེ་རའེ་ལགས་ཀྱིས། རྒྱ་ནག་གཞུང་གིས་ཉི་ཧོང་དང་རྒྱ་ནག་བར་འཕྲད་བཞིན་པའི་ཞི་བདེ་རྒྱ་མཚོའི་མངའ་ཁུལ་མཚོ་གླིང་རྙོག་དྲ་དེ་རྒྱལ་སྤྱིའི་ཁྲིམས་ལུགས་དང་མཐུན་པར་ཞི་བའི་ངང་ནས་ཐག་གཅོད་བྱས་ཏེ། ཨེ་ཤ་ཡའི་མངའ་ཁུལ་ནང་ཞི་བདེ་བརྟན་བརླིང་ཡོང་བ་བྱ་དགོས་སྐོར་བརྗོད་འདུག   ཨེ་ཤ་ཡའི་སྟོབས་ཆེན་རྒྱལ་ཁབ་ཉི་ཧོང་དང་རྒྱ་ནག་གཉིས་ཀྱི་འབྲེལ་ལམ་དེ་ ༢༠༡༣ ལོའི་ཟླ་ ༡༡ པའི་ཚེས་ ༧ ཉིན་རྒྱ་གཞུང་གིས་ཞི་བདེ་རྒྱ་མཚོའི་མངའ་ཁུལ་མཚོ་གླིང་ཐོག་མཁའ་དབྱིངས་དབྱེ་མཚམས་ས་ཐིག་གསར་བཟོ་དང་། ༢༠༡༣ ལོའི་ཟླ་ ༡༢ པའི་ཚེས་ ༢༩ ཉིན་སྲིད་བློན་ཤིང་ཛོ་ཨ་བྷེ་ལགས་ཀྱིས་རྒྱ་ནག་གིས་འཁོན་འཛིན་ཤུགས་ཆེན་བྱེད་བཞིན་པའི་འཛམ་གླིང་དམག་ཆེན་གཉིས་པའི་ཁྲོད་འདས་གྲོངས་སུ་ཕྱིན་པའི་ཉི་ཧོང་གི་དམག་མི་ཚོའི་རྗེས་དྲན་མཆོད་འབུལ་ལྷ་ཁང་ཡཱ་སུ་ཁཱོ་ཎེ་ (Yasukuni shrine) ཞེས་པའི་ནང་ཆེས་མཐོའི་མཆོད་འབུལ་དུ་བསྐྱོད་པ་སོགས་ལ་བརྟེན་རྒྱལ་ཁབ་གཉིས་ཀྱི་འབྲེལ་ལམ་ལ་དོ་ཕོག་ཆེན་པོ་བྱུང་ཡོད། གསར་འགོད་པ། དོན་གྲུབ།
