སང་ཉིན་ཧིན་སྒོར་གསར་པ་ ༢༠༠
ཅན་བསླེབ་རྒྱུ་རེད།

རྒྱ་གར་དབུས་དངུལ་ཁང་གིས་ཧིན་སྒོར་ ༢༠༠༠ ཅན་བརྗེ་ལེན་གྱི་དཀའ་ངལ་སེལ་ཆེད་མི་འགྱངས་བར་ཧིན་སྒོར་ ༢༠༠ ཅན་སྤྱི་ཚོགས་ནང་དངོས་སུ་འགྲེམ་སྤེལ་བྱ་འཆར་ཡོད་འདུག དེ་རིང་གི་རྒྱ་གར་ཞེས་པའི་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན།  ཁ་སང་རྒྱ་གར་དཔལ་འབྱོར་ལྷན་ཁང་གིས་རྒྱ་གར་ནང་ཧིན་སྒོར་ ༢༠༠ ཅན་གསར་པ་ཞིག་གང་མགྱོགས་སྤྱི་ཚོགས་ནང་དངོས་སུ་འགྲེམ་འཆར་ཡོད་སྐོར་བསྒྲགས་ཡོད་པ་དང་།  དེ་ནི་རྒྱ་གར་གྱི་ལོ་རྒྱུས་ནང་ཧིན་སྒོར་  ༢༠༠ ཅན་ཐོག་མར་གསར་བཟོ་བྱས་པ་ཞིག་ཆགས་ཡོད་འདུག རྒྱ་གར་དཔལ་འབྱོར་ལྷན་ཁང་གིས་བསྒྲགས་པ་ལྟར་ན།  རྒྱ་གར་གྱི་ཧིན་སྒོར་ ༢༠༠ ཅན་དེ་རྒྱ་གར་དབུས་དངུལ་ཁང་གི་འགན་འཛིན་ལྷན་ཚོགས་ཀྱིས་ལམ་སྟོན་འོག་གསར་བཟོ་བྱས་པ་ཞིག་ཡིན་པ་དང་།  རྒྱ་གར་གྱི་དངུལ་ལོར་བཟོ་སྐྲུན་ཁང་(Bharatiya Reserve Bank Note Mudran)ཞེས་པ་དེས། རྒྱུན་དུ་མ་སུ་རི་དང་སལ་བྷོ་ནེ་(Salboni)་གཉིས་ནས་ཧིན་སྒོར་གསར་བཟོ་བྱས་ཏེ་རྒྱ་གར་དབུས་དངུལ་ཁང་ལ་རྩིས་སྤྲོད་བྱེད་བཞིན་ཡོད་སྟབས།  ཐེངས་འདིའི་ཧིན་སྒོར་ ༢༠༠ ཅན་གསར་བཟོའི་འགན་དབང་དངུལ་ལོར་བཟོ་སྐྲུན་ཁང་དེར་སྤྲད་ཡོད་འདུག དེ་རིང་ཉིན་རྒྱབ་ཏུ་རྒྱ་གར་བང་ཆེན་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན།  རྒྱ་གར་དབུས་དངུལ་ཁང་གིས་ཕྱི་ཟླ་ ༨ པའི་ཚེས་  ༢༥ ཉིན་ནས་མགོ་བཙུགས་ཏེ། ཧིན་སྒོར་གསར་པ་ ༢༠༠ ཅན་དེ་སྤྱི་ཚོགས་ནང་དངོས་སུ་འགྲེམས་སྤེལ་བྱ་རྒྱུ་ཡིན་སྐོར་བརྗོད་འདུག   འོན་ཀྱང་།  རྒྱ་གར་གཞུང་གིས་ཧིན་སྒོར་གསར་པ་དེའི་ཁ་དོག་གཙོས་རྒྱུ་ཆ་སོགས་ངོ་སྤྲོད་བྱས་མེད་སྟབས།  བརྒྱུད་ལམ་ཁག་ཏུ་མི་མང་པོ་ཞིག་གིས།  ན་ནིང་ཕྱི་ཟླ་ ༡༡ པའི་ཚེས་  ༨ ཉིན་གསར་བཟོ་བྱས་བའི་ཧིན་སྒོར་  ༥༠༠ ཅན་དང་  ༢༠༠༠ ཅན་གཉིས་དང་འདྲ་བའི་མདོག་དམར་སྨུག་ཡིན་པའི་ཧིན་སྒོར་  ༢༠༠ ཅན་ཞིག་ཁྱབ་སྤེལ་གཏོང་བཞིན་ཡོད་པ་རེད། དེ་སྔ་མུ་དྷེ་གཞུང་གིས་རྒྱ་གར་ནང་ནག་ཚོང་དང་ལྐོག་ཟ་རུལ་སུངས།  དེ་བཞིན་ཁྲིམས་འགལ་གྱིས་དངུལ་ཉར་བ་དང་དངུལ་རྫུན་མ་བཀྲམ་པ་སོགས་བཀག་སྡོམ་བྱེད་ཆེད་ཧིན་སྒོར་གསར་བཟོའི་ལས་འགུལ་དེ་བརྩམས་ཡོད་སྐོར་བརྗོད་ཡོད་ཀྱང་། སྐབས་དེར་བཏོན་པའི་ཧིན་སྒོར་གསར་པ་ ༢༠༠༠ ཅན་ལ་སྤྱི་ཚོགས་ནང་བརྗེ་ལེན་བྱེད་པར་དཀའ་ངལ་ཡོད་པའི་སྐྱོན་བརྗོད་ཤུགས་ཆེན་བྱས་ཡོད་སྟབས།  རྒྱ་གར་གཞུང་གིས་ད་ལྟ་ཧིན་སྒོར་ ༢༠༠ ཅན་གསར་བཟོ་བྱས་པ་དེས་སྤྱི་ཚོགས་ནང་གི་དངུལ་བརྗེ་ལེན་གྱི་དཀའ་ངལ་དེ་སེལ་ཐུབ་པའི་རེ་བ་རྒྱག་གི་ཡོད་འདུག རྒྱ་གར་དབུས་དངུལ་ཁང་གིས་ཧིན་སྒོར་གསར་པ་ ༢༠༠ ཅན་བཟོས་པ་དང་ལྷན་དུ།  ད་དུང་ཧིན་སྒོར་ ༥༠ ཅན་གསར་བཟོ་བྱས་ཏེ་སྤྱི་ཚོགས་ནང་འགྲེམས་སྤེལ་བྱ་རྒྱའི་འཆར་གཞི་ཡོད་པ་མ་ཟད།  ཧིན་སྒོར་ ༥༠ ཅན་རྙིང་པ་དེ་སྔར་བཞིན་བེད་སྤྱོད་གཏོང་ཆོག་གི་ཡོད་ཅེས་བརྗོད་ཡོད་པ་རེད། ཕྱི་ལོ་ ༡༩༣༥ ལོར་རྒྱ་གར་དབུས་དངུལ་ཁང་ཐོག་མར་བཙུགས་ཡོད་པ་དང་། ཕྱི་ལོ་  ༡༩༣༨ ལོར་རྒྱ་གར་ནང་ཐོག་མར་ཧིན་སྒོར་ ༥ ཅན་དང་། དེ་ནས་རིམ་བཞིན་ཧིན་སྒོར་ ༡༠ ཅན་དང་། ༡༠༠ ཅན།  ༡༠༠༠ ཅན།  དེ་བཞིན།  ཧིན་སྒོར་ ༡༠༠༠༠ ཅན་སོགས་གསར་བཟོ་བྱས་ཡོད་སྟབས།  ཧིན་སྒོར་ ༡༠༠༠༠ ཅན་དེ་རྒྱ་གར་གྱི་ཧིན་སྒོར་གནས་རིམ་མཐོ་ཤོས་ཆགས་ཡོད་ཀྱང་། ཕྱི་ལོ་༡༩༧༨ ལོར་ཧིན་སྒོར་ ༡༠༠༠༠ ཅན་དེ་མེད་པ་བཟོས་ཡོད་འདུག  གསར་འགོད་པ།  སྐལ་བཟང་སྤྱིན་པ།
