ཨ་ཁུ་འཇིགས་མེད་ལགས་སྨན་ཁང་དུ་ལྷག་འདུག

བོད་ནང་གིི་གནས་སྟངས་ཕྱིར་བསྒྲགས་བྱས་ཏེ་རྒྱའི་བཙོན་འོག་ཏུ་ཚུད་པའི་ཆབ་སྲིད་བཙོན་ཟུར་བླ་བྲང་ཨ་ཁུ་འཇིགས་མེད་ལགས་སུ་སྙིང་དང་ཁྲག་ཤེད་སོགས་ཀྱི་ནད་གཞི་སྣ་ཚོགས་ཀྱི་རྐྱེན་པས་ད་ལྟ་སྨན་ཁང་དུ་ལྷག་ཡོད་འདུག ད་ལྟ་རང་གནས་ཚུལ་འབྱོར་བ་ལྟར་ན། ཨ་ཁུ་འཇིགས་མེད་གློད་གྲོལ་བཏང་བའི་ཉིན་བཅུ་གྲངས་ཤིག་གི་རྗེས་ཏེ་ ༢༠༡༦ ཟླ་ ༡༡ ཚེས་ ༡༡ ཉིན་ཁོང་གཟུགས་གཞིར་བརྟག་དཔྱད་ཆེད་ལན་ཀྲུའུ་གྲོང་ཁྱེར་དུ་ཕྱིན་ཡོད་པ་དང་། དེ་གའི་བརྟག་དཔྱད་ཁང་ནས་རྗེས་སུ་ནད་རྟགས་ཐོན་གསལ་ལྟར་ན། ཟ་ཁུ་གཅིན་སྙི(糖尿病)དང་། སྙིང་ནད། མཆིན་ནད། ཁྲག་ཤེད་མཐོ་བ། མིག་ནད་བཅས་ན་ཚ་ཁག་ཅིག་ངོས་འཛིན་བྱས་འདུག རྗེས་སུ་མགོ་ན་བ་སོགས་གཟུགས་ཁམས་ཧ་ཅང་མ་བདེ་བའི་རྐྱེན་གྱིས་དེ་རིང་(༢༠༡༦/༡༢/༢༦)ཉིན་བླ་བྲང་བསང་ཆུ་རྫོང་མི་དམངས་སྨན་ཁང་དུ་བསྐྱར་དུ་བརྟག་དཔྱད་བཏང་བར། སྨན་ཁང་གིས་ན་ཚ་ཚབས་ཆེན་ཡིན་པས་ངེས་པར་སྨན་ཁང་ནང་འདུག་དགོས་ཚུལ་བཤད་པ་ལྟར་ད་ལྟ་རྫོང་སྨན་ཁང་སྨན་བཅོས་བྱེད་བཞིན་ཡོད་འདུག གནས་ཚུལ་ནང་འཁོད་དོན། ཨ་ཁུ་འཇིགས་མེད་ཁོང་འཛིན་བཟུང་མ་བྱས་སྔོན་ལ་ནད་ཚ་གང་ཡང་མེད་ཀྱང་ཕྱི་ལོ་ ༢༠༡༡ ཟླ་ ༨ ཚེས་ ༢༠ ཉིན་འཛིན་བཟུང་བྱས་ཏེ། ཟླ་གཉིས་ཀྱི་མཚམས་སུ་གཟུགས་གཞི་མི་བདེ་བའི་གནས་སྟངས་འོག་གོང་རིམ་ལ་ཡང་ཡང་སྨན་པར་བསྟེན་དགོས་པའི་རེ་སྐུལ་བྱས་ཀྱང་། ཟླ་བཞིའི་རྗེས་ནས་མ་གཏོགས་བརྟག་དཔྱད་བཏང་པར་དེ་མུར་བཞག་ཅིང་། རྗེས་སུ་བཙོན་པའི་སྨན་ཁང་ནས་བརྟག་དཔྱད་བཏང་བར་གཅིན་སྙིའི་ན་ཚ་ཡོད་པར་ངོས་འཛིན་གྱིས་བཙོན་པའི་སྨན་ཁང་ནང་ཟུར་འཇོག་བྱས་ཡོད་འདུག ༢༠༡༦ ཟླ་ ༡༠ ཚེས་ ༢༦ ཉིན་ལོ་ལྔའི་ཁྲིམས་ཆད་དུས་ཡུན་རྫོགས་ཏེ་བཙོན་གྲོལ་བཏང་ཡོད་ཀྱང་། འགྲོ་འདུག་གི་བྱ་སྤྱོད་སོགས་ལ་སྟངས་འཛིན་བྱེད་བཞིན་ཡོད་པ་དང་། དགོན་པར་སྡོད་བཅུག་གི་མེད་པ་མ་ཟད། གཟན་ཤམ་ཡང་གྱོན་དུ་འཇུག་གི་མེད་པ་དང་། དེ་སྔ་འཛིན་བཟུང་བྱས་སྐབས་བཙན་འཕྲོག་བྱས་པའི་ཁོང་གི་ཐོབ་ཐང་ལག་འཁྱེར་དང་གློག་ཀླད། ཡིག་ཆ་ཁག དེ་བཞིན། སྒོར་ཁུག་སོགས་ཀྱང་ཕྱིར་རྩིས་སྤྲོད་བྱས་མེད་ཅིང་། དེའི་ཐད་ཐེངས་གསུམ་ལ་ཅ་དངོས་དེ་དག་ཕྱིར་སྤྲོད་དགོས་པའི་རེ་འདུན་བཏོན་ཡང་ཡ་ལན་གང་ཡང་སྤྲད་མི་འདུག དེང་སྐབས་ཁོང་ལ་ཐོབ་ཐང་ལག་འཁྱེར་མེད་སྟབས་ལན་ཀྲུའུ་གྲོང་ཁྱེར་གྱི་སྨན་ཁང་སོགས་སུ་སྡོད་ཐབས་མེད་པའི་གནས་སྟངས་ཆགས་ཡོད་འདུག  གནས་སྟངས་འདིའི་ཐོག་ནང་མི་ཉེ་གྲོགས་ཚོས་སེམས་འཚབ་ཆེན་པོ་བྱེད་བཞིན་ཡོད་འདུག ཅེས་གོང་གི་གནས་ཚུལ་དེ་ཨ་ཁུ་འཇིགས་མེད་ལགས་ཀྱི་དགེ་ཕྲུག་དགེ་འདུན་ཕུན་ཚོགས་ཀྱིས་བཏང་འབྱོར་བྱུང་།
