རྒྱ་ནག་ནང་འབར་གས་དོན་རྐྱེན་བྱུང་།

ཧུའུ་པེ་ཞིང་ཆེན་ཏང་ཡང་གྲོང་ཁྱེར་གྱི་རྡོ་སོལ་གློག་འདོན་ཁང་ཞིག་ཏུ་འབར་གས་དོན་རྐྱེན་བྱུང་སྟེ་ཉུང་མཐར་མི་ ༢༡ རྐྱེན་འདས་དང་མི་ ༥ ལ་རྨས་སྐྱོན་བཟོས་ཡོད་འདུག དེ་རིང་རྒྱ་ནག་ལྷོ་རྒྱུད་ཞོགས་པའི་ཚགས་པར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཁ་སང་ཉིན་རྒྱབ་པེ་ཅིན་གྱི་ཆུ་ཚོད་ ༣།༢༠ སྟེང་རྒྱ་ནག་དབུས་རྒྱུད་ཧུའུ་པེ་ཞིང་ཆེན་ཏང་ཡང་གྲོང་ཁྱེར་གྱི་མཱ་ཏིན་རྡོ་སོལ་གློག་འདོན་ཁང་གི་སོལ་རླངས་ཀྱི་སྦུག་ལམ་ལ་ཆག་སྐྱོན་ཤོར་ཏེ་འབར་གས་དོན་རྐྱེན་བྱུང་ཡོད་པ་དང་། དོན་རྐྱེན་ཁྲོད་ཉུང་མཐར་མི་ ༢༡ རྐྱེན་འདས་སུ་ཕྱིན་ཡོད་པ་དང་། མི་ ༥ ལ་རྨས་སྐྱོན་བྱུང་ཡོད་པའི་ཁྲོད་ནས་མི་གསུམ་ལ་རྨས་སྐྱོན་ཚབས་ཆེན་ཕོག་ཡོད་འདུག ཏང་ཡང་གྲོང་ཁྱེར་སྤྱིའི་བདེ་འཇགས་ལྟ་ཞིབ་ཁང་གིས། ལས་བཟོ་བས་སོལ་རླངས་འདྲེན་བྱེད་ཀྱི་སྦུག་ལམ་ལ་ཚོད་ལྟ་བྱེད་སྐབས་གས་འཐོར་བྱུང་བ་ཡིན་སྐོར་བརྗོད་འདུག རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱིས་སྲིད་གཞུང་གི་དཔོན་རིགས་ཚོས་འདས་ལོའི་ཕྱི་ཟླ་ ༨ ཚེས་ ༡༢ ཉིན་ཐེན་ཅིན་གྲོང་ཁྱེར་གྱི་རྫས་འགྱུར་མཛོད་ཁང་ལ་འབར་གས་བྱུང་བའི་ཁྲོད་དུ་མི་ ༡༧༠ ལྷག་རྐྱེན་ལམ་དུ་ཤོར་བའི་དོན་རྐྱེན་དེའི་བསླབ་བྱ་བླངས་ཏེ་སྤྱིའི་བདེ་འཇགས་ལ་སྔར་བས་ལྟ་ཞིབ་ནན་མོ་བྱེད་དགོས་ལུགས་བཤད་འདུག རོའུ་ཊར་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བའི་ནང་། ཏང་ཡང་གྲོང་ཁྱེར་གྱི་གློག་འདོན་ཁང་དེ་ནི་ཕྱི་ལོ་ ༢༠༠༠ ལོར་བཙུགས་པ་དང་། གློག་འདོན་ཚོང་གཉེར་ཁང་དེས་སོལ་རླངས་དང་ཆུ་གློག རྡོ་སྣུམ་དང་ལྕགས་རིགས་སོགས་ཐོན་གཉེར་བྱེད་བཞིན་ཡོད་པ་དང་། ན་ནིང་ཐེན་ཅིན་གྲོང་ཁྱེར་གྱི་རྫས་འགྱུར་མཛོད་ཁང་དུ་འབར་གས་བྱུང་རྗེས། རྒྱ་ནག་གཞུང་གིས་ཐེན་ཅིན་གྲོང་ཁྱེར་གྱི་ཉེན་ཆས་རྫས་འགྱུར་མཛོད་ཁང་ ༥༨༣ ལ་བཟོ་བཅོས་བྱས་པ་དང་། རྫས་འགྱུར་ཁང་ ༦༨ ས་གནས་གཞན་དུ་སྤར་བ་དང་ཁག་ཅིག་གི་སྒོ་བརྒྱབ་ཡོད་ནའང་། འདས་པའི་ལོ་ངོ་ ༣༠ རིང་རྒྱ་ནག་གིས་དཔལ་འབྱོར་ཐོན་སྐྱེད་གོང་སྤེལ་བཏང་བ་དང་མཉམ་དུ། བཟོ་གྲྭ་ཁག་ཏུ་མེ་སྐྱོན་དང་གཏེར་ཁ་རྡིབ་སྐྱོན་བཅས་ཀྱི་དོན་རྐྱེན་རིམ་པར་འབྱུང་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
