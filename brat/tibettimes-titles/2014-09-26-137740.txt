སྲིད་བློན་མུ་དྷེ་ཨ་རིར་སྐོར་སྐྱོད།

ཁ་སང་ནས་བཟུང་རྒྱ་གར་སྲིད་བློན་མུ་དྷེ་ཨ་རིའི་ནང་སྐོར་སྐྱོད་མགོ་འཛུགས་བྱས་ཡོད་ཅིང་ཚེས་ ༢༩ ཉིན་སྲིད་འཛིན་ཨོ་བྷ་མ་དང་ཐུག་འཕྲད་གྱིས་རྒྱལ་ཁབ་གཉིས་བར་བདེ་སྲུང་དང་ཚོང་དོན་སྐོར་ལ་གྲོས་ཆོད་འཇོག་རྒྱུ་ཡིན་འདུག ཟླ་འདིའི་ཚེས་ ༢༥ ནས་ ༣༠ ཉིན་བར་རྒྱ་གར་གྱི་སྲིད་བློན་ནཱ་རེན་ཌ་མུ་དྷེ་ཨ་རིར་གཟིགས་མགོ་འཛུགས་བྱས་ཡོད་ཅིང་། ཚེས་ ༢༧ ཉིན་ནེའུ་ཡོག་གྲོང་ཁྱེར་ནང་ཚོགས་བཞིན་པའི་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་ཚོགས་མིའི་རྒྱལ་ཁབ་ཁག་གི་དབུ་ཁྲིད་ལྷན་ཚོགས་ཐོག་གཏམ་བཤད་དང་། ཚེས་ ༢༩ དང་ ༣༠ བཅས་ཉིན་གཉིས་རིང་རྒྱལ་ས་ཝ་ཤིན་ཌོན་དུ་ཕེབས་ཏེ། སྲིད་འཛིན་ཨོ་བྷ་མ་དང་མཐོ་རིམ་དཔོན་རིགས་ཁག་ལ་ཐུག་འཕྲད་ཀྱིས་རྒྱལ་ཁབ་གཉིས་བར་དྲ་རྒྱའི་བདེ་འཇགས་དང་རྡུལ་ཕྲན་གློག་ཤུགས་མཁོ་འདོན། འཕྲོད་བསྟེན་དང་རིག་གཞུང་བརྗེ་ལེན། ཨ་རིས་རྒྱབ་སྐྱོར་འོག་རྒྱ་གར་གྱི་འཛུགས་སྐྲུན་འཕེལ་རྒྱས་གཏོང་ཐབས། རྒྱ་གར་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་གཏན་འཇགས་ཚོགས་མིའི་ཁོངས་སུ་གཞུག་ཐབས་བཅས་ཀྱི་སྐོར་ལ་གྲོས་མོལ་གནང་རྒྱུ་ཡིན་འདུག འདི་ནི་རྒྱ་གར་གྱི་སྲིད་བློན་ཨ་རིར་ཕེབས་ཏེ་སྲིད་འཛིན་ཨོ་བྷ་མ་དང་ཐུག་འཕྲད་གནང་བ་ཐོག་མ་དེ་ཆགས་ཡོད། འདིའི་ཐད་ནཱ་རེན་ཌ་མུ་དྷེ་ཡིས། “ངས་ད་ཐེངས་ཨ་རིའི་སྲིད་འཛིན་ལ་ཐུག་འཕྲད་བྱས་པ་ཐོག་མ་ཡིན། ཁོང་ད་ལྟ་རྒྱལ་སྤྱིའི་ཐོག་སྙན་གྲགས་ཆེ་བའི་མི་སྣ་ཆགས་དགོས་དོན་ནི་དམངས་གཙོའི་ཐོབ་ཐང་དང་གོ་སྐབས་བཀོལ་སྤྱོད་བྱས་པ་ལས་བྱུང་བ་ཞིག་ཡིན་སྟབས། འཛམ་གླིང་མི་མང་ཚོས་མིག་དཔེ་ལྟ་ས་ཞིག་ཆགས་ཡོད། ཨ་རི་ནི་ཤེས་ཡོན་དང་ལས་རྩོལ། འཕྲུལ་ལས་ཚན་རྩལ། གསར་གཏོད་ཀྱི་ཤེས་བྱས་ཕྱུག་པའི་རྒྱལ་ཁབ་ཅིག་ཡིན་སྟབས། ཨ་རི་དང་ཚོང་དོན། དམག་དོན་གཙོས་གནད་དོན་ཁག་ལ་མཉམ་ལས་བྱས་ཏེ། འཛམ་གླིང་ནང་ཞི་བདེ་དང་བརྟན་ལྷིང་ཡོང་བ་བྱ་རྒྱུ་ཡིན།”ཞེས་བརྗོད་འདུག   ནཱ་རེན་ཌ་མུ་དྷེ་ ༡༩༩༠ ལོར་བྷེ་ཇེ་པི་ཚོགས་པའི་སྤྱི་ཁྱབ་དྲུང་ཆེ་གནང་སྐབས་ཨ་རིར་སྐོར་སྐྱོད་ཐེངས་དང་པོ་བྱས་ཡོད་ཅིང་། དེའི་རྗེས་ ༡༩༩༣ ལོར་ཐེངས་གཉིས་པ་དང་། ༢༠༠༥ ལོར་སྐོར་སྐྱོད་ཐེངས་གསུམ་པ་བྱ་རྒྱུའི་ཐག་གཅོད་གནང་ཡོད་ཀྱང་། སྐབས་དེར་ཨ་རིའི་གཞུང་གིས་ནཱ་རེན་ཌ་མུ་དྷེ་གྷུ་ཅུ་ར་ཌེ་མངའ་སྡེའི་སྤྱི་ཁྱབ་བློན་ཆེན་བྱེད་སྐབས་ཁ་ཆེ་མི་རིགས་ ༢༠༠༠ ལྷག་དམར་གསོད་བཏང་ཡོད་པའི་སྐྱོན་འཛུགས་འོག ཨ་རིར་སྐྱོད་པའི་ཐོངས་མཆན་ལག་འཁྱེར་སྤྲོད་རྒྱུ་མིན་པའི་ཐག་གཅོད་བྱས་རྗེས་ད་བར་ཨ་རིར་སྐྱོད་ཐུབ་མེད་སྟབས། ད་རེབ་ནཱ་རེན་ཌ་མུ་དྷེ་ཨ་རིར་ཕེབས་པ་དང་བསྟུན་ཏེ་སྲིད་འཛིན་ཨོ་བྷ་མས་དགོང་དྲོའི་གསོལ་སྟོན་ཞིག་དང་། སྲིད་འཛིན་གཞོན་པ་ཇོའོ་བྷེ་ཌིན་གྱིས་གུང་ཚིགས་ཤིག་འདེགས་འབུལ་ཞུ་རྒྱུ་ཡིན་འདུག  ཟླ་འདིའི་ཚེས་ ༢༦ ཉིན་ནཱ་རེན་ཌ་མུ་དྷེ་ཡིས་ནེའུ་ཡོག་གྲོང་ཁྱེར་གྱི་སྤྱི་ཁྱབ་པ་བྷེལ་ཌི་བྷེ་ལཱ་སེའོ་ (Bill de Blasio) དང་སྨན་རྗས་རིག་པའི་ཐོག་ནོ་སྦེལ་གཟེངས་རྟགས་བཞེས་མྱོང་མཁན་(Harold Varmus) ཧཱ་རོལ་ཌི་ཝར་མུ་སེ་རྣམ་གཉིས་མཇལ་འཕྲད་བྱས་ཡོད་པ་དང་། སང་ཉིན་རྒྱ་གར་སྲིད་བློན་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་ལྷན་ཚོགས་ཐོག་གཏམ་བཤད་དང་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་སྤྱི་ཁྱབ་དྲུང་ཆེ་བྷན་ཁེ་མོན་དང་། རྒྱལ་ཚོགས་ཀྱི་ལྷན་ཚོགས་སུ་མཉམ་བཞུགས་གནང་བཞིན་ཡོད་པའི་བྷང་ལ་དྷེ་ཤི་དང་བལ་ཡུལ་གྱི་སྲིད་བློན་དང་། སིང་གྷ་ལའི་སྲིད་འཛིན་བཅས་ལ་དང་ཐུག་འཕྲད་བྱ་རྒྱུ་ཡིན་པ་མ་ཟད། ཚེས་ ༢༩ ཉིན་རྒྱལ་ས་ཝ་ཤིན་ཌོན་དུ་སྲིད་འཛིན་ཨོ་བྷ་མ་དང་། སྲིད་འཛིན་ཟུར་པ་བྷེལ་ཁེ་ལོན་ཌོན་དང་ལྕམ་ཁེ་ལོན་ཌོན་སོགས་དང་མཇལ་འཕྲད་བྱ་རྒྱུ་ཡིན་འདུག  གསར་འགོད་པ། དོན་གྲུབ།
