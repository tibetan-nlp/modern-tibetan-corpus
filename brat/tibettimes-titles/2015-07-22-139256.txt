ལིན་ཅི་ཧྭ་ཏང་ནས་ཕྱིར་འཕུད་བཏང་།

ཁ་སང་རྒྱ་ནག་དབུས་ཆབ་སྲིད་ལྷན་ཁང་གིས་གྲོས་མོལ་ཚོགས་འདུ་ཞིག་བསྐོངས་ཏེ། རྒྱ་ནག་མི་དམངས་ཆབ་སྲིད་གྲོས་ཚོགས་ཀྱི་ཚོགས་གཙོ་གཞོན་པ་དང་འཐབ་ཕྱོགས་གཅིག་གྱུར་གྱི་འགན་འཛིན་གཞོན་པ་ཟུར་བ་ལིན་ཅི་ཧྭ་དངོས་སུ་ཏང་དང་རྒྱལ་ཁབ་ཀྱི་གོ་གནས་ནས་ཕབ་པ་ཡོངས་ཁྱབ་གསལ་བསྒྲགས་བྱས་ཏེ་ཁྲིམས་འཛིན་ཁང་དུ་བསྐྱལ་ཡོད་པ་ཁྱབ་བསྒྲགས་བྱས་འདུག སྔ་ལོར་རྒྱ་ནག་དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་དང་ཆེས་མཐོའི་མི་དམངས་ཁྲིམས་ཁང་། སྤྱི་བདེ་ལས་ཁུངས་བཅས་ཀྱི་ཞིབ་བཤེར་སྐབས་ལིན་ཅི་ཧྭ་(Ling Jihua) ཡི་གྱོད་གཞི་ཐོན་རྗེས། སྔ་ལོའི་ཕྱི་ཟླ་ ༡༢ པའི་ཚེས་ ༢༢ ཉིན་རྒྱ་ནག་དབུས་ཆབ་སྲིད་རྒྱུན་ལས་ཚོགས་ཆེན་སྟེང་ལིན་ཅིན་ཧྭ་ལ་ཞིབ་བཤེར་བྱ་རྒྱུའི་ཐག་གཅོད་བྱས་ཡོད་པ་དང་། ཁ་སང་རྒྱ་ནག་དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གིས་༼ ལིན་ཅི་ཧྭ་ཡི་ཚབས་ཆེའི་སྒྲིག་འགལ་གྱོད་དོན་སྙན་ཐོ།༽ ཞེས་པ་ཞིག་བཏོན་རྗེས། ཆབ་སྲིད་ལྷན་ཁང་གིས་ལིན་ཅི་ཧྭ་གུང་ཁྲན་ཏང་དང་རྒྱལ་ཁབ་ཀྱི་ལས་གནས་ནས་གནས་དབྱུང་བཏང་སྟེ། གྱོད་གཞི་ཡོད་ཚད་ཁྲིམས་འཛིན་ཁང་དུ་བསྐྱལ་ཏེ་ཁྲིམས་ལུགས་ལྟར་ལག་བསྟར་བྱ་རྒྱུ་ཡིན་པ་ཡོངས་ཁྱབ་ཏུ་བསྒྲགས་འདུག ཁ་སང་རྒྱ་ནག་ཤིན་ཧྭ་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ལིན་ཅི་ཧྭ་ལ་ཞིབ་བཤེར་བྱས་པའི་བརྒྱུད་རིམ་ཁྲོད། དབང་ཆ་སྒེར་སྤྱོད་བྱས་ཏེ་རང་ཉིད་དང་ནང་མི་ཚོས་གཞན་གྱི་ལྐོག་རྔན་འབོར་ཆེན་བླངས་པ་དང་། བུད་མེད་མང་པོར་ཚུལ་མིན་ཆགས་སྲེད་བྱས་པ། གཞན་གྱི་ནག་ཉེས་འགེབ་སྲུང་བྱས་པ་སོགས་ཀྱི་སྒྲིག་འགལ་ཚབས་ཆེན་བྱས་ཡོད་པ་དང་། ཁོའི་ནང་མི་དང་སྐྱེས་སྨན་སོགས་ཀྱིས་ཀྱང་ཁྲིམས་འགལ་ཚོང་ལས་གཉེར་ཏེ་ཁེ་སངས་ཆེན་པོ་བླངས་པ་སོགས་བགྲང་ཡོད་ཅིང་། ད་དུང་དོགས་གཞི་གཞན་ཡང་ཡོད་པ་སོགས་འཁོད་འདུག ཧོང་ཀོང་གསར་འགྱུར་བརྒྱུད་ལམ་ཐོག་འབྲེལ་ཡོད་མི་སྣ་ཞིག་གིས་བརྗོད་རྒྱུར། ལིན་ཅི་ཧྭ་ལ་དབུས་བརྙན་འཕྲིན་ཆབ་སྲིད་སྡེ་ཚན་གྱི་མདོ་འཛིན་པར་ལྐོག་འབྲེལ་ཡོད་པ་མ་ཟད། རྒྱ་ནག་དབུས་གཞུང་སྒྲུབ་ཁང་གི་འགན་འཛིན་བཞུགས་སྐབས། ཀྲུང་ནན་ཧེ་ཡི་ཁ་པར་ལས་ཁུངས་ཀྱི་ཞབས་ཞུ་མ་ཞིག་ཀྱང་ཟུར་བསྟན་དུ་ཉར་ཏེ་མཐོ་རིམ་དཔོན་རིགས་ཚོར་ལྟ་སྲུང་སྟངས་འཛིན་བྱས་ཡོད་པ་སོགས་བརྗོད་འདུག ཟླ་འདིའི་ཚེས་ ༧ ཉིན་པའོ་ཞོན་དྲ་བས་གནས་ཚུལ་སྤེལ་བའི་ནང་། ལིན་ཅི་ཧྭ་ཡི་ཁྲིམས་འགལ་ལུགས་འགལ་གྱི་གྱོད་དོན་དེ་རྒྱ་ནག་དབུས་གཞུང་གིས་པེ་ཏཱེ་གཙང་བོའི་ཚོགས་འདུའི་སྔོན་ལ་ཐག་གཅོད་བྱ་སྲིད་པ་དང་། གུང་ཁྲན་ཏང་གི་མཐོ་རིམ་དཔོན་རིགས་ཚོས་ལིན་ཅི་ཧྭ་ལ་ནག་ཉེས་དྲུག་འགེལ་སྲིད་སྐོར་བཀོད་པའི་ནང་། གོང་རིམ་ལ་དོན་དངོས་སྦ་གསང་བྱས་པ་དང་། ཏང་གི་མཐོ་རིམ་དཔོན་རིགས་ཚོའི་འཛིན་སྐྱོང་འདེམ་སྒྲུག་གི་ཁེ་དབང་རྒྱུན་རིང་བླངས་པ། ཕྱིའི་བརྒྱུད་ལམ་བེད་སྤྱོད་བཏང་སྟེ་དཀྲོག་གཏམ་དང་ཏང་ནང་གི་འཐབ་རྩོད་བསྒྲགས་པ།  རྒྱལ་ཁབ་ཀྱི་གསང་བ་ཕྱི་གྱར་བྱས་པ། ཁོའི་བུ་ལིན་ཀོ་ (Ling Gu)ཡི་རླངས་འཁོར་དོན་རྐྱེན་གྱི་དཔང་རྟགས་བུད་མེད་བསད་པ། རང་ཉིད་སྒེར་དང་ཁྱིམ་རྒྱུད་ཀྱིས་ལྐོག་ཟ་རུལ་སུངས་ཚབས་ཆེན་བྱས་པ་བཅས་བགྲངས་སྲིད་ཅེས་འཁོད་ཡོད་འདུག ཟླ་འདིའི་ཚེས་ ༨ ཉིན་ཨ་རིའི་རླུང་འཕྲིན་རྒྱ་སྐད་སྡེ་ཚན་གྱིས་པའོ་ཞོན་གསར་འགྱུར་ལས་ཁང་གསར་གཏོད་མཁན་ཝེ་ཧྲི་ (Wei Shi)ལ་བཅའ་འདྲི་བྱས་པའི་ནང་། དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གིས་ཕྱི་ལོགས་ཀྱི་གླེང་བཞིན་པའི་ནག་ཉེས་དྲུག་དེ་ལིན་ཅི་ཧྭ་ལ་བགྲང་མི་སྲིད། ཡིན་ན་ཡང་། མཐའ་མཇུག་རྒྱ་གཞུང་དང་ལིན་ཅི་ཧྭ་ཡི་ནུ་བོ་ལིན་ཝན་ཁྲིན་གཉིས་ཀྱི་ཐག་གཅོད་ལ་རག་ལས་ཡོད། དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ཁང་དང་བདེ་འཇགས་ལས་བྱེད་ཚོས་དུས་རྒྱུན་ལིན་ཝན་ཁྲིན་ལ་ཁ་ཏ་བྱས་ཏེ། ཕྱིར་རང་རྒྱལ་དུ་ལོགས་འཐབ་བྱེད་བཞིན་ཡོད་སྟབས་རྒྱ་ནག་གཞུང་གིས་ལིན་ཅི་ཧྭ་ལ་གུ་ཡངས་ཏོག་ཙམ་གཏོང་ངེས་ཡིན་སྐོར་སོགས་བརྗོད་འདུག གསར་འགོད་པ། འཇིགས་མེད་རྒྱ་མཚོ།
