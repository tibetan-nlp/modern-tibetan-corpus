ཀེམ་ཇོན་ཨང་གི་སྐྱོན་ཚང་བརྟོལ།

དེ་རིང་ལྷོ་ཀོ་རི་ཡའི་ཕྱི་སྲིད་བློན་ཆེན་གྱིས་ད་བར་བྱང་ཀོ་རི་ཡའི་མགོ་ཁྲིད་ཀེམ་ཇོན་ཨང་གིས་མི་ ༧༠ དམར་གསོད་བཏང་ཡོད་པའི་སྐྱོན་འཛུགས་བྱས་འདུག དེ་རིང་ལྷོ་ཀོ་རི་ཡའི་ཕྱི་སྲིད་བློན་ཆེན་ཡུང་བྷེ་ཡང་སེ་ (Yun Byung-se) ཡིས་རྒྱལ་ས་སོལ་ནང་འཚོགས་བཞིན་ཡོད་པའི་གླེང་སྟེགས་ཤིག་གི་སྟེང་། ༢༠༡༡ ལོ་ནས་ཀེམ་ཇོན་ཨང་གིས་བྱང་ཀོ་རི་ཡའི་དབང་ཆ་བཟུང་རྗེས། ད་བར་ཁོ་རང་དང་ཆབ་སྲིད་ལྟ་ཕྱོགས་མི་མཐུན་པའི་མཐོ་རིམ་དཔོན་རིགས་ཐེ་པའི་གནད་ཡོད་མི་སྣ་དང་མང་ཚོགས་ ༧༠ དམར་གསོད་བཏང་ཡོད་པ་མ་ཟད། འགྲོ་བ་མིའི་ཐོབ་ཐང་རྡོག་རོལ་བཏང་ནས་ལྟ་ཕྱོགས་མི་མཐུན་པའི་མང་ཚོགས་ལ་སྡུག་སྦྱོང་མནར་གཅོད་གཏོང་གི་ཡོད་པའི་སྐྱོན་འཛུགས་བྱས་འདུག ལྷོ་ཀོ་རི་ཡའི་ཡོན་ཧེབ་ (Yonhap) གསར་ཁང་གིས་ཕྱི་སྲིད་བློན་ཆེན་ཡུང་བྷེ་ཡང་སེའི་གཏམ་བཤད་ལུང་འདྲེན་བྱས་པའི་ནང་། ཀེམ་ཇོན་ཨང་གིས་བྱང་ཀོ་རི་ཡའི་དབང་ཆ་བཟུང་རྗེས། རྒྱལ་ཁབ་ནང་དྲག་པོའི་ཁྲིམས་གཞི་བསྡམས་ནས་མི་སྤྱོད་ལས་འདས་པའི་དྲག་གནོན་སྲིད་བྱུས་བཟུང་སྟེ། རང་ཉིད་དང་ལྟ་ཕྱོགས་མི་མཐུན་པའི་དཔོན་རིགས་དང་མི་སེར་དུ་མ་དམར་གསོད་བཏང་ཡོད་པ་མ་ཟད། བྱང་ཀོ་རི་ཡ་ནས་ལྷོ་ཀོ་རི་ཡ་ནང་བྲོས་བྱོལ་ལ་འགྲོ་མཁན་ཚོར་ལྷོ་ཀོ་རི་ཡའི་གཞུང་ལ་ངོ་ལོག་རྒྱག་དགོས་པའི་བརྒྱུད་སྐུལ་བྱས་ཏེ། རྒྱལ་ཁབ་གཉིས་ཀྱི་འཕེལ་རྒྱས་འབྲེལ་ལམ་ལ་སྔར་ལྷག་དོ་ཕོག་གཏོང་བཞིན་འདུག ཅེས་བརྗོད་འདུག ཨེ་ཕེ་གསར་ལམ་ཐོག་མིང་བརྗོད་འདོད་མེད་པའི་ལྷོ་ཀོ་རི་ཡའི་གསང་བའི་ལས་ཁུངས་ཀྱི་ལས་བྱེད་ཞིག་གིས། ད་བར་ཀེམ་ཇོན་ཨང་གིས་བྱང་ཀོ་རི་ཡའི་མི་ ༩༠ ཙམ་དམར་གསོད་བཏང་ཡོད་པའི་ཡིད་ཆེས་བྱེད་བཞིན་ཡོད་པ་མ་ཟད། སྨྲ་བརྗོད་རང་དབང་ཚོད་འཛིན་དང་དམག་ཤུགས་ཀྱི་ནུས་སྟོབས་ཆེ་རུ་གཏོང་བ་སོགས་ཀྱི་བྱ་འགུལ་ལུ་གུ་རྒྱུད་སྤེལ་ཏེ། ཁྱིམ་མཚེས་རྒྱལ་ཁབ་དང་རྒྱལ་སྤྱིའི་བདེ་འཇགས་ལ་ཉེན་ཁ་བཟོ་བཞིན་ཡོད། ཅེས་བརྗོད་འདུག ༡༩༤༨ ལོ་ནས་བཟུང་ཀེམ་གྱི་ཁྱིམ་བརྒྱུད་ཀྱིས་བྱང་ཀོ་རི་ཡ་དབང་སྒྱུར་བྱས་རྗེས། ཀེམ་རིམ་བྱོན་གྱིས་ཕྱི་ཕྱོགས་ལ་སྒོ་མོ་ཕྱེ་བའི་སྲིད་བྱུས་ལག་བསྟར་མ་བྱས་པར་རྒྱལ་ཁབ་བསྐྱངས་ཡོད་པ་མ་ཟད། ལྟ་ཕྱོགས་མ་མཐུན་པ་དང་གཞུང་ལ་དགག་པ་རྒྱག་མཁན་ཚོ་རྩ་མེད་བཟོས་ཏེ། མི་གཅིག་དབང་སྒྱུར་རིང་ལུགས་ཀྱི་དར་ཆ་འཕྱར་ནས་སྲིད་ཀྱི་འཁོར་ལོ་བསྐོར་བ་མ་ཚད། འགྲོ་བ་མིའི་ཐོབ་ཐང་ལ་རྡོག་རོལ་དང་སྨྲ་བརྗོད་རང་དབང་བཀག་སྡོམ་སོགས་བྱས་སྟབས། མཉམ་འབྲེལ་རྒྱལ་ཚོགས་གཙོས་པའི་རྒྱལ་ཁབ་ཁག་གིས་སྐྱོན་བརྗོད་ཤུགས་ཆེ་བྱས་ཡོད། ༡༩༩༤ ནས་ ༢༠༡༡ ལོ་བར་ཀེམ་ཇོན་གཉིས་པས་བྱང་ཀོ་རི་ཡ་དབང་སྒྱུར་བྱས་རྗེས། རྒྱལ་ཁབ་གཞན་དང་ཚོང་དོན་དང་དམག་དོན་གྲོས་མཐུན་བཞག་སྟེ། རྒྱལ་ཁབ་འཕེལ་རྒྱས་དང་ཁྱིམ་མཚེས་རྒྱལ་ཁབ་ལྷོ་ཀོ་རི་ཡ་དང་འབྲེལ་མོལ་བསྐྱར་གསོ་ཐུབ་པ་སོགས་བྱས་ཏེ། རྒྱལ་སྤྱིའི་སྡིངས་ཆའི་སྟེང་བྱང་ཀོ་རི་ཡའི་རྣམ་པ་གསར་པ་ཞིག་བསྐྲུན་ཡོད། ཀེམ་ཇོན་གཉིས་པ་གྲོངས་རྗེས་ཁོང་གི་བུ་དང་ད་ལྟའི་བྱང་ཀོ་རི་ཡའི་མགོ་ཁྲིད་ཀེམ་ཇོན་ཨང་གིས་ཕ་འདས་པའི་བྱས་རྗེས་ལ་མིག་དཔེ་མ་བལྟས་པར། རང་ཉིད་ཀྱི་དབང་ཆ་བརྟན་སྲུང་བྱ་ཆེད་དབང་ཤུགས་ཆེ་བའི་མཐོ་རིམ་དཔོན་རིགས་རྒན་གྲས་ཚོར་ཉེས་པ་འདྲ་མིན་བགྲངས་ཏེ་གནས་དབྱུང་གཏོང་མགོ་བཙུགས་པ་མ་ཟད། ༢༠༡༣ ལོར་བྱང་ཀོ་རི་ཡ་ནང་གི་དབང་ཤུགས་ཆེ་ཤོས་གཉིས་པ་ཡིན་པའི་ཨ་ཁུ་ཅང་སུན་ཋག་ (Chang Song-thaek ) ལ་རྒྱལ་ཁབ་ལ་རྒྱབ་གཏོད་བྱས་པའི་ཉེས་མིང་བཀལ་ཏེ་དམར་གསོད་བཏང་ཡོད། ༢༠༡༥ ལོའི་ཟླ་ ༤ ཚེས་ ༣༠ ཉིན་བྱང་ཀོ་རི་ཡའི་བདེ་སྲུང་བློན་ཆེན་དང་དམག་སྤྱི་གཅིག་ལྕོགས་ཡཱང་ཅོལ་ (Hyon Yong Chol) གྱིས། ཨུ་རུ་སུར་ཚོགས་པའི་རྒྱལ་སྤྱིའི་བདེ་སྲུང་ལྷན་ཚོགས་ཐོག་གཏམ་བཤད་སྤེལ་སྐབས། ཀེམ་ཇོན་ཨང་ (Kim Jong Un) ལ་བརྩི་བཀུར་མ་བྱས་པར་གང་བྱུང་སྐད་ཆ་བཤད་ཡོད་པ་མ་ཟད། དེ་སྔོན་བྱང་ཀོ་རི་ཡའི་དམག་སྦྱོང་འགྲེམས་སྟོན་ཞིག་གི་སྐབས་ལ། ཀེམ་ཇོན་ཨང་གི་བཀའ་ལ་མ་བརྩིས་པར་གཉིད་ཅོག་བརྒྱབ་ནས་དམག་དོན་འཐུས་ཤོར་བཏང་བ་སོགས་རྒྱལ་ཁབ་ལ་རྒྱབ་གཏོད་བྱས་པའི་ཉེས་མིང་ཁག་གཅིག་བགྲངས་ཏེ། མང་ཚོགས་ཀྱི་མཐོང་སར་དམར་གསོད་བཏང་བ་སོགས་ད་བར་མི་ ༧༠ ལྷག་དམར་གསོད་དང་བུད་མེད་མང་པོ་ཞིག་ལ་བཙན་གཡེམ་བཏང་ཡོད། དེ་བཞིན་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་བཀའ་ལ་མ་བརྩི་འཇོག་མ་བྱས་པར་ཤུགས་ཚད་འབྲིང་བའི་སྒྱོགས་མདེལ་འདྲ་མིན་ཚོད་ལྟ་བྱས་ཏེ། ཁྱིམ་མཚེས་རྒྱལ་ཁབ་ལ་ཐེངས་མང་འཇིགས་སྐུལ་བྱས་ཡོད་སྟབས། ཨ་རིས་དཔལ་འབྱོར་ཉེས་ཆད་ཐེངས་གཉིས་བཀལ་ཡོད་པ་མ་ཟད། མཉམ་འབྲེལ་རྒྱལ་ཚོགས་བདེ་འཇགས་ལྷན་ཚོགས་ཀྱི་ཉེན་བརྡ་བཏང་བ་སོགས་བྱས་ཏེ། ད་ལྟ་བྱང་ཀོ་རི་ཡ་གྲོགས་མེད་ཁེར་རྐྱང་གི་རྒྱལ་ཁབ་དང་ཀེམ་ཇོན་ཨང་མི་གསོད་ལག་དམར་བ་ཆགས་ཡོད། ཆབ་སྲིད་དབྱེ་ཞིབ་པ་ཚོས། ད་ལྟ་ཀེམ་ཇོན་ཨང་གིས་སོ་སོའི་དབང་ཆ་བརྟན་སྲུང་བྱ་ཆེད་ནང་དགྲ་འདུལ་བཞིན་ཡོད་ཀྱང་། དཀའ་ངལ་ནང་ལྷུང་བའི་རྒྱལ་ཁབ་ཀྱི་དཔལ་འབྱོར་གནས་སྟངས་ཡར་རྒྱས་མ་གཏང་ཚེ། ཁོའི་རང་རྒྱལ་གཞན་ཕམ་གྱི་སྲིད་བྱུས་འདི་ཕམ་ཉེས་འགྲོ་ངེས་རེད་ཅེས་བརྗོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
