སྲོག་གི་ཉེན་ཁ་ལས་སླར་ཡང་གསོན་པ།

ཆབ་སྲིད་བཙོན་ཟུར་བླ་བྲང་ཨ་ཁུ་འཇིགས་མེད་ལགས་ཕྱི་ལོ་ ༢༠༡༦ ཟླ་ ༡༠ ཚེས་ ༢༦ ཉིན་རྒྱའི་བཙོན་འོག་ནས་ཀློད་གྲོལ་ཐོབ་སྟེ་ཟླ་གཅིག་ཏག་ཏག་ཕྱིན་མཚམས་ཁོང་གིས་གྲོགས་པོ་ཞིག་༼སྲོག་གི་ཉེན་ཁ་ལས་སླར་ཡང་གསོན་པ།༽ཞེས་པའི་ཡི་གེ་ཞིག་བྲིས་ཡོད་པ་ཕྱི་ལོགས་སུ་འབྱོར་པ་ཁོང་གི་དགེ་ཕྲུག་དགེ་འདུན་ཕུན་ཚོགས་ལགས་ཀྱིས་བཏང་འབྱོར་བྱུང་། “ང་ཡི་སྙིང་སྟོབས་དང་ཆོད་སེམས། རིགས་ཞེན་དང་ལ་རྒྱ། རེ་བ་དང་གསོན་ཤུགས་སོགས་ཉམས་སོང་བ་ཡིན་ན་ནི་ཁོ་ཚོའི་ “འདོད་ཐོག་ལ་རྟ་རྒྱུག” པ་ཡིན་པས་ཁོ་ཚོས་ང་ཚོ་ཕམ་ཉེས་གཏོང་བྱེད་ཀྱི་ཐབས་ལམ་ལའང་འཇིགས་སྣང་དང་བསླུ་བྲིད། རྫུན་གཏམ་དང་གཡོ་ཇུས། མདོར་ན་བཙན་པོའི་ཁྲོ་གཉེར། གཡོ་སྒྱུའི་བཅོས་འཛུམ་སོགས་ཀྱི་ཐབས་སྣ་ཚོགས་ཤིག་བེད་སྤྱོད་པ་ལ་བྱང་ཆ་དེ་ལྟར་མངའ་བ་རེད་ཨང་། དེ་ལྟ་བུའི་ལམ་ག་ནས་རྒྱ་ཆེའི་མང་ཚོགས་ལ་མགོ་བསྐོར་གཏོང་བ་ལས་གཞན་དུ་ང་ཚོར་ནག་ཉེས་ཀྱི་ཞྭ་མོ་གང་ཆེ་གང་ལྕི་བཟོས་ཏེ་ཁྲིམས་ལ་སྦྱར་ནས་མ་བྱས་པ་མང་པོ་ཞིག་ཁས་ལེན་དགོས་པའི་སྐུལ་མའང་ནན་ཏན་དུ་བཏང་སྟེ་ང་ཚོ་ཤི་མིན་རོ་མིན་གྱི་བེམ་གཟུགས་ཙམ་དང་། ཡང་ན་འཇིག་རྟེན་འདི་ནས་རྩ་མེད་དུ་གཏོང་བའི་ལས་སྦྱོར་ཡང་མུ་ནས་མུར་བརྩམས་པ་རེད་མོད། ཁོ་ཚོས་ངའི་བསམ་གཞིགས་ཀྱི་ཤེད་མོང་དང་། ངའི་བློ་ཕུགས་ཀྱི་དམ་བཅའ་ལ་འགྱུར་བ་འེན་ཙམ་ཡང་གཏོང་མ་ནུས་ཅེས་ཤོད་ཆོག་པ་མ་ཟད། ལྡོག་སྟེ་བདེན་དང་གནས་ལུགས་ཀྱི་སླད་དུ་དང་། དྲང་བདེན་བློ་དཀར་གྱི་ཕྱིར་དུ་ང་ལ་སྟོབས་ཤུགས་དང་ནུས་མཐུ།   སྙིང་རུས། བཟོད་བསྲན་སོགས་སྔར་མ་བྱུང་བ་མང་པོ་བསྩལ་སོང་བ་འདི་ནི་བདེན་དང་གནས་ལུགས་ཀྱི་སྟོབས་ཤུགས་ཡིན་པར་ཏན་ཏན་ཡིད་ཆེས་པ་ཡིན། ངའི་རིགས་ལ་བཅིངས་པའི་ལྷག་བསམ།    ངའི་ཡུལ་ལ་བཟུང་བའི་བརྩེ་བ། ངའི་སྤུན་ཟླར་བཅོལ་བའི་རེ་བ་སོགས་ནི་མི་འགྱུར་བྲག་རི་བཞིན་སུས་ཀྱང་གཞོམ་གཞིག་གཏོང་མི་ནུས་པ་བཞིན་དུ། འཛམ་གླིང་ཞི་བདེ་དང་། རྒྱལ་ཁམས་དར་རྒྱས། སེམས་ལྡན་བདེ་སྐྱིད་སོགས་ཀྱི་སླད་དུ་སྡུག་རུས་མཐའ་འཁྱོལ་བྱེད་ངེས་པའང་བདག་གིས་བདེན་དོན་གྱིས་བཞེངས་པའི་ཚེ་སྲོག་གི་དམ་བཅའ་ཡིན་ལ། ནམ་ཞིག་ཚེ་སྲོག་གི་མར་མེ་རླུང་གིས་བསད་པའམ།  བླ་སྲོག་གི་དར་ལྕོག་གྲི་ཡིས་གཏུབ་པ་གང་ཡིན་ན་ཡང་འཆི་ཁའི་ཁ་ཆེམས་སམ་སྙིང་གཏམ་དྲོན་མོའང་འདི་ཡིན། བླ་བྲང་འཇིགས་མེད་ཀྱིས་ ༢༠༡༦.༡༡.༢༦ ཉིན་བྲིས།”  ཞེས་སྤེལ་འདུག བླ་བྲང་འཇིགས་མེད་ལགས་དེ་སྔོན་རྒྱའི་བཙོན་ཁང་དུ་སྡོད་སྐབས་སྙིང་དང་ཁྲག་ཤེད་སོགས་ཀྱི་ན་ཚ་ཕོག་ཡོད་པ་དང་། ༢༠༡༦ ཟླ་ ༡༢ ཚེས་ ༢༦ ཉིན་བླ་བྲང་བསང་ཆུ་རྫོང་སྨན་ཁང་དུ་བརྟག་དཔྱད་བཏང་བར་སྨན་ཁང་དུ་སྨན་བཅོས་བྱས་ཏེ་སྡོད་དགོས་བྱུང་ཡོད་པའི་གནས་ཚུལ་ཐོན་ཡོད། གྲྭ་འཇིགས་མེད་ལགས་ཀྱིས་ ༢༠༠༨ ལོར་བོད་ནང་གི་དངོས་ཡོད་གནས་སྟངས་དང་། ལྷག་པར་དུ་ལོ་དེར་རྒྱ་ནག་གཞུང་གིས་བླ་བྲང་ཕྱོགས་སུ་དྲག་གནོན་བྱས་པའི་གནས་ཚུལ་ངོ་སྤྲོད་གནང་ཡོད་པའི་བརྙན་ཕབ་ཅིག་ཕྱི་ལོགས་སུ་འགྲེམ་སྤེལ་བྱས་ཏེ་རྒྱལ་སྤྱིའི་ཐོག་གླེང་གཞི་ཆེན་པོ་ཆགས་ཡོད་པ་རེད། ཁོང་དེ་སྔོན་རྒྱ་གཞུང་གིས་ཐེངས་མང་པོར་འཛིན་བཟུང་བཀག་བསྐྱིལ་བྱས་ཡོད་པ་དང་། ༢༠༡༡ ལོའི་ཟླ་ ༨ པའི་ཚེས་ ༢༠ ཉིན་འཛིན་བཟུང་བྱས་པ་ནས་བཟུང་ལོ་གསུམ་རིང་ཁྲིམས་ཆད་མ་བཅད་པར་སྨན་ཁང་སོགས་སུ་གསང་བའི་སྒོ་ནས་བཀག་སྡོམ་བྱས། ༢༠༡༤ ཕྱི་ཟླ་བརྒྱད་པའི་ནང་དངོས་སུ་ལོ་ལྔའི་ཁྲིམས་ཐག་བཅད་ཡོད་པ་ལྟར། འདས་པའི་ལོ་ལྔའི་རིང་བཙོན་འོག་ཏུ་མནར་གཅོད་མང་པོ་བཏང་མཐར་ ༢༠༡༦ ཟླ་ ༡༠ ཚེས་ ༢༦ ཉིན་གསང་སྟབས་ཀྱིས་བཙོན་གྲོལ་བཏང་སྟེ་ཡུལ་དུ་ལོག་ཡོད་པ་རེད། འདི་ནི་ཁོང་བཙོན་འཇུག་བྱེད་ཐེངས་བཞི་བ་དེ་ཆགས་ཡོད་སྐོར་སྐབས་དེར་བོད་ནང་གི་འབྲེལ་ཡོད་མི་སྣས་ཕྱི་ལོགས་སུ་བརྗོད་ཡོད།
