གསར་འགྱུར་ལས་རིཊ་གཉེར་མཁན་དམར་གསོད།

ཁ་སང་ཕྱི་དྲོར་ཨབ་གྷ་ནི་སི་ཐན་གྱི་སྒེར་གཉེར་གསར་ཁང་ཞིག་གི་ལས་བྱེད་ ༧ དམར་གསོད་བཏང་བ་དང་མི་ ༢༥ ལྷག་ལ་རྨས་སྐྱོན་བཟོས་ཡོད་འདུག   ཁ་སང་ཕྱི་དྲོར་ཨབ་གྷ་ནི་སི་ཐན་གྱི་རྒྱལ་ས་ཁཱ་བྷུལ་(Kabul)དུ་ཡོད་པའི་སྒེར་གཉེར་གསར་ཁང་ཐོ་ལོ་(Tolo News)ཡི་ལས་བྱེད་རྣམས་སྤྱི་སྤྱོད་རླངས་འཁོར་ཆུང་ཆུང་ཞིག་ཏུ་བསྡད་དེ་ཁྱིམ་དུ་ལོག་པའི་ལམ་བར་དུ། ཐ་ལི་བྷན་གྱི་དྲག་སྤྱོད་པ་ཞིག་གིས་རླངས་འཁོར་ནང་འབར་རྫས་གཡུགས་ཏེ། ཉུང་མཐར་ལས་བྱེད་དང་འགྲུལ་བ་ ༢༥ ལ་རྨས་སྐྱོན་བཟོས་ཡོད་པ་མ་ཟད། བརྙན་འཕྲིན་ཁང་དེའི་ལས་བྱེད་ ༧ དམར་གསོད་བཏང་ཡོད་སྟབས། ཨབ་གྷན་ནང་གི་གསར་འགོད་པ་མང་པོ་ལྷན་དུ་འདུས་ཏེ་དྲག་སྤྱོད་ལ་ངོ་རྒོལ་བྱེད་བཞིན་ཡོད་འདུག   དམར་གསོད་དོན་རྐྱེན་བྱུང་རྗེས། ཐོ་ལོ་གསར་ཁང་གིས་ཊི་ཊིར་ཐོག་ཏུ། དེ་རིང་སྟབས་མ་ལེགས་པར་དྲག་སྤྱོད་པའི་དམར་གསོད་ཁྲོད་ང་ཚོའི་ལས་བྱེད་ ༧ གྱི་སྲོག་ཤོར་སོང་ཞེས་བསྒྲགས་ཡོད་པ་དང་། འདས་ལོའི་ཕྱི་ཟླ་ ༩ པའི་ནང་གསར་ཁང་དེས་ཨབ་གྷན་གྱི་བྱང་ཕྱོགས་སུ་ཡོད་པའི་ཐ་ལི་བྷན་དྲག་སྤྱོད་ཚོགས་པས་བུད་མེད་མང་པོར་བཙན་གཡེམ་དང་བཙན་འཁྲིད་བྱེད་ཀྱི་ཡོད་པའི་གནས་ཚུལ་སྤེལ་བར་བརྟེན། དྲག་སྤྱོད་ཚོགས་པས་གནས་ཚུལ་དེ་ཚོ་དཀྲོག་གཏམ་ཡིན་ལུགས་དང་། གནས་ཚུལ་སྤེལ་མཁན་ལས་བྱེད་དང་གཙོ་སྐྱོང་བ། གསར་འགོད་པ་སོགས་ལ་དབྱེ་བ་མེད་པར་འཁོན་ལན་སློག་རྒྱུ་ཡིན་པའི་ཡོངས་ཁྱབ་འཇིགས་སྐུལ་བྱས་ཡོད་པ་རེད།   རྒྱལ་སྤྱིའི་གསར་ལམ་ཁག་ཏུ་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༨ ཉིན་ཨབ་གྷན་གྱི་རྒྱལ་ས་ཁཱ་བྷུལ་ནང་པཱ་ཀི་སི་ཐན་དང་ཨབ་གྷན། རྒྱ་ནག ཨ་རི་བཅས་ཀྱི་སྐུ་ཚབ་ཚོགས་པས་ཐ་ལི་བྷན་དྲག་སྤྱོད་ཚོགས་པ་དང་ལྷན་དུ་འབྲེལ་མོལ་བསྐྱར་གསོ་ཡོང་ཐབས་ཀྱི་གྲོས་མོལ་ཐེངས་གཉིས་པ་བསྡུས་ཏེ། ཨབ་གྷན་ནང་ཞི་བདེ་འགྲིག་འཇགས་དགོས་པའི་འབོད་སྐུལ་བྱས་ཡོད་ཀྱང་། འདས་པའི་ཉིན་ ༢༡ ནང་རྒྱལ་ས་ཁཱ་བྷུལ་ནང་དྲག་སྤྱོད་ཚོགས་པ་དེས་འབར་རྫས་ཐེངས་ ༦ གཡུགས་ནས་ཡུལ་མི་མང་པོ་དམར་གསོད་བཏང་ཡོད་པ་དང་། དྲག་སྤྱོད་ཚོགས་པ་དེའི་གཏོར་རྒོལ་འོག་ཕྱི་ལོ་ ༢༠༠༡ ནས་ལྟའི་བར་ཉུང་མཐར་ཨབ་གྷན་ནང་མི་གྲངས་ ༢༦༠༠༠ དང་། པཱ་ཀི་སི་ཐན་ནང་མི་གྲངས་ ༢༡༠༠༠ གིས་ཚེ་སྲོག་ཤོར་ཡོད་པ་དང་། མི་ ༡༦༢༠༠༠ ལྷག་ལ་རྨས་སྐྱོན་བཟོས་ཡོད་འདུག   ཐོ་ལོ་གསར་ཁང་ནི་ཨབ་གྷན་གྱི་མོ་བྷེ་གསར་འགོད་ལྷན་ཚོགས་(Moby Media Group)ཀྱིས་ཕྱི་ལོ་ ༢༠༡༠ ལོར་གསར་དུ་བཙུགས་པ་དང་། ཨབ་གྷན་གྱི་ཆུ་ཚོད་ ༢༤ རིང་གསར་འགྱུར་ཐད་གཏོང་བྱེད་མཁན་གྱི་བརྙན་འཕྲིན་ཐོག་མ་དེ་ཡིན་ཞིང་། གསར་ཁང་དེར་ཐོ་ལོ་བརྙན་འཕྲིན་དང་ལི་མཱར་བརྙན་འཕྲིན་ཟེར་བ་གཉིས་ཡོད་པ་དང་། གསར་ཁང་དེས་གཉེར་བའི་དྲ་རྒྱ་ལ་ལྟ་ཀློག་པ་ཚོས་དགའ་མོས་ཆེན་པོ་བྱེད་ཀྱི་ཡོད་པ་མ་ཟད། ཧ་ཅང་ཡིད་ཆེས་བྱེད་སའི་གསར་འགྱུར་དྲ་བ་ཞིག་ཡིན་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
