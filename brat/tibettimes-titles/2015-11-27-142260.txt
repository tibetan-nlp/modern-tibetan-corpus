རྒྱ་ནག་གིས་དྲག་ཆས་དམག་དཔུང་རྒྱ་བསྐྱེད།

  རྒྱ་ནག་གཞུང་གིས་ཕྱིའི་དཔལ་འབྱོར་འཚོལ་སྒྲུབ་དང་མཚོ་ཁུལ་བདག་འཛིན་ཆེད་སྐམ་མཚོ་མཁའ་གསུམ་གྱི་དམག་དཔུང་རྒྱ་སྐྱེད་གཏོང་འཆར་ཡོད་འདུག   དེ་རིང་ནེའུ་ཡོག་དུས་བབ་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཁ་སང་རྒྱ་ནག་གཞུང་གིས་དམག་དོན་སྒྲིག་འཛུགས་གསར་བ་ཞིག་གཏན་འབེབས་བྱ་འཆར་ཡོད་སྐོར་ཕྱི་བསྒྲགས་བྱས་ཡོད་པ་དང་། དེ་ནི་ཕྱི་ལོ་ ༢༠༢༠ ལོའི་བར་ད་སྔའི་དམག་དོན་སྒྲིག་འཛུགས་ལ་འཕོ་འགྱུར་བཏང་སྟེ་རྒྱ་ནག་མི་དམངས་བཅིངས་འགྲོལ་དམག་དཔུང་ཟེར་བའི་གནམ་ས་མཚོ་གསུམ་གྱི་དམག་འཐབ་སྟོབས་ཤུགས་རྒྱ་བསྐྱེད་གཏོང་འཆར་ཡོད་པ་ཞིག་ཡིན་འདུག   ཁ་སང་ཕྱི་ཚེས་ ༢༦ ཉིན་རྒྱ་ནག་དམག་ཁུལ་གྱི་ཆེས་མཐོའི་དམག་དཔོན་ ༢༠༠ བསྡུ་བསྐོང་གིས་ཚོགས་འདུ་ཞིག་ཚོགས་ཏེ། རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱིས་ད་སྔའི་དམག་དོན་སྒྲིག་འཛུགས་ལ་འཕོ་འགྱུར་བཏང་སྟེ་རྒྱ་ནག་མི་དམངས་བཅིངས་འགྲོལ་དམག་དཔུང་གིས་ཕྱིའི་དགྲ་བོའི་སྟོབས་ཤུགས་ལ་གདོང་གཏད་དགོས་པའི་འོས་འགན་ཕྲག་ཏུ་བབ་ཡོད་སྐོར་དང་། རྒྱལ་སྲུང་དང་དམག་ཤུགས་རྒྱ་བསྐྱེད་དགོས་པ་དེ་ལོ་རྒྱུས་དང་དུས་རབས་གསར་བའི་དགོས་མཁོ་ཞིག་ཡིན་སྟབས། རྒྱ་ནག་མི་དམངས་བཅིངས་འགྲོལ་དམག་དཔུང་གིས་སྔར་ལས་ལྷག་པར་བློ་སྤོབས་ཆེར་བསྐྱེད་དང་སྟོབས་ཤུགས་གཅིག་སྒྲིལ་སྒོས་རྒྱ་ནག་གི་དམིགས་འབེན་མངོན་འགྱུར་དང་། མ་འོངས་འགྲན་རྩོད་རང་བཞིན་གྱི་དེང་རབས་དམག་ཤུགས་རྒྱ་བསྐྱེད་པར་དམིགས་འབེན་གཏོད་དགོས་སྐོར་སོགས་བརྗོད་འདུག   གནས་ཚུལ་ལྟར་ན། ཉིན་གསུམ་རིང་གི་དམག་དོན་ཚོགས་འདུ་དེའི་དམིགས་འབེན་གཙོ་བོ་རྒྱ་ནག་མི་དམངས་བཅིངས་འགྲོལ་དམག་དཔུང་གི་སྒྲིག་འཛུགས་ལ་འཕོ་འགྱུར་བཏང་སྟེ། རྒྱ་ནག་གཞུང་གིས་ཕྱིའི་དཔལ་འབྱོར་འཚོལ་སྒྲུབ་དང་རྒྱ་ནག་གི་ལྷོ་རྒྱུད་རྒྱ་མཚོའི་མངའ་ཁུལ། དེ་བཞིན། ཉི་ཧོང་གི་འདབས་འབྲེལ་མཚོ་ཁུལ་བཅས་བདག་འཛིན་བྱེད་ཐབས་སྐོར་ལ་གྲོས་མོལ་བྱས་པ་ཞིག་ཡིན་ཞིང་། འདི་ལོའི་ཕྱི་ཟླ་ ༩ ཚེས་ ༣ ཉིན་གྱི་རྒྱ་ནག་གི་དམག་དཔུང་རྔམ་སྟོན་དུས་ཆེན་ཐོག ཞི་ཅིན་ཕིན་གྱིས་གནམ་ས་མཚོ་གསུམ་གྱི་དམག་དཔུང་གིས་གྲུབ་པའི་རྒྱ་ནག་མི་དམངས་བཅིངས་འགྲོལ་དམག་མི་འབུམ་ ༣ ཇེ་ཉུང་དུ་གཏོང་རྒྱུ་ཡིན་ལུགས་བརྗོད་ཀྱང་། ཉེ་བའི་ཆར་རྒྱ་ནག་མི་དམངས་བཅིངས་འགྲོལ་དམག་ཁོངས་སུ་དམག་མི་ས་ཡ་ ༡ ལྷག་ཡོད་འདུག   ཨ་རིའི་ཝལ་སི་ཊི་རེ་ཊི་རྗོར་ནོལ་ (WSJ) གསར་ཁང་གིས་སྤེལ་བའི་གནས་ཚུལ་ནང་། རྒྱ་ནག་གཞུང་གིས་འཛམ་གླིང་ཤར་ནུབ་ཀྱི་རྒྱལ་ཁབ་ཁག་ནང་དཔལ་འབྱོར་འཚོལ་སྒྲུབ་དང་རྒྱལ་སྲུང་བདེ་འཇགས་ལ་དམིགས་ཏེ་ཤར་ཨ་ཧྥེ་རི་ཁའི་མཚོ་འགྲམ་རྒྱལ་ཁབ་ཌ་ཇེ་བྷུ་ཊེ་ (Djibouti) ནང་རྒྱ་ནག་གི་དྲག་ཆས་དམག་དཔུང་རྩ་འཛུགས་དང་རྒྱ་བསྐྱེད་གཏོང་གི་ཡོད་པ་མ་ཟད། ཁ་སང་རྒྱལ་ཁབ་དེའི་ནང་རྒྱ་ནག་གི་དམག་དཔུང་རྩ་འཛུགས་དང་རྒྱ་བསྐྱེད་ཆོག་པའི་གྲོས་ཆོད་ཅིག་ཀྱང་བཞག་ཡོད་སྟབས། རྒྱལ་སྤྱིའི་སྲིད་དོན་དབྱེ་ཞིབ་པ་ཚོས་འདས་པའི་ལོ་ངོ་འདི་དག་རིང་པེ་ཅིན་གཞུང་གིས་ནུབ་རྒྱ་མཚོ་ཆེན་མོ་དང་ཞིའི་རྒྱ་མཚོ་ཆེན་མོ། དེ་བཞིན། རྒྱ་གར་རྒྱ་མཚོ་སོགས་ཀྱི་མཚོ་ཁུལ་ཁག་བདག་གེར་འཛིན་ཆེད། ཕྱིའི་རྒྱལ་ཁབ་ཁག་གི་སྲུང་དམག་མེད་སའི་ཤར་ནུབ་ཀྱི་མཚོ་གླིང་རྒྱལ་ཁབ་ཁག་ནང་དྲག་ཆས་དམག་དཔུང་འབོར་ཆེན་རྩ་འཛུགས་བྱེད་བཞིན་པར་བློ་འཚབ་ཡོད་སྐོར་བརྗོད་འདུག   གསར་འགོད་པ། པདྨ་རྒྱལ།
