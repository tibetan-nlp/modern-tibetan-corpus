ཆོས་དཔོན་རཱ་ཧེམ་སིང་ལ་ཁྲིམས་ཆད་ལོ་ ༢༠
ཕོག 

རྒྱ་གར་གྱི་ཆོས་དཔོན་རཱམ་རཱ་ཧེམ་སིང་གིས་བུད་མེད་གཉིས་ལ་བཙན་གཡེམ་བྱས་པའི་ནག་ཉེས་འོག་བུད་མེད་རེ་རེར་ལོ་ ༡༠ རེ་བྱས་ནས་ཁྱོན་ལོ་ ༢༠ ཡི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད། རྒྱ་གར་དུས་བབ་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན།  རྒྱ་གར་གྱི་ཆོས་དཔོན་རཱམ་རཱ་ཧེམ་སེང་གིས་ལོ་ ༢༠༠༢ ལོར་རང་གི་སློབ་ཕྲུག་བུ་མོ་གཉིས་ལ་བཙན་གཡེམ་བཏང་བའི་ནག་ཉེས་འོག  ཁ་སང་རྒྱ་གར་མངའ་སྡེ་ཧར་ཡ་ཎའི་རའོ་ཊི་ཁུ་(Rhotak)རྫོང་ཁྲིམས་ཁང་དུ།  རྒྱ་གར་ཉེས་དོན་ཞིབ་བཤེར་དམིགས་བསལ་ཁྲིམས་ཁང་ཞིག་གིས། བཙན་གཡེམ་བཏང་བའི་བུ་མོ་རེ་རེའི་ཆེད་དུ་ལོ་ ༡༠ རེ་བྱས་ཏེ་ཆོས་དཔོན་རཱམ་རཱ་ཧེམ་སེང་ལ་ཁྱོན་བསྡོམས་ལོ་ ༢༠ ཡི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་པ་དང་ཆབས་ཅིག བཙན་གཡེམ་ཐེབས་པའི་བུད་མེད་གཉིས་ལ་གུན་གསབ་ཏུ་ཧིན་སྒོར་འབུམ་ ༢༨ སྤྲོད་གཏན་འཁེལ་ཡོད་འདུག དེ་ཡང་རྒྱ་གར་ཉེས་དོན་ཞིབ་བཤེར་ཁང་གི་ཞིབ་བཤེར་པ་སཱོ་མཱེ་ཊི་(Sumeet Goel)ལགས་ཀྱིས།  གལ་སྲིད་ཆོས་དཔོན་རཱམ་རཱ་ཧེམ་སེང་གི་ཕྱོགས་གཏོགས་ཚོས་བཙོན་འཇུག་དུས་ཡུན་ནར་འགྱངས་བཏང་བར་སྐྱོན་བརྗོད་དང་ངོ་རྒོལ་བྱས་ཚེ།  ཁོ་ཚོའི་བྱ་སྤྱོད་ལ་དཔག་པའི་ཆོས་དཔོན་ལ་ཉེས་ཆད་གཅོད་རྒྱུ་ཡིན།  ཞེས་བརྗོད་ཡོད་པ་དང་།  ཡང་ཁྲིམས་དཔོན་ཇཱགི་སི་ཧིན་(Jagdeep Singh)ལགས་ཀྱིས། ཆོས་དཔོན་རཱམ་རཱ་ཧེམ་སེང་ལ་ཉེས་ཆད་ཧིན་སྒོར་འབུམ་ ༣༠ བཅད་དེ། འདས་པའི་ལོ་ ༡༥ རིང་གཞུང་དང་འབྲེལ་བའི་ལྷན་ཁང་ཁག་མང་པོ་བརྒྱུད་དེ་བཙན་གཡེམ་གྱི་གྱོད་གཞི་གཏུག་བཤེར་བྱེད་མཁན་བུད་མེད་གཉིས་ལ་ཧིན་སྒོར་འབུམ་ ༢༨ སྤྲོད་རྒྱུ་ཐག་བཅད་བྱུང་ཡོད་སྐོར་བརྗོད་འདུག ཆོས་དཔོན་རཱམ་རཱ་ཧེམ་སེང་གི་ཁྲིམས་རྩོད་པ་ནཱར་ཝ་ནཱ་(Garg Narwana)ལགས་ཀྱིས།  ཁྲིམས་ཁང་གིས་ཆོས་དཔོན་རཱམ་རཱ་ཧེམ་སེང་ལ་བཙན་གཡེམ་གྱི་གྱོད་གཞི་རེ་རེར་ལོ་ ༡༠ རེའི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་ཀྱང་།  ང་ཚོས་ལོ་ ༡༠ དེ་སྡེབ་གཅིག་ཏུ་ཉེས་ཆད་བཏང་ཡོད་པར་ངོས་འཛིན་བྱས་ཡོད།  འོན་ཀྱང་།  ད་ལྟ་ང་ཚོས་ཐོག་མའི་ལོ་ ༡༠ ཡི་བཙོན་འཇུག་དུས་ཡུན་རྫོགས་རྗེས་བསྐྱར་དུ་ལོ་ ༡༠ བཙོན་འཇུག་ཉེས་ཆད་འཁུར་དགོས་པ་ཤེས་སོང་།  ཞེས་བརྗོད་འདུག བཙན་གཡེམ་གྱོད་གཞིའི་ཁྲིམས་རྩོད་པ་ཨོ་ཊི་སི་ཧིན་(Utsave Singh)ལགས་ཀྱིས།  ད་ལྟ་བུད་མེད་ ༥༠ མདུན་དུ་ཐོན་ཏེ། ཁོ་ཚོས་ཁྲིམས་ཁང་དུ་བཙན་གཡེམ་བཏང་བའི་གྱོད་གཞི་ཁྲིམས་གཏུགས་བྱེད་ཀྱི་ཡོད་པ་དང་། ང་ཚོས་ཤེས་གསལ་ལྟར་ན།  ཆོས་དཔོན་རཱམ་རཱ་ཧེམ་སེང་གིས་ད་དུང་བུད་མེད་ ༤༨ ལྷག་ཅིག་ལ་བཙན་གཡེམ་བཏང་ཡོད་ཀྱང་།  བུད་མེད་དེ་ཚོས་རཱམ་རཱ་ཧེམ་སེང་ཁྲིམས་གཏུགས་བྱེད་པར་ཞེད་སྣང་སྐྱེས་ཀྱི་ཡོད་འདུག  གཅིག་བྱས་ན་བུད་མེད་རེ་ཟུང་ཞིག་ལྐོག་གསོད་ཡང་བཏང་ཡོད་ཚོད་འདུག ཅེས་བརྗོད་འདུག་ལ། དེ་སྔོན་ཆོས་དཔོན་དེས་མི་བསད་ཡོད་པའི་ཉེས་དོན་ཁྲིམས་གཏུག་བྱེད་མཁན་ཡང་བྱུང་མྱོང་ཡོད་འདུག ཆོས་དཔོན་རཱམ་རཱ་ཧེམ་སེང་བཙན་གཡེམ་གྱི་གྱོད་དོན་འོག་ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༥ ཉིན་ཁྲིམས་ཁང་དུ་སྐད་བཏང་སྟེ་ནག་ཉེས་ངོས་ལེན་དང་ཉེས་ཆད་དང་ལེན་བྱེད་པའི་རེ་སྐུལ་ཞུས་ཡོད་ཀྱང་།  ཁོ་བོའི་རྗེས་འཇུག་པ་ཚོས་ང་ཚོའི་བླ་མས་སྡིག་པའི་ལས་མི་བྱེད་ཅེས་ཆོས་དཔོན་ལ་ནག་ཉེས་མེད་སྐོར་བརྗོད་དེ་མངའ་སྡེ་ཧར་ཡ་ཎའི་ནང་ཟིང་འཁྲུག་ཆེན་པོ་བསླངས་ཡོད་པ་དང་། ཟིང་འཁྲུག་ཁྲོད་ད་བར་མི་གྲངས་ ༣༨ རྐྱེན་འདས་སུ་བཏང་བ་དང་། མི་ ༢༠༠ ལྷག་ཅིག་ལ་རྨས་སྐྱོན་བཟོས་ཡོད་པ་རེད། གསར་འགོད་པ། སྐལ་བཟང་སྦྱིན་པ།
