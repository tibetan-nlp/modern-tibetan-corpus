བོད་ཀྱི་དུས་བབ་དྲ་བའི་མཉེན་ཆས་གསར་པ།

དེ་རིང་༧གོང་ས་རྒྱལ་བ་རིན་པོ་ཆེ་མཆོག་དགུང་གྲངས་ ༨༢ ལ་ཕེབས་པའི་སྐུའི་འཁྲུངས་སྐར་དང་བསྟུན་ཏེ། བོད་ཀྱི་དུས་བབ་དྲ་བའི་ལག་ཐོག་ཁ་པར་དང་ལག་འཁྱེར་གློག་དེབ་སོགས་ཀྱི་པར་གཞི་གསར་པའི་མཉེན་ཆས་བཏང་ཡོད་པ་དང་། མཉེན་ཆས་འདི་མ་འོངས་པའི་ཉིན་གྲངས་ ༡༠༠ རིང་རིན་མེད་ཕབ་ལེན་བྱེད་ཆོག བོད་ཀྱི་དུས་བབ་ཀྱི་མཉེན་ཆས་གསར་པ་(tibettimes APP)འདི་ཁྱོད་ཀྱི་ལག་ཐོག་ཁ་པར་(Mobile)དང་། ལག་འཁྱེར་གློག་དེབ་(ipad)སོགས་ཀུ་ཤུ་(IOS)རྟགས་ཅན་ཡིན་ན་(app store)དང་། ཨན་ཌོར་ཌི་(Android)རྟགས་ཅན་ཡིན་ན་(Play Store)ནས་ཕབ་ལེན་བྱེད་ཐུབ་པ་མ་ཟད། དེ་གཉིས་ཀྱི་བེད་སྤྱོད་དང་ཇུས་འགོད་སོགས་ཀྱང་གཅིག་གྱུར་ཡིན། རིམ་སྤར་(Upgrade)ཞེས་པའི་མིང་ཚིག་འདི་དེ་རིང་གི་བརྡ་འཕྲིན་སྤྱིའི་འཕེལ་ཕྱོགས་ཤིག་ཆགས་ཡོད་པ་ལྟར། བོད་ཀྱི་དུས་བབ་མཉེན་ཆས་འདིའང་སྐྱོན་ཀུན་སེལ་ཡོན་ཏན་ཀུན་ལྡན་ཞིག་ཡིན་མི་སྲིད་སྟབས། དུས་ནས་དུས་སུ་སྤུས་ཇེ་དག་དང་སྣ་ཇེ་འཛོམ་དུ་གཏོང་རྒྱུ་ཡིན་པས། ཁྱེད་ཀྱི་ཁ་པར་ནང་གི་བོད་ཀྱི་དུས་བབ་མཉེན་ཆས་ཀྱང་བདུན་ཕྲག་འགའ་རེ་དང་། ཡང་ན་ཟླ་རེའི་མཚམས་སུ་ངེས་པར་དུ་ཁ་གསབ་(Update)དང་། ཡང་ན། རྦད་དེ་ནག་གསུབ་(Delet)བཏང་སྟེ་བསྐྱར་དུ་ཕབ་ལེན་བྱས་ན་ཁྱེད་ལ་དུས་བབ་མཉེན་ཆས་མ་ལག་གསར་ཤོས་དེ་རེག་པ་མ་ཟད། ཆ་རྐྱེན་ཇེ་འཛོམས་དང་འཁོར་རྒྱུག་ཀྱང་ཇེ་མགྱོགས་སུ་འགྲོ་བའི་དགེ་མཚན་ཡོད། བོད་ཀྱི་དུས་བབ་ཀྱི་མཉེན་ཆས་གསར་པ་འདི་ནི། རྒྱ་གར་ལྷོ་ཕྱོགས་གྷོ་ཝར་རྟེན་གཞི་བྱེད་པའི་ཀུ་ཤུ་འཆར་འགོད་བཟོ་གྲྭ་དང་། དྷ་རམ་ས་ལའི་བོད་ཀྱི་བརྡ་འཕྲིན་ལས་ཁུངས། བོད་ཀྱི་དུས་བབ་ལས་ཁུངས་བཅས་ཚོགས་པ་གསུམ་མཉམ་འབྲེལ་ཐོག་འདས་པའི་ལོ་ངོ་གཅིག་དང་ཕྱེད་ལྷག་རིང་བཟོས་ཤིང་། མཉེན་ཆས་འདི་ནི་(NDK)ཞེས་པའི་དཔར་གཞི་དང་ (SDK) ཞེས་པའི་དཔར་གཞི་གཉིས་ཀྱི་རྨང་གཞིའི་ཐོག་ཀུ་ཤུ་(apple)དང་གྷོ་གྷིལ་(Google)གཉིས་གཅིག་གྱུར་དུ་བཟོས་པའི་བོད་ཡིག་ཁ་པར་བརྒྱུད་ཁུངས་མཉེན་ཆས་ཐོག་མ་དེ་ཆགས་ཡོད། ༼   བོད་ཀྱི་དུས་བབ་མཉེན་ཆས་གསར་པ་བེད་སྤྱོད་གཏོང་ཕྱོགས། ༽ (1.) བོད་ཀྱི་དུས་བབ་དྲ་བའི་འཆར་ངོས། བོད་ཀྱི་དུས་བབ་དྲ་ཚིགས་ཨ་མ་དང་། ཐེངས་འདིར་གསར་དུ་སྒོ་འབྱེད་བྱས་པའི་མཉེན་ཆས་(App)གང་ཡིན་རུང་། ཁྱོད་ཀྱི་གློག་ཀླད་དང་ཁ་པར་སོགས་ནང་བོད་ཡིག་མེད་ཀྱང་བདེ་བླག་ཏུ་འཆར་ཐུབ་པའི་ཁྱད་ཆོས་ཡོད། (2.) བོད་ཀྱི་དུས་བབ་མཉེན་ཆས་ཕབ་ལེན། ཁྱོད་ཀྱི་ལག་ཐོག་ཁ་པར་དང་ལག་འཁྱེར་ཀློག་དེབ་ཀུ་ཤུ་(IOS)རྟགས་ཅན་ཡིན་ན་(app store)དང་། ཨན་ཌོར་ཌི་(Android)རྟགས་ཅན་ཡིན་ན་(Play Store)ནས་ཕབ་ལེན་བྱེད་དགོས། (3.) བོད་ཀྱི་དུས་བབ་འཚོལ་བཤེར། ཁྱོད་ཀྱིས་བོད་ཀྱི་དུས་བབ་མཉེན་ཆས་ཕབ་ལེན་བྱེད་པར། ཁྱོད་ཀྱི་ཁ་པ་ཀུ་ཤུ་ཡིན་ན་(app store)ནང་བོད་ཡིག་ཐོག་ ‘བོད་ཀྱི་དུས་བབ།’ དང་། ཡང་ན་དབྱིན་ཡིག་ནང་ tibettimes ཞེས་བཏགས་ཏེ་འཚོལ་དགོས་པ་དང་། ཁྱོད་ཀྱི་ཁ་པར་ཨན་ཌོར་ཌི་ཡིན་ན་(Play Store)ནང་དབྱིན་ཡིག་ཐོག་ tibettimes ཞེས་བཏགས་ན་རྙེད་ཐུབ། (4.) བོད་ཀྱི་དུས་བབ་མཉེན་ཆས། བོད་ཀྱི་དུས་བབ་མཉེན་ཆས་གསར་པ་འདི་བེད་སྤྱོད་གཏོང་བར། ཁྱོད་ཀྱི་ལག་ཐོག་ཁ་པར་དང་ལག་འཁྱེར་གློག་དེབ་སོགས་ཐོན་རིམ་(Verion)གསར་བའི་རིགས་ཡིན་དགོས་པ་མ་ཟད། ཁོག་པའི་ཤོང་ཚད་ཀྱང་གང་འཚམ་ཆེན་པོ་ཡོད་པ་དང་། མ་ཐར་སྟོང་ཆ་(MP 50)ལས་མ་ཉུང་བ་ཡོད་དགོས། (5.) བོད་ཀྱི་དུས་བབ་སྟེང་མཆན་འགོད་སྟངས། ཁྱོད་ཀྱིས་བོད་ཀྱི་དུས་བབ་སྟེང་མཆན་དང་བསམ་ཚུལ་སོགས་འགོད་ཚེ། བོད་ཀྱི་དུས་བབ་ལ་སྔར་དེབ་སྐྱེལ་བྱས་མེད་ན་གསར་དུ་དེབ་སྐྱེལ་བྱེད་དགོས་པ་དང་། སྔར་དེབ་སྐྱེལ་བྱས་ཡོད་ན་ནང་འཛུལ་ནང་ཕྱིན་ཏེ་ཐོ་མིང་དང་གསང་ཚིག་མ་ནོར་བར་བཏགས་རྗེས་མཆན་འགོད་ཐུབ། (6.) ནང་འཛུལ་བྱེད་སྟངས། ཁྱོད་ཀྱི་ཁ་པར་ནང་གློག་འཕྲིན་དང་ངོ་དེབ་སོགས་བེད་སྤྱོད་གཏོང་གི་ཡོད་ཚེ། དུས་བབ་ཁ་པར་མཉེན་ཆས་སྟེང་གི་ནང་འཛུལ་ནང་ཕྱིན་ན། ངོ་དེབ་ (Facebook)ཀྱི་རྟགས་ཤིག་དང་། གློག་འཕྲིན་ (Gmail)གྱི་རྟགས་ཤིག་ཡོད་པ་དེ་གཉིས་གང་གི་སྟེང་གནན་ན་ནང་འཛུལ་བྱེད་ཐུབ། (7.) དེབ་སྐྱེལ་བྱེད་སྟངས། ཁྱོད་ཀྱིས་དེབ་སྐྱེལ་གསར་དུ་བྱེད་པར་དུས་བབ་ཁ་པར་མཉེན་ཆས་མདུན་ངོས་སྒང་གི་གཡོན་ཟུར་གྱི་ནང་འཛུལ་ཞེས་པར་ཕྱིན་ཞིང་། དེའི་འོག་གི་གཡོན་ཟུར་དུ་ ‘དེབ་སྐྱེལ’ ཞེས་པ་ཞིག་ཡོད་པ་དེ་གནན་ཏེ། མིང་དང་། གློག་འཕྲིན། ཐོ་མིང་སོགས་རེ་རེ་བཞིན་བསྐོངས་ཏེ་འོག་གི་དེབ་སྐྱེལ་ཞེས་པར་མནན་རྗེས། སོ་སོའི་གློག་འཕྲིན་ནང་ཕྱིན་ན་ཁྱོད་ལ་བོད་ཀྱི་དུས་བབ་ཀྱིས་གློག་འཕྲིན་ཞིག་འབྱོར་ཡོད་པའི་ནང་གི་འབྲེལ་ཐག་སྒང་མ་དེར་མནན་ཏེ་གསང་ཚིག་བཟོ་དགོས། གསང་ཚིག་གསར་པ་ཡར་འཇུག་བྱས་རྗེས་བསྐྱར་དུ་ནང་འཛུལ་ནང་ཕྱིན་ཏེ་གློག་འཕྲིན་དང་གསང་ཚིག་གཉིས་མ་ནོར་བར་བཏགས་ཏེ་ནང་འཛུལ་བྱེད་དགོས། (8.) ཐོ་མིང་དང་གསང་ཚིག བོད་ཀྱི་དུས་བབ་དྲ་ཚིགས་ཨ་མའི་ནང་དེབ་སྐྱེལ་བྱས་པ་དང་དུས་བབ་ཁ་པར་མཉེན་ཆས་ནང་དེབ་སྐྱེལ་བྱས་པ་གཅིག་པ་ཡིན་ཞིང་། དེ་གཉིས་ཕན་ཚུན་གཅིག་གྱུར་དུ་བེད་སྤྱོད་གཏོང་ཐུབ། དྲ་ཚིགས་ཨ་མའི་སྟེང་དེབ་སྐྱེལ་བྱས་པའི་ཐོ་མིང་དང་གསང་ཚིག་དུས་བབ་མཉེན་ཆས་ནང་བེད་སྤྱོད་གཏོང་ཐུབ་ལ། དུས་བབ་ཁ་པར་མཉེན་ཆས་སྟེང་དེབ་སྐྱེལ་བྱས་པའི་ཐོ་མིང་དང་གསང་ཚིག་དུས་བབ་དྲ་ཚིགས་ཨ་མའི་ནང་དུའང་བེད་སྤྱོད་གཏོང་ཐུབ། (དུས་བབ་མཉེན་ཆས་ནང་གློག་འཕྲིན་དང་གསང་ཚིགས་གཉིས་བེད་སྤྱོད་བཏང་སྟེ་ནང་འཛུལ་བྱེད་དགོས་ལ། དུས་བབ་དྲ་ཚིགས་ཨ་མའི་སྟེང་གློག་འཕྲིན་དང་ཐོ་མིང་གང་རུང་དང་། འོག་ཏུ་གསང་ཚིག་བེད་སྤྱོད་བཏང་སྟེ་ནང་འཛུལ་བྱེད་དགོས།) (9.) གསང་ཚིག་བརྗེད་པ། གལ་སྲིད་ཁྱོད་ཀྱིས་སྔོན་ཆད་དེབ་སྐྱེལ་བྱས་པའི་གསང་ཚིག་བརྗེད་པ་ཡིན་ན། དུས་བབ་མཉེན་ཆས་ཀྱི་ནང་འཛུལ་ནང་གི་ཁ་སྐོང་འོག་ཏུ་ ‘གསང་ཚིག་བརྗེད་ན་འདིར་གནོན།’ཞེས་པ་ཞིག་ཡོད་པ་དེ་བརྒྱུད་ནས་ཁྱོད་ཀྱིས་དེ་སྔ་དུས་བབ་ནང་བེད་སྤྱོད་བཏང་ཡོད་པའི་གློག་འཕྲིན་དེ་ཁ་སྐོང་ནང་བཏགས་ཏེ་གསང་ཚིག་བསྐྱར་གསོ་བྱེད་དགོས། (10.) མིང་བརྗེ་བོ་རྒྱག་པ། ཁྱོད་རང་ནང་འཛུལ་བྱས་རྗེས། ‘ནང་འཛུལ་བྱེད་ས་དེར་ཁྱོད་ཀྱི་མིང་བོད་ཡིག་དང་དབྱིན་ཡིག་གང་རུང་ཐོག་འཆར་ཡོད། ཁྱོད་ཀྱིས་མིང་དེ་བརྗེ་བོ་རྒྱག་འདོད་ན། མིང་དེའི་སྟེང་གནན་ན་བརྗེ་བོ་རྒྱག་ཐུབ། རྗེས་མ་མིང་བསྐྱར་དུ་བརྗེ་བོ་རྒྱག་པར་ཡང་གོང་ལྟར་མིང་དེའི་སྟེང་གནན་ན་བརྗེ་བོ་རྒྱག་ཐུབ། བོད་ཀྱི་དུས་བབ་མཉེན་ཆས་བེད་སྤྱོད་གནང་མཁན་རྣམ་པའི་དཀའ་ངལ་དང་དོགས་འདྲི་ལ་དུས་ནས་དུས་སུ་ལམ་སྟོན་གསལ་བཤད་ཞུ་བཞིན་ཡོད། མཐའ་མའི་ཁ་སྣོན་ ༢༠༡༧།༧།༡༥
