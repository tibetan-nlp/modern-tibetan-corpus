རྩམ་པའི་ནུས་པ་བསམ་གྱིས་མི་ཁྱབ།

ཁ་སང་རྒྱ་ནག་གི་གནའ་རྫས་ཞིབ་འཇུག་པ་འགས་མཐོ་སྒང་དུ་བོད་མི་རིགས་འཚོ་སྡོད་བྱེད་ཐུབ་པ་དེ་ནི་ནས་ཀྱི་རྩམ་པ་ཟོས་པའི་ནུས་པ་ལ་འབྲེལ་བ་ཡོད་པ་ར་འཕྲོད་ཡོད་སྐོར་བསྒྲགས་འདུག ཟླ་འདིའི་ཚེས་ ༢༠ ཉིན་ཚན་རིག་ (Science) ཞེས་པའི་དྲ་རྒྱའི་ཐོག་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། རྒྱ་ནག་ལན་ཀྲུག་གཙུག་ལག་སློབ་ཆེན་གྱི་གནའ་རྫས་ཞིབ་འཇུག་པ་ཁྲིན་ཧྥ་ཧུའུ་དང་ (Chen Fahu) དང་ཏུང་ཀང་ཧེ་ (Dong Guanghui)ཞུ་བ་གཉིས་ཀྱིས་ད་ལྟའི་རྒྱ་ནག་འོག་གི་མཚོ་སྔོན་ཞིང་ཆེན་གྱི་ནུབ་བྱང་ཁུལ་གྱི་གནའ་ཤུལ་ཁག་ནས་རྙེད་སོན་བྱུང་བའི་གདོད་མའི་གནའ་རྫས་མང་པོའི་ཁྲོད། འབྲུ་རྙིང་དག་ལ་བརྟག་དཔྱད་བྱས་པ་བརྒྱུད་དེ། བོད་རིགས་ཚོ་མཐོ་ཚད་ལ་རྒྱ་མཚོའི་ངོས་ནས་ཧྥི་ཊི་ ༨༢༠༠ ལྷག་ལ་འཚོ་སྡོད་བྱེད་ཐུབ་པ་འདི་ནི། ནས་ཀྱི་རྩམ་པ་ཟོས་པར་བརྟེན་ཏེ་བྱུང་བ་ཞིག་ཡིན་ཞིང་། དེ་ནི་ནས་རྩམ་གྱིས་ཟས་བཅུད་དང་མཁའ་དབུགས་མ་འདང་བར་འཚོ་རླུང་སྟེར་གྱི་ཡོད་ཚུལ་བསྒྲགས་འདུག འདིའི་ཐད་ཁྲིན་ཧྥ་ཧུའུ་དང་ཏུང་ཀང་ཧྲེ་གཉིས་ཀྱིས་བརྗོད་དོན། “ང་ཚོས་མཚོ་སྔོན་ས་ཁུལ་གྱི་གནའ་རྫས་དག་ལ་ཞིབ་འཇུག་བྱས་ཏེ། ལོ་ངོ་ ༢༠༠༠༠ ཡར་སྔོན་ནས་མཐོ་སྒང་དུ་འགྲོ་བ་མིའི་རིགས་འགྲོ་འོང་བྱས་ཡོད་ཀྱང་སྐབས་དེར་གྲང་ངར་ཧ་ཅང་ཆེ་སྟབས། ཡུལ་ལུང་གཞན་ནང་ཕྱིར་ལོག་དགོས་བྱུང་ཡོད་པ་དང་། ལོ་ངོ་ ༣༦༠༠ སྔོན་ལ་མཐོ་ཚད་ལ་རྒྱ་མཚོའི་ངོས་ནས་རྨི་ཊར་ ༢༥༠༠ དང་དེ་ལས་ཉུང་བའི་རྨ་ཆུའི་ས་ཁུལ་མ་གཏོགས་བོད་ཀྱི་ས་མཐོར་ཞིང་པ་མེད་སྟབས། ཞིང་པ་ཚོས་མོན་ཆག་འདེབས་ལས་ཀྱིས་འཚོ་བ་བསྐྱལ་ཡོད་ཀྱང་། ལོ་ངོ་ ༣༦༠༠ རྗེས་ཞིང་པ་ཚོ་མཐོ་ཚད་ལ་རྒྱ་མཚོའི་ངོས་ནས་རྨི་ཊར་ ༣༤༠༠ ལྷག་ཡོད་པའི་བོད་མཐོ་སྒང་དུ་གནས་སྤོ་བྱས་ཏེ། སོ་ནམ་འདེབས་ལས་བྱས་ཏེ་འཚོ་སྡོད་བྱེད་མགོ་བཙུགས་ཡོད་ཅིང་། དེ་ནི་ཞིང་པ་ཚོས་བོད་ཀྱི་གནམ་གཤིས་ལ་དཔག་པའི་ནས་ཀྱི་སོན་གསར་རྙེད་བྱུང་སྟེ། ནས་འདེབས་ལས་དང་རྩམ་པ་ཟོས་རྗེས་བོད་རིགས་མི་རིགས་དེ་འཛམ་གླིང་གི་ཆེས་མཐོ་བའི་ས་བབ་ཐོག་འཚོ་གནས་ཐུབ་ཡོད་པ་ཤེས་རྟོགས་བྱུང་སོང་།”ཞེས་བརྗོད་འདུག ཨ་རིའི་ཝ་ཤིན་ཊོན་གཙུག་ལག་སློབ་ཆེན་གྱི་གནའ་རྫས་རིག་པ་ཇུ་ཌི་ (Jade d’Alpoim Guedes) ལགས་ཀྱིས། “ལོ་ངོ་ ༡༠༥༠༠ ཡར་སྔོན་དུ་ནས་དང་གྲོ་གཉིས་ཐོག་མར་ཨེ་ཤ་ཡའི་ནུབ་ཁུལ་དུ་འདེབས་ལས་བྱས་ཡོད་ཀྱང་། དེའི་རྗེས་རིམ་གྱིས་གནམ་གཤིས་དང་ཁོར་ཡུག་མི་འདྲ་བའི་ཆ་རྐྱེན་འོག་མཐོ་སྒང་དུ་འདེབས་ལས་བྱས་པ་ནི། འཛམ་གླིང་འདི་གཅིག་ལ་གཅིག་རྟེན་ཡིན་པ་གསལ་བཤད་བྱས་ཡོད།”ཅེས་བརྗོད་འདུག འཇར་མན་ཏུའུ་བྷེན་གཙུག་ལག་སློབ་ཆེན་གྱི་གནའ་རྫས་རིག་པ་བ་ཀར་ཌི་ (Kurt Rademaker) ལགས་ཀྱིས། “ཐོག་མ་འཛམ་གླིང་ཐོག་གི་སོ་ནམ་ཞིང་ལས་དེ་མཐོ་ཚད་ལ་རྒྱ་མཚོའི་ངོས་ནས་མི་ཊར་ ༣༤༠༠ ལྷག་ཡོད་པའི་གྲང་ངར་ཆེ་བའི་མཐོ་སྒང་དུ་ཁྱབ་ཡོད་པ་མ་ཟད། ནས་འདེབས་ལས་བྱེད་ཐུབ་པ་དེས་གནམ་གཤིས་དེ་འགོག་རྐྱེན་ཞིག་མིན་པ་ར་སྤྲོད་བྱས་པ་རེད།”ཅེས་དང་། ཨ་རིའི་ཁེ་ལི་ཧྥོར་ནི་ཡ་གཙུག་ལག་སློབ་ཆེན་གྱི་གནའ་རྫས་རིག་པ་མག་ (Mark Aldenderfer) ལགས་ཀྱིས། “ད་རེབ་ཉམས་ཞིབ་འདི་བརྒྱུད་ནས་བོད་ནང་ལོ་ངོ་ ༣༦༠༠ ལྷག་གི་གོང་ནས་ཞིང་ལས་ཡོད་པ་ར་སྤྲོད་བྱས་ཡོད་ཀྱང་། དེ་སྔོན་ཚན་རིག་པས་བོད་མིའི་རིགས་རྒྱུན་ལ་བརྟག་དཔྱད་བྱས་ཏེ། བོད་མི་མི་རིགས་དེ་མཐོ་ཚད་རྒྱ་མཚོའི་ངོས་ནས་རྨི་ཊར་ ༤༠༠༠ ནས་ ༥༠༠༠ ཡན་གྱི་ས་བབ་ནང་འཚོ་སྡོད་བྱེད་ཐུབ་ཀྱི་ཡོད་པ་མ་ཟད། བོད་མིའི་རིགས་རྒྱུན་ནང་ཡོད་པའི་ས་བབ་མཐོ་སར་འཚོ་ཐུབ་པའི་རིགས་རྫས་ (EPAS1) ཞེས་པ་དེར་འཚོ་རླུང་སྟངས་འཛིན་བྱེད་པའི་ནུས་པ་གཅིག་མཚུངས་ཡོད་པ་ར་སྤྲོད་བྱས་ཡོད། རིགས་རྫས་དེ་ནི་ལུས་ཟུངས་ཀྱི་སྐྱེ་ལྡན་ཕྲ་ཕུང་ཁོར་ཡུག་ལ་དཔག་པའི་གོམས་འདྲིས་བྱེད་ཐུབ་པའི་རིགས་རྫས་ཤིག་ཡིན་སྟབས། བོད་མི་ཚོ་མི་རིགས་གཞན་འཚོ་སྡོད་བྱེད་དཀའ་བའི་འཚོ་རླུང་ཉུང་བའི་ས་ཁུལ་ནང་འཚོ་ཐུབ་ཀྱི་ཡོད་པ་བདེན་དཔང་བྱས་ཡོད་པ་མ་ཟད། བོད་མི་ཚོ་ལོ་ངོ་སྟོང་ཕྲག་མང་པོའི་སྔོན་ལ་བོད་ནང་འཚོ་སྡོད་བྱེད་ཀྱི་ཡོད་པའང་གསལ་བཤད་བྱས་ཡོད།”ཅེས་བརྗོད་འདུག དབུས་ཤར་ཨེ་ཤ་ཡའི་ནང་སྟོན་ཐོག་འདེབས་ལས་བྱས་ཏེ་རྒྱལ་ཁབ་ཕན་ཚུན་བར་འབྲུ་རིགས་འགྲེམས་ཚོང་བྱེད་མགོ་བཙུགས་པ་ནས་བཟུང་སྟེ། ཕྱི་ལོ་མ་བྱུང་སྔོན་གྱི་ལོ་ ༤༠༠༠ ནས་ ༤༥༠༠ གོང་དུ་དབུས་ཤར་ཨེ་ཤ་ཡ་ནས་བོད་དང་རྒྱ་ནག་ནང་ནས་དང་གྲོ་བྱུང་ཡོད་པ་དང་། རྒྱ་ནག་ནས་དབུས་ཤར་ཨེ་ཤ་ཡའི་ནང་མོན་ཆག་དང་ལོ་གསུམ་ཁམ་བུ་དང་ཀུ་ཤུ། འབྲས་སོགས་ཕན་ཚུལ་ཚོང་འབྲེལ་བྱས་ཡོད་པ་མ་ཟད། དེ་བཞིན། ཕྱི་ལོ་མ་བྱུང་སྔོན་གྱི་ལོ་ ༢༠༠༠ ནས་ ༢༥༠༠ ནང་རྒྱ་ནག་ནས་ཡུ་རོབ་བར་གྱི་དར་གོས་ཚོང་ལམ་ (silk road) སྒོ་ཕྱེ་སྟེ། རྒྱལ་ཁབ་ཕན་ཚུན་བར་ཚོང་འབྲེལ་བྱེད་མགོ་བཙུགས་ཡོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
