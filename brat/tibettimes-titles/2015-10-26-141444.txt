སྐྱབས་བཅོལ་བའི་ཛ་དྲག་གནས་སྟངས།

  ཡུ་རོབ་མཐུན་ཚོགས་ཀྱིས་དཀྱིལ་ཤར་ཡུལ་གྲུ་ནས་ཡུ་རོབ་ཀྱི་རྒྱལ་ཁབ་ཁག་ཏུ་སྐྱབས་བཅོལ་ཡོང་མཁན་ཚོའི་དཀའ་སྡུག་སེལ་ཐབས་སྐོར་ལ་གྲོས་མོལ་བྱས་འདུག   ཁ་སང་སྦེལ་ལྗམ་རྒྱལ་ས་སྦར་སིལ་དུ་ཡུ་རོབ་མཐུན་ཚོགས་ཀྱི་ཚོགས་གཙོ་ཇེན་ཁཱོ་ལོ་ཌི་ཇུན་ཁཱར་ (Jean-Claude Juncker) ལགས་ཀྱིས། མཐུན་ཚོགས་ཀྱི་ཚོགས་མིའི་རྒྱལ་ཁབ་ ༡༠ ཡི་མགོ་ཁྲིད་དང་། ཡུ་རོབ་ཀྱི་ལྷོ་ཤར་རྒྱལ་ཁབ་ཨལ་བྷ་ནི་ཡ་དང་སར་བྷེ་ཡ། མ་སི་ཌོ་ནི་ཡ་བཅས་ཀྱི་མགོ་ཁྲིད་བར་ཛ་དྲག་གི་ཚོགས་འདུ་བསྡུས་ཏེ། ཡུ་རོབ་མཐུན་ཚོགས་ཀྱིས་བདུན་ཕྲག་གཅིག་ནང་ཚུད་དུ་སི་ལོ་ཝེ་ནི་ཡར་ཉེན་རྟོག་པ་ ༤༠༠ བཏང་ནས་སྐྱབས་བཅོལ་བར་རོགས་རམ་བྱ་རྒྱུ་དང་། སྐྱབས་བཅོལ་བ་རྒྱལ་ཁབ་གཅིག་ནས་རྒྱལ་ཁབ་གཞན་ལ་གནས་སྤོ་སྐབས་རྒྱལ་ཁབ་དེའི་ཆོག་མཆན་ཞུ་དགོས་པ་སོགས་ཀྱི་གྲོས་འཆར་ཁག་ ༡༦ ལ་གྲོས་མོལ་གནང་ཡོད་འདུག   ཉེ་ལམ་འཇར་མན་དང་ཨོ་སི་ཊིར་ཡ། ཧང་གྷ་རི་བཅས་ཀྱིས་ས་མཚམས་ཁུལ་དུ་ལྕགས་རྭ་བསྐོར་ཏེ། ཡུ་རོབ་ཀྱི་ནུབ་བྱང་རྒྱལ་ཁབ་བྷུལ་གྷེ་རི་ཡ་དང་རོ་མ་ནི་ཡ། སར་བྷེ་ཡ་བཅས་བརྒྱུད་དེ། སི་རི་ཡ་དང་ཨབ་གྷན་གྱི་སྐྱབས་བཅོལ་བ་ཚོ་ཡུ་རོབ་རྒྱལ་ཁབ་ཁག་ཏུ་འཛུལ་འཇུག་གི་མེད་སྟབས། བདུན་ཕྲག་སྔོན་མའི་རེས་གཟའ་སྤེན་པའི་ཉིན་བྷུལ་གྷེ་རི་དང་རོ་མ་ནི་ཡ། སར་བྷེ་ཡ་བཅས་ཀྱི་རྒྱལ་ཁབ་ཀྱི་མགོ་ཁྲིད་ཚོས་གསར་འགོད་ལྷན་ཚོགས་བསྐོངས་ཏེ། འཇར་མན་དང་ཨོ་སི་རི་ཡ་བཅས་ཀྱི་རྒྱལ་ཁབ་ཁག་གིས་ས་མཚམས་བཀག་ཡོད་སྟབས། ང་ཚོས་ཀྱང་ས་མཚམས་ཁག་ལ་ལྟ་སྲུང་ནན་མོ་བྱས་ཏེ་སྐྱབས་བཅོལ་བ་ཡོང་དུ་འཇུག་རྒྱུ་མིན་སྐོར་བརྗོད་ཡོད་པ་རེད།   ཡུ་རོབ་ཀྱི་རྒྱལ་ཁབ་ཁག་ནང་སྐྱབས་བཅོལ་དུ་ཡོང་མཁན་མང་ཆེ་བ་སི་རི་ཡ་དང་དཀྱིལ་ཤར་ཡུལ་གྲུ་ཨབ་གྷན་དང་ལྷོ་ཨ་ཧྥི་རི་ཁའི་རྒྱལ་ཁབ་ཁག་ཡིན་པ་དང་། བདུན་ཕྲག་སྔོན་མའི་ནང་སི་ལོ་ཝེ་ནི་ཡའི་ནང་སྐྱབས་བཅོལ་བ་ ༦༠༠༠༠ ལྷག་དང་གྷེར་སིའི་ནང་ ༤༨༠༠༠ འབྱོར་ཡོད་ཅིང་། འདི་ལོ་གྷེར་སི་དང་ཀྲར་ཀི། སི་ལོ་ཝེ་ནི་ཡ་བརྒྱུད་ཡུ་རོབ་རྒྱལ་ཁབ་ཁག་ཏུ་སྐྱབས་བཅོལ་བ་ ༦༧༠༠༠༠ འབྱོར་ཡོད་འདུག   ཉིན་ཤས་གོང་ཧང་གྷེ་རི་དང་འཇར་མན་ནེ་སོགས་ཀྱིས་ས་མཚམས་ཁག་བཀག་སྟེ་སྐྱབས་བཅོལ་བ་འཛུལ་བཅུག་གི་མེད་པ་དང་། ཨལ་བྷ་ནི་ཡ་དང་སར་བྷེ་ཡ། མ་སི་ཌོ་ནི་ཡ་བཅས་ཀྱིས་ཀྱང་ས་མཚམས་ཁག་བཀག་ནས་སྐྱབས་བཅོལ་བ་ཡོང་འཇུག་གི་མེད་པར། ད་ལྟ་སྐྱབས་བཅོལ་བ་འབུམ་གཅིག་ཙམ་ཞིག་གིས་དགུན་དུས་ཀྱི་གྲང་ངར་ལ་མ་ལྟོས་པར་ས་མཚམས་ཁུལ་དུ་སྒར་བརྒྱབ་སྟེ་དཀའ་སྡུག་གི་འཚོ་བ་རོལ་བཞིན་ཡོད་སྟབས། རྒྱལ་སྤྱིའི་ཉེས་ཡང་ཚོགས་པས་རྒྱལ་སྤྱིའི་སྒྲིག་འཛུགས་ཁག་གིས་མཉམ་རུབ་བྱས་ནས་ཁོ་ཚོའི་དཀའ་ངལ་མ་སེལ་ན། ང་ཚོ་འགྲོ་བ་མིའི་རིགས་དམར་གསོད་གཏོང་མཁན་གྱི་གཤེད་མ་ངོ་མ་ཆགས་རྒྱུ་ཡིན་སྐོར་སྐྱོན་བརྗོད་བྱས་འདུག   ཉིན་ཤས་གོང་དུ་བྷུལ་གྷེ་རི་ཡའི་སྲིད་བློན་བྷོ་ཀཱོ་ (Boyko Borisov) ཡིས་འཇར་མན་དང་ཨོ་སི་ཊིར་ཡ་གཙོས། རྒྱལ་ཁབ་ཁག་གིས་ས་མཚམས་བཀག་ཚེ། ང་ཚོས་ཀྱང་སྒོ་རྒྱག་རྒྱུ་ཡིན་སྐོར་བརྗོད་ཡོད་པ་མ་ཟད། སར་བྷེ་རི་ཡའི་སྲིད་བློན་ཨཱ་ལེག་ཛན་ཌར་ (Aleksandar Vucic) གྱིས་ཀྱང་ང་ཚོས་འཕྱར་བ་གྲུ་འདེགས་བྱས་ནས་སྐྱབས་བཅོལ་བའི་དཀའ་སྡུག་མ་སེལ་ན། མ་འོངས་པར་ང་ཚོ་རྒྱལ་སྤྱིའི་ལོ་རྒྱུས་ཁྲོད་ཀྱི་ནག་ཉེས་ཅན་ཆགས་རྒྱུའི་ཉེན་ཁ་ཡོད་སྟབས། ད་ལྟ་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱིས་ཁ་བཤད་ལག་བསྟར་བྱེད་དགོས་པའི་དུས་ལ་བབ་འདུག་ཅེས་བརྗོད་འདུག   འདི་ལོ་དཀྱིལ་ཤར་ཡུལ་གྲུའི་དམག་འཁྲུག་གི་རྐྱེན་པས་ཡུ་རོབ་རྒྱལ་ཁབ་ཁག་ཏུ་སྐྱབས་བཅོལ་ལ་ཡོང་མཁན་ཇེ་མང་དུ་ཕྱིན་ཡོད་པ་མ་ཟད། སྔོན་ཆད་དང་བསྡུར་ན་དཔལ་འབྱོར་གནས་སྟངས་ཡིན་ཡང་ད་ལྟ་ཞན་པོ་ཆགས་ཏེ། གཞོན་སྐྱེས་ལས་མེད་ས་ཡ་ ༥ མ་ཟིན་ཙམ་བྱུང་ཡོད་སྟབས། ད་ལྟ་ཡུ་རོབ་ཀྱི་རྒྱལ་ཁབ་ཁ་ཤས་ཀྱིས་སྐྱབས་བཅོལ་བ་ལེན་རྒྱུར་ཁས་ལེན་བྱེད་ཀྱི་མེད་པ་དང་། ཁ་ཤས་ཀྱིས་སྐྱབས་བཅོལ་བའི་གནད་དོན་དང་མཐའ་མཚམས་བདེ་འཇགས་ཐོག་རྟོག་ཞིབ་ཤུགས་ཆེ་རུ་གཏོང་གི་ཡོད་སྟབས། ད་ལྟ་སྐྱབས་བཅོལ་གནད་དོན་ཐད་ཡུ་རོབ་རྒྱལ་ཁབ་ཁག་གྲོས་འཆམ་ཡོང་ཁག་པོ་ཆགས་ཡོད་པ་མ་ཟད། ཡུ་རོབ་མཐུན་ཚོགས་ནང་ཁུལ་བར་ལྟ་བ་ལངས་ཕྱོགས་མ་མཐུན་པའི་དཀའ་ངལ་འཕྲད་བཞིན་ཡོད་འདུག   ད་ལྟའི་བར་རྒྱལ་སྤྱིའི་ཐོག་རང་ཡུལ་ནས་བྲོས་བྱོལ་གྱིས་རྒྱལ་ཁབ་གཞན་ནང་སྐྱབས་བཅོལ་དུ་འགྲོ་མཁན་ས་ཡ་ ༦༠ དང་འབུམ་ ༥༠ ལྷག་བྱུང་ཡོད་སྟབས། མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་སྤྱི་ཁྱབ་དྲུང་ཆེ་བྷན་ཁེ་མོན་མཆོག་དང་རྒྱལ་སྤྱིའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལྷན་ཚོགས་ཀྱི་ཚོགས་གཙོས། རྒྱལ་ཁབ་ཁག་གིས་གཅིག་ཕན་གཅིག་གྲོགས་བྱས་ཏེ་དཀའ་ངལ་སེལ་ཐུབ་བྱེད་དགོས་པ་དང་། ཡུ་རོབ་མཐུན་ཚོགས་ངོས་ནས་སྐྱབས་བཅོལ་ལ་ཡོང་མཁན་ཚོར་འགྲོ་ལམ་གསར་པ་ཞིག་ཕྱེ་དགོས་སྐོར་བརྗོད་ཡོད་པ་དང་། རྒྱལ་སྤྱིའི་སྐྱབས་བཅོལ་ལྷན་ཚོགས་ནས་ཀྱང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་ཚོགས་འདུའི་སྐབས་སྐྱབས་བཅོལ་གནས་སྟངས་གང་འདྲ་ཆགས་མིན་གྱི་གྲུབ་དོན་སྙན་ཐོ་བཏོན་ཏེ་ཁོ་ཚོའི་དཀའ་སྡུག་གསལ་སྟོན་བྱ་རྒྱུ་ཡིན་འདུག   དེ་རིང་ཨ་རབ་གསར་འཕྲིན་ཐོག་འཇར་མན་གྱི་སྲིད་བློན་ལྕམ་གྷང་གྷེ་ལ་མེར་གྷལ་མཆོག་གིས། འདི་ནི་ཡུ་རོབ་མཐུན་ཚོགས་ཀྱི་གདོང་ལེན་བྱེད་དགོས་པ་ཆེ་ཤོས་གྲས་ཤིག་ཆགས་ཡོད་སྟབས། ད་ལྟ་དུས་ཚོད་ལ་དཔག་པའི་དམིགས་བསལ་ཐབས་ལམ་ཞིག་འཚལ་རྒྱུ་གལ་ཆེན་པོ་ཆགས་ཡོད། ཅེས་དང་། ཨོ་སི་ཌོ་རི་ཡའི་སྲིད་བློན་ཝར་ནེ་ (Werner Faymann) ཡིས་གསར་འགྱུར་བརྒྱུད་ལམ་ (Kronen Zeitung daily) ཞེས་པས་བཅར་འདྲི་བྱེད་སྐབས། ད་ལྟ་ང་ཚོས་ཡུ་རོབ་གཅིག་སྒྲིལ་བཟོ་དགོས་མིན་དང་། ཡང་ན་རིམ་པས་འཐོར་བཤིག་འགྲོ་བར་བལྟས་ནས་སྡོད་དགོས་མིན་གཉིས་གདམ་ཀ་བྱེད་དགོས་པའི་གནས་ལ་གྱར་ཡོད་པ་རེད། ཅེས་བརྗོད་འདུག   གསར་འགོད་པ། དོན་གྲུབ།
