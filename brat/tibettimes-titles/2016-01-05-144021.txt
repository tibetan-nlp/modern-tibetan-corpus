བྱ་བྲལ་རིན་པོ་ཆེ་མཆོག་དགོངས་པ་རྫོཊ།

ཕྱི་ཟླ་ ༡༢ ཚེས་ ༣༠ ཉིན་ྋསྐྱབས་རྗེ་བྱ་བྲལ་རིན་པོ་ཆེ་སངས་རྒྱས་རྡོ་རྗེ་མཆོག་བལ་ཡུལ་གྱི་ཡང་ལེ་ཤོད་གཟིམ་ཤག་ནས་དགོངས་པ་ཆོས་དབྱིངས་སུ་འཐིམས་ཡོད་པ་དང་། དེ་རིང་ཐུགས་དམ་གྲོལ་ནས་དད་ལྡན་གདུལ་བྱ་ཡོངས་ལ་ཐུགས་ཕམ་ཆེན་པོས་ཡོངས་བསྒྲགས་གནང་ཡོད་འདུག   དེ་རིང་འདི་གའི་གསར་འགོད་པར་འབྲེལ་ཡོད་ཅིག་གིས་འགྲེལ་བརྗོད་གནང་བ་ལྟར་ན། བྱ་བྲལ་རིན་པོ་ཆེ་མཆོག་ལ་དམིགས་བསལ་གྱི་སྙུང་གཞི་གང་ཡང་མེད་པར་སྐུ་ཞི་བར་གཤེགས་ཡོད་པ་མ་ཟད། ཐུགས་དམ་དུ་བཞུགས་ཡོད་སྟབས་བདུན་ཕྲག་གཅིག་རིང་གསང་བ་བྱས་ནས་ཕྱིར་བསྒྲགས་གནང་མེད་སྐོར་བརྗོད་བྱུང་།   ྋསྐྱབས་རྗེ་བྱ་བྲལ་སངས་རྒྱས་རྡོ་རྗེ་མཆོག་ནི་ཕྱི་ལོ་ ༡༩༡༣ ཟླ་ ༦ ཚེས་ ༡༨ ཉིན་དཀར་མཛེས་ཁུལ་ཉག་རོང་དུ་ཡབ་པདྨ་དོན་གྲུབ་དང། ཡུམ་བསོད་ནམས་མཚོ་གཉིས་ཀྱི་སྲས་སུ་སྐུ་འཁྲུངས་ཤིང་། ད་ལོ་བོད་ལུགས་ལྟར་དགུང་གྲངས་ ༡༠༤ ལ་ཕེབས་ཡོད་འདུག རྗེ་འདི་ཉིད་ཀྱིས་རིས་སུ་མ་ཆད་པའི་མཁས་གྲུབ་གཉིས་ལྡན་གྱི་སྐྱེས་ཆེན་དམ་པ་བཞི་བཅུ་ལྷག་གི་ཞབས་ལ་གཏུགས་ནས་གདམས་ངག་མང་པོར་གསན་བསམ་མཛད་ཡོད་པ་མ་ཟད། བོད་ཀྱི་རྒྱལ་ཚབ་རྭ་སྒྲེང་རིན་པོ་ཆེའི་དབུ་བླ་མཛད་པ་སོགས་གདུལ་བྱ་དྲག་ཞན་ཕོ་མོ་མང་པོའི་རྒྱུད་ལ་རྣམ་དཀར་དགེ་བའི་ལས་ཀྱི་ས་བོན་བཏབ་ཡོད་འདུག བོད་ས་ཡོངས་སུ་དུས་འགྱུར་བྱུང་རྗེས་ཁོང་རྒྱ་གར་ནང་ཕེབས་ཏེ་མུ་མཐུད་རྒྱལ་བའི་བསྟན་པ་རིན་པོ་ཆེ་འཛིན་སྐྱོང་སྤེལ་གསུམ་གྱི་ཐད་མཛད་པ་རྒྱ་ཆེན་པོ་བསྐྱངས་ཡོད་པ་མ་ཟད། རྒྱུ་དངོས་ཡོད་ཚད་ཉ་དང་ར་ལུག་སོགས་ཉམ་ཆུང་སེམས་ཅན་འབུམ་ཕྲག་མང་པོའི་སྲོག་བླུ་ཚེ་ཐར་ཆེད་ལོ་ལྟར་གཏོང་གནང་མཛད་པ་དང་འབྲེལ། ཁོང་རང་ཉིད་ཀྱི་སྐུ་ཚེའི་རིང་དམར་ཟས་གཏན་ནས་མི་མཆོད་ཅིང་གཞན་དག་ལའང་སྲོག་བཅད་པ་དང་དམར་ཟས་བསྟེན་པའི་ཉེས་དམིགས་ཀྱི་སྐོར་ལ་བཀའ་སློབ་དང་ལྗགས་རྩོམ་མཛད་པ་སོགས་སྐུ་ཚེ་ཧྲིལ་པོ་འདོད་ཆུང་ཆོག་ཤེས་དང་རྣལ་འབྱོར་གྱི་སྤྱོད་པ་ཁོ་ནས་འགྲོ་དོན་མཛད་ཡོད་འདུག   ཕྱི་ལོ་ ༢༠༠༥ ཟླ་ ༥ ཚེས་ ༢༩ ཉིན་དུད་འགྲོའི་རང་དབང་ཚོགས་ཆུང་གིས་རིན་པོ་ཆེ་མཆོག་ལ་བཅར་འདྲི་ཞུ་སྐབས། ཁོང་གི་བཀའ་སློབ་ལམ་སྟོན་འོག་རྙིང་མའི་སྨོན་ལམ་ཆེན་མོའི་སྐབས་སུ་དམར་ཟས་སྤངས་དུ་བཅུག་པ་དང་འབྲེལ་བའི་བཀའ་སློབ་གནང་དོན། ང་ཚོ་རྒྱ་གར་ལ་ཡོང་ནས་ཤ་སྤོང་མཁན་དང་པོ་དེ་ང་ཡིན། རྒྱ་གར་རྡོ་རྗེ་གདན་ལ་རྙིང་མའི་སྨོན་ལམ་ཆེན་མོ་དང་པོ་བཙུགས་པའི་ལོ་དེར་ཤ་བཟས་པ་རེད། གཉིས་པའི་ལོ་དེར་ང་སླེབས་ནས་དེ་ནས་ཡོངས་སུ་རྫོགས་པ་ཚོགས་འདུའི་ནང་ལ་རྙིང་མ་བའི་བླ་ཆེན་ཐམས་ཅད་ཡོད་ས་འདིའི་ནང་ལ་ཤ་མ་སྤང་ན། རྒྱ་གར་རྡོ་རྗེ་གདན་སྟོན་པ་སངས་རྒྱས་པའི་གནས་ནང་པ་སངས་རྒྱས་པ་ཡོད་ཚད་ཀྱི་ཡོངས་གྲགས་ཀྱི་གནས་ཁྱད་པར་ཅན་རེད། དེའི་ནང་ལ་སྔ་འགྱུར་སྨོན་ལམ་ཆེན་མོ་ཚོགས་ཀྱི་ཡིན་ཟེར་ནས་ཤ་བཟས་ན་ནི་དེ་ལས་ཞབས་འདྲེན་ཆེ་བ་གཅིག སངས་རྒྱས་བསྟན་པ་ལ་དེ་ལས་ཐབས་སྡུག་པ་ཞིག་མེད་པ་རེད། ཁྱེད་རང་ཚོ་ཡོངས་སུ་རྫོགས་པ་ད་ནས་བཟུང་སྟེ་ཇི་སྲིད་སྨོན་ལམ་ཚོགས་རྒྱུན་རིང་ཤ་ཡི་ལས་ཀ་རྩ་བ་ནས་མ་བྱེད། བོད་ཀྱི་གྲྭ་པ་ཚོས་ཤ་བཟས། བླ་མ་ཚོས་སྤང་མ་ཐུབ་ན་ད་རུང་དེ་ལས་ཐབས་སྡུག་གི་རེད། དང་པོ་བླ་མ་ཚོའི་སྐུ་མདུན་དུ་འགྲོ་ནས་ཁྱེད་རང་ཚོ་འདི་ཡི་ཚེ་ལ་ཤ་དྲངས་ཀྱི་མིན། ཁྱེད་རང་ཚོ་ཡང་ཤ་མ་བཞེས་རོགས་ཞུ། ད་དུང་བཞེས་ན་མི་མང་ལ་སྐད་ཆ་ཤོད་ས་མི་འདུག གྲྭ་མང་ལ་ཡང་བཤད་ས་མི་འདུག་ཟེར་ནས་གསུང་རོགས་ཞུས་མ་གཏོགས། ཤེས་མཁན་ཡོན་ཏན་ཅན་ཚོས་ཤ་ཟ། གང་ཡང་མི་ཤེས་མཁན་མི་མང་ལུག་ཁྱི་འདྲ་བ་དེ་ཚོ་ལ་ཤ་མ་ཟ་ཟེར་ནས་བཀག་ན། དེ་རྩ་བ་ནས་ཡོང་ཐུབ་ཀྱི་མ་རེད་ཟེར་ནས་བཤད་རྒྱུ་བྱུང་བ་རེད། ཅེས་གསུངས་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
