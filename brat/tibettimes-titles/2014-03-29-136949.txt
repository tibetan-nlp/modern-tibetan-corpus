ཁྲག་ཡིག་རྩོམ་པ་པོ་ཀློད་གྲོལ།

དེ་རིང་གཟའ་སྤེན་པའི་ཉིན་རྒྱ་གཞུང་གིས་ཁྲག་ཡིག་རྩོམ་པ་པོ་བཀྲ་ཤིས་རབ་བརྟན་ནམ་ཐེའུ་རང་ལགས་བཙོན་གྲོལ་བཏང་ཡོད་འདུག ཕྱི་ལོ་ ༢༠༡༠ ཟླ་ ༤ ཚེས་ ༤ ཉིན་རྒྱ་གཞུང་གིས་༼ཁྲག་ཡིག༽ རྩོམ་པོ་བཀྲ་ཤིས་རབ་བརྟན་ནམ་ཐེའུ་རང་ལགས་ནུབ་བྱང་མི་རིགས་སློབ་ཆེན་དུ་འཛིན་བཟུང་གིས་ལོ་གཅིག་རིང་བཀག་ཉར་བྱས་པ་དང་། ༢༠༡༡ ལོའི་ཟླ་ ༦ ཚེས་ ༢ ཉིན་རྔ་བ་ཁུལ་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་གིས་ཁོང་ནི་མི་རིགས་ཁ་བྲལ་གྱི་ལས་འགུལ་སྤེལ་མཁན་དང་། ༢༠༠༨ ལོའི་ནུབ་བྱང་མི་རིགས་སློབ་ཆེན་བོད་རིགས་སློབ་མའི་ལས་འགུལ་གྱི་སྣེ་ཁྲིད་པ་སོགས་ཀྱི་ཉེས་མིང་འོག་ལོ་ ༤ བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་པ་ལྟར། དེ་རིང་ཕྱི་ཚེས་ ༢༩ ཉིན་གྱི་སྔ་དྲོ་ཁོང་ཟི་ཁྲོན་མེན་ཡཱང་ཙོན་ཁང་ནས་བཙོན་གྲོལ་ཐོབ་ཏེ་ཁྲིན་ཏུའུ་གྲོང་ཁྱེར་དུ་འབྱོར་ཡོད་འདུག་ལ།  སང་ཉིན་སྔ་དྲོ་མཛོད་དགེ་རྫོང་ཐང་སྐོར་ཞང་སོག་ཚང་དགོན་པའི་ཉེ་འདས་སུ་འབྱོར་རྒྱུ་ཡིན་འདུག  ན་གཞོན་རྩོམ་པ་པོ་བཀྲ་ཤིས་རབ་བརྟན་ནམ་ཐེའུ་རང་ལགས་ནི། ཟི་ཁྲོན་རྔ་བ་ཁུལ་མཛོད་དགེ་རྫོང་ཐང་སྐོར་ཞང་མགོ་སྐོར་སྡེ་བ་ནས་ཡིན་པ་དང་། དེ་སྔོན་ནུབ་བྱང་མི་རིགས་སློབ་གྲྭ་ཆེན་མོའི་ནང་ཟས་བཅད་ངོ་རྒོལ་གྱི་ལས་འགུལ་སྤེལ་བ་དང་། ཤར་དུང་རི་དུས་དེབ་ཀྱི་རྩོམ་སྒྲིག་པ་གནང་བ་བཅས་ལ་བརྟེན་ནས་རྒྱ་གཞུང་གིས་སྔ་ཕྱི་འདྲི་རྩད་བཀག་ཉར་བྱས་ཡོད་ལ། ༢༠༠༩ ལོར་ཁོང་གིས་བརྩམས་ཆོས་༼ཁྲག་ཡིག༽ སྤྱི་ཚོགས་ནང་འགྲེམས་སྤེལ་བྱས་རྗེས་འདྲི་བརྩད་བཀག་ཉར་བྱས་ཡོད་པ་མ་ཟད། ༢༠༡༡ ཟླ་ ༤ ཚེས་ ༤ ཉིན་ནུབ་བྱང་མི་རིགས་སློབ་ཆེན་ནས་འཛིན་བཟུང་གིས་ལོ་ ༡ རིང་བཀག་ཉར་བྱས་པ་དང་། ༢༠༡༡ ལོར་མི་རིགས་ཁ་བྲལ་གཙོས་པའི་ཉེས་མིང་ཁག་ཅིག་འོག་ཁྲིམས་ཐག་བཅད་ཡོད། ཁོང་གིས་བརྩམས་པའི་༼ ཁྲག་ཡིག ༽དཔེ་དེབ་ནང་བསྡོམས་པའི་ཚན་པ་ལྔ་དང་རྩོམ་ཡིག་སོ་གཉིས་ཡོད་པ་དག་ནི། ཚན་པ་དང་པོ། དམྱལ་ཁམས་ཀྱི་ཟིན་ཐོ་ཞེས་པའི་ནང༌། མྱལ་ཁམས་ཀྱི་ཆིག་ལབ། དམྱལ་ཁམས་ཀྱི་བཙོན་མ། དམྱལ་ཁམས་ཀྱི་གསར་འགྱུར། དམྱལ་ཁམས་ཀྱི་ཡར་ལངས། དམྱལ་ཁམས་ཀྱི་ཚོགས་འདུ། བཅས་རྩོམ་ཡིག་ ༥ ཡོད་པ་དང༌། ཚན་ པ་གཉིས་པ། རྣམ་ཤེས་ཀྱི་དབྱངས་ཞེས་པའི་ནང༌། གློ་བུར་དུ་ལྷ་ས་དྲན་པ། སྤྱི་ཟླ་བཅུ་གཅིག་པའི་སྙན་ངག་འདིའང༌། གློ་བུར་བའི་སྙན་ངག་ཚན་པ་གསུམ། ང་རང་རྨང་ནས་སྒུལ་བྱུང༌། རྣམ་ཤེས་ཀྱི་དབྱངས། མིག་ཆུའི་ཚབ་དུ་ཐོལ་བའི་སྙན་ངག རྣམ་ཤེས་ཁྲི་ཕྲག་གི་དོད་དུ། བཅས་རྩོམ་ཡིག་ ༧ ཡོད། ཚན་པ་གསུམ་པ། ང་ཡི་བོད་ཅེས་པའི་ནང༌། ཀྭ་ཡེ་བོད། ང་ཡིས་ཁྱོད་ལ་འདྲི། གཙོས་གྲོང་དུ་སླེབས་པའི་སེམས་ཚོར་ཞིག མདོ་དབུས་ལྕགས་ལམ། ང་ཡི་བོད། བཙན་པོའི་རལ་གྲི་ཐོགས་དང༌། འགྲུལ་བཞུད་ཐེངས་གཅིག་གི་སྐྱོ་ངལ། དམར་ནག་གི་ཟླ་གསུམ་པ། བཅས་རྩོམ་ཡིག་ ༨ བཀོད་ཡོད། ཚན་པ་བཞི་བ། སེམས་ཀྱི་དམར་མཆོད་ཅེས་པའི་ནང༌། སེམས་པ་འཁྱག་པས་བསྡམས་པའི་དུས་ཚིགས་སུ། གློག་ཀླད་མདུན་དུ་སྲིད་གཞུང་ལ་ཕུལ་བའི་གཏམ་ལྐོག་གྱུར་མ། གྲོགས་པོའི་སྐྱེ་སྐར་ལས་སྤུན་ཟླའི་འཆི་སྐར་དྲན་པ། གྲོགས་ཁ་ཤས་ལ་ཕུལ་བ། གློག་ཀླད་དྲ་བར་སྡིགས་དམོད་གཏོང་བ། ཐེའུ་རང༌། ཁྱོད་ཀྱི་ལོ་གསར་གང་ན་ཡོད་བཅས། རྩོམ་ཡིག་ ༦ བཀོད་ཡོད། ཚན་པ་ལྔ་བ། བདེན་པའི་གཏུག་བཤེར་ཞེས་པའི་ནང༌། ཁྲག་དྲི་བྲོ་བའི་དོན་རྐྱེན་དང་དེའི་ལྟག་རྒྱབ་ཀྱི་གསང་བ། འུ་ཚོས་ཀྱང་རང་ས་ནས་བསྒྲགས་གཏམ་སྤེལ་ཆོག ཟིན་ཐོ་ཁར་སྒུམ་ཁེར་སྒུམ། མྱ་ངན་དང་འགྱོད་གཤགས། ད་དུང་རྒོལ་གཏམ་ཁག་གཉིས། སྙིང་རླུང་ལྡང་བའི་གསར་འགྱུར། སེམས་ཀྱི་དམར་དམར་པོ་བཤད་ན། བཅས་འབོལ་རྩོམ་ཁག་ ༦ བཅས་བསྡོམས་པའི་རྩོམ་ཡིག་ ༣༢ བཀོད་ཡོད་འདུག དཔེ་དེབ་དེའི་ནང་རྩོམ་པ་པོ་ཁོང་རང་ཉིད་ཀྱིས་གསལ་བཤད་ཅིག་བྱས་པའི་ནང་བཀོད་དོན། དེབ་ཆུང་འདི་ནི་ཁོ་བོ་རང་ཉིད་ཀྱི་ན་ཚོད་དང༌། ཡོན་ཚད་ཀྱི་ངོས་ནས་བཤད་ན་འཕུད་རྒྱུ་སྔ་དྲགས་པའི་ལས་ཀ་ཞིག་ཡིན་སྲིད་མོད། ནམ་རྒྱུན་བླ་སྲོག་གི་ཕྱི་ནང་ཡོད་ཚད་ལ་ལྕག་ཚན་གྱིས་བྲབ་པའི་དམིགས་བསལ་བའི་ལོ་ཟླ་ཞིག་གི་ངོས་ནས་བཤད་ན། ཅིས་ཀྱང་མི་བཤད་ཐུ་མེད་དུ་གྱུར་བའི་སྐད་ཆ་འབའ་ཞིག་ཏུ་ངེས་པས་ན། ཁོ་བོ་ནས་ཀྱང་སྨྲ་སྒོ་རང་ས་ནས་བསྡམས་ཏེ་འདུག་པ་ལས་སྨྲ་བ་གྲོལ་ཏེ་སུ་ཞིག་ བཅད་པ་ཡིན་ན་རིན་དོད་ཆེ་བར་འདོད་པའི་གདམ་ག་ཞིག་ལ་འཕྲེང་ཏེ། ད་གཟོད་རང་གི་ན་ཚོད་ལས་ཀྱང་གཉོམ་ལག་མ་རྒྱས་པའི་དེབ་ཆུང་འདི་ཆེད་དུ་སྟོང་ལོ་བརྒྱད་དགུ་གཉིས་ཀྱི་སོ་མཚམས་སུ་ཁྲག་གི་ཐིགས་པ་བཞིན་གཏར་བ་ལགས་སོ། ཞེས་བཀོད་ཡོད། གསར་འགོད་པ། དོན་གོ
