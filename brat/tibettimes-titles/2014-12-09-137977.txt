བོད་རིཊ་མགོ་ཁྲིད་ཅིག་ལ་ལོ་ ༡༧
ཁྲིམས་ཆད།

ཉེ་སྔོན་བོད་རང་སྐྱོང་ལྗོངས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གིས་བོད་རིགས་མགོ་ཁྲིད་ཁག་ཅིག་གིས་ལྐོག་ཟ་བྱས་པའི་ནག་ཉེས་ཕྱིར་བསྒྲགས་བྱས་འདུག རྒྱ་ནག་ཤིན་ཧྭ་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། བོད་རང་སྐྱོང་ལྗོངས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གིས་ཟླ་འདིའི་ཕྱི་ཚེས་ ༥ ཉིན་བོད་རིགས་མགོ་ཁྲིད་ཁག་ཅིག་གིས་དབང་ཆ་སྒེར་སྤྱོད་ཀྱིས། གཞུང་དངུལ་ལྐོག་ཟ་དང་ལོག་པར་སྤྱད་པའི་གནས་ཚུལ་ཞིག་ཕྱིར་བསྒྲགས་བྱས་པའི་ནང་། མངའ་རིས་ས་ཁུལ་སྒེར་རྩེ་རྫོང་མ་མིག་ཤང་རྒོད་ཚང་གྲོང་ཚོའི་མགོ་ཁྲིད་བཀྲ་ཤིས་འཕྲིན་ལས་ཟེར་བ་ཞིག་གིས། ཕྱི་ལོ་ ༢༠༠༩ ནས་ ༢༠༡༤ ལོའི་ཟླ་དང་པོའི་བར་དབང་ཆ་སྒེར་སྤྱོད་བྱས་ཏེ། སྤྱི་དངུལ་རྒྱ་སྒོར་ཁྲི་ ༡༢༤ ལྷག་ལོག་སྤྱོད་ལྐོག་ཟ་བྱས་ཡོད་སྟབས། ཁོ་བོ་ཏང་ནང་ནས་ཕྱིར་འབུད་བཏང་སྟེ། སྒེར་རྩེ་རྫོང་མི་མང་ཁྲིམས་ཁང་གིས་ལོ་ ༡༧ གྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག ཡང་ནག་ཆུ་རྫོང་སྟག་གསར་ཤང་ཏང་གི་མགོ་ཁྲིད་ཟུར་པ་པ་མཚམས་ཟེར་བ་དེས། སྟག་གསར་ཤང་བདེ་སྡོད་ལས་གྲྭའི་སྤྱིའི་དངུལ་ཁྲི་ ༡༠༨།༦༣ གཞུང་དོན་འགྲོ་གྲོན་དུ་བཏང་བ་དང་། ལས་གྲྭ་དེའི་སྤྱི་དངུལ་དང་ཕྱུགས་ཟོག་བཤས་གཏོང་རོགས་དངུལ་སྒོར་ཁྲི་ ༩༦ ལྷག་སྒེར་གྱིས་བེད་སྤྱོད་བཏང་བ་དང་། དེ་སྔོན་ཁོ་བོས་ནག་ཆུ་རྫོང་ཁོར་ཡུག་སྲུང་སྐྱོང་ཁང་གི་མགོ་ཁྲིད་ཀྱི་འགན་འཁུར་སྐབས། ཁོར་ཡུག་གཙང་བཟོ་དང་འབྲེལ་བའི་འགྲོ་གྲོན་རྒྱ་སྒོར་ཁྲི་ ༩ བཅས་སྒེར་སྤྱོད་བྱས་ཡོད་སྟབས། ཁོ་བོ་ཏང་ནས་ཕྱིར་འབུད་བཏང་སྟེ་ཁྲིམས་འཛིན་ལས་ཁུངས་ལ་ཁྲིམས་ལྟར་ཐག་གཅོད་བྱེད་པར་རྩིས་སྤྲད་ཡོད་སྐོར་བརྗོད་འདུག གཞན་ཡང་། ཉིང་ཁྲི་ས་ཁུལ་ཏང་གི་སློབ་གྲྭའི་ཚན་རིག་ཞིབ་འཇུག་སྡེ་ཚན་གྱི་མགོ་ཁྲིད་ཟུར་པ་ལུའེ་ཅིན་ཧའེ་ཟེར་བ་ཞིག་གིས། ས་གནས་ཀྱི་རྨང་བརྟན་དམངས་ཕན་གྱི་མ་དངུལ་ལྐོག་ཟ་བྱས་པར་བརྟེན་ལས་གནས་ནས་གནས་དབྱུང་བཏང་ཞིང་། ཏང་ནང་ནས་ལྟ་ཞིབ་ལོ་གཅིག་བྱ་རྒྱུ་དང་། ཡང་མེ་ཏོག་རྫོང་འབྲས་སྤུངས་ཤང་གི་རྒྱུ་ནོར་དང་འབྲུ་རིགས་ཉི་མ་དོན་གྲུབ་དང་། རིན་སྤུངས་རྫོང་འཕྲོད་བསྟེན་ཁང་གི་ལས་བྱེད་རྡོ་རྗེ་ཐར་ཟེར་བ་གཉིས་ཀྱིས་ལྐོག་ཟ་བྱས་པའི་དོགས་ཡོད་ཀྱི་ཉེས་དོན་ཞིག་ལ་མུ་མཐུད་ཞིབ་བཤེར་བྱེད་བཞིན་ཡོད་ལུགས་བསྒྲགས་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
