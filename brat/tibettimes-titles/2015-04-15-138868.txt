གྲྭ་འཇིཊ་མེད་རྒྱ་མཚོར་ཉེས་འཛུཊ།

ཀན་སུའུ་ཞིང་ཆེན་ཅོ་ནེ་རྫོང་བདེ་ཐང་དགོན་གྱི་དབུ་མཛད་གྲྭ་འཇིགས་མེད་རྒྱ་མཚོ་ལགས་གནས་སྐབས་རིང་ཀློད་གྲོལ་བཏང་ཡོད་ཀྱང་མ་འོངས་པར་ཁྲིམས་བཤེར་བྱ་རྒྱུ་ཡིན་འདུག འདི་གར་འབྲེལ་ཡོད་ཅིག་གིས་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། འདི་ལོའི་གསུམ་བཅུའི་དུས་དྲན་སྐབས་རྒྱལ་ཁབ་ཁ་བྲལ་གཏོང་ཐབས་ཀྱི་བྱ་སྤྱོད་སྤེལ་བའི་ཉེས་མིང་འོག ཀན་སུའུ་ཞིང་ཆེན་ཅོ་ནེ་རྫོང་བདེ་ཐང་དགོན་གྱི་དབུ་མཛད་གྲྭ་འཇིགས་མེད་རྒྱ་མཚོ་ལགས་རྒྱ་གཞུང་ཉེན་རྟོག་པས་བཙན་འཁྲིད་བྱས་ཡོད་པ་དང་། འདི་ལོའི་ཕྱི་ཟླ་ ༣ ཚེས་ ༢༠ ཉིན་ཡཱང་ཏུང་ཧྲིན་ཟེར་བའི་དཔང་མི་ཞིག་གི་འགན་ལེན་འོག་ཀློད་གྲོལ་བཏང་ཡོད་ཀྱང་། ཉེན་རྟོག་ཁང་གིས་ཁོང་གི་དབུ་མཛད་ཀྱི་ལས་ཡུན་རྫོགས་རྗེས་ཁྲིམས་བཤེར་བྱ་རྒྱུ་ཡིན་པའི་ཡི་གེ་སྤྲད་ཡོད་འདུག གྲྭ་འཇིགས་མེད་རྒྱ་མཚོ་ལགས་ཕྱི་ལོ་ ༡༩༧༦ ཟླ་ ༣ ཚེས་ ༢༣ ཉིན་ཅོ་ནེ་རྫོང་ཁོངས་སུ་སྐྱེས་ཤིང་། ཕྱི་ལོ་ ༢༠༠༩ ལོར་པེ་ཅིན་ནང་བསྟན་སློབ་གླིང་ནས་མཐར་ཕྱིན་ཡོད་པ་དང་། ༢༠༡༠ ལོར་བདེ་ཐང་དགོན་པའི་དབུ་མཛད་དུ་བསྐོ་བཞག་གནང་སྟེ་ལོ་ ༦ རིང་གི་ལས་འཁུར་ཞུ་བཞིན་ཡོད་པ་དང་།  འདི་ལོའི་བོད་ཟླ་དང་པོའི་ཚེས་ ༡༦ ཉིན་(ཕྱི་ཟླ་ ༣ ཚེས་ ༦ ཙམ་ལ)གྲྭ་ཚང་གི་དབུ་ཁྲིད་དེ་བླ་བྲང་བཀྲ་ཤིས་འཁྱིལ་ཏུ་རྩ་བའི་བླ་མ་བདེ་ཐང་མཁན་རིན་པོ་ཆེ་མཆོག་གི་ཞབས་བརྟན་སྒྲུབ་ཏུ་ཕེབས་ཡོད་ཀྱང་།  བོད་ཟླ་དང་པོའི་ཚེས་ ༡༩ ཉིན་གྱི་མཚན་མོར་ཁོང་གར་སོང་ཆ་མེད་དུ་གྱུར་ཡོད་པ་དང་། ཉིན་ཞག་ ༡༡ འགོར་རྗེས་ད་གཟོད་ཅོ་ནེ་རྫོང་གི་བཙོན་ཁང་དུ་ཡོད་པ་ཤེས་རྟོགས་བྱུང་ཡོད་འདུག ཡང་གནས་ཚུལ་མཁོ་སྤྲོད་པས་བརྗོད་པ་ལྟར་ན། གྲྭ་འཇིགས་མེད་རྒྱ་མཚོ་ལགས་ཀྱིས་རྒྱ་ནག་གཞུང་གི་དམ་དྲག་སྲིད་བྱུས་ལ་སྐྱོན་བརྗོད་ཐེངས་མང་གནང་མྱོང་བ་མ་ཟད། ཕྱི་ལོ་ ༢༠༠༨ ལོར་བོད་ཁམས་ཡོངས་ལ་དམག་མི་འབོར་ཆེན་འགྲེམས་འཇོག་བྱེད་སྐབས། དམག་མི་ཚོས་ས་གནས་མང་ཚོགས་ཀྱིས་ལྷ་མཆོད་བསང་གསོལ་གཏོང་སའི་མཚོ་ཁ་ནས་ཉ་བཟུང་སྟེ། ས་གནས་མི་མང་གི་སེམས་ལ་ཕོག་ཐུག་ཆེན་པོ་བཟོས་པར་བརྟེན། ཁོང་གིས་རྒྱལ་ཁབ་ཀྱིས་གཏན་འབེབས་བྱས་པའི་མི་རིགས་སྲིད་བྱུས་དང་ཁོར་ཡུག་སྲུང་སྐྱོབ་ཀྱི་རྩ་ཁྲིམས་གཞིར་བཟུང་གནས་ཚུལ་དེ་གོང་རིམ་ལ་ཞུ་གཏུག་གནང་ཡོད་འདུག ཕྱི་ལོ་ ༢༠༡༢ ལོར་ཁོང་ལྷ་སར་ཕྱིན་སྐབས་ཉེན་རྟོག་པས་འཛིན་བཟུང་བྱས་ཏེ། སྟག་རྩེ་རྫོང་བཙོན་ཁང་དུ་བཀག་ཉར་གྱིས་ཟ་མ་དང་གྱོན་གོས་མ་སྤྲད་པར་ཞག་མ་ ༣ རིང་མནར་གཅོད་བཏང་ཡོད་ཀྱང་། ཁོང་གིས་གཞི་མེད་ཀྱི་ཉེས་འཛུགས་དང་ལེན་མ་བྱས་པར་བརྟེན་མཐར་ཀློད་གྲོལ་གཏོང་དགོས་ཁེལ་ཡོད་འདུག གཞན་ཡང་། ཕྱི་ལོ་ ༢༠༡༠ དང་ ༢༠༡༡ ལོར་རྒྱ་གཞུང་གིས་ས་གནས་ཀྱི་དམའ་འབྲིང་སློབ་གྲྭའི་ནང་བོད་ཡིག་གི་བསླབ་གཞི་རྩ་བ་ནས་མེད་པར་བཟོས་སྟབས། གྲྭ་འཇིགས་མེད་རྒྱ་མཚོ་ལགས་ཀྱིས་ཡིག་ངག་གཉིས་ཐོག་ནས་གོང་རིམ་ལ་ཡང་ཡང་ཞུ་གཏུག་བྱས་པ་མ་ཟད། ས་གནས་མི་མང་ལ་སོ་སོའི་སྐད་ཡིག་གི་དགོས་གལ་རང་བཞིན་དྲན་སྐུལ་སོགས་གནང་སྟེ་འབད་བརྩོན་གནང་བརྒྱུད། འདི་ལོའི་སློབ་སྐབས་དང་བོ་ནས་བཟུང་ས་གནས་དམའ་འབྲིང་ནང་བོད་ཡིག་འཛིན་གྲྭ་ཞིག་གསར་འཛུགས་གནང་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
