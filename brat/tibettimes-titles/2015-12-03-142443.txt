གྲྭ་བ་གཉིས་ལ་ཁྲིམས་ཐག་བཅད་འདུག

  གསར་དུ་ཤེས་རྟོགས་བྱུང་བར་རྒྱ་ནག་གཞུང་གིས་རྔ་པ་ཀིརྟི་དགོན་པའི་གྲྭ་དགེ་འདུན་ཕུན་ཚོགས་ལ་ལོ་ ༤ དང་གྲྭ་བློ་བཟང་སྐལ་བཟང་ལ་ལོ་ ༣ དང་ཕྱེད་བཅས་ཀྱི་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག   འདི་ལོའི་ཕྱི་ཟླ་ ༡༠ ཟླ་མཇུག་ཙམ་ལ་རྔ་པ་ཀིརྟི་དགོན་པའི་གྲྭ་དགེ་འདུན་ཕུན་ཚོགས། རང་ལོ་ ༡༨ ཡིན་པ་ཁོང་ལ་རྔ་པ་ཁུལ་ཁོངས་ཀྱི་ལས་རྫོང་ནས་ལོ་ ༤ བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་པ་དང་། མི་འགྱང་བར་སི་ཁྲན་ཕྱོགས་ཀྱི་ཆབ་སྲིད་བཙོན་པ་མང་ཆེ་བ་བཙོན་འཇུག་བྱས་ཡོད་པའི་མེན་ཡང་བཙོན་ཁང་དུ་བསྐྱལ་རྒྱུ་ཡིན་འདུག   དགེ་འདུན་ཕུན་ལགས་ནི་འདི་ལོའི་ཟླ་ ༣ ཚེས་ ༨ ཉིན་ ༸གོང་ས་མཆོག་གི་སྐུ་པར་ཆེན་པོ་ཞིག་ཁ་བཏགས་སེར་པོ་ཞིག་གིས་ལྕིབས་ཏེ་མགོ་ཐོག་ཏུ་འཕྱར་ནས་རྒྱལ་བ་རིན་པོ་ཆེ་ “བོད་ལ་ཕེབས་འཇུག་དགོས། བོད་ལ་རང་དབང་དང་འདྲ་མཉམ་དགོས།” ཞེས་སྐད་འབོད་བྱེད་བཞིན་རྔ་པ་རྫོང་མཁར་གྱི་གཞུང་ལམ་སྟེང་གོམ་བགྲོད་གནང་མཁན་དེ་ཡིན་ཞིང་། ཉིན་དེར་འཛིན་བཟུང་བཙན་ཁྲིད་བྱས་པ་ནས་ད་བར་གར་ཡོད་ཅི་བྱུང་གི་གནས་ཚུལ་གཅིག་ཀྱང་ཤེས་རྟོགས་མ་བྱུང་བར་ལྷག་ཡོད་འདུག་ལ། ཁོང་ནི་རྔ་བ་རྫོང་གཅའ་ཡུལ་ཚོའི་ཕྱུགས་ལས་འགབ་མའི་ཕ་བཀྲ་རྒྱ་དང་། མ་རིག་གོ་གཉིས་ཀྱི་བུ་ཡིན་པ་དང་། ཆུང་ཆུང་སྐབས་ནས་རྔ་བ་ཀིརྟི་དགོན་པར་ཆོས་ཞུགས་བྱས་ཏེ་ལོ་སྔོན་མའི་ནང་སྔོན་འགྲོའི་འཛིན་གྲྭ་ཁག་མཐར་ཕྱིན་ཡོད་འདུག   གཞན། འདི་ལོའི་ཕྱི་ཟླ་ ༡༡ ཚེས་ ༢ ཉིན་རྔ་པ་ཀིརྟི་དགོན་པའི་གྲྭ་བློ་བཟང་སྐལ་བཟང་། རང་ལོ་ ༡༩ ཡིན་པ་ཁོང་ལ་ལོ་ ༣ དང་ཟླ་ ༦ གི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་ཀྱང་ཁྲིམས་ཐག་གཅོད་ས་རྔ་པ་ཁུལ་ཁོངས་ཀྱི་ལས་རྫོང་ཡིན་པ་དང་མི་འགྱང་བར་མེན་ཡང་བཙོན་ཁང་དུ་བསྐྱེལ་རྒྱུ་ཡིན་འདུག   བློ་བཟང་སྐལ་བཟང་ལགས་ནི་འདི་ལོའི་ཟླ་ ༣ ཚེས་ ༡༧ ཉིན་རྔ་པ་རྫོང་མཁར་གྱི་གཞུང་ལམ་སྟེང་དུ་ ལག་པ་གཡོན་པས་༸གོང་ས་མཆོག་གི་སྐུ་པར་ཞིག་མཐོན་པོར་བཏེགས་ཤིང་། ལག་པ་གཡས་པས་རླུང་རྟ་གཏོར་བཞིན་ “བོད་ལ་རང་དབང་དགོས།” ཞེས་སྐད་འབོད་གནང་མཁན་དེ་ཡིན་ཞིང་། ཉིན་དེར་འཛིན་བཟུང་བཙན་ཁྲིད་བྱས་པ་ནས་ད་བར་འབྲེལ་ཡོད་རྣམས་ཀྱིས་ཀྱང་གར་ཡོད་ཅི་བྱུང་གི་གནས་ཚུལ་ཤེས་རྟོགས་མ་ཐུབ་པར་ལུས་ཡོད་འདུག ཁོང་གི་ཕ་ཡུལ་གཅའ་རུ་བ་ཡུལ་ཚོའི་ཕྱུགས་ལས་འགབ་མ་ཡིན་ཞིང་། ཕ་ཚེ་རིང་དང་། མ་བདེ་ཆེན་གཉིས་ལ་བུ་བུ་མོ་གསུམ་པའི་གཉིས་པ་དེ་ཡིན་འདུག   གོང་གསལ་གྱི་དགེ་འདུན་ཕུན་ཚོགས་དང་ཁོང་གཉིས་ཕ་མ་མིང་སྲིང་གི་སྤུན་ཡིན་པ་དང་། ཆུང་ཆུང་སྐབས་ནས་རྔ་པ་ཀིརྟི་དགོན་པར་ཆོས་ཞུགས་བྱས་ཏེ་དེང་སྐབས་འཛིན་གྲྭ་ཁ་ཏོག་གོང་མ་ཡིན་འདུག ཅེས་ཀིརྟིའི་བྱེས་པ་གྲྭ་ཚང་གི་བོད་གནས་འབྲེལ་མཐུད་པ་བློ་བཟང་ཡེ་ཤེས་དང་ཀ་ཉག་ཚེ་རིང་གཉིས་བཏང་འབྱོར།
