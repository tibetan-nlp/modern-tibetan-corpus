
དབྱིན་ཇིའི་ཚིག་མཛོད་རྩོམ་སྒྲིག་པས་ལོ་བརྒྱར་ལས་ཀ་བྱས་པ་དེ་ང་ཚོས་ལོ་དགུ་ལ་འགྲུབ་དགོས།

ཁ་སང་དེ་རིང་ཉིན་གཉིས་རིང་སྨོན་ལམ་བོད་ཀྱི་ཀུན་བཏུས་ཚིག་མཛོད་ཆེན་མོའི་ཆོས་བརྒྱུད་ཁག་གི་རྩོམ་སྒྲིག་པའི་ཚོགས་འདུ་ཐེངས་ ༣ པ་གངས་སྐྱིད་དུ་ཚོགས་ཏེ་འདས་པའི་ཉམས་མྱོང་དང་མ་འོངས་པའི་ལས་འཆར་བཅས་ཀྱི་སྐོར་ལ་བགྲོ་གླེང་གནང་སོང་། ཁ་སང་དེ་རིང་ཉིན་གཉིས་རིང་། གངས་སྐྱིད་བོད་ཀྱི་དཔེ་མཛོད་ཚོགས་ཁང་དུ། སྨོན་ལམ་བོད་ཀྱི་བརྡ་འཕྲིན་ཞིབ་འཇུག་ཁང་གིས་གོ་སྒྲིག་འོག ༼སྨོན་ལམ་བོད་ཀྱི་ཀུན་བཏུས་ཚིག་མཛོད་ཆེན་མོ།༽ཡི་ཆོས་ལུགས་དང་འབྲེལ་བའི་ཐ་སྙད་རྩོམ་སྒྲིག་བྱེད་ཕྱོགས་སྐོར་ལ་ཆོས་བརྒྱུད་ཁག་གི་རྩོམ་སྒྲིག་པ་འདུ་འཛོམས་ཀྱི་བགྲོ་གླེང་ཐེངས་ ༣ པ་དེ་ཚོགས་ཡོད། ཉིན་གཉིས་ཀྱི་བགྲོ་གླེང་སྟེང་ས་དགེ་བཀའ་རྙིང་བོན་དང་ཇོ་ནང་བཅས་པའི་ཆོས་ལུགས་ཆོས་བརྒྱུད་ཁག་གི་ཐ་སྙད་བསྡུ་རོབ་གནང་མཁན་རྩོམ་སྒྲིག་པ་ ༡༧ ཙམ་ཚོགས་ཞུགས་གནང་ཡོད་པ་དང་། ཚོགས་ཚོས་འདས་པའི་ཉམས་མྱོང་དང་དཀའ་ངལ། མ་འོངས་པར་ཡར་རྒྱས་གཏོང་ཕྱོགས་བཅས་ཀྱི་ཐོག་སོ་སོའི་དགོངས་འཆར་གསུངས་རྗེས་དེ་ལ་ཞིབ་འཇུག་ཁང་གི་འབྲེལ་ཡོད་ལས་བྱེད་ཚོས་ལམ་སྟོན་འཆར་འགོད་བཅས་གནང་ཡོད་པ་རེད། བགྲོ་གླེང་དབུ་འཛུགས་ཐོགསྨོན་ལམ་བརྡ་འཕྲིན་ཞིབ་འཇུག་ཁང་གི་འགན་འཛིན་གྲྭ་བློ་བཟང་སྨོན་ལམ་ལགས་ཀྱིས་བགྲོ་གླེང་ངོ་སྤྲོད་གནང་དོན། ཆོས་བརྒྱུད་གྲུབ་མཐའ་ཚང་མས་མཉམ་གཅིག་ཏུ་རླབས་ཆེན་གྱི་ལས་ཀ་ཞིག་མགོ་བཙུགས་ཐུབ་ཡོད་པ་འདིར། ཕྱོགས་གཅིག་ནས་བལྟས་ན་ང་ཚོ་ཚང་མར་སྐལ་བ་བཟང་པོ་ཐོབ་པ་རེད་དྲན་གྱི་འདུག དཔེར་ན། ཨོག་སི་ཧྥོར་ཌི་(Oxford)ཡི་ཚིག་མཛོད་ལྟ་བུར་མཚོན་ན། དབྱིན་ཡིག་གི་ཁུངས་གཏུག་ས་ཞིག་ཆགས་ཡོད་པ་རེད། ཨོག་སི་ཧྥོར་ཌི་གཙུག་ལག་སློབ་གྲྭའི་ནང་དུ་རྩོམ་སྒྲིག་སྡེ་ཚན་བཙུགས་ནས་ལོ་ ༡༢༧ ཕྱིན་ཡོད། དེ་འདྲའི་ལོ་མང་ཞིག་ལ་ཁོ་ཚོས་དཀའ་ལས་བརྒྱབ་སྟེ་ཚིག་མཛོད་བསྒྲིགས་པ་དེར་དགོས་པ་རྒྱུ་མཚན་མང་པོ་ཡོད། ང་ཚོ་བཙན་བྱོལ་དུ་ལོ་བརྒྱ་ཟེར་ན་དཀའ་ལས་ཁག་པོ་རེད་དེ་ཆོས་བརྒྱུད་ཁག་གི་རོགས་འདེགས་འོག ད་ཆར་ང་ཚོའི་ཚིག་མཛོད་འདིར་ལས་ཀ་བྱེད་མཁན་བརྒྱ་ལྷག་ཡོད་ལ།བོད་ནང་དུའང་རྩོམ་སྒྲིག་སྡེ་ཚན་གཉིས་ཡོད་ཀྱང་ཁོ་ཚོ་དེ་རིང་འདིར་ཕེབས་ཐུབ་ཡོད་པ་མ་རེད། ཨོག་སི་ཧྥོར་ཌི་ཡིས་ལོ་བརྒྱ་ལྷག་ལ་ལས་ཀ་བྱས་པ་དེ་ང་ཚོའི་མི་ ༡༡༥ ཡིས་འབད་བརྩོན་བྱས་ན་ཕལ་ཆེར་ལོ་ ༩ ཡི་ནང་འགྲུབ་ཐུབ་པ་ཞིག་རེད། ཅེས་བརྗོད་སོང་། ཁོང་གིས་མུ་མཐུད་དུ་ཚིག་མཛོད་བཟོ་དགོས་པའི་རྒྱུ་མཚན་སྐོར་བརྗོད་དོན། དེ་རིང་འདིར་ང་ཚོས་ཚིག་མཛོད་སྒྲིག་པའི་ཕན་ཡོན་དོན་ཚན་གསུམ་གྱི་སྒང་ནས་ཞུ་རྒྱུ་ཡིན། དང་པོ།དེ་རིང་འདིར་ཕེབས་མཁན་ཚང་མ་ཆོས་པ་རྐྱང་རྐྱང་རེད། སྤྱི་ཚོགས་འགྱུར་བ་འགྲོ་བཞིན་ཡོད་པ་དང་། དེང་གི་སྤྱི་ཚོགས་རིག་གནས་ཐད་ལ་གྲྭ་བ་ལས་མི་སྐྱ་དེ་ཚོ་ཤེས་ཡོན་ཡག་པོ་ཆགས་ཡོད། མང་ཚོགས་ཀྱི་དགོས་མཁོ་ང་ཚོས་སྐུང་མ་ཐུབ་ན་མང་ཚོགས་སྐད་ཡིག་གཞན་པའི་སྟེང་མགོ་འཁོར་འགྲོ་བ་རེད། དེ་ལ་ང་ཚོས་མཐུན་རྐྱེན་ཞིག་བསྐྲུན་མ་ཐུབ་ན། མི་སྐྱ་ཚོར་གྲྭ་བ་ལྟར་པོ་ཏི་མང་པོར་སློབ་སྦྱོང་བྱེད་པའི་དུས་ཚོད་དང་མཐུན་རྐྱེན་ཡོད་པ་མ་རེད། གཉིས་པ།ཡིག་སྒྱུར་བྱེད་མཁན་མང་པོ་ཞིག་སྐད་ཡིག་གཉིས་ལ་རང་མགོ་ཐོན་ཙམ་ལས། ཆོས་ལུགས་དང་ལྟ་གྲུབ་ཀྱི་མཁས་པ་ཆེན་པོ་དཀོན་པོ་མིན་སྟབས། ཁོ་ཚོ་མང་ཆེ་བས་སྐྱབས་བཅོལ་ས་འདྲ་བོ་དེ་ཧ་ལམ་ཚིག་མཛོད་ཆགས་འདུག ལྷག་པར་དུ་ཆོས་ལུགས་དང་བོད་བརྒྱུད་ནང་བསྟན་ལ་ཆ་བཞག་ན། ཡིག་སྒྱུར་བ་ཚོས་གང་བྱུང་མང་བྱུང་བསྒྱུར་བཞག་པ་དེ་ཉེན་ཁ་ཞིག་རེད། འཕྲུལ་ལས་ཀྱི་བརྒྱུད་རིམ་ནང་དུ་བོད་ཀྱི་རིག་གཞུང་ཆ་ཚང་པ་ཞིག་དགོས་ཀྱི་ཡོད་ན། ཐ་སྙད་ཆ་ཚང་བ་ཞིག་དགོས་པ་དེ་མ་རྩ་རེད། དེ་དག་གི་འགྲེལ་བཤད་མ་ནོར་བ་ཞིག་དགོས་པ་དེ་ལས་ཀྱང་གལ་ཆེན་པོ་རེད། གསུམ་པ། ང་ཚོས་ ༡༩༥༠སྔོན་ལ་ཡིག་ཐོག་ཏུ་བཀོད་ཡོད་པའི་ཐ་སྙད་དེ་དག་གང་ཐུབ་ཀྱིས་གཙང་མ་བཟོ་ཐུབ་ན་བསམ་ཡོད། དེ་ང་ཚོའི་མི་ཐོག་འདིས་བསྒྲུབ་དགོས་པའི་ལས་འགན་གལ་ཆེན་ཞིག་རེད། ལྷག་པར་དུ་ཆོས་ལུགས་དང་ལྟ་གྲུབ་ཀྱི་ཐ་སྙད་རྙིང་པ་དེ་དག་ད་ལྟ་ང་ཚོའི་མི་ཐོག་འདིས་འགྲེལ་བཤད་རྒྱག་མ་ཐུབ་པ་ཡིན་ན། རིམ་གྱིས་ཇེ་སྡུག་ཏུ་འགྲོ་ངེས་རེད། ༡༩༥༠ ལོའི་རྗེས་ཀྱི་ཐ་སྙད་དག་ང་ཚོས་བཟོ་མ་ཐུབ་ཀྱང་མི་རབས་རྗེས་མ་ཚོས་བཟོ་ཐུབ་པ་ཞིག་རེད། ད་ལྟའི་སྐབས་འདིར་རྣམ་པ་ཚོ་དང་ང་ཚོས་མ་ཤེས་པ་དེ་འདྲ་བྱུང་ན་སྐད་ཆ་འདྲི་སའང་ཡོད་པ་རེད། ང་ཚོའི་མི་ཐོག་འདི་བོད་ཀྱི་ཆོས་ལུགས་རིག་གཞུང་གང་ཐད་ནས་གལ་ཆེན་པོ་ཆགས་འདུགང་ཚོའི་ལས་ཀ་འདིས་སྤྱི་ཚོགས་ལ་ཕན་ཐོགས་དགོས་པ་རེད། ཕན་ཐོགས་ས་དེའང་ང་ཚོ་བོད་པ་གཅིག་པུ་མ་རེད། ང་ཚོའི་ཚིག་མཛོད་ཐོན་རིམ་དང་པོ་དེ་སོག་ཡིག་ཐོག་བསྒྱུར་རྒྱུའི་སྐད་ཆ་ཞིག་འགྲོ་བསྡད་ཡོད། དེ་ནས་རྒྱ་ཡིག་ནང་བསྒྱུར་རྒྱུར་ཡང་ཧ་ལམ་བརྒྱ་ཆ་ ༨༠ ནས་ ༩༠ ཙམ་གཏན་ཁེལ་བསྡད་ཡོད།རིམ་བཞིན་དབྱིན་ཡིག་སོགས་ཀྱི་ནང་དུའང་བསྒྱུར་རྒྱུ་ཡོང་གི་རེད། ཅེས་བརྗོད་བྱུང་། ཁོང་གིས་ལས་གཞི་འདིའི་ཐད་འབད་བརྩོན་གནང་དགོས་པའི་འབོད་སྐུལ་གནང་དོན། ལས་གཞི་འདིའི་ཐོག་ལ་རྣམ་པ་ཚོས་ཤུགས་སྐྱོན་རྒྱུ་ཧ་ཅང་གལ་ཆེན་པོ་རེད། གཅིག་ནས་སྤྱི་བའི་ལས་དོན་རེད་ལ་གཉིས་ནས་སོ་སོ་སྒེར་ལའང་ཕན་ཐོགས་ཆེན་པོ་ཡོད་པ་རེད། བཙན་བྱོལ་སྤྱི་ཚོགས་ནང་དུ་ཚིག་མཛོད་ཡག་པོ་བསྒྲིག་ཤེས་མཁན་ཧ་ཅང་དཀོན་པོ་རེད། གྲངས་ཀ་མི་ཐེབས་པ་མ་ཞིག་མ་རེད། དཔེར་ན། ཀརྨ་སྨོན་ལམ་དང་། ལྡོང་ཐོག་སྤྲུལ་སྐུ། བྲག་གཡབ་སྐྱབས་མགོན་སོགས་རེད། དེ་ཙམ་མ་གཏོགས་ཡོད་པ་མ་རེད། ལྷག་པར་དུ་ཚིག་མཛོད་རིག་པའི་འགྲོ་སྟངས་ཤེས་མཁན་དེ་ལས་ཀྱང་ཁག་པོ་རེད།ལས་གཞི་འདི་ལ་མི་བརྒྱ་ལྷག་གིས་ཕྱག་ལས་གནང་བ་བརྒྱུད་དེ། མ་འོངས་པར་ཚིག་མཛོད་སྒྲིག་ཤེས་མཁན་མི་ ༣༠ ཙམ་ཐོན་པ་ཡིན་ན་བཙན་བྱོལ་གྱི་གཟི་བརྗིད་རེད་ལ། དམིགས་བསལ་ཆོས་བརྒྱུད་ཁག་སོ་སོར་ཡང་ཆེད་ལས་མི་སྣ་འཐོན་རྒྱུར་ཐུགས་ཕན་སོས་པ་ཡོང་གི་རེད། ཅེས་བརྗོད་སོང་། ང་རང་ལ་མཚོན་ན་ཐད་ཀར་འགན་འཁུར་རྒྱུ་དེ་འཛིན་སྐྱོང་དང་། གློག་ཀླད་ནང་གི་འཕྲུལ་ལས་འགྲོ་སྟངས་དང་མཉེན་ཆས་བཟོ་རྒྱུ་སོགས་རེད། ཡི་གེ་ཞུ་དག་གཏོང་བྱེད་ཀྱི་མཉེན་ཆས་ཤིག་བཟོ་བཞིན་ཡོད།  སྤྱིར་མཉེན་ཆས་དེས་ཡི་གེ་བརྒྱ་ཆ་ ༥༠ ནས་ ༦༠ ཙམ་ཞུ་དག་གཏོང་ཐུབ་པ་བྱུང་ན་སྤྱི་ཚོགས་ནང་འགྲེམ་སྤེལ་བྱེད་དགོས་བསམ་ཡོད་ཀྱང་། ད་ལོ་བཀྲམ་ཐུབ་ས་མ་རེད། འཐུས་སྒོ་ཇེ་ཚང་དུ་བཏང་སྟེ་ལོ་རྗེས་མ་ཚོགས་འདུ་ཚོགས་སྐབས་རྩོམ་སྒྲིག་པ་ཚོར་འཕུལ་དགོས་བསམ་ཡོད། བློ་བཟང་སྨོན་ལམ་ལགས་ཀྱིས་མ་འོངས་པའི་ལས་འཆར་སྐོར་བརྗོད་དོན། ད་ལོའི་ཕྱི་ཟླ་ ༡༡ དང་ ༡༢ ཙམ་ལ་ང་ཚོའི་ཚིག་མཛོད་ཐོན་རིམ་དང་པོ་དེ་འགྲེམས་སྤེལ་བྱ་རྒྱུ་ཡིན། ཐོག་མའི་རེ་བར་ཐོན་རིམ་དང་པོའི་ནང་ཐ་སྙད་འབུམ་ ༡ ལྷག་བྱུང་ན་སེམས་ཡོད་པ་དང་། ང་ཚོའི་དྷ་རམ་ས་ལའི་ལས་ཁུངས་ཀྱིས་ཐ་སྙད་ཁྲི་ ༨ དང་། ས་གནས་ཆོས་བརྒྱུད་ཁག་ནས་ཐ་སྙད་ཁྲི་ ༢ ལྷག་འབྱོར་རྒྱུར་རེ་བ་བརྒྱབ་ཡོད། ཕལ་ཆེར་རེ་བ་ལྟར་འགྲུབ་ཀྱིན་འདུག ཐ་སྙད་འབུམ་གཅིག་དེ་དྲ་ཐོག་ཏུ་བཞག་སྟེ་འོག་ཏུ་མཆན་རྒྱག་ས་བཟོས་པ་ཡིན་ན། འབྱུང་འགྱུར་ལོ་གསུམ་རིང་མང་ཚོགས་ཀྱིས་ཞིབ་བསྡུར་དང་གླེང་སློང་མང་པོ་བྱས་རྗེས། ལོ་དྲུག་གི་རྗེས་སུ་བསྐྱར་བཅོས་དང་ཐ་སྙད་གསར་པ་རྣམས་ཁ་སྣོན་རྒྱག་རྒྱུའི་འཆར་གཞི་ཡིན། ང་ཚོའི་རེ་བར་ལོ་དང་པོ་དང་གཉིས་པ། གསུམ་པར་ང་ཚོའི་ལས་ཁུངས་ཀྱིས་ཤུགས་རྒྱག་རྒྱུ་དང་། ལོ་རྗེས་མ་དེ་དག་ནང་གཙོ་བོ་ཆོས་བརྒྱུད་ཁག་གི་རྩོམ་སྒྲིག་རྣམ་པ་ཚོར་རེ་བ་རྒྱག་གི་ཡོད། རྗེས་མའི་ལོ་གསུམ་དེར་རྣམ་པ་ཚོར་རེ་བ་བཅོལ་ཏེ་ཐ་སྙད་ཁྲི་ ༨ ཞིག་བཟོ་ཐུབ་རྒྱུར་འབོད་སྐུལ་ཞུ་བཞིན་ཡོད། ང་ཚོའི་ཚིག་མཛོད་ཐོན་རིམ་དང་པོ་དབུ་འབྱེད་དང་བསྟུན་ཏེ། གཅིག་བྱས་ན་མཛད་སྒོ་གྲ་རྒྱས་པོ་ཞིག་བྱེད་ཐུབ་ན་བསམ་ཡོད། བྱུང་ན་ཆོས་བརྒྱུད་ཁག་གི་བླ་ཆེན་གང་ཐུབ་ཐུབ་ཅིག་གདན་ཞུ་གནང་རྒྱུ་དང་། ༧རྒྱལ་བ་རིན་པོ་ཆེ་མཆོག་གིས་དབུ་འབྱེད་གནང་གི་རེད། ཡིན་ནའང་ཚིག་མཛོད་འདི་བོད་ནང་སོགས་སུ་འགྲོ་དགོས་དུས་བསམ་བློ་གཏོང་དགོས་ས་མང་པོ་འདུག ཉིན་ཁ་ཤས་རྗེས་ང་ཚོར་སྐུ་མདུན་ལ་མཇལ་ཁ་ཞིག་ཡོད། སྐབས་དེར་ད་བར་གང་བྱུང་བ་ཁག་ཡར་སྙན་གསེང་ཞུ་རྒྱུ་དང་། མ་འོངས་པར་ཚིག་མཛོད་དབུ་འབྱེད་གནང་རྒྱུ་སོགས་ཞུ་རྒྱུ་ཡིན། ཞེས་བརྗོད་བྱུང་། ཞེས་སོགས་བརྗོད་བྱུང་། དེའི་རྗེས་སྨོན་ལམ་བརྡ་འཕྲིན་ཞིབ་འཇུག་ཁང་གི་རྩོམ་སྒྲིག་འགན་འཛིན་འཇུ་བསྟན་སྐྱོང་ལགས་ཀྱིས་ཚིག་མཛོད་རྩོམ་སྒྲིག་བྱེད་ཕྱོགས་སྐོར་ལ་གཏམ་བཤད་གནང་དོན།ང་ཚོའི་ཚིག་འཛོད་དེ་རྩ་བའི་འགྲོ་སྟངས་ཆེན་པོ་གསུམ་དུ་བཀར་ཡོད་པ་རེད། དང་པོ། ཚིག་མཛོད་ལ་བརྡ་སྤྲོད་ཀྱི་དབྱེ་བ་བཀར་རྒྱུ་དང་། གཉིས་པ། བརྡ་ཆད་ཀྱི་དབྱེ་པ་ངོ་འཛིན་བྱ་རྒྱུ། གསུམ་པ། འགྲེལ་བཤད་རྒྱག་རྒྱུ་བཅས་རེད། དཔེར་ན། ཐ་སྙད་ཅིག་འཁྱེར་ཡོང་སྐབས། དང་པོ། བརྡ་སྤྲོད་ཀྱི་རིམ་པ་ལ་བསམ་བློ་གཏོང་དགོས། དཔེར་ན། མིང་རེད་དམ། ཚིག་རེད། ཡང་ན་རྒྱན་ཚིག་རེད། ང་ཚོས་བརྡ་སྤྲོད་ཀྱི་རིམ་པ་བཅུ་གྲངས་ཤིག་གཏན་འབེབས་བྱས་ཡོད། དེ་དག་ལས་གང་གི་ཁུངས་སུ་གཏོགས་མིན་ཐག་གཅོད་དགོས། གཉིས་པ། བརྡ་ཆད་ཀྱི་སྐོར་ལ་བསམ་བློ་གཏོང་དགོས། དཔེར་ན། ཚིག་འདི་མདོ་ཡི་ཚིག་ཅིག་རེད་དམ། སྔགས་ཀྱི་ཚིག་ཅིག་རེད། མདོ་ཡི་ཚིག་ཡིན་ན་བྱེ་བྲག་སྨྲ་བ་རེད་དམ། མདོ་སྡེ་སྨྲ་བ་རེད། དབུ་མ་ལྟ་བུར་མཚོན་ན་མཐའ་འགྱུར་བ་རེད་དམ་རང་རྒྱུད་པ་རེད། དེ་བཞིན། རྙིང་མའི་ནང་ནས་ཚིག་ཅིག་རྙེད་པ་ཡིན་ན། དེ་རྙིང་མའི་ཐུན་མོང་མ་ཡིན་པའི་ཚིག་ཅིག་རེད་དམ། བོད་བརྒྱུད་ནང་བསྟུན་ཐུན་མོང་གི་ཚིག་གཅིག་རེད། གསུམ་པ། ཚིག་དེ་ཚུར་བླངས་པ་དང་གོ་རིམ་ཡོད་པའི་སྒང་ནས་བསྒྲིགས་རྗེས་མཆན་རྒྱག་དགོས་པ་རེད། དཔེར་ན། ཚིག་དེ་གང་ནས་བྱུང་མིན་ལ་འགྲེལ་བཤད་རྒྱག་དགོས་པ་མ་ཟད། གཞུང་ལུགས་གང་དུ་ཡོད་པ་དང་མཁས་པ་སུས་གསུངས་པ། གང་ནས་དཔར་སྐྲུན་བྱས་པའི་དེབ་གང་ཞིག་གི་ཤོག་ངོས་ག་ཚོད་ནང་འདུག་པ་སོགས་ཀྱི་མཆན་དེ་རྒྱག་རྒྱུ་གལ་ཆེན་པོ་རེད། ཅེས་བརྗོད་སོང་། ཁོང་གིས་མུ་མཐུད་དུ་བརྗོད་དོན། རྣམ་པ་ཚོར་མཚོན་ན་ཆོས་ལུགས་ནང་གི་མཐུན་མོང་མ་ཡིན་པའི་ཐ་སྙད་དང་ཆོས་ལུགས་ཀྱི་ལོ་རྒྱུས་སྐོར་འགྲེལ་བཤད་རྒྱག་རྒྱུ་དེ་ཡིན་སྟབས། དང་པོ། མདོ་སྔགས་གཉིས་ཀྱི་བརྡ་ཆད་དབྱེ་བ་བཀར་དགོས། དཔེར་ན། མདོ་དང་སྔགས་གཉིས་ཀའི་ནང་དུ་ཡོད་པའི་ཐ་སྙད་ཅིག་ཡིན་ན། དེ་མདོ་ཡི་ཐ་སྙད་རེད་མ་གཏོགས་སྔགས་ཀྱི་ཐ་སྙད་རེད་ལབ་ན་འགྲིག་གི་ཡོད་པ་མ་རེད། སྔགས་ཀྱི་ནང་དུ་ཡོད་པའི་ཐ་སྙད་གང་ཞིག་མདོའི་ནང་དུ་རྩ་བ་ནས་མ་རྙེད་ཚེ། སྔགས་ཀྱི་ཐ་སྙད་ཡིན་པ་གཏན་ཁེལ་བ་རེད། སྔགས་ཀྱི་ནང་ནས་ཀྱང་། བྱ་རྒྱུད་སྤྱོད་རྒྱུད། རྣལ་འབྱོར་རྒྱུད་དང་རྣལ་འབྱོར་བླ་མེད་རྒྱུད་སོགས་ཡོད་པ་རེད། བྱ་རྒྱུད་དང་སྤྱོད་རྒྱུད་ཀྱི་ཐ་སྙད་དེ་གོང་གི་མདོ་སྔགས་ཀྱི་རིག་པ་དེ་བཏང་ན་འགྲིག་གི་རེད། རྣལ་འབྱོར་བརྒྱུད་དང་རྣལ་འབྱོར་བླ་མེད་བརྒྱུད་ཧ་ལམ་འདྲ་བོ་ཡིན་ས་རེད། སྔ་མ་སྔ་མའི་ནང་གི་ཐ་སྙད་རྣམས་ཟུར་དུ་བཏོན་པ་ཡིན་ན། ཕྱི་མ་ཕྱི་མའི་ནང་དུ་ཐ་སྙད་མ་འདྲ་བ་གང་ཡོད་པ་དེ་རང་བཞིན་གྱིས་གསལ་བོ་ཆགས་ཡོང་གི་རེད། དེ་ལྟར་བྱས་ན་ད་གཟོད་རྙིང་མ་དང་བོན་གྱི་རྫོགས་པ་ཆེན་པོའི་ནང་དུ་ཐུན་མོང་མ་ཡིན་པའི་ཐ་སྙད་ག་ཚོད་ཡོད་མེད་གསལ་བོ་ཆགས་ཡོང་གི་རེད། དེ་བཞིན་དུ་བྱེ་བྲག་སྨྲ་བ་དང་མདོ་སྡེ་སྨྲ་བའི་ཐ་སྙད་ཀྱང་དེ་ལྟར་ལྟ་དགོས་པ་རེད། རྙོག་གྲ་ཆེ་ཤོས་དེ་སེམས་ཙམ་པ་རེད།གང་ཡིན་ཟེར་ན། སེམས་ཙམ་པའི་ཆོས་འབྱུང་ཧ་ཅང་ཁག་པོ་ཡོད་དུས་རྣམ་པ་ཚོས་ཞིབ་འཇུག་གནང་དགོས་པ་ཞིག་རེད། ཅེས་བརྗོད་བྱུང་། འཇུ་བསྟན་སྐྱོང་ལགས་ཀྱིས་མུ་མཐུད་དོན། འགྲེལ་བཤད་རྒྱག་དུས་རྣམ་པ་ཚོས་འགྲེལ་བྱ་དང་འགྲེལ་བྱེད་གཉིས་ལ་ཐུགས་ཉམས་འཇོག་དགོས་པ་གལ་ཆེན་པོ་རེད། གང་ཡིན་ཞེས་ན་སྤྱི་ཡིས་བྱེ་བྲག་ལ་འགྲེལ་བཤད་བརྒྱབ་ཡོང་དུས་བྱེ་བྲག་གིས་སྤྱི་ལ་འགྲེལ་བཤད་རྒྱག་གི་ཡོད་པའི་རྣམ་པ་དེ་འགྲིག་གི་ཡོད་པ་མ་རེད།དཔེར་ན། རྣམ་དབྱེ་ཟེར་བ་དེ་བྱེད་སྒྲ་རེད་ལབ་ན། སྤྱི་ཞིག་ལ་བྱེ་བྲག་གིས་འགྲེལ་བཤད་རྒྱག་གི་ཡོད་སྟབས་དེ་འགྲིག་གི་ཡོད་པ་མ་རེད། སྡུག་བསྔལ་བདེན་པ་ཞེས་པ་དེ་བདེན་པ་གཞི་རེད་ལབ་ན་བྱེ་བྲག་ཅིག་ལ་སྤྱིས་འགྲེལ་བཤད་རྒྱག་གི་ཡོད་སྟབས་དེའང་འགྲིག་གི་ཡོད་པ་མ་རེད། གཞན། ལྡོག་འགྲེལ་དེར་ཡང་བསམ་བློ་གཏོང་དགོས་པ་ཞིག་དང་ལྡོག་ཕྱོགས་དེ་ངོས་འཛིན་ཐུབ་པ་གཞིར་བཞག་གིས་འགྲེལ་བཤད་རྒྱག་དགོས་པ་ལས། དཔེར་ན། གྲང་མོ་ལབ་ན་ཚ་བོའི་ལྡོག་ཕྱོགས་རེད་དང་། ཚ་བོ་ལབ་ན་གྲང་མོའི་ལྡོག་ཕྱོགས་རེད་ཟེར་དུས་གཉིས་ཆ་སྐོར་བ་ཞིག་བརྒྱབ་པ་ལས་གཅིག་ཀྱང་ངོས་འཛིན་ཐུབ་ཀྱི་མེད་སྟབས། ཕྱོགས་གཅིག་ལ་འགྲེལ་བཤད་ཡག་པོ་ཞིག་བརྒྱབ་པའི་རྗེས་ལ་གཞན་པ་དེ་དེའི་ལྡོག་ཕྱོགས་རེད་ཅེས་ལབ་ན་གོ་བརྡ་འཕྲོད་པ་ཞིག་རེད། ཅེས་སོགས་ཀྱི་ལམ་སྟོན་བྱས་སོང་། མཇུག་ཏུ་གསར་འགོད་པས་གྲྭ་བློ་བཟང་སྨོན་ལམ་ལགས་སུ་བཅར་འདྲི་ཞུས་དོན་ལྟར་ན། ད་ལྟའི་ཆར་ལས་གཞི་དེར་བོད་པའི་ཆོས་ལུགས་ཆོས་བརྒྱུད་ཁག་ཏུ་ཡན་ལག་ལྟ་བུའི་ཐ་སྙད་བསྡུ་རུབ་གནང་མཁན་གྱི་རྩོམ་སྒྲིག་ཁང་ ༢༡ ཙམ་ཡོད་པ་དང་། མ་འོངས་པར་རྩོམ་སྒྲིག་ཁང་གཉིས་གསར་དུ་འཛུགས་འཆར་ཡོད་འདུག་ལ། སྨན་དང་རྩིས་སོགས་རིག་ཚན་གཞན་དག་ལ་སྤྱི་ཚོགས་ནང་གི་ཆེད་མཁས་པར་ཕོགས་སྤྲད་དེ་ཆེད་ལས་འགན་སྤྲད་ཡོད་འདུག
