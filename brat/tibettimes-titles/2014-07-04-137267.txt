མི་ཚང་གསུམ་གྱི་ལས་འགུལ་ ༣༡ པ།

ཁ་སང་ཚེས་ ༣ ཉིན། སུད་སིར་གནས་སྡོད་ལྷུན་གྲུབ་ཚང་ཀརྨ་བཀྲ་ཤིས་ཚེ་རིང་དང་སྟག་ཚང་དཀོན་མཆོག བཀྲ་བསྟན་ཚང་པ་སངས་ནང་མི་གསུམ་གྱིས་རྒྱལ་ཚོགས་ལ་བོད་དོན་ཞུ་གཏུག་གི་ལས་འགུལ་ཐེངས་ ༣༡ པ་དེ་སྤེལ་ཡོད་འདུག ཐེངས་འདིའི་ལས་འགུལ་ནང་ཁྱིམ་ཚང་གསུམ་གྱི་ནང་མི་མ་ཡིན་པའི་གདུང་སེམས་མཉམ་སྐྱེ་ཆེད་ཨ་རིག་ཀརྨ་རྒྱ་མཚོ་ལགས་དང་། རྒྱལ་ཚང་བཀྲིས་ལྷ་མོ། ཡ་རུང་ཚང་བློ་གྲོས་རྒྱ་མཚོ་བཅས་མི་ ༦ མཉམ་ཞུགས་གནང་ཡོད་འདུག  ནང་མི་གསུམ་གྱི་ལས་འགུལ་འདི་སུད་སི་ཇི་ནི་ཝར་རྟེན་གཞི་བྱས་པའི་མཉམ་འབྲེལ་ཚོགས་ལས་ཁུངས་མདུན་དང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་འགྲོ་བ་མིའི་ཐོབ་ཐང་མདུན་ཐང་བཅས་སུ་བོད་ཀྱི་རྒྱལ་དར་སྒྲེང་འཛུགས་དང་། བོད་ནང་གི་དཀའ་སྡུག་གི་གནས་བབ་འཁོད་པའི་ཡིག་ཆ་ཁག་འགྲེམས་སྤེལ། ལྷག་པར་སྐུ་ལུས་མེར་སྲེག་གནང་མཁན་རྒྱལ་གཅེས་པ་རྣམས་ཀྱི་འདྲ་པར་འགྲེམས་འཇོག དེ་བཞིན་ཁོང་རྣམས་ཀྱི་སྐུ་ལུས་མེར་བསྲེགས་གནང་དགོས་པའི་རྒྱབ་ལྗོངས་འབྲེལ་ བརྗོད་དང་ཁོང་རྣམས་ཀྱི་འབོད་སྒྲའམ་དགོས་འདུན་འཁོད་པའི་ཡིག་ཆ་ཁག་དང་འཕྲེད་དར། དེ་བཞིན་སྐུ་ལུས་མེར་བསྲེགས་གནང་བའི་ཐད་༧གོང་ས་མཆོག་དང་དབུས་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་ལངས་ཕྱོགས་དང་དགོངས་ཚུལ་ག་རེ་ཡིན་པའི་སྐོར་གྱི་ཡིག་ཆ་འགྲེམ་སྤེལ་བཅས་གནང་གི་ཡོད་འདུག  ཁོང་ཚོས་ལས་འགུལ་སྤེལ་ཐེངས་རེར་ཉིན་གཅིག་ཡིག་ཆ་དཔར་སྐྲུན་སོགས་ཀྱི་ཆེད་གྲ་སྒྲིག་བྱེད་དགོས་པ་དང་། དངོས་གཞིའི་ཉིན། སྡོད་ས་རང་ནས་ཆུ་ཚོད་གསུམ་གྱི་ས་ཐག་ཡོད་པའི་མེ་འཁོར་ནང་བསྡད་དེ་བསྐྱོད་ དགོས་པ་དང་། སྐབས་རེར་རླུང་གཡུག་པ་དང་། ཆར་པ་བབ་པ་སོགས་དཀའ་ངལ་རབས་དང་རིམ་པ་འཕྲད་ཀྱི་ཡོད་ཀྱང་ད་ལྟའི་བར་ཞུ་གཏུག་ལས་འགུལ་ཐེངས་ ༣༡ གནང་ཡོད་པ་མ་ཟད། དེ་བཞིན་སུད་སིའི་གཞུང་དང་། གྲོས་ཚོགས། ས་གནས་གྲོང་ཁྱེར་ནང་དུ་ཡང་བོད་དོན་ཞུ་གཏུག་དང་། དྲིལ་བསྒྲགས། རྒྱབ་སྐྱོར་འབོད་སྐུལ་གྱི་གོམ་བགྲོད་སོགས་ཀྱི་ལས་འགུལ་ཡང་གང་ཙམ་སྤེལ་ཡོད་པ་ རེད་འདུག ཉིན་འདིར་ལས་འགུལ་བ་རྣམས་ཀྱིས་བོད་གནས་འགྲེལ་བརྗོད་ཡིག་ཆ་གྲངས་ ༦༠༠ ལྷག་ཕྱི་རྒྱལ་མི་རིགས་རྣམས་ལ་འགྲེམས་སྤེལ་གནང་ཐུབ་ཡོད་པ་མ་ཟད། རྒྱ་རིགས་ཡུལ་སྐོར་སྤྲོ་འཆམ་པ་ཚོར་བོད་མི་ཡོངས་ལ་མིང་དོན་མཚུངས་པའི་རང་ སྲིད་རང་སྐྱོང་ཡོང་ཐབས་ཀྱི་བསམ་འཆར་ཡིག་ཆ་སོགས་འགྲེམས་སྤེལ་གང་མང་གནང་ཡོད་ པ་དང་། དེར་ཉི་ཧོང་གི་ཡུལ་སྐོར་སྤྲོ་འཆམ་པ་མང་པོ་ཞིག་གི་གདུང་སེམས་མཉམ་སྐྱེད་དང་རྒྱབ་སྐྱོར་གནང་ཡོད་འདུག
