ས་རུད་ཀྱིས་མི་ ༤༥ ལྷག་གི་ཚེ་སྲོག་ཤོར།

ཧི་མ་ཅལ་མངའ་སྡེའི་མན་ཌི་རྫོང་དུ་ས་རུད་བྱུང་སྟེ་ཉུང་མཐར་མི་གྲངས་ ༤༥ རྐྱེན་འདས་སུ་ཕྱིན་ཡོད་འདུག ལོ་རེའི་ཕྱི་ཟླ་ ༧ པ་ནས་ ༩ པའི་བར་རྒྱ་གར་ནང་ཆར་དུས་ཡིན་པས་ཆར་བ་ཤུགས་ཆེན་བབས་ཏེ།  ཆུ་ལོག་དང་ས་རུད་ཀྱི་འབྱུང་བའི་གནོད་སྐྱོན་མང་པོ་བྱུང་གི་ཡོད་པ་དང་། དེ་རིང་བྷི་བྷི་སི་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན།  རྒྱ་གར་ཧི་མ་ཅལ་མངའ་སྡེའི་མན་ཌི་རྫོང་དུ་ས་རུད་བྱུང་སྟེ་སྤྱི་སྤྱོད་རླངས་འཁོར་གཉིས་ལ་རྡབ་སྐྱོན་དང་མི་ ༤༥ ལྷག་ཅིག་རྐྱེན་འདས་སུ་ཕྱིན་ཡོད། ས་རུད་བྱུང་སྐབས་སྤྱི་སྤྱོད་རླངས་འཁོར་དེ་གཉིས་འགྲོ་མཚམས་བཞག་སྟེ་ནང་གི་འགྲུལ་བ་ཚོ་སྐྱོབ་ཐབས་བྱས་ཡོད་ཀྱང་ས་རུད་ཤུགས་ཆེན་བྱུང་སྟབས་འགྲུལ་བ་མང་པོ་རྐྱེན་འདས་སུ་ཕྱིན་ཡོད་འདུག  རྒྱ་གར་ཉེན་སྐྱོབ་དམག་མིས་བརྗོད་པ་ལྟར་ན།  ཁོ་ཚོས་ད་བར་རྨས་སྐྱོན་བྱུང་བའི་མི་ལྔ་སྐྱོབ་ཐུབ་ཡོད་པ་དང་། མི་ཁག་ཅིག་ད་དུང་གར་ཡོད་ཅི་བྱུང་མེད་པར་གྱུར་ཡོད་པས་རྐྱེན་འདས་ཀྱི་གྲངས་འབོར་མུ་མཐུད་འཕར་གྱི་ཡོད་སྐོར་བརྗོད་འདུག ཐེངས་འདིའི་ས་རུད་ཁྲོད་ཧི་མ་ཅལ་མངའ་སྡེའི་རྒྱལ་ས་སིམ་ལཱ་ནས་ལེ་དབར་ ༡༢༤ ཡི་ལམ་ཁར་གནོད་སྐྱོན་ཐེབས་ཡོད་པ་དང་། དེ་བཞིན་ཁང་པ་དང་རླངས་འཁོར་གཞན་ཁག་ཅིག་ལའང་གནོད་སྐྱོན་བཟོས་ཡོད་འདུག་ལ། ས་རུད་གནོད་སྐྱོན་བྱུང་རྗེས།  རྒྱ་གར་སྲིད་བློན་མུ་དྷེ་མཆོག་གིས་ཊོའུ་ཊིར་ནང་འདས་གྲོངས་སུ་ཕྱིན་པའི་འགྲུལ་བ་ཚོའི་ཁྱིམ་མིར་གདུང་སེམས་མཉམ་སྐྱེད་ཞུ་ཡི་ཡོད། ཅེས་གསུངས་ཡོད་འདུག ཆར་དུས་ཀྱིས་རྒྱ་གར་མངའ་སྡེ་ ༢༠ ལྷག་གི་མི་མང་ས་ཡ་མང་པོར་གནོད་སྐྱོན་ཐེབས་ཀྱིན་ཡོད་པ་དང་། ཁ་སང་གཟའ་ཉི་མ་ཉིན་ཨ་སམ་ནང་ཅུ་ལོག་བྱུང་སྟེ་ཉུང་མཐར་མི་ ༡༠ རྐྱེན་འདས་སུ་ཕྱིན་འདུག རྒྱ་གར་གྱི་བརྒྱུད་ལམ་ཁག་གི་གནས་ཚུལ་ལྟར་ན། ད་ལོའི་ཆར་དུས་མགོ་བཙུགས་ནས་ད་ལྟའི་བར་དུ་ཉུང་མཐར་མི་གྲངས་ ༥༧ ས་རུད་ཆུ་ལོག་ཁྲོད་རྐྱེན་འདས་སུ་ཕྱིན་ཡོད་འདུག གསར་འགོད་པ། སྐལ་བཟང་སྦྱིན་པ།
