དགའ་ལྡན་ཁྲི་ཐོག་ ༡༠༣ པ་བསྐོ་བཞག་མཛད།

༧གོང་ས་མཆོག་ལ་དགོངས་ཞུ་མཚན་འཁོད་སྙན་ཞུ་ཕུལ་བ་ལྟར་ཤར་པ་ཆོས་རྗེ་རྗེ་བཙུན་བློ་བཟང་བསྟན་འཛིན་མཆོག་དགའ་ལྡན་ཁྲི་ཐོག་བརྒྱ་དང་གསུམ་པར་བསྐོ་གཞག་མཛད་ཡོད་འདུག བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་དྲ་བའི་ཐོག་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། དགའ་ལྡན་ཁྲི་ཐོག་ ༡༠༢ རི་རྫོང་སྲས་ཐུབ་བསྟན་ཉི་མ་ལུང་རྟོགས་བསྟན་འཛིན་ནོར་བུ་མཆོག་གིས་དགའ་ལྡན་ཁྲི་པའི་མཛད་འཁུར་ལོ་བདུན་ཚང་བ་དང་། ༧གོང་ས་མཆོག་ལ་དགོངས་ཞུ་མཚན་འཁོད་སྙན་ཞུ་ཕུལ་བ་ལྟར། དགའ་ལྡན་ཁྲི་ཐོག་ ༡༠༣ པར་ཤར་པ་ཆོས་རྗེ་རྗེ་བཙུན་བློ་བཟང་བསྟན་འཛིན་མཆོག་བསྐོ་བཞག་མཛད་ཡོད་སྟབས། འདི་ལོའི་ཕྱི་ཟླ་ ༡༡ ཚེས་ ༥ ཉིན་རྒྱ་གར་ལྷོ་ཕྱོགས་མོན་གྷོ་འདོད་རྒུ་གླིང་བོད་མིའི་གཞིས་ཆགས་སུ་རྟེན་གཞི་བྱས་པའི་དགའ་ལྡན་ཁྲི་ཐོག་ཁང་དུ་ཁྲི་འདོན་མཛད་སྒོ་འཚོགས་རྒྱུ་ཐུགས་གཏན་འཁེལ་ཡོད་འདུག ཤར་པ་ཆོས་རྗེ་རྗེ་བཙུན་བློ་བཟང་བསྟན་འཛིན་མཆོག་དགུང་གྲངས་ ༨༠ ཕེབས་ཡོད་པ་དང་། བོད་ཁམས་ཡ་རར་སྐུ་འཁྲུངས་ཞིང་ཆུང་དུས་སུ་ཡ་ར་ཚེམ་དགོན་པར་རབ་ཏུ་བྱུང་སྟེ། དགུང་ལོ་ ༡༦ ཐོག་འབྲས་སྤུངས་བློ་གསལ་གླིང་དུ་ཆོས་ཞུགས་ཀྱིས་གཞུང་ཆེན་ཁག་ལ་བསླབ་པ་སློབ་ གཉེར་གནང་། ཕྱི་ལོ་ ༡༩༥༩ ལོར་བཙན་བྱོལ་དུ་ཕེབས་ནས་སྦག་ས་ཆོས་སྒར་དུ་སློབ་གཉེར་གནང་བ་དང་། དེ་ནས་འབྲས་སྤུངས་བློ་གསལ་གླིང་གྲྭ་ཚང་གི་སྒྲིག་ཁོངས་སུ་ཞུགས་ནས་མུ་མཐུད་སློབ་གཉེར་གནང་འདུག རིན་པོ་ཆེ་མཆོག་གིས་གྲྭ་ཚང་གི་ཕྱག་མཛོད་དང་། གྲྭ་ཚང་གི་ཞིང་ལས་བྱེད་མཁན་གྱི་རྒྱ་དཔོན་དང་བཅུ་དཔོན། དགེ་བསྐོས། དཔེ་ཁྲིད་དགེ་རྒན་སོགས་ཀྱི་ཐུགས་འགན་ཡང་བཞེས་ཡོད་པ་མ་ཟད། དགེ་བཤེས་ལྷ་རམས་པ་དང་སྔགས་རམས་པའི་མཚན་རྟགས་བཞེས་རྗེས། ལོ་གསུམ་རིང་རྒྱུད་སྟོད་བླ་མ་དབུ་མཛད་དང་ལོ་གསུམ་རིང་མཁན་པོ་བཅས་ཀྱང་གནང་ འདུག ཕྱི་ལོ་ ༢༠༠༣ ལོར་༸གོང་ས་༸སྐྱབས་མགོན་ཆེན་པོ་མཆོག་གིས་གནས་མཆོག་རྡོ་རྗེ་གདན་དུ་ཤར་པ་ཆོས་རྗེའི་མཚན་གནས་སྩལ་རྗེས། ད་བར་ལོ་ངོ་ ༡༤ རིང་ཤར་པ་ཆོས་རྗེ་གནང་ཡོད་པ་རེད། གསར་འགོད་པ། པདྨ་མཚོ།
