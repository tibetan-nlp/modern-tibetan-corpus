སུ་ཞིག་ཡར་བསླེབས་པ་དང་། སུ་ཞིག་མར་ཟེགས་པ།
སུ་ཞིག་ཕྱུར་འཕུད་བཏང་བ།

གཟའ་མིག་དམར་ཉིན་བྷི་བྷི་སི་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་དོན། རྒྱ་ནག་མི་དམངས་འཐུས་མིའི་ཚོགས་ཆེན་ནི་རྒྱ་ནག་གི་སྲིད་ཇུས་གཏན་འབེབས་བྱེད་མཁན་ལྟེ་བ་ཡིན་པར་མཐོང་གི་ཡོད་ཀྱང་། ང་ཚོས་པེ་ཅིན་གཞུང་ལ་བསྐྱར་དུ་དོ་སྣང་ཞིག་བྱས་ཏེ་སུ་ཞིག་གིས་སྲིད་དབང་བེད་སྤྱད་དེ་བསྐྱར་དུ་སྲིད་སྟེགས་སྟེང་ལང་གི་ཡོད་པ་དང་སུ་ཞིག་པེ་ཅིན་གྱི་སྲིད་སྟེགས་ལས་ཕྱིར་འཕུད་བཏང་ཡོད་པ་རེད། ཡར་ཐོན།  སྲོག་ཤིང་། རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་ལ་ལོ་སྔོན་མའི་ཕྱི་ཟླ་ ༡༠ ནང་རྒྱ་ནག་གི་སྲོག་ཤིང་ཞེས་བའི་མིང་དེ་ཐོབ་སྐབས། ཁྱེད་ཀྱིས་ཞི་ཅིན་ཕིན་ནི་རྒྱ་ནག་གཞུང་གི་ལས་དོན་ཡོངས་རྫོགས་ཀྱི་བདག་དབང་ཡོད་མཁན་ཁོ་ན་དེ་ཡིན་པར་ངོས་འཛིན་བྱས་ཡོད་སྲིད་ཀྱང་། ད་བར་རྒྱ་ནག་གི་མགོ་ཁྲིད་སུ་ཡང་རྐུབ་སྟེགས་དེའི་སྟེང་ལས་སླ་བོའི་ངང་འདུག་ཐུབ་མེད། རྒྱ་ནག་གི་མགོ་ཁྲིད་ཚོས་མིང་བྱང་དེའི་ཆེད་དུ་དཀའ་ལས་མང་པོ་རྒྱག་གི་ཡོད་པ་དང་། མིང་བྱང་དེ་ནི་ཁོ་ཚོ་ཚང་མས་ལྷག་བསམ་རྣམ་དག་གིས་གཞུང་ལ་སྲི་ཞུ་ཡང་དག་སྒྲུབ་པའི་མཚོན་རྟགས་ཤིག་ལ་ངོས་འཛིན་བྱེད་ཀྱིན་ཡོད། ད་ལོའི་རྒྱ་ནག་མི་དམངས་འཐུས་མིའི་ཚོགས་ཆེན་སྟེང་ཐེངས་དང་པོར་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་ནི་རྒྱ་ནག་དམར་ཤོག་ཚོགས་པའི་སྲོག་ཤིང་གི་མགོ་ཁྲིད་ཡིན་ཞེས་ཐེངས་མང་པོ་བཏོན་ཡོད་པ་དང་། ཞི་ཅིན་ཕིན་ལ་རྒྱ་ནག་གི་གཙོ་འཛིན་མཱའོ་ཙི་ཏུང་ལས་ལྷག་གི་གོང་བཀུར་མཚན་གནས་མང་པོ་ཡོད་པ་མ་ཟད། ཞི་ཅིན་ཕིན་ད་བར་གྱི་རྒྱ་ནག་གི་ཡར་ཐོན་ཅན་གྱི་མགོ་ཁྲིད་གཅིག་པུ་དེའང་ཆགས་ཡོད། ཉམས་རྒུད།  ཧོང་ཀོང་གི་རྒྱལ་གཅེས་བློ་མཐུན་ཚོ། དེ་སྔ་དབྱིན་ཇུའི་གཞུང་གིས་བཙན་བཟུང་བྱས་མྱོང་བའི་ཧོང་ཀོང་གཞུང་གིས་ཟླ་འདིའི་ཟླ་མཇུག་ལ་ཧོང་ཀོང་སྤྱི་ཁྱབ་འོས་འདེམས་བྱེད་དགོས་ཀྱང་། དེར་རྒྱ་ནག་གཞུང་གི་སྟངས་འཛིན་འོག་གནོན་ཤུགས་ཆེན་པོ་སྤྲོད་བཞིན་ཡོད། རྒྱ་ནག་འཐབ་ཕྱོགས་གཅིག་གྱུར་ཁང་གི་མགོ་ཁྲིད་ཀྲང་ཏི་ཅཱང་ (Zhang Dejiang)གིས་ལོ་ངོ་གཉིས་ཀྱི་ནང་ཚུད་དུ་རྒྱ་ནག་གི་གྲོང་ཁྱེར་ཧྲིན་ཊིན་ (Shenzhen) གྱིས་ཧོང་ཀོང་གི་དཔལ་འབྱོར་ལ་སྟངས་འཛིན་བྱེད་ཐུབ་ཀྱི་ཡོད་པའི་ཉེན་བརྡ་བཏང་ཡོད། ཧྲིན་ཊིན་ནི་ཧོང་ཀོང་གི་ཁྱིམ་མཚེས་དང་། ལོ་ངོ་ ༣༠ ཡར་སྔོན་ལ་ཉ་བཟུང་ནས་འཚོ་བ་སྐྱེལ་བཞིན་པའི་གྲོང་གསེབ་ཆུང་ཆུང་ཞིག་ཡིན། ཡར་ཐོན།  རྒྱ་ནག་དམར་ཤོག་ཚོགས་པ། རྒྱ་ནག་དམར་ཤོག་ཚོགས་པས་ལོ་མང་རིང་རྒྱ་ནག་གི་མི་མང་དང་དམིགས་བསལ་གྱིས་ན་གཞོན་ཚོར་དམར་ཤོག་གཞུང་དང་རྒྱལ་ཁབ་ལ་རྒྱལ་ཞེན་བསླངས་མ་ཐུབ་པར་ལྷག་ཡོད་ཀྱང་། ཉམས་ཞིབ་བྱས་པ་བརྒྱུད་དེ་ད་ལོ་གཞི་ནས་མང་ཚོགས་ལ་རྒྱལ་ཞེན་གྱི་ཚོར་བ་ཤུགས་ཆེན་པོ་བསླངས་ཡོད། རྒྱ་ནག་དམར་ཤོག་ཚོགས་པས་སྤྱི་ཚོགས་དྲ་ལམ་གྱི་གལ་ཆེའི་རང་བཞིན་གནད་དུ་བཟུང་སྟེ། རྒྱ་ནག་ནང་དར་སྲོལ་ཆེ་བའི་སྐད་འཕྲིན་སྟེང་རྒྱལ་ཞེན་གྱི་བརྙན་དང་གཞས། འདྲ་པར་དང་ཡི་གེ་སོགས་བཀྲམས་ཏེ་མི་མང་ལ་རྒྱལ་ཞེན་སློང་ཐབས་བྱས་ཡོད། ད་དུང་རྒྱ་ནག་དམར་ཤོག་ཚོགས་པས་དམངས་སེམས་འགུག་ཐབས་སུ། སྐད་འཕྲིན་བརྒྱུད་དེ་མི་མང་གིས་ཐད་ཀར་རྒྱ་ནག་མི་མང་གི་འཐུས་མི་ཚོར་དྲི་བ་དྲིས་ལན་བྱེད་ཆོག་པའི་ལམ་ཁ་བཟོས་ཡོད། ད་དུང་རྒྱ་ནག་དམར་ཤོག་ཚོགས་པས་ “ངའི་མི་ཚེ་དང་ངའི་རྒྱ་ནག་” ཅེས་བའི་ཚན་པ་ཞིག་ཟུར་དུ་བཙུགས་ཏེ། ན་གཞོན་ཚོ་ནུབ་ཕྱོགས་པའི་ཤུགས་རྐྱེན་ལས་གཡོལ་ཐབས་བྱས་ཡོད། ཉམས་རྒུད།  རྒྱ་ནག་གི་འབྱོར་ལྡན་ཚོ། སྲིད་བློན་ལེ་ཁི་ཆིང་ (Li Keqiang) གིས་རྒྱ་ནག་མི་དམངས་ཚོགས་ཆེན་སྟེང་དཔལ་འབྱོར་སྙན་ཐོ་སྒྲོགས་སྦྱང་སྐབས། རྒྱ་ནག་གི་དངུལ་གྱི་འཛའ་མར་ཆག་པའི་དཀའ་ངལ་སེལ་ཆེད་དུ་མ་རྩ་ག་ཚོད་ཅིག་ཉར་ཡོད་མེད་གསལ་བཤད་བྱས་མེད་སྟབས།  མི་མང་ཚོགས་ཆེན་གྱི་ཟ་དང་ཟུར་ན་མི་མང་པོ་ཞིག་གིས་ཚོགས་ཆེན་དེ་ནི་རྒྱ་ནག་མི་མང་དམངས་གཙོའི་ཚོགས་ཆེན་རེད་དམ་ཞེས་ཤབ་ཤུབ་བྱེད་བཞིན་ཡོད། རྒྱ་ནག་གི་བཟོ་གྲྭ་མང་པོ་ཞིག་གིས་རྒྱ་ནག་གཞུང་ལ་སྐྱོན་བརྗོད་བྱེད་ཀྱིན་པའི་ཁྲོད། རྒྱ་ནག་གི་སྨན་གྱི་བཟོ་གྲྭ་ (Creat Group) ཞེས་བའི་ཚོགས་གཙོ་ཊོན་ཡི་བཱན་ (Zheng Yuewen) ལགས་ཀྱིས། རྒྱ་ནག་གཞུང་གིས་ཕྱི་འབྲེལ་ཚོང་ལས་ཐོག་ཧ་ཅང་དམ་དྲག་དང་། མ་དངུལ་ཕྱིར་གཏོང་བཀག་སྡོམ་བྱེད་ཀྱིན་ཡོད་པར་སྐྱོན་བརྗོད་བྱས་འདུག རྒྱ་ནག་གི་གློག་བརྙན་སྒྱུ་རྩལ་བཟོ་གྲྭ་ཆེ་ཤོས་ཏཱ་ལེན་ཝིན་ཌ་ (Dalian Wanda) ཡིས་ཨ་རིའི་ཧོ་ལཱི་ཝོ་ཌའི་ཁི་ཁཱེ་ལཱ་ཁི་ཐོན་རྫས་ (Hollywood’s Dick Clark Production) ཞེས་བ་ཉོ་རྩིས་བྱས་པའང་བཀག་འགོག་བྱས་ཏེ་རྒྱ་ནག་གི་ཁྲོམ་རྭའི་མ་དངུལ་ཕྱིའི་རྒྱལ་ཁབ་ཁག་ཏུ་གཏོང་གི་མེད། ཡར་ཐོན།  གློག་བཞུ་དམར་པོ་བཟུང་སྟེ་ངོ་རྒོལ་བྱེད་མཁན། དེ་ནི་ཁྱོད་དང་ང་ལ་གསལ་ཆ་ཡོད་པའི་རྒྱ་ནག་གི་ཕྱི་སྲིད་བློན་ཆེན་བཱང་ཡཱི་ (Wang Yi ) རེད། ད་ལོའི་རྒྱ་ནག་མི་དམངས་ཚོགས་ཆེན་གྱིས་དམར་ཤོག་གཞུང་དང་རྒྱལ་ཁབ་ལ་ཕྱོགས་ཞེན་དགོས་པའི་སྐུལ་སློང་བྱས་ཡོད་སྟབས། ཉེ་ཆར་རྒྱ་ནག་ནང་གི་རྒྱལ་ཞེན་ཅན་སྐོར་ཞིག་གིས་ཁོང་ཁྲོ་ཆེན་པོས་ལྷོ་ཀོ་རི་ཡ་དང་ཨ་རིའི་གཞུང་ལ་ངོ་རྒོལ་བྱས་ཡོད། ཨ་རིའི་གཞུང་གིས་ལྷོ་ཀོ་རི་ཡའི་ནང་མེ་ཤུགས་འཕུར་མདའ་སྐྱེལ་འདྲེན་བྱས་ཏེ། རྒྱ་ནག་གཞུང་གི་ཁ་གཏད་དུ་ལངས་བར་མཐའ་གཅིག་ཏུ་ངོ་རྒོལ་ཡོད་ཅེས་སྐད་འབོད་བརྒྱབ་ཡོད་པ་དང་། ངོ་རྒོལ་བ་ཚོས་ལྷོ་ཀོ་རི་ཡའི་ཐོན་རྫས་གཏོར་བཅོམ་དང་མེ་བསྲེགས་གཏོང་བའི་ལས་འགུལ་སྤེལ་ཡོད། སྤྱིར་ངོ་རྒོལ་དེའི་རྣམ་པར་བལྟས་བ་ཙམ་གྱིས་རྒྱ་ནག་གི་དམིགས་ཡུལ་གཙོ་བོ་ལྷོ་ཀོ་རི་ཡའི་ཚོང་རྫས་བཀག་སྡོམ་བྱ་རྒྱུ་དེ་ཡིན་པ་གསལ་བོ་ཡིན་ཡང་། རྒྱ་ནག་ཕྱི་སྲིད་བློན་ཆེན་གྱིས་མི་མང་ལ་ཞི་བདེ་གལ་ཆེན་ཡིན་པར་ཁ་གཡར་ནས་བཞག་ཡོད། རྒྱ་ནག་ཕྱི་སྲིད་བློན་ཆེན་གྱིས་ལྷོ་ཀོ་རི་ཡ་དང་བྱང་ཀོ་རི་ཡར་རེ་བསྐུལ་ཞུས་དོན། བྱང་ཀོ་རི་ཡས་མེ་ཤུགས་འཕུར་མདའ་ཚོད་ལྟ་དང་། ལྷོ་ཀོ་རི་ཡས་ཨ་རི་དང་མཉམ་འབྲེལ་གྱིས་མེ་ཤུགས་འཕུར་མདའ་སྐྱེལ་འདྲེན་བྱས་བ་དེས་ལྷོ་བྱང་ཀོ་རི་ཡའི་རྒྱལ་ཁབ་གཉིས་ཀྱི་འབྲེལ་ལམ་ཇེ་སྡུག་ཏུ་བཏང་བ་ལས་ཕན་ཐོགས་གང་ཡང་མེད་ཅེས་བརྗོད་ཡོད་པ་དང་། རྒྱལ་ཁབ་གཉིས་ཀྱིས་ལས་དོན་དེ་དག་ཕྱིར་འཐེན་མཚམས་འཇོག་བྱེད་དགོས་པའི་རེ་སྐུལ་ཞུས་འདུག དོན་དངོས་ཐོག་རྒྱལ་ཁབ་འདི་གཉིས་ཀྱིས་འཐབ་འཁྲུག་རྒྱག་སྲིད་དམ་ཞེས་ན། རྒྱ་ནག་གི་གློག་བཞུ་དམར་པོ་བཟུང་བའི་མི་མང་ཚོས་དམར་ཤོག་གི་དར་ཆ་དགོང་ལ་འཕྱར་ཏེ་སྐད་འབོད་བརྒྱབ་པ་དེས་གནས་སྟངས་དེ་མཚམས་འཇོག་བྱེད་དུ་བཅུག་ཡོད། ཉམས་རྒུད།   སློབ་གསོ་བའི་སྡེ། སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱིས་ལོ་སྔོན་མའི་ནང་རྒྱ་ནག་གི་སློབ་ཕྲུག་ཚོར་ཆབ་སྲིད་སློབ་གསོ་གལ་ཆེན་ཡིན་སྐོར་བརྗོད་ཡོད་མེད་ལ་མ་ལྟོས་པར། རྒྱ་ནག་གི་ཤེས་རིག་བློན་ཆེན་ཊང་པཱོ་ཧྲན་  (Chen Baosheng)གྱིས་རྒྱ་ནག་གི་ཤེས་ཡོན་གྲུབ་འབྲས་ཇེ་ཞན་དུ་སོང་ཡོད་པའི་གནས་ཚུལ་བསྒྲགས་ཡོད་པ་དེས་རྒྱ་ནག་གི་སློབ་གསོ་ཉམས་རྒུད་དུ་ཕྱིན་ཡོད་པ་བསྟན་ཡོད། “ང་ཚོས་མཐོ་སློབ་ཁག་ལ་ཞིབ་འཇུག་བྱས་སྐབས། སློབ་ཕྲུག་ཚོའི་བསམ་བློ་དང་ཆབ་སྲིད་གོ་རྟོགས་ཡག་པོ་མེད་པའི་ཁར།  དེར་ཡོད་པའི་མི་ཚོར་ལུས་ཕུང་ཡོད་པ་ཙམ་ལས་གསོན་ཤུགས་མི་འདུག་” ཅེས་ཞི་ཅིན་ཕིན་གྱིས་སྐྱོན་བརྗོད་བྱས་ཡོད། ད་དུང་ཁོས་རྙིང་ཞེན་དང་བསམ་བློ་མཁྲེགས་འཛིན་ཅན་གྱི་དགེ་རྒན་རྣམས་ལས་གནས་ནས་ཕྱིར་འཐོན་དགོས་པ་དང་། ཆབ་སྲིད་སློབ་གསོའི་མཇུག་འབྲས་ནི་ནམ་ཡང་རྒྱལ་ཁ་ལེན་མཁན་དེ་ཡིན་སྐོར། སྤྱི་ཚོགས་རིང་ལུགས་སློབ་གསོས་རྒྱ་ནག་གི་སློབ་ཕྲུག་ཚོའི་ཀུན་སྤྱོད་འཚར་ལོངས་སྐྱེད་བསྲིང་བྱེད་དགོས་ཚུལ་སོགས་བརྗོད་འདུག ང་ཚོས་རྒྱ་ནག་གི་རྒྱལ་ཞེན་སློང་ཐབས་བྱེད་པའི་བརྙན་རིས་དག་དང་། ཀླད་པ་བསྒྱུར་བཅོས་གཏོང་བྱེད་ཀྱི་ལག་ཆ་དག་ལ་བལྟས་པ་ཙམ་གྱིས་རྒྱ་ནག་གི་སློབ་ཕྲུག་ཚོའི་གནས་སྟངས་ཤེས་ཡོང་གི་རེད། ད་ལྟ་སུ་ཞིག་ཕྱི་ལ་འཕུད་ཡོད་དམ། རྒྱ་ནག་གཞུང་གིས་ལས་བྱེད་གལ་ཆེན་མང་པོ་ཞིག་ཕྱིར་འཕུད་བཏང་ཡོད་ཀྱང་། འདིར་མི་གཅིག་གི་མིང་བཏོན་པ་ཡིན་ན་དེ་ནི་པོ་ཤཱི་ལཱེ་ (Bo Xilai) རེད། རྒྱ་ནག་ནང་ཆབ་སྲིད་ཀྱི་ཤུགས་རྐྱེན་ཆེན་པོ་ཡོད་པའི་ཆབ་སྲིད་པ་པོ་ཤཱི་ལཱེ་ནི་ལོ་ངོ་ལྔའི་ཡར་སྔོན་གྱི་མི་མང་ཚོགས་ཆེན་སྐབས་སྲིད་སྟེགས་ལས་ཕབ་སྟེ། ད་ལྟ་པེ་ཅིན་བྱང་ཕྱོགས་སུ་ཡོད་པའི་བཙོན་ཁང་ཞིག་དུ་བཅུག་ཡོད། ཁོས་ལྐོག་ཟ་རུལ་སུངས་དང་སྲིད་དབང་སྒེར་གཅོད་བྱས་ཏེ། དབྱིན་ཇུའི་ཚོང་པ་ཞིག་བསད་པའི་ཉེས་མིང་འོག་ཚེ་བཙོན་ཁྲིམས་ཐག་བཅད་ཡོད། རྒྱ་ནག་གི་གྲོང་ཁྱེར་ཁྲོན་ཆིན་ (Chongqing) གྱི་དམར་ཤོག་ཚོགས་པའི་དྲུང་ཆེ་ཟུར་བ་པོ་ཤཱི་ལཱེ་ཡིས་རྒྱ་ནག་གི་སྲིད་སྟེགས་གལ་ཆེན་ཞིག་བཟུང་ཡོད་ཀྱང་། ལྐོག་ཟ་རུལ་སུངས་ལ་ཁ་གཡར་ཏེ་བཙོན་འཇུག་བྱས་བ་དེས་ཞི་ཅིན་ཕིན་ལ་ང་ཚོས་ད་ལྟ་མཐོང་བཞིན་པའི་ཆབ་སྲིད་གར་སྟེགས་ཆེན་པོ་འདི་ལེགས་འགྲུབ་བྱུང་ཡོད་པ་རེད།  ད་ལྟ་ཡོད་ཚད་ལ་བསྐྱར་དུ་མཇུག་སྡོམ་ཞིག་རྒྱབ་ཚེ་ཞི་ཅིན་ཕིན་ཡར་ཐོན་ཅན་གྱི་སྲིད་འཛིན་དེ་ཆགས་ཡོད། སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱིས་མི་དམངས་འཐུས་མིའི་ཚོགས་ཆེན་སྟེང་སྐད་ཆ་འཇམ་པོ་ཞིག་བཤད་དེ།  ཏང་གི་ཚོགས་ཆེན་བཅུ་དགུ་པ་དེ་འབྱུང་འགྱུར་གྱི་སྟོན་དུས་སུ་འཚོགས་འཆར་ཡོད་སྟབས། ཁོས་བསྐྱར་དུ་ཐེངས་གཉིས་བའི་སྲིད་འཛིན་རྐུབ་སྟེགས་རྩོད་ལེན་བྱེད་པར་གྲ་སྒྲིག་བྱེད་ཀྱིན་ཡོད་པ་དང་།  དེས་ཁོའི་ཁ་གཏད་ཀྱི་ཆབ་སྲིད་པ་ཚོར་ཞེ་སྣང་སློང་གི་ཡོད།
