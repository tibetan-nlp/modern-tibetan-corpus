སྦྲང་རྩི་བླངས་ཏེ་རྩྭ་ཐང་གི་བཅུད་འཕྲོག

རྒྱ་མི་སྤྲང་རྩི་ལེན་མཁན་ཚོས་སྤང་ཐང་མེ་ཏོག་གིས་བརྒྱན་པའི་འབྲོག་ས་དག་ནས་སྦྲང་རྩི་བླངས་ཏེ་སྐྱེ་ཁམས་ཁོར་ཡུག་ལ་གནོད་སྐྱོན་ཆེན་པོ་ཐེབས་ཀྱི་ཡོད་འདུག བོད་ནས་འདི་གར་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། འདས་པའི་ལོ་མང་རིང་ནས་བཟུང་དབྱར་དུས་བསླེབས་པ་དང་བསྟུན། ཡུལ་སྐོར་སྤྲོ་འཆམ་པ་མང་པོ་ཡོང་སའི་འབྲོག་ཁུལ་མཛོད་དགེ་རྫོང་དང་ཁྱུང་མཆུ་རྗོང་། རྔ་བ་རྫོང་སོགས་ཀྱི་འབྲོག་པ་རྣམས་རུ་བ་དབྱར་སར་ཕྱིན་རྗེས། རྒྱ་མི་ཚོས་ས་ཆ་ལ་རྒྱ་སྒོར་ ༡༠༠༠ རེ་སྤྲད་ནས་དགུན་སའི་རྩྭ་ཐང་ཐོག་སྦྲང་མ་ཁྲི་སྟོང་མང་པོ་བཀྲམ་སྟེ། རྩི་ཤིང་དང་མེ་ཏོག་གི་རྩི་བཅུད་ལེན་གྱི་ཡོད་སྟབས། ས་ཆ་དེ་དག་གི་གསར་བཞད་མེ་ཏོག་རྣམས་ཡུན་རིང་གནས་མི་ཐུབ་པར་རྙིད་འགྲོ་བ་མ་ཟད། སྔོ་ལྗང་ལྗང་གི་རྩྭ་ཐང་ཡང་མགྱོགས་པོར་སྐྱ་ཟད་སྔོ་ཟད་དུ་གྱུར་ཏེ། རྩྭ་ཐང་གི་སྐྱེ་ཁམས་ཁོར་ཡུག་ལ་གནོད་སྐྱོན་ཆེན་པོ་ཐེབས་ཀྱིན་ཡོད་འདུག གནས་ཚུལ་མཁོ་སྤྲོད་པས། རྒྱ་མི་སྦྲང་རྩི་ལེན་མཁན་དེ་ཚོས་རྩྭ་སར་རྒྱ་སྒོར་ཆིག་སྟོང་རེ་སྤྲད་ནས་རྩྭ་ཐང་དུ་སྒམ་ཆུང་བརྒྱ་ཕྲག་བརྒལ་བ་སྒྲིག་སྟེ་སྦྲང་རྩི་བསྡུ་བཞིན་ཡོད་པ་མ་ཟད། ལམ་འགྲམ་ནས་ས་གནས་བོད་མི་ཚོ་དང་ཡུལ་སྐོར་བ་ཚོར་སྦྲང་རྩི་འཚོང་བཞིན་ཡོད་པ་དང་། རིན་གོང་མཐོ་དམན་གསུམ་ཡོད་པ་ལས་རིན་གོང་ཆེ་ཤོས་དེར་རྒྱ་སྒོར་ལྔ་བརྒྱ་ནས་དྲུག་བརྒྱ་དང་། རིན་གོང་ཆུང་ཤོས་ཀྱི་སྦྲང་རྩི་དམ་བུ་གང་ལ་རྒྱ་སྒོར་ ༣༥ ཙམ་དགོས་ཀྱི་འདུག རིན་གོང་ཆེ་ཤོས་དེ་འབུ་དཀར་པོ་ཞིག་ཡིན་པ་དང་། དེ་ཨ་རག་ནང་དུ་བླུགས་ནས་འཐུང་ན་མིའི་ལུས་ཀྱི་འཚོ་བཅུད་རྒྱས་པར་ཕན་ཐོགས་ཆེན་པོ་ཡོད་སྐོར་འགྲེལ་བཤད་རྒྱག་གི་འདུག ཅེས་དང་། ཁོང་གིས་མུ་མཐུད། རྩྭ་ཐང་དང་ཞིང་ཁའི་གསར་བཞད་མེ་ཏོག་དང་རྩི་ཏོག་སོགས་ཉིན་གཉིས་གསུམ་རིང་ལ་མ་གཏོགས་མི་འདུག་པར་སྐམ་རྙིད་དུ་འགྱུར་འགྲོ་བ་དང་། སྔོ་ལྗང་རྩྭ་ཐང་གི་ཁ་མདོག་ཀྱང་རིམ་བཞིན་ཤོར་འགྲོ་གི་འདུག གང་ལྟར་སྦྲང་རྩི་བླངས་བ་འདིས་སྐྱེ་ཁམས་ཁོར་ཡུག་ལ་གནོད་འཚེ་ཚབས་ཆེན་བཟོ་ཡི་འདུག ང་རང་གི་བསམ་ཚུལ་ཡིན་ན། ང་ཚོས་འཕྲལ་གྱི་ཁེ་བཟང་ཆུང་ཆུང་ཞིག་གི་ཆེད་དུ་ཕུགས་སུ་ཕམ་ཁ་ཆེན་པོ་ཞིག་ཉོ་ཡི་འདུག གཅིག་ནས་རྒྱ་མི་སྦྲང་རྩི་མཁན་དེ་ཚོས་མ་དངུལ་ཕྲན་ཙམ་ཡང་གཏོང་མི་དགོས་པར་གཞན་གྱི་ས་ཐོག་ནས་ཁེ་སྤོགས་ཆེན་པོ་གཉེར་གྱི་ཡོད་པ་དང་། གཉིས་ནས་ང་རང་ཚོའི་རྩྭ་ཐང་གཙོས་རང་བྱུང་ཁོར་ཡུག་གི་འཚོ་བཅུད་ཉིན་རེ་བཞིན་ཉམས་འགྲོ་ཡི་འདུག ང་ཚོས་རྒྱ་སྒོར་ཆིག་སྟོང་ཙམ་གྱི་ཁེ་བཟང་ལ་མི་ལྟ་བར་རྩི་ཤིང་མེ་ཏོག་ལ་ཐེབས་བཞིན་པའི་གནོད་སྐྱོན་འདི་རིགས་སྔོན་འགོག་བྱས་ཏེ། རང་བྱུང་ཁོར་ཡུར་གི་འཚོ་བཅུད་ལ་སྲུང་སྐྱོབ་བྱེད་དགོས་པ་རེད་བསམ་གྱི་འདུག ཅེས་བརྗོད་བྱུང་། ཡང་ཁོང་གིས། ང་ཚོའི་ས་ཆ་འདིར་རྒྱ་མི་སྦྲང་མ་གསོ་མཁན་སྐོར་ཁག་གཉིས་ཡོད་པ་དང་། ཁོ་ཚོ་ས་ཆ་འདིའི་ཡུལ་མི་ཚོའི་ངོ་ཤེས་རེད་འདུག ཁོ་ཚོ་ལོ་རེ་བཞིན་ས་ཆ་འདིར་ཡོང་ནས་ས་རིན་རྒྱ་སྒོར་ཆིག་སྟོང་རེ་སྤྲད་ནས་སྡོད་ཀྱི་འདུག ཉེ་སྔོན་ཡང་ང་ཚོའི་རྫོང་ཐོག་ཏུ་རྒྱ་མི་སྦྲང་རྩི་ལེན་མཁན་སྐོར་ཁག་གཅིག་བསླེབས་སོང་ཡང་ཡུལ་མི་ཚོས་འདུག་ཏུ་བཅུག་མ་སོང་། དེ་མིན་ཁྱུང་མཆུ་རྫོང་ནས་ངའི་ཕ་ཡུལ་བར་དུ་སྤྱི་ལེ་ ༡༠༠ ལྷག་ཡོད་པ་དང་། དེའི་བར་ཁོ་ནར་སྦྲང་རྩི་ལེན་མཁན་སྐོར་ཁག་ཉི་ཤུ་ཡས་མས་ཤིག་ཡོད་ཅིང་། སྐོར་ཁག་ཆུང་ཆུང་རེ་ལ་སྒམ་ཆུང་ ༡༠༠ ཙམ་དང་། སྐོར་ཁག་ཆེན་པོ་རེར་སྒམ་ཆུང་ཉིས་བརྒྱ་སུམ་བརྒྱ་ཡོད་ས་རེད། ཅེས་བརྗོད་བྱུང་།
