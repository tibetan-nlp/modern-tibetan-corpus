ཡང་དགེ་རྒན་ ༧༠༠ བོད་དུ་གཏོང་རྒྱུ་རེད།

འདས་ལོར་རྒྱ་ནག་ནང་ལོགས་ནས་བོད་རང་སྐྱོང་ལྗོངས་ཁོངས་སུ་དགེ་རྒན་ ༧༠༠ བཏང་ཡོད་པ་དང་། ད་ལོར་སི་ཁྲོན་གྱི་བོད་རིགས་ས་ཁུལ་ཁག་ཏུ་དགེ་རྒན་ ༧༠༠ གཏོང་རྒྱུ་རེད། དེ་ཡང་བོད་ནང་གི་བརྒྱུད་ལམ་ཁག་ཏུ་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། འདི་ལོའི་སི་ཁྲོན་ཞིང་ཆེན་གྱི་སློབ་གསོའི་ལས་དོན་ཚོགས་འདུའི་ཐོག བོད་རིགས་ས་ཁུལ་གྱི་མི་སྟོང་གི་སློབ་སྐྱོར་འཆར་གཞི་ཟེར་བ་ཞིག་གཏན་འབེབས་བྱས་པའི་ནང། རྒྱ་ནག་ནས་སི་ཁྲོན་བོད་རིགས་ས་ཁུལ་གྱི་རྫོང་ཁག་ ༣༢ ནང་ཕུལ་བྱུང་དགེ་རྒན་ ༧༠༠ འདེམས་གཏོང་གིས་སློབ་སྐྱོར་བྱ་རྒྱུ་ཡིན་པ་དང་ཆབས་ཅིག བོད་ཁུལ་ནས་དགེ་རྒན་ ༣༠༠ རྒྱ་ནག་ནང་སློབ་སྦྱོང་ཆེད་གཏོང་དགོས་སྐོར་དང་། ད་ལོར་རྒྱ་ནག་ནོར་སྲིད་ཁང་ནས་གཞུང་བཙུགས་མཐོ་རིམ་ལས་རིགས་སློབ་གྲྭ་དང་། མཐོ་རིམ་ཆེད་སྦྱོང་སློབ་གྲྭའི་སློབ་མར་ཆ་སྙོམ་དངུལ་སྤྲོད་ཚད་གཞི་རྒྱ་སྒོར་ ༡༢༠༠༠ ལོན་པ་དང་། དེ་བཞིན་གཞུང་བཙུགས་འབྲིང་རིམ་ལས་རིགས་སློབ་གྲྭའི་སློབ་མར་ཆ་སྙོམ་དངུལ་སྤྲོད་ཚད་གཞི་རྒྱ་སྒོར་ ༦༠༠༠ ལོན་པར་འགན་ལེན་བྱ་རྒྱུ་ཡིན་ལ། སི་ཁྲོན་ཞིང་ཆེན་ཡོངས་ཀྱི་ནང་ཚད་གཞིར་མ་བསླེབ་པའི་རང་སྐྱོང་ཁུལ་དང་རྫོང་གྲོང་ཁྱེར་གྲོང་རྡལ་ཁག་ལ་དག་ཐེར་དོན་འཁྱོལ་ཡོང་བར་སྐུལ་འདེད་དང་སྦྲགས། མི་རིགས་རང་སྐྱོང་རྫོང་ཚང་མའི་ནང་གྲོང་ཚོ་གཅིག་ལ་བུ་བཅོལ་ཁང་གཅིག་འཛུགས་པའི་འཆར་གཞི་རྒྱ་བསྐྱེད་དགོས་པ་སོགས་ཁུལ་ཆེན་པོ་བཞི་ཡི་སློབ་གསོའི་ཞབས་ཞུའི་ཆུ་ཚད་ཞིང་ཆེན་གྱི་ཆུ་ཚད་དང་གཅིག་མཚུངས་ཡོང་བར་སྐུལ་འདེད་གཏོང་དགོས་སྐོར་བརྗོད་འདུག རྒྱ་ནག་ནང་ལོགས་ནས་དགེ་རྒན་ཕོན་ཆེད་བོད་ཁུལ་གྱི་སློབ་གྲྭ་ཁག་ནང་ཆེད་དུ་གཏོང་བའི་ལས་གཞི་དེ་ད་ལོར་གསར་དུ་མགོ་བཙུགས་པ་ཞིག་མ་ཡིན་པར་འདས་ལོའི་ཕྱི་ཟླ་ ༤ པའི་ནང་མགོ་བཙུགས་ཏེ། རྒྱ་ནག་པེ་ཅིན་སོགས་གྲོང་ཁྱེར་དང་ཞིང་ཆེན་ཁག་ ༡༧ ནས་དགེ་རྒན་ ༧༠༠ བོད་རང་སྐྱོང་ལྗོངས་ནང་ཆེད་གཏོང་གིས། བོད་ཀྱི་རྒྱལ་ས་ལྷ་སའི་གཙོས་གཞིས་ཀ་རྩེ་དང་ཉིང་ཁྲི། ལྷོ་ཁ། ཆབ་མདོ་ས་ཁུལ་གྱི་སློབ་གྲྭ་ཁག་ ༢༠ ནང་ལས་ཞུགས་བྱས་ཡོད་པ་དང་སྦྲགས། ལས་གཞི་དེའི་འོག་ལོ་ལྟར་བོད་ཁུལ་གྱི་གཞི་རིམ་སློབ་གྲྭ་ཁག་ནས་ཁྱོན་དགེ་རྒན་ ༤༠༠ རེ་རྒྱ་ནག་ནང་ལོགས་ཀྱི་འཕེལ་རྒྱས་ཆེ་བའི་ས་ཁུལ་ཁག་ཏུ་འགྲོ་དགོས་པ་བཟོས་ཏེ། དཀྱིལ་སྙིང་དམ་དུ་བཟུང་བའི་ཐོག་ནས་སློབ་གསོའི་བྱ་བ་སྤེལ་རེས་དང་སློབ་གསོའི་བྱ་བ་གོང་དུ་སྤེལ་དགོས་སྐོར་བརྗོད་ཀྱི་ཡོད་པ་རེད། ཡིན་ནའང་། བོད་ཁུལ་གྱི་སློབ་ཐོན་པ་མང་པོར་བྱ་བ་རག་དཀའ་བ་དང་། སློབ་གྲྭ་ཁག་ནང་དངོས་ཤུགས་བརྒྱུད་གསུམ་ནས་རྒྱ་ཡིག་རྒྱ་སྐད་སློབ་སྦྱོང་བྱེད་པར་ཤུགས་གནོན་རྒྱག་གི་ཡོད་པ་དེས་བོད་མི་ཚོར་སེམས་འཚབ་དང་དཀའ་ངལ་མང་དག་ཅིག་བཟོ་ཡི་ཡོད་པ་དཔེར་ན། འདས་ལོའི་ཕྱི་ཟླ་ ༧ པའི་ནང་རྨ་ལྷོ་བོད་རིགས་རང་སྐྱོང་ཁུལ་གཅན་ཚ་རྫོང་མི་རིགས་སློབ་འབྲིང་གཉིས་པའི་བོད་རིགས་སློབ་མ་ཡོངས་ནས་སློབ་གྲྭར་སྙན་ཞུ་ཕུལ་ཏེ། ཉེ་བའི་ལོ་ཤས་ནས་བཟུང་མི་རིགས་སློབ་འབྲིང་དེའི་ནང་རྒྱ་རིགས་དགེ་རྒན་ཇེ་མང་ ནས་ཇེ་མང་དུ་སོང་སྟེ་མི་རིགས་སློབ་འབྲིང་ཞེས་པ་མིང་གི་ལྷག་མ་ཙམ་དུ་གཏོང་ ཉེན་ཆེ་བས། མི་རིགས་རང་སྐྱོང་ས་ཁུལ་དུ་མི་རིགས་རང་གི་སྐད་དང་ཡི་གེ་བཀོལ་སྤྱོད་དགོས་ པའི་རྒྱལ་ཁབ་ཀྱི་གཏན་འབེབས་སྲིད་བྱུས་ལག་བསྟར་བྱེད་དགོས་པ་མ་ཟད། ད་ཡོད་ཀྱི་རྒྱ་རིགས་དགེ་རྒན་ཚོར་ཆེད་དུ་འཛིན་གྲྭ་བཙུགས་ཏེ་བོད་ཡིག་བོད་སྐད་ སྦྱོང་དུ་འཇུག་དགོས་པའི་རེ་བ་བཏོན་ཡོད་པ་རེད། དེ་བཞིན་འདས་ལོར་རྒྱ་གར་ནང་གསར་དུ་འབྱོར་བའི་སི་ཁྲོན་གྱི་བོད་རིགས་གཞོན་ནུ་ཞིག་གིས་འདི་གའི་གསར་འགོད་པར། ཁོང་ཚོའི་ཕ་ཡུལ་གྱི་སློབ་གྲྭ་ཁག་གི་དགེ་རྒན་མང་ཆེ་བ་རྒྱ་རིགས་ཡིན་པ་དང་། རྒྱ་རིགས་དགེ་རྒན་ཚོས་བོད་སྐད་ཤེས་ཀྱི་མེད་སྟབས། དང་ཐོག་ཞིང་འབྲོག་ཁུལ་གྱི་བྱིས་པ་ཚོ་དང་ཕན་ཚུན་བརྡ་འཕྲོད་ཁག་པོ་ཡོད་ཀྱང་། སློབ་གྲྭའི་ནང་ངེས་པར་དུ་རྒྱ་ཡིག་རྒྱ་སྐད་སྦྱོང་དགོས་པ་དང་། ཁོར་ཡུག་གིས་རྒྱ་ཡིག་རྒྱ་སྐད་སྦྱོང་རྒྱུ་ཏོག་ཙམ་ལས་སླ་པོའང་ཡོད་སྐོར་བརྗོད་བྱུང་།   གསར་འགོད་པ། པདྨ་མཚོ།
