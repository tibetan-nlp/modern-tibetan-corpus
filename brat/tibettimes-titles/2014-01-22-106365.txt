ལོ་སར་སྐབས་རོཊ་དངུལ་སྤྲོད་འཆར།

མཚོ་སྔོན་ཞིང་ཆེན་སྲིད་གཞུང་གིས་སྔར་ལོ་སྒོར་དུང་ཕྱུར་ ༡༥༤།༥ བཏང་སྟེ་རོང་འབྲོག་མི་མང་ལ་རོགས་སྐྱོར་བྱས་ཡོད་པ་དང་། འདི་ལོའི་ལོ་སར་སྐབས་མི་མང་གི་གཞི་རྩའི་འཚོ་བར་འགན་སྲུང་ཁ་གསབ་མ་དངུལ་སྒོར་ཁྲི་ ༧༢༧༣༣ སྤྲོད་འཆར་ཡོད་འདུག ཉེ་བའི་ལོ་འགའི་རིང་རྒྱ་གཞུང་གིས་བོད་ནང་གི་ས་ཆ་གང་སར་རོགས་དངུལ་མང་དུ་བཏང་སྟེ། མི་མང་གི་བསམ་བློ་བསྒྱུར་ཐབས་དང་། རྒྱལ་སྤྱིའི་སྤྱི་ཚོགས་ལ་བོད་ནང་གི་དཀའ་ངལ་འགེབ་ཐབས་ཀྱི་ཐབས་ལམ་འདྲ་མིན་ལག་བསྟར་བྱེད་ཀྱིན་པ་ལྟར། ཉེ་ལམ་མཚོ་སྔོན་ཞིང་ཆེན་སྲིད་གཞུང་གིས། ༢༠༡༣ ལོར་རོང་འབྲོག་མང་ཚོགས་ལ་རོགས་དངུལ་ག་ཚོད་ཕུལ་མིན་དང་། ས་གནས་ཀྱི་གཞུང་ལམ་དང་གློག སྡོད་ཁང་གི་མཐུན་རྐྱེན་ཇེ་ལེགས་སུ་བཏང་བའི་ཐད་གནས་ཚུལ་ཞིག་བཀོད་ཡོད་པའི་ནང་། ཕྱི་ལོ་ ༢༠༡༣ ལོར་མཚོ་སྔོན་ཞིང་ཆེན་སྲིད་གཞུང་གིས་སྒོར་དུང་ཕྱུར་ ༡༥༤།༥ བཏང་སྟེ་རོང་འབྲོག་ཁུལ་དུ་གཞུང་ལམ་སྤྱི་ལེ་ ༩༩༧༠ གསར་འཛུཊ་དང་། ཟམ་པ་ ༢༢༣ འཛུགས་སྐྲུན། སྡེ་བ་ ༡༠༧ ལ་ཁ་པར་གྱི་མཐུན་རྐྱེན་བསྐྲུན་ཡོད་ཚུལ་དང་། རོང་འབྲོག་ཁུལ་གྱི་ཁྱིམ་ཚང་ཁྲི་ ༣།༩༢ ཡི་གློག་གི་དཀའ་ངལ་སེལ་ཡོད་ཚུལ་སོགས་བཀོད་ཡོད་པ་མ་ཟད། དབུལ་སྐྱོར་གྱི་སྲིད་ཇུས་ལག་བསྟར་བྱས་ཏེ་སྡེ་བ་ ༣༧༦ དག་ཐེར་ཡར་སྐུལ་བྱས་ཡོད་པ་དང་། དབུལ་ཕོངས་རོང་འབྲོག་ ༥༤༠༠༠ ལྷག་དབུལ་སྐྱོར་གནས་སྤོར། སྒོར་དུང་ཕྱུར་ ༢།༡ བཏང་སྟེ་ཆུ་དང་གཞུང་ལམ་སོགས་གསར་བསྐྲུན་བཅས་བྱས་ཏེ། དབུལ་ཕོངས་ ༢༡༣༠༠༠ ཇེ་ཉུང་དུ་བཏང་ཚུལ་སོགས་བཀོད་ཡོད་འདུག་ལ། ༢༠༡༤ ལོའི་སྐོར་དང་པོའི་དཀའ་ངལ་མང་ཚོགས་ཀྱི་གཞི་རྩའི་འཚོ་བར་འགན་སྲུང་ཁ་གསབ་མ་དངུལ་སྒོར་ཁྲི་ ༧༢༧༣༣ ཡོད་པ་ལས། གྲོང་ཁྱེར་གྱི་མ་མཐའི་འཚོ་བའི་འགན་སྲུང་ཁ་གསབ་མ་དངུལ་སྒོར་ཁྲི་ ༣༥༤༥༥ གཏོང་འཆར་ཡོད་པ་དང་། ཞིང་སྡེའི་མ་མཐའི་འཚོ་བའི་འགན་སྲུང་ཁ་གསབ་མ་དངུལ་སྒོར་ཁྲི་ ༣༧༢༧༨ གཏོང་རྒྱུའི་གཏན་འབེབས་བྱས་ཏེ། ལོ་སར་སྐབས་མ་དངུལ་བགོ་འགྲེམས་བྱ་འཆར་ཡོད་ཚུལ་སོཊ་བཀོད་འདུག དེ་བཞིན། ཟླ་འདིའི་ཚེས་ ༡༩ ཉིན་མཚོ་སྔོན་ཞིང་ཆེན་དམངས་འཐུས་ཚོགས་ཆེན་སྐབས་བཅུ་གཉིས་པའི་གྲོས་ཚོགས་ཐེངས་གསུམ་པའི་ཐོག ཞིང་ཆེན་གྱི་དམར་ཤོག་མགོ་གཙོ་ཧུའུ་ཕིན་གྱིས་སྲིད་གཞུང་བྱ་བའི་སྙན་ཞུ་སྒྲོགས་སྦྱོང་གནང་སྟེ། ཉེ་བའི་ལོ་འགའི་རིང་དམངས་འཚོའི་ཐད་མཚོ་སྔོན་གྱི་སྤྱིའི་འགྲོ་སོང་གི་བརྒྱ་ཆ་ ༧༥ ཡན་ཟིན་གྱི་ཡོད་པ་དང་། ༢༠༡༣ ལོར་འགྲོ་སོང་སྒོར་ཐེར་ཆེན་དགུ་དང་ཐེར་འབུམ་བཞི་ (༩༤།༠༠༠།༠༠༠།༠༠༠) ལྷག་བཏང་བ་ལས། དམངས་འཚོའི་དོན་དངོས་བཅུའི་ཐད་སྒོར་ཐེར་ཆེན་བཞི་དང་ཐེར་འབུམ་གཅིག་དུང་ཕྱུར་ལྔ (༤༡།༥༠༠།༠༠༠།༠༠༠) ལྷག་བཏང་སྟེ་ལག་བསྟར་བྱས་ཡོད་ཚུལ་བརྗོད་འདུག ཁོས་བསྒྲགས་པའི་དམངས་འཚོའི་དོན་དངོས་བཅུ་ནི། དང་པོ། ལས་ཞུགས་ཡོང་སྒོ་འཕར་ཚད་ཀྱིས་ལོ་རྒྱུས་ཀྱི་ཆུ་ཚད་གསར་བ་བསྐྲུན་ཡོད་པ་དང་། གཉིས་པ། གྲོང་གསེབ་ཀྱི་བདེ་སྡོད་བཟོ་སྐྲུན་ཕྱོགས་ཡོངས་ནས་ལག་བསྟར་བྱས་པ། གསུམ་པ། སྤྱི་ཚོགས་འགན་སྲུང་ཆུ་ཚད་རྒྱུན་མཐུད་སྒོས་འཕེལ་རྒྱས་སུ་བཏང་བ། བཞི་བ། སློབ་གསོའི་ལས་དོན་མཐུན་སྦྱོར་དང་དོ་མཉམ་སྒོས་འཕེལ་རྒྱས་སུ་བཏང་བ། ལྔ་བ། སྨན་བཅོས་འཕྲོད་བསྟེན་སྒྲིག་སྲོལ་བཅོས་སྒྱུར་གཏིང་ཟབ་ཏུ་སྤེལ་བ། དྲུག་པ། རིག་གནས་དམངས་ཕན་སྲིད་ཇུས་མུ་མཐུད་ནས་གཏིང་ཟབ་ཏུ་སྤེལ་བ། བདུན་པ། ཚན་རྩལ་གསར་སྐྲུན་ཤུགས་ཚད་ཇེ་ཆེར་བཏང་བ། བརྒྱད་པ། དབུལ་སྐྱོར་བྱ་བའི་ཐད་ནས་ཤུགས་སྣོན། དགུ་བ། རོང་འབྲོག་ཁུལ་གྱི་ཐོན་སྐྱེད་འཚོ་བའི་ཆ་རྐྱེན་ལེགས་བཅོས། བཅུ་བ། གྲོང་གསེབ་སྡོད་དམངས་ཀྱི་ཁོར་ཡུག་ལེགས་བཅོས་བཅས་ཡིན་འདུག གསར་འགོད་པ། དོན་གོ
