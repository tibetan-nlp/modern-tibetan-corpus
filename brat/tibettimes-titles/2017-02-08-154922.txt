རྒྱ་ནག་གཞུང་གིས་མཚེ་མ་ ༡༥༠༠
ཙམ་ཁ་བྲལ་དུ་བཏང་།

རྒྱ་ནག་གཞུང་གིས་འཆར་ལྡན་བུ་སྐྱེས་ཀྱི་སྲིད་བྱུས་གདུག་རྩུབ་ཆེན་པོས་ལག་བསྟར་བྱས་རྐྱེན་ཉུང་མཐར་ཕྲུ་གུ་འབུམ་གཅིག་བདག་མེད་དྭ་ཕྲུག་ཏུ་གྱུར་ཡོད་པ་དང་། དེའི་ཁྲོད་དུ་ཕྲུ་གུ་མཚེ་མ་ ༡༥༠༠ ཙམ་ཁ་བྲལ་དུ་བཏང་ཡོད་པ་རེད་ཟེར། ཏ་ཅི་ཡན་རྒྱ་ཡིག་དྲ་བའི་ཐོག་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། ཉེ་སྔོན་ལོན་ཏུན་ཧི་ཊི་རོ་གནམ་ཐང་དུ། རང་ལོ་ ༡༡ ཡིན་པའི་བུ་མོ་ཨི་ཝི་ཧན་ལོན་མུ་རི་སི་ (Evie Hanlon-Moores)ལགས་རང་གི་མ་ཚབ་ཀྱི་མཁྲེས་སུ་ལངས་ཏེ། ལོ་གཅིག་རིང་མཐོང་མ་ཐུབ་པའི་ཨོ་སི་ཊོ་ལི་ཡ་ནས་ཡོང་བའི་བུ་མོ་ཨི་ཝཱ་ཆི་ཡཱ་(Eva Chia) བསྒུགས་ནས་བསྡད་ཡོད་པ་དང་། བུ་མོ་ཆུང་ཆུང་གཉིས་ཀྱིས་གནམ་ཐང་ནས་ཕན་ཚུན་མཐོང་མ་ཐག་ཡུན་རིང་འགྲོགས་འདྲིས་བྱས་མྱོང་བ་ལྟར་དགའ་སྤྲོ་ཆེན་པོས་ཁེངས་ཡོད་པ་རེད། དེ་ཡང་ལོ་ངོ་བཅུའི་ཡར་སྔོན་ཞིག་ཏུ། ཨོ་སི་ཊོ་ལི་ཡ་ནས་ཡིན་པའི་བུད་མེད་ཌི་ཁུར་ལན་ཌ་(Dee Cridland)ཞུ་བ་ཞིག་གིས་གསོས་ཕྲུག་བུ་མོ་(Eva Chia)རྒྱ་ནག་ནས་འཁྲིད་བསླེབས་སྐབས། དབྱིན་ཇིའི་རྒྱལ་ས་ལོན་ཏོན་གྱི་ཉེ་འཁྲིས་བྷ་ཊ་( Bath)གྲོང་ཁྱེར་ནང ་རྟེན་གཞི་བྱས་པའི་བུད་མེད་ཇོ་ཧན་ལིན་མུ་རི་སི་(Jo Hanlen-Moores)ཞུ་བ་ཞིག་གིས། གསོས་ཕྲུག་བདག་ཉར་བའི་དྲ་ཐོག་ཏུ་བུ་མོ་ཆུང་ཆུང་ཞིག་གི་འདྲ་པར་བཞག་ཡོད་པ་མཐོང་བ་དང་། པར་ནང་དུ་ཡོད་པའི་བུ་མོ་དེ་མོ་རང་གིས་གསོས་པའི་བུ་མོ་ཆུང་ཆུང་དེ་ཡིན་པའི་དོགས་པ་ཡ་མཚར་ཞིག་བྱུང་། དེ་མ་ཐག་མོས་བུད་མེད་དེར་རང་གི་བུ་མོའི་འདྲ་པར་དང་བཅས་འཕྲིན་ཡིག་ཅིག་བཏང་ཡོད་པ་རེད། ཇོ་ཧན་ལིན་མུ་རི་སི་ལགས་ཀྱིས་འཕྲིན་ཡིག་ནང་གི་འདྲ་པར་མཐོང་སྐབས་བུ་མོ་གཉིས་ཀྱི་སྐྱེས་གཟུགས་གཅིག་པ་གཅིག་རྐྱང་ཡིན་པ་དེར་ཧ་ལས་པ་དང་། སྐབས་དེར་མོ་རང་གིས་བུ་མོ་ཨི་ཝི་ཧན་ལོན་མུ་རི་སི་གསོས་ཕྲུག་ཏུ་བླངས་ནས་ཟླ་ ༣ ལྷག་འགྲོ་བཞིན་ཡོད་པ་རེད། ཕྱིས་སུ་མ་ཚབ་བུད་མེད་དེ་གཉིས་ཀྱིས་བུ་མོ་གཉིས་ཀྱི་རིགས་རྒྱུད་འཕེལ་རིམ་གྱི་(DNA Test)བརྟག་དཔྱད་བྱས་པ་བརྒྱུད། ལེ་བར་ཁྲི་ཕྲག་གིས་བར་ཐག་བཅད་པའི་བུ་མོ་ཆུང་ཆུང་དེ་གཉིས་མཚེ་མ་ཡིན་པ་ཤེས་རྟོགས་བྱུང་ཡོད་པ་དང་། རྟེན་འབྲེལ་མ་བསྒྲིགས་རང་འགྲིག་གིས་བུ་མོ་གཉིས་ཀྱི་མིང་ཡང་ཨི་ཝི་ཧན་ལོན་མུ་རི་སི་(Evie Hanlon-Moores)དང་། ཨི་ཝཱ་ཆི་ཡཱ་(Eva Chia )ཞེས་འདྲ་པོ་ཡོད་པ་རེད། དེ་ནས་བཟུང་ཁྱིམ་ཚང་གཉིས་ཀྱིས་འབྲེལ་བ་མུ་མཐུད་དུ་ཉར་བ་མ་ཟད། བུ་མོ་གཉིས་ཡང་ཡང་ཐུག་འཕྲད་བྱེད་དུ་འཇུག་རྒྱུའི་ཐག་བཅད་དེ། བུ་མོ་ཨི་ཝི་དང་ཨི་ཝཱ་གཉིས་ལོ་ ༣ དང་ཕྱེད་ཀར་ལོན་སྐབས་ཐེངས་དང་པོར་གཞན་ཡུལ་ནས་ལྷན་འཛོམས་བྱུང་ཡོད་པ་དང་། ཉེ་སྔོན་གྱི་ཐུག་འཕྲད་དེ་ཐེངས་ ༤ པ་ཡིན་འདུག གནས་ཚུལ་དེའི་ནང་དུས་རབས་ ༢༠ པའི་ལོ་རབས་ ༧༠ མཇུག་ཏུ། རྒྱ་ནག་གཞུང་གིས་འཆར་ལྡན་བུ་སྐྱེས་ཞེས་བྱིས་པ་གཅིག་གི་ལམ་ལུགས་གདུག་རྩུབ་ཆེན་པོས་ལག་བསྟར་བྱས་པའི་རྐྱེན་གྱིས་ཉུང་མཐར་རྒྱ་ནག་གུ་ཕྲུ་གུ་འབུམ་གཅིག་ལྷག་བདག་མེད་དྭ་ཕྲུག་ཏུ་གྱུར་ཡོད་པ་མ་ཟད། བདག་མེད་དུ་གྱུར་བའི་ཕྲུ་གུ་མང་ཆེ་བ་ཞིག་བུ་མོ་ཡིན་འདུག ལོ་རབས་ ༩༠ པའི་མགོ་ཙམ་ཞིག་ནས་རྒྱ་ནག་ནང་གི་དྭ་ཕྲུག་གཉོར་སྐྱོང་ཁང་དེ་ཚོ་བདག་མེད་ཕྲུ་གུས་ཁེངས་ཏེ་ལྟ་སྐྱོང་མ་ཐུབ་པའི་དཀའ་ངལ་འོག ཕྱིའི་རྒྱལ་ཁབ་ཁག་ནས་རྒྱ་ནག་གི་བདག་མེད་བྱིས་པ་གསོས་ཕྲུག་ཏུ་ལེན་ཆོག་པ་བཟོས་པར་བརྟེན། ཨ་རི་དང་ཁ་ན་ཌ། ཡུ་རོབ་སོགས་ཕྱི་རྒྱལ་གྱི་ཡུལ་ལུང་ཁག་ནས་རྒྱ་ནག་གི་བདག་མེད་དྭ་ཕྲུག་མང་པོ་གསོས་ཕྲུག་ཏུ་བླངས་ཡོད་པ་རེད། ཞིབ་འཇུག་པས་ཚོད་རྩིས་བྱས་ན། རྒྱ་ནག་གཞུང་གིས་བདག་མེད་དྭ་ཕྲུག་ཏུ་གྱུར་བའི་མཚེ་མ་ ༡༥༠༠ ཙམ་ཁ་བྲལ་དུ་བཏང་ཡོད་པ་དང་། ཉེ་སྔོན་ལོན་ཏོན་གནམ་ཐང་ནས་ཐུག་འཕྲད་བྱུང་བའི་བུ་མོ་ཨི་ཝི་དང་ཨི་ཝཱ་གཉིས་བདག་མེད་མཚེ་མ་ཁ་འཐོར་དུ་གྱུར་བའི་དཔེ་མཚོན་ཞིག་ཡིན་ལ། ད་དུང་ཨ་རི་དང་ཁ་ན་ཌའི་ནང་དུའང་རྒྱ་ནག་གི་མཚེ་མ་བུ་མོ་ཁ་འཐོར་བ་ཁག་ཅིག་གསོ་སྐྱོང་གནང་མཁན་གྱི་ཁྱིམ་མིའི་རོགས་རམ་འོག་ཕན་ཚུན་ངོ་ཤེས་ཏེ་འབྲེལ་འདྲིས་བྱེད་བཞིན་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
