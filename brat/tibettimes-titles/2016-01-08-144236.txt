
སྲིད་སྐྱོང་དང་སྤྱི་འཐུས་ལ་འོས་འཕེན་དེབ་སྐྱེལ་གྱི་གོ་སྐབས་གཉིས་པ།

  བོད་མིའི་སྒྲིག་འཛུགས་དབུས་འོས་བསྡུ་ལས་ཁང་གིས། སྲིད་སྐྱོང་དང་སྤྱི་འཐུས་ཀྱི་སྔོན་འགྲོའི་འོས་བསྡུའི་སྐབས་དེབ་སྐྱེལ་གནང་མེད་མཁན་ཚོར་བསྐྱར་དུ་འོས་བསྡུའི་དེབ་སྐྱེལ་གནང་བཅུག་སྟེ། དངོས་གཞིའི་སྐབས་འོས་ལྡུག་ཐུབ་པའི་གོ་སྐབས་ཤིག་སྤྲད་ཡོད་པ་དང་འབྲེལ་ཏེ་དོགས་འདྲི་ཁག་ཅིག་ཞུས།   འདི་གའི་གསར་འགོད་པས་དབུས་འོས་བསྡུའི་འགན་འཛིན་ཤོ་ཟུར་བསོད་ནམས་ཆོས་འཕེལ་མཆོག་ལ། ཉེ་ལམ་ ༢༠༡༥ ལོའི་ཟླ་ ༡༠ ཚེས་ ༡༨ ཉིན་སྲིད་སྐྱོང་དང་སྤྱི་འཐུས་ཀྱི་སྔོན་འགྲོའི་འོས་བསྡུ་གནང་སྐབས། དེབ་སྐྱེལ་གནང་མེད་མཁན་མང་པོ་ཞིག་གིས་འོས་བླུགས་ཐུབ་མེད་སྟབས། དབུས་འོས་བསྡུ་ལས་ཁང་གིས་ ༢༠༡༦ ཟླ་ ༣ ཚེས་ ༢༠ ཉིན་དངོས་གཞིའི་འོས་བསྡུའི་སྔོན་ལ། བསྐྱར་དུ་དེབ་སྐྱེལ་གྱི་གོ་སྐབས་ཤིག་གསལ་བསྒྲགས་གནང་ཡོད་པའི་སྐོར་ངོ་སྤྲོད་ཅིག་གནང་རོགས་ཞུས་པར་ཁོང་གིས། དེ་ལ་གསལ་བཤད་ཞུ་དགོས་པ་རིགས་གཉིས་ཆགས་ཀྱི་འདུག དང་པོ། འོས་བསྡུའི་དེབ་སྐྱེལ་བྱས་ཡོད་ཀྱང་། སྔོན་འགྲོའི་ནང་འོས་ལྡུག་ཐུབ་མེད་ན། དངོས་གཞིར་བླུག་ཆོག་གི་མ་རེད་ཟེར་བའི་སྐད་ཆ་ཞིག་འདུག དེ་བདེན་པ་མ་རེད། འོས་བསྡུའི་དེབ་སྐྱེལ་བྱས་ཡོད་ན་སྔོན་འགྲོ་དངོས་གཞི་གཉིས་ཆར་འོས་བླུག་ཆོག་གི་ཡོད་པ་མ་ཟད། ལས་ཀའི་དབང་གིས་སྔོན་འགྲོའི་ནང་འོས་ལྡུག་མ་ཐུབ་ཀྱང་དངོས་གཞིར་བླུག་ཆོག་གི་ཡོད་པ་རེད། གཉིས་པ། སྔོན་འགྲོའི་སྐབས་སུ་འོས་བསྡུའི་དེབ་སྐྱེལ་བྱས་མེད་པའི་དབང་གིས་འོས་བླུག་ཐུབ་མེད་ན། དངོས་གཞིའི་སྔོན་ལ་དེབ་སྐྱེལ་གནང་ཐུབ་པ་ཡིན་ན། སྔོན་འགྲོར་འོས་ལྡུག་ཐུབ་མེད་ཀྱང་དངོས་གཞིའི་སྐབས་སུ་འོས་བླུག་ཆོག་གི་ཡོད་པ་རེད། དེ་ཚང་མས་གསལ་བོར་མཁྱེན་དགོས་པ་ཞིག་དང་། གོ་སྐབས་ལེན་དགོས་པ་ཞིག་རེད།   ཚང་མས་དོ་སྣང་གནང་དགོས་པ་ཞིག་ལ། སྔོན་འགྲོའི་ནང་འོས་བླུག་པའི་གོ་སྐབས་མ་རག་པ་དག་གིས་གང་མགྱོགས་སོ་སོའི་ས་གནས་སུ་འོས་བསྡུའི་དེབ་སྐྱེལ་གནང་དགོས་པ་རེད། ང་ཚོ་དབུས་ནས་ཉེ་ལམ་ཕྱི་ལོ་ ༢༠༡༥ ཟླ་ ༡༡ ཚེས་ ༦ ཉིན་ནས་ས་གནས་ཁག་ལ་འོས་བསྡུའི་དེབ་སྐྱེལ་གྱི་གོ་སྐབས་གཉིས་པ་ཕུལ་ཡོད་པ་རེད། གོ་སྐབས་དེ་ཕྱི་ལོ་ ༢༠༡༦ ཟླ་ ༢ ཚེས་ ༢༠ ནས་རྫོགས་ཀྱི་སྟབས། དུས་ཚོད་དེའི་ནང་ཚུན་ལ་དེབ་སྐྱེལ་བྱེད་ཐུབ་པ་གནང་དགོས་པ་རེད།   ཕྱི་ལོ་ ༢༠༡༦ ཕྱི་ཟླ་ ༣ པའི་ཚེས་ ༢༠ ནང་ཚུན། རང་ལོ་ ༡༨ ལ་སོན་ཡོད་ཚེ། དངོས་གཞི་ལ་དེབ་སྐྱེལ་བྱ་ཆོག་གི་ཡོད་པ་དང་། དེ་སྔ་སྔོན་འགྲོའི་སྐབས་སུ་རང་ལོ་ ༡༨ ལ་སོན་མེད་ཀྱང་དངོས་གཞིའི་སྐབས་སུ་རང་ལོ་ ༡༨ ལ་སོན་ཡོད་ནའང་དངོས་གཞིར་དེབ་སྐྱེལ་གནང་ཆོག་གི་ཡོད་པ་རེད། མདོ་དོན། སྔོན་འགྲོ་ལ་འོས་ཤོག་བླུགས་ཐུབ་མེད་པ་ཡིན་ན་ཡང་། དངོས་གཞིར་འོས་ཤོག་བླུག་ཐུབ་པའི་ཆེད་དུ་གོ་སྐབས་སྤྲོད་ཀྱི་ཡོད་པ་རེད། ཅེས་གསུངས་སོང་།   སྲིད་སྐྱོང་དང་སྤྱི་འཐུས་ཀྱི་དངོས་གཞིའི་འོས་མི་ག་དུས་གསལ་བསྒྲགས་གནང་རྒྱུ་དང་། དངོས་གཞིའི་འོས་མིའི་ཁྲོད་ཆོལ་ཁ་དང་ཆོས་བརྒྱུད། ཕྱི་རྒྱལ་ཁག་བཅས་ནས་འོས་མི་ག་ཚོད་རེ་གསལ་བསྒྲགས་གནང་འཆར་ཡོད་མེད་དྲིས་པར་ཁོང་གིས། སྲིད་སྐྱོང་དང་སྤྱི་འཐུས་ཀྱི་དངོས་གཞིའི་འོས་མི་དེ་ ༢༠༡༦ ལོའི་ཕྱི་ཟླ་ ༢ པའི་ཚེས་ ༣ ཉིན་གསལ་བསྒྲགས་བྱ་རྒྱུ་རེད། སྲིད་སྐྱོང་འོས་གཞི་དེ་སྔ་ནས་གསལ་བསྒྲགས་ཞུས་པ་ལྟར་རེད། སྒྲིག་གཞིའི་དགོངས་དོན་ལྟར་འོས་མི་ ༢ གསལ་བསྒྲགས་བྱ་རྒྱུ་རེད། སྤྱི་འཐུས་ཆོལ་ཁ་སོ་སོར་འོས་མི་ ༢༠ རེ་དང་། ཆོས་བརྒྱུད་ཁག་ལ་འོས་མི་ ༤ རེ། ཨ་རི་དང་ཡུ་རོབ་ལ་འོས་མི་ ༤ རེ། རྒྱ་བལ་འབྲུག་གསུམ་ཕུད་ཨོ་སི་ཊིར་ལེ་ཡ་དང་ཨེ་ཤ་ཡའི་ནང་འོས་མི་ ༢ བཅས་གསལ་བསྒྲགས་ཞུ་རྒྱུ་རེད། དེའི་སྟེང་ལ་སོ་སོ་དང་བླངས་ཀྱིས་བཞེངས་མཁན་དག་ཁ་སྣོན་རྒྱག་རྒྱུ་རེད། ཅེས་གསུངས་སོང་།   ཡང་ཁོང་ལ། མང་ཚོགས་ནང་ནམ་རྒྱུན་སྡོད་ས་དང་། འོས་ལྡུག་ས་གཉིས་གཅིག་པ་མིན་ན་འོས་ལྡུག་ཆོག་གི་མ་རེད་ཟེར་བའི་སྐད་ཆ་ཞིག་ཡོང་གི་འདུག དེའི་ཐད་གསལ་བཤད་འདྲ་ཡོད་དམ་ཞེས་ཞུས་པར་ཁོང་གིས། དེའི་སྐོར་ལ་མང་ཚོགས་ནང་གསལ་བོ་ཆགས་མ་ཐུབ་པའི་གནས་ཚུལ་ཞིག་ཡོང་གི་ཡོད་ས་རེད། འོས་བསྡུའི་དེབ་སྐྱེལ་བྱས་ཡོད་པ་ཡིན་ན། འོས་བསྡུ་གནང་སྐབས་སོ་སོ་གང་དུ་ཡོད་ན་འོས་བླུག་ཆོག་གི་རེད། ཐོག་མ་སོ་སོ་སྡོད་ས་ཟེར་ན་འདྲ་འོས་བསྡུའི་དེབ་སྐྱེལ་གནང་ས་དང་། དངོས་སུ་འོས་ལྡུག་ས་གཉིས་ས་གནས་གཅིག་ཡིན་དགོས་པ་མ་རེད། ཡིན་ནའང་གསལ་བོར་མཁྱེན་དགོས་པ་ཞིག་ལ། ནམ་རྒྱུན་ཨ་རིར་བཞུགས་ཏེ་འོས་བསྡུའི་དེབ་སྐྱེལ་ཡང་ཨ་རིར་གནང་ཡོད་མཁན་ཞིག འོས་ཤོག་ལྡུག་པའི་ཉིན་མོར་དྷ་རམ་ས་ལར་ཕེབས་བསྡད་ཡོད་ན། སྲིད་སྐྱོང་ལ་འོས་ལྡུག་ཆོག་གི་རེད། ཡིན་ནའང་སྤྱི་འཐུས་ལ་འོས་ལྡུག་པའི་གོ་སྐབས་ཡོད་པ་མ་རེད། གང་ཡིན་ཟེར་ན། ཁོ་རང་ཨ་རིར་སྡོད་མཁན་ཞིག་ཡིན་སྟབས་དྷ་རམ་ས་ལ་ནས་ཨ་རིའི་སྤྱི་འཐུས་ལ་འོས་ལྡུག་པའི་གོ་སྐབས་ཡོད་པ་མ་རེད། རྒྱ་བལ་འབྲུག་གསུམ་ནང་ནས་ཆོལ་ཁ་ཆོས་ལུགས་ཀྱི་སྤྱི་འཐུས་ལ་འོས་ལྡུག་གི་ཡོད་པ་རེད། དེ་བཞིན། དྷ་རམ་ས་ལར་སྡོད་མཁན་ཞིག་དངོས་སུ་འོས་ཤོག་ལྡུག་པའི་ཉིན་མོར་ཨ་རིར་འབྱོར་ཡོད་ནའང་དེ་བཞིན་རེད།   དོ་སྣང་བྱེད་དགོས་པ་གཙོ་བོ་ས་གནས་གང་ནས་འོས་ལྡུག་ནའང་ ༢༠༡༥ ལོའི་དཔྱ་ཁྲལ་ཆད་ལྷག་མེད་པ་ཕུལ་ཟིན་པ་མ་ཟད། འོས་བསྡུའི་དེབ་སྐྱེལ་ངེས་པར་དུ་གནང་ཡོད་པ་ཞིག་ཡིན་དགོས་པ་རེད། ཅེས་གསུངས་སོང་།   མཇུག་ཏུ་འོས་བསྡུའི་འགན་འཛིན་མཆོག་ལ། སྔོན་འགྲོའི་འོས་བསྡུའི་སྐབས་སུ་ས་གནས་རེ་ཟུང་ནང་འོས་བསྡུ་ཚགས་ཚུད་ཡོང་དང་མ་ཡོང་གི་སྐད་ཆ་རེ་ཟུང་འདུག དབུས་འོས་བསྡུ་ལྷན་ཁང་གི་ངོས་ནས་དངོས་གཞིའི་འོས་བསྡུའི་སྐབས་ལ། སྔར་ལས་ལྷག་པར་ཚགས་ཚུད་ཡོང་ཐབས་ལ་གྲ་སྒྲིག་དང་ལས་འཆར་གང་འདྲ་ཡོད་མེད་དྲིས་པར་ཁོང་གིས། སྔོན་འགྲོའི་སྐབས་སུ་ལས་དོན་བྱེད་ཕྱོགས་ཐད་དབུས་འོས་བསྡུ་ལས་ཁང་ལ་ཐད་ཀར་འགྲིག་དང་མ་འགྲིག་འདི་དང་འདི་འདུག་ཟེར་ནས་ཞུ་གཏུག་རྒྱག་མཁན་དེ་འདྲ་བྱུང་མ་སོང་། ཡིན་ནའང་ང་ཚོས་སྐྱོན་གནད་དེ་འདྲ་ཆགས་ཡོད་པ་དང་། སྐྱོན་གནད་དེ་འདྲ་མཐོང་པ་གཞིར་བཞག་སྟེ་དངོས་གཞིའི་སྐབས་སུ་སྐྱོན་གནད་དང་། དཀའ་ངལ་དེ་འདྲ་མི་ཡོང་བའི་ཆེད་དུ་ང་ཚོ་དབུས་འོས་བསྡུ་ལས་ཁང་ནས་ས་གནས་འོས་བསྡུ་ལས་ཁང་ཁག་ལ་ལམ་སྟོན་ཆ་ཚང་ཞིག་ཞུ་བཞིན་ཡོད།   གསར་འགོད་པ། པདྨ་རྒྱལ།
