དམག་སྡེའི་སྟག་རྒན་ ༤༣ རྙི་རུ་ཚུད།

  ཞི་ཅིན་ཕིན་གྱིས་ཅང་ཙེ་མིན་གྱི་དམག་ཁུལ་གཙང་སྦྲ་བཟོས་ཏེ་ད་ལྟའི་བར་ཁྱོན་དམག་སྡེའི་སྟག་རྒན་ ༤༣ གྱི་གོ་གནས་ཕབ་ཡོད་འདུག   ཁ་སང་རྒྱ་ནག་དམག་དོན་དྲ་བས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཉེ་ཆར་རྒྱ་ནག་དམག་དོན་ཞིབ་བཤེར་ལྷན་ཁང་གིས་ཀོང་ཀྲུའུ་དམག་ཁུལ་གྱི་དམག་སྤྱི་འབྲིང་བ་ཟུར་པ་ཝང་ཡས་ཧྥ་ (Wang Yufa) ཁྲིམས་འགལ་གྱི་དོགས་གཞིའི་འོག་ཞིབ་བཤེར་བྱེད་ཀྱི་ཡོད་སྐོར་ཕྱིར་བསྒྲགས་བྱས་ཡོད་པ་མ་ཟད། འདི་ལོར་འཛིན་བཟུང་བྱས་པའི་དབུས་དམག་དོན་རྒྱབ་ཕྱོགས་བཀོད་འདོམས་ཁང་གི་འགན་འཛིན་གཞོན་པ་ཟུར་བ་གུ་ཅུན་ཧྲན་(Gu Junshan) སོགས་དམག་དཔོན་གསུམ་ལའང་ཁོ་བོ་འབྲེལ་བ་ཡོད་པའི་དོགས་གཞི་ཡོད་ཚུལ་བཀོད་འདུག   ཝང་ཡས་ཧྥ་ལ་ཞིབ་བཤེར་བྱེད་ཀྱི་ཡོད་པ་ཕྱིར་བསྒྲགས་བྱས་པ་དང་མཉམ་དུ་རྒྱ་ནག་གི་བརྒྱུད་ལམ་ཁག་གཞན་དུ། ལོ་འདིའི་ཕྱི་ཟླ་ ༨ ཚེས་ ༢༨ ཉིན་རྒྱ་ཡོངས་སྲིད་གྲོས་རྒྱུན་ལས་ཚོགས་འདུ་ཐེངས་ ༡༢ བ་མཇུག་བསྡུ་སྐབས། མཐོ་རིམ་དཔོན་རིགས་ཀྲང་ལི་ཅོན་ (Zhang Lijun)དང་ཝང་ལང་ (Wang Liang) སོགས་ཚོགས་མི་ ༥ ལས་གནས་ནས་ཕྱིར་འཕུད་བཏང་བ་མ་ཟད། མི་ ༤ ལ་ཁྲིམས་འགལ་བྱས་པའི་བརྡ་ཁྱབ་བཏང་ཡོད་ཀྱང་གཅིག་ལ་བཏང་མེད་པ་དེ་ཝང་ཡས་ཧྥ་ཡིན་སྐོར་སོགས་བརྗོད་འདུག   ད་ལོའི་ཟླ་ ༣ ཚེས་ ༢ ཉིན་རྒྱལ་ཡོངས་འཐུས་མི་ཚོགས་ཆེན་སྐབས་ ༡༢ བའི་ཚོགས་དུས་ ༣ པ་མགོ་མ་བཙུགས་སྔོན་རྒྱ་ནག་གཞུང་གིས་མཐོ་རིམ་དམག་དཔོན་ ༡༤ ལྷག་ཞིབ་བཤེར་བྱ་རྒྱུ་ཁྱབ་བསྒྲགས་བྱས་ཡོད་པ་དང་། རྒྱ་ནག་ནང་དུ་དམག་དཔོན་ཞིབ་བཤེར་གླེང་སློང་ཤུགས་ཆེན་པོ་ཆགས་ཡོད་པ་མ་ཟད། ཟླ་དེའི་ཚེས་ ༣ ཉིན་ཚོགས་ཆེན་ཚོགས་སྐབས་ཝང་ཡས་ཧྥ་ལ་གསར་འགོད་པའི་བཅར་འདྲི་བྱེད་སྐབས། དམག་དཔོན་ཚོའི་ལྐོག་ཟ་རུལ་སུངས་ལ་རྒོལ་རྡུང་གཏོང་བར་རྒྱབ་སྐྱོར་ཡོད་ཅེས་སོགས་བརྗོད་ཡོད་འདུག   དེ་རིང་ཀྲུང་གོར་ལྟ་བ་ཞེས་པའི་དྲ་བའི་སྟེང་། ཀོང་ཀྲུའུ་ཞིང་ཆེན་གྱི་དམག་ཁུལ་ཐབས་ཇུས་ལྷན་ཁང་གི་མགོ་གཙོ་གཞོན་པ་ཟུར་བ་ཁྲིན་ཅན་ཧྥིན་ད་ལོའི་ལོ་སྟོད་ནས་ཞིབ་བཤེར་བྱས་ཡོད་པ་དང་། ཝང་ཡས་ཧྥ་ཡིས་ཀོང་ཀྲུའུ་ཞིང་ཆེན་དམག་ཁུལ་གྱི་མཁའ་དམག་རྒྱུན་ལས་བྱས་པའི་ལོ་ ༣ གྱི་བརྒྱུད་རིམ་ཁྲོད་ཁོང་གཉིས་ལ་འབྲེལ་བ་ཟབ་མོ་ཡོད་ཚུལ་དང་། ཕྱི་ལོ་ ༢༠༡༤ ལོར་གོ་གནས་ནས་ལྷུང་བའི་ཁྲིན་ཏུའུ་དམག་ཁུལ་གྱི་དམག་སྤྱི་གཞོན་པ་ཟུར་བ་ཡཱང་ཅིན་ཧྲན་ (Yang Jinshan)་དང་། ད་ལོ་གོ་གནས་ནས་ལྷུང་བའི་དབུས་དམག་དོན་རྒྱབ་ཕྱོགས་བཀོད་འདོམས་ཁང་གི་འགན་འཛིན་གཞོན་པ་ཟུར་བ་གུ་ཅུན་ཧྲན་སོགས་ནི་ཝང་ཡས་ཧྥ་ཡི་ཡུལ་ཕྱོགས་གཅིག་པ་ཡིན་ཚུགས་དང་། ཞི་ཅིན་ཕིན་གྱིས་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་རྒོལ་རྡུང་ཁྲོད་ད་ལྟའི་བར་ཁྱོན་དམག་སྡེའི་སྟག་རྒན་༤༣ ལ་ཞིབ་བཤེར་བྱས་ཏེ་གོ་གནས་ཕབ་ཡོད་སྐོར་སོགས་བརྗོད་འདུག   ཝང་ཡས་ཧྥ་ནི་ ༡༩༤༨ ལོའི་ཟླ་ ༨ པར་ཧི་ནན་ནན་ཀྲའོ་ནས་སྐྱེས་བ་དང་། ཕྱི་ལོ་ ༡༩༦༣ ལོའི་ཟླ་ ༣ པའི་ནང་རྒྱ་ནག་མི་མང་བཅིངས་གྲོལ་དམག་མིའི་ནང་ཞུགས་ཡོད་འདུག   གསར་འགོད་པ། འཇིགས་མེད་རྒྱ་མཚོ།
