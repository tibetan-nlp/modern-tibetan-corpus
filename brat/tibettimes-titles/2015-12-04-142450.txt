མུ་ར་སྦྱིན་པ་ཀློད་གྲོལ་བཏང་།

ཁ་སང་ཕྱི་ཚེས་ ༣ ཉིན་ཀན་ལྷོ་རྨ་ཆུ་རྫོང་ནས་ཡིན་པའི་ཆབ་སྲིད་བཙོན་པ་མུ་ར་སྦྱིན་པ་ལགས་ཀློད་གྲོལ་བཏང་སྟེ་རང་ཡུལ་དུ་འབྱོར་སྐབས་ས་གནས་སེར་སྐྱ་མི་མང་གིས་ཕེབས་བསུ་གཟབ་རྒྱས་ཞུས་ཡོད་འདུག   འདི་གར་བོད་མི་ཞིག་གིས་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། ཕྱི་ལོ་ ༢༠༡༢ ཟླ་ ༡༠ ཚེས་ ༢༥ ཉིན་གྱི་བོད་ནང་གི་མཚན་མོའི་ཆུ་ཚོད་ ༢།༠༠ པ་ཡས་མས་སུ། རྨ་ཆུ་རྫོང་མུ་ར་ངག་སྐོར་སྡེ་བའི་དགེ་འདུན་པ་སྦྱིན་པ་ལགས་ཁྲིན་འདུ་གྲོང་ཁྱེར་ནས་གློ་བུར་དུ་འཛིན་བཟུང་བྱས་ཡོད་པ་དང་། ཟླ་ ༥ ལྷག་ལ་གཅར་རྡུང་འདྲི་གཅོད་བྱས་རྗེས། ཕྱི་ལོ་ ༢༠༡༣ ཟླ་ ༤ པ་ཙམ་ལ། ཕྱི་ལོ་ ༢༠༠༨ ནས་བཟུང་རྒྱ་ནག་ཉེན་རྟོག་པ་དང་དམག་མིས་བོད་ཁུལ་དུ་ཚད་ལས་བརྒལ་བའི་དྲག་གནོན་དང་། མནར་གཅོད་བཏང་བ་སོགས་ཀྱི་གནས་ཚུལ་རྣམས་ཕྱིར་བསྒྲགས་བྱས་པ་དང་། བདེན་པའི་བསྒྲགས་གཏམ་ཞེས་པའི་ཡི་གེ་ཞིག་བྲིས་ཏེ་རྨ་ཆུ་རྫོང་དུ་བཀྲམ་པ་སོགས་ཀྱི་ནག་ཉེས་གཡོགས་ཏེ། ལོ་ ༣ གྱི་ཁྲིམས་ཐག་བཅད་ནས་ཀན་སུའུ་ཞིང་ཆེན་པའེ་ཡེན་གྲོང་ཁྱེར་པའེ་ཝན་གྲོང་རྡལ་གྱི་བཙོན་ཁང་ཞིག་ཏུ་བཅུག་ཡོད་པ་ལྟར། ཁ་སང་ཕྱི་ཚེས་ ༣ ཉིན་ཀློད་གྲོལ་བཏང་ཡོད་སྟབས། ཁྱིམ་མི་གཉེན་འབྲེལ་སོགས་ཀྱིས་རླངས་འཁོར་མང་པོས་ཁོང་རང་ཡུལ་དུ་བསུས་ཡོད་པ་དང་། སྐབས་དེར་ས་གནས་བོད་རིགས་སེར་སྐྱ་མང་པོས་གཞུང་ལམ་འགྲམ་དུ་བང་བསྒྲིགས་ནས་ཁ་བཏགས་བཟུང་སྟེ་ཕེབས་བསུ་གཟབ་རྒྱས་ཞུས་ཡོད་པ་དང་། གནས་ཚུལ་མཁོ་སྤྲོད་པས་སྦྱིན་པ་ལགས་ཀྱི་སྐུ་གཟུགས་གནས་སྟངས་སྐྱོ་བོ་ཡོད་སྐོར་བརྗོད་བྱུང་།   མུ་ར་སྦྱིན་པ་ལགས་ད་ལྟ་དགུང་ལོ་༤༣ ཡིན་ཞིང་། ཕའི་མིང་ལ་ནོར་ཕུན་དང་མའི་མིང་ལ་མཚོ་མོ་ཟེར་གྱི་ཡོད་པ་དང་། ཁོང་གིས་དེ་སྔོན་དགུན་གནང་སོགས་ཀྱི་སྐབས་སུ། ས་གནས་ཀྱི་སློབ་གྲྭ་དང་གྲྭ་བཙུན་རྣམས་ལ་སུམ་རྟགས་སོགས་བོད་ཀྱི་རིག་གནས་ཟབ་སྦྱོང་འཛིན་གྲྭ་ཐེངས་མང་པོར་བཙུགས་ནས་སློབ་ཁྲིད་གནང་ཡོད་པ་དང་། ལོ་རེ་རེ་བཞིན་ཕ་སྐད་དང་ཡིག་རྨོངས་སེལ་བའི་འཛིན་གྲྭ་བསྡུ་བ་སོགས་ཕ་ཡུལ་ལ་ཕན་པའི་ཕྱག་ལས་གང་མང་གནང་ཡོད་ཀྱང་། ཕྱི་ལོ་ ༢༠༠༨ ལོར་རྨ་ཆུ་ས་གནས་སུ་བྱུང་བའི་ཞི་རྒོལ་གྱི་གནས་ཚུལ་ཕྱི་ཕྱོགས་སུ་མཁོ་སྤྲོད་བྱས་པ་དང་། རྒྱལ་ཁབ་ཀྱི་གསང་བ་ཕྱིར་སྒྲོག་བྱས་པའི་ཉེས་མིང་འོག རྒྱ་གཞུང་ཉེན་རྟོག་པས་ཁོང་ཐེངས་མང་པོར་འཛིན་བཟུང་བྱས་ཏེ་བཀག་ཉར་འདྲི་བརྩད་དང་རྗེས་འདེད་ལྟ་རྟོག་སོགས་བྱས་ནས་སུན་གཙེར་མང་པོ་བྱས་མཐར་དངོས་སུ་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
