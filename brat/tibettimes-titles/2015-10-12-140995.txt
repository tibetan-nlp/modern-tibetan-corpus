གད་སྙིཊ་ཉུང་འཁྲིའི་བསམ་བློ་དགོས།

  རྒྱ་ནག་ཁྲོན་ཝན་གྲོང་བརྡར་གྱི་ཡུལ་མི་ཚོས་ཨར་འདམ་བཟོ་གྲྭའི་གད་སྙིགས་སྲེག་སར་ངོ་རྒོལ་བྱེད་སྐབས། ཉེན་རྟོག་པ་འབོར་ཆེན་ཡོང་སྟེ་མི་མང་ལ་དྲག་གནོན་དང་འཛིན་བཟུང་བྱས་ཡོད་འདུག   ཕྱི་ཚེས་ ༡༠ ཉིན་གྱི་སྔ་དྲོར་ཧྲན་ཏུང་ཞིང་ཆེན་ཡཱང་ཁྲོན་གྲོང་ཁྱེར་ཁྲོན་ཝན་གྲོང་བརྡར་རང་དབང་དོ་དམ་ས་ཁུལ་གྱི་ཡུལ་མི་སྟོང་ཕྲག་མང་པོས། དུང་དཀར་ཨར་འདམ་བཟོ་གྲྭས་གད་སྙིགས་སྲེག་ས་བཟོས་པར་ངོ་རྒོལ་བྱེད་སྐབས། ཉེན་རྟོག་པ་དང་དྲག་ཆས་ཉེན་རྟོག་པ་འབོར་ཆེན་ཡོང་སྟེ་མི་མང་ནང་གང་བྱུང་མིག་ཚ་དུག་རྫས་གཡུགས་པ་སོགས་དྲག་གནོན་བྱས་པར་བརྟེན། ཉེན་རྟོག་པ་དང་མི་མང་བར་འཁྲུག་འཛིང་ཤོར་ཏེ་མི་མང་པོར་རྨས་སྐྱོན་བཟོས་ཤིང་། ངོ་རྒོལ་བ་གང་མང་འཛིན་བཟུང་བྱས་ཡོད་པ་དང་། ཉེན་རྟོག་པའི་རླངས་འཁོར་དང་། ས་གནས་སྲིད་གཞུང་གི་རླངས་འཁོར། དེ་བཞིན་སྐྱེལ་འདྲེན་རླངས་འཁོར་སོགས་ཁ་ཤས་མེར་བསྲེགས་བཏང་བ་སོགས་ངོ་རྒོལ་ཤུགས་ཆེན་བྱས་ཡོད་འདུག   ཟླ་འདི་ཚེས་ ༣ ཉིན་ནས་བཟུང་ངོ་རྒོལ་བྱས་ཏེ་བཟོ་གྲྭའི་ལམ་ཁ་བཀག་པ་སོགས་བྱས་ཡོད་སྟབས། ཉེན་རྟོག་པས་ཡུལ་མི་ཁ་ཤས་འཛིན་བཟུང་བྱས་ཡོད་པ་དང་། དེ་བཞིན། ཕྱི་ཚེས་ ༩ ཉིན་ཡང་བསྐྱར་ངོ་རྒོལ་བྱས་ནས་ཡུལ་མི་ཚོ་དང་ཉེན་རྟོག་པའི་བར་དུ་འཁྲུག་འཛིང་ཤོར་ནས་ཉེན་རྟོག་པའི་རླངས་འཁོར་གཅིག་མེར་བསྲེགས་བཏང་བ་སོགས་ཀྱི་གནས་ཚུལ་བྱུང་ཡོད་ཀྱང་། ས་གནས་སྲིད་གཞུང་གིས་མི་མང་གི་རེ་འདོད་ལ་ཉན་འཇོག་མ་བྱས་པར་སྣང་མེད་དུ་བཞག་ཡོད་འདུག   དྲ་མིང་ལ་ “ཁེ་ཁེ་ཆན་ཆན་” ཟེར་བ་ཞིག་གིས། ངེད་ཚང་གི་ཉེ་འགྲམ་ལ་ཡོད་པའི་དུང་དཀར་ཨར་འདམ་བཟོ་གྲྭས་གད་སྙིགས་སྲེག་ས་བཟོས་པར་བརྟེན་མང་ཚོགས་ཀྱིས་ངོ་རྒོལ་བྱས་ཤིང་། ཉེན་རྟོག་རླངས་འཁོར་ ༢ དང་གཞུང་ཞབས་སྒེར་སྤྱོད་ཀྱི་རླངས་འཁོར་ ༢ མེར་བསྲེགས་སོང་། མང་ཚོགས་དང་ཉེན་རྟོག་པའི་འཁྲུག་འཛིང་ཁྲོད་ཉུང་མཐའ་ཡང་ཡུལ་མི་ ༡༠ ལྷག་འཛིན་བཟུང་བྱས་ཡོད་པ་མ་ཟད། མང་པོ་ཞིག་ལ་རྨས་སྐྱོན་བྱུང་འདུག ཅེས་བརྗོད་ཡོད་པ་རེད།   ཡུལ་གནས་དངོས་མཐོང་གི་དྲ་མིང་ལ་ “རྒད་པོ་མགོ་རྡོག” ཟེར་བ་ཞིག་གིས། ཡཱང་ཁྲོན་གྲོང་ཁྱེར་གྱི་ཁྲོན་ཝན་གྲོང་བརྡལ་རང་དབང་དོ་དམ་ས་ཁུལ་གྱི་དུང་དཀར་ཨར་འདམ་སྒོ་ཁ་རུ་ཉེན་རྟོག་པ་འབོར་ཆེན་བསླེབས་ནས་མང་ཚོགས་དྲག་གནོན་བྱས་པ་དང་འཁྲུག་འཛིང་ཤོར་ཏེ་མིག་ཚ་དུག་རྫས་གཡུགས་པ་དང་། གཞུང་ལམ་ཡོད་ཚད་བཀག་པ་མ་ཟད་གསར་འགོད་པར་གནས་ཚུལ་རྒྱུས་ལོན་བྱེད་བཅུག་མེད་སྟབས། གསར་འགོད་པས་ཀྱང་གནས་ཚུལ་རྒྱུས་ལོན་བྱེད་ཕོད་ཀྱི་མི་འདུག ཅེས་བརྗོད་འདུག   ཅང་སུའུ་ཞིང་ཆེན་ཝོ་ཡེ་གྲོང་ཁྱེར་གྱི་ཁོར་ཡུག་སྲུང་སྐྱོབ་བྱེད་མཁན་ཝོ་ལེ་ཧོན་ (Wu Lihong) གྱིས། གད་སྙིགས་མེར་བསྲེགས་པ་དང་ས་ལ་སྦས་པས་ཁོར་ཡུག་སྦག་བཙོག་བཟོ་ཡི་རེད། ད་ལྟ་རྒྱ་ནག་གཞུང་གིས་གད་སྙིགས་གང་སར་སྤུངས་ཡོད་སྟབས། གད་སྙིགས་ཀྱི་འབྱུང་རྐྱེན་ལ་བསམ་མནོ་གཏང་སྟེ། གད་སྙིགས་ཕྱིར་བསྡུ་བྱེད་དགོས་པ་དང་ཡང་ན་གད་སྙིགས་ཉུང་འཁྲི་གཏོང་བའི་བསམ་བློ་གསར་པ་ཞིག་གཏང་དགོས་ཀྱི་འདུག ཉི་ཧོང་སོགས་རྒྱལ་ཁབ་མང་པོར་གད་སྙིགས་ཕྱིར་བསྡུ་བྱེད་པར་ཉམས་མྱོང་ཡོད་པ་རེད། རྒྱ་ནག་གིས་གད་སྙིགས་ཡོད་ཚད་མཉམ་གཅིག་ཏུ་བསྡུས་ཏེ། ལྕགས་རིགས་དང་ཟས་ལྷག་རིགས་ཚང་མ་འདྲེས་ཏེ་མེར་སྲེག་བཞིན་འདུག སྤྱིར་བཏང་ལྕགས་རིགས་ཕྱིར་བསྡུ་བྱེད་ཆོག་པ་དང་། ཟས་ལྷག་རིགས་ཀྱང་ལུད་སྙོལ་བ་སོགས་བྱས་ཏེ་བསྐྱར་དུ་བེད་སྤྱོད་གཏོང་ཐུབ་ཀྱི་ཡོད་པ་རེད། ཅེས་སོགས་བརྗོད་འདུག གསར་འགོད་པ། འཇིགས་མེད་རྒྱ་མཚོ།
