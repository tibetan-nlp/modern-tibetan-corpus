ལང་ཚོ་དར་བབ་ཀྱི་ཟབ་སྦྱོང་ཚོགས།

བྱིས་པ་ལོ་ ༡༠ ནས་ ༡༩ བར་ལུས་སེམས་གཉིས་ལ་འགྱུར་བ་ཆེན་པོ་ཐེབས་ཀྱི་ཡོད་པས་སྐབས་དེར་དགེ་རྒན་དང་ཕ་མ་སོགས་ཚང་མས་དོ་སྣང་སྤྲད་དེ་ལམ་སྟོན་ཡག་པོ་ཞིག་གནང་ཐུབ་ན་མི་ཚེའི་མདུན་ལམ་ལ་ཕན་རླབས་ཆེན་པོ་ཡོད་སྐོར་བརྗོད་སོང་། དེ་རིང་ཕྱི་ཚེས་ ༢༤ ནས་༢༧ བར་རྒྱ་གར་ལྷོ་ཕྱོགས་གཞིས་ཆགས་སྦེལ་ཀོབ་སེར་བྱེས་མགྲོན་ཁང་ཡིད་དགའ་ཆོས་གླིང་དུ། བོད་མིའི་སྒྲིག་འཛུགས་འཕྲོད་བསྟེན་ལས་ཁུངས་ཀྱི་གོ་སྒྲིག་འོག དབུས་བོད་སློབ་འཛིན་ཚོགས་དང་སམ་བྷོ་ཊའི་འཛིན་རིམ་ལྔ་པ་ཡན་གྱི་དགེ་ལས་རྣམས་ལ་ལང་ཚོ་དར་བབ་ཀྱི་འཕྲོད་བསྟེན་དང་འབྲེལ་བའི་ཟབ་སྦྱོང་ཞིག་མགོ་འཛུགས་གནང་ཡོད་པ་དང་། མཛད་སྒོའི་ཐོག་སྐུ་མགྲོན་གཙོ་བོ་བོད་མི་མང་སྤྱི་འཐུས་བསམ་གཏན་ཆོས་སྒྲོན་ལགས་དང་། ལྡི་ལིར་ཡོད་པའི་རྒྱ་གར་ན་གཞོན་ལམ་སྟོན་ཚོགས་པ་(Time Expression India)ཡི་ལམ་སྟོན་པ་ལྕམ་ཨུ་མཱ་ཤར་མཱ་(Uma sharma)ལགས་དང་།  རི་ཐཱ་ཤར་མཱ་(Ritu shrma)ལགས། དེ་བཞིན་ས་གནས་ཁག་ནས་ཕེབས་པའི་དགེ་རྒན་ ༥༠ ལྷག་བཅས་ཟབ་སྦྱོང་ནང་མཉམ་ཞུགས་གནང་འདུག དབུས་འཕྲོད་བསྟེན་ལས་ཁུངས་ཀྱི་དྲུང་གཞོན་མི་འགྱུར་གཡུ་སྒྲོན་ལགས་ནས་ཟབ་སྦྱོང་སྐོར་ངོ་སྤྲོད་གནང་སྐབས། བུ་ཕྲུག་གི་མི་ཚེའི་ནང་སློབ་གྲྭ་དང་ཕ་མ་གལ་ཆེན་པོ་ཡིན་སྟབས། ཟླ་བ་གཉིས་གུང་སང་གི་སྐབས་དང་བསྟུན་ཏེ་དྷ་སྟེང་བོད་ཁྱིམ་བཅས་སློབ་གྲྭ་ཁག་༦ ལ་མདུན་ལམ་སློབ་སྟོན་པ་བརྒྱུད་དེ། སློབ་ཕྲུག་གུང་སེང་དུ་ལེན་མཁན་ཕ་མ་དང་སྤུན་མཆེད་རྣམས་ལ་གཞོན་ནུ་ལང་ཚོ་དར་བབ་ཀྱི་ཟབ་སྦྱོང་ཞིག་ཕུལ་རོགས་ཞུས་པ་ལྟར། དེ་རིང་སློབ་གྲྭ་ཁག་ ༦ དང་ཕ་མ་ལ་ཟབ་སྦྱོང་འདི་ཕུལ་ཐུབ་པ་བྱུང་ཡོད་ལ། རྗེས་མ་རྒྱ་གར་བྱང་ཤར་ཁུལ་གྱི་དགེ་རྒན་ཡོངས་ལ་ཟབ་སྦྱོང་འདི་གོ་སྒྲིག་ཞུ་རྒྱུ་ཡིན་སྐོར་དང་། བུ་ཕྲུག་གི་མི་ཚེའི་ནང་ལོ་ ༡༠ ནས་ ༡༩ བར་འཚར་ལོངས་ཀྱི་དབང་གིས་ལུས་སེམས་གཉིས་ལ་འགྱུར་བ་ཆེན་པོ་འགྲོ་བཞིན་ཡོད་སྟབས། ང་ཚོ་ཚང་མས་དོ་སྣང་དང་ལམ་སྟོན་ཡག་པོ་ཞིག་གནང་ཐུབ་པ་ཡིན་ན། བུ་ཕྲུག་ལུས་བདེ་སེམས་སྐྱིད་དང་མདུན་ལམ་ཡག་པོ་ཡོང་བར་ཕན་ཐོགས་ཡོད་སྟབས། འཕྲོད་བསྟེན་ལས་ཁུངས་ཀྱི་ངོས་ནས་གཞོན་ནུ་ལང་ཚོ་དར་བབ་ཀྱི་ལས་གཞི་འདི་གལ་ཆེན་ཞིག་ཏུ་རྩི་བཞིན་ཡོད་སྐོར་བརྗོད་སོང་།   སྐུ་མགྲོན་གཙོ་བོ་བོད་མི་མང་སྤྱི་འཐུས་བསམ་བསྟན་ཆོས་སྒྲོན་ལགས་ཀྱིས། བོད་རིགས་ཕ་མ་ཁ་ཤས་ཀྱིས་རང་གི་བུ་ཕྲུག་སློབ་གྲྭར་བཞག་ཚར་བ་ཙམ་གྱིས་འགྲིག་སོང་བསམ་གྱི་ཡོད་རེད་ལ། ང་རང་ཡིན་ནའང་དེ་འདྲ་བསམ་མྱོང་། ཡིན་ནའང་། དེ་འདྲ་རྩ་བ་ནས་རེད་མི་འདུག ཕྲུ་གུ་ཚོ་འཛིན་རིམ་ལྔ་པ་བར་དུ་སློབ་གྲྭར་ཡག་པོ་འགྲོ་གི་འདུག་ལ། སློབ་སྦྱོང་ཡང་ཡག་པོ་བྱེད་ཀྱི་འདུག འོན་ཀྱང་། འཛིན་གྲྭ་ལྔ་པ་ནས་ཕ་མ་དང་དགེ་རྒན་ཚོས་ཡག་པོ་བལྟས་ནས་འགན་ལེན་མ་ཐུབ་ན་སློབ་གྲྭར་ཡག་པོ་འགྲོ་ཡི་མི་འདུག སློབ་ཕྲུག་ཚོར་འགྱུར་བ་ཆེན་པོ་འགྲོ་ཡི་འདུག་པས་ཕ་མ་དང་དགེ་རྒན་མཉམ་རུབ་ཀྱི་ཐོག་ནས་ལྟ་སྐྱོང་ཡག་པོ་བྱེད་དགོས་ཀྱི་འདུག ཅེས་སོགསབརྗོད་སོང་།   དེའི་རྗེས་ལྕམ་ཨུ་མཱ་ཤར་མཱ་Uma sharma)ལགས་ཀྱིས། རྒྱ་གར་གྱི་ན་གཞོན་སློབ་སྟོན་ཚོགས་པ་(Time Expression India)ནས་བྱིས་པ་དང་ཕ་མ་དགེ་རྒན་ཚང་མར་འཁོར་ཡུག་བཟང་པོ་ཞིག་བསྐྲུན་ཐབས་སུ་འབད་བརྩོན་བྱེད་བཞིན་ཡོད་ལ། ད་ལྟ་བར་དུ་རྒྱ་གར་དང་ཕྱིའི་རྒྱལ་ཁབ་མང་པོའི་ནང་ན་གཞོན་ལམ་སྟོན་གྱི་ལས་འགུལ་ ༤༠༠༠ ལྷག་སྤེལ་ཐུབ་ཡོད་ལ། ད་ལན་ཟབ་སྦྱོང་འདིའི་ཐོག་བཅར་བའི་གོ་སྐབས་བྱུང་བར་དགའ་བོ་ཞེ་དྲགས་བྱུང་ལ། མ་འོངས་པར་ཡང་ཟབ་སྦྱོང་འདི་འདྲར་མཉམ་འཛོམས་ཡོང་བའི་རེ་བ་ཡོད་སྐོར་སོགས་བརྗོད་སོང་ལ། དེའི་རྗེས་མཁས་དབང་ལྕམ་རི་ཐཱ་ཤར་མཱ་(Ritu sharma)ལགས་ནས་ཟབ་སྦྱོང་མགོ་བཙུགས་གནང་སོང་།  རྒྱ་གར་ལྷོ་ཕྱོགས་ས་གནས་གསར་འགོད་པ་དོན་འགྲུབ་བཀྲ་ཤིས།
