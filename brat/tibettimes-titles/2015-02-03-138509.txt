རུལ་རྒོལ་མདུང་རྩེ་ནང་ལ་སྐོར།

ཉེ་ལམ་རྒྱ་ནག་དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གི་ཚོགས་འདུ་ཚོགས་རྗེས་རུལ་རྒོལ་གྱི་མདུང་ཁ་དེ་ད་ལྟ་ལས་ཀ་བྱེད་བཞིན་པའི་གཞུང་མཐོ་རིམ་དཔོན་རིགས་ ༩༦ ལ་གཏད་ཡོད་པ་རེད་ཟེར། ཟླ་སྔོན་མའི་ཚེས་ ༡༢ ནས་ ༡༤ ཉིན་བར་རྒྱ་ནག་དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གི་ཚོགས་ཆེན་ཐེངས་ ༥ བ་དེ་བསྐོངས་སྐབས། དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གི་མགོ་གཙོ་ཝོང་ཆེ་ཧྲན་ (Wang Qishan)གྱིས་ཞི་ཅིན་ཕིན་གྱི་སྐད་ཆ་ལུང་འདྲེན་བྱས་ཏེ། ལྐོག་ཟ་རུལ་སུངས་ག་ཚོད་མང་ཡང་རྒོལ་རྡུང་ལས་འགུལ་མཚམས་འཇོག་བྱ་རྒྱུ་མིན་པ་དང་། དུས་ཚོད་དང་ས་མཚམས། གནས་རིམ་སོགས་ལ་ངེས་གཏན་མེད་པར་མུ་མཐུད་ལག་བསྟར་བྱ་རྒྱུ་ཡིན་ཚུལ་བསྒྲགས་བྱས་ཡོད་སྟབས། ཕྱི་ཕྱོགས་ཀྱི་གསར་ལམ་ཁག་ཏུ་ཞི་ཅིན་ཕིན་གྱི་རུལ་སུངས་རྒོལ་རྡུང་གི་ཁ་ཕྱོགས་གསར་བའི་མདུང་རྩེ་ནང་ལ་གཏད་དེ། ད་ལྟ་རྒྱ་གཞུང་གི་ལས་གནས་སུ་ཡོད་པ་དང་། དགོངས་པ་ཞུས་ཟིན་པའི་མཐོ་རིམ་དཔོན་རིགས་ ༩༦ ཙམ་དམིགས་ཡུལ་དུ་བཟུང་ཡོད་ཅིང་། དེ་དག་ལས་ཁག་ཅིག་ལ་ཞིབ་བཤེར་བྱེད་བཞིན་ཡོད་པ་དང་ཁག་ཅིག་ལ་ཞིབ་བཤེར་བྱ་རྒྱུ་ལྷག་ཡོད་སྐོར་བརྗོད་འདུག ཟླ་འདིའི་ཚེས་ ༡ ཉིན་རྒྱ་ནག་ལྟ་ཞིབ་ཅེས་པའི་དྲ་ཚིགས་སྟེང་གནས་ཚུལ་སྤེལ་པ་ལྟར་ན། དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ལྷན་ཁང་གི་ཚོགས་འདུའི་སྐབས་ལྐོག་ཟ་རུལ་སུངས་ལ་རིམ་པ་ ༣ ཙམ་གཏན་འབེབས་བྱས་ཡོད་ཅིང་། དེ་དག་གི་ཁ་ཕྱོགས་གཙོ་བོ་ད་ལྟ་ལས་གནས་སུ་བཞུགས་བཞིན་པའི་མཐོ་རིམ་དཔོན་རིགས་དག་བཟུང་ཡོད་ཅིང་། རིམ་པ་དང་པོའི་ཁྲོད་དབུས་དྲིལ་བསྒྲགས་ལས་ཁུངས་ཀྱི་འགན་འཛིན་དང་སྔོན་ཆད་སི་ཁྲོན་ཞིང་ཆེན་གྱི་མགོ་གཙོ་ལའུ་ཆེ་པའོ་ (Liu Qibao)ནས་མགོ་བཙུགས་ཡོད་འདུག ལའུ་ཆེ་པའོ་ལ་ལོ་སྔོན་མ་ནས་ཞིབ་བཤེར་མགོ་བཙུགས་ཏེ་ད་བར་ཐེངས་བཞི་བྱས་པའི་བརྒྱུད་རིམ་ཁྲོད། ཁོས་ཀྲོའུ་ཡོན་ཁང་དང་ལིན་ཅུ་ཧྭ་སོགས་ཀྱི་ནག་ཉེས་རྩིས་སྤྲོད་བྱས་ཡོད་འདུག གནས་ཚུལ་དེའི་ནང་། ལའུ་ཆེ་པའོ་ལ་ལས་ཀ་བྱེད་ཞོར་དུ་ཞིབ་བཤེར་བྱས་པ་དེས་རུལ་སུངས་རྒོལ་རྡུང་གི་ལས་འགུལ་དེ་ཅུང་གུ་ཡངས་དང་ཚོད་འཛིན་ལྟ་བུ་བྱས་པ་མཚོན་ཡང་། ཞིབ་བཤེར་ལྷན་ཁང་གི་ཚོགས་འདུའི་སྐབས། ཝོང་ཆེ་ཧྲན་གྱིས་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱི་སྐད་ཆ་བསྐྱར་བཤད་བྱས་པ་དང་ལྐོག་ཟ་རུལ་སུངས་རྒོལ་རྡུང་གི་ལས་འགུལ་མུ་མཐུད་སྤེལ་རྒྱུ་ཡིན་སྐོར་བསྒྲགས་པ་དེར་གཞིགས་ན། རུལ་རྒོལ་ལས་འགུལ་དེ་ད་དུང་མུ་མཐུད་སྤེལ་རྒྱུ་ཡིན་པ་མ་ཟད། སྔར་ལས་ཤུགས་ཆེ་རུ་བཏང་སྟེ་ད་ལྟ་དབུས་གཞུང་ནང་ལས་ཀ་བྱེད་བཞིན་པ་ཚོ་ཡང་ཞིབ་བཤེར་ནང་ཚུད་རྒྱུ་ཡིན་སྐོར་འཁོད་འདུག ལའུ་ཆེ་པའོ་ནི་ཕྱི་ལོ་ ༢༠༠༧ ནས་ ༢༠༡༢ བར་སི་ཁྲོན་ཞིང་ཆེན་གྱི་མགོ་གཙོ་དང་ཞིང་ཆེན་ཨུ་ཡོན་རྒྱུན་ལས་འགན་འཛིན་ཡིན་འདུག ༢༠༡༢ ལོའི་ཏང་གི་ཚོགས་ཆེན་བཅོ་བརྒྱད་པ་ནས་ཁོའི་གོ་གནས་འཕར་ཏེ་དབུས་ཆབ་སྲིད་སྡེ་ཚན་ནང་མགོ་གཙོ་དང་དབུས་དྲིལ་བསྒྲགས་ལས་ཁུངས་ཀྱི་འགན་འཛིན་ཡིན་འདུག ཧོང་ཀོང་གི་འགྲན་གླེང་དུས་དེབ་ཀྱིས་གནས་ཚུལ་སྤེལ་པ་ལྟར་ན། ད་ལྟ་ཞིབ་བཤེར་འོག་ཚུད་ཡོད་པའི་ལའུ་ཆེ་པའོ་དང་གནས་བབ་གཅིག་པ་ཡིན་པའི་མཐོ་རིམས་དཔོན་རིགས་ ༥ ཡོད་པ་དང་། དེ་དག་ནི་ཝང་ཧྲིན་ཅོན་དང་ཀྲོང་ཕེན། ལོ་ཀྲན་ཀོན། ཝང་ཅཱ་རའུ། ཀྲོང་ཆིན་ལེ་བཅས་ཡིན་འདུག ཝང་ཧྲིན་ཅོན་ནི་རྒྱ་ནག་དབུས་མི་དམངས་རྒྱུན་ལས་ཨུ་ཡོན་འགན་འཛིན་གཞོན་པ་ཡིན་པ་དང་། ཁོ་སྔོན་ཆད་རྒྱ་ནག་ཆེས་མཐོའི་ཁྲིམས་ཁང་གི་འགན་འཛིན་དང་ཏང་རྩ་འཛུགས་མགོ་གཙོ། དབུས་ཆབ་སྲིད་ལྷན་ཁང་གི་དྲུང་ཡིག་སོགས་ཡིན་འདུག ཀྲོང་ཕེན་ནི་རྒྱ་ནག་འཕེལ་རྒྱས་ཨུ་ཡོན་འགན་འཛིན་ཟུར་པ་ཡིན་འདུག ལོ་ཀྲན་ཀོན་ནི་ད་ལྟ་རྒྱ་ནག་སྲིད་གྲོས་འགན་འཛིན་གཞོན་པ་དང་། སྔོན་ཆད་ཧི་ནན་ཞིང་ཆེན་གྱི་མགོ་གཙོ་ཡིན་འདུག ཝང་ཅཱ་རའུ་ནི་ད་ལྟ་རྒྱ་ནག་སྲིད་གྲོས་འགན་འཛིན་གཞོན་པ་དང་སྔོན་ཆད་རྒྱ་ནག་ཕྱི་འབྲེལ་ལྷན་ཁང་གི་འགན་འཛིན་ཡིན་འདུག ཀྲོང་ཆིན་ལེ་ནི་ད་ལྟ་སྲིད་གྲོས་འགན་འཛིན་གཞོན་པ་དང་སྔོན་ཆད་ཧི་པའེ་ཞིང་ཆེན་གྱི་མགོ་གཙོ་ཡིན་འདུག
