རྒྱ་ནག་གཞུང་ལ་ཐེ་ཇུས་བྱེད་དབང་མེད།

བོད་རང་སྐྱོང་ལྗོངས་ཀྱི་མགོ་གཙོས་༧གོང་ས་མཆོག་ལ་ཡང་སྲིད་གནད་དོན་ཐད་སྐྱོན་བརྗོད་བྱས་པར་སྲིད་སྐྱོང་མཆོག་གིས་ལྷ་མེད་སྨྲ་བ་དང་ཆོས་དུག་ཏུ་འཛིན་མཁན་ཚོར་ཐེ་ཇུས་བྱེད་དབང་མེད་སྐོར་གསུངས་འདུག ཟླ་འདིའི་ཚེས་ ༩ ཉིན་བོད་རང་སྐྱོང་ལྗོངས་ཀྱི་སྤྱི་ཁྱབ་པ་པདྨ་འཕྲིན་ལས་ཀྱིས་གལ་སྲིད་ཏཱ་ལའི་བླ་མས་ཁོང་གི་ཡང་སྲིད་ཕེབས་རྒྱུ་མ་རེད་གསུངས་པ་ཡིན་ཚེ། དེ་ནི་ནང་པ་སངས་རྒྱས་པའི་ཆོས་ལ་སྐུར་འདེབས་བྱས་པ་ཞིག་ཡིན་སྟབས། རྒྱ་ནག་གཞུང་གིས་བོད་བརྒྱུད་ནང་བསྟན་ནང་བླ་ཆེན་གང་ཞིག་འདས་གྲོངས་ཕྱིན་ཚེ་ཡང་སྤྲུལ་ངོས་འཛིན་གནང་སྲོལ་ཡོད་པ་དེ་རྒྱུན་འཁྱོངས་བྱས་ཏེ། ཏཱ་ལའི་བླ་མའི་ཡང་སྲིད་ངོས་འཛིན་བྱ་རྒྱུ་ཡིན་སྐོར་བརྗོད་ཡོད་སྟབས། ཁ་སང་ཚེས་ ༡༠ ཉིན་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་སྲིད་སྐྱོང་བློ་བཟང་སེང་གེ་མཆོག་གིས་སྐྱོན་བརྗོད་དང་འབྲེལ་ལན་འདེབས་གནང་འདུག སྲིད་སྐྱོང་མཆོག་གིས་རོའུ་ཌར་གསར་ལམ་ཐོག “༧གོང་ས་ཏཱ་ལའི་བླ་མ་མཆོག་གི་ཡང་སྲིད་གནད་དོན་ལ་པདྨ་འཕྲིན་ལས་གཙོས་ལྷ་མེད་སྨྲ་བ་དང་ཆོས་དུག་ཏུ་འཛིན་མཁན་གྱི་རྒྱ་ནག་དམར་ཤོག་ཚོགས་པའི་མགོ་ཁྲིད་སུ་གང་ལའང་ཐེ་ཇུས་བྱེད་དབང་མེད་ལ། དེ་ནི་ཕེ་ཌལ་ཁཱ་སེ་ཌོར་ (Fidel Castro) གྱིས་ངས་བླ་ཆེན་པོབ་རྗེས་མ་དེ་འདེམ་རྒྱུ་ཡིན་པས་ཡེ་ཤུའི་ཆོས་ལུགས་རྗེས་འབྲངས་པ་ཡོངས་ཀྱིས་ངོས་འཛིན་བྱེད་དགོས་ཞེས་བརྗོད་པ་དང་གཅིག་མཚུངས་ཆགས་ཡོད”ཅེས་གསུངས་འདུག ཉིན་འདི་གར་པེ་ཅིན་དུ། རྒྱ་ནག་རྒྱལ་ཡོངས་འཐུས་མི་ཚོགས་ཆེན་ཚོགས་བཞིན་ཡོད་པ་དང་བསྟུན་ཏེ། ཁོས་གསར་འགོད་པར་ “གལ་སྲིད་རྒྱ་ནག་དབུས་གཞུང་གིས་༧གོང་ས་ཏཱ་ལའི་བླ་མ་མཆོག་ངོས་འཛིན་གནང་མེད་ཚེ་ཁོང་སྐུ་ཕྲེང་ ༡༤ པ་ཆགས་ཐབས་མེད། ད་ལྟ་ཏཱ་ལའི་བླ་མས་པེ་ཅིན་གཞུང་གིས་ཡང་སྲིད་ངོས་འཛིན་བྱ་རྒྱུ་བཙན་འཕྲོག་བྱས་ཏེ། ཁོང་གི་ཡང་སྲིད་ཕེབས་རྒྱུ་མ་རེད་གསུངས་ན་ཡོང་ཐབས་མེད་པ་མ་ཟད། དེ་ནི་བོད་བརྒྱུད་ནང་བསྟན་པ་སུས་ཀྱང་ཁས་ལེན་བྱེད་ཐབས་བྲལ་བ་ཞིག་ཡིན་སྟབས། ལོ་རྒྱུས་ལ་བརྩི་བཀུར་བྱས་ནས་བོད་བརྒྱུད་ནང་བསྟན་དང་ཆོས་དད་ལ་སྐུར་འདེབས་བྱེད་མི་རུང་།” ཞེས་སོགས་བརྗོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
