འཛམ་གླིང་ཞི་བདེ་བསྐྲུན་དགོས།

  ཕྱི་ཚེས་ ༣༠ ཉིན་་༧གོང་ས་མཆོག་གིས་ཆོས་ལུགས་ཕན་ཚུན་མཐུན་སྒྲིལ་བྱས་ནས་འཛམ་གླིང་ཞི་བདེ་བསྐྲུན་ཐུབ་པ་བྱེད་དགོས་པ་གསུངས་འདུག   ཕྱི་ཟླ་ ༨ ཚེས་ ༢༩ ཉིན་༧གོང་ས་མཆོག་བཞུགས་སྒར་ནས་ཆིབས་ཐོན་གྱིས་རྒྱ་གར་གྲོང་ཁྱེར་ན་སིག་ཏུ་ཧིན་དུའི་ཆོས་ཕྱོགས་ཀྱི་འདུ་འཛོམས་ཆེན་མོ་ཀུམ་བྷ་མེ་ལ་ཞེས་པའི་དུས་སྟོན་ཐོག་ཆིབས་བསྒྱུར་བསྐྱངས་ཡོད་པ་མ་ཟད། ཕྱི་ཚེས་ ༣༠ ཉིན་ན་སིག་གྲོང་ཁྱེར་དུ་ཕེབས་འབྱོར་གནང་སྐབས་ཧིན་དུའི་བླ་ཆེན་སོ་མི་ཀར་ཤི་ནི་མཆོག་ (Swami Karshni Guruasharanandaji Maharaj) དང་གུ་རུ་ཤ་ར་ཎན་ཌ་མཆོག(Guru Sharananand)། སོ་མི་པར་མཱ་ཎཎ་ཌ་མཆོག་(Swami Parmanand)། སོ་མི་གྷ་ཡ་ཎཎ་ཌ་མཆོག་ (Swami Gyananand) བཅས་ཀྱི་ཕེབས་བསུའི་ཁྲོད རེམ་བྷ་ཁེ་ཤེ་ཝར་གྲོང་རྡལ་ (Trimbakeshwar) གྱི་ཁཱར་ཤེ་ཎེ་ཨ་ཤ་རམ་ (Karshni Ashram) དུ་ཕེབས་ནས་སོ་མི་ཀར་ཤི་ནི་མཆོག་དང་ལྷན་ཤིང་སྣ་འདེབས་འཛུགས་དང་གསོལ་འདེབས་སྨོན་ལམ་མཛད་ཡོད་པ་མ་ཟད། བླ་ཆེན་རྣམ་པ་དང་ལྷན་དུ་དྲི་བ་དྲིས་ལན་གྱི་ལམ་ནས་བགྲོ་གླེང་མཛད་པ་དང་གསར་འགོད་ལྷན་ཚོགས་སོགས་བསྐྱངས་ཡོད་འདུག   ཕྱི་ཚེས་ ༣༡ ཉིན་རྒྱ་གར་བང་ཆེན་གྱིས་༧གོང་ས་མཆོག་གིས་གསར་འགོད་ལྷན་ཚོག་ཐོག་གསུངས་པའི་བཀའ་མོལ་ལུང་འདྲེན་བྱས་པའི་ནང་། རྒྱ་གར་ནང་དམངས་གཙོའི་ལམ་སྲོལ་བཟང་པོ་དང་རིན་ཐང་དག་ཁ་ཞེ་གཉིས་མེད་ཐོག་ལག་བསྟར་དོན་འཁྱོལ་གནང་བཞིན་ཡོད་པར་བརྟེན། ད་ལྟ་རྒྱལ་ཡོངས་ནང་ཞི་བདེ་དང་ལྷིང་འཇགས་བྱུང་ཡོད་པ་མ་ཟད། ཆོས་ལུགས་དང་རིགས་རུས་མི་གཅིག་པའི་མི་རིགས་ཕན་ཚུན་དབར་འཆམ་མཐུན་ངང་ཞི་བདེའི་འཚོ་བ་རོལ་བཞིན་ཡོད་སྟབས། རྒྱ་གར་དང་ཁྱིམ་མཚེས་རྒྱལ་ཁབ་ཁག་བསྡུར་ན་རྒྱ་གར་ནི་ཞི་བདེའི་རྒྱལ་ཁབ་ཅིག་ཡིན། དེ་རིང་ངོས་འདིར་བཅར་དགོས་དོན་ནི་ཧིན་དུའི་བླ་ཆེན་དང་མཁས་དབང་གཙོས་རྩ་རླུང་དང་གཏུམ་མོའི་ཉམས་ལེན་གནང་མཁན་གྱི་ཆོས་གྲོགས་ཚོ་ཐུག་སྟེ། ང་ཚོས་མཉམ་རུབ་ཐོག་འཛམ་གླིང་ཞི་བདེ་ཡོང་ཐབས་ལ་འབད་བརྩོན་གང་འདྲ་བྱེད་དགོས་མིན་ཐོག་གྲོས་བསྡུར་བྱ་རྒྱུ་ཡིན། ངོས་ཀྱིས་རྒྱ་གར་ནི་ཆོས་ལུགས་རིས་མེད་ཀྱི་རྒྱལ་ཁབ་ཅིག་ཡིན་སྟབས། ཆོས་ལུགས་དང་རིགས་རུས་མི་གཅིག་པའི་མང་ཚོགས་དབར་འཆམ་མཐུན་དང་ཞི་བདེ་ཡོང་དགོས་པར་བརྟེན། ངོས་ཀྱིས་ཆོས་ལུགས་ཁག་གི་དབུ་ཁྲིད་རྣམ་པ་ཚོ་ངེས་པར་དུ་མཉམ་འཛོམས་གནང་སྟེ། རྒྱ་ཆེའི་མང་ཚོགས་ལ་བཟང་པོའི་བསླབ་བྱ་དེ་དག་རྒྱུད་སྤྲོད་གནང་རྒྱུ་ཁ་ཚ་དགོས་གཏུགས་ཆགས་ཡོད་པར་ངོས་འཛིན་བྱེད་བཞིན་ཡོད། ཅེས་གསུངས་འདུག   ཁོང་གིས་ཆོས་ལུགས་ཕན་ཚུན་བར་མཐུན་ལམ་གནང་དགོས་སྐོར་གསུངས་དོན། དེང་སང་ང་ཚོས་འཇིགས་སྐུལ་རིང་ལུགས་པ་དང་གནམ་གཤིས་འགྱུར་ལྡོག་སོགས་རྒྱལ་སྤྱིའི་ཚབས་ཆེན་གནད་དོན་ཁག་ལ་གདོང་ལེན་བྱེད་དགོས་སྟབས། ང་ཚོས་ཐབས་མཁས་ཐོག་འགྲོ་བ་མིའི་རིགས་ཀྱི་བདེ་སྡུག་རྨང་གཞིར་བཟུང་ནས་འཕྱར་གྲུ་མཉམ་འདེགས་ཐོག་དཀའ་ངལ་དེ་དག་སེལ་ཐུབ་པར་རྩོད་ལེན་བྱེད་དགོས། འཛམ་གླིང་ཐོག་གི་ཆོས་ལུགས་ཚང་མས་ཞི་བདེ་དང་འཆམ་མཐུན། བདེ་ཐབས་བཅས་ཀྱི་བཟང་པོའི་བསླབ་བྱ་དག་ང་ཚོར་གཅིག་མཚུངས་སྤྲོད་བཞིན་ཡོད་ཀྱང་། ད་ལྟ་ཆོས་ལུགས་ཀྱི་མིང་ཐོག་ནས་མང་ཚོགས་ཕན་ཚུན་བར་ཞེ་འཁོན་དང་འཁྲུག་འཛིང་ཤོར་བཞིན་ཡོད་སྟབས། ང་ཚོས་མཉམ་རུབ་ཐོག་ལས་དོན་བསྒྲུབ་ནས་དཀའ་རྙོག་དེ་དག་སེལ་ཐུབ་པ་བྱས་ཏེ། འཛམ་གླིང་ནང་ཞི་བདེ་བསྐྲུན་ཐུབ་པ་བྱེད་དགོས། ཞེས་གསུངས་འདུག   ཀུམ་བྷ་མེ་ལ་ཟེར་བའི་ཧིན་དུའི་ཆོས་ཕྱོགས་ཀྱི་འདུ་འཛོམས་ཆེན་མོ་དེ་ནི། ཆུ་མོ་གང་གཱ་དང་ཡ་མུ་ན་གཙང་པོ་གཉིས་མཉམ་དུ་འདུས་པའི་སངྒམ་གཙང་ཆུ་དང་། ས་ར་སི་ཝ་ཏི། སྒོ་དྷ་ཝ་རི། ཤིཔ་ར་བཅས་ཀྱི་གཙང་པོ་རྒྱུག་ཡུལ་གྱི་གྲོང་ཁྱེར་ཆེ་ཁག་ཨི་ལཱ་བཱཌ་ (Allahabad) དང་ཨུ་ཇན་ (Ujjain)། ཧཱ་རི་ཌ་ཝར་ (Haridwar) དང་ནཱ་སིག་ (Nasik) བཅས་སུ་ལོ་གསུམ་རེའི་མཚམས་སུ་རེས་མོས་ཀྱིས་གྲོང་ཁྱེར་དེ་དག་ནང་ཐེངས་རེ་ཚོགས་བཞིན་ཡོད། ཐེངས་འདིའི་འདུ་འཛོམས་དེ་ནི་སྒོ་དྷ་ཝ་རིའི་གཙང་པོ་རྒྱུག་ཡུལ་ན་སིག་གྲོང་ཁྱེར་ནང་ཚོགས་བཞིན་ཡོད་པ་མ་ཟད། དེར་མི་གྲངས་འབུམ་བརྒྱ་བརྒལ་བ་འདུ་འཛོམས་བྱ་རྒྱུའི་ཚོད་དཔག་བྱེད་བཞིན་འདུག   གསར་འགོད་པ། དོན་གྲུབ།
