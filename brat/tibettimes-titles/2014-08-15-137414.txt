རྒྱ་གར་རང་བཙན་ཉིན་མོ་སྲུང་གཙི།

དེ་རིང་ཟླ་ ༨ པའི་ཚེས་ ༡༥ ཉིན་གྱི་ཞོགས་པའི་ཆུ་ཚོད་ ༩།༠༠ བའི་ཐོག་དབུས་བོད་མིའི་སྒྲིག་འཛུགས་བཀའ་ཤག་མདུན་ཐང་དུ། སྲིད་སྐྱོང་བློ་བཟང་སེང་གེ་མཆོག་གཙོས་ཕྱག་རོགས་བཀའ་བློན་རྣམ་པ་དང་། བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་ལྷན་ཁང་ཁག་གི་ཟུང་དྲུང་ཡན་གྱི་སྲི་ཞུ་ལས་བྱེད་བཅས་འདུ་འཛོམས་ཀྱིས་རྒྱ་གར་གྱི་རྒྱལ་གླུ་གྱེར་འདོན་དང་སྦྲགས། སྲིད་སྐྱོང་མཆོག་གིས་རྒྱ་གར་གྱི་རྒྱལ་དར་མཁའ་དབྱིངས་སུ་སྒྲེང་འཛུགས་བཅས་གནང་སྟེ། གནས་ཡུལ་རྒྱ་གར་གཞུང་དབྱིན་ཡུལ་བཙན་རྒྱལ་རིང་ལུགས་ཀྱིས་མི་ལོ་ཉིས་བརྒྱ་ལྷག་རིང་བདག་བཟུང་བྱས་པའི་འོག་ནས་རང་བཙན་ཐོབ་སྟེ་མི་ལོ་ ༦༨ འཁོར་བའི་དུས་དྲན་སྲུང་གཙི་གནང་སོང་། སྲིད་སྐྱོང་མཆོག་གིས་དུས་དྲན་མཛད་སྒོའི་ཐོག་གསུངས་དོན།Â  དེ་རིང་རྒྱ་གར་རང་བཙན་ཐོབ་ནས་ལོ་ ༦༨ འཁོར་བའི་ཉིན་མོར་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱིས་ཚབ་ཞུས་ཏེ་རྒྱ་གར་གཞུང་དང་མི་མང་ཚང་མར་འཚམས་འདྲིའི་བཀྲ་ཤིས་བདེ་ལེགས་ཞུ་རྒྱུ་དང་། དམིགས་བསལ་སྐུ་ཞབས་མུ་དྷིར་ (Nerandra Modi) གིས་ད་ལམ་རྒྱ་གར་སྲིད་བློན་གྱི་འགན་དབང་བཞེས་ནས་ལོ་དང་པོ་ཡིན་དུས་སྐབས། ཁོང་དང་ཁོང་གི་བཀའ་ཤག་ལ་བོད་མི་ཡོངས་ནས་འཚམས་འདྲི་དང་བྱ་བ་ལམ་ལྷོང་ཡོང་བའི་སྨོན་འདུན་ཡོད་ཅེས་ཞུ་རྒྱུ་ཡིན། ཞེས་གསུངས་སོང་། ཁོང་གིས་མུ་མཐུད་དུ། ད་ལྟའི་བར་ང་ཚོ་བོད་པ་ཉག་ཕྲ་བའི་གནས་སྐབས་ལ་རྒྱ་གར་གཞུང་དམངས་ལྟར་བཀའ་དྲིན་ཆེ་བ་གཞན་སུ་ཡང་ཡོད་པ་མ་རེད། བོད་མིའི་སྒྲིག་འཛུགས་ད་ལྟའི་བར་དུ་གནས་ཐུབ་པ་དང་གཞིས་ཆགས་དང་དགོན་པ། སློབ་གྲྭ སྨན་ཁང་སོགས་གང་ས་ཅི་ཐད་དུ་རྒྱ་གར་གཞུང་གིས་མཐུན་རྐྱེན་གནང་ཡོད་པ་རེད། འདི་དག་རྒྱ་གར་གྱི་གནད་ཡོད་མི་སྣ་མང་པོས་སྐུ་ལས་སྐྱོན་པའི་བཀའ་དྲིན་ལ་བརྟེན་ནས་བྱུང་ཡོད་སྟབས། ཁོང་རྣམ་པ་ཚོས་ང་ཚོར་ཕྱག་རོགས་དང་མཐུན་སྦྱོར་གནང་བར་ཐུགས་རྗེ་ཞུ་རྒྱུ་ཡིན་ཞེས་གསུངས་སོང་། གངས་ཅན་སྐྱིད་གཤོང་གི་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱིས་རྒྱ་གར་གྱི་རང་བཙན་ཉིན་མོ་སྲུང་གཙི་གནང་བའི་ལས་རིམ་གྲུབ་རྗེས། སྲིད་སྐྱོང་མཆོག་བློ་བཟང་སེང་གེ་མཆོག་དྷ་ཤོད་ས་གནས་སྲིད་གཞུང་གིས་མགྲོན་འབོད་གནང་བ་ལྟར་དྷ་ཤོད་ཀ་ཅཱ་རིའི་ཉེན་རྟོག་རྩེད་ཐང་དུ་ཕེབས་ཏེ་རྒྱ་གར་རང་བཙན་ཉིན་མོ་སྲུང་གཙི་གནང་ཡོད་འདུག དེ་བཞིན་ངྷ་རམ་ས་ལའི་གཞུང་འབྲེལ་མ་ཡིན་པའི་ཚོགས་པ་ཁག་གིས་ཀྱང་དྷ་ས་མེག་གྷན་དུ་རྒྱ་གར་ཉིན་མོ་སྲུང་གཙི་གནང་ཡོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
