ཉེས་ཅན་ཆིག་སྟོང་ལྷག་སྲོག་ཐོག་བཏང་།

འདས་པའི་ལོ་ངོ་གཉིས་རིང་ཨེ་རན་སྲིད་གཞུང་གིས་རྒྱལ་ནང་གི་གནས་འཕྲིན་བརྒྱུད་ལམ་ཁག་བཀག་སྡོམ་དང་ལོ་ཆུང་བྱིས་པ་ཚོ་ཁྲིམས་སར་འབོད་འགུག་གིས་སྲོག་ཐོག་ཏུ་གཏོང་བཞིན་ཡོད་འདུག དེ་རིང་ཨེ་ཕེ་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཨེ་རན་རྒྱལ་ནང་དུ་འདས་པའི་མི་ལོ་གཉིས་རིང་ཉེས་ཅན་མི་སྣ་ཆིག་སྟོང་ལྷག་སྲོག་ཐོག་ཏུ་བཏང་ཡོད་པ་དང་། དེའི་གྲས་ཀྱི་མི་ ༧༣ བྱིས་པ་ཡིན་འདུག མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ཚན་པའི་ཆེད་མཁས་པ་ཨ་ཧ་མི་ཌ་ཤཱ་ཧི་ཌ་ (Ahmed Shaheed) ཟེར་བ་ཞིག་གིས། ཨེ་རན་རྒྱལ་ནང་དུ་ཕྱི་ལོ་ ༢༠༡༤ ལོར་ཉེས་ཅན་ ༧༥༣ དང་ཕྱི་ལོ་ ༢༠༡༥ ལོར་ཉེས་ཅན་ ༩༦༦ སྲོག་ཐོག་ཏུ་བཏང་ཡོད་པའི་གྲས་ཀྱི་ཉེས་ཅན་ ༧༣ བྱིས་པ་ཡིན་ཞིང་། ད་ཆར་ཉེས་ཅན་ ༡༦༠ སྲོག་ཐོག་ཏུ་གཏོང་རྒྱུ་ཡིན་པ་མ་ཟད། ཨེ་རན་སྲིད་གཞུང་གིས་ལོ་ ༡༨ ལ་མ་སོན་པའི་བྱིས་པ་ཚོར་སྲོག་ཆད་གཅོད་རྒྱུ་ཚབས་ཆེ་རུ་འགྲོ་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་ཨེ་རན་ནང་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལྟ་ཞིབ་ཚན་པའི་འབྲེལ་མཐུད་པས། ན་ནིང་གཅིག་པུར་ཨེ་རན་རྒྱལ་ནང་དུ་གསར་འགོད་པ་ ༤༧ འཛིན་བཟུང་བཀག་བསྐྱིལ་བྱས་ཡོད་པ་དང་། གནས་འཕྲིན་དྲ་རྒྱ་ ༢༧༠ སྒོ་བརྒྱབ་ཡོད་སྐོར་བརྗོད་འདུག གཞན། རྒྱལ་སྤྱིའི་ཉེས་ཡངས་ཚོགས་པས། ཨེ་རན་སྲིད་གཞུང་གིས་རྒྱལ་སྤྱིའི་རྩ་ཁྲིམས་རྩིས་མེད་བཏང་སྟེ་བུ་མོ་ལོ་ ༩ ཡས་མས་དང་བུ་ལོ་ ༡༥ ཡས་མས་རིགས་ཁྲིམས་སར་འབོད་འགུག་གིས་སྲོག་ཐོག་ཏུ་གཏོང་བཞིན་པར་སྐྱོན་བརྗོད་ཤུགས་ཆེན་བྱས་འདུག གནས་ཚུལ་ལྟར་ན། ད་ལོའི་ཕྱི་ཟླ་དང་པོའི་ནང་བཏོན་པའི་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་འགྲོ་བ་མིའི་ཐོབ་ཐང་སྙན་ཐོ་ཞིག་གི་ནང་། རྒྱལ་སྤྱིའི་ཉེས་ཡངས་ཚོགས་པ་དང་འགྲོ་བ་མིའི་ཐོབ་ཐང་ལྟ་ཞིབ་ཚན་པ་གཉིས་ནས། ཨེ་རན་རྒྱལ་ཁབ་ནི་འཛམ་གླིང་སྟེང་གི་ལོ་ཆུང་བྱིས་པ་ཉེས་ཅན་སྲོག་ཐོག་ཏུ་བཏང་བ་ཆེས་མང་ཤོས་དེ་ཆགས་ཀྱི་ཡོད་སྐོར་འཁོད་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
