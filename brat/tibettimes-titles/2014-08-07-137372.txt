རྒྱ་གར་སྲིད་བློན་བལ་ཡུལ་དུ་གཟིཊ་སྐོར།

ཁ་སང་དང་དེ་རིང་ཉིན་གཉིས་རིང་རྒྱ་གར་སྲིད་བློན་བལ་ཡུལ་ནང་གཟིགས་སྐོར་བྱས་ཏེ་བལ་ཡུལ་ལ་རོགས་དངུལ་སྤྲོད་པ་དང་རྒྱལ་ཁབ་གཉིས་བར་ཚོང་དོན་དང་ཚན་རྩལ་ཐོག་གྲོས་མཐུན་མང་པོ་ཞིག་བཞག་ཡོད་འདུག ཁ་སང་དང་དེ་རིང་ཉིན་གཉིས་རིང་རྒྱ་གར་གྱི་སྲིད་བློན་ནཱ་རེན་ཌ་མུ་དྷེ་བལ་ཡུལ་དུ་བསྐྱོད་དེ་བལ་ཡུལ་གྱི་སྲིད་འཛིན་རམ་བྷ་རན་ (Ram Baran Yadav)དང་། སྲིད་བློན་སོ་ཤེལ་ཀོ་རེ་ལཱ་ ( Susshil Koirala)། སྲིད་བློན་གཞོན་པ་བྷམ་གྷོ་དྷམ་ (Bam Dev Gautam) སོགས་དང་ཐུག་འཕྲད་ཀྱིས་བལ་ཡུལ་ནང་དཀའ་ངལ་ཆེན་པོ་འཕྲད་བཞིན་པའི་སྐེ་ལྦའི་ནད་གཞི་འགོག་ཐབས་ཆེད་བལ་སྒོར་ས་ཡ་ ༦༩ ཡི་རོགས་རམ་དང་། བལ་ཡུལ་དང་རྒྱ་གར་བར་ཆུ་མཛོད་གློག་ཁང་གསར་རྒྱག་ལས་གཞི། བལ་ཡུལ་གྱི་བརྙན་འཕྲིན་ཚན་པ་དང་རྒྱ་གར་གྱི་ཌུར་དྷ་ཤིན་(Doordarshan) བར་མཉམ་ལས་བྱེད་ཕྱོགས་བཅས་ལ་མིང་རྟགས་བཀོད་ཡོད་འདུག ད་དུང་། རྒྱ་གར་གཞུང་གིས་ ༡༩༥༠ ལོར་བཞག་པའི་རྒྱ་གར་དང་བལ་ཡུལ་ཀྱི་མཛའ་འབྲེལ་གྲོས་མཐུན་ (India-Nepal friendship treaty) ལ་བསྐྱར་ཞིབ་བྱ་རྒྱུ་དང་། ཁོར་ཡུག་དང་ཡུལ་སྐོར་སྤྲོ་འཆམ། ཚོང་དོན་དང་བདེ་འཇགས་བཅས་ཀྱི་ཐད་གྲོས་མོལ་བྱས་ཡོད་པ་མ་ཟད། ཁ་སང་ཉིན་རྒྱབ་ནཱ་རེན་ཌ་མུ་དྷེ་བལ་ཡུལ་གྲོས་ཚོགས་ནང་ཕེབས་ཏེ་སྐར་མ་ ༤༥ ཙམ་རིང་གཏམ་བཤད་བྱས་ཡོད་པ་དང་། ཁོང་གིས་བལ་ཡུལ་འཕེལ་རྒྱས་གཏོང་ཐབས་ (HIT formula ) ཞེས་པ་ཞིག་བསྒྲགས་ཏེ། རྒྱ་གར་གཞུང་གིས་བལ་ཡུལ་གྱི་གཞུང་ལམ་འཛུགས་སྐྲུན་དང་བརྡ་འཕྲིན་འཕེལ་རྒྱས། གློག་ལམ་རྒྱུད་གཏོང་བཅས་ཀྱི་ཐད་ཨ་སྒོར་ཐེར་འབུམ་ ༡ གི་སྐྱེད་མེད་བུ་ལོན་གཡར་ཡོད་འདུག འདི་ནི་ལོ་ངོ་ ༡༧ རྗེས་རྒྱ་གར་གྱི་སྲིད་བློན་བལ་ཡུལ་ནང་ཕེབས་ཐེངས་ཐོག་མ་དེ་ཆགས་ཡོད་འདུག སྲིད་བློན་མུ་དྷེ་ཡིས། “རྒྱ་གར་དང་བལ་ཡུལ་གྱི་འབྲེལ་ལམ་ནི་ཧི་མ་ལ་ཡ་དང་ཆུ་བོ་གངྒཱ་ལས་རྙིང་པ་ཆགས་ཡོད་སྟབས། ང་ཚོའི་འབྲེལ་ལམ་དེ་ཡིག་ཐོག་ཏུ་གང་འཁོད་ཙམ་ལ་གཞི་མ་བཅོལ་བར། མི་དང་མིའི་བར་གྱི་འདུ་ཤེས་ལས་བྱུང་བའི་མཛའ་མཐུན་འབྲེལ་ལམ་ཞིག་ཡོང་བ་བྱ་དགོས” ཞེས་དང་། ད་དུང་ཁོང་གིས། “ད་ལྟ་ངའི་སེམས་ནང་བལ་ཡུལ་ལ་རྡོག་རྒྱག་ (hit) ཅིག་ཞུ་རྒྱུ་བྱུང་ན་བསམ་གྱི་འདུག ངའི་སྐད་ཆ་དེའི་ཚིག་ངོས་ནས་ཁྱེད་ཚོར་ཕོག་ཐུག་འགྲོ་སྲིད་ཀྱང་། ངས་ང་ཚོས་རོགས་རམ་བྱས་ཏེ་བལ་ཡུལ་གྱི་གཞུང་ལམ་དང་། བརྡ་འཕྲིན་ཚན་རྩལ་སོགས་འཕེལ་རྒྱས་མདུན་སྐྱོད་ཤུགས་ཆེ་ཙམ་གཏོང་འདོད་པ་དེ་ཡིན” ཞེས་སོགས་བརྗོད་འདུག བལ་ཡུལ་གྱི་སྲིད་འཛིན་རམ་བྷ་རན་གྱིས།  “ཁ་སང་ཁྱེད་ཀྱིས་བལ་ཡུལ་གྲོས་ཚོགས་ནང་གཏམ་བཤད་སྤེལ་བ་དེས། ང་ཚོའི་བློ་སེམས་གཉིས་ཆ་འགུག་ཐུབ་པ་བྱུང་ཡོད་སྟབས། མ་འོངས་པར་རྒྱལ་ཁབ་གཉིས་འབྲེལ་ལམ་ཡར་རྒྱས་སུ་འགྲོ་བར་རྨང་གཞི་བརྟན་པོ་ཞིག་བསྐྲུན་ཡོད།”ཅེས་དང་། བལ་ཡུལ་སྲིད་བློན་གྱིས། “རྒྱ་གར་སྲིད་བློན་བལ་ཡུལ་ནང་ཕེབས་པ་དེས། རྒྱལ་ཁབ་གཉིས་བར་འཕྲད་བཞིན་པའི་དཀའ་ངལ་མང་དག་ཅིག་སེལ་ཡོད་སྟབས། མ་འོངས་པར་ལས་འཆར་གང་མང་ཐོག་མཉམ་ལས་བྱ་རྒྱུ་ཡིན” ཞེས་བརྗོད་འདུག ༡༩༥༠ ལོར་རྒྱ་གར་དང་བལ་ཡུལ་བར་མཛའ་འབྲེལ་གྲོས་མཐུན་ (India-Nepal friendship treaty) ཞེས་པ་བཞག་སྟེ་ཚོང་དོན་དང་བདེ་སྲུང་། བལ་ཡུལ་གྱི་མི་སེར་རྒྱ་གར་གྱི་མི་སེར་གཅིག་མཚུངས་ལྟར་རྒྱ་གར་ནང་སློབ་སྦྱོང་དང་ལས་ཀའི་ཐོབ་ཐང་སྤྲོད་རྒྱུ་དང་། ཆ་རྐྱེན་གང་ཡང་མེད་པའི་ཐོག་རྒྱལ་ཁབ་གཉིས་ཀྱི་མི་སེར་རྒྱལ་མཚམས་ཕར་ཚུར་འགྲོ་འོང་ཆོག་པ་བཅས་བྱུང་ཡོད་ཀྱང་། ༡༩༦༢ ལོར་རྒྱ་དཀར་ནག་བར་དམག་འཁྲུག་བྱུང་རྗེས་བལ་ཡུལ་གྱིས་རྒྱ་ནག་དང་མཉམ་ལས་ཤུགས་ཆེན་བྱས་ཤིང་། ༡༩༧༥ ལོར་འབྲས་ལྗོངས་རྒྱ་གར་ཁོངས་སུ་བཅུག་པ་སོགས་ཀྱིས་འབྲེལ་ལམ་ལ་དོ་ཕོག་ཕྱིན་ཡོད། ༡༩༩༠ དང་ ༡༩༩༥ ལོར་རྒྱ་གར་རྒྱལ་ས་ལྡི་ལིར་བལ་ཡུལ་གྱི་སྲིད་བློན་བྷ་ཋ་རེར་ (Krishna Prasad Bhattarai) དང་རྒྱ་གར་གྱི་སྲིད་བློན་ཝུ་པི་སིང་ (V.P. Singh) ཐུག་འཕྲད་དང་། ༡༩༩༡ ལོར་བལ་ཡུལ་གྱི་སྲིད་བློན་ཀོ་རེ་ལ་ (Girija Prasad Koirala) མཆོག་རྒྱ་གར་ནང་སྐོར་སྐྱོད་བྱས་ཏེ་ཚོང་དོན་དང་བདེ་སྲུང་། ས་མཚམས་བཅས་ལ་གྲོས་ཆོད་བཞག་སྟེ་རྒྱལ་ཁབ་གཉིས་ཀྱི་འབྲེལ་ལམ་བསྐྱར་གསོ་ཕྲན་བུ་བྱུང་ཡོད་ཀྱང་། ༢༠༠༡ ལོའི་ཟླ་ ༦ ཚེས་ ༡ ཉིན་བལ་ཡུལ་གྱི་རྒྱལ་པོ་དང་བཙུན་མོ་སོགས་དམར་གསོད་དོན་རྐྱེན་བྱུང་སྟེ། མའོ་དམར་ཤོག་ཚོགས་པ་དང་མི་མང་བར་ཟིང་འཁྲུག་བྱུང་བ་དང་། རྒྱ་ནག་དང་བལ་ཡུལ་བར་ལོ་རེར་ཨ་སྒོར་ཐེར་འབུམ་ ༡༠ ལྷག་གི་ཚོང་འབྲེལ་བྱེད་བཞིན་པ་བཅས་ཀྱིས་རྒྱ་གར་ལ་ཕོག་ཐུག་ཆེན་པོ་བཟོས་ཡོད། སྐབས་སྔོན་མའི་བྷེ་ཇེ་པི་ཚོགས་པའི་མཉམ་སྡེབ་སྲིད་གཞུང་ ( NDA) གི་ཕྱི་སྲིད་བློན་ཆེན་ཇཱ་སེ་ཝང་སིང་ (Jaswant Singh)གིས། སྐབས་དེའི་བལ་ཡུལ་གྱི་སྲིད་བློན་ཤེར་བྷ་ཧ་དྷུར་ (Sher Bahadur Deuba )དང་ ༡༩༥༠ ལོར་བཞག་པའི་རྒྱལ་ཁབ་གཉིས་བར་གྱི་གྲོས་མཐུན་བསྐྱར་གསོ་ཡོང་ཐབས་བྱས་བར་བལ་ཡུལ་སྲིད་བློན་གྱིས་ངོས་ལེན་བྱས་མེད་ལ། ༢༠༠༧ ལོར་རྒྱ་གར་དང་འབྲུག་བར་གྲོས་མཐུན་འཇོག་སྐབས་ཀྱང་བལ་ཡུལ་གྱིས་ཆ་ཤས་བླངས་མེད། ༢༠༠༩ ལོར་བལ་ཡུལ་དམར་ཤོག་ཚོགས་པའི་མགོ་ཁྲིད་དང་སྲིད་བློན་གཅིག་ལྕོགས་པར་ཅན་ཌ་(Prachanda) རྒྱ་གར་ནང་སྐོར་སྐྱོད་ཀྱིས་རྒྱ་གར་སྲིད་བློན་དང་གྲོས་མཐུན་ཁག་ ༢༢ ཙམ་ལ་མིང་རྟགས་བཀོད་པ་མ་ཟད། རྒྱ་གར་གཞུང་གིས་བལ་ཡུལ་ལ་ཧིན་སྒོར་ཐེར་འབུམ་ ༡ ལྷག་དང་། ༢༠༡༠ ལོར་ཨ་སྒོར་ས་ཡ་ ༥༠ ཡི་རོགས་དངུལ་སྤྲད་ཡོད་སྟབས། རྒྱལ་ཁབ་གཉིས་ཀྱི་འབྲེལ་ལམ་སླར་གསོ་བྱུང་ཡོད། ཆབ་སྲིད་དབྱེ་ཞིབ་པ་ཚོས། ལོ་ ༡༧ རྗེས་རྒྱ་གར་གྱི་སྲིད་བློན་བལ་ཡུལ་ནང་སྐོར་སྐྱོད་གནང་པ་དེས། རྒྱལ་ཁབ་གཉིས་ཀྱི་འབྲེལ་ལམ་ལ་བཟང་ཕྱོགས་ཀྱི་རྣམ་པ་ཐོན་ཡོད་ཀྱང་། ད་ལྟ་བལ་ཡུལ་ནང་དམར་ཤོག་ཚོགས་པའི་སྟོབས་ཤུགས་ཆེན་པོ་ཡོད་པ་དེས་རྒྱལ་ཁབ་གཉིས་ཀྱི་མ་འོངས་པའི་འབྲེལ་ལམ་ལ་དཀའ་ངལ་བཟོ་སྲིད་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
