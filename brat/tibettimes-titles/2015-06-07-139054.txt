ཨང་སིན་སུའུ་ཀེ་རྒྱ་ནག་ལ་ཕེབས་རྒྱུ།

ཟླ་འདིའི་ཚེས་ ༡༠ ནས་ ༡༤ ཡི་བར་འབར་མའི་དམངས་གཙོ་ཚོགས་པའི་དབུ་ཁྲིད་ལྕམ་ཨང་སིན་སུའུ་ཀེ་རྒྱ་ནག་ཏུ་གཟིགས་སྐོར་གནང་རྒྱུ་རེད།   གཟའ་པ་སངས་ཉིན། འབར་མ་གཞུང་གི་ཁ་གཏད་ཚོགས་པ་དང་རྒྱལ་ཡོངས་དམངས་གཙོ་ཚོགས་པའི་མགྲིན་ཚབ་པ་ནེན་ཝིན་ (Nyan Win) གྱིས། ཟླ་འདིའི་ཚེས་ ༡༠ ནས་ ༡༤ ཉིན་བར་འབར་མའི་དམངས་གཙོ་ཚོགས་པའི་ཚོགས་གཙོ་ལྕམ་ཨང་སིན་སུའུ་ཀེ་རྒྱ་ནག་ལ་གཟིགས་སྐོར་དུ་འགྲོ་རྒྱུ་ཡིན་པ་དང་སྐབས་དེར་རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་དང་སྲིད་བློན་ལི་ཁེ་ཆང་སོགས་ལ་མཇལ་འཕྲད་བྱ་རྒྱུ་ཡིན་པ་བསྒྲགས་འདུག དེའི་རྗེས་རྒྱ་ནག་ཕྱི་འབྲེལ་ལྷན་ཁང་གི་དྲ་བའི་སྟེང་། རྒྱ་ནག་གུང་ཁྲན་ཚོགས་པས་གདན་ཞུ་བྱས་པ་བཞིན། ཨང་སིན་སུའུ་ཀེ་ཡིས་སྣེ་ཁྲིད་བྱེད་པའི་འབར་མ་རྒྱལ་ཡོངས་དམངས་གཙོ་ཚོགས་པའི་འཐུས་མི་རྣམས་ཟླ་ ༦ ཚེས་ ༡༠ ནས་ཚེས་ ༡༤ བར་རྒྱ་ནག་ལ་གཟིགས་སྐོར་དུ་ཕེབས་རྒྱུ། ཞེས་འཁོད་འདུག་ཀྱང་། གཟིགས་སྐོར་སྐབས་རྒྱ་ནག་གི་མི་སྣ་སུར་ཐུག་རྒྱུ་ཡིན་མིན་གསལ་བོ་བཀོད་མི་འདུག གཟའ་པ་སངས་ཀྱི་ཕྱི་དྲོར་རྒྱ་ནག་ཕྱི་འབྲེལ་ལྷན་ཁང་གི་མགྲིན་ཚབ་པ་ཧོ་ལེ་ (Hong Lei) ཡིས། ད་རེབ་འབར་མའི་དམངས་གཙོ་ཚོགས་པའི་ཚོགས་གཙོ་ཨང་སིན་སུའུ་ཀེ་རྒྱ་ནག་ལ་གཟིགས་སྐོར་དུ་ཕེབས་རྒྱུ་དེ་རྒྱ་ནག་དང་འབར་མའི་བར་གྱི་འབྲེལ་ལམ་ཡར་རྒྱས་འགྲོ་རྒྱུའི་རེ་བ་ཡོད་ཚུལ་བརྗོད་དོན། “ཚོགས་པ་དང་ཚོགས་པའི་བར་འབྲེལ་བ་བྱ་རྒྱུ་དེ་ནི་རྒྱ་ནག་དང་འབར་མིའི་འབྲེལ་ལམ་ཁྲོད་ཀྱི་གལ་གནད་གཙོ་གྲས་ཤིག་ཡིན་པ་དང་། ཉེ་བའི་ལོ་མང་རིང་དམངས་གཙོ་ཚོགས་པ་གཙོས་བའི་འབར་མའི་ནང་གི་ཚོགས་པ་ཚང་མར་ང་ཚོས་འབྲེལ་བ་རྒྱུན་འཁྱོངས་ཡག་པོ་བྱེད་བཞིན་ཡོད་ལ། ད་རེབ་དམངས་གཙོ་ཚོགས་པའི་གཟིགས་སྐོར་དེའང་། ཕྱོགས་གཉིས་ཀྱིས་མཐུན་ལམ་ཤེས་རྟོགས་ཐོག་ནས་རྒྱ་ནག་དང་འབར་མའི་བར་གྱི་མཛའ་འབྲེལ་ཡར་རྒྱས་གཏང་རྒྱུའི་རེ་བ་ཡོད།” ཅེས་བརྗོད་འདུག ཁ་སང་བྷེ་བྷེ་སི་ཡིས་གནས་ཚུལ་སྤེལ་བའི་ནང་། བྷེ་བྷེ་སིའི་ཨེ་ཤ་ཡ་དང་ཞི་བདེ་རྒྱ་མཚོའི་ས་ཁུལ་གྱི་རྩོམ་སྒྲིག་པ་ཀྲུའུ་ཧྥེ་ཕིན་ (Zhou Feipeng) གྱིས། ནོ་སྦེལ་ཞི་བདེ་གཟེངས་རྟགས་ཐོབ་མཁན་གྱི་ཨང་སིན་སུའུ་ཀེ་རྒྱ་ནག་ནང་གཟིགས་སྐོར་སྐབས་ཁོ་མོས་གལ་སྲིད་རྒྱ་ནག་ནོ་སྦེལ་ཞི་བདེ་གཟེངས་རྟགས་ཐོབ་མཁན་ལུའུ་ཞོ་པོ་ (Liu Xiaobo) ཡི་སྐོར་གླེང་མ་ཐུབ་པ་ཡིན་ན། ཁོ་མོ་རང་གི་ཆེ་མཐོང་ཀུན་སྤྱོད་ཡོངས་རྫོགས་ཆུད་ཟོས་སུ་འགྲོ་རྒྱུ་དང་། གལ་སྲིད་ཁོ་མོས་གླེང་སློང་བྱུང་བ་ཡིན་ན་རྒྱ་ནག་ལ་ཕོག་ཐུག་གཏང་འགྲོ་གི་རེད། ཅེས་བརྗོད་འདུག གནས་ཚུལ་དེའི་ནང་དུ། ད་ལོ་ཕྱི་ཟླ་ ༡༡ པའི་ནང་འབར་མའི་འོས་བསྡུ་ཆེན་མོའི་ཁྲོད་ཨང་སིན་སུའུ་ཀེ་ཡིས་དབུ་ཁྲིད་པའི་རྒྱལ་ཡོངས་དམངས་གཙོ་ཚོགས་པར་རྒྱལ་ཁ་ཐོབ་སྲིད་པ་དང་། གལ་སྲིད་དམག་མིའི་སྲིད་གཞུང་གིས་ཨང་སིན་སུའུ་ཀེ་ལ་འོས་ཤོག་ལྡུག་མི་ཆོག་པའི་ཁྲིམས་བཟོས་པ་ཡིན་ན། ཁོ་མོའི་དབང་ཤུགས་དང་ཤུགས་རྐྱེན་སྔར་ལས་རྒྱ་ཆེ་རུ་འགྲོ་རྒྱུ་ཡིན་ཚུལ་སོགས་འཁོད་འདུག རོའུ་ཊར་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། སྤྱིར་བཏང་ཕྱི་ལོ་ ༢༠༡༤ ལོའི་ཟླ་ ༡༢ པའི་ནང་ཨང་སིན་སུའུ་ཀེ་ཡི་རྒྱ་ནག་ལ་གཟུགས་སྐོར་དུ་ཕེབས་རྒྱུའི་འཆར་གཞི་ཡོད་ཀྱང་། གྲོས་མོལ་གནད་དོན་གྱི་དབང་གིས་ཕྱིར་འཐེན་བྱ་དགོས་བྱུང་ཡོད་འདུག ཅེས་འཁོད་འདུག ཨང་སིན་སུའུ་ཀེ་ལོ་ངོ་ ༢༡ རིང་འབར་མ་དམག་མིའི་སྲིད་གཞུང་གིས་བཟང་བཙོན་འོག་བཅུག་ཡོད་པ་དང་ཕྱི་ལོ་ ༡༩༩༡ ལོར་ཁོ་མོར་ནོ་སྦེལ་ཞི་བདེ་གཟེངས་རྟགས་ཐོབ་ཡོད། འབར་མའི་རང་དབང་འཐབ་རྩོད་པ་ཨང་སིན་སུའུ་ཀེ་དང་བོད་མིའི་སྐྱབས་མགོན་༧་གོང་ས་མཆོག་རྣམ་གཉིས་སྔ་ཕྱི་ཐེངས་གཉིས་ལ་མཇལ་འཕྲད་གནང་ཡོད་པ་དང་། ཐོག་མའི་མཇལ་འཕྲད་དེ་ཕྱི་ལོ་ ༢༠༡༢ ལོའི་ཟླ་ ༦ ཚེས་ ༢༢ ཉིན་དབྱིན་ཇིའི་རྒྱལ་ས་ལན་ཊོན་ནས་གནང་ཡོད་པ་དང་། སྐབས་དེར་ཁོང་གཉིས་གྱིས་གླེང་མོལ་སྐར་མ་ ༣༠ ཡས་མས་ལ་བྱས་ཡོད་འདུག ཐེངས་གཉིས་པ་ ༢༠༡༣ ལོའི་ཟླ་ ༩ ཚེས་ ༡༥ ཉིན་ཆེག་སྤྱི་མཐུན་རྒྱལ་ཁབ་ (Czech Republi) ཀྱི་རྒྱལ་ས་ཕརོ་གྷུའོ་ (Prague) ནང་དུས་རབས་ཉེར་གཅིག་པའི་སྤྱི་ཚོགས་འཕོ་འགྱུར་ཞེས་པའི་གླེང་སྟེགས་ཐེངས་ ༡༧ པ་སྐོང་ཚོགས་གནང་བའི་སྟེང་༧གོང་ས་མཆོག་དང་ཨང་སིན་སུའུ་ཀེ་གཉིས་མཉམ་ཞུགས་གནང་བའི་ཞོར་དུ་མཇལ་འཕྲད་གནང་ཡོད་འདུག   གསར་འགོད་པ། འཇིགས་མེད་རྒྱ་མཚོ།
