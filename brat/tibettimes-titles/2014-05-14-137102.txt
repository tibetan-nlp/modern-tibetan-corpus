རྒྱ་ནག་དང་ཨ་ཧྥི་རི་ཁའི་མཉམ་ལས།

རྒྱ་ནག་གི་སྲིད་བློན་ལི་ཨ་ཧྥི་རི་ཁའི་རྒྱལ་ཁབ་ཁེན་ཡའི་ནང་སྐོར་སྐྱོད་བྱས་ཏེ། མྱུར་མགྱོགས་མེ་འཁོར་ལྕགས་ལམ་བཟོ་སྐྲུན་གཙོས་གྲོས་མཐུན་དོན་ཚན་ ༡༧ ལ་མིང་རྟགས་བཀོད་ཡོད་འདུག ཟླ་འདིའི་ཚེས་ ༩ ནས་བཟུང་ཉིན་གསུམ་རིང་། རྒྱ་ནག་གི་སྲིད་བློན་ལི་ཁེ་ཆང་གིས་ཨ་ཧྥ་རི་ཁའི་རྒྱལ་ཁབ་ཁེན་ཡའི་སྲིད་འཛིན་ཁེན་ཡཱ་ཌ་ (Uhuru Kenyatta)དང་སྲིད་འཛིན་གཞོན་པ་རུ་ཌོ་ (William Ruto) གཙོས་གནད་ཡོད་མི་སྣ་ཁག་དང་ཐུག་འཕྲད་བྱས་ཏེ། རྒྱ་གཞུང་གིས་ཨ་ཧྥི་རི་ཁ་དང་རྒྱ་ནག་གི་མཉམ་སྡེབ་དངུལ་ཁང་འཕེལ་རྒྱས་གཏོང་ཆེད་ཨ་སྒོར་ཐེར་འབུམ་ ༢ དང་། ཁེན་ཡའི་རྒྱལ་ས་ནེ་རོ་བྷི་ནང་གསར་འཛུགས་བྱེད་བཞིན་པའི་ཨ་ཧྥི་རི་ཀ་དང་རྒྱ་ནག་གི་རིག་གཞུང་ཞིབ་འཇུག་ལྟེ་གནས་ཁང་དུ་ཁེན་ཡའི་དངུལ་ལོར་ཤི་ལིང་ ( Shilling) ཐེར་འབུམ་ ༥།༡ དང་། ཁོར་ཡུག་དང་རི་དྭགས་སྲུང་སྐྱོབ་ཆེད་ཤི་ལིང་ས་ཡ་ ༨༨༠། སོ་ནམ་ཐོན་ལས་ཡར་རྒྱས་ཆེད་ཤི་ལིང་ཐེར་འབུམ་ ༥།༡ ཁེན་ཡའི་སཱོ་མ་ལི་ཡའི་སྐྱབས་བཅོལ་བ་ཚོར་རོགས་རམ། རྡོ་སྣུམ་ཚོང་འབྲེལ། ཤེས་ཡོན་སློབ་གསོ་གོང་འཕེལ། འཕྲོད་བསྟེན་ལྟ་སྐྱོང་ལས་གཞི། རྒྱལ་ཁབ་གཉིས་ཀྱི་ཚོང་འབྲེལ་ཁེ་ཕན་དོ་མཉམ་ཡོང་ཐབས་སོགས་དོན་ཚན་ ༡༧ ལ་མིང་རྟགས་བཀོད་ཡོད་ཅིང་། ཁ་སང་ལི་ཁེ་ཆང་གིས་ཡུ་གྷན་ཌའི་སྲིད་འཛིན་མུ་སི་ཝུ་ནི་ (Yoweri Museveni)་དང་། ཌན་ཛ་ནི་ཡའི་སྲིད་འཛིན་ཇཱ་ཁཱ་ཡ་ (Jakaya Kikwete)། རུ་ཝན་ཌའི་སྲིད་འཛིན་པོལ་ (Paul Kagame)། ལྷོ་སུ་ཌན་གྱི་སྲིད་འཛིན་སེལ་ཝ་ (Salva Kiir)བཅས་ལྷན་འཛོམས་ཐོག ཁེན་ཡའི་རྒྱལ་ས་ནེ་རི་བྷི་ནས་མོམ་བྷ་སཱ་བར་ཤར་གཏོང་ཐུབ་པའི་མྱུར་མགྱོགས་མེ་འཁོར་ལྕགས་ལམ་བཟོ་སྐྲུན་ཆེད་ཤི་ལིང་ཐེར་འབུམ་ ༣༢༧ གྱི་རོགས་སྐྱོར་སྤྲོད་རྒྱུ་གསལ་བསྒྲགས་དང་འབྲེལ། ཨ་ཧྥི་རི་ཀའི་མངའ་ཁུལ་བརྟན་བརླིང་སྲུང་སྐྱོབ་དང་འཛུགས་སྐྲུན་འཕེལ་རྒྱས་སོགས་ཀྱི་སྐོར་ལ་གོ་བསྡུར་བྱས་ཡོད་འདུག འདིའི་ཐད་ལི་ཁེ་ཆང་གིས། “ རྒྱ་ནག་དང་ཁེན་ཡ་ནི་ཡར་རྒྱས་འགྲོ་བཞིན་པའི་རྒྱལ་ཁབ་ཅིག་ཆགས་ཡོད་སྟབས། རྒྱལ་ཁབ་གཉིས་པོས་གཅིག་ཕན་གཅིག་གྲོགས་བྱས་ཏེ། དཔལ་འབྱོར་འཕེལ་རྒྱས་གཏོང་རྒྱུ་ནི་གལ་འགངས་ཆེ་ཤོས་ཤིག་ཆགས་ཡོད་སྟབས། མ་འོངས་པར་སྔར་ལྷག་མཉམ་ལས་མཉམ་འབྲེལ་བྱས་ཏེ་དཔལ་འབྱོར་འཕེལ་རྒྱས་གཏོང་རྒྱུ་ཡིན།” ཞེས་དང་། ད་ཐེངས་སྐོར་སྐྱོད་འདི་གྲུབ་འབྲས་ལྡན་པ་ཞིག་བྱུང་ཡོད་སྐོར་སོགས་བརྗོད་འདུག ཁེན་ཡའི་སྲིད་འཛིན་ཁེན་ཡཱ་ཌ་ཡིས་ཀྱང་། “ རྒྱལ་སྤྱིའི་སྤྱི་ཚོགས་ཀྱིས་རྒྱ་ནག་གི་འཕེལ་རྒྱས་ལ་བལྟས་ཏེ། མ་འོངས་པར་རྒྱལ་ཁབ་འཛུགས་སྐྲུན་བྱེད་ཕྱོགས་ཐད་ཞིབ་འཇུག་མཐིལ་ཕྱིན་བྱ་དགོས་པ་ཞིག་ཆགས་ཡོད་སྟབས། རྒྱ་ནག་དང་མཉམ་འབྲེལ་བྱས་ཏེ་འཕུར་མཆོང་ལྟ་བུའི་འཕེལ་རྒྱས་གཏོང་ཐབས་བྱ་རྒྱུ་ཡིན།” ཞེས་བརྗོད་འདུག  ༡༩༦༣ ལོའི་ཟླ་ ༡༢ ཚེས་ ༡༤ ཉིན་ནས་བཟུང་རྒྱ་ནག་དང་ཁེན་ཡའི་ཐོག་མར་འབྲེལ་བ་བྱུང་ཡོད་ཅིང་། ༡༩༩༦ ལོར་རྒྱ་ནག་དམག་སྤྱི་ལེའུ་ཅང་སུང་ (Liu Jingsong) གིས་སྣེ་ཁྲིད་པའི་དམག་དཔུང་སྐུ་ཚབ་ཚོགས་ཆུང་ཞིག་ཁེན་ཡར་སྐོར་སྐྱོད་དང་། ༡༩༩༧ ལོར་ཁེན་ཡའི་གནམ་དམག་དམག་སྤྱི་ནིག་ལེ་ཤན་ (Nick Leshan)རྒྱ་ནག་ནང་སྐོར་སྐྱོད་བྱས་པ་བཅས་ལ་བརྟེན། རྒྱལ་ཁབ་གཉིས་ཀྱི་འབྲེལ་ལམ་ཡར་རྒྱས་ཕྱིན་ཏེ་ཁེན་ཡ་རྒྱ་ནག་གི་མཐུན་གྲོགས་རྒྱལ་ཁབ་ཅིག་ཆགས་ཡོད། ༢༠༠༢ ལོར་རྒྱ་ནག་དང་ཁེན་ཡ་བར་ཨ་སྒོར་ས་ཡ་ ༡༨༦།༣༧ ཚོང་འབྲེལ་བྱས་ཡོད་ཅིང་། དེ་ལས་རྒྱ་གཞུང་གིས་ཁེན་ཡར་ཨ་སྒོར་ས་ཡ་ ༡༨༠།༥༧༦ གི་ཕྱིར་ཚོང་དང་། ཁེན་ཡས་རྒྱ་ནག་ལ་ཨ་སྒོར་ས་ཡ་ ༥།༧༩༨ ཀྱི་ཚོང་འབྲེལ་བྱས་ཡོད་འདུག ༢༠༠༥ ལོར་ཁེན་ཡའི་སྲིད་འཛིན་ཀི་བྷ་ཀི་ (Mwai Kibaki)རྒྱ་ནག་ནང་སྐོར་སྐྱོད་དང་། ༢༠༠༦ ལོར་རྒྱ་ནག་གི་སྲིད་འཛིན་ཧུའུ་ཅིང་ཐའོ་ཁེན་ཡར་སྐོར་སྐྱོད་བྱས་ཏེ། ཨ་སྒོར་ས་ཡ་ ༡༤༥།༦༨ ཡི་ཚོང་དོན་གྲོས་ཆོད་བཞག་པ་དང་། ༢༠༠༦ ནས་ ༢༠༡༠ བར་ཨ་སྒོར་ས་ཡ་ ༥༣།༣༢ ལྷག་གི་ཚོང་འབྲེལ་བྱས་ཡོད་འདུག ༢༠༠༠ ནས་ ༢༠༡༤ ལོའི་བར་རྒྱ་གཞུང་གིས་ཨ་སྒོར་ས་ཡ་ ༡༠༨ ཀྱི་འགྲོ་གྲོན་བཏང་སྟེ། ཁེན་ཡའི་ལས་གྲྭ་ ༧༠ ཙམ་ལ་རོགས་སྐྱོར་དང་། ད་ལྟ་ཁེན་ཡར་རྒྱ་ནག་གི་ཁེ་ལས་ ༥༠ ལྷག་གིས་ཤི་ལིང་ཐེར་འབུམ་ ༡༧༢ ཀྱི་མ་གནས་ལས་གྲྭ་ ༨༠ ཙམ་ཞིག་ལ་ཐེ་ཇུས་བྱེད་བཞིན་ཡོད་པ་མ་ཟད། ༢༠༡༡ ལོར་རྒྱལ་ཁབ་གཉིས་བར་ཨ་སྒོར་ས་ཡ་ ༥༢༣།༥ དང་། ༢༠༡༢ ལོར་ཨ་སྒོར་ཐེར་འབུམ ༢།༧༦༣ ། དེ་བཞིན། ༢༠༡༣ ལོར་ཨ་སྒོར་ཐེར་འབུམ་ ༣།༢༦༥ ཡི་ཚོང་འབྲེལ་བྱས་ཡོད་འདུག ཆབ་སྲིད་དབྱེ་ཞིབ་པ་ཚོས། ད་ལྟ་རྒྱ་ནག་དང་ཨ་རིའི་འབྲེལ་ལམ་ཇེ་སྡུག་ཏུ་འགྲོ་བཞིན་པ་དང་བསྟུན་ཏེ། རྒྱ་ནག་གི་སྲིད་བློན་ལི་ཁེ་ཆང་ཨ་ཧྥི་རི་ཀའི་རྒྱལ་ཁབ་ཁག་ཏུ་སྐོར་སྐྱོད་བྱས་ཏེ། ཚོང་དོན་གྲོས་མཐུན་བཞག་པ་དེ་ནི་ཨ་རིར་ཁ་གཏད་གཅོག་ཐབས་ཀྱི་སྲིད་ཇུས་ཤིག་ཡིན་སྐོར་བརྗོད་འདུག ལི་ཁེ་ཆང་གི་བདུན་ཕྲག་གཅིག་རིང་གི་ཨ་ཧྥི་རི་ཀའི་རྒྱལ་ཁབ་བཞིའི་སྐོར་སྐྱོད་དེ་མཇུག་སྒྲིལ་རྗེས། ཁ་སང་རྒྱ་ནག་ལ་ཕྱིར་ལོག་བྱས་ཡོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
