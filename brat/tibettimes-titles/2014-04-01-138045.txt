བྱིས་པ་ཁྲི་གཅིག་ལྷག་གཡུག་གི་འདུག

རྒྱ་ནག་ནང་ལོ་རེར་ཕ་མས་གཡུག་པའི་བྱིས་པ་ཆིག་ཁྲི་ལྷག་ཡོད་པ་དང་། བྱིས་པ་དེ་དག་ལས་བརྒྱ་ཆ་ ༩༩ ནད་པ་དང་དབང་པོ་སྐྱོན་ཅན་ཡིན་འདུག རོའུ་ཊར་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ད་ལོའི་ཕྱི་ཟླ་ ༡ ཚེས་ ༢༨ ཉིན་རྒྱ་ནག་ཀྲང་ཀྲིག་གྲོང་ཁྱེར་བྱིས་པའི་ཕན་བདེ་གླིང་གིས་བདག་མེད་བྱིས་པ་བསྡུ་ལེན་གྱི་ལས་གཞི་ཞིག་མགོ་འཛུགས་བྱས་ཡོད་ཀྱང་། བྱིས་པ་བདག་མེད་ཀྱི་གྲངས་འབོར་ཚད་ལས་བརྒལ་ཏེ་ལས་གཞི་མཚམས་འཇོག་བྱས་ཡོད་འདུག བདག་མེད་བྱིས་པ་བསྡུ་ལེན་བྱས་ནས་ཉིན་གྲངས་ ༤༨ ནང་ནད་གཞི་འདྲ་མིན་ཕོག་པའི་བྱིས་པ་ ༢༦༢ ཕན་བདེ་གླིང་དེར་བསྐྱལ་ཡོད་པ་དང་། བྱིས་པ་ཆུང་ཤོས་ཉིན་ ༢ དང་ཆེ་ཤོས་ལོ་ ༥ ཡིན་ལ། ཕན་བདེ་གླིང་དུ་བསླེབས་རྗེས་ནད་གཞི་ལྕི་མོ་ཕོག་པའི་བྱིས་པ་ ༢༣ འདས་གྲོངས་སུ་གྱུར་ཡོད་འདུག ཀོང་ཀྲིག་གྲོང་ཁྱེར་བྱིས་པའི་ཕན་བདེ་གླིང་གི་མགོ་ཁྲིད་ཞི་ཅོའུ་ཡིས། སྤྱིར་བཏང་ཕན་བདེ་གླིང་དེར་ཁྱོན་བྱིས་པའི་ཉལ་ཁྲི་ ༡༠༠༠ ལྷག་ཡོད་ཀྱང་། བདག་མེད་བྱིས་པའི་གྲངས་འབོར་ཚད་ལས་བརྒལ་ཏེ་བྱིས་པའི་ཉལ་ཁྲི་དང་ཡོ་ཆས། དེ་བཞིན་སྨན་བཅོས་སོགས་ཀྱི་ཆ་རྐྱེན་འདང་ངེས་མེད་སྟབས། བདག་མེད་བྱིས་པ་བསྡུ་ལེན་གྱི་ལས་གཞི་མཚམས་འཇོག་མི་བྱེད་ཐབས་མེད་བྱུང་སྐོར་བརྗོད་འདུག དེ་བཞིན། ཝུན་ཀྲུའུ་གྲོང་ཁྱེར་བྱིས་པའི་ཕན་བདེ་གླིང་གི་མགོ་ཁྲིད་གཞོན་པ་ཀྲང་ཧེ་ཡཱན་གྱིས་བྱིས་པའི་ཕན་བདེ་གླིང་དེར་བདག་མེད་བྱིས་པ་ ༣༠༠ ལྷག་ཡོད་ཀྱང་བྱིས་པའི་ཉལ་ཁྲི་ ༡༠༠ ལས་མེད་སྟབས། ཕྱི་ལོགས་སུ་ཁང་པ་གླས་ནས་བྱིས་པ་མང་པོ་སྡོད་དུ་བཅུག་ཡོད་པ་དང་། ཡང་ཁ་ཤས་ཁྱིམ་ཚང་ནང་གནས་ཚང་གཡར་ནས་བསྡད་ཡོད་སྐོར་བརྗོད་འདུག རྒྱ་ནག་བུད་མེད་དང་བྱིས་པའི་རེ་བ་ཐེབས་རྩ་ཚོགས་པ་གསར་འཛུགས་གནང་མཁན་ཀྲང་ཝིན་གྱིས། དམངས་ཁྲོད་ཀྱི་རོགས་སྐྱོབ་ཚོགས་པ་མང་པོ་བདག་མེད་བྱིས་པ་སྙོར་སྐྱོང་གི་བྱ་བའི་ནང་ཞུགས་འདོད་ཡོད་ཀྱང་། སྒེར་གཉེར་བྱིས་པ་སྙོར་སྐྱོང་ཁང་ཚོར་ཁྲིམས་མཐུན་ཐོབ་ཐང་རག་རྒྱུ་ཁག་པོ་ཡོད་པར་བརྟེན། བྱིས་སྙོར་ཁང་མང་པོར་མཐུན་རྐྱེན་ཐད་དཀའ་ངལ་གང་མང་འཕྲད་ཀྱི་ཡོད། རྒྱལ་ཡོངས་བདག་མེད་བྱིས་པ་སྙོར་སྐྱོང་གི་བྱ་བར་ཕན་འབྲས་ཐོབ་དགོས་ན། སྤྱི་ཚོགས་ཧྲིལ་པོའི་ནུས་པ་གཅིག་ཏུ་སྤུངས་ཆོག་པའི་དཀྱེལ་ཡངས་ཀྱི་ཁོར་ཡུག་བསྐྲུན་ཏེ། མི་སྒེར་དང་ཚོགས་སྡེ་གང་མང་ལས་གཞི་དེའི་ནང་མཉམ་ཞུགས་བྱེད་དུ་འཇུག་དགོས་པའི་རེ་སྐུལ་བྱས་ཡོད་འདུག རྒྱ་ནག་རྒྱལ་ཡོངས་བྱིས་པ་ལྟ་སྐྱོང་ཁང་གི་བསྡོམས་རྩིས་སྙན་ཐོ་ལྟར་ན། རྒྱ་ནག་ནང་ལོ་རེར་ཕ་མས་བདག་སྐྱོང་བྱེད་མ་ཐུབ་པ་དང་བྱེད་འདོད་མེད་པར་གཡུག་པའི་བྱིས་པ་དམར་འབྱར་སོགས་ ༡༠༠༠༠ ལྷག་རེ་ཡོད་པ་དང་། དེ་དག་ལས་བརྒྱ་ཆ་ ༩༩ ནད་པ་དང་དབང་པོ་སྐྱོན་ཅན་ཡིན་འདུག ༢༠༡༡ ལོའི་ཕྱི་ཟླ་ ༦ པའི་ནང་རྒྱ་ནག་ཧུའུ་པའེ་ཞིང་ཆེན་ཧྲི་ཅཱ་ཀྲོང་གྲོང་ཁྱེར་དུ་ཆེས་ཐོག་མར་བྱིས་པ་བདག་མེད་སྙོར་སྐྱོང་ཁང་གསར་འཛུགས་བྱས་པ་ནས་ད་ལྟའི་བར་བདག་མེད་བྱིས་སྙོར་ཁང་ ༢༥ ལྷག་བཙུགས་ཡོད་པ་དང་། མ་འོངས་པར་གྲོང་ཁྱེར་རེ་རེའི་ནང་དུ་ཉུང་མཐར་བྱིས་པ་བདག་མེད་བསྡུ་ལེན་ཁང་ ༢ རེ་འཛུགས་འཆར་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
