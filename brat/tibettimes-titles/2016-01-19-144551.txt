
རྩོད་རྙོག་ག་རེ་ཡོད་ཀྱང་ཕན་ཚུན་གུས་ཞབས་བྱེད་ཐུབ་དགོས་པ་དེ་གལ་ཆེ་ཤོས་རེད།

  བོད་ཀྱི་རྩོད་རྙོག་འདུམ་འགྲིག་ལྟེ་གནས་ཁང་གི་འགན་འཛིན་ཀརྨ་ལེགས་བཤད་ལགས་སུ། ལྟེ་གནས་ཁང་གི་ངོས་ནས་བོད་མིའི་སྤྱི་ཚོགས་ནང་རྩོད་རྙོག་འདུམ་འགྲིག་གི་ཕྱག་ལས་ག་རེ་གནང་ཡོད་པ་དང་། དེས་ཕན་ནུས་ཅི་ཙམ་ཐོན་ཡོད་པ་བཅས་ཀྱི་སྐོར་ལ་བཅར་འདྲི་ཞུས།   ( བོད་ཀྱི་རྩོད་ཉོག་འདུམ་འགྲིག་ཁང་གི་འགན་འཛིན་ཀརྨ་ལེགས་བཤད། ) དྲི་བ། འདས་པའི་ལོ་ ༡༤ ལྷག་གི་རིང་རྩོད་རྙོག་འདུམ་འགྲིག་ཁང་གི་ངོས་ནས་བོད་མིའི་སྤྱི་ཚོགས་ནང་འདུམ་འགྲིག་གི་ཕྱག་ལས་གཙོ་བོ་ག་རེ་ག་རེ་གནང་ཡོད་པ་རེད།   ལན། ང་ཚོའི་ཚོགས་པ་བཙུགས་ཏེ་ལོ་བཅུ་བཞི་བཅོ་ལྔ་ཙམ་འགྲོ་ཡི་ཡོད་པ་དང་། དེའི་རིང་ལ་བོད་པའི་སྤྱི་ཚོགས་ནང་ཐུགས་ཕན་གསོ་བའི་ཞབས་ཞུ་མང་པོ་ཞུས་ཐུབ་ཡོད་པ་རེད། རྩ་བའི་ཆ་ནས་རྩོད་རྙོག་འདུམ་འགྲིག་གི་ཐབས་ལམ་དང་སྒྱུ་རྩལ་ངོ་སྤྲོད་བྱེད་པ་དང་། ཐབས་ལམ་དང་སྒྱུ་རྩལ་དེ་ཚོ་སོ་སོའི་ཉིན་རེའི་འཚོ་བའི་ནང་དང་ལྷག་པར་དུ་ཕྱག་ལས་གནང་མཁན་མང་པོ་ཞིག་གིས་ཉིན་རེའི་ལས་ཀའི་ཐོག་ལ་བེད་སྤྱོད་བྱས་ཏེ། སྤྱི་ཚོགས་ཁྱོན་ཡོངས་ཀྱི་ནུས་པ་ཆེ་རུ་འགྲོ་ཐུབ་པ་དང་དམངས་གཙོ་དེ་ཁ་ནས་བཤད་པ་ཙམ་མ་ཡིན་པར་དོན་དངོས་སུ་སོ་སོའི་སྤྱི་ཚོགས་དང་ཁྱིམ་ཚང་། ཚོགས་པ་དང་སྡེ་ཚན་སོ་སོའི་ནང་ལག་བསྟར་ཐུབ་པར་ང་ཚོས་གང་ཡོང་གིས་འབད་བརྩོན་བྱས་ཡོད་པ་རེད། ལས་ཀ་གཙོ་བོ་ཞིག་སྦྱོང་བརྡར་དང་ཟབ་སྦྱོང་སྤྲོད་རྒྱུ་དེ་རེད། ད་ལྟའི་བར་དུ་གཞིས་ཆགས་དང་སློབ་གྲྭ། དགོན་སྡེ། ཚོགས་སྡེ་ཁག་སོགས་ལ་ཁྱོན་ཟབ་སྦྱོང་ཁག་ ༢༣༨ སྤྲད་ཡོད་པ་དང་། དེའི་ནང་མི་ ༨༠༠༠ ལྷག་ལ་གོ་རྟོགས་སྤེལ་ཡོད།   དྲི་བ། རྣམ་པ་ཚོས་གཞིས་ཆགས་ཁག་ནང་སྐོར་སྐྱོད་སྐབས། གཞིས་ཆགས་ཁག་ནང་རྩོད་རྙོག་ཤུགས་ཆེ་ཤོས་ག་རེ་ཡོང་གི་འདུག  རྩོད་རྙོག་འདུམ་འགྲིག་གི་ཟབ་སྦྱོང་བརྒྱུད་ནས་དཀའ་རྙོག་དེ་དག་ལ་ཕན་ནུས་ཅི་ཙམ་ཐོན་སོང་།     ལན། དྲང་པོར་ཞུས་ན། ད་ལྟའི་བར་དུ་ང་ཚོས་དམིགས་བསལ་གྱིས་དཀའ་ངལ་ཆེ་ཤོས་གང་ཡིན་པའི་ཐོག་ཉམས་ཞིབ་དེ་འདྲ་བྱེད་རྒྱུ་བྱུང་མེད་ཀྱང་། གཞིས་ཆགས་ག་བར་ཕྱིན་པ་ཡིན་རྒྱ་གར་ནའང་ཡུལ་མི་དང་འབྲེལ་བའི་དཀའ་ངལ་མང་པོ་འདུག དེ་ནས་ས་ཞིང་གི་དཀའ་རྙོག་དོག་ཙམ་རེ་འདུག ང་ཚོས་བོད་པ་དང་རྒྱ་གར་བ་མཉམ་རུབ་ཀྱི་ཐོག་ནས་དཀའ་ངལ་སེལ་ཐབས་བྱས་པ་ཡིན། ཡུལ་མིར་མཐུན་ལམ་ག་འདྲ་བྱེད་དགོས་པ་དང་། ཁོང་ཚོའི་དགོས་མཁོ་ག་རེ་རེད། ང་རང་ཚོས་ག་འདྲ་བྱས་ནས་སོ་སོའི་གནས་སྟངས་གསལ་པོ་ཞུ་ཐུབ་ཀྱི་རེད་དེའི་ཐད་ང་ཚོས་འབད་བརྩོན་བྱེད་བཞིན་ཡོད། ང་ཚོའི་ཟབ་སྦྱོང་གིས་ཕན་ཐོགས་ཀྱི་ཡོད་པ་ག་རེ་ར་འཕྲོད་པ་རེད་ཟེར་ན། ༢༠༠༡ ལོར་ང་ཚོས་ཚོགས་པ་བཙུགས་རྗེས་ ༢༠༠༢ དང་ ༢༠༠༣ ལོའི་ནང་གཞིས་ཆགས་ཁག་ནང་ཟབ་སྦྱོང་མགོ་བཙུགས་སྐབས། གཞིས་ཆགས་ཁ་ཤས་ཀྱིས་ང་ཚོའི་གཞིས་ཆགས་ཞེ་དྲགས་འཆམ་པོ་དང་མཐུན་པོ་ཡོད། རྩོད་རྙོག་འདུམ་འགྲིག་གི་ཟབ་སྦྱོང་མི་དགོས། ཟབ་སྦྱོང་སྤྲོད་དུ་ཡོང་ཆོག་གི་མེད་ཅེས་བཤད་མཁན་བྱུང་སོང་། རྩོད་རྙོག་ཟེར་དུས་ཞེད་སྣང་ཚ་བོ་དང་དངངས་སྐྲག་སྐྱེས་དགོས་པ་ཞིག་གི་ལྟ་སྟངས་འཛིན་གྱི་འདུག ཡིན་ནའང་ཕྱིས་སུ་ང་ཚོ་གཞིས་ཆགས་དེ་དག་ནང་དངོས་སུ་བཅར་སྐབས་རྩོད་རྙོག་མང་པོ་ཡོད་པ་མ་ཟད། ཐ་ན་གཞིས་མགོ་ལ་ཉེས་རྡུང་བཏང་བ་སོགས་ཀྱི་གནས་ཚུལ་ཁོ་རང་ཚོས་ཚུར་བཤད་ཡོང་གི་འདུག ཟབ་སྦྱོང་ཐེངས་གཉིས་གསུམ་སྤྲད་རྗེས་ཁོང་ཚོས་ཡང་བསྐྱར་ཕེབས་རོགས་གནང་། ཐུགས་ཕན་གསོ་ཡི་འདུག་གསུང་གི་འདུག   དྲི་བ། ང་ཚོའི་སྤྱི་ཚོགས་ནང་དཀའ་རྙོག་ག་རེ་བྱུང་བ་ཡིན་ནའང་ཁྲིམས་ཁང་དང་ཆེད་གཉེར་འདུམ་འགྲིག་ཁང་སོགས་ལ་ཞུ་གཏུག་བྱེད་པའི་གོམས་གཤིས་ཞན་པོ་རེད། གནད་དོན་ཆུང་ཆུང་ལའང་མཇུག་འབྲས་ཚབས་ཆེན་འབྱུང་གི་ཡོད་པ་རེད། ཁྱེད་རང་ཚོ་ལྟེ་གནས་ཁང་གི་ངོས་ནས་དེའི་ཐད་ལམ་སྟོན་ག་འདྲ་གནང་གི་ཡོད།   ལན། ང་ཚོས་གལ་ཆེ་ཤོས་ཤིག་ག་རེ་ལབ་ཀྱི་ཡོད་ཟེར་ན། རྩོད་རྙོག་ལབ་དུས་མ་མཐུན་པ་ཞིག་རེད། མ་མཐུན་པ་ཡོད་པ་ཙམ་གྱིས་རྩོད་རྙོག་ཡིན་པའི་ཁྱབ་ཀྱི་ཡོད་པ་མ་རེད། རྩོད་རྙོག་ག་རེ་ཡོད་ཀྱང་ཕན་ཚུན་གུས་ཞབས་བྱེད་ཐུབ་དགོས་པ་དེ་གལ་ཆེ་ཤོས་རེད། ང་ཚོས་མ་མཐུན་པ་དེར་ཞེད་སྣང་དང་དངངས་སྐྲག སྡུག་ཅག་ཡིན་པ་རྒྱུ་མཚན་དུ་བྱས་ནས་སྦེད་དེ་འཇོག་གི་ཡོད་པ་རེད། སྦེད་བཞག་ན་རིམ་གྱིས་རྩོད་རྙོག་གཅིག་གི་སྒང་ལ་གཅིག་རྩེག་ནས་ཆེན་པོ་ཆགས་ཏེ་མཐའ་མར་འཛིང་དགོས་པ་ཆགས་ཀྱི་ཡོད་པ་རེད། ཐབས་ལམ་གལ་ཆེ་ཤོས་དེ་རྩོད་རྙོག་ཅིག་ཡོད་པ་ཡིན་ན། དང་པོ་གནད་དོན་དང་གང་ཟག་གཉིས་བར་གྱི་འབྲེལ་བ་དེར་དབྱེ་བ་ཕྱེ་དགོས་པ་དང་། དེ་ནས་དཀའ་ངལ་དེ་ག་བར་འདུག་དེར་མཉམ་དུ་གདོང་ལེན་བྱས་ཏེ་ཕན་ཚུན་ལ་འགྲེལ་བཤད་བརྒྱབ་ནས་གསལ་པོ་བཟོ་དགོས་པ་རེད། ཡིན་ནའང་། མི་དང་མིའི་བར་གྱི་འབྲེལ་བ་དང་། ཕན་ཚུན་བར་གྱི་ཆེ་མཐོང་དང་གུས་འདུད་རང་འཇགས་གནས་པ་ཡིན་ན། རྩོད་རྙོག་དེ་བཟང་པོའི་ཕྱོགས་སུ་བསྒྱུར་ཐུབ་པ་རེད་ལ། དེས་སྤྱི་ཚོགས་ཡང་ཡར་རྒྱས་འགྲོ་ཐུབ་པ་རེད།   དྲི་བ། སྔོན་ཆད་དང་བསྡུར་ན་ང་ཚོའི་སྤྱི་ཚོགས་གང་ལ་གང་འཚམ་རྒྱུ་མཚན་བཤད་ས་ཡོད་པ་དང་། བསམ་བློ་གུ་ཡངས་པོ་ཆགས་ཀྱི་ཡོད་པ་རེད། རྣམ་པ་ཚོས་ད་ལྟའི་བར་དེའི་ཐད་ནས་ནུས་པ་ག་རེ་ཐོན་སོང་།   ལན། སྔོན་མ་དང་བསྡུར་ན་ང་ཚོའི་སྤྱི་ཚོགས་ཡར་རྒྱས་ཕྱིན་ཡོད་ཀྱང་། ད་ལྟ་ཡིན་ནའང་སོ་སོའི་སྒང་ལ་དངོས་སུ་དཀའ་ལས་ཤིག་འཕྲད་ཡོང་སྐབས་བསྐྱར་དུ་ལམ་ཀ་རྙིང་པ་དེ་རང་ལ་འགྲོ་བའི་འབད་བརྩོན་ཤུགས་ཆེ་བ་བྱེད་ཀྱི་ཡོད་སྟབས། ང་ཚོའི་སར་དངོས་སུ་འདུམ་འགྲིག་བྱེད་པར་ཡོང་མཁན་བྱུང་ཡོད་པ་མ་རེད་ལབ་ནའང་འགྲིག་འདུག ཡིན་ནའང་ང་ཚོ་དངོས་སུ་ཟབ་སྦྱོང་སྤྲོད་དུ་འགྲོ་སྐབས་ཚོགས་པ་སྡེ་ཚན་ནང་ལོགས་ཡིན་ན་རེད། མི་སྒེར་ཕན་ཚུན་ཡིན་ན་རེད། སྔོན་ཆད་བརྡ་མ་འཕྲོད་པ་དང་ཕན་ཚུན་ལ་གོ་བ་ལོག་པ་དེ་འདྲ་སེལ་ཐུབ་པ་བྱུང་སྟེ་ཡག་པོ་ཆགས་སོང་ཟེར་མཁན་ཡོང་གི་འདུག ད་ལྟ་ང་ཚོས་བྱེད་ཐུབ་པ་གཙོ་བོ་དེ་ཟབ་སྦྱོང་སྤྲད་ནས་གོ་རྟོགས་སྤེལ་བ་དང་། རྩོད་རྙོག་འདུམ་འགྲིག་གི་ཐབས་ལམ་རྣམས་ངོ་སྤྲོད་བྱས་ཏེ་མི་མང་ནང་སྦྱོང་བརྡར་ཐོབ་པ་བྱ་རྒྱུ་དང་། གཉིས་པ་དེ་ང་ཚོར་སྤྱི་ཚོགས་ནང་བར་འདུམ་བྱེད་པའི་ནུས་པ་ཡོད་པ་རེད། དཔེར་ན། ཕྱི་རྒྱལ་བ་བོད་པ་དང་འབྲེལ་བ་ཡོད་པའི་གནས་ཚུལ་ཞིག་ང་ཚོས་འདུམ་འགྲིག་བྱས་ཏེ་ཡག་པོ་བྱུང་ཡོད་པ་དང་། སྤྱི་ཚོགས་ནང་དཀའ་ངལ་མང་པོ་འཕྲད་བསྡད་པའི་མི་ཞིག་ལའང་ང་ཚོས་ལམ་སྟོན་བྱས་པ་བརྒྱུད་དེ་ཁོ་རང་གི་རྩོད་རྙོག་སེལ་ཐུབ་ཡོད་པ་རེད།   དྲི་བ།  དེང་སྐབས་ཆ་འཕྲིན་གྱི་བརྒྱུད་ལམ་སྟབས་བདེར་ཕྱིན་པ་དང་བསྟུན་ང་ཚོའི་སྤྱི་ཚོགས་ནང་འགན་ལེན་གྱི་འདུ་ཤེས་མེད་པ་དང་ཕྱོགས་རེས་འཐེན་འཁྱེར་བྱས་ཏེ་རྙོག་ཟིང་མང་པོ་འབྱུང་བཞིན་ཡོད་སྟབས། རྣམ་པ་ཚོ་ཆེད་དམིགས་ཀྱིས་གནད་དོན་འདིའི་ཐོག་ཕྱག་ལས་གནང་མཁན་ཞིག་ཡིན་པའི་ཆ་ནས་འདི་ལ་ཞིབ་འཇུག་དང་བཟང་ཕྱོགས་སུ་སྣེ་ཁྲིད་པའི་ནུས་པ་ཇི་ལྟར་འདོན་གྱི་ཡོད།   ལན། དེང་སང་གནས་ཚུལ་དེ་རིགས་གང་ལ་གང་འཚམ་ཡོང་གི་ཡོད་པ་རེད། ཡིན་ནའང་ངོ་དེབ་སོགས་བརྡ་འཕྲིན་རིགས་སྟངས་འཛིན་བྱེད་ཐུབ་རྒྱུ་ཁག་པོ་རེད། མི་སོ་སོའི་ངོས་ནས་སྤྱི་ཚོགས་ཡར་རྒྱས་ཀྱི་ཆེད་དུ་སྐད་ཆ་བཤད་རྒྱུ་དེ་དམངས་གཙོའི་འཕེལ་རིམ་ནང་གལ་ཆེན་པོ་རེད། ཡིན་ནའང་། སོ་སོའི་སྐད་ཆ་དེ་ཚོར་འགན་འཁུར་ཏེ་ག་བར་ཐུག་གི་ཡོད་ནའང་དོན་དག་འཁྱོལ་ཐུབ་པ་དང་ཐུག་ས་ཡོད་པ་ཞིག་བྱེད་དགོས་པ་དེ་དཔེ་མི་སྲིད་པའི་གལ་ཆེན་པོ་རེད་བསམ་གྱི་འདུག དེ་ནས་གལ་ཆེ་ཤོས་གཞན་ཞིག་ནི་ང་རང་ཚོའི་སྤྱི་ཚོགས་ནང་གསར་འགྱུར་བརྒྱུད་ལམ་མང་པོ་ཞིག་ཡོད་པ་དང་། གསར་འགྱུར་དེ་དམངས་གཙོའི་ཀ་བ་བཞི་པ་ཡིན་པའི་ཆ་ནས་གནས་ཚུལ་ག་རེ་ཡིན་ནའང་ཕར་ཕྱོགས་ཚུར་ཕྱོགས་གཉིས་ཆ་གསལ་པོ་གསལ་རྐྱང་ཞིག་སྟོན་ཐུབ་པ་ཡིན་ན། བཟང་ཕྱོགས་ཀྱི་ནུས་པ་ཤུགས་ཆེན་པོ་ཞིག་འདོན་ཐུབ་ངེས་རེད་བསམ་གྱི་འདུག དྲང་པོ་བཤད་ན། ངོ་དེབ་སོགས་ཀྱི་ནང་སྒྲ་དང་པར་འདྲ་མི་འདྲ་བཟོས་བསྡད་པ་དེ་ཚོས་སྤྱི་ཚོགས་ལ་ཕན་ཅི་གང་ཐོགས་ཀྱི་མེད་ལ། སོ་སོ་སྒེར་ལ་ཡིན་ནའང་ཕན་ཅི་ཐོགས་ཀྱི་ཡོད་པ་མ་རེད་དྲན་གྱི་འདུག དེར་བརྟེན་བཟང་སྤྱོད་ལྡན་པ་དང་འགན་འཁུར་བའི་ཐོག་ནས་གནས་ཚུལ་ག་རེ་ཡིན་ནའང་སྟོན་ཐུབ་པ་ཡིན་ན། སྤྱི་ཚོགས་ཡར་རྒྱས་ཐད་རླབས་ཆེན་གྱི་ནུས་པ་ཐོན་ངེས་རེད། ང་ཚོའི་དམིགས་ཡུལ་གཅིག་དེ་གཞོན་སྐྱེས་ཡིན་སྟབས། མ་འོངས་ལོ་ ༥༠ རྗེས་སུ་བཙན་བྱོལ་སྤྱི་ཚོགས་ནང་གི་མི་བརྒྱ་ཆ་ ༦༠ ཞིག་ལ་རྩོད་རྙོག་འདུམ་འགྲིག་གི་སྒྱུ་རྩལ་དང་ཐབས་ལམ་དེ་ཚོ་ཡོད་བསྡད་པ་ཡིན་ན། རང་བཞིན་གྱིས་ད་ལྟའི་འགན་འཁུར་མེད་པའི་ལས་ཀ་དེ་ཚོ་ཉུང་དུ་འགྲོ་བའི་རེ་བ་བྱེད་བཞིན་ཡོད།   དྲི་བ། ང་ཚོས་འདི་ལྟར་བལྟས་བསྡད་ན། བོད་མིའི་སྤྱི་ཚོགས་ནང་བཟའ་ཚང་བར་གྱི་དཀའ་རྙོག་ཧ་ཅང་ཤུགས་ཆེ་རུ་འགྲོ་བཞིན་ཡོད་པ་ཞིག་མཐོང་གི་འདུག དེའི་རྒྱུ་རྐྱེན་གཙོ་བོ་ག་རེ་ཡིན་ནམ། དཀའ་རྙོག་དེ་སེལ་བར་ཁྱོ་ཤུག་གཉིས་དང་ཡང་ན་དགའ་རོགས་ཤིག་གི་ངོས་ནས་ངེས་པར་དུ་གཟབ་ནན་དགོས་པ་ག་རེ་ག་རེ་མཐོང་གི་འདུག   ལན། ཕན་ཚུན་ལ་གུས་བརྩི་དང་ཆེ་མཐོང་བྱེད་དགོས་པ་གལ་ཆེ་ཤོས་དང་པོ་དེ་རེད་བསམ་གྱི་འདུག དེ་ནས་གཉིས་པ་དེ་ཁ་གསང་པོ་བྱེད་དགོས་པ་རེད། གོང་དུ་ཞུས་པ་ལྟར་རྩོད་རྙོག་བྱུང་ཚེ་སུ་ཡིན་ནའང་སྐྱིད་པོ་མེད་པས་སྦེད་གསང་བྱས་ཏེ། སོ་སོར་དངོས་གནས་ག་རེ་དགོས་ཀྱི་ཡོད་པ་དེ་ལབ་མ་ཐུབ་པའི་གནས་སྟངས་མང་པོ་ཞིག་ཡོང་གི་ཡོད་སྟབས། དེ་ཕན་ཚུན་འཆམ་མཐུན་མི་ཡོང་བའི་འགལ་རྐྱེན་གཅིག་ཕལ་ཆེར་ཆགས་ཀྱི་ཡོད་པ་རེད། ང་ཚོ་བོད་པའི་གོམས་གཤིས་ནང་སོ་སོའི་ཚོར་བ་དང་དགོས་མཁོ་དེ་ཚོ་ཁ་གསང་པོ་བྱས་ནས་བཤད་མཁན་ཉུང་ཉུང་ཞེ་དྲགས་རེད། དེ་ནས་རྒྱུ་རྐྱེན་གཞན་པ་ཞིག་ལ་དེང་སང་ཆུ་ཚོད་ཀྱི་ཆ་ནས་མཉམ་དུ་ཡུན་རིང་པོར་གཤིབ་མ་ཐུབ་པ་དང་། དེ་བཞིན་སོ་སོའི་ལས་ཀ་དང་ཐག་རིང་ཐུང་སོགས་ཀྱིས་དཀའ་རྙོག་ཡོང་གི་ཡོད་སྲིད་པ་རེད་བསམ་གྱི་འདུག ཡིན་ནའང་། ད་ལྟའི་ཆར་ནུབ་ཕྱོགས་ཀྱི་རྒྱལ་ཁབ་ཁག་དང་བསྡུར་ན་ང་ཚོའི་ཁྱིམ་ཚང་གི་གནས་སྟངས་དེ་མིག་དཔེ་ལྟ་འོས་པ་འདྲ་པོ་ཞིག་ཆགས་བསྡད་ཡོད་པ་རེད།   གསར་འགོད་པ། པདྨ་མཚོ།
