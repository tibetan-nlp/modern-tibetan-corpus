སྤྱི་ཁྱབ་རྩོམ་སྒྲིག་པ་ཞིག་འདས་གྲོངས།

ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༨ ཉིན་གྱི་ཕྱི་དྲོར་རྒྱ་ནག་ཞིན་ཧྭ་གསར་ཁང་ཨན་ཝེ་ཞིང་ཆེན་ཡན་ལག་ཁང་གི་སྤྱི་ཁྱབ་རྩོམ་སྒྲིག་པ་སཱོང་པེན་ལས་ཁུངས་ནང་འདས་གྲོངས་སུ་གྱུར་ཡོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༨ ཉིན་གྱི་ཕྱི་དྲོ་པེ་ཅིན་གྱི་ཆུ་ཚོད་ ༧ ཙམ་ལ། རྒྱ་ནག་གཞུང་འབྲེལ་ཞིན་ཧྭ་གསར་ཁང་གི་ཨན་ཝེ་ཞིང་ཆེན་ཡན་ལག་ཁང་གི་སྤྱི་ཁྱབ་རྩོམ་སྒྲིག་པ་སཱོང་པེན་ལས་ཁུངས་ནང་འདས་གྲོངས་སུ་གྱུར་བ་ཤེས་རྟོགས་བྱུང་ཡོད་པ་དང་། ཁོང་འདས་གྲོངས་སུ་ཕྱིན་རྗེས། ཞིན་ཧྭ་གསར་ཁང་དབུས་ས་གཉིས་ནས་ཁོ་བོའི་ཤི་རྐྱེན་སོགས་ཕྱིར་བསྒྲགས་བྱས་མེད་ཀྱང་། བརྒྱུད་ལམ་གཞན་ནས་སཱོང་པེན་མ་ཤི་གོང་དུ་ཡི་གེ་ཞིག་བཞག་པའི་ནང་། ཕྱི་ལོ་ ༢༠༠༤ ནས་ཁོ་བོར་སྙིང་ཁམས་ཀྱི་ན་ཚ་ཕོག་ཡོད་པ་ཤེས་ཏེ་འཚོ་བར་རེ་ཐག་ཆད་སོང་བ་བྲིས་ཡོད་སྐོར་བརྗོད་འདུག  ཁོ་བོའི་ལས་གྲོགས་ཤིག་གིས། ཕྱི་ལོ་ ༡༩༨༤ ཙམ་ནས་སཱོང་པེན་གྱིས་ཨན་ཝེ་ཡན་ལག་ཁང་དུ་ལས་ཀ་བྱས་ཤིང་། ཁོང་གིས་འདས་པའི་ལོ་མང་ནས་བཟུང་དབྱར་ཁའི་གུང་གསེང་ལའང་ངལ་གསོ་མ་བྱས་པར་ཨན་ཝེ་རྡོ་སོལ་བདེ་འཇགས་ཐོན་སྐྱེད་དང་། ཁེ་ལས་འཕེལ་རྒྱས་ཐད་དོ་སྣང་ཆེན་པོ་བྱས་པ་མ་ཟད། རྒྱུན་དུ་རྡོ་སོལ་སྔོག་འདོན་ས་ཁུལ་དུ་ཕྱིན་ཏེ་འབྲེལ་ཡོད་ལས་བྱེད་མང་པོར་བཅར་འདྲི་བྱས་པ་སོགས། གསར་འགྱུར་ལས་རིགས་ཐོག་འབུར་དུ་ཐོན་པ་ཞིག་ཡིན་སྐོར་དང་། ཁོ་བོ་བསམ་བློ་གུ་ཡངས་པོ་ཡོད་སྟབས། ལས་ཁུངས་ནང་གློ་བུར་འདས་གྲོངས་སུ་གྱུར་བར་ཧང་སང་ཚུལ་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
