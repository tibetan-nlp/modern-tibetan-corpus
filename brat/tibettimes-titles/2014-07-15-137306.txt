དགེ་རྒན་ ༢༠༠༠ ལྷག་གི་ངོ་རྒོལ།

རྒྱ་ནག་ཧུའུ་ནན་ཞིང་ཆེན་གྱི་སྒེར་གཉེར་སློབ་གྲྭ་ཁག་གི་དགེ་རྒན་ ༢༠༠༠ ལྷག་གིས་ལས་མཚམས་བཞག་སྟེ་ངག་བཅད་ངོ་རྒོལ་བྱེད་བཞིན་ཡོད་འདུག ཁ་སང་རྒྱ་ནག་ཧུའུ་ནན་ཞིང་ཆེན་སྲིད་གཞུང་གི་སྒོ་འགྲམ་དུ། ཞིང་ཆེན་དང་གྲོང་ཁྱེར་ཁག་ ༡༤ ཡི་སྒེར་གཉེར་སློབ་གྲྭ་ཁག་གི་དགེ་རྒན་ ༢༠༠༠ ལྷག་གིས། རྒྱལ་ཁབ་ཀྱི་གཏན་འབེབས་གཞིར་བཟུང་རྒན་ཕོགས་འཕར་སྣོན་དང་ལྟ་སྐྱོང་སྲིད་བྱུས་ལག་བསྟར་བྱེད་དགོས་པའི་ངག་བཅད་ངོ་རྒོལ་བྱེད་བཞིན་ཡོད་ཀྱང་། སྲིད་གཞུང་གི་འབྲེལ་ཡོད་མི་སྣ་གཅིག་གིས་ཀྱང་ཐུག་འཕྲད་བྱས་མེད་པ་མ་ཟད། གློ་བུར་དུ་ཉེན་རྟོག་པ་མང་པོ་བསླེབས་ཏེ་འབོད་སྐུལ་གྱི་འཕྲེད་དར་བཙན་འཕྲོག་བྱས་ཡོད་པ་མ་ཟད། མི་གཅིག་ལ་རྨས་སྐྱོན་དང་མི་ ༥ འཛིན་བཟུང་བྱས་ཡོད་འདུག ངོ་རྒོལ་ནང་ཞུགས་པའི་དགེ་རྒན་ཞིག་གིས། ཐེངས་འདིའི་ངོ་རྒོལ་ནང་ཞིང་ཆེན་དང་གྲོང་ཁྱེར་ཁག་ ༡༤ ཡི་སྒེར་གཉེར་སློབ་གྲྭའི་དགེ་རྒན་ ༢༠༠༠ ནས་ ༣༠༠༠ ཙམ་མཉམ་ཞུགས་བྱས་ཡོད་པ་དང་། རེས་གཟའ་ཟླ་བ་ནས་སྤེན་པ་བར་སྲིད་གཞུང་གི་སྒོ་འགྲམ་དུ་ངག་བཅད་ངོ་རྒོལ་བྱེད་འཆར་ཡོད། ད་ལོའི་ཕྱི་ཟླ་ ༥ ཚེས་ ༢༩ ཉིན་དགེ་རྒན་ཚོའི་སྐུ་ཚབ་ཞིང་ཆེན་སློབ་གསོ་ཁང་དུ་ཆེད་བཅར་གྱིས། ཕྱི་ལོ་ ༡༩༩༧ ལོར་ཀྲུང་དབྱང་གིས་གཏན་འབེབས་བྱས་པའི་ཡིག་ཆ་ཨང་ ༣༢ པ་དང་། ཕྱི་ལོ་ ༢༠༡༡ ལོའི་ཡིག་ཆ་ཨང་ ༨ པ་ངེས་པར་དུ་ལག་བསྟར་དོན་འཁྱོལ་ཡོང་བ་དགོས་པའི་རེ་སྐུལ་བྱེད་སྐབས། སྲིད་གཞུང་གི་འབྲེལ་ཡོད་མི་སྣས་ཕྱི་ཟླ་ ༦ པའི་ནང་སྲིད་གཞུང་གི་ཚོགས་འདུ་བསྡུ་སྐབས་ཐག་གཅོད་བྱ་རྒྱུ་ཡིན་ལུགས་བརྗོད་ཡོད་ཀྱང་། ད་ལྟའི་བར་བརྡ་ལན་ཅི་ཡང་སྤྲོད་མཁན་མེད་སྟབས། ཉེ་སྔོན་ཡང་བསྐྱར་ཞིང་ཆེན་སློབ་གསོ་ཁང་དུ་བཅར་ནས་སྐད་ཆ་འདྲི་སྐབས། སྲིད་གཞུང་གི་ཚོགས་འདུ་བསྡུ་རྒྱུ་མིན་ལུགས་བརྗོད་འདུག་པས། སྒེར་གཉེར་སློབ་གྲྭ་ཁག་གི་དགེ་རྒན་རྣམས་ཁོང་ཁྲོ་ཆེན་པོ་ལངས་ཏེ། ཟླ་འདིའི་ཕྱི་ཚེས་ ༢ ཉིན་དགེ་རྒན་ ༩༠ ལྷག་ཞིང་ཆེན་སྲིད་གཞུང་གི་སྒོ་ཁར་འཛོམས་ཏེ་ངག་བཅད་ངོ་རྒོལ་བྱེད་སྐབས། ཉེན་རྟོག་པས་མི་ཁ་ཤས་འཛིན་བཟུང་བཀག་ཉར་བྱས་ཡོད་ལུགས་སོགས་བརྗོད་འདུག ཡང་ལས་འགུལ་ནང་ཞུགས་པའི་དགེ་རྒན་གཞན་ཞིག་གིས། རྒྱ་ནག་གི་གྲོང་ཁྱེར་ ༤༠ ལྷག་གི་སྒེར་གཉེར་སློབ་གྲྭའི་དགེ་རྒན་གྱི་རྒན་ཕོགས་འཕར་སྣོན་གྱི་གནད་དོན་ཐག་གཅོད་བྱས་ཟིན་ཡང་། ཧུའུ་ནན་ཞིང་ཆེན་ནང་ལག་བསྟར་མ་བྱས་པར་ལོ་ངོ་ ༡༠ གྲངས་འགྲོ་བཞིན་ཡོད་པས། སྲིད་གཞུང་གིས་དགེ་རྒན་ཚོར་སྒོར་མོ་ཧ་ཅང་མང་པོ་ཆད་ཡོད། ང་ཚོའི་ནང་ལོ་ན་ཆུང་ཤོས་ལོ་ ༦༠ ཡིན་པ་དང་། མང་ཆེ་བ་ལོ་ ༧༠ ནས་ ༨༠ རེད། སྲིད་གཞུང་གི་ལྟ་སྐྱོང་ཕྱིར་འགྱངས་བྱས་པའི་ཁྲོད་དགེ་རྒན་མང་པོ་ཞིག་འདས་གྲོངས་སུ་གྱུར་བ་དང་། ད་ལྟའི་བར་སྲིད་གཞུང་གི་ལྟ་སྐྱོང་ལ་རེ་སྒུག་བྱས་ནས་འདས་གྲོངས་སུ་གྱུར་བ་བརྒྱ་ཆ་ ༡༠ ནས་ ༡༥ ཟིན་གྱི་འདུག ང་ཚོས་ཟླ་རེའི་འཚོ་སྣོན་རྒྱ་སྒོར་ ༦༠༠ དེ་སྤྲོད་དགོས་པའི་རེ་སྐུལ་བྱས་ཀྱང་དེའང་སྤྲོད་ཀྱི་མི་འདུག སྤྱིར་བཏང་རྒན་ཡོལ་དགེ་རྒན་ཞིག་ལ་ཟླ་རེར་རྒྱ་སྒོར་ ༢༥༠༠ ནས་ ༢༦༠༠ ཡོད་པ་རེད། སྲིད་གཞུང་གིས་བརྗོད་ན། གནད་དོན་འདི་ཐག་གཅོད་མི་བྱེད་པའི་རྒྱུ་རྐྱེན་ནི་དེ་སྔོན་རྒྱལ་ཡོངས་ནང་སྒེར་གཉེར་སློབ་གྲྭའི་དགེ་རྒན་གྱི་གྲངས་འབོར་རྩིས་བཤེར་བྱེད་སྐབས་ཁྲི་ ༡༢ ཙམ་ལས་བྱུང་མེད་ཀྱང་། འདས་ལོར་བསྐྱར་དུ་ཞིབ་རྩིས་བྱེད་སྐབས་ཁྲི་ ༥༨ ལྷག་ཡོད་པ་ཤེས་རྟོགས་བྱུང་སྐབས་སྲིད་གཞུང་གི་མགོ་ཁྲིད་ཚོ་སྐྲག་ཏུ་བཅུག་སོང་ཟེར་གྱི་འདུག་ཅེས་བརྗོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༧ ཉིན་ཀྲེ་ཅང་ཞིང་ཆེན་གྱི་དགེ་རྒན་སྟོང་ཕྲག་བརྒལ་བས་ཞིང་ཆེན་སློབ་གསོ་ཁང་གི་སྒོ་འགྲམ་དུ་ངོ་རྒོལ་བྱས་ཏེ། ཉེན་རྟོག་པས་དགེ་རྒན་གཉིས་འཛིན་བཟུང་བྱས་ཡོད་པ་དང་། འདི་ལོའི་ཕྱི་ཟླ་ ༤ ནང་ཧུའུ་པེ་ཞིང་ཆེན་གྱི་གྲོང་ཁྱེར་ ༣༠ ལྷག་གི་དགེ་རྒན་སྟོང་ཕྲག་མང་པོས་སྲིད་གཞུང་ལ་ངོ་རྒོལ་བྱས་ཡོད་པ་དང་། དེ་བཞིན། འདི་ལོའི་ཕྱི་ཟླ་ ༤ ཚེས་ ༨ ཉིན་ཧི་ནན་ཞིང་ཆེན་ལོ་ཧི་གྲོང་ཁྱེར་གྱི་རྫོང་ཁག་གཉིས་ཀྱི་སློབ་གྲྭ་ཆུང་འབྲིང་གི་དགེ་རྒན་ ༥༠༠༠ ལྷག་གིས། ཟླ་ཕྱེད་ཙམ་གྱི་རིང་ལས་མཚམས་བཞག་སྟེ་ངོ་རྒོལ་བྱས་ཡོད་པ་དང་། སྐབས་དེར་རྒྱ་གཞུང་ཉེན་རྟོག་པས་ལས་འགུལ་སྣེ་ཁྲིད་པ་ཡིན་པའི་ཉེས་མིང་འོག་དགེ་རྒན་མང་པོ་འཛིན་བཟུང་བྱས་པ་སོགས་རྒྱ་ནག་ནང་གཞི་རིམ་གྱི་སློབ་གྲྭ་ཆུང་འབྲིང་དང་། ལྷག་པར་དུ་སྒེར་གཉེར་སློབ་གྲྭ་ཁག་གི་དགེ་རྒན་སྟོང་ཕྲག་མང་པོས་སྲིད་གཞུང་ལ་ངོ་རྒོལ་རིམ་པར་བྱེད་བཞིན་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
