ཨ་རིས་རྒྱ་ནག་གི་རྐུ་མ་འཇུས།

རྒྱ་ནག་གི་བཅིངས་གྲོལ་དམག་མིའི་རུ་ཁག་ཅིག་གི་མི་ལྔ་ཨ་རིའི་དྲ་ཚིགས་ནང་ལྐོག་འཛུལ་བྱས་ཏེ་གསང་བའི་གནས་ཚུལ་མང་པོ་ཞིག་རྐུ་འཁྱེར་བྱས་བར་འཛིན་བཟུང་བཀའ་རྒྱ་བཏང་ཡོད་འདུག ཁ་སང་ཨ་རིའི་གཞུང་གིས་གཞུང་འབྲེལ་གསར་འགོད་གསལ་བསྒྲགས་སྤེལ་ཏེ་ ༢༠༠༦ ལོ་ནས་ད་བར་རྒྱ་ནག་བཅིངས་གྲོལ་དམག་མིའི་རུ་ཁག་ཨང་རྟགས་ ༦༡༣༩༨ ཀྱི་རུ་མི་ལྔས་ཨ་རིའི་དྲ་ཚིགས་ནང་ལྐོག་ཉན་ལྐོག་འཛུལ་བྱས་ཏེ། ཨ་རིའི་ཧྥན་སེལ་ཝེ་ནེ་ཡ་མངའ་སྡེའི་(Westinghouse Electric)རྡུལ་ཕྲན་གློག་ཆས་བཟོ་སྐྲུན་ཁང་དང་། ཧ་ཡང་བཟོ་གྲྭ(Alcoa Inc)། དྭངས་ལྕགས་བཟོ་གྲྭ འཕྲུལ་རིག་ཚན་རྩལ་འཕེལ་རྒྱས་ཁེ་ལས་ (Allegheny Technologies)། ཉི་འོད་འཕྲུལ་རིགས་བཟོ་སྐྲུན་ཁང་ (SolarWorld)། དྭངས་ལྕགས་ལས་བཟོ་ཚོགས་པ་བཅས་ཀྱི་བཟོ་རིག་ལག་རྩལ་དང་ཁེ་ལས་ཚོང་དོན་གྱི་གསང་བ་མང་པོ་ཞིག་རྐུ་འཁྱེར་བྱས་པའི་ཉེས་མིང་འོག་འཛིན་བཟུང་བཀའ་རྒྱ་བཏང་ཡོད་འདུག འདིའི་ཐད་ཨ་རིའི་སྲིད་འཛིན་ཨོ་བྷ་མས། “ཨ་རིའི་དྲ་ཚིགས་ནང་ལྐོག་ཉན་ལྐོག་འཛུལ་བྱེད་བཞིན་པ་དག་ནི། ཨ་རིའི་དཔལ་འབྱོར་དང་སྲུང་སྐྱོབ་ལ་འགོག་རྐྱེན་བཟོ་མཁན་གྱི་བྱེད་གཏེ་གཙོ་བོ་དེ་ཡིན།” ཞེས་བརྗོད་འདུག རྒྱ་ནག་ཕྱི་སྲིད་མགྲིན་ཚབ་པ་ཆིང་ཀང་གིས། “ང་ཚོས་དྲ་ཐོག་བརྒྱུད་ཨ་རིའི་ཚོང་དོན་གསང་བ་རྐུ་འཁྱེར་གང་ཡང་བྱས་མེད་སྟབས། གལ་ཏེ་ཨ་རིས་མུ་མཐུད་གཞི་མེད་སྐྱོན་འཛུགས་བྱས་ཚེ་རྒྱལ་ཁབ་གཉིས་ཀྱི་འབྲེལ་ལམ་ལ་དོ་ཕོག་འགྲོ་རྒྱུ་རེད།” ཅེས་དང་། ཁོས་ད་དུང་ཨ་རིས་རྒྱ་ནག་ལ་ཁ་གསལ་ལན་འདེབས་རྒྱག་དགོས་པ་མ་ཟད། སྐྱོན་འཛུགས་དེ་འཕྲལ་དུ་མཚམས་འཇོག་བྱ་དགོས་སྐོར་སོགས་བརྗོད་འདུག ཨ་རིའི་ཁྲིམས་དོན་སྤྱི་ཁྱབ་པ་ཨི་རེག་ཧུ་ཌར་(Eric Holder )གྱིས། “རྒྱ་གཞུང་གིས་ཨ་རིའི་ཚོང་དོན་གསང་བ་རྐུ་འཁྱེར་བྱེད་དོན་ནི། ཁོ་ཚོའི་གཞུང་འཛིན་ཁེ་ལས་ཁག་ལ་ཁེ་ཕན་ཐོབ་ཐབས་གཙོས། དམིགས་ཡུལ་གཞན་ཁག་ཅིག་སྒྲུབ་ཐབས་བྱེད་བཞིན་ཡོད།” ཅེས་བརྗོད་འདུག ཨ་རིའི་ཁྲིམས་དོན་སྤྱི་ཁྱབ་གཞོན་པ་ཇོན་ཁར་ལིན་(John Carlin) གྱིས། “ རྒྱལ་སྤྱིའི་ཐོག་ཁྲོམ་རྭའི་འགྲན་རྩོད་ཤུགས་ཆེན་ཡོད་སྟབས། ཨ་རིའི་བཟོ་སྐྲུན་ལག་རྩལ་བ་ཚོས་ཉིན་མཚན་ལྟོས་མེད་དུ་དཀའ་ལས་བརྒྱབ་སྟེ། གསར་གཏོད་གསར་སྐྲུན་བྱེད་བཞིན་ཡོད་པ་དག་བཅིངས་གྲོལ་དམག་མིའི་རུ་ཁག་དེའི་རུ་མི་ཚོས་ཧྲང་ཧེ་བརྒྱུད་རྐུ་འཁྱེར་བྱས་ཏེ། ང་ཚོས་དཀའ་ལས་བརྒྱབ་པའི་འབྲས་བུ་ཁག་ལ་ལོངས་སྤྱོད་བྱེད་བཞིན་འདུག” ཅེས་བརྗོད་འདུག ཐེངས་འདིར་ཨ་རིའི་དྲ་ཚིགས་ནང་ལྐོག་འཛུལ་བྱེད་མཁན་དག་ནི་ཝང་ཏུང་(Wang Dong)དང་། སུང་ཁེ་ལིང་(Sun Kailiang)། ཝུན་ཤིན་ཡུ(WenXinyu)། ཧོང་ཀྲིན་ཡུ(Huang Zhenyu)།གྷོ་ཁྲེན་ཧེ་ (GuChunhui) བཅས་ཡིན་འདུག ༢༠༠༦ ལོ་ནས་བཟུང་རྒྱ་གཞུང་གིས་ཨ་རིའི་ཧྥན་སེལ་ཝེ་ནེ་ཡ་མངའ་སྡེའི་པི་སེ་བྷག་ནང་རྟེན་གཞི་བྱས་པའི་ཝེ་སེ་ཀྲིང་རྡུལ་ཕྲན་གློག་ཆས་དང་དྭངས་ལྕགས། ཧ་ཡང་བཟོ་གྲྭ ཉི་འོད་གློག་རྡུལ་ཁེ་ལས་སོགས་ཀྱི་དྲ་ཚིགས་ནང་ལྐོག་འཛུལ་བྱས་ཏེ། ཁེ་ལས་ཀྱི་རྒྱུ་ཆ་དང་ཚོང་དོན་ལས་གཞི། རིན་གོང་འཕར་ཆག སྤུས་ཚད། བཟོ་རིག་ལག་རྩལ་སོགས་ཀྱི་ཡིག་ཆ་རྐུ་འཁྱེར་བྱས་ཏེ། རྒྱ་གཞུང་གི་ཁེ་ལས་དག་ལ་མཁོ་སྤྲོད་དང་འབྲེལ། ཁྲོམ་རྭར་དགོས་མཁོ་ཆེ་བའི་འཕྲུལ་ཆས་གསར་བཟོ་བྱེད་བཅུག་སྟེ་རིན་གོང་ཆུང་བའི་དངོས་རྫས་བཟོས་ཤིང་། རྒྱལ་སྤྱིའི་ཐོག་ཕྱིར་ཚོང་བྱས་ཏེ་ཨ་རིའི་ཁྲོམ་རྭའི་བཟོ་སྐྲུན་ཐོན་ལས་དག་ལ་ཁ་གཏད་བཅག་ཡོད་པ་མ་ཟད། ཨ་རིའི་ཚོང་རིགས་འཕེལ་རྒྱས་ལའང་དོ་ཕོག་ཤུགས་ཆེན་བཏང་ཡོད་འདུག ཨ་རིའི་གསང་བའི་ལས་ཁུངས་ཀྱི་ལས་བྱེད་ཅིག་གིས། ༢༠༠༦ ལོ་ནས་བཟུང་ད་བར་རྒྱ་གཞུང་གིས་ཨ་རིའི་ཁེ་ལས་ཁག་གི་ཚོང་དོན་གསང་བ་རྐུ་འཁྱེར་བྱས་ཏེ། གྱོང་གུད་ཚབས་ཆེན་བཟོས་ཡོད་པ་ལྟར། ད་དུང་ང་ཚོས་ཤེས་རྟོགས་ཐུབ་མེད་པའི་ཁེ་ལས་མང་པོ་ཞིག་ལའང་དེ་བཞིན་གྱོང་གུད་ཆེན་པོ་སྤྲད་ཡོད་སྲིད་སྐོར་སོགས་བརྗོད་འདུག ཆབ་སྲིད་དབྱེ་ཞིབ་པ་ཚོས། རྒྱ་གཞུང་གིས་ཨ་རིའི་ཚོང་དོན་གསང་བ་རྐུ་འཁྱེར་བྱས་ཏེ། ཨ་རིའི་ཁེ་ལས་མང་པོ་ཞིག་ལ་གྱོང་གུད་ཤུགས་ཆེན་བཟོས་ཡོད་སྟབས། མ་འོངས་པར་སྟོབས་ཆེན་རྒྱལ་ཁབ་གཉིས་ཀྱི་འབྲེལ་ལམ་ལ་ངན་ཕྱོགས་ཀྱི་ཤུགས་རྐྱེན་ཐེབས་རྒྱུ་ཡིན་སྐོར་བརྗོད་འདུག རྒྱ་གཞུང་གི་མགོ་ཁྲིད་ཚོས་མདང་དགོང་རྒྱ་ནག་གི་ཨ་རིའི་གཞུང་ཚབ་ཆེན་མོ་བྷོ་ཁཱ་སེ་ (Max Baucus) ལགས་ལ་ཐུག་འཕྲད་བྱས་ཏེ། ཨ་རིའི་ཚོང་དོན་གསང་བ་རྐུ་འཁྱེར་བྱས་པའི་གནད་དོན་ཐད་གྲོས་མོལ་བྱས་ཡོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
