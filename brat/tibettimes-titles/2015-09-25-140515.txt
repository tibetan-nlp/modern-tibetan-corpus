ཡུལ་མི་ ༡༢ འཛིན་བཟུང་བྱས།

  རྔ་བ་ཁུལ་མཛོད་དགེ་རྫོང་ཐང་སྐོར་གྲོང་རྡལ་རྐ་བར་མ་སྡེ་བའི་དུད་ཁྱིམ་ ༢༠ ལྷག་གི་ས་ཆ་བཙན་འཕྲོག་བྱས་པ་མ་ཟད། ཉེས་མེད་མི་མང་ ༡༢ འཛིན་བཟུང་བྱས་ཡོད་འདུག   རྒྱ་ནག་ས་གནས་སྲིད་གཞུང་གིས་མདོ་སྨད་མཛོད་དགེ་རྫོང་ཐང་སྐོར་གྲང་རྡལ་རྐ་བར་མའི་མང་ཚོགས་སྡོད་གནས་བཙན་འཕྲོག་བྱས་པ་མ་ཟད། ཁང་པ་གཏོར་བཅོམ་བྱས་ཏེ་མ་བུ་གཉིས་ཀྱི་གཞན་ལ་གནས་ཚང་གཡར་ནས་བསྡད་འདུག   ཕྱི་ལོ་ ༢༠༡༠ ལོར་རྔ་བ་ཁུལ་མཛོད་དགེ་རྫོང་ཐང་སྐོར་གྲོང་རྡལ་ས་གནས་སྲིད་གཞུང་གིས་གྲོང་ཁྱེར་སྔོ་ལྗང་ཅན་དུ་བསྒྱུར་བའི་ལས་གཞི་ལ་ཁ་གཡར་ནས་ཐང་སྐོར་གྲོང་རྡལ་རྐ་བར་མ་སྡེ་བའི་དུད་ཁྱིམ་ ༢༠ ལྷག་གི་ས་ཆ་བཙན་འཕྲོག་བྱས་ཡོད་སྟབས། ས་གནས་མང་ཚོགས་ཀྱིས་གནད་དོན་དེ་ཐག་གཅོད་ཡོང་བའི་སྙན་ཞུ་དང་རེ་འབོད་ཞུས་ཡོད་ཀྱང་། ས་གནས་སྲིད་གཞུང་གིས་འབྲོག་ཁུལ་གྱི་ས་ཆ་བཙན་འཕྲོག་བྱས་པའི་གནད་དོན་ཐག་གཅོད་རྒྱུ་ལྟ་ཅི། དེ་ལས་ལྡོག་སྟེ་ཡུལ་མི་ ༡༢ འཛིན་བཟུང་བྱས་ཡོད་འདུག   སྤྱིར་བཏང་རྔ་བ་ཁུལ་སྲིད་གཞུང་གིས་ས་ཆའི་གནད་དོན་དེ་ཟླ་བ་ ༥ བའི་ནང་ཐག་གཅོད་བྱ་རྒྱུ་བྱས་ཡོད་ཀྱང་། ཟླ་བ་ ༤ ལྷག་ལ་གྲགས་འགུལ་གང་ཡང་བྱུང་མི་འདུག འོན་ཀྱང་ཉེ་ཆར་ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༠ ཉིན་སི་ཁྲོན་ཞིང་ཆེན་རྔ་བལ་ཁུལ་དང་མཛོད་དགེ་རྫོང་གི་འབྲེལ་ཡོད་ལས་བྱེད་ཀྱི་ཐང་སྐོར་གྲོང་རྡལ་གྱི་རྐ་བར་མའི་ས་ཆའི་གནད་དོན་ཐག་གཅོད་བྱ་རྒྱུ་བྱས་ཡོད་ཀྱང་། ཡུལ་མི་ཁག་ཅིག་གིས་ཉིན་གུང་བོ་སྒུག་ཡོད་ཀྱང་ཕྱི་དྲོ་ཆུ་ཚོད་ ༦ པ་ཙམ་ནས་ས་གནས་སྲིད་གཞུང་ཉེན་རྟོག་པ་ཡོང་ནས་ཡུལ་མི་ཚོར་འཇིགས་སྐུལ་བྱས་ཡོད་འདུག སྐབས་དེར་ཡུལ་མི་འཇིགས་བྱེད་སྐྱབས་ཀྱིས། ང་ཚོའི་ཁྱིམ་ཚང་ ༢༧༠ ལྷག་ཡོད་པའི་ཡུལ་མི་ཚོའི་མཛུབ་ཐམ་བརྒྱབ་ནས་སྲིད་གཞུང་ལ་བཙངས་ཡོད་ཟེར་བའི་བདེན་དབང་ཡི་གེ་ཡོད་ན་ང་ཚོས་ས་འདི་ཐང་ཀར་སྲིད་གཞུང་ལ་འབུལ་རྒྱུ་ཡིན། འོན་ཀྱང་། གལ་ཏེ་ཁྱིམ་ཚང་ ༢༧༠ ལྷག་གི་མཛུབ་ཐམ་མེད་པ་ཡིན་ན། ས་འདི་ཁྲིམས་ཀྱི་ནང་ལ་འདི་ལྟར་ལེན་ཆོག་ཆོག་ཅིག་རྩ་བ་མིན་སྟབས་ང་ཚོ་འགྲོ་རྒྱུ་མིན། དེ་རིང་རྔ་བ་ཁུལ་སྲིད་གཞུང་གི་མགོ་བ་ཚོ་ས་ཡི་གནད་དོན་ཐག་གཅོད་བྱ་ཡོང་ཡོད་ན་ས་ཐོག་འདིར་ཡོང་དགོས་རེད། ང་ཚོས་འདི་ནས་ཉིན་གུང་བོ་སྒུག་ཀྱང་བསླེབ་མ་སོང་། གལ་སྲིད་དེ་རིང་མ་ཡོང་བ་ཡིན་ན། ང་ཚོའི་ས་ཆ་ང་ཚོས་བདག་པོ་བརྒྱག་གི་ཡིན། ཞེས་བརྗོད་འདུག   ཕྱི་ཚེ་ ༢༡ ཉིན་ལ་རྒྱ་གཞུང་གིས་ས་གནས་ཡུལ་མི་ཚོ་འཛིན་རྒྱུའི་གྲ་སྒྲིག་ཨང་གསར་བྱས་ཡོད་ སྟབས། ཚེས་ ༢༢ ཉིན་ས་གནས་ཉེན་རྟོག་པ་བཀྲ་ཤིས་ཟེར་བ་གཙོ་བྱས་ཏེ་དྲག་ཆས་ཉེན་རྟོག་པའི་རླངས་ འཁོར་དྲུག་ལྷག་ཡོང་ནས། གནས་ལུགས་དང་རྒྱུ་མཚན་ཅི་ཡང་མ་བཤད་པར། མང་ཚོགས་ལ་གཅར་རྡུང་དང་མི་མང་གི་ལག་འཁྱེར་ཁ་པར་དང་པར་ཆས་སོགས་བཙན་འཕྲོག་ བྱས་ཡོད་པ་མ་ཟད། འཇིགས་བྱེད་སྐྱབས་དང་ཕུར་ཁོ། ཚེ་དཔག བཀྲ་ཤིས་རབ་བསྟན། བསོད་ནམས་རྒྱ་མཚོ། བཀྲ་བྷ། ཞི་ཕྲུག ཚེ་རིང་སྐྱབས། ཚེ་རིང་བཀྲ་ཤིས། འཚོ་སྐྱིད། ཡེ་ཤེས་བཅསན་ཉིས་མེད་ས་གནས་བོད་མི་ ༡༢ ལྷག་འཛིན་བཟུང་བཙན་ཁྲིད་བྱས་ཡོད་འདུག དེའི་མཚན་ལ་བུད་མེད་འཚོ་སྐྱིད་ལ་གཅར་རྡུང་དྲག་པོ་བཏང་རྗེས་གློད་བཀྲོད་བཏང་ འདུག ཉིན་དེའི་དགོང་དྲོ་ཆུ་ཚོད་བདུན་ཡས་མས་ནས་ཕུར་ཁོ་ཡི་སྡོད་ཁྱིམ་ཁང་པ་བཙན་ གྱིས་ཁྲབ་ལྡན་འདེད་འཁོར་ (推动机)གྱིས་དེད་དེ་གཏོར་བཅོམ་བྱས་ཡོད་པ་མ་ཟད། ཁང་པའི་ནང་དུ་གསར་གྱི་རྒྱན་ཆ་བྱུ་རུའི་སྐེ་རྒྱན་དང་རྒྱ་སྒོར་ ༡༠༠༠༠ ལྷག་ཡོད་པའང་ཁང་པ་དང་མཉམ་དུ་མེད་པར་བཟོ་འདུག་པ་མ་ཟད། ད་ལྟ་ཁོའི་བཟའ་ཟླ་ནོར་འཚོ་དང་བུ་མོ་ཡོན་འཚོ་མ་བུ་གཉིས་ཀྱིས་གཞན་ལ་གནས་ཚང་ གཡར་ནས་བསྡད་འདུག ཅེས་བརྗོད་སོང་།   དེ་བཞིན་ཕྱི་ཚེས་ ༢༣ ཉིན་མིང་ལ་སྒྲོལ་མ་སྐྱབས་ཟེར་བ་ཞིག་གིས་སི་ཁྲོན་ཞིང་ཆེན་ནས་རྔ་བོད་རིགས་རང་སྐྱོང་ཁུལ་གྱི་ཁུལ་དཔོན་ལེའུ་ཙོ་མིན་ (Liu Zuoming) ཟེར་བ་ཞིག་ལ་སྙན་ཞུ་ཕུལ་བའི་ནང་། ངའི་མིང་ལ་སྒྲོལ་མ་སྐྱབས་ཟེར། ང་ནི་སྟག་ཚང་གྲོང་རྡལ་མཐའ་བའི་གྲོང་མི་ཡིན། ༢༠༡༤ ལོར་ངས་མཐའ་བའི་ས་ཆའི་གནད་དོན་སྐོར་གྱི་ཞུ་སྙན་ཞིག་ཁྱེད་ལ་ཕུལ་ཡིན། ང་ཚོའི་ཡུལ་ཚོ་སྲིད་གཞུང་གི་འགོ་ཁྲིད་ཀྱིས་གསུང་བ་ལྟར་ན། ཞུ་འཕྲིན་དེ་སྟག་ཚང་གྲོང་རྡལ་སྲིད་གཞུང་ལ་འབྱོར་ཡོད་པ་དང་། སྟག་ཚང་གྲོང་རྡལ་སྲིད་གཞུང་གིས་མཐའ་བ་ས་ཆའི་གནད་དོན་ཐག་གཅོད་བྱེད་བཞིན་ཡོད་པ་དེར། ངས་ཀྱང་དྲང་བདེན་ཁྲིམས་མཐུན་གྱི་སྒོ་ནས་ཐག་གཅོད་བྱེད་པའི་རེ་སྒུག་བྱེད་བཞིན་ཡོད་ཅེས་དང་།   ཁོང་གིས་མུ་མཐུད་དུ། ༢༠༡༥ ལོའི་ཟླ་ ༩ ཚེས་ ༢༣ ཉིན་གྱི་སྔ་དྲོ་འཇིགས་བྱེད་སྐྱབས་ཀྱི་སྐྱེས་དམན་དབང་མོ་འཚོ་ཡིས་ང་ལ་ཁ་པར་བཏང་ནས། མཛོད་དགེ་རྫོང་སྤྱི་བདེ་ལས་ཁང་ཉེན་རྟོག་པས་དྲག་སྤྱོད་ཀྱིས་ཐང་སྐོར་གྲོང་རྡལ་རུ་ཁག་གཉིས་པའི་ས་ཆའི་ཁེ་དབང་ཞི་རྒོལ་བ་མི་ ༡༠ ལྷག་འཛིན་བཟུང་བྱས་ཡོད་སྟབས། ཐང་སྐོར་གྲོང་རྡལ་སྲིད་གཞུང་གིས་རྩ་ཁྲིམས་དང་ཁྲིམས་ཀྱི་གོ་རིམ་ཐོག་ནས་ཁྲིམས་མཐུན་གྱིས་ཐང་སྐོར་རུ་ཁག་གཉིས་པའི་རྩྭ་ཐང་གཞུང་བསྡུ་ཞུས་པ་རེད་དམ། གལ་སྲིད་ཁྲིམས་མཐུན་གྱིས་གཞུང་བསྡུ་ཞུས་པ་ཡིན་ན་ཅིའི་ཕྱིར་ཐོག་མར་ཁྲིམས་མཐུན་གཞུང་བསྡུའི་ཡིག་ཆ་ཡོད་ཚད་འཁྱེར་ཡོང་མེད་པ་རེད་དམ། ༼རྒྱ་ནག་མི་དམངས་སྤྱི་འཐུན་རྒྱལ་ཁབ་གྲོང་རྡལ་འཆར་འགོད་བཅའ་ཁྲིམས་༽ ཀྱི་བཅའ་ཁྲིམས་ཨང་ ༢༢ ཀྱི་ནང་དུ་གསལ་བོ་བཀོད་པ་ཞིག་ལ། གྲོང་ཚོ་དང་གྲོང་རྡལ་མི་དམངས་སྲིད་གཞུང་རྩ་འཛུགས་ཀྱིས་གྲོང་ཚོ་འཆར་བཀོད་བྱེད་པ་དང་གྲོང་ཚོའི་འཆར་བཀོད་བྱ་རྒྱུ་ཡིན་ན། མི་དམངས་སྲིད་གཞུང་གོང་རིམ་རིམ་པ་དང་པོའི་ཆོག་མཆན་དགོས་པ་དང་། གྲོང་ཚོའི་འཆར་བཀོད་ཀྱི་སྙན་སེང་དང་ཆོག་མཆན་མ་བྱུང་སྔོན་ལ། གྲོང་ཚོའི་མི་མང་ཚོགས་འདུ་དང་སྡེ་བའི་འཐུས་མི་རྣམས་ལ་གྲོས་མོལ་དང་མོས་མཐུན་བྱེད་དགོས་འདུག ཁྲིམས་ལུགས་ནང་གསལ་བོ་ཡོད་ཀྱང་ང་ཚོའི་ས་གནས་སྲིད་གཞུང་གིས་འདི་ལྟར་བྱས་ཡོད་དམ། ཞེས་སོགས་བཀོད་ནས་སྙན་ཞུ་ཞིག་ཀྱང་ཕུལ་ཡོད་འདུག   ལོ་འདིའི་ཕྱི་ཟླ་  ༡ ཚེས་ ༢༨ ཉིན། སི་ཁྲོན་ཞིང་ཆེན་མི་མང་འཐུས་མི་ཚོགས་ཆེན་དང་ཞིང་ཆེན་ཆབ་སྲིད་གྲོས་ཚོགས་གཉིས་འཚོག་བཞིན་པའི་སྐབས་དེར། རྒྱ་གཞུང་གིས་ཐང་སྐོར་གྲོང་རྡལ་གྱི་ས་ཆ་བཙན་འཕྲོག་བྱེད་པའི་དཀའ་ངལ་ཞུ་གཏུག་གི་ལས་འགུལ་ཞིག་སྤེལ་ཡོད་པ་དང་། དེའི་ཁྲོད་དུ་རྔ་བ་ཁུལ་མཛོད་དགེ་རྫོང་ཐང་སྐོར་འབྲོག་པའི་འཐུས་ཚབ་ ༡༠ ཡིས་ཀྱང་། “སྲོག་ལ་བབ་ཀྱང་ས་ཆ་སྲུང་སྐྱོབ་བྱ་རྒྱུ་ཡིན། ཁྲིམས་འགལ་གྱིས་ས་ཆ་འཕྲོག་པར་ངོ་རྒོལ་བྱ་རྒྱུ་ཡིན།” “ས་ཆ་བཙན་འཕྲོག་བྱས་པས་ང་ཚོ་རང་ཁྱིམ་ལ་ལོག་ས་མེད།” ཞེས་བྲིས་ཡོད་པའི་འཕྲེད་དར་སྒྲེང་སྟེ་ཞུ་གཏུག་གི་ཁྲོམ་སྐོར་བྱས་ཡོད་ཀྱང་། སྐབས་དེར་བོད་མི་ཞུ་གཏུག་མཁན་ཚོས་གོམ་བགྲོད་སྐར་མ་གཅིག་ལས་མ་སོང་བར་ཉེན་རྟོག་པ་ ༥༠ ཙམ་འབྱོར་ཏེ་བོད་མི་རྣམས་སྤྱི་སྤྱོད་རླངས་འཁོར་ཞིག་གི་ནང་དུ་བཙན་ཁྲིད་བྱས་ཡོད་འདུག   གནས་ཚུལ་དེའི་འབྱུང་རྐྱེན་ནི་ལོ་ངོ་ལྔ་ལྷག་ཙམ་གྱི་གོང་མཛོད་དགེ་ཐང་སྐོར་གྱི་རུ་ཁག་གཉིས་པའི་འབྲོག་སརགཞུང་གི་འཛུགས་སྐྲུན་བྱ་རྒྱུ་ཡིན་ལུགས་བཤད་ནས་ས་གནས་འབྲོག་པའི་ས་ཆ་བཙན་འཕྲོག་བྱས་པ་དང་། དེའི་རྗེས་ཀྱི་ལོ་མང་པོའི་རིང་ས་གནས་འབྲོག་པ་ཚོས་མཛོད་དགེ་རྫོང་ལ་ཞུ་གཏུག་ཐེངས་མང་བྱས་ཡོད་ཀྱང་དེར་ཁ་ཡ་དང་ཐག་གཅོད་བྱེད་མཁན་རྩ་བ་ནས་བྱས་མི་འདུག   གསར་འགོད་པ། འཇིགས་མེད་རྒྱ་མཚོ།
