གྲྭ་བ་གཉིས་ཀྱི་ཛ་དྲག་གནས་སྟངས།

ཡུལ་ཤུལ་ཁྲི་འདུ་རྫོང་ཉ་མཚོ་ཟིལ་དཀར་དགོན་གྱི་དགེ་འདུན་པ་ཚུལ་ཁྲིམས་སྐལ་བཟང་ལགས་དང་བློ་བཟང་སྦྱིན་པ་ལགས་གཉིས་ཀྱི་སྐུ་གཟུགས་གནས་སྟངས་ཛ་དྲག་ཆགས་ཡོད་འདུག བོད་ནས་ཁུངས་ལྡན་གནས་ཚུལ་འབྱོར་བ་ལྟར་ན། ཕྱི་ལོ་ ༢༠༡༢ ཟླ་ ༩ ཚེས་ ༡ ཉིན་རྒྱ་གཞུང་གིས་རང་ལུས་མེར་བསྲེགས་དང་འབྲེལ་བ་ཡོད་པའི་ཉེས་མིང་འོག ཡུལ་ཤུལ་ཁུལ་ཁྲིན་འདུ་རྫོང་ཉ་མཚོ་ཟིལ་དཀར་དགོན་གྱི་དགེ་འདུན་པ་ ༥ འཛིན་བཟུང་བྱས་ཏེ། གྲྭ་བ་བཞི་ལ་སྔ་གཞུག་ཏུ་ལོ་ ༢ ནས་ ༡༠ བར་གྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་པ་ལྟར། འདས་ལོའི་ཕྱི་ཟླ་ ༧ ཚེས་ ༡༢ ཉིན་རྒྱ་གཞུང་གིས་བསམ་བཞིན་མི་བསད་པའི་ཉེས་མིང་འོག གྲྭ་ཚུལ་ཁྲིམས་སྐལ་བཟང་ལགས་ལ་ལོ་ ༡༠ བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་དེ་ཟི་ལིང་བཙོན་ཁང་དུ་བཅུག་ཡོད་པ་དང་། ཕྱིས་སུ་ཁོང་ལ་མཆིན་པའི་ན་ཚ་བྱུང་སྟེ་ཐེངས་གཉིས་ལ་བཤག་བཅོས་བྱས་ཡོད་ཀྱང་། ནད་གཞི་དྲག་སྐྱེད་མ་བྱུང་བར་སྐུ་གཟུགས་གནས་སྟངས་ཇེ་སྡུག་ཏུ་ཕྱིན་ཡོད་སྟབས། ཉེ་སྔོན་ཁོང་གི་ཨ་མ་ལགས་ཁོང་ལ་ལྟ་རུ་ཕྱིན་ཡོད་ཀྱང་། བཙོན་ཁང་གིས་ཁོང་མོར་ཐོབ་ཐང་ལག་ཁྱེར་མེད་སྐོར་བརྗོད་དེ་ཐུག་འཕྲད་བྱེད་དུ་བཅུག་མེད་པ་རེད། དེ་བཞིན། འདས་ལོའི་ཕྱི་ཟླ་ ༣ པའི་ནང་རང་ལུས་མེར་བསྲེགས་དང་འབྲེལ་བའི་ཉེས་མིང་འོག ལོ་ ༥ བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་པའི་གྲྭ་བློ་བཟང་སྦྱིན་པ་ལགས་ལ་སྔོན་ཆད་ནས་མཆིན་པ་དང་མཁལ་མའི་ན་ཚ་ཡོད་པ་དང་། ཁོང་བཙོན་ཁང་དུ་བཅུག་རྗེས། བཟའ་བཏུང་གི་ཆ་རྐྱེན་ཞན་པའི་ཐོག་སྨན་བཅོས་བྱེད་པའི་གོ་སྐབས་མ་རག་པར་ནད་གཞི་ཇེ་སྡུག་ཏུ་ཕྱིན་ཡོད་སྟབས། ཁྱིམ་མི་གཉེན་འབྲེལ་ཚོ་བཙོན་ཁང་དུ་སྨན་དང་ཟ་མ་སྐྱེལ་དུ་ཕྱིན་ཡང་སྤྲོད་དུ་བཅུག་མེད་སྟབས། ཉ་མཚོ་དགོན་དང་ཁྱིམ་མི་གཉེན་འབྲེལ་ཚོས་ཁོང་གཉིས་ཀྱི་སྐུ་གཟུགས་གནས་སྟངས་ལ་སེམས་ཁྲལ་དང་། གང་མགྱོགས་ཀྱིས་སྨན་བཅོས་བྱེད་པའི་གོ་སྐབས་སྤྲོད་པར་རེ་བ་རྒྱག་བཞིན་ཡོད་འདུག གྲྭ་ཚུལ་ཁྲིམས་སྐལ་བཟང་ལགས་ནི་ད་ལྟ་རང་ལོ་ ༢༧ ཡིན་པ་དང་། ཁོང་གི་ཕ་མིང་ལ་ཨ་ཐུབ་དང་ཨ་མའི་མིང་ལ་སྒྲོལ་དཀར་ཟེར་གྱི་ཡོད་ལ། ཁྲིན་འདུ་རྫོང་མཁར་རོང་སྡེ་བ་ནས་ཡིན་པ་དང་། གྲྭ་བློ་བཟང་སྦྱིན་པ་ལགས་རང་ལོ་ ༣༢ ཡིན་ལ། ཕ་མིང་ལ་རྟ་རྒྱལ་དང་མ་མིང་ལ་པདྨ་མཚོ་མོ་ཟེར་བ་དང་། ཁྲིན་འདུ་རྫོང་གེ་ཤིང་སྡེ་པ་ནས་ཡིན་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
