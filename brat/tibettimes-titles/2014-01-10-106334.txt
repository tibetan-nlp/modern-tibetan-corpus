ཐེ་ཝན་གཞུང་ལ་བོད་མིའི་ཉམས་ཉེན་ཞུ་སྐུལ།

དེ་རིང་སྔ་དྲོར་བོད་རིགས་རྩོམ་པ་པོ་ཞིག་དང་རྒྱ་རིགས་རྩོམ་པ་པོ་ཞིག་བཅས་ཀྱིས་ཐེ་ཝན་ཁྲིམས་བཟོ་ལྷན་ཚོགས་ཀྱི་ལས་ཁུངས་ནང་ཐེ་ཝན་གཞུང་གིས་བོད་དོན་ལ་རྒྱབ་སྐྱོར་དགོས་པའི་འབོད་སྐུལ་ཞུས་འདུག འདི་གར་གནས་ཚུལ་བཏང་འབྱོར་བྱུང་བ་ལྟར་ན། བོད་རིགས་རྩོམ་པ་པོ་གནམ་ལོ་ཡག་དང་རྒྱ་རིགས་རྩོམ་པ་པོ་ཡོན་ཧོང་པེན་གཉིས་ཀྱིས་ཐེ་ཝན་ཁྲིམས་བཟོ་ལྷན་ཚོགས་ཀྱི་གཞུང་ལས་ཁང་དུ་གསར་འགོད་གསལ་བསྒྲགས་ཞིག་སྤེལ་ཏེ། ད་ལྟའི་ཐེ་ཝན་གཞུང་ལ། རྒྱ་ནག་དམར་ཤོག་གཞུང་གིས་སློབ་དཔོན་ཆེན་མོ་པཎ་ཆེན་སྐུ་ཕྲེང་བཅུ་བ་བཀྲོངས་པའི་ནག་ཉེས་བརྩད་ཞིབ་དང་ད་ལྟའི་བོད་ནང་གི་ཛ་དྲག་གནས་སྟངས་སེལ་བར་རྒྱབ་བསྐྱོར་ཡོང་བའི་ཞུ་བསྐུལ་ཞུས་ཡོད་འདུག གསར་འགོད་ལྷན་ཚོགས་གཙོ་སྐྱོང་བ་ཐེ་ཝན་དམངས་གཙོ་ཡར་ཐོན་ཚོགས་པའི་སྤྱི་འཐུས་ལིས་ཅོན་ཡེ་ལགས་ཀྱིས་བརྗོད་དོན།  རྒྱ་ནག་དམར་ཤོག་གཞུང་གི་གཉའ་གནོན་ལ་ངོ་རྒོལ་དང་མ་དགའ་བ་མཚོན་ཆེད། ན་ནིང་བོད་ནང་དུ་མེ་བསྲེགས་ངོ་རྒོལ་གྲངས་འབོར་ ༡༢༩ བསླེབས་ཡོད་ལ། འདི་ནི་མིའི་རིགས་ཀྱི་ལོ་རྒྱུས་སྟེང་གི་ཆེས་ཚ་རླབས་བརྡོལ་བ་ལྟ་བུའི་ངོ་རྒོལ་ལས་འགུལ་ཞིག་རེད། ཡིན་ནའང་། ཐེ་ཝན་མཱ་ཡོན་ཅིག་གཞུང་གིས་འདི་ལྟ་བུའི་ཛ་དྲག་གི་གནས་སྟངས་དེ་མ་མཐོང་མ་ཐོས་ཁུལ་བྱས་ཡོད་སྟབས། འདི་ནི་དམངས་གཙོའི་དར་ཆའི་འོག་གི་ནག་ཉེས་ཤིག་ཡིན། ཞེས་བརྗོད་ཡོད་འདུག སྐབས་དེར་ཡོན་ཧོང་པེན་གྱིས། ད་ལོ་ནི་སློབ་དཔོན་ཆེན་མོ་པཎ་ཆེན་སྐུ་ཕྲེང་བཅུ་བ་བཀྲོངས་ནས་ལོ་འཁོར་ ༢༥ དང་། པཎ་ཆེན་སྐུ་ཕྲེང་བཅུ་བའི་བཀྲོངས་རྐྱེན་ནི་ཆབ་སྲིད་ལྐོག་གསོད་རེད། དེའི་ཁར་པཎ་ཆེན་སྐུ་ཕྲེང་བཅུ་གཅིག་པ་ནི་འཛམ་གླིང་སྟེང་གི་ཆེས་ཆུང་བའི་ཆབ་སྲིད་བཙོན་མ་རེད་ལ། དེང་སྐབས་གར་ཡོད་སུས་ཀྱང་ཤེས་ཀྱི་མེད། ད་དུང་བོད་ནང་མེ་བསྲེགས་ངོ་རྒོལ་མུ་མཐུད་ནས་ཡོང་གི་ཡོད་སྟབས། རྒྱ་དམར་གཞུང་གིས་བོད་ནང་ནག་ཉེས་གནམ་སར་མི་ཤོང་བ་བསགས་ཡོད། རྩ་ཁྲིམས་ནང་བོད་དང་སོག་པོ་སོགས་རྒྱ་ནག་མི་སེར་རྒྱལ་ཁབ་ཀྱི་ཡིན་ཞེས་འཁོད་བཞིན་དུ་བོད་ནང་འབྱུང་བཞིན་པའི་ཛ་དྲག་གནས་སྟངས་ཐད་ཐེ་ཝན་མི་སེར་རྒྱལ་ཁབ་གཞུང་གིས་དོ་ཁུར་མི་བྱེད་པ་ནི་གཡོལ་ཐབས་མེད་པའི་དོན་དག་ཅིག་ཡིན། ཞེས་སོགས་བརྗོད་འདུག གནམ་ལོ་ཡག་གིས་བརྗོད་དོན། བོད་མིར་ཐེབས་ཀྱིན་པའི་གཉའ་གནོན་ནི་ཆབ་སྲིད་ཐད་ནས་གཉའ་གནོན། དཔལ་འབྱོར་ཐད་ནས་འཕྲོག་བཅོམ། ཤེས་རིག་དང་ཆོས་ལུགས་གཞན་དབང་དུ་གྱུར་བ་བཅས་ནས་མཚོན་ཐུབ། དཔེར་ན། བོད་ནང་རང་སྐྱོང་ཡོད་ལབ་བཞིན་དུ་ཧ་ལམ་ལས་ཁུངས་ཆེ་ཕྲ་ཚང་མའི་མགོ་བ་རྒྱ་རིགས་ཡིན་ལ། བོད་མིར་དབང་ཚད་ཅི་ཡང་ལྷག་མེད། ད་ལྟ་ཞཱང་རིམ་པའི་མགོ་ཁྲིད་ཀྱང་མི་རིགས་སློབ་ཆེན་ཁག་གི་བོད་རིག་པའི་སྡེ་ཁག་ནང་རྒྱ་རིགས་སྐྱེད་སྲིང་བྱེད་ཀྱིན་ཡོད། དཔྱད་གཞིའི་ཡིག་ཆ་ལྟར་ན། དུས་རབས་ཉི་ཤུའི་ལོ་རབས་དགུ་བཅུའི་མགོར། མཚོ་སྔོན་བོད་ཁུལ་གྱི་གཏེར་ཁའི་མ་གནས་རྒྱ་སྒོར་ཁྲི་དུང་ཕྱུར་ཆིག་འབུམ་བདུན་ཁྲི་བརྒལ་ཡོད་ཅིང་། ༢༠༡༠ ལོར་བོད་རང་སྐྱོང་ལྗོངས་ཀྱི་གཏེར་ཁའི་མ་གནས་ཀྱང་རྒྱ་སྒོར་ཁྲི་དུང་ཕྱུར་དྲུག་སྟོང་ལྷག་བརྒལ་ཡོད། ཡིན་ནའང་། བོད་མིའི་ལོ་རེའི་ཆ་སྙོམས་ཡོང་འབབ་ཨ་སྒོར་ ༤༠༠ ཡས་མས་ལས་མེད་ཅིང་། ཡོང་འབབ་དེ་དག་གི་བརྒྱ་ཆའི་ ༧༠ དབྱར་རྩ་དགུན་འབུ་ཡིན་སྟབས། གལ་སྲིད་དབྱར་རྩྭ་དགུན་འབུ་མེད་ཚེ། བོད་མི་ཕལ་ཆེ་བ་སྤྲང་བོར་གྱུར་ཡོད། ཁོང་གིས་མུ་མཐུད་དུ། ཤེས་རིག་ཐད་ནས་བཤད་ན། སློབ་གསོའི་མ་ལག་ནང་རྒྱའི་སྐད་ཡིག་གིས་བོད་ཡིག་གི་ཚབ་བྱས་ཡོད་སྟབས། བོད་ཕྲུག་ཚོས་བོད་ཡིག་སྦྱོང་བར་ཚད་བཀག་ཆེན་པོ་ཐེབས་ཡོད། ཆོས་ལུགས་ཚོགས་སྡེ་ཁག་གི་འཛིན་སྐྱོང་འགན་དབང་ངམ། སྤྲུལ་སྐུ་ངོས་འཛིན་དང་དགོན་པར་དོ་དམ་ཚུད་པའི་དབང་ཆ་ཚང་མ་གུང་ཁྲན་ཏང་གིས་གཞུང་བཞེས་བཏང་ཡོད། མདོར་ན། བོད་ནང་མེར་བསྲེགས་ཚུད་པའི་ངོ་རྒོལ་སྣ་མང་གི་འབྱུང་རྐྱེན་དངོས་ནི། ༼གང་དུ་གཉའ་གནོན་ཡོད་ན། དེ་གར་ངོ་རྒོལ་ཡོད༽ཅེས་པའི་བཤད་རྒྱུན་དེས་མཚོན་ཐུབ་ཀྱི་ཡོད། ཅེས་སོགས་བཀོད་འདུག གནས་ཚུལ་ཁོ་སྤྲོད་པའི་ངོ་སྤྲོད་གནང་བ་ལྟར་ན། དེ་རིང་གི་གསར་འགོད་གསལ་བསྒྲགས་དེ་རྒྱལ་སྤྱིའི་གསར་ལམ་ཁག་ ༣༠ ལྷག་ལ་ཁྱབ་སྤེལ་ཕྱིན་ཡོད་འདུག
