རྒྱ་གར་རང་བཙན་ཉིན་མོ་སྲུང་གཙི་གནང་།

བོད་མིའི་སྒྲིག་འཛུགས་ཀྱིས་རྒྱ་གར་རང་བཙན་ཐོབ་སྟེ་ལོ་ངོ་ཧྲིལ་པོ་ ༧༡ འཁོར་བའི་ཉིན་མོ་སྲུང་གཙི་གནང་སོང་། དེ་རིང་སྔ་དྲོའི་ཆུ་ཚོད་༩ ཐོག་བོད་མིའི་སྒྲིག་འཛུགས་བཀའ་ཤག་གི་མདུན་ཐང་དུ། རྒྱ་གར་དབྱིན་ཇིའི་དབང་སྒྱུར་འོག་ནས་རང་བཙན་ཐོབ་སྟེ་ལོ་ངོ་ཧྲིལ་བོ་༧༡ འཁོར་བའི་ཉིན་མོ་སྲུང་གཙི་གནང་ཡོད་པ་དང་། ཉིན་མོ་སྲུང་གཙིའི་ཐོག་སྲིད་སྐྱོང་བློ་བཟང་སེང་གེ་མཆོག་གི་དབུས་བཀའ་བློན་རྣམ་པ་དང་སྲི་ཞུ་ལས་བྱེད་ཁག་ཅིག་ནས། རྒྱ་གར་གྱི་རྒྱལ་གླུ་གྱེར་བཞིན་རྒྱལ་དར་སྒྲེང་འཛུགས་གནང་སོང་། རྒྱལ་དར་སྒྲེང་འཛུགས་ཀྱི་མཛད་སྒོ་གྲུབ་རྗེས་སྲིད་སྐྱོང་བློ་བཟང་སེང་གེ་མཆོག་གིས། རྒྱ་གར་དེ་འཛམ་གླིང་གི་རྒྱལ་ཁབ་སྟོབས་འབྱོར་ལྡན་པ་ཞིག་དང་དཔལ་འབྱོར་ཐད་ནས་འཛམ་གླིང་ཡོངས་ནས་འཕེལ་རྒྱས་འགྲོ་བཞིན་ཡོད་པས། རྒྱ་གར་མི་མང་ཚོས་དཔལ་འབྱོར་གྱི་ཁེ་ཕན་དང་སྟོབས་འབྱོར་མངའ་ཐང་གི་དཔལ་ཡོན་ལོངས་སུ་སྤྱོད་པ་དང་། དེ་བཞིན་རྒྱ་གར་གཞུང་མང་གཉིས་ཀྱིས་༧གོང་ས་མཆོག་གི་གཙོའི་བོད་མིའི་སྒྲིག་འཛུགས་དང་བོད་མི་ཚོར་གནས་ཚང་གཡར་བར་ཐུགས་རྗེ་ཆེ་ཞུས་པ་ཡིན་ལ། རྒྱ་གར་གཞུང་མང་གཉིས་ཀྱིས་ང་ཚོར་རོགས་རམ་གནང་བ་ཞིག་རྒྱལ་ཁབ་སུས་ཀྱང་གནང་མེད་པས། དེ་རིང་རྒྱ་གར་གྱི་རང་བཙན་དུས་དྲན་ཉིན་མོར་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་ཚབ་བྱས་ཏེ་རྒྱ་གར་གཞུང་མང་གཉིས་ལ་སེམས་གཏིང་ནས་ཐུགས་རྗེ་ཆེ་ཞུས་པ་ཡིན། ཞེས་གསུངས་སོང་། ཁོང་གིས་རྒྱ་གར་རང་བཙན་ཉིན་མོ་དང་འབྲེལ་ནས་བོད་མི་ཚོར། རྒྱ་གར་རང་བཙན་ཤོར་ནས་དབྱིན་ཇིའི་འོག་ཏུ་ལོ་གཉིས་བརྒྱ་ལྷག་ཅིག་བསྡད་པ་རེད། དེའི་སྔོན་ལ་ཁ་ཆེའི་འོག་ཏུ་ལོ་དེ་ལས་མང་བ་བསྡད་པ་རེད་དེ། རྒྱ་གར་མགོ་ཁྲིད་ཚོ་དང་མི་མང་ལ་སེམས་ཤུགས་མ་ཆགས་པའི་ཐོག་ནས། གཙོ་ཆེ་ཤོས་མི་རིགས་ཀྱི་ངོ་བོ་མ་བརླག་པ་བྱས་ཏེ་རྩ་དོན་རྩད་པས་རྒྱ་གར་རང་བཙན་དེ་ཕྱིར་ཐོབ་པ་བྱུང་ཡོད་པ་དང་། དེ་བཞིན་གཱན་དྷི་ཇིའི་ལྟ་གྲུབ་གཞིར་བཟུང་ནས་ཞི་བའི་ལམ་ནས་འཐབ་རྩོད་བྱས་ཏེ་རྩ་དོན་འགྲུབ་ཐུབ་པ་བྱུང་ཡོད་དུས། ང་ཚོ་ཡིན་ན་དེ་ལྟ་བུའི་ཉི་མ་ཞིག་ཏན་ཏན་ཤར་གྱི་ཡིན་དུས་ང་རང་ཚོའི་མི་རིགས་ཀྱི་ངོ་བོ་སྲུང་སྐྱོབ་དང་རྩ་དོན་རྩོད་དགོས་པ་གལ་ཆེན་པོ་རེད། ཅེས་གསུངས་སོང་། ཕྱི་ལོ་ ༡༩༤༧ ལོ་ནས་རྒྱ་གར་དབྱིན་ཇིའི་འོག་ནས་རང་བཙན་ཐོབ་ཡོད་པ་དང་། ཉིན་མོ་འདིར་གཙོ་ཆེ་ཤོས་རྒྱ་གར་དབྱིན་ཇིའི་འོག་ཏུ་ཡོད་སྐབས་རང་བཙན་འཐབ་རྩོད་བྱེད་མཁན་ཀྱི་མི་སྣ་ཚོར་རྗེས་དྲན་ཞུ་བཞིན་ཡོད་པ་རེད། གསར་འགོད་པ། སངས་རྒྱས་དཀོན་མཆོག
