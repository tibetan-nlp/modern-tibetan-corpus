རྡོ་རྗེ་རིན་ཆེན་ལཊ་འཛིན་བཟུང་།

ཁ་སང་དཀར་མཛེས་ཁུལ་གསེར་རྟ་རྫོང་ནས་བོད་མི་རྡོ་རྗེ་རིན་ཆེན་ཞུ་བ་ཞིག་གིས་ངོ་རྒོལ་སྐད་འབོད་གནང་སྟེ་འཛིན་བཟུང་བྱས་ཡོད་འདུག འདི་གར་ཆབ་སྲིད་བཙོན་ཟུར་མགོ་ལོག་འཇིགས་མེད་ལགས་ཀྱིས་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། ཁ་སང་ཉིན་རྒྱབ་བོད་ནང་གི་ཆུ་ཚོད་ ༢ ཡོལ་ཙམ་ལ། དཀར་མཛེས་ཁུལ་གསེར་རྟ་རྫོང་གི་བོད་མི་རྡོ་རྗེ་རིན་ཆེན་ཞུ་བ་ཞིག་གིས། རྫོང་ཁྲོམ་གཞུང་དུ་རླུང་རྟ་གཏོར་བཞིན་རྒྱ་གཞུང་ལ་ངོ་རྒོལ་སྐད་འབོད་གནང་ཡོད་པ་དང་། ཅུང་ཙམ་གྱི་རྗེས་སུ་རྫོང་ཉེན་རྟོག་པ་བསླེབས་ཏེ་འཛིན་བཟུང་བཙན་འཁྲིད་བྱས་ཡོད་པ་དང་། ད་ལྟའི་ཆར་ཁོང་གང་དུ་བཀག་ཉར་བྱས་ཡོད་མེད་དང་ས་གནས་ཀྱི་གནས་སྟངས་སོགས་ཤེས་རྟོགས་མ་ཐུབ་པར་ལྷག་ཡོད་འདུག ས་གནས་ནས་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་མཁན་གྱིས། རྒྱ་གཞུང་གི་རྒྱུ་འབྲས་དང་གནས་ལུགས་མེད་པའི་དྲག་གནོན་གྱི་བྱ་འགུལ་མང་པོ་མཐོང་རྒྱུ་ཡོད་པ་དང་། བོད་མིའི་མགོ་དཔོན་ཚོས་འོག་གི་འབངས་མི་ཡོངས་ལ་དམའ་འབེབས་དང་བརྙས་བཅོས། བཙན་གནོན། དབང་བསྒྱུར་བྱས་པ་སོགས་ཚད་ལས་བརྒལ་བའི་བྱ་སྤྱོད་དག་སེམས་ནས་མ་བཟོད་པར། ཞི་བདེ་དང་རང་དབང་ཆེད་ཁོང་གིས་རང་སྲོག་ལ་མ་འཛེམས་པར་ཞི་བདེའི་ངོ་རྒོལ་བྱས་ཏེ་འཛིན་བཟུང་བྱས་སོང་། ཞེས་བརྗོད་འདུག རྡོ་རྗེ་རིན་ཆེན་ནམ་(རྡོ་རིས)ལགས་ནི་དཀར་མཛེས་ཁུལ་གསེར་རྟ་རྫོང་ཧོར་ཤུལ་ཡུལ་ཚོ་ནས་ཡིན་ཞིང་། ཨ་མའི་མིང་ལ་རིག་མཚོ་ཟེར་བ་དང་། རྡོ་རྗེ་རིན་ཆེན་ལགས་འཛིན་བཟུང་མ་བྱས་གོང་གསེར་རྟ་བླ་རུང་སྒར་དུ་ཚོང་ལས་གཉེར་ཡོད་པ་དང་། དེ་སྔོན་གནུབས་ཟུར་དགོན་པའི་གྲྭ་པ་ཡིན་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
