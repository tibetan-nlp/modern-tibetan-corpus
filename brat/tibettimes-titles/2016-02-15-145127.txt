རྒྱ་ནག་གི་ཚོང་རའི་གཡོ་སྐྱོམ།

འདི་ལོའི་ཕྱི་ཟླ་དང་པོའི་ནང་རྒྱ་ནག་གི་ཚོང་ཟོག་ཕྱིར་གཏོང་ནང་འདྲེན་ཐད་གཏོང་མང་ཡོང་ཉུང་གི་ཧེ་བག་རྒྱ་སྒོར་དུང་ཕྱུར་ ༤༠༦༢ ཤོར་ཡོད་སྐོར་བསྒྲགས་འདུག   དེ་རིང་རྒྱ་ནག་མཚོ་འགག་སྤྱི་ཁྱབ་ལྟ་སྐུལ་དོ་དམ་ཁང་གིས། འདི་ལོའི་ཕྱི་ཟླ་དང་པོའི་ནང་རྒྱ་ནག་གིས་ཐོན་རྫས་ཕྱིར་ཚོང་ནང་འདྲེན་བྱས་པའི་གྲངས་འབོར་ཡོངས་བསྒྲགས་བྱས་པ་ལྟར་ན། འདས་པའི་ཟླ་གཅིག་ནང་ཁྱོན་རྒྱ་སྒོར་དུང་ཕྱུར་ཁྲི་ ༡.༨༨ ཀྱིས་ཐོན་རྫས་ཕྱིར་ཚོང་ནང་འདྲེན་བྱས་ཡོད་པའི་ཁྲོད། རྒྱ་ནག་གི་ཐོན་རྫས་ཕྱིར་ཚོང་བྱས་པ་རྒྱ་སྒོར་དུང་ཕྱུར་ཁྲི་ ༡.༡༤ ཟིན་ཡོད་ཀྱང་། སྔར་དང་བསྡུར་ན་བརྒྱ་ཆ་ ༦.༦ མར་ཆག་ཡོད་པ་དང་། དེ་བཞིན་ཕྱིའི་ཐོན་རྫས་ནང་འདྲེན་བྱས་པ་རྒྱ་སྒོར་དུང་ཕྱུར་ ༧༣༧༥.༤ ཡིན་པས་བརྒྱ་ཆ་ ༡༤.༤ མར་ཆག་པ་བཅས། ཕྱིར་ཚོང་ནང་འདྲེན་གྱི་གྲངས་འབོར་ཐད་སྔར་དང་བསྡུར་ན་བརྒྱ་ཆ་ ༩.༨ ཉུང་དུ་ཕྱིན་ཡོད་པ་མ་ཟད། ཕྱིའི་ཚོང་འབྲེལ་ཐད་གཏོང་མང་ཡོང་ཉུང་གི་ཧེ་བག་རྒྱ་སྒོར་དུང་ཕྱུར་ ༤༠༦༢ ཤོར་ཡོད་འདུག   མཚོ་འགག་སྤྱི་ཁྱབ་ལྟ་སྐུལ་དོ་དམ་ཁང་གི་དབྱེ་ཞིབ་ལྟར་ན། ཡུ་རོབ་མཉམ་འབྲེལ་མཐུན་ཚོགས་ནི་རྒྱ་ནག་གི་ཚོང་རོགས་ཤུགས་ཆེ་ཤོས་དེ་ཡིན་ཞིང་། འདི་ལོའི་ཕྱི་ཟླ་ ༡ ནང་རྒྱ་ནག་དང་ཡུ་རོབ་མཉམ་འབྲེལ་བར་རྒྱ་སྒོར་དུང་ཕྱུར་ ༢༩༠༣.༣ དང་། རྒྱ་ནག་དང་ཨ་རིའི་བར་རྒྱ་སྒོར་དུང་ཕྱུར་ ༢༦༩༨.༩ ཡི་ཚོང་འབྲེལ་བྱུང་ཡོད་ཀྱང་། སྔར་དང་བསྡུར་ན་བརྒྱ་ཆ་ ༩.༩ རེ་མར་ཆག་ཡོད་འདུག དེ་བཞིན་རྒྱ་ནག་དང་ཡ་གླིང་ཤར་ལྷོའི་རྒྱལ་ཁབ་མཉམ་འབྲེལ་མཐུན་ཚོགས་བར་རྒྱ་སྒོར་དུང་ཕྱུར་ ༢༣༤༢.༥ ཡི་ཚོང་འབྲེལ་བྱུང་ཡོད་ཀྱང་སྔར་ལས་བརྒྱ་ཆ་ ༡༠.༨ ཉུང་དུ་ཕྱིན་ཡོད་པ་དང་། རྒྱ་ནག་དང་ཉི་ཧོང་བར་རྒྱ་སྒོར་དུང་ཕྱུར་ ༡༣༡༥.༤ ཡི་ཚོང་འབྲེལ་བྱུང་ཡོད་ཀྱང་བརྒྱ་ཆ་ ༦ མར་ཆག་ཡོད་འདུག རྒྱ་ནག་གི་ཐོན་རྫས་རིགས་ལས་གློག་ཆས་དང་གྱོན་གོས། ཟས་རིགས་སོགས་ཕྱིར་ཚོང་གི་གྲངས་འབོར་མར་ཆག་ཡོད་ཀྱང་། གཏེར་རྡོ་ལྕགས་རིགས་དང་ལས་སྣོན་བྱས་པའི་སྣུམ་སོགས་ཀྱི་གྲངས་འབོར་ཡར་འཕར་ཡོད་པ་དང་། ནང་འདྲེན་གྱི་ཚོང་ཟོག་རིན་གོང་སྤྱི་ཡོངས་ནས་མར་ཆག་ཡོད་འདུག   ད་ལྟའི་ཆར་རྒྱ་ནག་འཛམ་གླིང་ཐོག་གི་ཕྱི་དངུལ་བརྗེ་བསྒྱུར་གྱི་མ་འཇོག་འབོར་ཆེ་ཤོས་ཀྱི་རྒྱལ་ཁབ་ཆགས་ཡོད་ཀྱང་། འདས་པའི་ཟླ་ ༦ ནང་ཨ་སྒོར་དུང་ཕྱུར་ ༤༢༠༠ མར་ཆག་སྟེ། ཕྱི་ལོ་ ༢༠༡༢ ལོ་ཚུན་ཆད་ཀྱི་ཆེས་དམའ་ཤོས་ཤིག་ཏུ་ལྷུང་ཡོད་པ་མ་ཟད། ཟླ་གཅིག་ཁོ་ནའི་ནང་རྒྱ་ནག་མི་མང་དངུལ་ཁང་གི་ཨ་སྒོར་མ་འཇོག་དུང་ཕྱུར་ ༩༩༥ མར་ཆག་ཡོད་འདུག གནད་དོན་དེའི་ཐད་རྒྱ་ནག་དབུས་དངུལ་ཁང་གི་མགོ་ཁྲིད་ཀྲུའུ་ཞོ་ཁྲན་གྱིས་གཞི་རྩའི་སྒྲིག་བཀོད་ལ་གནད་དོན་མ་བྱུང་ན། ཕྱི་དངུལ་མ་འཇོག་ཡར་འཕར་བ་དང་ཡང་ན་མར་ཆག་པ་རྒྱུན་ལྡན་ཡིན་སྐོར་བརྗོད་འདུག   རྒྱ་གཞུང་གིས་ཕྱི་ལོ་ ༢༠༡༥ ལོར་རྒྱའི་དཔལ་འབྱོར་ཐད་བརྒྱ་ཆ་ ༦.༩ ཡར་འཕར་ཡོད་སྐོར་བསྒྲགས་ཡོད་ཀྱང་། དེ་ནི་ཕྱི་ལོ་ ༡༩༩༠ ལོ་ནས་ད་ལྟའི་བར་གྱི་ཚད་གཞི་དམའ་ཤོས་ཤིག་ཆགས་ཡོད་སྟབས། འཛམ་གླིང་གི་མ་རྩ་འཇོག་མཁན་མང་པོ་ཞིག་གིས་རྒྱ་ནག་གི་མ་རྐང་ཁྲོམ་རའི་འཁོར་རྒྱུག་ལ་ཡིད་ཆེས་བརླགས་ཏེ་ཕྱིར་འཐེན་བྱས་ཡོད་པ་རེད། དེའི་ཐད་རྒྱ་ནག་དབུས་དངུལ་ཁང་གི་མགོ་ཁྲིད་ཀྲུའུ་ཞོ་ཁྲན་གྱིས་མ་རྩའི་འཁོར་རྒྱུག་དང་མ་རྩ་ཕྱིར་འཐེན་གཉིས་ཐ་དད་ཡིན་སྟབས། རྒྱ་ནག་གིས་གནད་དོན་དེའི་རྐྱེན་གྱིས་མ་རྩའི་འཁོར་རྒྱུག་ལ་སྟངས་འཛིན་ཤུགས་སྣོན་བྱེད་མི་སྲིད་སྐོར་བརྗོད་ཡོད་པ་རེད།   དཔལ་འབྱོར་དབྱེ་ཞིབ་པ་ཚོས་རྒྱ་སྒོར་གྱི་འཛའ་ཐང་ཆག་པ་དང་ཕྱི་དངུལ་བརྗེ་བསྒྱུར་གྱི་མ་འཇོག་མྱུར་ཚད་མཐོན་པོས་མར་ལྷུང་བ་དེ། མཐའ་གཅིག་ཏུ་དོན་ངན་ཞིག་མིན་པར་རྒྱའི་ཐོན་རྫས་ཕྱིར་ཚོང་གི་གྲངས་འབོར་ཡར་འཕར་བར་ཕན་ཐོགས་ཡོད་ཀྱང་། རྒྱ་ནག་གི་མ་རྐང་ཁྲོམ་རའི་ནང་མ་རྩ་འཇོག་མཁན་ཚོའི་ཡིད་ཆེས་བསྐྱར་གསོ་ཡོང་ཆེད། ངེས་པར་དུ་རྒྱ་ནག་གི་དཔལ་འབྱོར་སྒྲིག་གཞི་གཏིང་ཟབ་ས་ནས་བསྒྱུར་བཅོས་གཏོང་དགོས་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
