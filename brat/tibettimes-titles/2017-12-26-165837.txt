ཨ་རིས་མཉམ་འབྲེལ་རྒྱལ་ཚོཊ་ལ་ཉེས་ཆད་བཅད།

 ཨ་རིའི་མི་མང་གི་དངུལ་སྣང་མེད་ཀྱིས་གཞན་ལ་ཁེ་སྤོགས་ཟ་རུ་འཇུག་གི་མིན་ཞེས་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ལ་སྤྲོད་ཀྱི་ཡོད་པའི་ཨ་རིའི་རོགས་དངུལ་ལས་ཨ་སྒོར་ས་ཡ་ ༢༨༤ ཕྱིར་འཐེན་བྱས་ཡོད་འདུག ཉེ་སྔོན་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་ཚོགས་ཞུགས་རྒྱལ་ཁབ་ ༡༩༣ གིས་འོས་བླུགས་ཏེ་ཨ་རིའི་ཇེ་རོ་སཱ་ལམ་གྲོང་ཁྱེར་གྱི་ཐག་གཅོད་ཕྱིར་འཐེན་དགོས་པའི་མང་མོས་བྱུང་ཡོད་པ་དང་། ཨ་རིའི་སྲིད་འཛིན་ཊོམ་ཕུས་ཨ་རིར་རྒྱབ་འགལ་བྱས་ན་རོགས་དངུལ་མཚམས་འཇོག་བྱ་རྒྱུ་ཡིན་པའི་ཉེན་བརྡ་བཏང་ཡོད་པ་ལྟར། ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༤ ཉིན་ཨ་རིའི་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་སྐུ་ཚབ་ལྕམ་ནི་ཁི་ཧ་ལེ་ཡིས་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་རྩིས་ལོ་ ༢༠༡༨ དང་ ༢༠༡༩ ཡི་སྔོན་རྩིས་ལས་ཨ་རིའི་རོགས་དངུལ་ཨ་སྒོར་ས་ཡ་ ༢༨༤ ཕྱིར་འཐེན་བྱས་པ་མ་ཟད། མ་འོངས་པར་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་དོ་དམ་ནུས་པ་དང་རྒྱབ་སྐྱོར་ནུས་པ་ཆུང་དུ་གཏོང་རྒྱུ་ཡིན་སྐོར་བསྒྲགས་འདུག མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་དྲུང་ཆེ་གྷ་ཊིར་རི་སི་(António Guterres)ལགས་ཀྱིས་མ་འོངས་ལོ་གཉིས་ཀྱི་སྔོན་རྩིས་ཨ་སྒོར་དུང་ཕྱུར་ ༥.༤ ལས་བརྒལ་རྒྱུ་མེད་སྐོར་བརྗོད་ཡོད་པ་དང་། དེ་ནི་རྩིས་ལོ་ ༢༠༡༦ དང་ ༢༠༡༧ ལོའི་སྔོན་རྩིས་དང་བསྡུར་ན་ཨ་སྒོར་དུང་ཕྱུར་ ༢ ཉུང་དུ་ཕྱིན་ཡོད་པ་རེད། ཨ་རིའི་གཞུང་ཚབ་ལྕམ་ནི་ཁི་ཧ་ལེ་ལགས་ཀྱིས།  མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་འགྲོ་གྲོན་གཏོང་ཡུལ་ཁག་ཅིག་ནི་ཚད་ལས་བརྒལ་ཡོད་པ་དང་། ཡང་ཁག་ཅིག་ཉུང་དྲགས་པའི་གནས་སྟངས་མངོན་གསལ་དོད་པོ་ཆགས་ཡོད་སྟབས།  ང་ཚོས་མུ་མཐུད་ཨ་རིའི་མི་མང་གི་གཏོང་ཕོད་སྣང་མེད་དུ་བཏང་སྟེ་འཇོག་གི་མིན་ལ། མི་གཞན་ལ་ཁེ་སྤོགས་ཀྱང་ཟ་རུ་འཇུག་གི་མིན། ཐེངས་འདིའི་རོགས་དངུལ་ཕྱིར་འཐེན་བྱས་པ་དེས་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ལ་ཚོད་འཛིན་དང་འགན་ལེན་རང་བཞིན་གྱི་བརྡ་ལན་ཞིག་སྤྲོད་ཀྱི་ཡོད་སྟབས།  འདི་ནི་བཟང་ཕྱོགས་ཀྱི་ལམ་ཁར་བསྒྱུར་བ་ཞིག་ཆགས་ཡོད། འདི་ལོའི་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་སྔོན་རྩིས་གཏན་འབེབས་བྱུང་བར་ང་ཚོ་དགའ་པོ་བྱུང་ཡོད་ལ།  མ་འོངས་པར་ང་ཚོས་ཨ་རིའི་མི་མང་གི་མངོན་འདོད་ལ་སྲུང་སྐྱོབ་བྱས་པའི་ཐོག་ནས་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ལ་འགྲོ་གྲོན་འདང་ངེས་ཅན་ཞིག་ཡོང་བར་ཐབས་ཤེས་གང་ཡོད་བྱ་རྒྱུ་ཡིན།  ཞེས་བརྗོད་ཡོད་པ་རེད། ཨ་རི་ནི་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་རོགས་དངུལ་སྤྱིན་བདག་ཆེ་ཤོས་དེ་ཡིན་པ་དང་། ད་ལྟའི་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་ལོ་རེའི་རྒྱུན་ལྡན་འགྲོ་གྲོན་གྱི་བརྒྱ་ཆ་ ༢༢ ཏེ། ཨ་སྒོར་ས་ཡ་ ༣༣ ཙམ་ཨ་རིའི་གཞུང་ནས་གཏོང་བཞིན་ཡོད་པ་མ་ཟད། ཞི་བདེའི་སྲུང་དམག་གི་འགྲོ་གྲོན་བརྒྱ་ཆ་ ༢༨.༥ ཨ་རིས་གཏོང་གི་ཡོད་འདུག ཡིན་ནའང་། ཧྥ་རན་སིའི་རྒྱལ་ས་ཕ་རི་སིར་རྟེན་གཞི་བྱས་པའི་(Agence France-Presse)གསར་ཁང་གི་གནས་ཚུལ་ལྟར་ན། འདི་ལོའི་ཕྱི་ཟླ་ ༦ པའི་ནང་གཏན་འབེབས་བྱས་པའི་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་རྩིས་ལོ་ ༢༠༡༧ དང་ ༢༠༡༨ ལོའི་ཞི་བདེའི་སྲུང་དམག་གི་འགྲོ་གྲོན་ལས་ཨ་རིའི་རོགས་དངུལ་དུང་ཕྱུར་ ༦ ཕྱིར་འཐེན་བྱས་ཡོད་འདུག ཨ་རིའི་རོགས་དངུལ་ཐེབས་རྩ་ཁང་གི་སྙན་ཐོའི་ནང་།  ཕྱི་ལོ་ ༢༠༡༦ ལོར་ཨ་རིའི་གཞུང་གིས་དཀྱིས་ཤར་ཡུལ་གྲུའི་རྒྱལ་ཁབ་ཁག་ཏུ་དཔལ་འབྱོར་འཛུགས་སྐྲུན་དང་དམག་དོན་ཆེད་ཨ་སྒོར་དུང་ཕྱུར་ ༡༣ གྱི་འགྲོ་གྲོན་བཏང་ཡོད་པ་དང་། བྱང་དང་ལྷོ་ཨ་ཧྥེ་རི་ཁའི་རྒྱལ་ཁབ། དེ་བཞིན། ཨེ་ཤ་ཡའི་ལྟེ་གནས་རྒྱལ་ཁབ་ཁག་ཏུ་ཨ་སྒོར་དུང་ཕྱུར་ ༦།༧ ། ཡུ་རོབ་དང་ཨེ་རཱ་ཤི་ཡར་ཨ་སྒོར་དུང་ཕྱུར་ ༡།༥། ནུབ་ཕྱོགས་རྒྱལ་ཁབ་དབྱིན་ཇི་དང་། ཨུ་རུ་སུ། ནེའུ་ཛི་ལན་ཌི་སོགས་ལ་ཨ་སྒོར་དུང་ཕྱུར་ ༢།༢ ཀྱིས་རོགས་སྐྱོར་བྱས་ཡོད་པ་རེད། གསར་འགོད་པ།  སྐལ་བཟང་སྦྱིན་པ།
