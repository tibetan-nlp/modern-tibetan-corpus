རྒྱ་ནག་ནང་ཨ་ཧྥི་རི་ཀའི་འཁྱམ་པོ་ཁྲི་ ༣༠
འདུག

རྒྱ་ནག་ཀོང་ཀྲུའུ་གྲོང་ཁྱེར་འཕེལ་རྒྱས་ཞིབ་འཇུག་ཁང་གི་སྙན་ཐོ་ལྟར་ན་ཀོང་ཀྲུའུ་རུ་ཨ་ཧྥི་རི་ཀའི་འཁྱམ་པོ་ཁྲི་ ༣༠ ཙམ་ཡོད་འདུག ཉེ་སྔོན་ཀོང་ཀྲུའུ་གྲོང་ཁྱེར་འཕེལ་རྒྱས་ཞིབ་འཇུག་ཁང་གིས་༼ཀོང་ཀྲུའུ་ཕྱི་མི་འཁྱམ་པོ་དོ་དམ་གྱི་གནས་སྟངས་ཞིབ་འཇུག༽ཅེས་པའི་རྟོག་ཞིབ་སྙན་ཐོ་ཞིག་བཏོན་པའི་ནང་། ཀོང་ཀྲུའུ་གྲོང་ཁྱེར་ནི་ཨེ་ཤ་ཡ་ཁུལ་གྱི་ཨ་ཧྥི་རི་ཀ་ལྷོ་ནུབ་ཁུལ་གྱི་ཨིན་གྷོ་ལཱ་དང་མཱ་ལི། གྷ་ནཱ། ནི་གྷར་སོགས་རྒྱལ་ཁབ་ཆུང་ཆུང་ཁག་ནས་ཡོང་བའི་འཁྱམ་པོ་གནས་སྡོད་བྱ་ཡུལ་ཆེ་ཤོས་ཤིག་ཆགས་ཡོད་པ་དང་། ད་ལྟའི་ཆར་ཨ་ཧྥི་རི་ཀའི་འཁྱམ་པོ་ཁྲི་ ༢༠ ནས་ ༣༠ ཙམ་གྲོང་ཁྱེར་དེར་གནས་སྡོད་བྱས་ཡོད་པ་དང་། ཁོ་ཚོ་མང་ཆེ་བ་ཁྲིམས་འགལ་གྱིས་འཚོ་སྡོད་བྱས་ཏེ། རྒྱུན་དུ་ཨ་ཧྥི་ཀ་དང་རྒྱ་ནག་བར་འགྲོ་འོང་བྱེད་བཞིན་ཡོད་སྟབས། ས་གནས་ཀྱི་གྲོང་མི་ཚོས་དེང་སྐབས་ཨ་ཧྥི་རི་ཀའི་ནང་ཁྱབ་ཀྱིན་པའི་ཨེ་བྷོ་ལ་འགོས་ནད་རྒྱ་ནག་ནང་ཁྱབ་པར་སེམས་ཁུར་བྱེད་བཞིན་ཡོད་འདུག ཀོང་ཀྲུའུ་ཡི་ཨ་ཧྥི་རི་ཀའི་འཁྱམ་པོ་མང་ཆེ་བ་ཚོང་འབྲེལ་ཐོག་ལས་ཀ་བྱེད་བཞིན་པའི་ཚོང་པ་དང་། རྒྱ་ནག་ནང་རྒྱུན་སྤྱོད་ཅ་དངོས་རིགས་དང་བཟོ་སྐྲུན་རྒྱུ་ཆ་སོགས་ཐོན་སྐྱེད་བྱས་ཏེ་རྒྱུན་དུ་རྒྱ་ནག་དང་ཨ་ཧྥི་རི་ཀའི་བར་འགྲོ་འོང་བྱེད་ཀྱི་ཡོད་པ་དང་། ཡང་ཁ་ཤས་རྒྱ་ནག་དང་ཨ་ཧྥི་རི་ཀའི་ཚོང་པ་ཚོའི་གཡོག་པོ་བྱས་ཏེ། ཚོང་ཟོག་ཐོན་སྐྱེད་དང་སྐྱེལ་འདྲེན་སོགས་ཀྱི་ལས་ཀ་ཤུགས་ཆེན་བྱེད་ཀྱི་ཡོད་པ་དང་། དེ་བཞིན་ཀོང་ཀྲུའུ་རུ་སློབ་སྦྱོང་བྱེད་བཞིན་པའི་སློབ་མ་ཡང་ཡོད་འདུག ཀོང་ཀྲུའུ་གྲོང་ཁྱེར་གྱི་རྒྱ་མོ་ཞིག་གིས། དེང་སྐབས་ཨ་ཧྥི་རི་ཀའི་ནང་བཅོས་ཐབས་མེད་པའི་ཨེ་བྷོ་ལ་འགོས་ནད་ཁྱབ་ཀྱི་ཡོད་སྟབས། ཀོང་ཀྲུའུ་ནང་ཨ་ཧྥི་ཀའི་མི་ཡིས་འགྲོ་འོང་ཤུགས་ཆེན་བྱེད་བཞིན་པར་སེམས་ཁུར་ཡོད་པ་དང་། སྲིད་གཞུང་གིས་གནད་དོན་དེའི་ཐད་ཕྱི་ལྷོད་ཅིང་ནང་དམ་པའི་སྲིད་བྱུས་ལག་བསྟར་བྱེད་ཀྱི་ཡོད་པའི་རེ་བ་ཡོད་ལུགས་བརྗོད་འདུག་ལ། སྤྱི་ཚོགས་གནས་འཕྲིན་དྲ་བ་ཁག་ནས་ཀོང་ཀྲུའུ་ནང་ཨ་ཧྥི་རི་ཀའི་འཁྱམ་པོ་ཇེ་མང་དུ་སོང་བ་དང་བསྟུན་ཏེ་གྲོང་ཁྱེར་ནང་གནད་དོན་ཡང་རིམ་པར་བྱུང་བཞིན་ཡོད། ཕྱི་ལོ་ ༢༠༠༩ ལོར་ཨ་ཧྥེ་རུ་ཁའི་མི་ཚོས་ཀོང་ཀྲུའུ་གྲོང་ཁྱེར་ཉེན་རྟོག་ཁང་གི་མཐའ་སྐོར་ཏེ་ངོ་རྒོལ་བྱས་པ་དང་། ཕྱི་ལོ་ ༢༠༡༢ ལོར་ཡང་ངོ་རྒོལ་དོན་རྐྱེན་བསླངས་ཡོད་པ་དང་། གཞན་ཡང་། སྤྱི་ཚོགས་ཐོག་མངོན་གསལ་མིན་པའི་གནད་དོན་བཙན་གཡེམ་དང་གཉེན་སྒྲིག་མ་བྱས་པར་བྱིས་པ་སྐྱེས་བ། དེ་བཞིན་ཨེ་ཙི་འགོས་ཁྱབ་སོགས་ཀྱི་གནད་དོན་མང་པོ་བཟོས་ཏེ། ཀོང་ཀྲུའུ་གྲོང་ཁྱེར་མི་མང་གི་སྤྱི་ཚོགས་འཚོ་བ་དང་གྲོང་ཁྱེར་བདེན་འཇགས་ཐད་རྙོག་དྲ་དང་དཀའ་ངལ་མང་པོ་བཟོ་བཞིན་ཡོད་ཀྱང་། སྲིད་གཞུང་གིས་སྟངས་འཛིན་གང་ཡང་བྱེད་ཀྱི་མེད་པའི་སྐྱོན་བརྗོད་བྱེད་བཞིན་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
