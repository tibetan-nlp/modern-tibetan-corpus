
བོད་པའི་དགོན་སྡེ་ཁག་ནང་ཧི་མ་ལ་ཡའི་རི་རྒྱུད་བརྒྱ་ཆ་ ༥༥།༨
ཆགས་འདུག

དེ་རིང་གྲོས་ཚོགས་ནང་བཙན་བྱོལ་བོད་མིའི་དགོན་སྡེ་ཁག་ནང་བོད་རིགས་དང་ཧི་མ་ལ་ཡའི་རི་རྒྱུད་ག་ཚོད་ཆགས་ཀྱི་ཡོད་མེད་དྲི་བ་དྲིས་ལེན་སྐབས་ཆོས་རིག་བཀའ་བློན་མཆོག་གིས་གསལ་བཤད་གནང་སོང་། དེ་རིང་ཉིན་རྒྱབ་སྤྱི་འཐུས་གྲོས་ཚོགས་ནང་། བོད་མིའི་སྒྲིག་འཛུགས་ཆོས་རིག་ལས་ཁུངས་ཀྱི་སྔོན་རྩིས་གཏན་འབེབས་དང་འབྲེལ་ཏེ། བཀའ་བརྒྱུད་སྤྱི་འཐུས་བསྟན་པ་ཡར་འཕེལ་ལགས་ཀྱིས། ཉེ་ལམ་བོད་པའི་ཆོས་ལུགས་ཆོས་བརྒྱུད་ཁག་གི་བླ་ཆེན་རྣམས་དྷ་རམ་ས་ལར་ཚོགས་འདུ་ཚོགས་པའི་སྟེང་། ཚོགས་ཞུགས་པ་ཡོངས་ཀྱི་སེམས་འཚབ་ཆེ་ཤོས་དེ་བཙན་བྱོལ་ནང་གི་བོད་པའི་དགོན་སྡེ་ཁག་ནང་བོད་རིགས་གྲྭ་རྒྱུན་ཇེ་ཉུང་དུ་འགྲོ་བཞིན་ཡོད་པའི་དཀའ་ངལ་དེ་མགྲིན་དབྱངས་གཅིག་གྱུར་གྱིས་གསུངས་ཡོད་སྟབས། བོད་མིའི་སྒྲིག་འཛུགས་ཆོས་རིགས་ལས་ཁུངས་ངོས་ནས་གནད་དོན་དེར་ཞིབ་འཇུག་གནང་ཡོད་མེད་དང་། གནང་ཡོད་ན་དཀའ་ངལ་སེལ་ཐབས་གང་འདྲ་གནང་གི་ཡོད་མེད་དྲི་བ་བཏོན་པའི་ལན་དུ། ཆོས་རིག་བཀའ་བློན་པདྨ་ཆོས་འབྱོར་མཆོག་གིས་གསུངས་དོན། ང་ཚོས་དགོན་སྡེ་ཁག་དང་བཙུན་དགོན་ཁག་ལ་ཞིབ་འཇུག་བྱས་པ་ཡིན། ད་ལྟ་ཁྱོན་བསྡོམས་གྲྭ་བཙུན་ཁྲི་བཞི་ ༤༠༠༠༠ ལྷག་ཅིག་འདུག དེ་དག་ཁྲོད་བོད་རིགས་ག་ཚོད་ཡོད་པ་དང་། ཧི་མ་ལ་ཡའི་རི་རྒྱུད་ག་ཚོད་ཡོད་མེད་ལ་ཞིབ་འཇུག་བྱས་སྐབས་བརྒྱ་ཆ་ ༥༥།༨ ཙམ་ཧི་མ་ལ་ཡའི་རི་རྒྱུད་ཆགས་ཚར་འདུག དེའི་ནང་དགེ་ལུགས་འོ་ཙམ་རེད་མ་གཏོགས་ཆོས་བརྒྱུད་གཞན་པ་མང་ཆེ་བ་ཧི་མ་ལ་ཡའི་རི་རྒྱུད་ཆགས་ཚར་འདུག དེ་ལ་ཚང་མས་ཐུགས་འཚབ་གནང་གི་ཡོད་པ་རེད། ཅེས་གསུངས་སོང་། ཆོས་རིག་བཀའ་བློན་མཆོག་གིས་མུ་མཐུད་དུ་གསུངས་དོན། སྔོན་ཆད་ ༡༩༧༩ ལོའི་རྗེས་ལ་ཏུན་ཞོ་ཕིན་གྱིས་སྲིད་བྱུས་གུ་ཡངས་བཏང་བའི་རྗེས་ལ་བཙན་བྱོལ་བོད་མིའི་དགོན་སྡེ་ཁག་གི་གྲྭ་བཙུན་བརྒྱ་ཆ་ ༨༠ ལྷག་ཅིག་བོད་ནས་ཡོང་ཆགས་ཡོད་པ་རེད། ད་ལྟ་རྒྱ་ནག་གཞུང་གིས་བཀག་འགོག་བྱས་ཏེ། བོད་ནས་ཡོང་མཁན་ཞེ་དྲགས་ཉུང་ཉུང་ཆགས་འདུག བཙན་བྱོལ་སྤྱི་ཚོགས་ནང་ནས་སོ་སོའི་ཕྲུ་གུ་གྲྭ་བ་བྱེད་མཁན་ཧ་ཅང་དཀོན་པོ་རེད། དཀའ་ངལ་དེའི་ཐོག་ལ་ཆོས་བརྒྱུད་ཁག་གི་དབུ་བླ་ཚོས་བཀའ་མོལ་མང་པོ་གནང་བ་རེད་དེ་མཐའ་མར་དགོན་པའི་ནང་གི་དཔེ་ཆ་ཡག་པོ་ཡོད་པའི་དགེ་འདུན་པ་ཚོ་ཕྱི་རྒྱལ་ཁག་ཏུ་མ་ཤོར་བར་དགོན་སྡེ་ཁག་ནང་བཞུགས་ཐུབ་ཐབས་ལ་འབད་བརྩོན་གནང་དགོས་ཀྱི་འདུག་ཟེར་བ་ཙམ་ལས་གཞན་འདི་དང་དེ་བྱེད་དགོས་པ་རེད་ཟེར་བ་ཞིག་གསུང་རྒྱུ་མི་འདུག ༧གོང་ས་མཆོག་གིས་ག་རེ་དགོངས་པ་བཞེས་ཀྱི་འདུག་ཟེར་ན། བོད་ནང་བོད་མི་གནས་ཐུབ་དང་མི་ཐུབ་དོགས་པ་ཆགས་ཡོད་དུས་བོད་པའི་ནང་ཆོས་གནས་ཐུབ་ཀྱི་མེད་པ་གསལ་བོ་ཆགས་ཡོད་པ་རེད། ཧི་མ་ལ་ཡའི་རི་རྒྱུད་དེ་ཕ་མེས་ཡང་མེས་ནས་ནང་པའི་ཆོས་མི་རིགས་ཤིག་ཆགས་ཡོད་པ་རེད། ན་ནིང་༧གོང་ས་མཆོག་གིས་རྒྱ་གར་ལྷོ་ཕྱོགས་འབྲས་སྤུངས་དགོན་དུ། བླ་མ་ཆོས་འཕེལ་བཟོད་པ་མཆོག་ལ། ཁྱེད་རང་ཚོའི་གྲྭ་རྒྱུན་མང་པོ་ཐེབས་ཏེ་ནང་ཆོས་ལ་དོ་སྣང་བྱས་ཏེ་དགོན་པར་བཞུགས་པ་དེ་ཞེ་དྲགས་ཡག་པོ་རེད། ཐུགས་རྗེ་ཆེ་གསུངས་ཡོད་པ་རེད། དེ་དུས་སོ་སོའི་བསམ་བློ་ལའང་འགྱུར་བ་ཕྱིན་སོང་། ང་ཚོའི་བསམ་བློ་དེ་ཐབས་སྐྱོ་བོ་དང་ཚད་དམའ་བོ་རེད་འདུག ང་ཚོའི་བསམ་པར་ང་ཚོ་ཧི་མ་ལ་ཡའི་རི་རྒྱུད་ལ་བཀའ་དྲིན་ཆེན་པོ་ཡིན་པ་ཞིག་དྲན་བསྡད་ཡོད་པ་རེད། ཡིན་ནའང་༧སྐྱབས་མགོན་རིན་པོ་ཆེ་མཆོག་གི་དགོངས་གཞི་དེ་འདྲ་ཞིག་རེད་མི་འདུག ཧ་ཅང་རྒྱ་ཆེན་པོ་རེད། ནང་ཆོས་འདི་གནས་དགོས་ཀྱི་ཡོད་ན། བོད་གངས་ཅན་གྱི་ལྗོངས་སུ་གནས་ཐུབ་དང་མི་ཐུབ་དྲི་བ་རེད། ཧི་མ་ལ་ཡའི་རི་རྒྱུད་དུ་གནས་དགོས་པ་ད་ལྟ་སྐབས་སུ་བབ་པའི་གནད་དོན་གལ་ཆེན་པོ་ཆགས་ཡོད་སྟབས། ཧི་མ་ལ་ཡའི་རི་རྒྱུད་ནས་ནང་ཆོས་ལ་བསླབ་པ་སློབ་གཉེར་གནང་སྟེ་དོ་སྣང་བྱས་པར་ཐུགས་རྗེ་ཆེ་ཨཱ་ཞེས་གསུངས་ཡོད་པ་རེད། ཅེས་གསུངས་སོང་། གསར་འགོད་པ། དོན་འགྲུབ་བཀྲ་ཤིས།
