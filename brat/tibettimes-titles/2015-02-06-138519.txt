ཨོ་བྷ་མས་༧གོང་ས་མཆོག་ལ་ཆེ་བསྟོད་ཞུས།

དེ་རིང་ཨ་རིའི་ཕྱི་ཚེས་ ༥ ཉིན་གྱི་སྔ་དྲོའི་ཆུ་ཚོད་ ༨ སྟེང་། ཨ་རིའི་རྒྱལ་ས་ཝ་ཤཱིན་ཊོན་དུ། ཨ་རིའི་རྒྱལ་ཡོངས་སྨོན་ལམ་གསོལ་སྟོན་སྟེང་། སྐུ་མགྲོན་གཙོ་བོ་སྲིད་འཛིན་ཨོ་བྷ་མ་མཆོག་གིས་གཏམ་བཤད་དབུ་འཛུགས་དང་འབྲེལ་ཏེ། བོད་མིའི་འདུ་ཕྱིའི་སྐྱབས་གནས་༧གོང་ས་མཆོག་ལ་དམིགས་བསལ་ཆེ་བསྟོད་གསུང་བཤད་གནང་འདུག སྐབས་དེར་ཨོ་བྷ་མས་གསུངས་དོན། “ངས་འདིར་དམིགས་བསལ་གྱིས་འཚམས་འདྲི་ཞུ་ཡུལ་ཞིག་ནི། ངའི་གྲོགས་པོ་བཟང་པོ་༧གོང་ས་ཏཱ་ལའི་བླ་མ་མཆོག་ཡིན། ཁོང་ནི་བྱམས་བརྩེ་ཉམས་ལེན་གནང་མཁན་གྱི་རླབས་ཆེན་དཔེ་མཚོན་ཞིག་ཡིན་ལ། ཁོང་གིས་ང་ཚོར་འགྲོ་བ་མིའི་བྱམས་བརྩེའི་རིན་ཐང་གི་ཆེད་དུ་སྐད་ཆ་ཤོད་རྒྱུའི་སེམས་ཤུགས་སྤར་གནང་གི་ཡོད། ངས་དེ་སྔོན་༧གོང་ས་མཆོག་ཕོ་བྲང་དཀར་པོར་གདན་འདྲེན་ཞུ་རྒྱུའི་གོ་སྐབས་ཐེངས་མང་ཐོབ་བྱུང་ལ། དེ་རིང་འདིར་སླར་ཡང་༧གོང་ས་མཆོག་མཉམ་ཞུགས་གནང་རྒྱུ་བྱུང་བར་དགའ་བོ་བྱུང་།” ཞེས་སོགས་བརྗོད་འདུག ཐེངས་འདིའི་སྨོན་ལམ་གསོལ་སྟོན་དེ་ནི། ཨ་རིའི་གྲོས་ཚོགས་གོང་མིའི་འཐུས་མི་ཁག་ཅིག་གིས་གོ་སྒྲིག་ཞུ་བཞིན་པའི་ཡེ་ཤུའི་ཆོས་ཕྱོགས་ཀྱི་བྱེད་སྒོ་ཞིག་ཡིན་པ་དང་། གསོལ་སྟོན་དེར་ཆོས་ལུགས་ཁག་གི་མི་སྣ་གདན་ཞུ་བྱེད་སྲོལ་ཡོད་ཅིང་། ལོ་ལྟར་གསོལ་སྟོན་གྱི་སྐུ་མགྲོན་གཙོ་བོར་ཨ་རིའི་སྲིད་འཛིན་དང་། གལ་སྲིད་འཛིན་ཕེབས་མ་ཐུབ་ཚེས་དམིགས་བསལ་སྐུ་མགྲོན་ཆོས་དཔོན་ཚོས་གཏམ་བཤད་གནང་བའི་ལུགས་སྲོལ་ཡོད་འདུག ཨ་རིའི་རྒྱལ་ཡོངས་སྨོན་ལམ་གསོལ་སྟོན་འདི་ནི་ ༡༩༡༥ ལོར་མགོ་འཛུགས་གནང་སྟེ་ད་རེས་ཀྱི་མཛད་སྒོ་དེ་ཐེངས་ ༡༠༠ པ་དེ་ཡིན་པ་དང་། ཐེངས་འདིར་སྲིད་འཛིན་ཨོ་བྷ་མ་དང་ཁོང་གི་སྐྱེས་དམན་མེ་ཤཱལ་ཨོ་བྷ་མ་གཙོས་རྒྱལ་ཁབ་བརྒྱ་ལྷག་གི་ཆོས་ལུགས་དང་ཆབ་སྲིད་མི་སྣ་བཅས་སྐུ་མགྲོན་ ༣༥༠༠ ལྷག་མཉམ་ཞུགས་གནང་ཡོད་འདུག
