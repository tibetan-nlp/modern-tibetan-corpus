སྐྱིད་གྲོང་འགག་སྒོ་བསྐྱར་དུ་ཕྱེས།

  ཁ་སང་ནས་བཟུང་བོད་བལ་ས་མཚམས་ཀྱི་སྐྱིད་གྲོང་འགག་སྒོ་བསྐྱར་དུ་སྒོ་ཕྱེས་ཏེ་ཚོང་ཟོག་སྐྱེལ་འདྲེན་བྱེད་བཞིན་ཡོད་འདུག   བོད་ནང་གི་གསར་འགྱུར་བརྒྱུད་ལམ་ཁག་ཏུ་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། འདི་ལོའི་ཕྱི་ཟླ་ ༤ པའི་ནང་བལ་ཡུལ་དུ་ས་ཡོམ་ཤུགས་ཆེན་བྱུང་བར་བརྟེན། བོད་དང་བལ་པོའི་ས་མཚམས་ཁུལ་དུའང་གནོད་སྐྱོན་ཆེན་པོ་ཐེབས་ཡོད་སྟབས། རྒྱ་གཞུང་གིས་བོད་ཀྱི་སྐྱིད་གྲོང་འགག་སྒོ་ཟླ་ ༥ ལྷག་རིང་སྒོ་བརྒྱབ་ཡོད་ཀྱང་། ཁ་སང་ཕྱི་ཚེས་ ༡༣ ཉིན་ནས་བཟུང་བསྐྱར་དུ་སྒོ་ཕྱེས་ཡོད་ལུགས་དང་། ཚོང་ཟོག་སྐྱེལ་འདྲེན་ཁོ་ན་མ་གཏོགས་གནས་སྐབས་ཡུལ་སྐོར་བ་འགྲོ་རུ་འཇུག་རྒྱུ་མིན་སྐོར་བསྒྲགས་འདུག   སྐྱིད་གྲོང་འགག་སྒོའི་འབྲེལ་ཡོད་འགན་འཁུར་བས། སྐྱིད་གྲོང་འགག་སྒོ་བརྒྱུད་ཚོང་ཟོག་གཙོ་བོ་གྱོན་གོས་དང་། རྒྱུན་སྤྱོད་དངོས་རྫས། ལུག་སོགས་ཕྱིར་འཚོང་བྱ་རྒྱུ་དང་། འབྲས་དང་གྲོ་ཞིབ། ཟངས་ཆས་སོགས་ནང་འདྲེན་བྱ་རྒྱུ་ཡིན་ལ། འགག་སྒོ་དེ་བརྒྱུད་ཚོང་ཟོག་རྣམས་བལ་ཡུལ་གྱི་རྒྱལ་ས་ཀ་ཐ་མན་ཀྲུའུ་དང་པཱོ་ཁེ་ལ་ཟེར་བའི་བར་སྐྱེལ་འདྲེན་བྱེད་ཆོག་པ་དང་། པཱོ་ཁེ་ལ་ནས་རྒྱ་གར་ནང་ཚོང་ཟོག་སྐྱེལ་རྒྱུ་ཡང་ཧ་ཅང་སྟབས་བདེ་ཆགས་ཡོད་སྐོར་དང་། འགག་སྒོ་བསྐྱར་དུ་ཕྱེས་རྗེས་སྔོན་ལ་བལ་པོར་རོགས་རམ་དངོས་ཟོག་ཕྱིར་གཏོང་བྱ་རྒྱུ་ཡིན་པས། གནས་སྐབས་རིང་ཡུལ་སྐོར་བ་ཚོ་སྐྱིད་གྲོང་འགག་སྒོ་བརྒྱུད་ནས་བལ་པོར་སྐྱོད་ཐུབ་ཀྱི་མེད་སྐོར་སོགས་བརྗོད་འདུག   སྐྱིད་གྲོང་གྲོང་རྡལ་ནས་བལ་བོའི་རྒྱལ་ས་བར་སྤྱི་ལེ ༡༣༡ ཙམ་ཡོད་པ་དང་། ས་གནས་དེ་བོད་བལ་ས་མཚམས་ཀྱི་ལོ་རྒྱུས་ཡུན་རིང་ལྡན་པའི་རང་དབང་གི་ཚོང་ལམ་ཞིག་ཆགས་ཡོད་ཀྱང་། རྒྱ་གཞུང་གིས་བོད་ས་ཡོངས་རྫོགས་བཙན་བཟུང་བྱས་རྗེས། ལོ་མང་རིང་ཚོང་འགྲུལ་སྒོ་བརྒྱབ་པ་སོགས་འགོག་རྐྱེན་མང་པོ་བཟོས་ཡོད་འདུག ཕྱི་ལོ་ ༡༩༦༢ ལོར་བོད་ནས་བལ་དུ་སྐྱོད་པའི་སྐམ་སའི་ཚོང་ལམ་ཆེ་གྲས་ཤིག་ཏུ་བརྩིས་ཏེ་དོ་དམ་འགག་འགོག་དངོས་སུ་བཙུགས་ཡོད་པ་མ་ཟད། ད་ལྟའི་ཆར་རྒྱལ་ཁབ་ཀྱི་རིམ་པ་དང་པོའི་སྐམ་སའི་ཚོང་ལམ་འགག་སྒོ་ཞེས་པའི་ནང་བཅུག་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
