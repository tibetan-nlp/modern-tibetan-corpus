དོན་འགྲུབ་རྒྱལ་རྗེས་དྲན་བགྲོ་གླེང་བྱས།

  ཡུ་རོབ་ནང་གི་བོད་མི་ཚོས་ཐེངས་དང་པོར་དཔལ་དོན་འགྲུབ་རྒྱལ་ལ་རྗེས་དྲན་གྱི་བགྲོ་གླེང་ཞིག་ཚོགས་ཏེ་ཡུ་རོབ་རྒྱལ་ཁབ་འདྲ་མིན་དང་ཨ་རི་སོགས་ནས་བོད་པའི་རྩོམ་པ་པོ་ཚོས་ཆ་ཤས་བླངས་ཡོད་འདུག   ཕྱི་ཚེས་ ༢༨ ཉིན། ཡུ་རོབ་བོད་རིགས་གཞོན་ནུའི་མཐུན་ཚོགས་ཀྱི་གོ་སྒྲིག་འོག སུད་སི་གྲོང་ཁྱེར་ཛུ་རིག་གི་Volkshaus ཚོགས་ཁང་ནང་། དཔལ་དོན་གྲུབ་རྒྱལ་ལམ་རང་གྲོལ་ལགས་སྐུ་གཤེགས་ནས་མི་ལོ་སུམ་བཅུ་འཁོར་གྱིན་པའི་དུས་དང་བསྟུན་ཏེ། ༼ ལང་ཚོའི་རྦབ་ཆུ་དང་བོད་ཀྱི་དེང་རབས་རྩོམ་རིག ༽ ཅེས་པའི་བརྗོད་གཞིའི་ཐོག བོད་ཀྱི་དེང་རབས་རྩོམ་རིག་དང་འབྲེལ་བའི་བགྲོ་གླེང་དང་གླེང་མོལ་གནང་སོང་།   དུས་དྲན་འདུ་འཛོམས་ཐོག་དབྱིན་ཡུལ་གྱི་ཨོ་ཁི་སི་ཧྥོར་ཌི་(Oxford)གཙུག་ལག་སློབ་གྲྭ་ཆེན་མོའི་བོད་ཀྱི་རྩོམ་རིག་དགེ་རྒན་དང་། བོད་དང་ཧི་མ་ལའི་རིག་གནས་ཉམས་ཞིབ་པ་བླ་མ་སྐྱབས་ལགས་དང་། བོད་ཀྱི་རྩོམ་པ་པོའི་མཐུན་ཚོགས་ཀྱི་ཚོགས་གཙོ་དང་སྤྱི་ཟུར་ཆབ་བྲག་ལྷ་མོ་སྐྱབས་ལགས་ཀྱིས་གཙོས་ཡུ་རོབ་རྒྱལ་ཁབ་འདྲ་མིན་ནས་ཕེབས་པའི་གཞོན་སྐྱེས་རྩོམ་རིགས་ལ་དོ་སྣང་ཅན་ ༤༠ མཉམ་ཞུགས་གནང་སོང་།   རྗེས་དྲན་བགྲོ་གླེང་འདིའི་ཐོག་ཨ་རིར་བཞུགས་པའི་བོད་ཀྱི་རྩོམ་པ་པོ་དང་མདོ་སྨད་སྤྱི་ཟུར་ཧོར་གཙང་འཇིགས་མེད་ལགས་ཀྱིས་དྲ་ཐོག་བརྒྱུད་དོན་གྲུབ་རྒྱལ་ལ་ངོས་འཛིན་གནང་སྟངས་དང་། བོད་ཀྱི་དེང་རབས་རྩོམ་རིག་སྐོར་དང་འབྲེལ་བའི་གཏམ་བཤད་དང་། སྐུ་ཞབས་བླ་མ་སྐྱབས་ལགས་ཀྱིས་ལང་ཚོའི་རྦབ་ཆུ་གཙོས་དོན་གྲུབ་རྒྱལ་གྱི་རྩོམ་ཡིག་ཁག་ཅིག་དབྱིན་སྐད་ཐོག་ངོ་སྤྲོད་འགྲེལ་བཤད། རྒན་ཆབ་བྲག་ལྷ་མོ་སྐྱབས་ལགས་ཀྱིས་དོན་གྲུབ་རྒྱལ་དང་དོན་གྲུབ་རྒྱལ་གྱི་རྩོམ་ཡིག་ཁག་ལ་གདེང་འཇོག་ཇི་ལྟར་གནང་མིན་སོགས་ལ་གཏམ་བཤད་དང་གླེང་མོལ་གནང་ཡོད་པ་དང་། དེ་བཞིན། མཉམ་ཞུགས་པ་ཚོས་དྲི་བ་དང་བསམ་ཚུལ་བརྗེ་རེས་སོགས་གནང་སྟེ། ཡུ་རོབ་ཁུལ་དུ་བོད་ཀྱི་དེང་རབས་རྩོམ་རིག་ཁག་ལ་དོ་སྣང་དང་། གོ་རྟོགས་སླེབས་དགོས་པའི་བསམ་ཚུལ་སྣ་ཚོགས་བརྗོད་སོང་།   རྗེས་དྲན་བགྲོ་གླེང་གོ་སྒྲིག་པ་མིག་དམར་སྒྲོལ་མ་ལགས་ཀྱིས་བརྗོད་དོན། དེ་རིང་ང་ཚོས་རྩོམ་རིག་སྐོར་ལ་དཔལ་དོན་གྲུབ་རྒྱལ་ལགས་སྐུ་གཤེགས་པའི་རྗེས་དྲན་དང་སྟབས་བསྟུན་བགྲོ་གླེང་ཞིག་གནང་དགོས་དོན་ནི། དོན་གྲུབ་རྒྱལ་ལགས་ནི་བོད་ཀྱི་རྩོམ་པ་པོ་སྐད་གྲགས་ཅན་ཞིག་ཡིན་པ་མ་ཟད། ཁོང་གི་བསམ་བློའི་སྙིང་སྟོབས་སོགས་ལ་རྗེས་དྲན་བྱ་དགོས་པ་དང་། ལྷག་དོན་དུ་ཁོང་གི་ཚེ་སྲོག་ཀྱང་བོད་མི་རིགས་གཉིད་སད་ཆེད་བློས་གཏོང་གནང་མཁན་ཞིག་ཡིན་པའི་ཆ་ནས་ང་ཚོས་ཁོང་རྗེས་དྲན་ཞུ་རྒྱུ་གལ་ཆེན་ཆགས་ཡོད་པ་རེད། གཉིས་ནས་ཁོང་དང་ཁོང་ལྟ་བུའི་བོད་ཀྱི་ཉེ་རབས་རྩོམ་པ་པོ་རྣམས་ཕྱི་རྒྱལ་ཁག་ནང་ཡོད་པའི་བོད་པ་ང་ལྟ་བུ་མང་པོ་ཞིག་གིས་ངོ་ཤེས་ཀྱིན་མེད་ལ། ཁོང་ཅག་གི་བསམ་བློ་དང་ལྟ་བ་གང་ཡིན་དང་། ཁོང་ཚོས་བོད་མི་རིགས་ཀྱི་ཆེད་དུ་བསམ་བློ་ག་འདྲ་གཏོང་གིན་ཡོད་མེད། དེ་བཞིན། བོད་ཀྱི་རྩོམ་རིག་ཟེར་ན་ག་འདྲ་ཞིག་ཡིན་མིན་སོགས་ཧ་གོ་གི་མེད་དུས། རྩོམ་པ་པོ་ཁོང་དང་འབྲེལ་བའི་བོད་ཀྱི་རྩོམ་རིག་སྐོར་དང་། ཡང་ན། དེང་རབས་ཀྱི་བོད་ཀྱི་རྩོམ་པ་པོ་དག་གི་བསམ་བློ་དང་རྩོམ་ཡིག་གི་སྒྱུ་རྩལ་སོགས་གླེང་སློང་གནང་རྒྱུ་འདི་གལ་ཆེན་ཡིན་ལ། བོད་ནང་གི་རྩོམ་པ་པོ་དང་ཕྱི་ཕྱོགས་སུ་ཡོད་པའི་གཞོན་སྐྱེས་རྩོམ་པ་པོ་ཁག་གི་བར་འབྲེལ་ལམ་དང་བསམ་བློ་བརྗེ་རེས་ལྟ་བུའི་ནུས་པ་ཡོད་པ་ཞིག་ལ་མཐོང་ནས་དེ་རིང་འདི་ལྟ་བུའི་བགྲོ་གླེང་ཞིག་གོ་སྒྲིག་ཞུས་པ་ཡིན། ཞེས་ངོ་སྤྲོད་གནང་སོང་།   ཁོ་མོས་མུ་མཐུད་དུ་བརྗོད་དོན། བཙན་བྱོལ་ནང་རྩོམ་རིག་ལ་དོ་སྣང་སྤྲོད་མཁན་ཧ་ཅང་ཉུང་ཉུང་ཡིན་ལ། ཕྱིའི་སྐད་ཡིག་སྦྱངས་ནས་མི་ཚེ་སྐྱེལ་གྱིན་ཡོད་ཀྱང་ཕྱིའི་སྐད་ཡིག་ནང་རྩོམ་ཡིག་ཡག་པོ་འབྲི་མཁན་དེ་ལས་ཀྱང་ཉུང་། རེ་འགའ་ཡོད་པ་ནི་འཇམ་དབྱངས་ནོར་བུ་ལྟ་བུ་ཡིན་ཡིན། འཇམ་དབྱངས་ནོར་བུ་ལགས་ནི་ཧ་ཅང་གནས་ཚད་མཐོ་བའི་རྩོམ་པ་པོ་ཞིག་དང་། བོད་ཀྱི་རང་བཙན་དང་རྒྱལ་རབས། འཐབ་རྩོད་སོགས་ལ་ཤུགས་སྐྱོན་གྱིན་པའི་རྩོམ་པ་པོ་ཡག་པོ་ཞིག་རེད། ཁོང་གིས་དབྱིན་ཡིག་ནང་བྲིས་པ་དག་གིས་གཞོན་སྐྱེས་རྣམས་ལ་བོད་དོན་དང་བོད་དོན་འཐབ་རྩོད་སོགས་ལ་གོ་རྟོགས་དང་ལམ་ཁ་སྟོན་གྱིན་ཡོད་རེད། འོན་ཀྱང་། ཁོང་ལྟ་བུ་ཉུང་ཉུང་རེད་འདུག་པས་མ་འོངས་པར་བཙན་བྱོལ་ནང་རྩོམ་རིག་ཐོག་དོ་སྣང་དགོས་རྒྱུ་གལ་ཆེན་བསམ་གྱིན་འདུག ཅེས་བརྗོད་བྱུང་།   མཇུག་ཏུ་མོས་རེ་སྐུལ་དང་བཅས་བརྗོད་དོན། མ་འོངས་པར་ཡུ་རོབ་ནང་བོད་ཀྱི་ན་གཞོན་རྩོམ་པ་པོ་རྣམས་དང་། ལྷག་དོན་དུ་བོད་ནང་སློབ་སྦྱོང་དང་རྩོམ་ཡིག་སྦྱོང་མྱོང་མཁན་དང་། ནུབ་ཕྱོགས་རྒྱལ་ཁབ་ཁག་ནང་སྐྱེས་པའི་བོད་ཀྱི་གཞོན་སྐྱེས་ཚོའི་བར་སྙན་ངག་དང་། སྒྱུ་རྩལ། ཆབ་སྲིད། རིག་གནས། བསམ་བློ་བརྗེ་རེས་བྱ་དགོས་རྒྱུ་གལ་ཆེན་དུ་མཐོང་བས། གལ་སྲིད་སྟབས་བདེ་བྱུང་ན་མུ་མཐུད་ཟླ་བ་གསུམ་རེའི་ནང་དང་། ཡང་ན་ལོ་གཅིག་རེའི་ནང་ལ་རྩོམ་རིག་ལ་དོ་སྣང་ཡོད་མཁན་ཚོ་མཉམ་འཛོམས་དང་བགྲོ་གླེང་ཐེངས་རེར་བྱ་ཐུབ་རྒྱུའི་རེ་བ་རྒྱག་གིན་ཡོད། ཅེས་སོགས་བརྗོད་སོང་།   གསར་འགོད་པ། ཙ་ཡུས་ཕུན་ཚོགས།
