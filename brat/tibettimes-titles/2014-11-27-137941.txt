ཞི་རྒོལ་བ་ ༡༥༠ ཙམ་འཛིན་བཟུང་།

ཧོང་ཀོང་གཞུང་གིས་ཝང་ཇོ་ཁྲོམ་གཞུང་གི་ཞི་རྒོལ་བ་ཚོ་ཕྱིར་འབུད་བཏང་སྟེ་གཙང་སེལ་བྱས་ཡོད་པ་མ་ཟད། ལས་འགུལ་སྣེ་ཁྲིད་པ་སོགས་མི་ ༡༥༠ ཙམ་འཛིན་བཟུང་བཀག་ཉར་བྱས་ཡོད་འདུག ཧོང་ཀོང་ནང་མི་མང་གིས་ཞི་རྒོལ་བྱས་ཏེ་ཟླ་གཉིས་ཏག་ཏག་ཕྱིན་ཡོད་པ་དང་། ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༥ ཉིན་གྱི་དགོང་མོ་ནས་བཟུང་ཧོང་ཀོང་ཆེས་མཐོའི་ཁྲིམས་ཁང་གི་བཀའ་བཀོད་འོག ཉེན་རྟོག་པ་བརྒྱ་ཕྲག་བརྒལ་བས་ཞི་རྒོལ་གྱི་ལྟེ་གནས་ཝང་ཇོ་ཁྲོམ་གཞུང་དུ་གཙང་སེལ་ལས་འགུལ་སྤེལ་ཏེ། ཞི་རྒོལ་བའི་སྡོད་གུར་གཏོར་བཤིག་བཏང་ནས་བཙན་གྱིས་ཕྱིར་འབུད་བཏང་ཡོད་པ་དང་། སྐབས་དེར་ཉེན་རྟོག་པ་དང་ལས་འགུལ་བའི་བར་འཁྲུག་འཛིང་ཤོར་ཡོད་པ་མ་ཟད། ཁྲིམས་འགལ་ འདུ་འཛོམས་དང་ཉེན་རྟོག་པ་ ༧ ཙམ་ལ་གཅར་རྡུང་བཏང་བ། གཙང་སེལ་གྱི་བྱ་བར་འགོག་རྐྱེན་བཟོས་པ་སོགས་ཀྱི་ཉེས་མིང་འོག ཞི་རྒོལ་རྒྱབ་སྐྱོར་བ་ཁྲིམས་བཟོ་ལྷན་ཁང་གི་འཐུས་མི་ཉིང་ཀོ་ཤོང་དང་སློབ་ཕྲུག་མཐུན་ཚོགས་ཀྱི་མགོ་ཁྲིད་གཞོན་པ་ཚིན་ཨོ་ཧེ། སློབ་མའི་ལས་འགུལ་སྣེ་ཁྲིད་པ་ཧང་ཀྲི་སོགས་ཀྱི་གཙོས་ཞི་རྒོལ་བ་ ༡༥༠ ཙམ་འཛིན་བཟུང་བཀག་ཉར་བྱས་ཡོད་པ་མ་ཟད། གཙང་སེལ་སྐབས་གསར་འགོད་པ་ཞིག་འཛིན་བཟུང་བྱས་ཡོད་སྟབས། ཧོང་ཀོང་གསར་འགོད་མཐུན་ཚོགས་སོགས་རྒྱལ་ཕྱི་ནང་ནས་གསར་འགོད་རང་དབང་ལ་གཏོར་སྐྱོན་བཏང་བའི་སྐྱོན་བརྗོད་ཤུགས་ཆེན་བྱེད་བཞིན་འདུག ཁ་སང་ཕྱི་དྲོར་ཧོང་ཀོང་གསར་འགོད་མཐུན་ཚོགས་ཀྱིས་བསྒྲགས་གཏམ་ཞིག་སྤེལ་བའི་ནང་། ཧོང་ཀོང་དཀྱིལ་དབུས་བཙན་བཟུང་གི་ལས་འགུལ་སྤེལ་བ་ནས་ཉེན་རྟོག་པས་ལས་འགུལ་ལ་འགོག་རྒོལ་རིམ་པར་བྱས་ཏེ་ངོ་རྒོལ་བ་ཕྱིར་འབུད་གཏོང་ཐབས་བྱས་པས་མ་ཚད། འཛེམས་མེད་ཀྱིས་གསར་འགྱུར་ལས་རིགས་གཉེར་མཁན་ཚོར་དྲག་ཤུགས་བཀོལ་ཏེ་ཉུང་མཐར་གསར་འགོད་པ་ ༢༥ ལ་རྨས་སྐྱོན་བཟོས་ཡོད་པ་མ་ཟད། ཐ་ན་རྒྱུ་མཚན་ཅི་ཡང་མེད་པར་གསར་འགོད་པ་འཛིན་བཟུང་བྱས་པ་དེས་མི་རྣམས་ལ་འཁོན་སེམས་ཆེན་པོ་བསླངས་ཡོད་སྟབས། ཉེན་རྟོག་ཕྱོགས་ནས་ཉེན་རྟོག་པའི་བཙན་སྐུལ་བྱ་སྤྱོད་ལ་སྟངས་འཛིན་དང་། གསར་འགོད་པའི་བྱ་བར་འགོག་རྐྱེན་བཟོ་བཞིན་པ་འཕྲལ་དུ་མཚམས་འཇོག་དགོས་པའི་ནན་སྐུལ་བྱས་ཡོད་པ་མ་ཟད། ཧོང་ཀོང་ཉེན་རྟོག་ལས་དོན་ལྷན་ཁང་གི་མགོ་ཁྲིད་ཙིན་ཝེ་ཤོང་གིས་གནད་དོན་ལ་དཔྱད་ཞིབ་ལེགས་བཅོས་བྱས་ཏེ། རྗེས་ཕྱོགས་གནད་དོན་དེ་རིགས་མི་འབྱུང་པར་ཁག་ཐེག་དགོས་པའི་སྐུལ་མ་གནང་འདུག གཞན་ཡང་། ཁ་སང་ཕྱི་ཚེས་ ༢༥ ཉིན་དབྱིན་ཇི་གྲོས་ཚོགས་ཀྱི་རྒྱ་ནག་ལས་དོན་ཚོགས་ཆུང་ཞིག་ཧྲང་ཧེ་རུ་རྒྱལ་དོན་གཟིགས་སྐོར་དུ་ཕེབས་འཆར་ཡོད་ཀྱང་། ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༢ ཉིན་ལས་དོན་ཚོགས་ཆུང་དེའི་ཚོགས་གཙོ་(Richard Graham)ལགས་ཀྱིས། དབྱིན་གཞུང་གིས་ཕྱི་ལོ་ ༡༩༨༤ ལོའི་རྒྱ་དབྱིན་མཉམ་འབྲེལ་གསལ་བསྒྲགས་ཀྱི་རྩ་དོན་ལ་སྲུང་སྐྱོབ་བྱེད་པའི་འོས་འགན་ཡོད་པ་མ་ཟད། ཧོང་ཀོང་ཁྲིམས་སྐྱོང་འོག་སྨྲ་བརྗོད་རང་དབང་དང་ཁྲིམས་འཛིན་ལམ་ལུགས་རང་སྐྱོང་ལ་ཁག་ཐེག་དགོས་སྐོར་བརྗོད་ཡོད་པར་བརྟེན། དབྱིན་ཇིར་གནས་སྡོད་རྒྱ་ནག་གཞུང་ཚབ་ཁང་གིས་ཁོང་རྒྱ་ནག་ནང་གཟིགས་སྐོར་དུ་ཕེབས་པར་ཐོངས་མཆན་སྤྲད་མེད་སྟབས། ཁ་སང་ཕྱི་ཚེས་ ༢༥ ཉིན་གྲོས་ཚོགས་འཐུས་མི་ཚོགས་པ་དེས། རྒྱ་ནག་གཞུང་ཚབ་ཁང་གིས་རེས་ཆར་ཌི་གྷ་ར་ཧམ་(Richard Graham)ལགས་ལ་ཐོངས་མཆན་མ་སྤྲད་ན། འཐུས་མི་ཚོགས་པ་རྒྱ་ནག་ནང་གཟིགས་སྐོར་དུ་ཕེབས་རྒྱུ་ཕྱིར་འཐེན་བྱ་རྩིས་ཡོད་སྐོར་བརྗོད་ཡོད་ཀྱང་། རྒྱ་ནག་གཞུང་ཚབ་ཀྱིས་མོས་མཐུན་མ་བྱས་པར་ཐེངས་འདིའི་གཟིགས་སྐོར་ཕྱིར་འཐེན་བྱས་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
