རྒྱ་ནག་གིས་ལོ་ངོ་ ༢༥
རིང་ཨ་རིར་མགོ་སྐོར་བཏང་སོང་ཟེར། 

རྒྱ་ནག་གཞུང་གིས་ཨ་རི་དང་བྱང་ཀོ་རེ་ཡ་བར་རྐང་པ་གཉིས་བཙུགས་ཏེ་སོ་སོའི་ཁེ་བཟང་གི་ཆེད་དུ་ཨ་རི་ལ་འདས་པའི་ལོ་ ༢༥ རིང་མགོ་སྐོར་བཏང་ཡོད་སྐོར་བརྗོད་འདུག རྒྱ་ནག་ལྷོ་རྒྱུད་ཞོགས་པའི་ཆ་རྐྱེན་གྱིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན།  ཁ་སང་ཨ་རིའི་གྲོས་ཚོགས་འཐུས་མི་དང་བདེ་སྲུང་ལྷན་ཁང་གི་ཚོགས་མི་སྐུ་ཞབས་ཊོམ་ཁོ་ཊོན་(Tom Cotton)ལགས་ཀྱིས།  རྒྱ་ནག་གིས་འདས་པའི་ལོ་ངོ་ ༢༥ རིང་བྱང་ཀོ་རི་ཡའི་རྡུལ་ཕྲན་མཚོན་ཆ་བཀག་འགོག་ལ་ཁ་གཡར་ཏེ་ཨ་རིར་མགོ་སྐོར་བཏང་ཡོད་པ་མ་ཟད། རྒྱ་ནག་གིས་བྱང་ཀོ་རི་ཡ་བེད་སྤྱོད་བཏང་སྟེ་སོ་སོའི་དཔལ་འབྱོར་རྒྱ་སྐྱེད་དང་ཨ་རིའི་དཔལ་འབྱོར་དང་གནས་བབ་གཞན་མང་པོ་ཞིག་གི་ཐོག་ཁ་གཏད་གཅོག་ཐབས་བྱེད་ཀྱི་ཡོད་སྐོར་སྐྱོན་བརྗོད་བྱས་འདུག ཨ་རིའི་སྲིད་འཛིན་ཊི་རུམ་ཕིས་ཡང་ནས་བསྐྱར་དུ་བྱང་ཀོ་རི་ཡའི་རྡུལ་ཕྲན་མཚོན་ཆ་ཚོད་ལྟ་མཚམས་འཇོག་དགོས་སྐོར་ལ་རྒྱ་ནག་གཙོས་པའི་རྒྱལ་སྤྱིིའི་སྲིད་གཞུང་ཁག་ལ་འབོད་སྐུལ་ཞུས་ཡོད་པ་མ་ཟད།  རྒྱ་ནག་གིས་བྱང་ཀོ་རི་ཡ་དང་ལྷན་དུ་ཚོང་འབྲེལ་མཚམས་འཇོག་དགོས་པའི་རེ་འདུན་ནན་པོ་བཏོན་ཡོད་པ་དང་།  ཟླ་སྔོན་མའི་ཕྱི་ཚེས་ ༢༢ ཉིན་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ནང་རྒྱ་ནག་གིས་བྱང་ཀོ་རི་ཡར་ཉེས་ཆད་གཅོད་པར་རྒྱབ་སྐྱོར་བྱས་ཡོད་སྟབས།  སྲིད་འཛིན་ཊི་རུམ་ཕིས་རྒྱ་ནག་ལ་བསྔགས་བསྟོད་དང་སྐད་ཆ་སྙན་པོ་བཤད་ཡོད་ཀྱང་།  ཨ་རིའི་གྲོས་ཚོགས་འཐུས་མི་སྐུ་ཞབས་ཁོ་ཊོན་ལགས་ཀྱིས། བྱང་ཀོ་རི་ཡ་རྡུན་ཕྲན་མཚོན་ཆ་ཅན་གྱི་རྒྱལ་ཁབ་ཅིག་ཏུ་གྱུར་ཚེ་རྒྱ་ནག་ལ་ཁེ་བཟང་ཡོད་སྟབས།  རྒྱ་ནག་གིས་ཨ་རི་དང་བྱང་ཀོ་རི་ཡའི་བར་རྐང་པ་གཉིས་བཙུགས་ཏེ། ཕྱོགས་གཅིག་ནས་སྤྱི་ཚོགས་ནང་ནུབ་ཕྱོགས་སྲིད་གཞུང་ཁག་ལ་སྐད་ཆ་གཅིག་བཤད་ཀྱི་ཡོད་པ་དང། ཕྱོགས་གཞན་ཞིག་ནས་རྒྱ་ནག་གིས་ད་བར་བྱང་ཀོ་རི་ཡའི་རྡུན་ཕྲན་མཚོན་ཆ་ཚོད་ལྟ་འགོག་ཐབས་བྱས་མེད་པ་མ་ཟད།  ད་ལྟ་བྱང་ཀོ་རི་ཡའི་རྡུན་ཕྲན་མཚོན་ཆ་མེད་པ་བཟོ་ཐབས་ཡང་གང་ཡང་བྱེད་ཀྱི་མེད།  ཅེས་བརྗོད་འདུག ཉེ་ཆར་བྱང་ཀོ་རི་ཡས་རྡུན་ཕྲན་མཚོན་ཆ་ཚོད་ལྟ་བྱས་ཏེ་ལེགས་འགྲུབ་ཡོད་པ་མ་ཟད། དེར་ཨ་རིའི་རྒྱལ་ཡོངས་ནང་སྙོབ་ཐུབ་པའི་ནུས་པ་ཡོད་སྐོར་བརྗོད་ཡོད་སྟབས། ཟླ་འདིའི་ཕྱི་ཚེས་ ༥ ཉིན་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་ཆབ་སྲིད་འབྲེལ་མཐུད་པའི་འགན་འཛིན་ཇེ་ཧྥི་རའེ་ཧྥིལ་ཊི་མན་གཙོས་པའི་སྐུ་ཚབ་མི་སྣ་ ༡༥ ཛ་དྲག་ཐོག་བྱང་ཀོ་རི་ཡར་ཆེད་ཕེབས་གནང་ཡོད་པ་རེད།་ བྱང་ཀོ་རི་ཡས་ཨ་རི་གཙོས་པའི་རྒྱལ་སྤྱིའི་སྲིད་གཞུང་ཁག་གི་སྐྱོན་བརྗོད་བྱས་མིན་ལ་མ་ལྟོས་པར་ད་ལྟའི་བར་རྡུལ་ཕྲན་མཚོན་ཆ་ཚོད་ལྟ་ཐེངས་ ༦ བྱས་ཡོད་པ་དང་། ཟླ་འདིའི་ཕྱི་ཚེས་ ༦ ཉིན་བྱང་ཀོ་རི་ཡ་དང་ས་མཚམས་སུ་འཁེལ་བའི་རྒྱ་ནག་གི་མངའ་སྡེའི་གསར་ཁང་(Jilin Daily)ཞེས་པའི་སྟེང་། ས་གནས་སྲིད་གཞུང་གིས་ཡུལ་མི་ཚོར་རྡུལ་ཕྲན་མཚོན་ཆའི་དམག་འཁྲུག་བྱུང་སྐབས་འགེབས་སྲུང་ཇི་ལྟར་བྱེད་དགོས་མིན་གྱི་སྤྱོང་བརྡར་སྤྲད་ཡོད་སྐོར་བཀོད་ཡོད་པ་དང་། རྒྱ་ནག་གི་ཕྱི་སྲིད་བློན་ཆེན་ཝང་ཡས་ཟེར་བ་དེས་བྱང་ཀོ་རེ་ཡའི་འཇིགས་ཉེན་སླར་ཡང་འཕག་འཚག་བརྒྱབས་བྱུང་།  ཞེས་བརྗོད་ཡོད་པ་རེད། གསར་འགོད་པ།  སྐལ་བཟང་སྦྱིན་པ།
