མི་ ༡༧༡ གནས་དབྱུང་གཏོང་དགོས་ཟེར།

ཐེན་ཅིན་གྲོང་ཁྱེར་གྱི་འབར་གས་དོན་རྐྱེན་ལ་ཞིབ་བཤེར་བྱས་པ་བརྒྱུད་འབྲེལ་ཡོད་མི་གྲངས་ ༡༧༡ གནས་དབྱུང་གཏོང་དགོས་པའི་བསམ་ཚུལ་བཏོན་འདུག   འདས་ལོའི་ཕྱི་ཟླ་ ༨ ཚེས་ ༡༢ ཉིན་རྒྱ་ནག་ཐེན་ཅིན་གྲོང་ཁྱེར་དུ་བྱུང་བའི་འབར་གས་དོན་རྐྱེན་ཐད་རྒྱ་ནག་རྒྱལ་སྲིད་སྤྱི་ཁྱབ་ཁང་གིས། ཚོགས་ཆུང་ཆེད་འཛུགས་ཀྱིས་ཞིབ་བཤེར་བྱས་ཏེ་སྙན་ཐོ་ཞིག་བཏོན་པའི་ནང་། དོན་རྐྱེན་དང་འབྲེལ་བ་ཡོད་པའི་ཉེས་མིང་འོག་ཁྱོན་མི་གྲངས་ ༡༧༡ གནས་དབྱུང་གཏོང་དགོས་པའི་བསམ་ཚུལ་བཏོན་ཡོད་པ་དང་། མི་ ༡༠༠ གྲངས་དེའི་ནང་ཞིང་ཆེན་རིམ་པའི་མགོ་ཁྲིད་ ༥ དང་། ཁེ་ལས་འགན་འཁུར་བ་ ༣༥། ས་གནས་རིམ་པའི་ལས་བྱེད་ ༢༢ དང་རྫོང་རིམ་པའི་ལས་བྱེད་ ༢༢། གྲོང་ཚོ་རིམ་པ་དང་དེ་ལས་དམའ་བའི་ལས་བྱེད་ ༢༥ ཡོད་པ་དང་། གཞན་ཡང་འབྲེལ་ཡོད་ཁེ་ལས་པ་ ༢༤ ལ་ཁྲིམས་ལྟར་ཞིབ་བཤེར་དགོས་སྐོར་བརྗོད་ཡོད་འདུག   ཞིབ་འཇུག་བྱས་པའི་གྲུབ་དོན་ནང་། རེའུ་ཧེ་ཚོང་གཉེར་ཁང་གི་ཉེན་ཆས་མཛོད་ཁང་དུ་འཛམ་འདྲེད་རྫས་ཀྱི་ནང་ཟེ་ཚྭ་འདྲེས་འགྱུར་གྱི་སྲིང་བལ་བེད་སྤྱོད་བྱས་པ་དང་། གནམ་གཤིས་ཚ་བ་ཆེ་བ་སོགས་ཀྱིས་འབར་ཉེན་ཆེ་བའི་རྫས་རིགས་གཞན་བར་ཚ་དྲོད་རྒྱས་ཏེ་འབར་གས་བྱུང་ཡོད་སྐོར་བརྗོད་ཡོད་པ་དང་འབྲེལ། ཐེན་ཅིན་གྲོང་ཁྱེར་གྱི་དམར་ཤོག་དྲུང་ཆེ་དང་གྲོང་ཁྱེར་སྲིད་གཞུང་སོགས་ནས་བདེ་འཇགས་ཐད་དོ་ཁུར་མ་བྱས་པའི་སྐྱོན་བརྗོད་བྱས་ཡོད་འདུག   དེ་ཡང་སྙན་ཐོའི་ནང་། ཐེངས་འདིའི་དོན་རྐྱེན་གྱིས་ཁོར་ཡུག་དང་མཁའ་དབུགས། ས་ཁམས། ཆུ་བཅས་ལ་ཚད་རིམ་མི་འདྲ་བའི་འབག་བཙོག་གནོད་སྐྱོན་ཐེབས་ཡོད་པ་དང་། འབག་བཙོག་འདིས་དུས་ཡུན་རིང་པོར་བདེ་ཐང་ལ་གནོད་པ་ཐེབས་སྲིད་སྟབས། འབྲེལ་ཡོད་གནས་ཚུལ་ཐད་མུ་མཐུད་ཞིབ་འཇུག་དང་སྔོན་འགོག་གི་བྱེད་ཐབས་བཀོལ་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག   འབར་གས་དོན་རྐྱེན་བྱུང་རྗེས། རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་དང་སྲིད་བློན་ལི་ཁེ་ཆང་གཉིས་ཀྱིས་དོན་རྐྱེན་གྱི་བྱུང་རྐྱེན་ལ་ཞིབ་བཤེར་ནན་པོ་དགོས་སྐོར་ཐེངས་མང་བརྗོད་ཡོད་པ་ལྟར། རྒྱ་ནག་བདེ་འཇགས་ལྟ་ཞིབ་ཁང་གི་སྤྱི་ཁྱབ་དཔོན་པོ་དང་ཡུན་རིང་ཐེན་ཅིན་གྲོང་ཁྱེར་གྱི་སྤྱི་ཁྱབ་གཞོན་པའི་གོ་གནས་བཟུང་མྱོང་བའི་ཡང་ཏུང་ཉིང་གི་གོ་གནས་དེ་མ་ཐག་ཕབ་ཡོད་པ་དང་། དེ་བཞིན་ཐེན་ཅིན་འགྲིམ་འགྲུལ་ཁང་གི་འགན་འཛིན་གྱི་གཙོས་ས་གནས་གྲོང་ཁྱེར་རིམ་པའི་ལས་བྱེད་ ༡༠ གྲངས་ཤིག་ལ་ཞིབ་བཤེར་དང་གནས་དབྱུང་བཏང་ཡོད་འདུག   རྒྱ་ནག་གཞུང་གི་སྡོམ་རྩིས་ལྟར་ན། འབར་གས་དོན་རྐྱེན་དེའི་ཁྲོད་མེ་འགོག་ལས་བྱེད་ ༧༥ དང་ཉེན་རྟོག་པ་ ༡༡། ཁེ་ལས་པ་དང་ཉེ་འཁྲིས་ཀྱི་སྡོད་མང་ ༥༥་བཅས་ཁྱོན་མི་གྲངས་ ༡༦༥ རྐྱེན་ལམ་དུ་ཤོར་བ་དང་། མི་ ༨ གར་སོང་ཆ་མེད། ཁྱོན་མི་གྲངས་ ༧༩༨ ལ་རྨས་སྐྱོན་བཟོས་ཡོད་པ་དང་། ཁེ་ལས་སྐྱེལ་འདྲེན་རླངས་འཁོར་ ༡༢༤༢༨ ལ་ཚིག་སྐྱོན་ཐེབས་པ་སོགས་དཔལ་འབྱོར་ཐད་ཁྱོན་རྒྱ་སྒོར་དུང་ཕྱུར་ ༦༨.༦༦ གིས་གྱོང་གུད་བཟོས་ཡོད་འདུག   གསར་འགོད་པ། པདྨ་མཚོ།
