དྲག་སྤྱོད་པ་རྩ་མེད་བཟོ་དགོས།

ཁ་སང་ཧྥ་རན་སིའི་རྒྱལ་སར་རྒྱལ་སྤྱིའི་རྒྱལ་ཁབ་ ༥༠ ཙམ་གྱི་མགོ་ཁྲིད་དང་མང་ཚོགས་ས་ཡ་ ༡།༦ ལྷག་འདུ་འཛོམས་ཀྱིས་འཇིགས་སྐུལ་རིང་ལུགས་པའི་བྱ་སྤྱོད་ལ་ངོ་རྒོལ་ལས་འགུལ་སྤེལ་འདུག ཁ་སང་ཧྥ་རན་སིའི་རྒྱལ་སའི་ཐང་ཆེན་(Place de la Republique)ནས་ (Place de la Nation) ཞེས་པའི་བར་ཧྥ་རན་སིའི་སྲིད་འཛིན་ཧོ་ལན་ཌི་དང་། སྲིད་འཛིན་ཟུར་པ་སཱར་ཀཱོ་ཛཱི་དང་ཁོང་གི་སྐྱེས་དམན་ཀཱར་ལཱ་བྷུར་ཎེ། འཇར་མན་གྱི་སྲིད་བློན་ཨན་གྷེ་ལཱ་མེར་གྷལ། པ་ལི་སི་ཐན་གྱི་སྲིད་འཛིན་མཱོ་མཱ་ཌཱ་ཨཱ་བྷ་སེ་(Mahmoud Abbas)། དབྱིན་ཡུལ་གྱི་སྲིད་བློན་ཀེམ་རོན་གཙོས། དཀྱིལ་ཤར་ཡུལ་གྲུ་ཁག་གི་སྲིད་འཛིན་དང་སྲིད་བློན། དེ་བཞིན་ཡུ་རོབ་མཐུན་ཚོགས་ཀྱི་ཚོགས་གཙོ་བཅས་རྒྱལ་ཁབ་ ༥༠ ཙམ་གྱི་དབུ་ཁྲིད་མཉམ་ཞུགས་ཐོག རྒྱལ་སའི་མང་ཚོགས་ས་ཡ་གཅིག་དང་ཕྱེད་ལྷག་དང་། རྒྱལ་ཡོངས་གྲོང་ཁྱེར་ཁག་གི་མང་ཚོགས་ས་ཡ་ ༣།༧ ཙམ་གྱིས་ལག་ཏུ་རྒྱལ་དར་དང་༼ ང་ཚོ་ཆར་ལི་ཡིན། ཞེད་ཀྱི་མི་འདུག ༽ཟེར་བ་དང་༼གསར་འགོད་པའི་རང་དབང་དང་སྨྲ་བརྗོད་རང་དབང་ལ་སུས་ཀྱང་བཀག་འགོག་བྱེད་ཐུབ་ཀྱི་མ་རེད།༽ཟེར་བ་སོགས་ཀྱི་བརྗོད་ཚིག་འདྲ་མིན་འཁོད་པའི་འཕྲེད་དར་འཁྱེར་ཏེ། རང་དབང་དང་ཆར་ལི་ཞེས་སྐད་འབོད་རྒྱག་གིན་གོམ་བགྲོད་ལས་འགུལ་ཞིག་སྤེལ་ཡོད་པ་མ་ཟད། དབྱིན་ཡུལ་རྒྱལ་ས་ལོན་ཌོན་དང་སི་པན་རྒྱལ་ས་མེའུ་རོ་ཌོ། ཉི་ཧོང་རྒྱལ་ས་ཀྲོག་འཁྱོ། ཨ་རིའི་རྒྱལ་ས་ཝ་ཤིང་ཌོན་དང་འཇར་མན་གྱི་རྒྱལ་ས་སྦར་ལན། ཨོ་སི་ཌོ་ལི་ཡའི་གྲོང་ཁྱེར་སིད་ཌི་ཎེ་སོགས་རྒྱལ་སྤྱིའི་རྒྱལ་ས་དང་གྲོང་ཁྱེར་ཁག་ ༣༦ གིས་གཅིག་མཚུངས་ལས་འགུལ་སྤེལ་ཡོད་འདུག དེ་ནི་ཟླ་འདིའི་ཚེས་ ༧ ཉིན་ཧྥ་རན་སིའི་རྒྱལ་སའི་ཆར་ལི་ཧབ་དོ་ (Charlie Hebdo) ཟེར་པའི་ཟུར་ཟའི་དུས་དེབ་རྩོམ་སྒྲིག་ཁང་དུ་རང་ལོ་ ༣༤ ཡིན་པའི་ཤེ་རིབ་ཁཱོ་ཝ་ཆཱི་ (Cherif Kouachi) དང་། རང་ལོ་ ༣༣ ཡིན་པའི་ས་ཡི་ཌཱ་ཁཱོ་ཝ་ཆཱི་(Said Kouachi)ཕུ་ནུ་གཉིས་ཀྱིས་འཇབ་རྒོལ་བྱས་ཏེ། རྩོམ་སྒྲིག་པ་ཚུད་པའི་གསར་འགོད་པ་བཅུ་དང་། སྐོར་སྲུང་བ་གཉིས་བཅས་མི་ ༡༢ སྡེབ་གསོད་དང་། གཞན། ཨི་མེ་ཌི་(Amedy Coulibaly) ཟེར་བ་ཞིག་གིས་ཉེན་རྟོག་པ་བུད་མེད་ཞིག་བསད་དེ། ཁཱོ་ཤར་དགུ་ཚིགས་ཚོང་ཁང་(kosher supermarket) ནང་འཛུལ་ཏེ་མི་གཏའ་མར་ཉར་བ། གཞན་མི་བཞི་བསད་པ་སོགས་ཀྱི་དོན་རྐྱེན་བྱུང་ཡོད་སྟབས། ཧྥ་རན་སི་གཞུང་གིས་ཉེན་རྟོག་པ་མང་པོ་བཏང་སྟེ་དུས་དེབ་ཁང་ལ་གཏོར་རྒོལ་གཏོང་མཁན་ཡིབ་སྡོད་བྱེད་ས་ཌེམ་མར་ཌིན་ཨིན་གྷོ་ལེ་(Dammartin-en-Goele) མཐའ་བསྐོར་ནས་མཐར་ཁོ་གཉིས་བསད་ཡོད། འདིའི་ཐད་ཧྥ་རན་སིའི་སྲིད་འཛིན་ཧོ་ལན་ཌི་མཆོག་གིས། “དེ་རིང་ཧྥ་རན་སི་ནི་རྒྱལ་སྤྱིའི་རྒྱལ་ས་ཆགས་ཡོད་སྟབས། འཛམ་གླིང་རྒྱལ་ཁབ་དང་མང་ཚོགས་ཡོངས་ཀྱིས་ཡར་ལངས་བྱེད་བཞིན་ཡོད”ཅེས་དང་། ཧྥ་རན་སིའི་ནང་སྲིད་བློན་ཆེན་བྷེ་ཎར་ཌི་(Bernard Cazeneuve) ཡིས། “ད་ལྟ་ཧྥ་རན་སིའི་རྒྱལ་སའི་མང་ཚོགས་ས་ཡ་༡།༦ དང་རྒྱལ་ཡོངས་ཀྱི་མང་ཚོགས་ས་ཡ་ ༣།༧ ཡིས་ཞི་བའི་གོམ་བགྲོད་བྱེད་བཞིན་ཡོད་སྟབས། ང་ཚོས་ཉེན་རྟོག་པ་ ༢༢༠༠ དང་མེ་མདའ་རྒྱག་མཁན་(sniper) ༡༣༠༠ ལྷག་བཏང་སྟེ། ཞི་རྒོལ་བ་ཚོར་སྲུང་སྐྱོབ་བྱེད་བཞིན་ཡོད”ཅེས་བརྗོད་འདུག ཉིན་དེར་དྲག་སྤྱོད་སྤེལ་མཁན་འཇིགས་སྐུལ་རིང་ལུགས་པ་ཧཱ་ཡཱ་ཌི་ (Hayat Boumeddiene)ཟེར་བའི་བུད་མེད་ཅིག་རྒྱལ་ཁབ་གཞན་ནང་བྲོས་བྱོལ་ཕྱིན་ཏེ་འཇུ་བཟུང་ཐུབ་མེད། ཁ་སང་འཇིགས་སྐུལ་རིང་ལུགས་པ་ཨི་མེ་ཌི་ཡིས་བརྙན་ཐུང་ཞིག་ཁྱབ་བསྒྲགས་བྱས་ཡོད་པའི་ནང་། ང་ཚོས་ཡཱ་མན་གྱི་ཨལ་ཁལ་ཌཱ་འཇིགས་སྐུལ་ཚོགས་པའི་བཀོད་ཇུས་འོག་ཨཱ་ལཱ་ཡི་དགྲ་བོར་ཁ་གཏད་གཅོག་ཆེད། ཤོག་ཁག་གཉིས་ལ་ཆ་བགོས་བྱས་ཏེ་གཅིག་གིས་ཆར་ལི་ཧབ་ཌོ་དུས་དེབ་ཁང་དང་། གཞན་གྱིས་ཉེན་རྟོག་པར་ཁ་གཏད་གཅོག་རྒྱུ་བྱས་ཏེ་གཏོར་རྒོལ་ཤུགས་ཆེན་གཏོང་རྒྱུའི་འཆར་གཞི་ཡོད། ཅེས་བརྗོད་ཡོད་འདུག ཁ་སང་སྔ་དྲོར་ཡུ་རོབ་མཐུན་ཚོགས་ཀྱི་རྒྱལ་ཁབ་ཁག་ནང་སྲིད་བློན་ཆེན་དང་དབྱིན་ཡུལ་གྱི་ནང་སྲིད་བློན་ཆེན་བར་ལྷན་ཚོགས་ཤིག་ཚོགས་ཏེ། ཚོགས་ཐོག་ཏུ་རྒྱལ་ཁབ་ཁག་གིས་དྲ་རྒྱའི་བདེ་འཇགས་དང་ས་མཚམས་ལྟ་སྲུང་ནན་མོ་བྱས་ཏེ་འཇིགས་སྐུལ་རིང་ལུགས་པ་དང་འཐབ་རྒོལ་བྱེད་རྒྱུ་ཐག་གཅོད་བྱས་ཡོད་འདུག ཟླ་འདིའི་ཚེས་ ༡༡ ཉིན་ཧྥ་རན་སིའི་རྒྱལ་སར་མང་ཚོགས་དང་རྒྱལ་སྤྱིའི་དབུ་ཁྲིད་ཁག་གིས་འཇིགས་སྐུལ་རིང་ལུགས་པས་རྐྱེན་འདས་བཏང་བ་དག་གི་ནང་མི་ཚོར་གདུང་སེམས་མཉམ་སྐྱེད་ཀྱི་གྱོན་ཆས་གྱོན་ཞིང་། ལག་ཏུ་རྒྱལ་དར་དང་འཕྲེད་དར་བཟུང་སྟེ། དབྱིན་ཡིག་དང་ཧྥ་རན་སིའི་སྐད་ཡིག ཨི་རག་བྷེག་སྐད་ཡིག་སོགས་ནང་བརྗོད་ཚིག་འདྲ་མིན་བྲིས་ནས་ང་ཚོ་ཞེད་ཀྱི་མེད་ཅེས་པ་སོགས་བརྗོད་ནས་སྐད་འབོད་བརྒྱབ་ཡོད་པ་མ་ཟད། འཕྲེད་དར་ཐོག་ཏུ་ང་ཧྥ་རན་སིའི་མི་སེར་ཡིན། སུ་གང་ལ་ཞེད་ཀྱི་མི་འདུག བསྟན་བཤིག་སློང་བ་ལས་དམག་མ་རྒྱག ཁྲག་ཐོན་པ་ལས་སྣག་ཚ་བེད་སྤྱོད་བྱས་ཏེ་བསམ་ཚུལ་སྤེལ་ཐུབ་པ་བྱེད་དགོས་སོགས་བརྗོད་ཚིག་འདྲ་མིན་བྲིས་ཏེ་གཞི་རྒྱ་ཆེ་བའི་གོམ་བགྲོད་ཅིག་བྱས་ཡོད་ཅིང་། སྐབས་དེར་དབྱིན་ཡུལ་གྱི་སྲིད་བློན་ཀེམ་རོན་མཆོག་གིས་ད་རེབ་ང་ཚོ་འདིར་ཡོང་དགོས་དོན་ནི་དམངས་གཙོ་དང་རང་དབང་། སྨྲ་བརྗོད་རང་དབང་སོགས་ཀྱི་རིན་ཐང་ལ་བརྩི་སྲུང་ཞུ་ཆེད་འཛོམས་པ་ཞིག་ཆགས་ཡོད་སྟབས། མ་འོངས་པར་ང་ཚོས་རྒྱལ་སྤྱི་རྒྱལ་ནང་གི་བདེ་འཇགས་ལ་སྲུང་སྐྱོབ་དང་ལྟ་སྲུང་ནན་མོ་བྱས་ཏེ། འཇིགས་སྐུལ་རིང་ལུགས་པ་དག་ལ་ཁ་གཏད་གཅོག་ཐུབ་པ་བྱེད་དགོས་ཚུལ་དང་།  ཨི་ཌི་ལིའི་སྲིད་བློན་མཱ་ཌོ་རན་ཛི་ (Matteo Renzi) མཆོག་གིས་ཡུ་རོབ་མཐུན་ཚོགས་ཀྱིས་འཇིགས་སྐུལ་རིང་ལུགས་པར་ཁ་གཏད་གཅོག་རྒྱུའི་ཐག་གཅོད་བྱས་ཡོད་པ་མ་ཟད། དེ་རིང་ལས་འགུལ་འདིའི་རྐྱེན་པས་ང་ཚོ་ཚང་མ་ཧྥ་རན་སིའི་མི་སེར་ཆགས་ཡོད་སྟབས། ང་ཚོས་ནུས་ཤུགས་གཅིག་སྒྲིལ་བྱས་ཏེ་དགྲ་བོ་ཚོར་ང་ཚོས་ཁོ་ཚོར་ཞེད་ཀྱི་མེད་པ་གསལ་བཤད་བྱེད་ཐུབ་བྱ་དགོས་ཞེས་བརྗོད་ཡོད་འདུག དབྱིན་ཡུལ་གྱི་ཚགས་ཤོག་ཆ་འཕྲིན་གྱི་གསར་འགོད་པས་ཛཱ་ཁཱ་རེ་ཡཱ་ (Zakaria Moumni) ལ་གདུང་སེམས་མཉམ་སྐྱེད་ལས་འགུལ་ནང་མཉམ་ཞུགས་བྱེད་དགོས་དོན་གང་ཡིན་དྲིས་པར། ཁོ་མོས། ངས་འདིར་ཡོང་དགོས་དོན་ནི་འཇིགས་སྐུལ་རིང་ལུགས་པར་རྒྱལ་ཁ་ཐོབ་མེད་པ་མ་ཟད། དེ་ལས་ལྡོག་སྟེ་དོན་རྐྱེན་དེའི་རྐྱེན་པས་ཆོས་ལུགས་དང་མི་རིགས་འདྲ་མིན་གྱི་མང་ཚོགས་དག་གོང་བུ་གཅིག་སྒྲིལ་གྱིས་མཉམ་དུ་ཡར་ལངས་བྱས་ཡོད་པ་དེ་ཤེས་རྟོགས་ཐུབ་ཆེད་ཡོང་བ་ཡིན་ཞེས་དང་། ཡང་འཇིགས་སྐུལ་རིང་ལུགས་པ་ཕུ་ནུ་གཉིས་སྡོད་ཡུལ་གྷན་ཎེ་(Gennevilliers) ནང་སྡོད་མཁན་གྱི་ན་གཞོན་ཞིག་ལ་ལས་འགུལ་ནང་མཉམ་ཞུགས་མ་བྱས་པའི་རྒྱུ་མཚན་གང་ཡིན་དྲིས་པར་ཁོས། ཁཱོ་ཝ་ཆཱི་ཕུ་ནུ་གཉིས་ནི་ང་ཚོའི་ས་ཁུལ་གྱི་དཔའ་རྒོད་ཆ་གཅིག་ཡིན་སྟབས། དེ་རིང་ང་ཚོས་ཁོ་གཉིས་ཀྱི་བྱས་རྗེས་ལ་རྟེན་འབྲེལ་བྱེད་བཞིན་ཡོད་སྟབས། ལས་འགུལ་ནང་མཉམ་ཞུགས་བྱས་མེད་ཅེས་བརྗོད་འདུག  གནད་དོན་དབྱེ་ཞིབ་པ་ཚོས། ང་ཚོས་སི་རི་ཡི་དང་ཨི་རག་གི་མཉམ་འབྲེལ་དྲག་སྤྱོད་ཤོག་ཁག་ (ISIS) དང་ཨལ་ཁལ་ཌཱ་འཇིགས་སྐུལ་རིང་ལུགས་གཉིས་ཀྱི་འགྲོ་འདུག་སྤྱོད་གསུམ་ལ་རྟོག་ཞིབ་ནན་པོ་མ་བྱས་ཚེ། ལོ་འདི་འཇིགས་སྐུལ་རིང་ལུགས་པས་གཏོར་རྒོལ་དང་མི་གསོད་དོན་རྐྱེན་མང་ཤོས་བྱེད་ཡུལ་གྱི་ལོ་ཞིག་ཏུ་འགྱུར་རྒྱུ་ཡིན་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
