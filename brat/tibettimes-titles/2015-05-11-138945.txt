ཆོས་སྐྱིད་སྒྲོལ་མར་ཨང་གཉིས་པ་ཐོབ།

བོད་རིགས་བུ་མོ་ཆོས་སྐྱིད་སྒྲོལ་མ་ལགས་ཀྱིས་ཨེ་ཤ་ཡའི་ན་གཞོན་འདུར་རྒྱུག་འགྲན་སྡུར་ཁྲོད་ཨང་ ༢ པའི་བྱ་དགའ་བླངས་ཡོད་འདུག རྒྱ་ནག་ལུས་རྩེད་དྲ་བའི་ཐོག་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་ཁོ་ཊར(Qatar) གྱི་རྒྱལ་ས་ཌོ་ཧཱ་རུ། ཕྱི་ལོ་ ༢༠༡༥ ལོའི་ཨེ་ཤ་ཡའི་ན་གཞོན་མཆོང་རྒྱུག་འགྲན་ཚོགས་འཚོགས་བཞིན་ཡོད་པ་ལྟར། ཟླ་འདིའི་ཕྱི་ཚེས་ ༩ ཉིན་གྱི་སྔ་དྲོར་རྨིད་ ༥༠༠༠ ཅན་གྱི་སྐྱེད་མའི་འདུར་རྒྱུག་འགྲན་སྡུར་ཁྲོད། མཚོ་སྔོན་ཞིང་ཆེན་གྱི་ལུས་རྩལ་བ་ཆོས་སྐྱིད་སྒྲོལ་མ་ལགས་མགྱོགས་ཚད་སྐར་མ་ ༢༤ དང་སྐར་ཆ་ ༨ ལས་མ་འགོར་བར་ཨང་གཉིས་པའི་རྒྱལ་ཁ་ཐོབ་ཡོད་པ་དང་། ཁོང་མོ་ནི་རྒྱལ་སྤྱིའི་རྒྱུག་འགྲན་ཁྲོད་ཨང་རིམ་ལེན་མཁན་གྱི་བོད་རིགས་བུ་མོ་གསུམ་པ་དེ་ཆགས་ཡོད་པ་རེད། ཁོང་མོས་གསར་འགོད་པར་བརྗོད་རྒྱུར། ནམ་རྒྱུན་ང་བདུན་ཕྲག་རེའི་ནང་ཉུང་མཐར་སྤྱི་ལེ་ ༡༨༠ རེར་གོམ་བགྲོད་དུ་འགྲོ་ཡི་ཡོད་ལ། ཨེ་ཤ་ཡའི་ན་གཞོན་འགྲན་ཚོགས་ཀྱི་སྦྱོང་བརྡར་སྐབས། ཉིན་རེར་སྤྱི་ལེ་ ༣༠ རེར་གོམ་བགྲོད་དུ་ཕྱིན་པ་དང་། ང་རང་གཅིག་པུ་ཡིན་པས་ཁེར་རྐྱང་གི་ཚོར་བའང་བྱུང་། ཞེས་དང་། ཡང་ཁོང་མོས་ཨོ་རྩལ་འགྲན་ཚོགས་ནང་ཞུགས་རྒྱུ་ཁོང་མོའི་རེ་བ་ཆེ་ཤོས་དེ་ཡིན་སྟབས། དཀའ་ངལ་ཅི་ཙམ་ཆེ་ནའང་འབད་བརྩོན་བྱ་རྒྱུ་ཡིན་སྐོར་ཡང་བརྗོད་འདུག རང་ལོ་ ༡༧ ཡིན་པའི་ཆོས་སྐྱིད་སྒྲོལ་མ་ལགས་ནི། མགོ་ལོག་བོད་རིགས་རང་སྐྱོང་ཁུལ་རྨ་ཆེན་རྫོང་རྟ་བོ་གྲོང་རྡལ་ནས་ཡིན་ཞིང་། ཁོང་མོས་ཕྱི་ལོ་ ༢༠༡༤ ཟླ་ ༤ ནང་ཚོགས་པའི་སྐབས་ ༤ པའི་ལི་ལིན་བུམ་པའི་འགྲན་ཚོགས་སྐབས། སྤྱི་ལེ་ ༥ དང་ ༡༠ ཅན་གྱི་འགྲན་སྡུར་སོ་སོའི་ནང་ཨང་དང་པོ་བླངས་ཡོད་པ་དང་། དེ་བཞིན་ལོ་དེའི་ཕྱི་ ཟླ་ ༩ པའི་ནང་ཅི་ལིན་དུ་ཚོགས་པའི་རྒྱ་ནག་རྒྱལ་ཡོངས་འདུར་རྒྱུག་འགྲན་སྡུར་ཁྲོད་ཨང་གསུམ་པ་བླངས། ཡང་ལོ་དེའི་ཕྱི་ཟླ ༡༠ ནང་སྤེལ་བའི་ཡུལ་སྐོར་བུམ་པའི་རྒྱལ་སྤྱིའི་རྒྱུག་འགྲན་ནང་ཨང་དང་པོ་གསུམ་དང་ཨང་གཉིས་པ་གཅིག་བཅས་ཀྱི་བྱ་དགའ་ཐོབ་ཡོད་པ་ལྟར། ཁོང་མོ་ཕྱི་ལོ་ ༢༠༡༤ ལོའི་རྒྱ་ནག་གི་ཨོ་ལིན་ཕིག་གི་ཚད་མཐོའི་རྗེས་སྣོན་ལུས་རྩལ་པའི་འདུར་རྒྱུག་སྦྱོང་བརྡར་ཁྲོད་ཕུལ་བྱུང་གི་ས་བོན་དུ་བདམས་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
