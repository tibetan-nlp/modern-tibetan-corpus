ལྷོ་ཀོ་རི་ཡར་འཇིགས་སྐུལ་བྱས།

ཟླ་འདིའི་ཚེས་ ༨ ནས་བཟུང་ཉིན་གསུམ་རིང་ལྷོ་ཀོ་རི་ཡའི་ནང་སྟོན་འབྲས་བསྡུ་པའི་དུས་ཆེན་ (Harvest Festival) བསླེབ་ལ་ཉེ་བ་དང་བསྟུན་ཏེ། དེ་རིང་སྔ་དྲོར་བྱང་ཀོ་རི་ཡ་གཞུང་གིས་ཁེན་ཝང་མངའ་སྡེའི་ཝང་སང་གྲོང་ཁྱེར་གྱི་མཚོ་དམག་ས་ཚིགས་བརྒྱུད། ལྷོ་ཀོ་རི་ཡ་དང་ཉི་ཧོང་གི་མཚོ་ཁུལ་དུ་ཤུགས་ཚད་ཆུང་བའི་མེ་ཤུགས་འཕུར་མདའ་གསུམ་ཚོད་ལྟ་བྱས་ཏེ་ལྷོ་ཀོ་རི་ཡར་འཇིགས་སྐུལ་བྱས་ཡོད་པ་དང་། སྲིད་དོན་པ་ཚོས་དེ་ནི་ཉེ་ལམ་ལྷོ་ཀོ་རི་ཡས་ཨ་རི་དང་མཉམ་དུ་དམག་སྦྱོང་བྱ་རྒྱུའི་གྲོས་མཐུན་བཞག་པ་དང་། རྒྱ་ནག་གཙོས་རྒྱལ་སྤྱིའི་སྒྲིག་འཛུགས་ཁག་དང་འབྲེལ་ལམ་འཛུགས་བཞིན་ཡོད་སྟབས། བྱང་ཀོ་རི་ཡས་དེ་ལ་མ་འདོད་པའི་རྣམ་འགྱུར་བསྟན་པ་ཞིག་ལ་ངོས་འཛིན་བྱེད་བཞིན་ཡོད་འདུག ལྷོ་ཀོ་རི་ཡའི་དམག་སྤྱིས་ཡོན་ཧེབ་ (Yonhap) གསར་འགྱུར་བརྒྱུད་ལམ་ཐོག་བརྗོད་དོན། “དེ་རིང་བྱང་ཀོ་རི་ཡས་ཚོད་ལྟ་བྱས་པའི་མེ་ཤུགས་འཕུར་མདའ་དེ་དག་ནི། འདི་ལོའི་ཕྱི་ཟླ་ ༨ ཚེས་ ༡༤ དང་ཟླ་ ༩ ཚེས་ ༡ ཉིན་ཚོད་ལྟ་བྱས་པའི་མེ་ཤུགས་འཕུར་མདའ་དག་དང་གཅིག་མཚུངས་ཡིན་སྟབས། བྱང་ཀོ་རི་ཡས་ཤུགས་ཚད་ཆེ་བའི་མེ་ཤུགས་འཕུར་མདའ་གསར་བཟོ་བྱེད་བཞིན་ཡོད་པ་མཚོན་ཐུབ་ཅིང་། ད་ལྟ་ལྷོ་ཀོ་རི་ཡས་དམག་དཔུང་གིས་བྱང་ཀོ་རི་ཡའི་དམག་དཔུང་གི་འགུལ་སྐྱོད་ལ་རྟོག་ཞིབ་བྱེད་བཞིན་ཡོད” ཅེས་བརྗོད་འདུག ལྷོ་བྱང་ཀོ་རི་ཡའི་སྟོན་འབྲས་བསྡུ་བའི་དུས་ཆེན་བསླེབ་ལ་ཉེ་བའི་སྐབས་འདིར། ཟླ་འདིའི་ཚེས་ ༥ ཉིན་བྱང་ཀོ་རི་ཡས་དེ་སྔོན་འཇུ་བཟུང་བྱས་པའི་ལྷོ་ཀོ་རི་ཡའི་མི་སེར་ཞིག་ལྷོ་ཀོ་རི་ཡར་ཕྱིར་སློག་བྱས་ཡོད་སྟབས། ཕྱི་ཕྱོགས་སུ་ལྷོ་བྱང་ཀོ་རི་ཡའི་འབྲེལ་ལམ་ལེ་ལེགས་སུ་འགྲོ་སྲིད་པར་ངོས་འཛིན་བྱེད་མཁན་བྱུང་ཡོད་ཀྱང་། བྱང་ཀོ་རི་ཡས་དེ་རིང་གི་བྱེད་སྟངས་དེས་སླར་ཡང་རྒྱལ་ཁབ་གཉིས་བར་གྱི་འབྲེལ་ལམ་མུ་མཐུད་ཇེ་སྡུག་ཏུ་འགྲོ་བཞིན་ཡོད་པ་བསྟན་ཡོད། ༢༠༡༠ ལོར་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱིས་བྱང་ཀོ་རི་ཡས་རྡུལ་མཚོན་ཚོད་ལྟ་བྱེད་མི་ཆོག་པའི་གྲོས་ཆོད་བཞག་པ་དེར་བྱང་ཀོ་རི་ཡས་བརྩི་སྲུང་བྱས་མེད་པ་མ་ཟད། ལོ་ལྟར་མཚོན་ཆ་ཚོད་ལྟ་བྱས་ཏེ་རྒྱལ་ཁབ་སྲུང་སྐྱོབ་དང་ཨ་རིའི་དམག་ཤུགས་ལ་ཚོད་འཛིན་བྱེད་བཞིན་ཡོད་སྐོར་རྗོད་བཞིན་ཡོད་ལ། ༢༠༡༤ ལོའི་ཟླ་ ༦ ཚེས་ ༢༦ ཉིན་མེ་ཤུགས་འཕུར་མདའ་ ༣ དང་། ཚེས་ ༢༩ ཉིན་ ༢ བཅས་ད་བར་ལོ་འདིའི་ནང་ཤུགས་ཚད་འབྲིང་ཆུང་གཉིས་ཀྱི་སྒྱོགས་མདེལ་སོགས་ ༡༡༡ ཙམ་ཚོད་ལྟ་བྱས་ཡོད་འདུག ཆབ་སྲིད་དབྱེ་ཞིབ་པ་ཚོས། བྱང་ཀོ་རིས་ལྷོ་ཀི་རི་ཡར་འཇིགས་སྐུལ་བྱས་ཏེ་མཚོན་ཆ་ཚོད་ལྟ་བྱེད་བཞིན་ཡོད་པ་དག་ནི། རྒྱལ་སྤྱིའི་ཁྲིམས་ལུགས་དང་འགལ་བ་ཞིག་ཡིན་པ་མ་ཟད། ཨེ་ཤ་ཡ་གཙོས་རྒྱལ་སྤྱིའི་བརྟན་ལྷིང་ལ་དཀའ་ངལ་བཟོ་བཞིན་ཡོད་སྟབས། མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱིས་བྱང་ཀོ་རི་ཡར་བརྡ་ལན་སྤྲོད་དགོས་སྐོར་རྗོད་བཞིན་ཡོད། གསར་འགོད་པ། དོན་གྲུབ།
