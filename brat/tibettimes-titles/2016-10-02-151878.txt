སྦག་སྦག་སྐོར་སྐྱོད་ལས་འགུལ་སྤེལ།

དེ་རིང་རྒྱ་གར་ལྷོ་ཕྱོགས་སྦེལ་ཀོབ་ཏུ་དབུ་མའི་ལམ་ལས་འགུལ་ཚོགས་ཆུང་གི་གོ་སྒྲིག་འོག་སྦག་སྦག་སྐོར་སྐྱོད་ཀྱི་བོད་དོན་དྲིལ་བསྒྲགས་ལས་འགུལ་ཞིག་མགོ་འཛུགས་གནང་ཡོད་འདུག དེ་རིང་ཕྱི་ཚེས་ ༢ ཉིན་གྱི་ཞོགས་པའི་ཆུ་ཚོད་ ༩ པའི་སྟེང་། སྦེལ་ཀོབ་ས་གནས་དབུ་མའི་ལམ་མི་མང་ལས་འགུལ་ཁང་གི་གོ་སྒྲིག་འོག སྦེལ་ཀོབ་ལུགས་ཟུང་བསམ་འགྲུབ་གླིང་དང་བདེ་སྐྱིད་སླར་གསོ་གཉིས་ཀྱི་མི་མང་མཉམ་འཛོམས་ཐོག ལུགས་ཟུང་གཞིས་གྲོང་དང་པོའི་བྱང་ཆུབ་མཆོད་བརྟེན་མདུན་ཐང་དུ། མེས་པོ་དམ་པ་གྷན་དྷའི་འཁྲུངས་སྐར་དང་བསྟུན་ཏེ་སྦག་སྦག་སྐོར་སྐྱོད་ཀྱི་བོད་དོན་དྲིལ་བསྒྲགས་ལས་འགུལ་ཞིག་དབུ་འཛུགས་གནང་ཡོད་འདུག ལས་འགུལ་དེའི་ནང་སྦག་སྦག་ ༧ དང་མི་གྲངས་ ༡༢ བཅས་དང་དེ་ལ་རོགས་འདེགས་ལྟ་བུར་རླངས་འཁོར་གཅིག་བཅས་ཡོད་པ་དང་། སྦེལ་ཀོབ་ནས་བྷིང་ལོར་བརྒྱུད་རྒྱ་གར་རྒྱལ་ས་ལྡི་ལིར་བསྐྱོད་རྒྱུ་ཡིན་པ་དང་། ལྡི་ལི་ནས་ཟླ་འདིའི་ཚེས་ ༧ ཉིན་དྷ་རམ་ས་ལར་འབྱོར་རྒྱུ་ཡིན་འདུག སྦེལ་ཀོབ་དབུ་མའི་ལམ་ས་གནས་ལས་འགུལ་ཚོགས་ཆུང་གི་ཚོགས་གཞོན་ཆོས་ལྡན་ལགས་ཀྱིས་ངོ་སྤྲོད་གནང་དོན། ཕྱི་ལོ་ ༡༩༥༩ ལོར་རྒྱ་ནག་གཞུང་གིས་བོད་ཡུལ་འདི་ཉིད་བཙན་གནོན་དབང་བསྒྱུར་བྱས་པ་ནས་བཟུང་རྒྱ་གར་དང་ལྷོ་ཨེ་ཤ་ཡར་ཞི་བདེའི་གནས་འབབ་མེད་པ་ཆགས་ཡོད། ཐོག་མ་རྒྱ་ནག་གཞུང་གིས་བོད་ཡུལ་ཧམ་ཟོས་ཀྱི་ངན་སེམས་འཆང་སྐབས། བོད་གཞུང་གིས་བོད་དེ་ཞི་བདེ་ལྟེ་གནས་ཤིག་ཏུ་བསྒྱུར་ཐབས་ལ་ཐད་འབོད་སྐུལ་ཡང་ཡང་ཞུས་ཀྱང་། དོན་འབྲས་ལྡན་པའི་བཟང་ཕྱོགས་ཀྱི་རྣམ་པ་མེད་ཁར་ཕྱི་ལོ་ ༡༩༥༩ ལོར་བོད་ལྗོངས་ཡོངས་རྫོགས་དབང་བསྒྱུར་བྱས་རྗེས་༧གོང་ས་མཆོག་གིས་དབུས་བོད་མི་བརྒྱད་ཁྲི་ལྷག་རྒྱ་གར་དུ་བཙན་བྱོལ་དུ་གྱར་དགོས་བྱུང་ཡོད། སྤྱིར་བོད་ནི་འཛམ་གླིང་ཡང་ཐོག་གི་ཆུ་མཛོད་ལྟ་བུ་ཡིན་པ་དང་། ལྷག་པར་དུ་ལྷོ་ཕྱོགས་ཨེ་ཤ་ཡའི་ནང་གི་ཆུ་བོ་ཆེན་པོ་རྣམས་ཀྱི་འབྱུང་ཁུངས་ཡིན་ཡང་། རྒྱ་ནག་གཞུང་གིས་ཆུ་མཛོད་གསར་བསྐྲུན་དང་གཏེར་ཁ་སྔོག་འདོན་སོགས་བྱས་ཏེ། རྒྱུག་ཆུ་རྣམས་སྦགས་གཙོག་དང་འཁོར་ཡུག་ལ་གཏོར་བཅོམ་ཚད་མེད་བྱས་ཏེ། རྒྱ་གར་གཙོས་ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་ཁག་ལ་གནོད་སྐྱོན་ཆེན་པོ་ཐེབས་བཞིན་ཡོད། རྒྱ་ནག་གཞུང་གིས་འདས་པའི་ལོ་ངོ་ ༦༠ ལྷག་རིང་བོད་པའི་ཆོས་དང་རིག་གཞུང་རྩ་གཏོར་བཏང་སྟབས། ༢༠༠༩ ཚུན་ལ་བོད་མི་མང་དག་ཅིག་གིས་རང་ལུས་མེར་བསྲེགས་གནང་བ་དང་། དེ་དག་ལས་མི་ ༡༢༤ ཙམ་འདས་གྲོངས་སུ་ཕྱིན་པ། དེ་བཞིན། དགོན་སྡེ་ཁག་ལ་དམ་དྲག་བྱས་ཏེ་གྲྭ་བཙུན་ཕྱིར་འཕུད་དང་གྲྭ་ཤག་གཏོར་བཤིག་བཏང་དང་གཏོང་བཞིན་ཡོད་སྟབས། རྒྱལ་སྤྱིའི་དྲང་བདེན་ལ་དགའ་བའི་རྒྱལ་ཁབ་དང་མི་མང་ཚོས་བོད་དོན་ལ་རྒྱབ་སྐྱོར་ཡོང་བའི་འབོད་སྐུལ་ཞུ་བཞིན་ཡོད། ཅེས་སོགས་བརྗོད་སོང་། ཐེངས་འདིའི་སྦག་སྦག་སྐོར་སྐྱོད་ལས་འགུལ་ཁྲོད་དབུ་མའི་ལམ་དང་འབྲེལ་བའི་ཡིག་ཆ་ཁག་འགྲེམ་སྤེལ་དང་། འཛམ་གླིང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་དང་རྒྱ་གར་སྲིད་བློན་མཆོག དེ་བཞིན། ལྡི་ལིའི་རྒྱ་ནག་གཞུང་ཚབ་བཅས་སུ་ཡིག་ཆ་འབུལ་འཆར་ཡོད་འདུག རྒྱ་གར་ལྷོ་ཕྱོགས་ས་གནས་གསར་འགོད་པ་དོན་གྲུབ་བཀྲ་ཤིས།
