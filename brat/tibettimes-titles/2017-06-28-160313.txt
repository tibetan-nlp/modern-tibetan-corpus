སཱལ་མན་ཁཱང་གི་གློག་བརྙན་གསར་ཤོས།

རྒྱ་གར་གྱི་གློག་བརྙན་འཁྲབ་སྟོན་པ་གྲགས་ཆེན་དང་ཕོ་གསར་རྫིག་ཤོས་གྲས་སུ་བརྩི་བའི་སཱལ་མན་ཁཱང་གློག་བརྙན་གསར་པ་ཐྲེབ་ལེ་ཊི་ཞེས་པ་སྟོན་མགོ་བཙུགས་ཏེ་བདུན་ཕྲག་གཅིག་ནང་ཡོང་འབབ་ཧིན་སྒོར་དུང་ཕྱུར་བརྒྱ་ལ་བསླེབ་རྒྱུ་རེད། དེ་རིང་རྒྱ་གར་དུས་བབ་ཀྱིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཟླ་འདིའི་ཚེས་ ༢༣ ཉིན་རྒྱ་གར་གྱི་གློག་བརྙན་འཁྲབ་སྟོན་པ་སཱལ་མཱན་ཁཱང་(Salman Khan)གི་གློག་བརྙན་ཐྲེབ་ལེ་ཊི་ (Tubelight) ཞེས་པ་གློག་བརྙན་ཁང་ཁག་ཏུ་སྟོན་མགོ་བཙུགས་ཏེ། ཉིན་གཅིག་གི་ནང་ཡོང་འབབ་ཧིན་སྒོར་དུང་ཕྱུར་ ༡༡.༧༥(Rs 11.75 crore) ཟིན་ཡོད་པ་དང་། ཉིན་ལྔའི་ནང་ཧིན་སྒོར་དུང་ཕྱིར་ ༩༣ (Rs93crore)ཟིན་ཏེ། བདུན་ཕྲག་གཅིག་ནང་ཧིན་སྒོར་དུང་ཕྱུར་ ༡༠༠ ལ་བསླེབ་རྒྱུར་ཚོད་དཔག་བྱེད་བཞིན་ཡོད་འདུག གློག་བརྙན་དེ་ཟླ་འདིའི་ཚེས་ ༢༣ ཉིན་ཏེ་ཁ་ཆེ་ཆོས་ལུགས་ཀྱི་དུས་ཆེན་ཨེ་དྷི་(Eid)ཞེས་པའི་རྒྱ་གར་རྒྱལ་ཡོངས་གུང་གསེང་དང་བསྟུན་ཏེ་སྤྱི་ཚོགས་ཐོག་འགྲེམ་སྤེལ་བྱས་ཡོད་པ་དང་། གློག་བརྙན་དེ་ ༢༠༡༧ ལོའི་རྒྱ་གར་གྱི་གློག་བརྙན་ཁྲོམ་རྭའི་ནང་གི་འཕར་ཚད་མཐོ་ཤོས་ཨང་གཉིས་པ་དེ་ཆགས་ཡོད་འདུག ཨང་དང་པོ་དེ་ཉེ་སྔོན་ཟླ་སྔོན་མའི་ནང་འགྲེམ་སྤེལ་བྱས་པའི་གློག་བརྙན་འཁྲབ་སྟོན་པ་(Prabhas)ཡིས་འཁྲབ་སྟོན་བྱས་པའི་བྷ་ཧུ་བྷ་ལེ་གཉིས་པ་(baahubali 2)དེ་ཆགས་ཡོད་པ་རེད། ཐྲེབ་ལཱ་ཊི་གློག་བརྙན་གྱི་འཁྲབ་ཁྲིད་པ་ཀཱ་པོར་ཁཱང་(Kabir Khan)ཡིན་པ་དང་། ཁོང་ནི་རྒྱ་གར་གྱི་གློག་བརྙན་འཁྲབ་ཁྲིད་པ་དང་འཁྲབ་གཞུང་རྩོམ་པ་པོ། གློག་བརྙན་ཕབ་སྐྲུན་པ་སོགས་ཡིན་པ་མ་ཟད། རྒྱ་གར་ནེ་རུ་གཙུག་ལག་སློབ་ཆེན་གྱི་སྤྱི་ཚོགས་ཚན་རིག་སློབ་དཔོན་ཀྱང་ཡིན་འདུག ཁོང་གིས་འཁྲབ་ཁྲིད་བྱས་པའི་གློག་བརྙན་སྐད་གྲགས་ཆེ་ཤོས་དག་ནི། ༢༠༠༦ ལོའི་( Kabul Express)དང་ ༢༠༠༩ ལོའི་( New York)། དེ་བཞིན། ༢༠༡༢ ལོའི་(Ek Tha Tiger)། ༢༠༡༥ ལོའི་(Bajrangi Bhaijaan)སོགས་ཆགས་འདུག ད་ཐེངས་ཁོང་གི་འཁྲབ་ཁྲིད་བྱས་པའི་ཐྲེབ་ལཱ་ཊི་ཞེས་པའི་གློག་བརྙན་གྱི་ནང་དོན། ཁྱིམ་ཚང་ཕལ་བ་ཞིག་གི་གཏམ་རྒྱུད་ཡིན་པ་དང་། མི་ཞིག་གིས་རང་གི་ནང་མི་ལ་བཅང་བའི་ལྷད་མེད་ཀྱི་བརྩེ་བ་དང་། རང་ཉིད་ལ་འགྱུར་བ་མེད་པའི་ཡིད་ཆེས་བརྟན་པོ་ཡོད་པ་བཅས་ཀྱི་སློབ་གསོའི་ནུས་པའི་སྐོར་གྱི་གཏམ་རྒྱུད་ཅིག་ཡིན་འདུག གློག་བརྙན་འདིར་ཕར་འགྲོ་སོང་ཧིན་སྒོར་དུང་ཕྱུར་ ༡༠ (Rs 100 crore)བཏང་ཡོད་པ་དང་། འཛམ་གླིང་ཁྱོན་ཡོངས་ལ་བསྟན་ཏེ་བདུན་དང་པོའི་ནང་རྒྱ་གར་སྒོར་དུང་ཕྱིར་ ༢༢.༥༧(Rs 22.57 crore)ཟིན་ཏེ་དུས་ཐུང་ཞིག་གི་ནང་ཡོང་འབབ་འཕར་ཚད་མཐོ་ཤོས་ཀྱི་ཟིན་ཐོ་བསྐྲུན་ཡོད་འདུག གློག་བརྙན་འདི་ནི་འཁྲབ་སྟོན་པ་ཟཱལ་མཱན་ཁཱང་གི་སྒོར་དུང་ཕྱུར་བརྒྱ་ཟིན་པའི་གློག་བརྙན་ཐེངས་ ༡༡ པ་དེ་ཆགས་ཡོད་འདུག གསར་འགོད་། སངས་རྒྱས་དཀོན་མཆོག
