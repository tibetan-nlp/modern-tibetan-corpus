ཉིན་རེར་མི་ ༤༠༠༠ ལྷག་རྐྱེན་འདས།

ཨ་རིའི་དངོས་ཁམས་རིག་པ་བས་མཁའ་དབུགས་འབོག་བཙོག་གི་རྐྱེན་གྱིས་རྒྱ་ནག་ནང་ཉིན་རེར་མི་གྲངས་ ༤༠༠༠ ལྷག་རེ་འདས་གྲོངས་སུ་འགྲོ་བཞིན་ཡོད་སྐོར་བསྒྲགས་འདུག ཨ་རིའི་ཁེ་ལི་ཧྥོར་ཎི་ཡའི་གཙུག་ལག་སློབ་ཆེན་གྱི་དངོས་ཁམས་རིག་པའི་སྡེ་ཚན་གྱིས་ཞིབ་འཇུག་བྱས་པ་ལྟར། རྒྱ་ནག་ནང་མཁའ་དབུགས་འབག་བཙོག་གིས་ཉིན་རེར་མི་གྲངས་ ༤༠༠༠ རེ་འཆི་ལམ་དུ་འདེད་བཞིན་ཡོད་པ་དང་། དེ་བཞིན་ལོ་རེར་མི་གྲངས་ས་ཡ་གཅིག་དང་འབུམ་དྲུག་ཙམ་སྙིང་ནད་དང་གློ་ནད་སོགས་ཀྱིས་འདས་གྲོངས་སུ་འགྲོ་བཞིན་ཡོད་པ་དང་། དེ་ནི་ཡིད་ཆེས་བྱེད་དཀའ་བའི་མཁའ་དབུགས་འབག་བཙོག་གི་གནོད་སྐྱོན་ལས་བྱུང་བ་ཞིག་ཡིན་ལ། དམིགས་བསལ་གྱི་མཁའ་དབུགས་འབག་བཙོག་ཁྲོད་དུད་སྨུག་རྡུལ་ཕྲན་གྱི་རྐྱེན་ལས་བྱུང་བ་ཞིག་ཡིན་འདུག སྔོན་ཆད་ཉམས་ཞིབ་པ་ཚོས་ལོ་ལྟར་རྒྱ་ནག་གི་མཁའ་དབུགས་འབག་བཙོག་གི་རྐྱེན་གྱིས་འདས་གྲོངས་སུ་འགྲོ་བའི་གྲངས་འབོར་ནི་ས་ཡ་གཅིག་ནས་གཉིས་པར་ཡིན་པའི་ཚོད་རྩིས་བྱས་ཡོད་ཀྱང་། ད་རེས་ཀྱི་སྙན་ཐོ་དེ་ཞིབ་འཇུག་གི་བསྡོམས་རྩིས་གསར་ཤོས་ཤིག་ཆགས་ཡོད་འདུག ཞིབ་འཇུག་གི་གཙོ་གཉེར་རྩོམ་པ་པོ་རོ་བྷོར་ཊི་རོ་ཌི་ (Robert Rohde) གྱིས། རྒྱ་ནག་གི་མི་འབོར་བརྒྱ་ཆ་ ༣༨ དུས་ཡིན་རིང་པོ་ཞིག་ལ་ཨ་རིའི་ཁོར་ཡུག་སྲུང་སྐྱོབ་ལས་ཁུངས་(Environmental Protection Agency) ཀྱིས་བརྗོད་པའི་བདེ་ཐང་མིན་པའི་མཁའ་དབུགས་ཞེས་པའི་ས་ཆུ་རུ་གནས་སྡོད་བྱེད་ཀྱི་ཡོད་ལ། དེ་ནི་བློ་ཡུལ་ལ་ཤོང་དཀའ་བའི་གྲངས་འབོར་ཧ་ཅང་མང་པོ་ཞིག་ཡིན་པ་མ་ཟད། རྒྱ་ནག་གི་ལྷོ་ཤར་ས་ཆ་ནི་མཁའ་དབུགས་འབག་བཙོག་གི་གནས་སྟངས་ཆེས་སྡུག་ཤོས་རེད། ཅེས་བརྗོད་འདུག ཡང་ཁོང་གིས། ཨ་རི་དང་རྒྱ་ནག་གཉིས་ཀྱི་མཁའ་དབུགས་འབག་བཙོག་གི་གནས་སྟངས་མ་འདྲ་བའི་རྒྱུ་མཚན་ནི། གནམ་གཤིས་ཀྱི་གནས་སྟངས་ཁྲོད་དགུན་དུས་སུ་རྒྱ་ནག་ནང་རྡོ་སོལ་ཤུགས་ཆེན་བེད་སྤྱོད་བྱས་པ་སོགས་ཀྱིས་མཁའ་དབུགས་འབག་བཙོག་སྡུག་དུ་འགྲོ་བཞིན་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༢༢ ལོའི་རྒྱ་ནག་དགུན་དུས་ཀྱི་ཨོ་རྩལ་འགྲན་ཚོགས་སྐབས་མཁའ་དབུགས་འབག་བཙོག་ཉུང་དུ་གཏང་རྒྱུར་དཀའ་ངལ་ཆེན་པོ་ཡོད་པ་རེད་ཅེས་སོགས་བརྗོད་འདུག སྤྱིར་འཛམ་གླིང་ནང་མཁའ་དབུགས་འབག་བཙོག་གིས་ལོ་རེར་མི་གྲངས་ས་ཡ་ ༣ ལྷག་རྐྱེན་འདས་སུ་འགྲོ་བཞིན་ཡོད་པ་དང་། དེའི་གྲངས་ཚད་ནི་གཉན་ནད་རག་དུག་དང་འདར་ནད། གཅིན་པ་མངར་འགྱུར་གྱི་ན་ཚ། གློ་གཅོང་གི་ན་ཚ་སོགས་ལས་ཀྱང་མང་བ་ཡོད་འདུག གསར་འགོད་པ། འཇིགས་མེད་རྒྱ་མཚོ།
