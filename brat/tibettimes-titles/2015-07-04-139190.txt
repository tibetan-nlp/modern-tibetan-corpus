རྒྱ་ནག་གི་བརྒྱུད་ལམ་གླེང་པ།

ཨ་རིའི་སྲིད་དོན་གླེང་སྟེགས་ཤིག་གི་སྟེང་ཞི་ཅིན་ཕིན་རྒྱ་ནག་གི་སྲིད་འཛིན་དུ་བསྡད་རྗེས་རྒྱ་ནག་གི་སྨྲ་བརྗོད་རང་དབང་དང་བརྒྱུད་ལམ་རང་དབང་ཐོག་དམ་དྲག་ཆེ་རུ་ཕྱིན་ཡོད་པའི་གླེང་སློང་བྱས་ཡོད་འདུག ཟླ་སྔོན་མའི་ཚེས་ ༢༥ ཉིན་ནས་དེ་རིང་ཕྱི་ཚེས་ ༤ ཉིན་བར་ཨ་རིའི་ཨ་སི་ཕེན་བསམ་བློའི་དུས་སྟོན་ (Aspen Ideas Festival)ཞེས་པ་བསྐོངས་ཚོགས་བྱས་ཡོད་པ་དང་། ཨ་རི་གཙོས་པའི་རྒྱལ་ཁབ་མང་པོ་ཞིག་གི་མི་སྣ་གྲགས་ཅན་ཕེབས་ཏེ་དེང་སྐབས་རྒྱལ་སྤྱིའི་ཁྱོན་ཡོངས་ཀྱིས་གདོང་ལེན་བྱེད་དགོས་པའི་གནད་དོན་ཁག་ཅིག་གི་ཐོག་བསམ་བློ་བརྗེ་རེས་བྱས་ཡོད་འདུག སྐབས་དེར་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་རྒྱ་ནག་གི་སྲིད་སྟེགས་སྟེང་བསླེབས་པ་དང་བསྟུན་ཏེ་ཁོ་རང་གི་འགལ་ཟླ་ཕྱོགས་ལ་རྒོལ་རྡུང་གཏོང་གི་ཡོད་ཀྱང་། ཕྱོགས་གཞན་ཞིག་ནས་རྒྱ་ནག་དྲ་རྒྱའི་འབྲེལ་མཐུད་འཁྲུག་ཆ་དོད་པོ་ཡོད་པ་དེས་རྒྱ་ནག་གི་སྨྲ་བརྗོད་རང་དབང་ལ་ཤུགས་རྐྱེན་ཐེབས་ཀྱི་ཡོད་དམ། ཞེས་པའི་བརྗོད་གཞི་ཞིག་གླེང་སློང་བྱུང་ཡོད་འདུག དེ་རིང་པའོ་ཞོན་དྲ་བས་གནས་ཚུལ་སྤེལ་བའི་ནང་། ནུབ་ཕྱོགས་རྒྱལ་ཁབ་ཚོས་ཞི་ཅིན་ཕིན་སྲིད་གཞུང་གིས་རྒྱ་ནག་གི་གསར་འགྱུར་རང་དབང་དང་སྨྲ་བརྗོད་རང་དབང་ལ་སྟངས་འཛིན་བྱེད་ཀྱིན་ཡོད་པར་ངོས་འཛིན་བྱེད་བཞིན་པ་དེར། ད་རེབ་བགྲོ་གླེང་སྟེང་རྒྱ་ནག་ནས་ཕེབས་པའི་ཚད་ཞིན་བརྒྱུད་ལམ་གྱི་རྩོམ་པ་པོ་མཱ་ཁི་ཀྲང་ (Mike Zhao) ལགས་ཀྱིས་བརྗོད་དོན་ “ཞི་ཅིན་ཕིན་སྲིད་སྟེགས་སུ་བསླེབས་རྗེས་རྒྱ་ནག་གི་འབྲེལ་མཐུད་དྲ་རྒྱའི་རང་དབང་དང་སྨྲ་བརྗོད་རང་དབང་། གསར་འགྱུར་རང་དབང་སོགས་ཀྱི་རྒྱ་ཁྱོན་ཆུང་དུ་འགྲོ་བཞིན་འདུག” ཅེས་དང་། “ཁྱེད་ཅག་ཨ་རི་བ་ཚོར་རང་དབང་གི་འདེམས་བསྐོ་དང་། གསར་འགྱུར་འབྱུང་ཁུངས་དངོས་ཀྱི་སྤྲོ་སེང་ཆེན་པོ་ཞིག་ཡོད་པ་རེད། འོན་ཀྱང་རྒྱ་ནག་ལ་འདི་འདྲ་མེད་སྟབས་ཁོ་ཚོར་མཚོན་ན་དྲ་རྒྱའི་བརྒྱུད་ལམ་ནི་ཡོད་ཚད་རེད།” ཅེས་བརྗོད་འདུག ཨ་རིའི་རྒྱ་ནག་ཆེད་མཁས་པ་དང་རྩོམ་པ་པོ་ལྕམ་ཨེ་མི་ལེ་ (Emily Parker) ཡིས་བརྗོད་དོན་ “རྒྱ་ནག་མི་དམངས་ཀྱིས་དྲ་བའི་བརྒྱུད་ལམ་བེད་སྤྱོད་བྱེད་རྒྱུ་ཤུགས་ཆེ་རུ་ཕྱིན་པ་དང་བསྟུན་ཏེ། དྲ་རྒྱ་བརྒྱུད་ནས་ཚོགས་པ་འདྲ་མིན་བཟོ་བཞིན་ཡོད་པ་དང་། དྲ་དམངས་དུང་ཕྱིར་ ༦ ལྷག་གིས་དྲ་རྒྱ་བེད་སྤྱོད་བྱེད་ཀྱི་ཡོ་སྟབས་རྒྱ་ནག་གཞུང་ལ་སེམས་ཁྲལ་ཡོང་གི་ཡོད། གལ་སྲིད་ཁྱོད་ཀྱིས་དྲ་རྒྱ་བརྒྱུད་ནས་མི་མང་པོ་ས་ཆ་གཞན་ཞིག་ཏུ་འདུ་འཛོམས་བྱ་རྒྱུ་བྱས་པ་ཡིན་ན། དེ་ནི་རྒྱ་ནག་གཞུང་གི་ཞེད་སྣང་དང་དཀའ་ངལ་ཆེ་ཤོས་ཤིག་ཆགས་ཡོད།” ཅེས་སོགས་བརྗོད་འདུག གསར་འགོད་པ། འཇིགས་མེད་རྒྱ་མཚོ།
