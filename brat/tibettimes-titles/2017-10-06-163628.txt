སྒོ་སྲུང་བས་བུ་གསོ་ཁང་མེར་བསྲེཊ་བཏང་།

བུ་གསོ་ཁང་མེར་བསྲེགས་བཏང་སྟེ་དགེ་སློབ་ལྔ་འདས་གྲོངས་སུ་ཕྱིན་པ་དང་མི་ ༢༥ ལ་སྨན་བཅོས་བྱེད་བཞིན་ཡོད་འདུག   དེ་རིང་བྷི་བྷི་སི་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། བྷར་ཛིལ་ (Brazil) རྒྱལ་ཁབ་ཀྱི་མངའ་སྡེ་མཱི་ན་ས་གྷི་ར་ཡེ་སི་ (Minas Gerais)་ས་གནས་ཇཱ་ནུའུ་བྷ་ (Janauba)ཡི་བུ་གསོ་ཁང་གི་སྒོ་སྲུང་བས་སློབ་གྲྭ་མེར་བསྲེགས་བཏང་སྟེ་ཕྲུ་གུ་བཞི་དང་དགེ་རྒན་གཅིག་གི་སྲོག་ཤོར་བ་མ་ཟད། ཚིག་སྐྱོན་བྱུང་བའི་དགེ་ལས་སློབ་ཕྲུག་ ༢༥ ལ་ཉེ་འཁོར་གྱི་སྨན་ཁང་དུ་སྨན་བཅོས་བྱེད་བཞིན་ཡོད་པ་དང་།  དེའི་ནང་ནས་མང་ཆེ་བ་རང་ལོ་ ༤ དང་ ༥ ཅན་གྱི་ཕྲུ་གུ་ཡིན་ལ། རྨས་སྐྱོན་ཕོག་པ་ཁ་ཤས་ཀྱི་གནས་སྟངས་ཛ་དྲག་ཆགས་ཡོད་འདུག བུ་གསོ་ཁང་ལ་མེ་འབར་སྐབས་ཕྱི་རོལ་དུ་ཕྲུ་གུ་ཚོའི་ཕ་མ་རྣམས་དངངས་སྐྲག་ཆེན་པོས་འོ་དོད་བྱེད་བཞིན་འཚང་ཁ་བརྒྱབས་པ་སོགས་གནས་སྟངས་ཟིང་ཆ་ཆེན་པོར་གྱུར་ཡོད་པ་དང་། ས་གནས་བརྒྱུད་ལམ་གྱི་གནས་ཚུལ་ལྟར་ན། བུ་གསོ་ཁང་དེར་ཁྱོན་ཕྲུ་གུ་ ༧༠ ཙམ་ཡོད་པའི་ནང་ནས་ཕྲུ་གུ་ ༤ ཡི་ཚེ་སྲོག་ཤོར་བ་མ་གཏོགས་གཞན་རྣམས་སྐབས་ཐོག་དེར་སྐྱོབ་ཐུབ་ཡོད་འདུག  ཉེན་རྟོག་པས་བརྗོད་པ་ལྟར་ན། སྒོ་སྲུང་བ་དེའི་མིང་ལ་ཌ་མི་ཨོ་སཱོ་རིས་ཌོས་སན་ཏོས་ (Damiao Soares dos Santos) ཟེར་བ་དང་། ད་ལྟ་རང་ལོ་ ༥༠ ཡིན་འདུག ཁོས་བུ་གསོ་ཁང་མེར་བསྲེགས་བཏང་རྗེས་ཕྱི་རོལ་དུ་འབུད་ཐབས་བྱས་ཡོད་ཀྱང་། ཟིང་ཆའི་ཁྲོད་ཕྱིར་འབུད་མ་ཐུབ་པར་ཁོ་རང་ལའང་ཚིག་སྐྱོན་ཆེན་པོ་བྱུང་ཡོད་པ་དང་། ཕྱིས་སུ་སྨན་ཁང་དུ་བསྐྱལ་ཡོད་ཀྱང་འདས་གྲོངས་སུ་ཕྱིན་ཡོད་འདུག ད་ལྟའི་ཆར་སྒོ་སྲུང་བ་དེས་བུ་གསོ་ཁང་མེར་བསྲེགས་བཏང་དགོས་པའི་རྒྱུ་རྐྱེན་གང་ཡིན་ཞིབ་འཇུག་བྱེད་མུས་ཡིན་འདུག   ཡུལ་དེའི་གསར་འགྱུར་བརྒྱུད་ལམ་ཁག་ཏུ་གནས་ཚུལ་བཀོད་པ་ལྟར་ན། སྒོ་སྲུང་བས་རང་གི་ལས་ཀ་ནས་ལོ་གཅིག་གི་ཐོབ་ཐང་གུང་སེང་བླངས་ཏེ་ཁྱིམ་དུ་ཕྱིན་རྗེས། ཡང་བསྐྱར་བུ་གསོ་ཁང་དུ་ལོག་སྐབས་འབྲེལ་ཡོད་ལས་བྱེད་ཚོས་ཁོའི་གཟུགས་པོའི་འཕྲོད་བསྟེན་ཞན་པོ་ཡོད་ཚུལ་བརྗོད་དེ་ལས་ཀ་ནས་ཕྱིར་འཕུད་བྱས་ཡོད་སྟབས། ཁོ་རང་ཁོང་ཁྲོ་ཆེན་པོ་ལངས་ཏེ་བུ་གསོ་ཁང་མེར་བསྲེགས་བཏང་ཚུལ་བརྗོད་འདུག      དོན་རྐྱེན་དེ་བྱུང་རྗེས་བྷར་ཛིལ་གྱི་སྲིད་འཛིན་མའེ་ཀཱལ་ཏི་མིར་ (Michel Temer) ལགས་ཀྱིས། བུ་གསོ་ཁང་ལ་དེ་འདྲའི་ཡིད་སྐྱོ་བའི་གནས་ཚུལ་ཞིག་བྱུང་བར་ང་རང་བློ་ཕམ་ཆེན་པོ་ཡོད་ལ། དེ་ནི་ང་ཚོའི་ཕམ་ཁ་ཞིག་རེད། དེར་བརྟེན། ངས་ཤུལ་ལུས་ཕ་མ་དང་ནང་མི་ཚོར་མྱ་ངན་གུས་འདུད་ཞུ་ཡི་ཡོད། ཅེས་བརྗོད་ཡོད་པ་དང་། ཇ་ནུའུ་བྷ་ས་གནས་གཞུང་གིས་ཡིད་སྐྱོ་བའི་དོན་རྐྱེན་དེར་ཉིན་ ༧ རིང་མྱ་ངན་གུས་འདུད་ཞུ་རྒྱུའི་བཀའ་ཁྱབ་བཏང་ཡོད་པ་རེད། གསར་འགོད་པ། ཉི་མ་ཚེ་རིང་།
