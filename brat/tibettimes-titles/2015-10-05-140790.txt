རྔ་པའི་ཞི་རྒོལ་བ་གསུམ་བཙན་ཁྲིད་བྱས།

  ཉེ་སྔོན་རྔ་པ་ཀིརྟི་དགོན་གྱི་གྲྭ་བློ་བཟང་འཇམ་དབྱངས་དང་མི་སྐྱ་གཉིས་ཀྱིས་རྔ་པ་རྫོང་མཁར་དུ་ཞི་རྒོལ་སྐད་འབོད་གནང་སྟེ་ཉེན་རྟོག་པས་བཙན་ཁྲིད་བྱས་ཡོད་འདུག   དེ་རིང་བཞུགས་སྒར་ཀིརྟིའི་བྱེས་པ་གྲྭ་ཚང་གི་བོད་གནས་འབྲེལ་མཐུད་པ་བློ་བཟང་ཡེ་ཤེས་ལགས་དང་ཀ་ཉག་ཚེ་རིང་ལགས་གཉིས་ཀྱིས་འདི་གར་གནས་ཚུལ་མཁོ་གཏོང་གནང་བ་ལྟར་ན། ཕྱི་ཟླ་༩ ཚེས་ ༢༣ ཉིན་གྱི་བོད་ནང་གི་ཕྱི་དྲོའི་ཆུ་ཚོད་བཞི་པ་ཙམ་ལ། རང་ལོ་ ༡༥ ཡིན་པ་རྔ་པ་ཀིརྟི་དགོན་པའི་གྲྭ་བློ་བཟང་འཇམ་དབྱངས་ལགས་ཀྱིས། དཔའ་བོའི་སྲང་ལམ་ནས་རྔ་པ་རྫོང་མཁར་གྱི་གཞུང་ལམ་བར་གོམ་བགྲོད་ཀྱིས་”བོད་ལ་རང་དབང་དགོས། རྒྱལ་བ་རིན་པོ་ཆེ་བོད་ལ་གདན་འདྲེན་ཞུ་དགོས།” ཞེས་སྐད་འབོད་གནང་ནས་རྒྱ་ནག་གཞུང་ལ་ཞི་བའི་ངོ་རྒོལ་གནང་ཡོད་པ་དང་། དེ་མ་ཐག་དེར་ཡོད་ཉེན་རྟོག་པ་རྣམས་ཀྱིས་ཁོང་འཛིན་བཟུང་བཙན་ཁྲིད་བྱས་ཡོད་སྟབས། ད་ལྟའི་འཆར་ཁོང་བཀག་ཉར་བྱས་ཡོད་པ་ཙམ་ལས་གནས་ཚུལ་གཞན་གང་ཡང་ཞིབ་ཕྲ་ཤེས་རྟོགས་མ་ཐུབ་པར་ལྷག་ཡོད་འདུག   གྲྭ་བློ་བཟང་འཇམ་དབྱངས་ལགས་ནི་རྔ་པ་རྫོང་རྨེའུ་རུ་མ་ཡུལ་ཚོའི་རུ་ཆེན་གཉིས་པའི་བ་ཆོས་ཚང་གི་ཕ་ཆོས་འཕེལ་དང་མ་མཚོ་མོ་གཉིས་ཀྱི་བུ་ཡིན་ཞིང་། ཆུང་ཆུང་སྐབས་ནས་རྔ་པ་ཀིརྟི་དགོན་པར་ཆོས་ཞུགས་བྱས་ཏེ་དེང་སྐབས་སྔོན་འགྲོའི་འཛིན་གྲྭར་སློབ་གཉེར་གནང་བཞིན་ཡོད་འདུག   གཞན་ཡང་། ཕྱི་ཟླ་༩ ཚེས་༡༠ ཉིན་གྱི་བོད་ནང་གི་ཉིན་གུང་ཆུ་ཚོད་༡༢ ཙམ་ལ་རྔ་པ་རྫོང་སོ་རུ་སྡེ་བའི་ཚུར་རིས་ཚང་གི་བུ་འཕྲིན་ལས་ལགས་དང་། ཚེ་སྟོད་ཚང་གི་བུ་བློ་སངས་ལགས་གཉིས་ཀྱིས་”བོད་ལ་རང་དབང་དགོས། རྒྱལ་བ་རིན་པོ་ཆེ་སྐུ་ཚེ་ཁྲི་ལོ་བརྟན་པར་ཤོག” ཅེས་ངོ་རྒོལ་སྐད་འབོད་གནང་བཞིན་རྔ་པ་རྫོང་མཁར་གྱི་གཞུང་ལམ་དུ་གོམ་བགྲོད་བྱས་ཡོད་པ་དང་། དེ་མ་ཐག་ཉེན་རྟོག་པས་འཛིན་བཟུང་བཙན་ཁྲིད་ཀྱིས་ད་བར་གནས་ཚུལ་ཅི་ཡང་ཤེས་རྟོགས་མེད་པར་གྱུར་ཡོད་པ་དང་། ཁོང་གཉིས་ནི་རྔ་པ་རྫོང་ཆོས་རྗེ་མ་ཡུལ་ཚོའི་སོ་རུ་མ་སྡེ་བའི་ཡིན་པ་ཙམ་ལས་ཞིབ་ཕྲ་ཤེས་རྟོགས་ཐུབ་ཀྱི་མེད་པ་རེད།   ཕྱི་ཟླ་༩ ཚེས་༡༠ ཉིན་ནས་བཟུང་གཞུང་འབྲེལ་གྱི་ལས་ཁུངས་ཁག་ལས་གཞན་པའི་མི་སྒེར་གྱི་དྲ་ལམ་ཡོངས་རྫོགས་བཀག་སྡོམ་བྱས་ཡོད་པ་མ་ཟད། རྫོང་མཁར་གྱི་གཞུང་ལམ་ཁག་ལ་ཉེན་རྟོག་པ་མང་རུ་བཏང་ནས་རྒྱུན་དུ་ལྟ་རྟོག་བྱས་ཏེ་དམ་དྲག་ཆེན་པོ་བྱེད་བཞིན་ཡོད་འདུག   གསར་འགོད་པ། པདྨ་མཚོ།
