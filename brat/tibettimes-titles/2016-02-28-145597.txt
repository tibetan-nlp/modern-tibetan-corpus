རྩོམ་པ་པོ་ཞིག་བསྐྱར་དུ་འཛིན་བཟུང་བྱས།

ཉེ་སྔོན་རྒྱ་གཞུང་ཉེན་རྟོག་པས་བོད་ཀྱི་ན་གཞོན་རྩོམ་པ་པོ་བསོད་ནམས་ཚེ་འཕེལ་ལམ་གངས་ཅན་པ་གུ་དྲག་ལགས་ཡང་བསྐྱར་འཛིན་བཟུང་བྱས་ཡོད་འདུག   འདི་གར་དགེ་འདུན་པ་བྱང་ཆུབ་སེམས་པ་ལགས་ཀྱིས་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། གཞོན་སྐྱེས་རྩོམ་པ་པོ་བསོད་ནམས་ཚེ་འཕེལ་ལམ་སྨྱུག་མིང་གངས་ཅན་པ་གུ་དྲག་ལགས། ཕྱི་ཟླ་ ༢ ཚེས་ ༡༩ ཉིན་བོད་ཀྱི་རྒྱལ་ས་ལྷ་ས་རུ་ཉེན་རྟོག་པས་ཡང་བསྐྱར་འཛིན་བཟུང་བཙན་འཁྲིད་བྱས་ཡོད་པ་དང་། ད་ལྟའི་ཆར་ཁོང་བཀག་ཉར་འོག་ཡོད་པ་དང་ཡང་བསྐྱར་འཛིན་བཟུང་བྱེད་དགོས་པའི་རྒྱུ་མཚན་སྐོར་ཅི་ཡང་ཤེས་རྟོགས་ཐུབ་ཀྱི་མེད་པ་རེད།   གཞོན་སྐྱེས་རྩོམ་པ་པོ་བསོད་ནམས་ཚེ་འཕེལ་ལགས་ཚ་བ་དཔའ་ཤོད་རྫོང་ནས་ཡིན་ཞིང་། དེ་སྔོན་ལ་སྟོད་དིང་རི་རྫོང་ཤེལ་དཀར་གྲོང་རྡལ་དུ་ཚོང་ལས་གཉེར་ཏེ་བསྡད་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༠༩ ལོར་གཞིས་ཀ་རྩེ་ནས་རྒྱ་གཞུང་ཉེན་རྟོག་པས་ཁོང་འཛིན་བཟུང་བྱས་ཏེ་ཟླ་ ༩ ལྷག་རིང་ཉ་རི་བཙོན་ཁང་དུ་བཀག་ཉར་བྱས་ཡོད་པ་དང་། ཕྱིས་སུ་རྒྱལ་ཁབ་ཁ་བྲལ་གཏོང་བར་ངན་སྐུལ་བྱས་པའི་ཉེས་མིང་འོག ལོ་གཉིས་ཀྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་དེ་ངལ་རྩོལ་བསྒྱུར་བཀོད་ཁང་ཞིག་ཏུ་བཅུག་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༡༢ ལོའི་ལོ་སྟོད་ཙམ་ལ་གཞི་ནས་ཀློད་གྲོལ་ཐོབ་ཡོད་འདུག ཕྱི་ལོ་ ༢༠༡༥ ལོའི་ཕྱི་ཟླ་ ༦ ཚེས་ ༢༩ ཉིན་གྱི་བོད་ནང་གི་ཕྱི་དྲོའི་ཆུ་ཚོད་ ༦ པ་ཙམ་ལ་ཡང་བསྐྱར་རྒྱ་གཞུང་ཉེན་རྟོག་པས་ཁོང་གློ་བུར་དུ་འཛིན་བཟུང་བྱས་ཡོད་པ་མ་ཟད། མི་རིགས་བར་དཀྲུག་ཤིང་འགལ་ཟླ་སློང་བའི་རྩོམ་ཡིག་བྲིས་པ་དང་། ༧གོང་ས་མཆོག་དང་སྲིད་སྐྱོང་བློ་བཟང་སེང་གེ་མཆོག་སོགས་ཀྱི་གསུང་བཤད་ཉར་ཚགས་བྱས་པ། ༧གོང་ས་མཆོག་གི་སྐུ་པར་གང་སར་སྤྲད་ནས་སྤྱི་ཚོགས་ཀྱི་བདེ་འཇགས་ལ་གནོད་སྐྱོན་བཟོས་པ་སོགས་ཀྱི་ཉེས་མིང་འོག དུས་ཡུན་ཉིན་ ༢༠ རིང་གཞིས་ཀ་རྩེའི་ཉ་རི་བཙོན་ཁང་དུ་བཀག་ཉར་བྱས་ཡོད་པའི་ཁར། རྒྱ་སྒོར་ ༡༥༠༠ ཡིས་ཉེས་ཆད་བཅད་དེ་འདས་ལོའི་ཕྱི་ཟླ་ ༧ ཚེས་ ༢༤ ཉིན་ཀློད་གྲོལ་བཏང་ཡོད་ཀྱང་། རྒྱ་གཞུང་ཉེན་རྟོག་པ་དང་ས་གནས་ལས་བྱེད་མི་སྣས་མུ་མཐུད་ལྟ་ཞིབ་ནན་པོ་བྱས་ཡོད་པ་རེད། གསར་འགོད་པ། པདྨ་མཚོ།
