གྲྭ་བ་ཞིག་ལ་ལོ་ ༡༢ ཀྱི་ཁྲིམས་ཐག་བཅད།

སོག་རྫོང་ཁྲི་རུ་དྲིལ་མདའ་དགོན་པའི་གྲྭ་ཚངས་དབྱངས་རྒྱ་མཚོ་ལགས་སུ་རྒྱ་གཞུང་གིས་རྒྱ་ནག་ལ་ངོ་རྒོལ་སྐུལ་སློང་བྱས་པའི་ཉེས་མིང་འོག་ལོ་ ༡༢ ཀྱི་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག ཉེ་ལམ་གསར་དུ་ཤེས་རྟོགས་བྱུང་བ་ལྟར་ན། ཟླ་འདིའི་ཚེས་ ༡ ཡས་མས་ཤིག་ལ་སོག་རྫོང་ཁྲི་རུ་དྲིལ་མདའ་དགོན་པའི་དབུ་མཛད་ཚངས་དབྱངས་རྒྱ་མཚོ་ལགས་སུ་རྒྱ་གཞུང་གིས་ལོ་ ༡༢ གྱི་ཁྲིམས་ཐག་བཅད་དེ། ད་ལྟ་ལྷ་ས་ཆུ་ཤུལ་བཙོན་ཁང་དུ་བཅུག་ཡོད་པ་འབྲེལ་ཚོས་ཤེས་རྟོགས་བྱུང་ཡོད་པ་དང་། སྐབས་དེར་སོག་རྫོང་ཁྲི་རྡོ་ཡུལ་ཚོ་ལ་ཡོད་པའི་ནང་མི་ལ་ཁྲིམས་ཐག་བཅད་པའི་ཡི་གེ་ཞིག་འབྱོར་ཡོད་ཡོད་པའི་ནང་། གྲྭ་ཚངས་དབྱངས་རྒྱ་མཚོ་ལ་ཁྲིམས་ཐག་བཅད་ནས་ཟླ་གསུམ་མ་སོང་བར་ཁོང་ལ་བཙོན་དུ་ཐུག་འཕྲད་བྱེད་དུ་འགྲོ་མི་ཆོག་པ་དང་། གལ་སྲིད་ཟླ་གསུམ་རྗེས་ཐུག་འཕྲད་བྱེད་མཁན་ཡོད་ཚེ་ཐོབ་ཐང་ལག་ཁྱེར་ཡོད་པའི་ཐོག་དང་པོ་ཁྲི་རྡོ་ཡུལ་ཚོ་དང་། སོག་རྫོང་བཅས་ནས་རིམ་པ་བཞིན་འགན་ལེན་ཡི་གེ་ལེན་དགོས་སྐོར་འཁོད་སོགས་འཁོད་ཡོད་འདུག རྒྱ་གཞུང་གིས་ཚངས་དབྱངས་རྒྱ་མཚོ་ལ་ཕྱི་ཕྱོགས་སུ་འབྲེལ་བ་བྱས་པ་དང་དགེ་འདུན་པ་གཞན་ཚོར་རྒྱ་ནག་ལ་ངོ་རྒོལ་བྱ་དགོས་པའི་སྐུལ་སློང་བྱས་ཡོད་པའི་ཉེས་འཛུགས་བྱས་ཡོད་འདུག ད་ལོའི་ཕྱི་ཟླ་ ༣ ཚེས་ ༡༧ གྱི་མཚན་མོ་རྒྱ་གཞུང་གིས་སོག་རྫོང་ཁྲི་རུ་དྲིལ་མདའ་དགོན་པའི་གྲྭ་ཚངས་དབྱངས་རྒྱ་མཚོ་ལགས་དང་ཚེ་དབང་། ཨ་ཚེ། རྒྱལ་མཚན་བཅས་གྲྭ་པ་ ༤ འཛིན་བཟུང་བཙོན་འཇུག་བྱས་ནས་ད་བར་ནང་མི་རྣམས་ནས་ཁོང་རྣམས་གར་ཡོད་ཀྱི་གནས་ཚུལ་ཤེས་རྟོགས་ཐུབ་མེད།
