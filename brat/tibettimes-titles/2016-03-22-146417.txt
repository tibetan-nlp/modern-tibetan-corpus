བོད་མི་ལྷ་སར་འགྲོ་བར་བཀག་འགོག

རྒྱ་གཞུང་གིས་མདོ་སྟོད་སྨད་ཀྱི་བོད་རིགས་ལྷ་སར་འགྲོ་མཁན་ཚོར་བཀག་འགོག་གིས་གནམ་གྲུ་དང་མེ་འཁོར་གྱི་འཛིན་བྱང་འཚོང་གི་མེད་པ་རེད།   འདི་གའི་གསར་འགོད་པར་ཨོ་སི་ཊོ་ལི་ཡར་གནས་སྡོད་བོད་ཀྱི་ཆབ་སྲིད་བཙོན་ཟུར་ཤེལ་དགེ་ལགས་ཀྱིས། གསུམ་བཅུའི་དུས་དྲན་གྱི་སྔ་གཞུག་སྟེ་འདི་ལོའི་ཕྱི་ཟླ་ ༣ པའི་ཚེས་ ༡ ཙམ་ནས་བཟུང་ཕྱི་ཟླ་ ༤ ཚེས་ ༡༢ བར་དུ། མདོ་སྟོད་སྨད་ནས་ཡིན་པའི་བོད་རིགས་ལྷ་སར་འགྲོ་མཁན་ཚོར་རྒྱ་གཞུང་གིས་བཀག་འགོག་བྱས་ཏེ། ཟི་ལིང་དང་ལན་ཀྲུའུ་སོགས་ནས་མེ་འཁོར་དང་གནམ་གྲུའི་འཛིན་བྱང་གཅིག་ཀྱང་འཚོང་གི་མེད་པ་མ་ཟད། ཁྲིན་ཏུའུ་ནས་ལྷ་སར་འགྲོ་མཁན་བོད་རིགས་ཚོར་ཕྱི་ཟླ་ ༤ པའི་ཟླ་མཇུག་བར་དུ་མ་གཏོགས་གནམ་གྲུའི་འཛིན་བྱང་གཅིག་ཀྱང་འཚོང་གི་མི་འདུག་ལབ་བྱུང་། ཞེས་དང་།   ཉིན་གཉིས་གསུམ་གྱི་གོང་དུ་ངའི་ཕ་ཡུལ་གྱི་མི་ཁ་ཤས་ཤིག་ལྷ་སར་འགྲོ་རྩིས་ཀྱིས་ལན་ཀྲུའུ་ཡི་མེ་འཁོར་ས་ཚིགས་སུ་ཕྱིན་པ་དང་། མི་མང་པོར་ལྷ་སར་འགྲོ་བའི་འཛིན་བྱང་འཚོང་གིན་ཡོད་ཀྱང་། མེ་འཁོར་འཛིན་བྱང་འཚོང་མཁན་གྱིས་ཁོང་ཚོར་ཐོབ་ཐང་ལག་ཁྱེར་འཁུར་ཤོག་ལབ་སྟེ་ཐོབ་ཐང་ལག་ཁྱེར་སྟོན་སྐབས། ཁྱེད་ཚོ་བོད་པ་རེད་འདུག ལྷ་སར་འགྲོ་བའི་འཛིན་བྱང་འཚོང་གི་མ་རེད་ཅེས་བརྗོད་དེ་བཙོང་མི་འདུག་ལ། ལན་ཀྲུའུ་ཙམ་མིན་པར་ཟི་ལིང་དང་ཁྲིན་ཏུའུ་སོགས་ཀྱི་མེ་འཁོར་ས་ཚིགས་དང་གནམ་གྲུའི་འཛིན་བྱང་ཉོ་ས་ནས་ཀྱང་བོད་རིགས་ཚོར་ལྷ་སར་འགྲོ་བའི་འཛིན་བྱང་གཅིག་ཀྱང་འཚོང་གི་མེད་པ་སོགས་དམ་དྲག་ཆེན་པོ་བྱེད་བཞིན་ཡོད་འདུག མེ་འཁོར་དང་གནམ་གྲུའི་ས་ཚིགས་སོགས་ས་ཚིགས་ཆེན་པོ་དེ་འདྲ་ལའང་མི་རིགས་དབྱེ་འབྱེད་ཀྱི་བྱ་སྤྱོད་སྤེལ་བཞིན་ཡོད་འདུག ཅེས་བརྗོད་བྱུང་།   ཡང་ཁོང་གིས། སུམ་བཅུའི་དུས་དྲན་གྱི་སྔ་གཞུག་ཏུ་ལྷ་སར་ཡོད་པའི་མདོ་སྟོད་སྨད་ཀྱི་གནས་མཇལ་བ་དང་ཚོང་པ་སོགས་བོད་མི་མང་པོ་རང་ཡུལ་དུ་བཙན་གྱིས་ཕྱིར་ལོག་བཏང་ཡོད་པ་མ་ཟད། ཉིན་གྲངས་མང་པོར་མི་ཧ་ཅང་མང་པོ་བཀག་ཉར་བྱས་ཡོད་པ་དང་། ཕྱི་ཕྱོགས་ཀྱི་འབྲེལ་ལམ་གཅོད་ཆེད་མདོ་སྟོད་སྨད་ཕྱོགས་ཀྱི་ས་ཁུལ་ཧ་ཅང་མང་པོར་སྐད་འཕྲིན་སོགས་དྲ་ལམ་ཚང་མ་རྦད་དེ་སྒོ་བརྒྱབ་འདུག ཅེས་བརྗོད་བྱུང་།   གསར་འགོད་པ། པདྨ་མཚོ།
