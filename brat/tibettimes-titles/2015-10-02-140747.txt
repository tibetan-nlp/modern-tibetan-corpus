གྷན་དྷིའི་འཁྲུངས་སྐར་སྲུང་གཙི་ཞུས།

  བོད་མིའི་སྒྲིག་འཛུགས་བཀའ་ཤག་གིས་རྒྱ་གར་གྱི་མེས་པོ་དམ་པ་གྷན་དྷི་མཆོག་གི་འཁྲུངས་སྐར་སྲུང་གཙི་ཞུས་སོང་།   དེ་རིང་སྔ་དྲོའི་ཆུ་ཚོད་ ༩ བའི་སྟེང་། བོད་མིའི་སྒྲིག་འཛུགས་བཀའ་ཤག་མདུན་ཐང་དུ་སྲིད་ཚབ་ཆོས་རིག་བཀའ་བློན་པདྨ་ཆོས་འབྱོར་མཆོག་གི་གཙོས་བོད་མིའི་སྒྲིག་འཛུགས་ལྷན་ཁང་ཁག་གི་བཀའ་བློན་རྣམ་པ་དང་དང་དྲུང་ཆེ་རྣམ་པ། བཀའ་དྲུང་ཡིག་ཚང་གི་སྲི་ཞུའི་ལས་བྱེད་བཅས་འཛོམས་ཏེ། རྒྱ་གར་གྱི་རང་བཙན་འཐབ་རྩོད་པའམ་རྒྱལ་ཁབ་ཀྱི་ཡབ་ཆེན་དམ་པ་གྷན་དྷི་མཆོག་གི་སྐུའི་འཁྲུངས་སྐར་སྲུང་གཙི་ཞུས་སོང་། མཛད་སྒོའི་ཐོག་མར་སྲིད་ཚབ་ཆོས་རིག་བཀའ་བློན་མཆོག་གིས་རྒྱ་གར་རྒྱལ་དར་མཁའ་དབྱིངས་སུ་བསྒྲེངས་བ་དང་བསྟུན་རྒྱ་གར་གྱི་རྒྱལ་གླུ་ཕུལ་སོང་། དེའི་རྗེས་གྷན་དྷི་མཆོག་གི་སྐུ་པར་ལ་མཇལ་དར་དང་མཆོད་མེ། བདུག་སྤོས་སོགས་ཕུལ་ཏེ་རྗེས་དྲན་ཞུས་སོང་།   གྷན་དྷི་མཆོག་གི་སྐུའི་འཁྲུངས་སྐར་སྲུང་གཙིའི་སྐོར་ལ་སྲིད་ཚབ་ཆོས་རིག་བཀའ་བློན་པདྨ་ཆོས་འབྱོར་མཆོག་གིས། གྷན་དྷི་ཇི་རྒྱལ་ཁབ་ཀྱི་ཡབ་ཆེན་ (the father of nation) རེད། ཁོང་འཁྲུངས་པའི་སྐབས་ལ་རྒྱ་གར་དབྱིན་ཇིས་དབང་སྒྱུར་ནས་དཀའ་ངལ་འོག་གནས་ཡོད་པ་རེད། རྗེས་སུ་ཁོང་དབྱིན་ཇིའི་ལུང་པར་ཕྱིན་ཏེ་ཚུར་ཡོང་ནས་རྒྱ་གར་དབང་འཛིན་གྱི་འགན་ཁུར་ཞིང་། འཚེ་མེད་ཞི་བའི་ལྟ་གྲུབ་སྟེང་ལས་འགུལ་སྤེལ་པ་མ་ཟད། ཁོང་གིས་མི་ཚེ་དཀྱུས་མ་ཞིག་རོལ་ནས་རྒྱལ་ཁབ་ཆེད་བློས་གཏོང་མང་པོ་བྱས་ཤིང་། བཙོན་ཁང་སོགས་ལ་ཐེངས་མང་ཕྱིན་ཏེ་དཀའ་ངལ་ཚད་མེད་མྱངས་རྗེས་རྒྱ་གར་ལ་རང་བཙན་རག་པ་རེད། ཡིན་ན་ཡང་། ང་ཚོ་གྷན་དྷི་མཆོག་དང་ཏོག་ཙམ་འདྲ་གི་མ་རེད། གྷན་དྷི་ཇི་ཁོ་རང་གི་ལུང་པར་ཡོད་པ་རེད། སྐབས་དེར་དབྱིན་ཇི་རྒྱ་གར་ནང་ལ་ཉུང་ཉུང་ཞིག་མ་གཏོགས་མེད་སྟབས། སོ་སོའི་ལུང་པའི་ནང་ཁོང་རང་གི་མི་རིགས་ཀྱིས་ལས་འགུལ་སྤེལ་རྒྱུ་ལས་སླ་བ་དང་ལས་དོན་གྲུབ་པ་རེད། ད་ལྟ་ང་ཚོར་གྷན་དྷི་ནང་བཞིན་གྱི་ལས་རྒྱུ་མེད་ཀྱང་། བོད་པ་ཚོས་དཔེ་མཚོན་ཞུ་རྒྱུ་མང་པོ་ཡོད་པ་རེད། ང་ཚོའི་འཐབ་རྩོད་ཀྱི་གནས་སྟངས་དེ་གྷན་དྷི་མཆོག་གི་ཨ་ཧིམ་ས་དང་འདྲ་བོ་རེད། མི་གཞན་ལ་གནོད་འཚེ་མེད་པའི་སྟེང་རང་གི་བདེན་པ་འཐབ་རྩོད་བྱེད་པ་དེ་དཔེ་བསྟན་གནང་ཡོད་སྟབས། ༧ གོང་ས་མཆོག་གིས་དུས་རྒྱུན་གྷན་དྷི་ཇི་ཡི་ཨ་ཧིམ་ས་དེ་ང་ཚོས་གལ་ཆེན་པོ་རྩི་བཞིན་ཡོད་ཅེས་གསུང་གི་ཡོད་པ་རེད། ཅེས་གསུངས་སོང་།   གསར་འགོད་པ། འཇིགས་མེད་རྒྱ་མཚོ།
