རྒྱ་གར་རང་བཙན་ཉིན་མོའི་རྣམ་པ་གསར་བ།

ད་ལོའི་རྒྱ་གར་རང་བཙན་གྱི་ཉིན་མོར་རྒྱ་གར་ལྷོ་ཕྱོགས་བྷིང་ལོར་ཁུལ་དུ་གནས་སྡོད་བོད་པའི་སློབ་ཕྲུག་ཁག་ཅིག་གིས་དང་བླངས་ཐོག་ཁོར་ཡུག་གཙང་བཟོའི་ལས་འགུལ་ཞིག་སྤེལ་བར་མི་མང་གིས་དོ་སྣང་ཆེན་པོ་བྱུང་ཡོད་འདུག ཟླ་འདིའི་ཚེས་ ༡༥ ཉིན་རྒྱ་གར་རང་བཙན་དུས་དྲན་ཐེངས་ ༧༠ བའི་ཉིན་མོ་དང་བསྟུན་ཏེ། རྒྱ་གར་ལྷོ་ཕྱོགས་བྷིང་ལོར་ས་གནས་བོོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་ཀྱི་གོ་སྒྲིག་འོག རྒྱ་གར་གཞུང་དམངས་ཡོངས་ལ་ཐུགས་རྗེ་ཆེ་མཚོན་ཕྱིར། མཐོ་སློབ་འགྲིམ་བཞིན་པའི་སློབ་ཕྲུག་ ༣༠ ལྷག་གིས། བྷིང་ལོར་གྲོང་ཁྱེར་གྱི་ཁྲོམ་ལམ་སྦད་ཀན་རོལ་ཏྲི་སོགས་མི་འདུ་འཛོམས་ཆེ་ས་དང་། རྒྱལ་སའི་དབུས་ཁུལ་དག་ལ་འཁོར་ཡུག་གཙང་མ་བཟོ་བའི་ལས་འགུལ་ཞིག་སྤེལ་ཡོད་འདུག ཁོར་ཡུག་གཙང་བཟོའི་ལས་འགུལ་དེ་ནི། འདས་པའི་ལོ་ངོ་ལྔ་བཅུ་ལྷག་རིང་རྒྱ་གར་གཞུང་དམངས་གཉིས་ཀྱིས་བོོད་པ་སྐྱབས་བཅོལ་བ་ཚོར་གནས་ཚང་གཡར་བ་ཙམ་མ་ཟད། ཐོག་མ་དཀའ་ངལ་དང་ཛ་དྲག་གི་སྐབས་བོད་པ་ཚོ་དཀའ་སྡུག་ལས་བསྐྱབས་ཏེ། ད་ལྟ་སྐྱབས་བཅོལ་གཞན་དང་མི་འདྲ་བའི་མཐུན་རྐྱེན་འཛོམ་པོ་འདི་འདྲ་གནང་བ་དང་། གཉིས་ནས་རྒྱ་གར་གྱི་རྒྱལ་ཁབ་ཡར་རྒྱས་འགྲོ་བཞིན་པའི་གནས་སྐབས་འདི་ལ་དཀའ་ངལ་ཆེ་ཤོས་ནི་ཁོར་ཡུག་གཙང་སྦྲ་དེ་ཆགས་ཡོད་སྟབས། རྒྱ་གར་མི་དམངས་ལ་ཚོར་སྣང་གསར་བ་ཞིག་དང་། མཛའ་སེམས་ཀྱི་བརྡ་ལན་གསལ་བོ་ཞིག་གཏོང་བའི་སླད་དུ་ཡིན་འདུག གནས་ཚུལ་དེའི་སྐོར་དཀོན་མཆོག་རིན་ཆེན་ལགས་ཀྱིས་ངོ་སྤྲོད་གནང་དོན། དེ་རིང་ང་ཚོའི་གཙང་མ་བཟོས་པ་ལ་རྟེན་ནས་ཁོ་ཚོར་ཚོར་བ་ཤུགས་ཆེ་པོ་ཞེ་དྲགས་བྱུང་འདུག ང་ཚོས་གཙང་བཟོ་སྐབས་གྱོན་གོས་སེར་པོ་གྱོན་ཏེ། རྒྱབ་ལ་༧གོང་ས་མཆོག་གི་སྐུ་པར་དང་མཉམ་དུ། “རྒྱ་གར་མི་དམངས་ལ་དཀའ་དྲིན་ཆེ།  རང་བཙན་ཉིན་མོར་བཀྲ་ཤིས་བདེ་ལེགས།” ཞེས་པའི་ཡི་གེ་ཞིག་བྲིས་ཡོད། དེ་ནས་བོད་ཀྱི་རྒྱལ་དར་དང་རྒྱ་གར་གྱི་རྒྱལ་དར། གཞོན་ནུའི་དར་ཆ་སོགས་འཕྱར་ཏེ་ཐུགས་རྗེ་ཆེ་ཞུས་བ་ཡིན། ས་གནས་མི་མང་ཚོས། ཁྱེད་རང་ཚོས་ག་རེ་བྱེད་ནས་གཙང་མ་བཟོས་བ་ཡིན་ཞེས་དྲིས་བྱུང་། ང་ཚོས། ང་ཚོ་རྒྱ་གར་ལ་སྐྱབས་བཅོལ་དུ་བསྡད་ཡོད་སྟབས། རྒྱ་གར་གཞུང་དམངས་ཀྱི་བཀའ་དྲིན་དྲན་པའི་སྒོ་ནས་ཐུགས་རྗེ་ཆེ་ཞུ་བཞིན་ཡོད། ཅེས་ལབ་སྐབས། ཁྱེད་ཚོས་ང་ཚོར་ཚོར་སྣང་གསར་བ་ཞིག་སྤྲད་སོང་།  ཐུགས་རྗེ་ཆེ་ཞེས་ཟེར་གྱི་འདུག  དེའི་ནང་ནས་སྤོ་བོ་ལོ་དྲུག་བཅུ་ཡས་མས་ཅན་ཞིག་གིས། ཁྱེད་རང་ཚོས་རྒྱ་གར་རང་བཙན་ཉིན་མོར་ཁོར་ཡུག་གཙང་མ་བཟོས་བ་དེ་ཞེ་དྲགས་ཡག་པོ་རེད། རྒྱ་གར་རང་བཙན་གྱི་ཉིན་མོར་འདི་ལ་རང་བཙན་གྱི་འཐབ་རྩོད་པ་གྷན་དྷི་དང་ནེ་རུ་གཙོ་བོར་གྱུར་པའི་སྐབས་དེའི་མི་སྣ་ཚོས་དཀའ་ལས་ཅི་ཙམ་བརྒྱབ་པ་རྟོགས་མཁན་མང་པོ་ཡོད་པ་མ་རེད། ཁྱེད་རང་ཚོས་རང་བཙན་ཉིན་མོའི་རིན་ཐང་རྟོགས་འདུག དཔེ་ཡག་པོ་རེད། ཅེས་དགའ་འཛུམ་ངང་ནས་ལག་ཐལ་མོ་སྦྱར་ནས་ཐུགས་རྗེ་ཆེ་ཞུས་སོང་། སློབ་ཕྲུག་ཁ་ཤས་ཤིག་བསླེབས་ནས་ང་ཚོར་ལག་པ་གཏོང་རོགས་ལབ་སོང་། ང་ཚོས་ལག་པ་བཙོག་པོ་འདུག་ལབ་པར། སྐྱོན་ཡོད་པ་མ་རེད། ཁྱེད་རང་ཚོས་ཁོར་ཡུག་གི་གད་སྙིགས་བཙོག་པོར་ལག་པ་འཆིང་ཐུབ་ན། ང་ཚོས་ཁྱེད་རང་ཚོའི་ལག་པར་ལག་པ་འཆིང་ན་སྐྱོན་ག་རེ་ཡོད་པ་རེད། ཁྱེད་རང་ཚོ་དངོས་གནས་ཐུགས་རྗེ་ཆེ།  ཞེས་ལབ་སོང་། ཡང་མི་ཁ་ཤས་ཞིག་གིས།  ཁྱེད་རང་ཚོ་འཇར་པན་བ་ཡིན་ནམ། འཇར་པན་བ་ཚོས་རིན་མེད་ཐོག་ཁོར་ཡུག་གཙང་མ་བཟོ་རྒྱུར་དོ་སྣང་ཆེན་པོ་ཡོད་པ་རེད། ཅེས་ལབ་སོང་། ང་ཚོས་ང་ཚོ་འཇར་པན་བ་མིན། ང་ཚོ་བོད་པ་ཡིན་ལབ་དུས། ཨོ། དེས་ན་ཏཱ་ལའི་བླ་མའི་མི་ཡིན་པ་རེད།  དངོས་གནས་རིན་མེད་ཐོག་ཁོར་ཡུག་གཙང་མ་བཟོ་བ་དེ་ལ་ཐུགས་རྗེ་ཆེ། ཞེས་ལབ་ནས་རྒྱ་གར་ཡུལ་མི་ཚོས་དོ་སྣང་ཆེན་པོ་བྱས་སོང་། ཞེས་ངོ་སྤྲོད་གནང་བྱུང་། རྒྱ་གར་ལྷོ་ཕྱོགས་ས་གནས་གསར་འགོད་པ་དོན་གྲུབ་བཀྲ་ཤིས།
