གོ་པལ་པུར་བོད་ཁྱིམ་གྱི་རྩེད་འགྲན་ཐེངས་ ༡༩
པ་མགོ་འཛུགས་གནང་།

༧གོང་ས་མཆོག་ནི་ནུབ་ལ་ཉེ་བའི་གནའ་བོའི་རྒྱ་གར་གྱི་རིག་གཞུང་དང་ཧི་མ་ལ་ཡའི་མི་རིགས་ཀྱི་རིག་གཞུང་སྲུང་སྐྱོབ་གནང་མཁན་ཆགས་ཡོད། དེ་རིང་སྔ་དྲོའི་ཆུ་ཚོད་ ༩ སྟེང་། ཧི་མ་ཅལ་མངའ་སྡེའི་ཀང་ར་རྫོང་ཁུངས་གོ་པལ་པུར་བོད་ཁྱིམ་སློབ་གྲྭའི་ནང་དུ། གོ་པལ་པུར་བོད་ཁྱིམ་སློབ་གྲྭ་དབུ་བརྙེས་ནས་མི་ལོ་ཧྲིལ་བོ་ ༢༠ དང་། སློབ་གྲྭའི་ཁང་ཚན་ཕན་ཚུན་བར་མཆོང་རྒྱུགས་འཕེན་གསུམ་གྱི་རྩེད་འགྲན་ཐེངས་ ༡༩ པ་མགོ་འཛུགས་གནང་སོང་བ་དང་། སྐུ་མགྲོན་གཙོ་བོར་ཀང་ར་རྫོང་ཉེན་རྟོག་ཁང་འགན་འཛིན་སྐུ་ཞབས་ཆཱ་ཧ་རྗེ་ཊ་ (Dr.Ramesh Chander Chhajta) ལགས་དང་། བོད་ཁྱིམ་སྤྱི་ཁྱབ་ཤེས་ཡོན་འགན་འཛིན་བསོད་ནམས་སྲིད་ཆོས་ལགས། དེ་བཞིན་སློབ་གྲྭའི་ཉེ་འགྲམ་གྱི་དགོན་སྡེ་ཁག་དང་ཚོགས་སྡེ་ཁག་གི་སྐུ་མགྲོན། གོ་པལ་པུར་བོད་ཁྱིམ་གྱི་དབུ་ལས་སློབ་ཕྲུག་དང་། ཕ་མ་བཅས་མི་གྲངས་ ༡༠༠༠ ལྷག་འཛོམས་འདུག མཛོད་སྒོའི་ཐོག སྐུ་མགྲོན་གཙོ་བོ་ལགས་ནས་སློབ་ཕྲུག་རྒན་བདག་ཐུབ་བསྟན་རིན་ཆེན་དང་། དོན་འགྲུབ་སྒྲོལ་མ་རྣམ་གཉིས་ལ་ད་ལོའི་ཤེས་རིག་ལས་ཁུངས་ཀྱི་གསུམ་བཟང་སློབ་མ་རྩེད་ཕུད་ཀྱི་ལག་འཁྱེར་གསོལ་རས་གནང་སོང་། དེའི་རྗེས་གོ་པལ་པུར་བོད་ཁྱིམ་གྱི་སློབ་གྲྭ་ཆེ་བའི་དགེ་ཆེ་ཟླ་བ་རྒྱལ་མཚན་ལགས་ཀྱིས་རྩེད་འགྲན་ལས་རིམ་ངོ་སྤྲོད་གནང་སྐབས། ད་ལན་གོ་པལ་པུར་བོད་ཁྱིམ་གྱི་རྒྱུགས་མཆོང་འཕེན་གསུམ་གྱི་རྩེད་འགྲན་ཐེངས་ ༡༩ པ་ཚོགས་ཀྱི་ཡོད་པ་དང་། རྩེད་འགྲན་འདི་ནི་ཕྱི་ལོ་ ༡༩༩༦ ལོ་ནས་མགོ་བཙུགས་ཏེ་རིམ་པས་རྩེད་འགྲན་ཐེངས་ ༡༩ པ་དེ་ཆགས་ཀྱི་ཡོད། རྩེད་འགྲན་གྱི་དགོས་པ་རྒྱུ་མཚན་གཙོ་བོ་ནི། སློབ་ཕྲུག་ཚོའི་འཕྲོད་བསྟེན་དང་འཚར་ལོངས། དེ་བཞིན་སོ་སོའི་རྒྱལ་ཁབ་དང་མི་རིགས་ཀྱི་ཞེན་ཁོག་སྐྱེད་སྲིང་བྱེད་ཐབས་སུ་ཁང་ཚན་ཕན་ཚུན་བར་རྩེད་འགྲན་བྱེད་ཀྱི་ཡོད། རྩེད་འགྲན་ནང་དམའ་རིམ་ནས་འཛིན་གྲྭ་བཅུ་གཉིས་བར་གྱི་སློབ་ཕྲུག་ཆུང་བ་དང་འབྲིང་བ། ཆེ་བ་བཅས་སྡེ་ཚན་གསུམ་དུ་བགོས་ཏེ། ཁང་ཚན་བཞིའི་འགྲན་ཞུགས་སློབ་ཕྲུག་ ༢༤༠ ལྷག་དང་རྩེད་མོའི་རྣམ་གྲངས་ ༣༠ ལྷག་གོ་སྒྲིག་བྱས་ཡོད། རྩེད་འགྲན་པ་ཨང་དང་པོ་གཉིས་པ་གསུམ་པ་བཅས་ལ་ཨང་རིམ་ལྟར་ལག་འཁྱེར་གསོལ་རས་སྤྲོད་ཀྱི་ཡོད། སློབ་ཕྲུག་ཚང་མས་རྩེད་འགྲན་ནང་ཆ་ཤས་ལེན་ཐབས་སུ། ང་ཚོས་མཆོང་རྒྱུགས་འཕེན་གསུམ་གྱི་རྩེད་འགྲན་ནང་མ་ཚུད་པའི་སློབ་ཕྲུག་ཚང་མ་རུ་སྒྲིག་ནང་གོ་སྒྲིག་བྱས་ནས་ལོ་རེ་བཞིན་འཚོགས་ཀྱི་ཡོད། རྩེད་འགྲན་དེ་ཡང་རྒྱལ་སྤྱིའི་ཨོ་རྩེད་འགྲན་ཚོགས་ཀྱི་རྣམ་པའི་ཐོག་ནས་འཚོགས་ཀྱི་ཡོད་ལ། དེ་ཡང་སློབ་ཕྲུག་ཚོའི་ནུས་པ་དང་མོས་པ་གཞིར་བཟུང་ནས་གོ་སྒྲིག་ཞུ་ཡི་ཡོད། ཅེས་བརྗོད་སོང་། སྐུ་མགྲོན་གཙོ་བོ་སྐུ་ཞབས་ཆཱ་ཧ་རྗེ་ཊ་ལགས་ཀྱིས་བོད་དང་རྒྱ་གར་མི་རིགས་གཉིས་ཀྱི་གནའ་དེང་གི་རིག་གཞུང་འབྲེལ་ལམ་སྐོར་བརྗོད་སྐབས། སློབ་གྲྭའི་རྩིག་ངོས་སུ་བྲིས་པའི་རང་ལས་གཞན་གཅེས་ཅེས་པ་འདི་ནི་ཕལ་ཆེར་ཧི་མ་ལ་ཡའི་མི་རིགས་ཀྱི་རིག་གཞུང་ཞིག་རེད་བསམ་གྱི་འདུག ཧིན་རྡུ་ཆོས་ལུགས་ཀྱི་རྗེས་འབྲང་བའམ་ནང་པའི་ཆོས་ལུགས་ཀྱི་རྗེས་འབྲང་གང་ཡིན་ཡང་། ང་ཚོའི་རང་རྒྱུད་ཀྱི་རིག་གཞུང་གཅིག་པ་ཆགས་བསྡད་ཡོད་པ་རེད། ང་ཚོ་ཧི་མ་ལ་ཡའི་མི་རིགས་རྣམས་ནི་རང་ལས་གཞན་གཅེས་པའི་བསམ་བློ་ཡོད་མཁན་ཞིག་རེད། དེང་སྐབས་༧གོང་ས་ཏཱ་ལའི་བླ་མ་མཆོག་ནི་ཧི་མ་ལ་ཡའི་མི་རིགས་དང་རྒྱ་གར་གྱི་གནའ་བོའི་རིག་གཞུང་གི་བདག་པོ་ཆགས་ཡོད་པར་མ་ཟད། ནུབ་ལ་ཉེ་བའི་གནའ་བོའི་རྒྱ་གར་གྱི་རིག་གཞུང་སྲུང་སྐྱོབ་གནང་མཁན་ཞིག་ཀྱང་ཆགས་ཡོད་པ་རེད། ༧གོང་ས་ཏཱ་ལའི་བླ་མ་མཆོག་གི་ལམ་སྟོན་འོག བོད་ཕྲུག་ན་གཞོན་ཚོར་གནའ་དེང་གི་སློབ་གསོ་དང་ཤེས་ཡོན་ཐོབ་ཀྱི་ཡོད་སྟབས། ཁྱོད་རྣམ་པ་མ་འོངས་པར་མི་བཟང་པོ་ཞིག་ཆགས་རྒྱུའི་རེ་བ་ཡོད་ལ། ཁྱེད་རང་ཚོ་རྒྱལ་ཁབ་གང་དུ་འཚོ་གནས་བྱེད་ཀྱི་ཡོད་ནའང་། ཁྱེད་རང་ཚོས་གཞན་ལ་ཕན་པའི་བསམ་པ་ཙམ་མིན་པར་དངོས་སུ་ལག་བསྟར་གནང་མཁན་ཞིག་ཡོང་བའི་རེ་བ་ཡོད། ང་རང་དེ་སྔོན་སིམ་ལར་བསྡད་མྱོང་། བོད་པ་ཚོ་གང་དུ་གནས་ཀྱང་སྒྲིག་ཁྲིམས་ལ་བརྩི་མཐོང་བྱེད་མཁན་ཞིག་རེད་འདུག སྤྱིར་དེང་དུས་ཤེས་ཡོན་གྱི་ལག་འཁྱེར་གལ་ཆེན་པོ་ཞིག་རེད་དེ། རང་རྒྱུད་དུ་ཤེས་ཡོན་ལེགས་པོ་མེད་ཚེ། ལག་འཁྱེར་ནི་རྩིག་ངོས་སུ་འགྲེམས་སྟོན་བྱས་པ་ཙམ་མ་གཏོགས་དེས་ཕན་ཐོགས་ཡོད་པ་མ་རེད། དེར་བརྟེན་ཁྱེད་རང་སློབ་ཕྲུག་ཚོས་ཆོས་དང་རིག་གཞུང་གི་ལེགས་ཆ་དེ་རང་རྒྱུད་ལ་འཇགས་ཐུབ་པ་བྱེད་རྒྱུ་ནི་ཧ་ཅང་གལ་ཆེན་པོ་རེད། ཅེས་བརྗོད་སོང་། གསར་འགོད་པ། ཉི་མ་ཚེ་རིང་། པདྨ་རྒྱལ།
