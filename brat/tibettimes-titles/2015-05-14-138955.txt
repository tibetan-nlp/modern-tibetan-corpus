ཤིན་ཅང་གི་དྲག་རྒོལ་དམར་གསོད།

ཉེ་སྔོན་ཤིན་ཅང་གི་ཞིབ་བཤེར་ས་ཚིགས་ཤིག་ཏུ་དྲག་རྒོལ་དོན་རྐྱེན་ཐེངས་གཉིས་བྱུང་སྟེ་ཁྱོན་མི་གྲངས་ ༦ བསད་པ་དང་མི་ ༤ ལ་རྨས་སྐྱོན་ཕོག་ཡོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༡ ཉིན་གྱི་མཚན་མོ་པེ་ཅིན་གྱི་ཆུ་ཚོད་ ༩ པ་ཙམ་ལ། ཤིན་ཅང་ཡུ་གུར་མི་རིགས་རང་སྐྱོང་ཁུལ་ལོ་ཕུ་རྫོང་ལོ་ཕུ་གྲོང་ཚོའི་ཞིབ་བཤེར་ས་ཚིགས་ཏུ། མི་ཞིག་གིས་འབར་རྫས་གཡུགས་ཏེ་ཁོ་རང་དང་ཉེན་རྟོག་པ་གཉིས་རྐྱེན་ལམ་དུ་བཏང་ཡོད་པ་དང་། དོན་རྐྱེན་དེ་བྱུང་མ་ཐག་ས་གནས་ཉེན་རྟོག་ཁང་གིས་རིམ་པ་དང་པོའི་འཐབ་རྒོལ་གྲ་སྒྲིག་བྱས་ཡོད་ཀྱང་། ཕྱི་ཚེས་ ༡༢ ཉིན་གྱི་མཚན་མོའི་ཆུ་ཚོད་ ༨ ཐོག ཡང་བསྐྱར་ཞིབ་བཤེར་ས་ཚིགས་དེར་གཏོར་རྒོལ་ཐེབས་ཏེ། ལུས་ལ་འབར་རྫས་བཅངས་པའི་མི་ ༢ དང་ཉེན་རྟོག་པ་ ༡ བསད་པ་དང་། ཉེན་རྟོག་པ་ ༤ ལ་རྨས་སྐྱོན་ཕོག་ཡོད་འདུག ཡུ་གུར་གྱི་གསར་འགྱུར་བརྒྱུད་ལམ་ཁག་ནས་དོན་རྐྱེན་དེ་བྱུང་རྗེས། ངོ་རྒོལ་བའི་ཁྱིམ་མི་གཉེན་འབྲེལ་སོགས་མི་གྲངས་ ༢༠༠ ལྷག་ས་གནས་ཉེན་རྟོག་ཁང་གིས་འཛིན་བཟུང་བཀག་ཉར་བྱས་ཡོད་པ་མ་ཟད། རྒྱ་ནག་ཉེན་རྟོག་ཕྱོགས་ནས་བསྟུད་མར་བྱུང་བའི་ངོ་རྒོལ་དོན་རྐྱེན་དེ་འཆར་གཞི་ལྡན་པ་དང་རྩ་འཛུགས་ཡོད་པ་ཞིག་ཡིན་སྐོར་བསྒྲགས་ཡོད་ཀྱང་། ད་ལྟའི་ཆར་དོན་རྐྱེན་བཟོ་མཁན་གཙོ་བོ་སུ་ཡིན་དང་། འཇིགས་སྐུལ་རྩ་འཛུགས་ལ་འབྲེལ་བ་ཡོད་མེད་གསལ་བོ་ཆགས་མེད་སྐོར་བརྗོད་འདུག ཡང་ཡུ་གུར་གྱི་བརྒྱུད་ལམ་ཁག་ནས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཞིབ་བཤེར་ས་ཚིགས་དེ་ས་གནས་འཆར་ལྡན་བུ་སྐྱེས་ཁང་དང་སྨན་ཁང་གི་འགྲམ་དུ་ཡོད་ལ། ཞིབ་བཤེར་ས་ཚིགས་དེས་གཙོ་བོ་ཡུ་གུར་བུད་མེད་ཚོས་མགོ་རས་གྱོན་པ་དང་། སྐྱེས་པ་ཚོས་ཨ་ར་རིང་པོ་འཇོག་པའི་ཐལ་དྲགས་རིང་ལུགས་སུ་རྩི་བའི་བྱ་སྤྱོད་རིགས་ལ་ཞིབ་བཤེར་བྱེད་བཞིན་ཡོད་པ་དང་། ཐེངས་འདིའི་གཏོར་རྒོལ་བ་ ༣ དེ་ལོ་ཕུ་རྫོང་ནས་ཡིན་ཞིང་། ད་ལྟ་རང་ལོ་ ༡༨ ནས་ ༢༠ བར་ཡིན་འདུག དོན་རྐྱེན་བྱུང་རྗེས་ས་གནས་ཉེན་རྟོག་ཁང་གི་འགན་འཛིན་གྱིས། སྲིད་གཞུང་གིས་ཡུ་གུར་བའི་མགོ་རས་དང་ཨ་ར་ལ་ཞིབ་བཤེར་བྱེད་བཞིན་པ་དེར་མ་དགའ་བའི་རྣམ་འགྱུར་སྟོན་ཆེད། ཞིབ་བཤེར་ས་ཚིགས་དེ་དྲག་རྒོལ་གྱི་དམིགས་འབེན་དུ་བཟུང་བ་ཡིན་སྲིད་སྐོར་བརྗོད་འདུག རྒྱ་ནག་ནང་གི་གྲངས་ཉུང་མི་རིགས་ཀྱི་གནད་དོན་ལྟ་ཞིབ་པ་ཨ་རིའི་ཧྥོར་སི་བྷོར་གྷི་མངའ་སྡེའི་གཙུག་ལག་སློབ་ཆེན་(Frostburg State University)གྱི་དགེ་རྒན་ཆེན་མོ་མཱ་ཧི་ཡེན་གྱིས། ཤིན་ཅང་ནང་གི་ས་གནས་སྲིད་གཞུང་ཚང་མས་གནད་དོན་ཆུང་དུ་བཏང་སྟེ་མེད་པ་བཟོ་བའི་ཐབས་ལམ་མི་འཚོལ་བར་གོ་སྐབས་སྣ་ཚོགས་བཙལ་ཏེ་གནད་དོན་བཟོ་བ་དང་། གནད་དོན་ཤུགས་ཆེ་རུ་གཏོང་བཞིན་འདུག གནད་དོན་ཆུང་ཆུང་ཐ་མག་དང་ཆང་རག་བཙོང་བ། དཔང་ཡིག་བསྡུ་ལེན་ནས་ཞིབ་བཤེར་ས་ཚིགས་ལ་གཏོར་རྒོལ་བཏང་བའི་གནད་དོན་རིམ་པའི་ཁྲོད་མི་མང་པོ་འཛིན་བཟུང་བྱས་ཡོད་པ་རེད། ང་ཚོས་སེམས་ཁུར་བྱེད་སའི་གནད་དོན་ཞིག་ནི། རྒྱ་ནག་གི་གཅིག་གྱུར་ཆེན་པོའི་ཁོར་ཡུག་འོག ག་རེ་བྱས་ནས་ཤིན་ཅང་ཁུལ་གྱི་རྫོང་སྲིད་གཞུང་ནས་གྲོང་ཁྱེར་སྲིད་གཞུང་བར་གྱིས། རྒྱལ་ཁབ་ཀྱི་གཞི་རྩའི་སྒྲིག་ཁྲིམས་དང་འགལ་བའི་བྱ་སྤྱོད་སྤེལ་བཞིན་ཡོད་པ་རེད། ཅེས་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
