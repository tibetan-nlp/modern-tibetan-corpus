
གཞོན་པའི་ཤེས་ཡོན་གྱི་ཁྲག་ཚ་བོ་དང་རྒན་པའི་ཉམས་མྱོང་གི་གཟུགས་སྟོབས་གཉིས་མཉམ་རུབ་བྱེད་དགོས།

  བོད་ཀྱི་དུས་བབ་ཀྱིས་སྐབས་བཅུ་དྲུག་པའི་སྤྱི་འཐུས་འོས་མི་ཚོར་བཅར་འདྲིའི་ལས་རིམ་ཞིག་འགྲོ་བཞིན་ཡོད་པ་ལྟར། ཐེངས་འདིར་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་དྲུང་ཆེ་ཟུར་པ་མི་འགྱུར་རྡོ་རྗེ་ལགས་སུ། ཁོང་གིས་ལོ་ངོ་ ༣༩ རིང་བཙན་བྱོལ་བོད་མིའི་སྒྲིག་འཛུགས་ནང་ཕྱག་ལས་གནང་བ་དང་འབྲེལ་ཏེ། ཁོང་སྐབས་བཅུ་དྲུག་པའི་བོད་མི་མང་སྤྱི་འཐུས་ཀྱི་འོས་མིར་བཞེངས་དགོས་པའི་རྒྱུ་མཚན་སྐོར་ལ་བཅར་འདྲི་ཞུས།   དྲི་བ། སྐུ་ཉིད་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་དྲུང་ཆེའི་གནས་རིམ་ནས་རྒན་ཡོལ་ཞུས་ཏེ། ད་ལྟ་སྤྱི་འཐུས་ཀྱི་འོས་མིའི་ཁོངས་སུ་ཞུགས་ཡོད་པའི་རྒྱུ་མཚན་གང་ཡིན་ནམ།   ལན། རྩོམ་སྒྲིག་པའི་གཙོས་པའི་དུས་བབ་ནང་ཕྱག་ལས་གནང་མཁན་ཚང་མར་ཐུགས་རྗེ་ཆེ་གནང་ཞུ་རྒྱུ་ཡིན། རྒྱུ་མཚན་གང་ཡིན་ཞེ་ན། དེ་སྔ་བཙན་བྱོལ་བོད་མིའི་གཞུང་དང་ད་ལྟ་བཙན་བྱོལ་བོད་མིའི་སྒྲིག་འཛུགས་འདིའི་ནང་གི་ལས་བྱེད་མང་པོ་ཞིག་གིས་སོ་སོའི་མི་ཚེའི་དྭངས་མ་བློས་བཏང་སྟེ་བོད་བསྟན་པ་ཆབ་སྲིད་དང་མི་མང་གི་ཆེད་དུ་ལོ་ངོ་ ༢༠ དང་ ༤༠ མ་ཟིན་ཙམ་གྱི་རིང་ཞབས་ཞུ་ཞུས་པའི་སྲི་ཞུ་ལས་བྱེད་གང་འཚམ་ཞིག་ཡོད་པ་ཚོར་སོ་སོའི་མྱོང་ཚོར་ལ་གཞིགས་པའི་གནས་ཚུལ་ཕྲན་བུ་རེ་ཞུ་རྒྱུ་ཚང་མར་ཡོད་པ་རེད། དེ་རིང་ཁྱེད་རང་ཚོས་ང་ལ་བཅར་འདྲི་གནང་བ་དེ་ཞེ་དྲགས་ཡག་པོ་འདུག་ཞུ་བཞིན་ཡོད། མ་འོངས་པར་ཡང་དེ་འདྲའི་ལམ་སྲོལ་ཞིག་འཇོག་ཐུབ་པ་ཡིན་ན། ལས་བྱེད་པ་རྣམས་ལ་བརྒྱུད་ལམ་ཁག་ནས་ཆེ་མཐོང་གནང་གི་ཡོད་པའང་མཚོན་ཐུབ་པ་རེད།   དེ་སྔ་ངས་བོད་གཞུང་དགའ་ལྡན་ཕོ་བྲང་ཕྱོགས་ལས་རྣམ་རྒྱལ་གྱི་མང་གཙོའི་གཞུང་གི་ནང་ལོ་ ༣༦ རིང་དང། དེ་རྗེས་བོད་མིའི་སྒྲིག་འཛུགས་ནང་ལོ་ ༣ རིང་ཞབས་ཞུ་ཞུས་ཡོད། ད་ལྟ་ང་སྤྱི་འཐུས་ཀྱི་འོས་ཁོངས་སུ་བསླེབས་བསྡད་པའི་རྒྱུ་མཚན་གཙོ་ཆེ་ཤོས་དེ་མངའ་རིས་སྤྱི་མཐུན་ཚོགས་པ་དང་། ཆོལ་གསུམ་ཆིག་སྒྲིལ་ཚོགས་པ་བཅས་ནས་ང་སྤྱི་འཐུས་འོས་གཞིར་བཏོན་པ་ལྟར། དབུས་གཙང་ཆོལ་ཁའི་མི་མང་གིས་ངས་རྒྱུན་རིང་པོར་བཙན་བྱོལ་བོད་མིའི་སྒྲིག་འཛུགས་ནང་ཞབས་ཞུ་ཞུས་པའི་ཉམས་མྱོང་ལ་གཟིགས་ཏེ། ད་ལྟ་ང་རྩེ་ཕུད་བཅུའི་ནང་ཚུད་པ་དེ་ང་ལ་ཐུགས་རེ་གནང་བ་ཞིག་ཡིན་པར་ངོས་འཛིན་ཞུ་ཡི་ཡོད། ང་སྤྱི་འཐུས་འོས་མིའི་ཁོངས་ཞུགས་པའི་རྒྱུ་མཚན་དང་པོ་དེ་རྒྱ་ཆེའི་མང་ཚོགས་ཀྱི་ཐུགས་རེ་སྐོང་བའི་ཆེད་དུ་ཡིན། གཉིས་པ་དེ་ང་བཙན་བྱོལ་བོད་མིའི་སྒྲིག་འཛུགས་ནང་བསྡད་དུས་ང་གཞུང་ཞབས་པ་ཤིག་རེད། གཞུང་ཞབས་པ་ཟེར་བ་དེས་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱིས་གཏན་ལ་ཕབ་པའི་སྲིད་བྱུས་དེ་ཚོ་ལག་ལེན་བསྟར་རྒྱུ་མ་གཏོགས། སྒྲིག་འཛུགས་ཀྱི་སྲིད་བྱུས་ཁག་འཛིན་སྐྱོང་ལག་ལེན་གནང་ཕྱོགས་སོགས་ལ་དགེ་སྐྱོན་ཡོ་བསྲང་བྱེད་པའི་གོ་སྐབས་དེ་ཕལ་ཆེར་མེད་པ་འདྲ་པོ་རེད། བོད་མི་མང་སྤྱི་འཐུས་ཀྱི་སྡིངས་ཆ་འདིའི་སྟེང་ནས་བཙན་བྱོལ་བོད་མིའི་སྒྲིག་འཛུགས་འདིའི་ནང་དགེ་སྐྱོན་ཡོ་བསྲང་དང་ཆབ་སྲིད་དང་འབྲེལ་བའི་གནད་དོན་མང་པོ་ཞིག་གི་ཐོག་ལ་སོ་སོའི་ཉམས་མྱོང་དང་འཆར་སྣང་ལ་གཞིགས་པའི་བསམ་འཆར་འདོན་རྒྱུའི་གོ་སྐབས་ཡོད་སྟབས་ང་སྤྱི་འཐུས་འོས་མིའི་ནང་ལངས་པ་ཡིན།   དྲི་བ། སྐུ་ཉིད་ཀྱིས་ལོ་ངོ་ ༣༩ རིང་གཞུང་ལ་ཞབས་ཞུ་ཞུས་རྗེས་ད་ལྟ་སྤྱི་འཐུས་འོས་མིའི་ནང་ཞུགས་ཡོད་པ་རེད། ང་ཚོའི་སྤྱི་ཚོགས་ནང་མི་རྒན་པ་ཚོས་རྐུབ་སྟེགས་འཇུས་བསྡད་པ་ལས་གཞོན་སྐྱེས་ཚོར་གོ་སྐབས་སྤྲོད་དགོས་ཟེར་བའི་སྐད་ཆ་ཡོང་གི་ཡོད་སྟབས། སྐད་ཆ་དེར་ཁྱེད་རང་གིས་ངོས་འཛིན་གང་འདྲ་གནང་གི་ཡོད།   ལན། དགོངས་འཆར་དེ་འདྲ་གནང་མཁན་ཚོར་ངས་རྗེས་སུ་ཡི་རངས་ཞུ་ཡི་ཡོད། དགོངས་འཆར་དེ་ཡག་པོ་ཞེ་དྲགས་འདུག ཡིན་ནའང་ང་ཚོས་དགོངས་པ་བཞེས་ས་གཙོ་བོ་གཅིག་གང་ཆགས་ཀྱི་ཡོད་པ་རེད་ཟེར་ན། ཚང་མ་གསར་པ་བྱས་ནས་ཉམས་མྱོང་ཁྱོན་ནས་མེད་མཁན་རྐྱང་རྐྱང་ཆགས་ན། སྒྲིག་འཛུགས་འདི་དང་གྲོས་ཚོགས་འདི་ག་འདྲ་ཞིག་ཆགས་འགྲོ་ས་རེད། རྒན་པའི་ཉམས་ལ་ཆེ་མཐོང་མ་སྤྲད་པའི་དབང་གིས་ལས་དོན་མང་པོ་ཞིག་གི་ཐོག་ལ་ཏག་ཏག་འགྲོ་མ་ཐུབ་པའི་དཀའ་ངལ་ཡོད་པ་ད་ལྟ་ནས་མཐོང་གི་འདུག ངས་ག་དུས་ཡིན་ནའང་གཞོན་པའི་དེང་དུས་ཤེས་ཡོན་གྱི་ཁྲག་ཚ་བོ་དང་རྒན་པའི་ཉམས་མྱོང་གི་གཟུགས་སྟོབས་གཉིས་མཉམ་རུབ་ཐོག་ནས་འགྲོ་ཐུབ་དགོས་པ་དེ་ཞེ་དྲགས་གལ་ཆེན་པོ་རེད་བསམ་གྱི་ཡོད། དཔེར་ན། ང་འདྲ་བོ་ཞིག་ལ་ཆ་བཞག་ན། ཁྱེད་རང་རྒན་པ་རེད་ད་དུང་འདིའི་ནང་བསྡད་པ་ཡིན་ན་མི་གཞན་ཞིག་གི་ལམ་ཀ་འགོག་གི་མ་རེད་དམ་ཟེར་བ་དེ་འདྲ་བཤད་མི་སྲིད་པ་མ་རེད། ངས་ཀྱང་ཐུགས་ཕན་གསོ་ལུགས་ག་འདྲ་ཆགས་མིན་ཐོག་བསམ་བཏང་བ་ཡིན། ཡིན་ནའང་ ༢༠༡༥ ཟླ་ ༡༢ ཚེས་བཅུ་གྲངས་ཤིག་ནས་ངའི་བསམ་བློ་ལ་འགྱུར་བ་ཞེ་དྲགས་ཐེབས་སོང་། ངས་གནས་ཚུལ་གོ་བ་ལ་དབུས་གཙང་ཆོལ་ཁའི་འོས་འཕེན་པ་མ་ཡིན་པ་རེ་ཟུང་གིས་ང་དབུས་གཙང་སྤྱི་འཐུས་ནང་འགོག་ཐབས་བྱེད་དགོས་པ་རེད་ཟེར་བའི་སྐད་ཆ་བྱུང་འདུག གསུང་སྟངས་དེ་ཐབས་སྐྱོ་བོ་ཞེ་དྲགས་རེད་དེ། ངས་སྐད་ཆ་བཤད་པ་དང་ལས་དོན་བྱས་པ་དེ་དག་གིས་ཕན་ས་ཞིག་ལ་ཕན་གྱི་ཡོད་སྟབས། དབུས་གཙང་མི་མང་གིས་ང་ལ་ཐུགས་རེ་གནང་བ་རེད་བསམ་གྱི་འདུག དེ་ནས་བཟུང་ང་འོས་གཞིའི་ནང་སྡོད་རྒྱུ་གཏན་འཁེལ་ཏེ་བོད་མིའི་སྒྲིག་འཛུགས་ནས་དགོངས་པ་ཞུས་པ་ཡིན།   དྲི་བ། གོ་ཐོས་ལྟར་ན་སྐུ་ཉིད་དགའ་ལྡན་ཕོ་བྲང་གི་བཀའ་དྲུང་མཐའ་མ་དེ་ཆགས་ཡོད་འདུག འདིའི་ཐད་ལ་ཁྱེད་རང་ལ་དགོངས་ཚུལ་དང་ངོས་འཛིན་གང་འདྲ་ཡོད་དམ།   ལན། དངོས་གནས་དྲང་གནས་ངའི་མི་ཚེའི་ནང་ཕྱི་བལྟས་བྱེད་སྐབས། རྩ་བའི་ལོ་ངོ་ ༣༩ ནང་ངས་རྒྱལ་ཁབ་མ་འདྲ་བ་ཁག་བཞིའི་ནང་༧གོང་ས་སྐྱབས་མགོན་ཆེན་པོ་མཆོག་གི་སྐུ་ཚབ་དོན་གཅོད་ཞུ་བའི་གོ་སྐབས་བྱུང་བ་རེད། དེ་བཞིན་སྐབས་ཤིག་ལ་བཀའ་ཤག་གི་ཆབ་སྲིད་དྲུང་ཆེ་བྱས་པ་སོགས་གང་ལྟར་ངས་ལས་དོན་མང་པོ་ཞུས་པ་ཡིན། བཙན་བྱོལ་བོད་མིའི་གཞུང་ཡིན་ན་འདྲ་བཙན་བྱོལ་བོད་མིའི་སྒྲིག་འཛུགས་ཡིན་ནའང་འདྲ་གཞུང་ཞབས་པ་ཞིག་གི་ཚད་མཐོ་ཤོས་ཀྱི་འགྲོ་ས་དེ་བཀའ་དྲུང་ཆགས་བསྡད་ཡོད་པ་དང་། ང་རང་བོད་གཞུང་དགའ་ལྡན་ཕོ་བྲང་ཕྱོགས་ལས་རྣམ་རྒྱལ་གྱི་བཀའ་དྲུང་མཐའ་མ་དེ་ཆགས་བསྡད་ཡོད་པ་རེད། དེ་ནས་བཟུང་སྟེ་དགའ་ལྡན་ཕོ་བྲང་གི་གཞུང་གི་དུས་ཚོད་དེ་ཡལ་ནས་སྒྲིག་འཛུགས་གསར་པ་ཞིག་བསླེབས་བསྡད་ཡོད་པ་རེད། ང་དགའ་ལྡན་ཕོ་བྲང་གི་བཀའ་དྲུང་མཐའ་མ་དེ་ཆགས་པ་ངའི་མི་ཚེའི་ནང་གི་རྩ་ཆེ་ཤོས་ཤིག་དང་གལ་འགངས་ཆེ་ཤོས་ཤིག་རེད། གང་ཡིན་ཟེར་ན། ཆབ་སྲིད་ཀྱི་འགྱུར་ལྡོག་ཧ་ལས་པའི་ཆེན་པོ་ཞིག་གི་མཐུད་མཚམས་སུ་ལས་ཀ་བྱེད་མཁན་གྱི་མི་ཞིག་ཆགས་ཡོད་པའི་རྒྱུ་མཚན་ལ་བརྟེན་ནས་ང་རང་ཞེ་དྲགས་སྤོབས་པ་ཆེན་པོ་ཡོད།   དྲི་བ། བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་སྲི་ཞུ་ལས་བྱེད་ཚོས་གསར་འགྱུར་བརྒྱུད་ལམ་ཐོག་རང་དབང་གིས་སྐད་ཆ་བཤད་པ་དང་རྩོམ་ཡིག་འབྲི་བ་སོགས་ཆོག་གི་ཡོད་པ་མ་རེད། ད་ལྟའི་དུས་ཚོད་ཀྱི་ཆ་ནས་སྒྲིག་གཞི་འདི་ལ་འགྱུར་བ་གཏོང་དགོས་པར་གཟིགས་ཀྱི་ཡོད་དམ།   ལན། དེ་ཏོག་ཙམ་རྙོག་འཛིང་ཆེན་པོའི་བཀའ་མོལ་ཞིག་རེད། གང་ཡིན་ཟེར་ན། སྤྱིར་བཏང་རང་དབང་གི་ཐོག་ནས་རྩོམ་ཡིག་སྤེལ་རྒྱུ་དེ་གལ་ཆེན་པོ་རེད། ཕྱོགས་གཞན་ཞིག་ནས་བལྟས་ན། སྒྲིག་འཛུགས་ཟེར་བ་འདི་ཁྲིམས་དང་སྒྲིག་གཞི་ཞིག་གི་འོག་ལ་འགྲོ་དགོས་ཀྱི་ཡོད་སྟབས། ལས་བྱེད་པ་ཚང་མས་རང་སྣང་གང་དྲན་བྱས་ནས་ཅིག་བྲིས་བསྡད་པ་ཡིན་ན་དེ་ཏག་ཏག་བསྡད་ཡོད་པ་མ་རེད། ངས་གོང་དུ་ཞུས་པ་ལྟར་བོད་མིའི་སྒྲིག་འཛུགས་ཟེར་བ་འདིར་མགོ་ཁྲིད་སུ་སླེབ་མིན་སྐད་ཆ་ཅི་གང་མ་རེད། ༧རྒྱལ་བ་རིན་པོ་ཆེ་སྐུ་ཕྲེང་བཅུ་བཞི་པ་མཆོག་ནས་དགུང་ལོ་ ༡༦ ནས་བཟུང་སྐུ་ངལ་བསྐྱོན་པའི་སྒྲིག་འཛུགས་ཤིག་ཆགས་བསྡད་ཡོད་པ་རེད། ངས་སྲི་ཞུ་བ་ཟེར་བ་དེ་སྲི་ཞུ་བ་རང་གི་ངོ་བོར་སྡོད་དགོས་པ་རེད་བསམ་གྱི་འདུག ཡིན་ནའང་། ངས་གསལ་པོ་དྲན་གྱི་མི་འདུག་མ་གཏོགས་ང་ཚོའི་སྒྲིག་གཞིའི་ནང་། སྤྱི་ཚོགས་ཁྱོན་ཡོངས་ལ་ཕན་པའི་རིགས་བྲིས་ན་འགྲིག་གི་ཡོད་པ་རེད་དང་། ཆོས་དང་རིག་གཞུང་སོགས་ལ་འབྲེལ་བ་ཡོད་པའི་ཐོག་ལ་རྩོམ་ཡིག་དེ་འདྲ་བྲིས་ན་ཆོག་རྒྱུ་ཞེས་འཁོད་ཡོད་བསམ་གྱི་འདུག   དྲི་བ། སྒང་ལ་བཞུགས་མཁན་དང་གཤམ་འོག་ལ་འདུག་མཁན་ཚང་མས་བོད་པ་རང་གི་རིག་གཞུང་སྲུབ་སྐྱོབ་དགོས་སྐོར་གསུང་གི་ཡོད་པ་རེད། སྐུ་ཉིད་ནས་ང་ཚོས་སྲུང་སྐྱོབ་དགོས་པའི་རིག་གཞུང་དེ་གཙོ་བོ་གང་དང་གང་ཆགས་ཀྱི་ཡོད་པ་རེད་དགོངས་ཀྱི་ཡོད།   ལན། སྤྱིར་བཏང་རིག་གཞུང་ཟེར་བ་དེ་རྒྱ་ཆེན་པོ་ཞེ་དྲགས་རེད། ཡིན་ནའང་རིག་གཞུང་གི་རྩ་བ་དེ་སྐད་ཡིག་ལ་ཐུག་ཡོད་པ་རེད། ང་ཚོར་བོད་རང་ནས་དར་བའི་རིག་གཞུང་དང་རྒྱ་གར་སོགས་རྒྱལ་ཁབ་དང་མི་རིགས་གཞན་གྱི་ལེགས་ཆ་ཚུར་བླངས་པའི་རིག་གཞུང་སོགས་མང་པོ་ཞིག་ཡོད་པ་དེ་དག་རྒྱུན་འཁྱོངས་ཡོང་བའི་སྲོག་རྩ་དེ་ང་ཚོའི་སྐད་དང་ཡི་གེ་ལ་ཐུག་ཡོད་པ་རེད། བོད་མིའི་སྒྲིག་འཛུགས་ལས་ཀྱི་སྣེ་མོ་བ་སུ་ཞིག་ཡིན་ནའང་སོ་སོའི་སྐད་ཡིག་ལ་བདག་འཛིན་ཡག་པོ་ཞིག་བྱེད་མ་ཐུབ་པ་ཡིན་ན། ང་ཚོའི་མི་རབས་རྗེས་མ་ཚོས་སུ་ལ་དཔེ་ལྟ་དགོས་པ་རེད། མ་འོངས་པར་ང་ཚོའི་དབུ་ཁྲིད་ཁ་ཤས་ལ་བོད་སྐད་ཀྱིས་མ་འདང་བར་མང་ཚོགས་ལ་གཏམ་བཤད་སྐབས་སྐད་སྒྱུར་དགོས་པའི་དུས་ཚོད་ཅིག་ཤར་མི་སྲིད་པ་མ་རེད། ང་ཚོས་ཕྱོགས་གཅིག་ནས་རྒྱལ་སྤྱིའི་ནང་བོད་མིར་འགྲོ་བ་མིའི་ཐོབ་ཐང་མི་འདུག སོ་སོའི་སྐད་ཡིག་བཀོལ་སྤྱོད་ཀྱི་ཐོབ་ཐང་མི་འདུག་ཅེས་སྐད་རྒྱག་གི་ཡོད་པ་དང་། ཕྱོགས་གཞན་ཞིག་ནས་ང་ཚོ་རང་གི་སྤྱི་ཚོགས་ནང་འགན་འཁུར་མཁན་ཚོ་སོ་སོའི་སྐད་ཡིག་ཐོག་རང་རྐྱ་འཕེར་བ་ཞིག་མེད་ན། ང་ཚོར་གཞན་ཞིག་ལ་སྐད་རྒྱག་ས་ལྷག་གི་ཡོད་པ་མ་རེད།   དྲི་བ། སྐུ་ཉིད་ཀྱིས་འདས་པའི་ལོ་ངོ་ ༣༩ རིང་གཞུང་ནང་ཕྱག་ལས་གནང་བ་རེད། དུས་ཡུན་རིང་པོ་དེའི་ནང་བོད་མིའི་སྒྲིག་འཛུགས་ལ་འགྱུར་བ་ཆེ་ཤོས་གང་དག་ཕྱིན་སོང་།   ལན། འགྱུར་བ་མང་པོ་ཞེ་དྲགས་ཕྱིན་ཡོད་པ་རེད། དེ་དག་ངས་ཞུ་དགོས་པ་མ་རེད་ཡིག་ཐོག་ན་གསལ་བསྡད་ཡོད་པ་རེད། རྩ་བའི་སྐབས་སོ་སོར་བཀའ་ཤག་སུ་ཕེབས་པ་ཡིན་ནའང་སྐབས་སོ་སོར་སྲིད་བྱུས་ངེས་ཅན་རེ་རེ་ཏན་ཏན་ཡོང་གི་རེད་ལ། ཡོང་དགོས་པ་ཞིག་ཀྱང་རེད། ཡིན་ནའང་དེ་དག་ཚང་མ་ཕྱོགས་བསྡོམས་བྱས་ཡོང་ན། བཀའ་ཤག་ག་འདྲ་ཞིག་གི་སྐབས་ལ་ཁེལ་བ་ཡིན་ནའང་༧གོང་ས་སྐྱབས་མགོན་ཆེན་པོ་མཆོག་གི་དགོངས་གཞི་ག་རེ་རེད། དེ་ལྟེ་བ་རེད། དེ་ཧ་ཅང་གལ་ཆེན་པོའང་རེད། ང་ཚོ་བོད་རྒྱ་ཆེའི་མང་ཚོགས་དང་བོད་རྒྱལ་ཁབ་ཀྱི་ཁོག་པའི་སྙིང་དང་ལྟ་བའི་མིག་དེ་༧རྒྱལ་བ་རིན་པོ་ཆེ་ཆགས་ཀྱི་ཡོད་པ་རེད། དོན་དག་གཉིས་པ་དེ་བོད་རྒྱལ་ཁབ་དང་བོད་མི་རིགས་གནས་ཐུབ་དང་མི་ཐུབ་ཟེར་བ་དེ་བོད་མི་རིགས་ལ་ཡོད་པའི་ཐུན་མོང་མ་ཡིན་པའི་ཆོས་དང་སྐད་ཡིག་ལ་ཐུག་ཡོད་པ་དང་། དེ་དག་ད་ལྟ་ང་ཚོས་ག་ཚོད་ཅིག་སྲུང་སྐྱོབ་བྱེད་ཐུབ་ཀྱི་འདུག་དང་མི་འདུག བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་ལས་ཁུངས་ནང་ཁུལ་ཕན་ཚུན་བར་ཡིག་འབྲེལ་བྱེད་པ་ཚུན་ཆད་ཀྱང་བོད་ཡིག་ཐོག་འགྲོ་དགོས་ཀྱི་ཡོད་པ་དེ་གྲོས་ཚོགས་སྔ་རྗེས་ནང་བཀའ་མོལ་ཡོང་གི་ཡོད་པ་རེད། གསུམ་པ་དེ་བོད་མིའི་སྒྲིག་འཛུགས་ཟེར་བ་དེ་ཧ་ལས་པའི་གལ་ཆེན་པོ་རེད། བོད་མིའི་སྒྲིག་འཛུགས་འདི་དངོས་གནས་དྲང་གནས་ཚ་ཚ་འུར་འུར་བྱས་ནས་གནས་དགོས་པར་གནད་འགག་མང་པོ་ཞིག་ཡོད་པ་རེད། བོད་མི་ས་ཡ་དྲུག་ལ་བདེ་སྐྱིད་བསྐྲུན་པའི་བཟོ་གྲྭ་ཟེར་ན་འདྲ་ཐབས་ལམ་ལྟེ་བ་ཞིག་འདི་ཆགས་བསྡད་ཡོད་པ་རེད། བོད་མིའི་སྒྲིག་འཛུགས་འདི་སྣང་མེད་ངང་འཐུས་ཤོར་ཕྱིན་ན་ང་ཚོར་ཕངས་པོ་ཞེ་དྲགས་ཆགས་ཀྱི་རེད་བསམ་གྱི་འདུག ཕྱི་ལོགས་ལ་ཁྲ་ཁྲ་ཟང་ཟིང་ལས་གལ་ཆེ་བ་དེ་ནང་ལོགས་སྲ་ཐག་ཐག་ཅིག་བཟོ་རྒྱུ་དེ་གལ་ཆེ་ཤོས་ཆགས་ཡོད་པ་རེད།   དྲི་བ། བཀའ་ཤག་སྐབས་བཅུ་བཞི་པའི་འདས་པའི་ལོ་བཞི་ལྷག་རིང་གངས་ཅན་སྐྱིད་གཤོངས་ཟེར་ན་འདྲ། བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་སྲིད་བྱུས་འཛིན་སྟངས་དང་ལས་དོན་གནང་ཕྱོགས་ཐོག་འགྱུར་བ་ཆེ་ཤོས་གང་དག་ཕྱིན་སོང་།   ལན། གངས་ཅན་སྐྱིད་གཤོངས་ལ་འགྱུར་བ་ག་རེ་འགྲོ་ཡི་འདུག བོད་མིའི་སྒྲིག་འཛུགས་ལ་འགྱུར་བ་ག་རེ་འགྲོ་ཡི་འདུག བོད་མིའི་འཐབ་རྩོད་ལ་འགྱུར་བ་ག་འདྲ་འགྲོ་ཡི་འདུག འཐབ་རྩོད་ཀྱི་ལས་དོན་བྱེད་ཕྱོགས་ལ་འགྱུར་བ་ག་འདྲ་འགྲོ་ཡི་འདུག དེ་ཚོ་བོད་སྤྱི་པའི་དོན་དག་ལ་ཤ་ཚ་ཞེན་ཁོག་ཡོད་མཁན་ཞིག་ཡིན་ན་གསལ་པོ་ཆགས་བསྡད་ཡོད་པ་རེད། ལྷག་པར་དུ་གངས་ཅན་སྐྱིད་གཤོང་གིས་མཚོན་པའི་ཕྱོགས་མཐའི་ལས་ཀྱི་སྣེ་མོར་གཏོགས་མཁན་ཚོས་མཁྱེན་གྱི་ཡོད་པ་རེད། ངས་གཅིག་གཅིག་གཉིས་གཉིས་བྱས་ནས་འགྲེལ་བཤད་དགོས་པ་མཐོང་མ་སོང་། ཡིན་ནའང་སྤྱི་ཡོངས་ཀྱི་ཐོག་ནས་ཉེན་ཁ་ཆེ་ཤོས་ཟེར་ན་འདྲ། ཛ་དྲག་ཤོས་འདྲ་བོ་དེ་ག་རེ་མཐོང་སོང་ཟེར་ན། སྔོན་མ་རིན་པོ་ཆེ་ཡིན་པའི་ཁ་ཐུག་ལ་འོག་གི་སྲི་ཞུ་ལས་བྱེད་དེ་ཚོས་སྒང་ལ་བརྩི་བཀུར་དང་སྒང་ནས་འོག་ལ་མཐོང་དང་ཆ་འཇོག་ཡག་པོ་ཡོད་པ་རེད། གངས་ཅན་སྐྱིད་གཤོངས་དེ་ཚ་སོབ་སོབ་འདྲ་བོ་ཞིག་ཡོད་པ་རེད། ད་ལྟ་དྲོད་དེ་ཧ་ལམ་ཡལ་བསྡད་པ་འདྲ་བོ་ཞིག་དང་ཚང་མ་ཉོབ་བསྡད་པ་འདྲ་བོ་ཞིག་ཆགས་ཀྱི་འདུག དེ་གཅིག་རེད། གཉིས་ནས་ངས་གོང་དུ་ཞུས་པ་ལྟར་བོད་པ་རང་གི་སྐད་ཡིག་ཐོག་ཞེ་དྲགས་དམ་དོན་མེད་པའི་རྣམ་པ་ཞིག་གང་ས་གང་ནས་འཐོན་ཡོང་གི་འདུག དེ་ཐབས་སྐྱོ་བོ་ཞེ་དྲགས་རེད། ས་གནས་མི་མང་གིས་གངས་ཅན་སྐྱིད་གཤོང་ལ་ལྟ་བསྡད་ཡོད་པ་རེད། གནད་དོན་གཞན་པ་གཅིག་ག་རེ་འདུག་ཟེར་ན། སྔོན་མ་གཞི་རིམ་གྱི་སྲིད་བྱུས་ཁག་བཀའ་ཤག་གིས་གཏན་འབེབས་གནང་རྗེས་ལག་བསྟར་གྱི་རིམ་པ་དེ་འབྲེལ་ཡོད་སོ་སོས་ཐག་བཅད་ནས་འགྲོ་བསྡད་ཡོད་པ་རེད། ད་ལྟ་དབང་ཆ་ཚང་མ་ཡར་བསྡུ་གསོག་བརྒྱབ་སྟེ་ཐ་ན་བཀའ་བློན་དང་དྲུང་ཆེ་དེ་ཚོར་ཡང་བྱེད་རྒྱུ་གང་ཡང་མེད་པའི་བཟོ་འདྲ་བོ་ཞིག་ཆགས་བསྡད་པའི་ཚོར་སྣང་ཡོང་གི་འདུག ས་མཚམས་ལས་བརྒལ་བའི་གནས་ཚུལ་དེ་མང་གཙོའི་གཞུང་ཞིག་ལ་མཚོན་ན་ཁྱོན་ནས་འགྲིག་གི་ཡོད་པ་མ་རེད་ལ། སྤྱི་བའི་ལས་དོན་ཐོག་ལའང་འཐུས་ཤོར་དང་གྱོང་ཆེན་པོ་རེག་པའི་གཞི་རྩ་ཞིག་ཆགས་ཀྱི་ཡོད་པ་རེད།   དྲི་བ། སྲིད་སྐྱོང་གི་དངོས་གཞིའི་འོས་མིའི་ནང་བློ་བཟང་སེང་གེ་མཆོག་དང་སྤེན་པ་ཚེ་རིང་མཆོག་རྣམ་གཉིས་ཆགས་བསྡད་ཡོད་པ་རེད། ཁྱེད་རང་གིས་ཁོང་གཉིས་ལས་སུ་ལ་རྒྱབ་སྐྱོར་གནང་གི་ཡོད་པ་དང་། དེ་ལ་རྒྱབ་སྐྱོར་གནང་དགོས་པའི་རྒྱུ་མཚན་གང་ཡིན་ནམ།   ལན། ང་ད་བར་དུ་བོད་མིའི་སྒྲིག་འཛུགས་ནང་བསྡད་ཡོད་སྟབས་སུ་ལ་རྒྱབ་སྐྱོར་བྱེད་དགོས་ཀྱི་འདུག་དང་། སུ་ལ་རྒྱབ་སྐྱོར་བྱེད་དགོས་ཀྱི་མི་འདུག་ཅེས་པའི་སྐད་ཆ་དེ་འདྲ་སུ་གང་ལའང་ཞུས་མེད། ད་ལྟ་ང་སྒྲིག་འཛུགས་ལས་རྒན་ཡོལ་ཞུས་ཏེ་སྤྱི་འཐུས་ནང་ལང་མཁན་ཞིག་ཆགས་བསྡད་ཡོད་པ་རེད། ངས་ག་རེ་མཐོང་གི་འདུག་ཞུས་པ་ཡིན་ན། ཉེ་ཆར་ཚོགས་གཙོ་སྤེན་པ་ཚེ་རིང་ལགས་ཀྱིས་ཁོ་རང་སྲིད་སྐྱོང་ལ་བཞེངས་པའི་བསྒྲགས་གཏམ་ཞིག་མཐོང་སོང་། དེ་ཧ་ལས་པ་ཡག་པོ་མཐོང་སོང་། ཆབ་སྲིད་དང་སྤྱི་ཚོགས་སོགས་ལས་དོན་ག་རེ་ཡིན་ནའང་སྲིད་ཇུས་དང་ལངས་ཕྱོགས་བརྟན་པོ་དང་ཁ་གསལ་པོ། དམིགས་ཚད་དངོས་ཡོད་གནས་ཚུལ་དང་མཐུན་པ། དྲང་པོ་དྲང་བཞག་དང་མངོན་ལྐོག་ཕྱོགས་ལྷུང་མེད་པའི་ཐོག་ནས་གནང་རྒྱུ་ཡིན་པ་དང་། ལས་ཀ་གང་འདྲ་ཞིག་ལག་བསྟར་གནང་བའི་རྗེས་ལ་ངེས་པར་དུ་གྲུབ་འབྲས་འཐོན་པ་ཞིག་གནང་རྒྱུ་ཡིན་གསུངས་པ་དེ་དག་ང་ཚོ་སྐྱབས་བཅོལ་བ་ཡིན་པ་ངོ་ཤེས་པ་དང་། དངོས་ཡོད་གནས་ཚུལ་གཞི་ལ་བཞག་པའི་ཐོག་ནས་གསུངས་བཞག་པ་རྐྱང་རྐྱང་རེད། སྲིད་སྐྱོང་སུ་ཕེབས་པ་ཡིན་ནའང་དངོས་ཡོད་གནས་ཚུལ་གཞིར་བཞག་གནང་དགོས་པ་དེ་དཔེ་གལ་ཆེན་པོ་རེད། གང་ཡིན་ཟེར་ན། ང་ཚོས་གལ་ཆེ་ཤོས་དེ་ག་རེ་ཤེས་དགོས་ཀྱི་འདུག་ཟེར་ན། ང་ཚོའི་སྒྲིག་འཛུགས་འདི་མིའི་ལུང་པའི་ནང་ཡོད་པའི་སྒྲིག་འཛུགས་ཤིག་ཡིན་སྟབས་སོ་སོ་གནས་སའི་རྒྱལ་ཁབ་ཀྱི་དགོངས་པར་བསྟུན་དགོས་པ་རེད་མ་གཏོགས་ཧམ་པ་ཞེ་དྲགས་བྱེད་རྒྱུ་ཡོད་པ་མ་རེད། དེ་མ་གཏོགས་འདྲ་ཆགས་པོ་དང་གཏམ་བཤད་རྫིག་པོ་རྐྱང་རྐྱང་བྱས་ཏེ་དོན་དག་ངོ་མའི་ཐོག་འབབ་མ་ཐུབ་པ་ཡིན་ན། ནམ་ཞིག་ང་ཚོའི་གནས་སྟངས་དེ་ཐབས་སྐྱོ་པོ་ཞིག་ཆགས་མི་སྲིད་པ་མ་རེད།   དུས་དང་བསྟུན་པའི་གདོང་ལེན་གནང་དགོས་སར་འཛེམ་ཟོན་མེད་པའི་ཐོག་ནས་གདོང་ལེན་གནང་རྒྱུ་ཡིན་གསུངས་པ་དེ་གལ་ཆེན་པོ་ཞེ་དྲགས་རེད། བར་སྐབས་ཤིག་ནས་དོལ་རྒྱལ་གྱི་གནས་ཚུལ་བྱུང་ཡོད་པ་རེད། དེ་ཚོར་ད་ལྟ་དུས་དང་བསྟུན་པའི་གདོང་ལེན་གནང་ཐུབ་ཀྱི་འདུག་གམ་མི་འདུག དེ་༧རྒྱལ་བ་རིན་པོ་ཆེའི་དབུ་ལ་གཡུག་བཞག་ན་འགྲིག་གི་མ་རེད། དེ་དཔེ་འདྲ་པོ་ཞིག་རེད། དེ་བཞིན་བར་སྐབས་ཤིག་ལ་གཞོན་ནུའི་སྒྲིག་འཛུགས་ནང་དཀའ་ངལ་ཞིག་ཡོང་གི་ཡོད་པ་རེད། ངས་བསམ་ན་ང་ཚོ་ཚང་མར་སྒྲིག་འཛུགས་དེ་སྲུང་སྐྱོབ་བྱེད་དགོས་པའི་འགན་ཡོད་པ་རེད་བསམ་གྱི་འདུག སྤེན་པ་ཚེ་རིང་ལགས་ཀྱིས་ནུས་ཤུགས་ཚང་མ་མཉམ་སྤུང་གནང་གི་ཡིན་གསུངས་པ་དེའང་སེམས་ནང་འཇུག་པོ་བྱུང་སོང་། ང་རང་ཚོའི་གཏམ་དཔེའི་ནང་ལ་ཕོ་རབ་ཅིག་གི་བློ་རྩལ་ལས་ཕོ་འབྲིང་གསུམ་གྱི་གྲོས་བསྡུར་དགའ་ཟེར་བ་ལྟར། ང་ཚོ་ཚང་མ་མཉམ་རུབ་ཀྱི་ཐོག་ནས་གྲོས་ཚོགས་ཡིན་ན་འདྲ། འཛིན་སྐྱོང་ཡིན་ནའང་འདྲ། ཁྲིམས་ཞིབ་ཡིན་ནའང་འདྲ་དེ་ཚོ་ཚང་མ་ཐུན་མོང་གི་ཐོག་ནས་མཉམ་རུབ་བྱ་རྒྱུ་གལ་ཆེན་པོ་རེད། དབུ་ཁྲིད་ཟེར་བ་དེའི་ཁྱད་ཆོས་ཆེ་ཤོས་ངས་ག་རེ་རྩི་ཡི་ཡོད་ཟེར་ན། ཚང་མ་མཉམ་དུ་འཁྲིད་ཐུབ་མཁན་ཞིག་དགོས་པ་རེད་མ་གཏོགས་དབུ་ཁྲིད་གཏམ་བཤད་ཡག་པོ་བྱེད་ཐུབ་མཁན་ཞིག་དང་། མི་བབ་ཡག་པོ་ཡོད་པ་ཙམ་གྱིས་དབུ་ཁྲིད་ཡག་པོ་ཡོང་བའི་ཁྱབ་ཀྱི་མ་རེད།   གསར་འགོད་པ། པདྨ་མཚོ།
