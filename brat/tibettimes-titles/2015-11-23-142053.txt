གཏེར་ཁུངས་རྡིབ་སྐྱོན་ཁྲོད་མི་ ༡༠༤
རྐྱེན་ལམ་དུ་ཤོར།

  འབར་མའི་ནུབ་རྒྱུད་ཁུལ་དུ་ས་རུལ་ཉིལ་ཏེ་གཡུ་རྡོའི་གཏེར་ཁུངས་ཤིག་རྡིབ་སྐྱོན་བྱུང་སྟེ་ལས་བཟོ་བ་དང་ཡུལ་མི་ ༡༠༤ རྐྱེན་ལམ་དུ་ཤོར་བ་དང་མི་བརྒྱ་ཕྲག་མང་པོ་བོར་བརླག་ཕྱིན་ཡོད་འདུག   དེ་རིང་རྒྱལ་སྤྱིའི་བརྒྱུད་ལམ་ཆེ་ཁག་ཏུ་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། ཁ་སང་གཟའ་ཉི་མའི་དགོང་མོར་འབར་མའི་ནུབ་རྒྱུད་ཀྱི་ཁ་ཆེན་ (Kachin) མངའ་སྡེའི་ཧ་ཕ་ཁན་ཊི་ (Hpakant) གྲོང་རྡལ་ནང་ས་རུལ་ཉིལ་ཏེ་ཉེ་འཁྲིས་ཀྱི་གཡུ་རྡོའི་གཏེར་ཁུངས་ཤིག་རྡིབ་སྐྱོན་བྱུང་བའི་ཁྲོད་མི་ ༡༠༤ ཤི་རྨས་དང་མི་བརྒྱ་ཕྲག་མང་པོ་བོར་བརླག དེ་བཞིན་ཡུལ་མི་སྟོང་ཕྲག་མང་པོ་བདེ་འཇགས་ས་ཁུལ་དུ་གནས་སྤོར་དགོས་བྱུང་ཡོད་འདུག   དེ་ཡང་སིན་མེ་ (Seng Mai) ཟེར་བའི་ས་གནས་ཡུལ་མི་ཞིག་གིས་འབར་མའི་གསར་འགོད་པར། ཁ་སང་དགོང་མོ་ས་རུལ་ཉིལ་ཏེ་ཉེ་འཁྲིས་ཀྱི་གཡུ་རྡོའི་གཏེར་ཁུངས་ཤིག་རྡིབ་སྐྱོན་བྱུང་བའི་ཁྲོད་ལས་བཟོ་བ་དང་ཡུལ་མི་ ༩༡ ཙམ་གྱི་ཕུང་པོ་རྙེད་ཐུབ་ཡོད་ཀྱང་། ད་དུང་ཡུལ་མི་དང་ལས་མི་བརྒྱ་ཕྲག་མང་པོ་བོར་བརླག་ཕྱིན་ཡོད་ཡོད་སྐོར་བརྗོད་འདུག་ལ། ས་གནས་གཞུང་འབྲེལ་དཔོན་རིགས་ཤིག་གིས་ཉེ་ལམ་ནས་མངའ་ཁུལ་དེ་གར་རང་བྱུང་ཐོན་ཁུངས་མང་པོ་སྔོག་འདོན་བྱེད་ཀྱི་ཡོད་སྟབས་རི་སྒང་ཉིལ་བ་ཞིག་ཡིན་པ་ལས་ཆར་བ་བབ་ནས་ས་རུལ་ཉིལ་བ་མིན་སྐོར་བརྗོད་འདུག   གནས་ཚུལ་ལྟར་ན། ཁ་ཆེན་བྱ་བའི་མངའ་སྡེ་ནི་འབར་མའི་ནུབ་རྒྱུད་ཀྱི་རྒྱ་ནག་གི་མཐའ་མཚམས་སུ་འཁེལ་གྱི་ཡོད་པ་དང་། མངའ་སྡེ་དེའི་ནང་རང་བྱུང་ཐོན་ཁུངས་མང་པོ་ཡོད་པ་དང་། ལྷག་པར་དུ་འཛམ་གླིང་ནང་གི་གཡུ་རྡོ་ཡག་ཤོས་འབོར་ཆེན་འཐོན་བཞིན་ཡོད་སྟབས། འབར་མའི་ཡུལ་མི་དང་ཡུལ་ལུང་གཞན་ནས་ཡོང་བའི་ལས་མི་སྟོང་ཕྲག་མང་པོ་གཏེར་ཁ་སྔོག་འདོན་གྱི་ལས་ཀར་ཞུགས་ཡོད་ཀྱང་། གཏེར་ཁ་སྔོག་འདོན་གྱི་མཐུན་རྐྱེན་ཤུན་ཏུ་ཞན་སྟབས་ལས་བཟོ་བ་ཚོའི་ཚེ་སྲོག་ལ་ཉེན་ཁ་ཆེན་པོ་ཡོད་འདུག   ༢༠༡༤ ལོར་འབར་མའི་དམག་གཞུང་གཞོགས་གཏོགས་གཏེར་འདོན་ཚན་པ་ཞིག་གི་དོ་དམ་འོག་མངའ་ཁུལ་དེའི་རང་བྱུང་ཐོན་ཁུངས་ཁག་ནས་ཨ་སྒོར་ཐེར་འབུམ་ ༣༡ གཏེར་རྫས་ཐོན་སྐྱེད་བྱས་ཡོད་པ་ལས་བརྒྱ་ཆ་ ༤༨ འབར་མའི་དམག་གཞུང་གཞོགས་གཏོགས་དཔོན་རིགས་ཚོས་ལྐོག་ཟ་བྱས་ཡོད་པ་མ་ཟད། ཉེ་ལམ་ནས་འབར་མའི་དམག་གཞུང་གིས་དྲག་ཆས་དམག་དཔུང་འབོར་ཆེན་བཏང་སྟེ་རང་བྱུང་ཐོན་ཁུངས་ས་ཚིགས་ཁག་ཏུ་དམ་དྲག་བྱས་ཡོད་པ་དང་། དེང་སྐབས་རྒྱ་ནག་ནང་གཡུ་རྡོ་དེའི་རིགས་ལ་རིན་གོང་མཐོན་པོ་སྤྲོད་ཀྱི་ཡོད་སྟབས། མངའ་ཁུལ་དེའི་ཐོན་ཁུངས་ཁག་ནས་བཏོན་པའི་གཡུ་རྡོ་མང་ཆེ་བ་རྒྱ་ནག་ནང་སྐྱེལ་འདྲེན་བྱེད་ཀྱི་ཡོད་འདུག   གསར་འགོད་པ། པདྨ་རྒྱལ།
