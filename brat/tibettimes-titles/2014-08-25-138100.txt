རྒྱ་གཞུང་གི་གཡོ་འོག་ཏུ་མ་ཚུད་པ་བྱོས།

སྤྱི་འཐུས་ཚོགས་གཙོས་སྣེ་འཁྲིད་འོག་སྤྱི་འཐུས་ཁག་ཅིག་ཨོ་སི་ཊོ་ལི་ཡར་གཟིགས་སྐོར་དང་འབྲེལ་བོད་རིགས་མི་མང་ལ་གཏམ་བཤད་དང་རྒྱ་རིགས་ཤེས་ཡོན་ཅན་སོགས་དང་བགྲོ་གླེང་གནང་ཡོད། ཟླ་འདིའི་ཚེས་ ༢༣ ཉིན་སྤྱི་འཐུས་ཚོགས་གཙོ་སྤེན་པ་ཚེ་རིང་མཆོག་གི་གཙོས་བཀའ་ཟུར་སྤྱི་འཐུས་ཐུབ་བསྟན་ལུང་རིགས་ལགས་དང་། སྤྱི་འཐུས་བསོད་ནམས་རྒྱལ་མཚན་ལགས། སྤྱི་འཐུས་རྣམ་དག་གཙུག་ཕུད་ལགས་བཅས་ཨོ་སི་ཊོ་ལི་ཡའི་གྲོང་ཁྱེར་སི་ཌེ་ནི་ནས་མགོ་བཙུགས་ཏེ་འཚམས་གཟིགས་གནང་བཞིན་ཡོད་པ་དང་། སྐབས་དེར་སྤྱི་འཐུས་ཚོགས་གཙོ་མཆོག་གིས་ས་གནས་བོད་མི་རྣམས་ལ་གསུང་བཤད་གནང་དོན། འདི་གར་ཁྱེད་རང་ཚོ་དང་ཐུག་འཕྲད་བྱེད་རྒྱུ་དེ་ནི་བར་ལམ་ནས་རྒྱ་ནག་གཞུང་གིས་འཆར་གཞི་ལྡན་པའི་ཐོག་ནས་བོད་མི་ནང་ཁུལ་དཀྲུག་ཤིང་རྒྱག་རྒྱུའི་ལས་གཞིའི་ཐོག་ཨོ་སི་ཊོ་ལི་ཡར་གནས་སྡོད་ཆབ་སྲིད་བཙོན་ཟུར་ཁག་གཅིག་གིས་རྒྱ་ནག་གཞུང་དང་འབྲེལ་བ་བྱས་བ་དང་གཡོ་འོག་ལ་ཚུད་བཞིན་ཡོད་པ་དེ་ནི་ཧ་ཅང་ཐབས་སྐྱོ་པོ་རེད། སྤྱིར་བཏང་ཆབ་སྲིད་བཙོན་ཟུར་རྣམས་བོད་ནང་རྒྱ་ནག་གཞུང་གི་ཁ་གཏད་ལ་ལངས་ཏེ་བཙན་བྱོལ་དུ་ཡོང་བ་དང་། དེ་རྗེས་རྒྱལ་ཁབ་འདིའི་ནང་སྐྱབས་བཅོལ་དུ་བསླེབས་ཟིན་རྗེས་སླར་ཡང་རྒྱ་ནག་གཞུང་དང་འབྲེལ་བ་བྱེད་ཀྱི་ཡོད་པའི་གནས་ཚུལ་དེ་ནི་ཧ་ཅང་བློ་ཕམ་དགོས་པ་ཞིག་རེད། དཔེར་ན། དཔལ་འབྱོར་ཚེ་རིང་ཟེར་བ་དང་། འཕྲིན་ལས་རིན་ཆེན་ཟེར་བའི་རིགས་ནི་བོད་མི་དམངས་ཀྱི་ཞབས་འདྲེན་ཞུ་མཁན་རེད། མི་སྣ་དེ་དང་ལྷན་འབྲེལ་བས་མཚོན་མཉམ་དུ་འགྲོ་རྒྱུར་བསམ་བློ་ཏན་ཏན་གཏོང་དགོས་རྒྱུ་ཧ་ཅང་གལ་ཆེན་པོ་ཆགས་ཡོད། ཅེས་སོགས་ཀྱི་གསུངས་ཡོད་འདུག ཚེས་ ༢༤ ཉིན་གྱི་ཕྱི་དྲོར་ཨོ་སི་ཊི་ལི་ཡའི་རྒྱལ་ཡོངས་བོད་རིགས་སྤྱི་མཐུན་ཚོགས་པས་གོ་སྒྲིག་འོག་རྒྱ་རིགས་མཁས་པ་དང་བོད་དོན་རྒྱབ་སྐྱོར་བ་ ༦༠ ཡས་མས་དང་། བོད་མི་སྐོར་ཞིག་བཅས་འདུ་འཛོམས་བྱས་ཤིང་། ནུབ་ཨོ་སི་ཊི་ལི་ཡའི་གཙུག་ལག་སློབ་གཉེར་ཁང་གི་ཆབ་སྲིད་ཚན་རིག་གི་གཙུག་ལག་སློབ་དཔོན་གཞོན་པ་ཧྤང་ཆོན་ཡེན་ལགས་ཀྱིས་གཙོ་སྐྱོང་དང་། རྒྱ་བོད་མཛའ་འབྲེལ་ཚོགས་པའི་ཚོགས་གཞོན་མེའེ་ལེ་ལགས་ཀྱིས་བོད་རྒྱ་གཉིས་ཡུན་རིང་འབྲེལ་ཟབ་དང་ཉེ་རབས་རྒྱ་ནག་དམར་ཤོག་གིས་བོད་མིར་མནར་གཅོད་བཏང་བར་དགོངས་སེལ། ༧གོང་ས་མཆོག་གིས་བོད་རྒྱ་གཉིས་ཕན་གྱི་དབུ་མའི་ལམ་གྱི་སྲིད་བྱུས་བཏོན་པར་རྒྱབ་སྐྱོར་དང་རྒྱ་མི་རིགས་ལ་དེའི་གོ་རྟོགས་སྤེལ་དགོས་པ་གལ་ཆེ་སོགས་གཏམ་བཤད་གནང་ཡོད་འདུག དེ་ནས་སྤྱི་འཐུས་ཚོགས་གཙོ་མཆོག་གིས་ལོ་རྒྱུས་ཐོག་གི་བོད་རྒྱའི་འབྲེལ་བ་དང་། བོད་ནང་གི་ཛ་དྲག་གི་གནས་བབ། བོད་རྒྱའི་དཀའ་རྙོག་སེལ་ཐབས་ཀྱི་དབུ་མའི་ལམ་གྱི་སྲིད་ཇུས། བོད་ཀྱི་ཁོར་ཡུག་གི་གནས་བབ། བཙན་བྱོལ་བོད་མིའི་དམངས་གཙོའི་འཕེལ་རིམ། བོད་ཀྱི་ཁོར་ཡུག་ནི་ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་ཁག་དང་འབྲེལ་བ་ཅི་ཡོད་བཅས་ཀྱི་བཀའ་སློབ་ཟབ་རྒྱས་གནང་བ་དང་། དེ་རྗེས་དྲི་བ་དྲི་ལན་གྱི་ལས་རིམ་ཚུགས་ཤིང་དེར་རྒྱ་རིགས་ཚོགས་ཞུགས་པ་རྣམ་པས་རང་རང་གི་བསམ་ཚུལ་དང་དྲི་བ་དང་གཏམ་བཤད། རྒྱ་རིགས་ཚོས་བསམ་ཚུལ་གཅིག་མཐུན་ནི་དབུ་མའི་ལམ་ནི་བོད་རྒྱའི་དཀའ་རྙོག་སེལ་ཐབས་ཀྱི་ལམ་ཁ་ཉག་ཅིག་ཡིན་པ་དང་། དེར་རྒྱབ་སྐྱོར་ཞུ་རྒྱའི་ཆོད་སེམས་བརྟན་པོ་མ་ཟད། བོད་ཀྱི་ཁོར་ཡུག་དང་རིག་གཞུང་སྲུང་སྐྱོབས་བོད་ཙམ་མ་ཟད་རྒྱ་ནག་དང་འཛམ་གླིང་ཁྱོན་ཡོངས་ལ་གལ་ཆེ་ཡིན་པ་བཅས་ཀྱི་གཏམ་བཤད་སྤེལ་ཡོད་འདུག
