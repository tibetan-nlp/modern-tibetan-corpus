གཞས་མ་རིག་འཛིན་སྒྲོལ་མའི་སེམས་གཏམ།

སུད་སིའི་མདོ་སྨད་ས་གནས་རྒྱུན་ལས་དབུ་བརྙས་པའི་མཛད་སྒོའི་སྟེང་བོད་ནས་ཕེབས་པའི་གཞས་མ་རིག་འཛིན་སྒྲོལ་མ་ལགས་ཀྱིས་བོད་ཕྱི་ནང་གཉིས་སུ་གཞོན་སྐྱེས་ཚོས་བོད་ཡིག་བོད་སྐད་ཐོག་ཤུགས་བསྐྱོན་དགོས་ཚུལ་བརྗོད་འདུག ཟླ་འདིའི་ཚེས་ ༨ ཉིན། སུད་སི་ས་གནས་མདོ་སྨད་རྒྱུན་ལས་ཁང་དབུ་བརྙེས་ནས་ལོ་འཁོར་ ༣༠ འཁོར་བའི་དུས་དྲན་མཛད་སྒོ་ཞིག སུད་སི་ས་གནས་བུན་ཏར་ཐོར་ Winterthur ནང་བསྡུས་ཡོད་པ་དང་། མཛད་སྒོའི་ཐོག་ཆེད་དུ་བོད་ནས་གདན་འདྲེན་ཞུས་པའི་གླུ་མ་རིག་འཛིན་སྒྲོལ་མ་དང་། ཨ་རིར་གནས་སྡོད་གཞས་པ་དང་སྒྱུ་རྩལ་བ་རྟ་མགྲིན། སུད་སིར་གནས་སྡོད་གཞས་མ་བསམ་འགྲུབ་ལྷ་སྐྱིད་དང་འཇམ་དབྱངས་དཔལ་ལྡན། རྒན་གྱའི་ཚེ་རིང་སྐྱིད། བསྟན་འཛིན། བཀྲ་ཤིས་སྒྲོལ་མ། དཔལ་རྩེ་སོགས་གླུ་བ་མ་གླུ་མང་པོ་ཕེབས་ཡོད་འདུག མཛད་སྒོའི་སྐུ་མགྲོན་དུ་རི་ཁོན་ཆོས་འཁོར་དགོན་གྱི་མཁན་རིན་པོ་ཆེ་དང་། སུད་སི་སྐུ་ཚབ་དོན་གཅོད། དེ་བཞིན། སུད་སི་དང་ལིག་ཏེན་སིཊཡན་བོད་རིགས་མི་མང་ཚོགས་གཙོའི་སྐུ་ཚབ། གཞུང་འབྲེལ་དང་གཞུང་འབྲེལ་མ་ཡིན་པའི་ཚོགས་པ་ཁག་གི་ཚོགས་གཙོ་དང་སྐུ་ཚབ། སུད་སི་དང་ཡུ་རོབ་རྒྱལ་ཁབ་གཞན་ནས་ཕེབས་པའི་བོད་མི་ ༣༠༠ ལྷག་མཉམ་ཞུགས་གནང་ཡོད་འདུག མཛད་སྒོ་གྲུབ་མཚམས་བོད་མི་ཡོངས་རྫོགས་མཉམ་དུ་སྐོར་གཞས་སོགས་ལ་རོལ་ཏེ། སྐུ་མགྲོན་དང་མི་མང་ཡོངས་དགའ་མགུ་ཡི་རང་ཆེན་པོ་བྱུང་ཡོད་འདུག བོད་ཀྱི་གཞས་མ་གྲགས་ཅན་རིག་འཛིན་སྒྲོལ་མ་ལགས་སུ། ཁོང་མོ་ཡུ་རོབ་སུད་སིར་ཕེབས་ཏེ་ཕྱི་སྡོད་བོད་མིའི་སྤྲོ་ཚོགས་སྟེང་ཞུགས་པའི་ཚོར་བ་ཅི་བྱུང་དྲིས་པར་ཁོ་མོས། ང་རང་ཕྱི་རྒྱལ་གྱི་རྒྱལ་ཁབ་ཁ་ཤས་ཤིག་ལ་སོང་བ་ཡིན། ས་ཆ་གང་དུ་སོང་ནའང་དེ་གའི་མི་མང་ལ་བསམ་ཚུལ་རེ་རྗོད་ཀྱིན་ཡོད། ཐེངས་འདིར་ཡུ་རོབ་སུད་སི་ལ་སླེབས་རྗེས་ཚོར་བ་མི་འདྲ་བ་མང་པོ་སྤྲད་བྱུང་། ཡག་ས་ནས་བཤད་ན་ཕྱི་རྒྱལ་དུ་ཡོད་པའི་བོད་པ་ཚོས་མི་རིགས་རང་གི་ངོ་བོ་དང་རྣམ་པ་ཡག་པོ་ཟིན་ཐུབ་འདུག་པ་མ་ཟད། བོད་ནང་ལས་ལྷག་པའི་ཡག་པོ་འདུག་བསམ་པའི་ཚོར་བའང་ཡོང་གི་འདུག བོད་ནང་གི་བོད་པ་མང་པོ་ཞིག་གིས་ཀྱང་རང་ཉིད་ཀྱི་གོམས་གཤིས་དང་སྐད་ཡིག་སོགས་ལ་མཐོང་ཆེན་དང་གོམས་གཤིས་མཁྲེགས་པོ་འཛིན་ཐུབ་ཀྱིན་མེད། འོན་ཀྱང་སྡུག་ས་ནས་བཤད་ན། ང་རང་འདི་ནས་ཁྱིམ་ཚང་ཁག་ཅིག་གི་ནང་སོང་བ་ཡིན། ཁྱིམ་ཚང་ཁ་ཤས་ནང་གི་བུ་ཕྲུག་ཚོས་བོད་སྐད་ཡག་པོ་བཤད་ཤེས་ཀྱི་མེད་པར་ཕྱི་རྒྱལ་གྱི་སྐད་ཤོད་ཀྱིན་འདུག བོད་ནང་དུའང་དེ་བཞིན་རྒྱ་སྐད་མ་གཏོགས་མ་བོད་སྐད་ཁ་ནང་ལ་བདེ་བོ་མེད་པ་ལྟ་བུར་འགྱུར་བཞིན་མང་པོ་རེད། ཡིན་ནའང་ང་ཚོས་བོད་རང་གི་སྐད་དང་ཡི་གེ་ལ་བདག་པོ་ཡག་པོ་རྒྱག་དགོས་རྒྱུ་དང་། དེ་ལ་ཕ་མ་དེ་ཚོས་འགན་འཁུར་དགོས་རྒྱུ་དེ་གལ་ཆེན་རེད། ཅེས་ཟེར། གླུ་མ་རིག་འཛིན་སྒྲོལ་མ་ལགས་ནི་བོད་ཀྱི་གཞས་མ་ཞིག་ཡིན་པ་མ་ཟད། བོད་སྐད་ཡིག་སྲུང་སྐྱོབ་ཕྱག་ལས་གནང་མཁན་ཞིག་ཀྱང་ཡིན་འདུག་ལ། ཁོ་མོས་བོད་ནང་ཕ་སྐད་འཛིན་གྲྭ་སོགས་བོད་སྐད་ཡིག་སྲུང་སྐྱོབ་དང་འབྲེལ་བའི་ཕྱག་ལས་མང་པོ་གནང་གི་ཡོད་འདུག སུད་སིའི་ས་གནས་མདོ་སྨད་རྒྱུན་ལས་ཁང་འདི་ཕྱི་ལོ་ ༡༩༨༦ ལོའི་ཟླ་ ༥ ཚེས་ ༢༣ ཉིན་དབུ་བརྙེས་ཡོད་པ་དང་། ཐོག་མར་སྒྲིག་འཛུགས་འདི་ཉིད་གསར་འཛུགས་གནང་མཁན་གྲས་སུ་རི་ཀོང་ཆོས་འཁོར་དགོན་གྱི་མཁན་རིན་པོ་ཆེ་ཞིང་གཤེགས་དགེ་འདུན་བཟང་པོ་མཆོག་དང་གཅོད་པ་ཐར་རིན་ཆེན། གདོང་རྩེ་བསོད་ནམས་རབ་བརྟན་པ། བསོད་ནམས་རབ་རྒྱས། ཕྱོགས་དགེ་རྟ་མགྲིན། རིག་བདག་ཚང་ཡི་དམ་སོགས་ཡོད་འདུག ཕྱི་ལོ་ ༡༩༩༣ ལོའི་ ཟླ་ ༤ ཚེས་ ༢༩ ཉིན་མདོ་སྨད་སྒྲིག་འཛུགས་ཀྱི་ཚོགས་ཆེན་ཐེངས་བརྒྱད་པ་ཚོགས་རྗེས་ཚོགས་པའི་ངོ་བོའི་ཐོག་སུད་སི་གཞུང་གི་ཁྲིམས་ལུགས་ལྟར་ཐོ་འགོད་བྱས་ཡོད་པ་དང་། སུད་སི་མདོ་སྨད་རྒྱུན་ལས་ཁང་འདི་ཉིད་འཛུགས་དགོས་པའི་དམིགས་ཡུལ་ནི། ༧གོང་ས་སྐྱབས་མགོན་རིན་པོ་ཆེ་མཆོག་དང་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་རླབས་ཆེན་དབུ་ཁྲིད་འོག་དབུས་མདོ་སྨད་ཆོལ་ཁའི་བཀའ་ཁྱབ་བཞིན་སུད་སིར་གནས་བཞུགས་མདོ་སྨད་པ་ཡོངས་ཀྱིས་བོད་མིའི་ལས་འགན་ལེན་ཐུབ་དང་བོད་ཀྱི་རིག་གཞུང་ལ་ཞབས་འདེགས་བསྒྲུབ་ཐབས་བཅས་སུ་བཙུགས་པ་ཞིག་ཡིན་འདུག གསར་འགོད་པ། ཙ་ཡུས་ཕུན་ཚོགས།
