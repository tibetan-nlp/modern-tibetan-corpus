འགྲོ་བ་མིའི་ཐོབ་ཐང་གི་དཀའ་སྡུག

  མཉམ་འབྲེལ་རྒྱལ་ཚོགས་སྐྱབས་བཅོལ་རྟོགས་ཞིབ་ཚོགས་ཆུང་གིས་སྙན་ཐོ་བཏོན་པའི་ནང་རྒྱ་ནག་དང་ལྷོ་ཨ་ཧྥི་རི་ཁ། དེ་བཞིན་སེ་རི་ཡ་སོགས་ཀྱི་གནས་སྟངས་ཧ་ཅང་སྐྱོ་བོ་ཡིན་པའི་གསལ་བསྒྲགས་སྤེལ་འདུག   གཟའ་ཕུར་བའི་ཉིན་འཛམ་གླིང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱིས་འགྲོ་བ་མིའི་ཐོབ་ཐང་ཡོངས་ཁྱབ་གསལ་བསྒྲགས་སྤེལ་ནས་མི་ལོ་ ༦༧ འཁོར་བའི་ཉིན་མོར། མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་ཆེས་མཐོའི་སྐྱབས་བཅོལ་རྟོག་ཞིབ་ཚོགས་པའི་དྲ་ཚིགས་ (UNHCR) སྟེང་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། འཛམ་གླིང་སྟེང་མི་འབོར་ཆེ་ཤོས་དང་འགྲོ་བ་མིའི་ཐོབ་ཐང་གི་གནས་སྟངས་སྡུག་ཤོས་ཡིན་པའི་རྒྱ་ནག་ནང་སྨྲ་བརྗོད་རང་དབང་དང་གསར་འགོད་རང་དབང་། འདུ་འཛོམས་རང་དབང་གི་གནས་སྟངས་སྡུག་ཏུ་ཕྱིན་ཡོད་པ་མ་ཟད། གསར་དུ་འབྱུང་བཞིན་པའི་གནད་དོན་ཞིག་ནི་འདི་ལོའི་ཕྱི་ཟླ་ ༦ པ་ནས་བཟུང་རྒྱ་གཞུང་གིས་འགྲོ་བའི་ཐོབ་ཐང་གི་ཁྲིམས་རྩོད་པ་དང་རང་དབང་དམངས་གཙོ་དོན་གཉེར་བ་མི་ ༢༠༠ ལྷག་འཛིན་བཟུང་བྱས་ཡོད་པའི་གྲས་ཀྱི་མི་ ༢༠ ཙམ་ད་ལྟ་ད་དུང་བཀག་བསྐྱིལ་བྱས་ཡོད་འདུག   མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་མནར་གཅོད་བཀག་སྡོམ་ཚོགས་ཆུང་ (UNCAT) གིས། རྒྱ་ནག་གཞུང་གིས་རྒྱལ་ནང་གི་ཆབ་སྲིད་བཙོན་པ་ཚོའི་ལུས་སེམས་གཉིས་གར་མནར་གཅོད་དང་འཚེ་བ་བྱེད་བཞིན་པར་སེམས་ཁུར་ཡོད་སྐོར་དང་། ཕྱི་ལོ་ ༢༠༠༨ ལོ་ནས་བཟུང་རྒྱ་གཞུང་གིས་རྒྱལ་ནང་གི་ཆབ་སྲིད་བཙོན་པ་ཚོར་མི་སྤྱོད་ལས་འདས་པའི་མནར་གཅོད་གཏོང་མུས་ཡིན་ཡང་། རྒྱ་ནག་གཞུང་གིས་ད་དུང་དོན་དངོས་ལ་གཡོལ་ཐབས་བྱེད་ཀྱི་ཡོད་པ་དང་། དེང་སྐབས་རྒྱ་ནག་གི་བཙོན་ཁང་ཁག་ཏུ་བཙོན་པ་ཚོ་གློ་བུར་འདས་གྲོངས་དང་བོར་བརླག་འགྲོ་བཞིན་པའི་གནས་སྟངས་ཧ་ཅང་ཚབས་ཆེན་ཡིན་སྐོར་སོགས་བརྗོད་འདུག   གཞན། འདས་པའི་ལོ་ངོ་གསུམ་གྱི་གོང་ནས་དབུས་ཨ་ཧྥི་རི་ཁ་སྤྱི་མཐུན་རྒྱལ་ཁབ་ (CAR) ནང་ཟིང་འཁྲུག་ལངས་ཏེ་འགྲོ་བ་མིའི་ཐོབ་ཐང་གི་གནས་སྟངས་སྡུག་ཏུ་ཕྱིན་པ་དང་། འདི་ལོའི་ཟླ་བ་ ༡༠ ནང་འགྲོ་བ་མིའི་ཐོབ་ཐང་རྡོག་རོལ་ཐེབས་པའི་གནད་དོན་ ༦༠༠༠ ལྷག་བྱུང་ཡོད་པ་ལས་མི་ ༤༤༧༠༠༠ གནས་དབྱུང་བཏང་ཡོད་འདུག་ལ། དོན་རྐྱེན་འདིའི་བརྒྱ་ཆ་ ༧༠ ཕྱོགས་འགལ་ཚོགས་ཁག་བར་གྱི་ཞེ་འཁོན་དང་འཁྲུག་འཛིང་ལས་བྱུང་བ་ཡིན་ཞིང་། འགྲོ་བ་མིའི་ཐོབ་ཐང་དང་འབྲེལ་བའི་གནད་དོན་ ༦༠༠༠༠ ལྷག་ནི་རྒྱལ་ཁབ་དེའི་ནང་ཕོ་མཆོག་མོ་དམན་གྱི་རྩོད་གླེང་ཡིན་པ་དང་། དེའི་ཁྲོད་ཀྱི་གནད་དོན་ ༣༠༠༠༠ ལྷག་བཙན་གཡེམ་དང་ཆགས་སྲེད་ཀྱི་བརྙས་བཅོས་དང་འབྲེལ་བ་ཡོད་འདུག་ལ། རྒྱལ་ཁབ་དེའི་ནང་ཉིན་གཅིག་ལ་ཡུལ་མི་བརྒྱ་ཕྲག་མང་པོ་འགྲོ་བ་མིའི་ཐོབ་ཐང་གི་དཀའ་རྙོག་འཕྲད་མུས་ཡིན་འདུག   དེ་མིན་ཟླ་འདིའི་ཕྱི་ཚེས་ ༨ ཉིན་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་སྐྱབས་བཅོལ་ལྟ་རྟོག་ཚན་པས། ཨི་རག་དང་སི་རེ་ཡ། ཇོར་ཌན་རྒྱལ་ཁབ་བཅས་ཀྱི་མཐའ་མཚམས་ཁག་ནས་སི་རེ་ཡའི་སྐྱབས་བཅོལ་བ་མི་ ༦༣༠༠༠༠ སྣེ་ལེན་བྱས་ཡོད་པ་དང་། ད་ཆར་སི་རེ་ཡའི་ཡུལ་མི་ ༡༢༠༠༠ གཞན་ཡུལ་དུ་བྲོས་བྱོལ་དུ་འགྲོ་འཆར་ཡོད་པ་དང་། འདས་པའི་བདུན་ཕྲག་འདི་དག་ནང་ཨི་རག་དང་སི་རེ་ཡ། ཇོར་ཌན་རྒྱལ་ཁབ་བཅས་ཀྱི་མཐའ་མཚམས་ཁག་ཏུ་སྐྱབས་བཅོལ་བ་ ༡༢༠༠༠ ཡོད་པའི་གྲས་ཀྱི་མང་ཆེ་བ་རྒན་པ་དང་བྱིས་པ། བུད་མེད་དང་རྨས་མ། ནད་པ་སོགས་ཡིན་འདུག   གསར་འགོད་པ། པདྨ་རྒྱལ།
