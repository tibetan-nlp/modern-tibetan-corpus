ཆུ་ལོག་གི་གནོད་སྐྱོན་ཚབས་ཆེན་ཕོག

རྒྱ་ནག་ཧྥུ་ཅན་ཞིང་ཆེན་མེན་ཆིན་རྫོང་ཁོངས་ཀྱི་ཆུ་མཛོད་ཁང་གསུམ་གློ་བུར་དུ་ཤོར་ཏེ་མི་རྒྱུ་གཉིས་ལ་གནོད་སྐྱོན་ཚབས་ཆེན་བཟོས་ཡོད་འདུག ད་ལོའི་ཕྱི་ཟླ་ ༦ པའི་ཟླ་མཇུག་ནས་བཟུང་སྟེ་རྒྱ་ནག་ལྷོ་བརྒྱུད་ཀྱི་ཞིང་ཆེན་ཁག་གི་གཙོས་རྒྱལ་ཡོངས་ཀྱི་ཞིང་ཆེན་མང་པོའི་ནང་བསྟུད་མར་དྲག་ཆར་བབས་ཏེ། ཆུ་ལོག་ས་རུད་ཀྱི་གནོད་སྐྱོན་རིམ་པར་བྱུང་བཞིན་ཡོད་པ་ལྟར། ཁ་སང་ཨ་རིའི་ནང་ཡོད་པའི་ཞིན་ཐང་རིན་བརྙན་འཕྲིན་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། རྒྱ་ནག་ཤར་ལྷོའི་ཕྱོགས་སུ་ཡོད་པའི་ཧྥུ་ཅན་ཞིང་ཆེན་ནང་ཉིན་གྲངས་མང་པོར་དྲག་ཆར་བབས་ཏེ། མེན་ཆིན་རྫོང་ཁོངས་ཀྱི་ཆུ་མཛོད་ཁང་གསུམ་གློ་བུར་དུ་ཤོར་ཏེ་མི་རྒྱུ་གཉིས་ལ་གནོད་སྐྱོན་ཚབས་ཆེན་ཕོག་ཡོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༡༡ ཉིན་ཧྥུ་ཅན་ཞིང་ཆེན་ཆུ་ལོག་སྔོན་འགོག་ལས་སྒྲུབ་ཁང་གིས། ཞིང་ཆེན་ཡོངས་ཀྱི་གྲོང་ཁྱེར་ཁག་ ༩ ཡི་རྫོང་ ༥༦ ནང་མི་ཁྲི་ ༦༠ ལྷག་ལ་གནོད་སྐྱོན་ཐེབས་ཡོད་པ་མ་ཟད། གྲོང་ཁྱེར་ ༦ ཆུས་བསྣུབས་ཡོད་པའི་ཁྲོད། མི་ ༡༡ རྐྱེན་འདས་དང་མི་ ༢༣ གར་སོང་ཆ་མེད་དུ་གྱུར་ཡོད་པ་དང་། ཐེངས་འདིའི་གནོད་སྐྱོན་ཁྲོད་རྒྱ་སྒོར་དུང་ཕྱུར་ ༦༥ ལྷག་གི་གྱོང་གུད་ཐེབས་ཡོད་སྐོར་བསྒྲགས་ཡོད་ཀྱང་། མེན་ཆིན་རྫོང་པན་ཏུང་གྲོང་བརྡལ་གྱི་ཡུལ་མི་ཚོས་གྲོང་བརྡལ་དེ་གཅིག་པུའི་ནང་ནས་མི་གྲངས་ ༢༠༠ ལྷག་རྐྱེན་འདས་སུ་ཕྱིན་པ་ཤེས་གསལ་ཡིན་ལ། ཚོད་རྩིས་བྱས་ན་མེན་ཆིན་རྫོང་ཡོངས་ནས་མི་སྟོང་ཕྲག་གཅིག་གཉིས་ཤིག་རྐྱེན་འདས་སུ་ཕྱིན་ཡོད་སྐོར་འགྲེལ་བརྗོད་བྱས་འདུག ཕྱི་ཚེས་ ༡༢ ཉིན་པན་ཏུང་གྲོང་བརྡལ་གྱི་ཡུལ་མི་ཞིག་གིས། ཆུ་མཛོད་ཤོར་བའི་གནད་དོན་ལ་སྔོན་བརྡ་གཏོང་ཐུབ་པ་ཞིག་ཡིན་ལ། གནོད་སྐྱོན་བྱུང་རྗེས་མི་མང་གིས་སོ་སོར་མྱུར་སྐྱོབ་བྱེད་ཀྱི་ཡོད་པ་མ་གཏོགས་སྲིད་གཞུང་གི་རོགས་སྐྱོབ་གང་ཡང་བྱེད་ཀྱི་མེད་པ་དང་། ད་ལན་གནོད་སྐྱོན་ཐེབས་ཡུལ་ཚབས་ཆེ་ཤོས་དེ་གཞི་རིམ་གྲོང་བརྡལ་ཁག་ཡིན་ལ། མི་རྐྱེན་འདས་ཕྱིན་པ་དང་རྨས་སྐྱོན་བྱུང་བ་མང་དྲགས་ནས་རོ་ཁང་དང་སྨན་ཁང་སོགས་སུ་མ་ཤོང་བར་རོ་ཁང་མེད་པའི་དམག་མིའི་སྨན་ཁང་དུའང་ཕུང་པོ་བཅོལ་འཇོག་བྱེད་དགོས་བྱུང་ཡོད་པ་དང་། ཕུང་པོ་མང་པོ་བདག་པོ་མེད་ལ། ངོ་ཐག་མ་ཆོད་པར་མེར་བསྲེགས་བཏང་ཡོད་པ་དང་། ད་ལྟའི་ཆར་གྲོང་བརྡལ་ཡོངས་ཀྱི་འགྲོ་འོང་བཀག་སྡོམ་བྱས་ཏེ་དུག་སེལ་བྱེད་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག རྒྱ་ནག་དམངས་སྲིད་ཁང་གི་བསྡོམས་རྩིས་ལྟར་ན། འདི་ལོའི་ཕྱི་ཟླ་ ༦ པའི་ཟླ་མཇུག་ནས་ཟླ་འདིའི་ཕྱི་ཚེས་ ༨ ཉིན་བར་དུ། རྒྱ་ནག་རྒྱལ་ཡོངས་ཀྱི་གྲོང་དང་གྲོང་བརྡལ་ ༥༠༠ བརྒལ་བའི་ཡུལ་མི་ཁྲི་ ༣༡༠༠ ལྷག་ལ་ཆུ་ལོག་གི་གནོད་སྐྱོན་ཐེབས་ཡོད་པ་དང་། ཆུ་ལོག་གི་གནོད་སྐྱོན་ཁྲོད་མི་གྲངས་ ༡༦༤ རྐྱེན་འདས་དང་མི་ ༢༦ བར་བརླག ཁང་པ་ཁྲི་ ༧ ལྷག་ལ་གནོད་སྐྱོན་ཐེབས་ཡོད་སྐོར་བསྒྲགས་ཡོད་པ་དང་། རྒྱ་ནག་གཞུང་འབྲེལ་ལས་ཁུངས་ཁག་གི་སྔོན་བརྡ་ལྟར་ན། ད་ལོར་ཕྱི་ལོ་ ༡༩༩༨ ལོར་འབྲི་ཆུའི་ཆུ་ལོག་གི་གནོད་སྐྱོན་ལྟ་བུ་ཞིག་འབྱུང་བའི་ཉེན་ཁ་ཆེན་པོ་ཡོད་སྐོར་བརྗོད་འདུག་ལ། ཕྱི་ལོ་ ༡༩༩༨ ལོར་རྒྱ་ནག་གི་ཞིང་ཆེན་དང་གྲོང་ཁྱེར་ཁག་ ༢༩ ནང་འབྲི་ཆུའི་ཆུ་ལོག་གི་གནོད་སྐྱོན་ཆེན་པོ་བྱུང་སྟེ། ཉུང་མཐར་མི་གྲངས་ ༤༡༥༠ ཡི་ཚེ་སྲོག་ཤོར་ཡོད་པ་དང་། དཔལ་འབྱོར་ཐད་རྒྱ་སྒོར་དུང་ཕྱུར་ ༢༥༥༡ གི་གྱོང་གུན་ཕོག་ཡོད་པ་རེད། གསར་འགོད་པ། པདྨ་མཚོ།
