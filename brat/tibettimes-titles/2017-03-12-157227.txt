ཀླུ་མཁར་བྱམས་ལ་ངོ་རྒོལ་བྱས།

དྷ་རམ་ས་ལར་ཚོགས་པ་གཉིས་ཀྱིས་སྔ་རྗེས་སུ་བསྒྲགས་གཏམ་སྤེལ་ཏེ་ཉེ་ལམ་ཨ་ཚོགས་ཀླུ་མཁར་བྱམས་ཀྱིས་དབྱིན་ཇི་བ་ཞིག་གྲོངས་པར་མྱ་ངན་རྩོམ་ཡིག་སྤེལ་བ་དེ་༧གོང་ས་མཆོག་ལ་ཟུར་ཟ་ཡིན་པར་ངོས་འཛིན་ཐོག་ངོ་རྒོལ་གསལ་བསྒྲགས་སྤེལ་སོང་། དེ་རིང་ཉིན་རྒྱབ་ཆུ་ཚོད་ ༢།༣༠ སྟེང་། དྷ་ས་ནོར་བུ་གསོལ་མགྲོན་ཁང་དུ། བོད་ཀྱི་རྩ་དོན་གླིང་སྐད་འཕྲིན་ཚོགས་པའི་གསར་འགོད་གསལ་བསྒྲགས་ཤིག་སྤེལ་ཏེ། ཉེ་ལམ་དྷ་སར་གནས་སྡོད་ཨ་ཚོགས་ཀླུ་མཁར་བྱམས་ཀྱིས་༼ ལོ་ ༡༡༣ ལ་བཞུགས་དགོས་པའི་མི་དེ་ཁྱེད་ཡིན། ༽ཞེས་པའི་རྩོམ་ཡིག་ཅིག་བྲིས་ཏེ་༧གོང་ས་མཆོག་ལ་ཟུར་ཟ་བྱས་པར་ངོ་རྒོལ་ཡོད་སྐོར་བརྗོད་སོང་། རྩ་དོན་གླིང་སྐད་འཕྲིན་ཚོགས་པའི་དབུས་གཙང་སྐུ་ཚབ་སྤྱི་ཟུར་བུ་ཡོན་ཏན་རྒྱ་མཚོ་ལགས་དང་། མདོ་སྨད་སྐུ་ཚབ་རྔ་པ་ཚེ་ཕྲུག་ལགས། མདོ་སྟོད་སྐུ་ཚབ་འཕྲིན་ལས་བྱམས་པ་ལགས་གསུམ་གྱིས་གསར་འགོད་གསལ་བསྒྲགས་སྒྲོགས་སྦྱང་གནང་བ་ལྟར་ན། ཉེ་ལམ་རྩ་དོན་གླིང་སྐད་འཕྲིན་ཚོགས་པས་སྣེ་འཁྲིད་འོག བོད་ཆོལ་ཁ་གསུམ་གྱི་མི་མང་ ༢༢༢༡ ཙམ་གྱིས་ཀླུ་མཁར་བྱམས་ལ་ངོ་རྒོལ་མཚན་རྟགས་བཀོད་ཡོད་པ་དང་། བསྒྲགས་གཏམ་དང་ས་ཡིག་གི་ངོ་བཤུས་སོགས་དགའ་ལྡན་ཕོ་བྲང་དང་བཀའ་ཤག བོད་མི་མང་སྤྱི་འཐུས་དང་གཞུང་འབྲེལ་མིན་པའི་ཚོགས་པ་ཁག་ལ་ཕུལ་ཡོད་འདུག ཁོང་ཚོའི་བསྒྲགས་གཏམ་ནང་། ད་ལོའི་དུས་འཁོར་དབང་ཆེན་སྐབས་༧གོང་ས་མཆོག་གིས་ལྷ་དང་བླ་མའི་ལུང་བསྟན་དང་རྨི་ལྟས་སོགས་ལ་གཞིགས་ན་ཁོང་ཉིད་དགུང་གྲངས་ ༡༡༣ ལ་བཞུགས་རྒྱུའི་རྟགས་མཚན་བྱུང་ཡང་། དོན་དངོས་དེ་ལྟར་བཞུགས་མིན་བཤད་དཀའ་ཡང་ལོ་ ༡༠༠ ལ་བཞུགས་ཐུབ་ངེས་ཡིན་སྐོར་གསུངས་པ་དེ་བོད་མི་ཡོངས་དགའ་སྤྲོའི་ཁེངས་ཡོད་སྐབས། ཀླུ་མཁར་བྱམས་ཀྱིས་དབྱིན་ཇི་བ་ཞིག་གྲོངས་སྐབས་༼ ལོ་ ༡༡༣ ལ་བཞུགས་དགོས་པའི་མི་དེ་ཁྱེད་ཡིན། ༽ཞེས་པའི་རྩོམ་ཡིག་བྲིས་ཏེ་༧གོང་ས་མཆོག་ལ་ཟུར་ཟ་བྱས་པ་དེར་བོད་མི་ཡོངས་བློ་ཕམ་ཆེན་པོ་བྱུང་ཡོད་སྐོར་བཀོད་འདུག ༧གོང་ས་མཆོག་ནི་བོད་མི་ཡོངས་ཀྱི་མགོན་སྐྱབས་དང་། བོད་མིའི་ངོ་བོ་དང་ཆིག་བསྒྲིལ་གྱི་འབྱུང་རྩ་ཆགས་ཡོད་ཀྱང་། ཀླུ་མཁར་བྱམས་ཀྱིས་བསམ་བཞིན་དུ་བོད་རྒྱ་ཆེའི་མི་མང་གི་སེམས་ལ་ཕོག་ཐུག་འགྲོ་བའི་བའི་བྱ་སྤྱོད་སྤེལ་ཡོད་སྟབས། ཁོ་རང་གི་རྩོམ་ཡིག་དང་ལས་འགུལ་ཅི་སྤེལ་བོད་པའི་རྩ་དོན་དང་འབྲེལ་བ་མེད་པར་རྒྱ་ནག་གཞུང་གི་བྱེད་སྟངས་ཤིག་ལ་ངོས་འཛིན་བྱ་རྒྱུ་ཡིན་སྐོར་དང་། བོད་གཞིས་བྱེས་མི་མང་ཡོངས་དང་བོད་མིའི་སྒྲིག་འཛུགས་གཙོས་པའི་ཚོགས་སྒེར་ཁག་གིས་ཡིད་གཟབ་བྱེད་དགོས་པ་མ་ཟད། བདེ་སྲུང་ལས་ཁུངས་ཀྱིས་ཆབ་སྲིད་བཙོན་ཟུར་གྱི་མིང་ཁོངས་ནས་ཕྱིར་འདོན་གནང་དགོས་པ་དང་། དགུ་བཅུ་གསུམ་ཚོགས་པས་ཀྱང་ཚོགས་པའི་ཆེ་མཐོང་ཆེད་ཞུ་དག་གཏོང་དགོས་སྐོར་སོགས་བཀོད་འདུག གཞན། དེ་རིང་སྔ་དྲོའི་ཆུ་ཚོད་ ༡༡།༠༠ པའི་སྟེང་དྷ་ས་ཡོངས་གླང་སློབ་གྲྭའི་ཚོགས་ཁང་དུ། དྷ་ས་ས་གནས་མདོ་སྨད་རྒྱུན་ལས་ཀྱིས་ཀླུ་མཁར་བྱམས་ལ་ངོ་རྒོལ་གསལ་བཤད་ཅིག་སྤེལ་བ་དང་མཉམ་དུ་དྷ་སའི་མདོ་སྨད་མི་མང་ལ་ས་རྟགས་བསྡུ་རུབ་བྱས་སོང་། གསལ་བཤད་ནང་ཀླུ་མཁར་བྱམས་ཀྱིས་གཅིག་མིན་གཉིས་མིན་ཐེངས་མང་པོར་༧གོང་ས་མཆོག་དང་བོད་མིའི་སྒྲིག་འཛུགས་ལ་ནག་འགེལ་ཟུར་ཟ་དང་མཚན་སྨོད་བྱས་ཡོད་སྟབས། བོད་པའི་གནས་སྟངས་ཛ་དྲག་གཉེན་འཕྲེང་གི་སྐབས་འདིར། བཤད་སྤྱོད་དེ་རིགས་ལ་མི་མང་གིས་མ་སྐུལ་དང་བླངས་ཐོག་འགན་འཁྱེར་ཏེ། བློ་ཕམ་གྱི་རྣམ་འགྱུར་གསལ་བོར་སྟོན་དགོས་པའི་དུས་ལ་བབ་ཡོད་པ་དང་། ཀླུ་མཁར་བྱམས་ཀྱི་བཤད་སྤྱོད་དེ་རིགས་སྨྲ་བརྗོད་རང་དབང་དང་དམངས་གཙོ་ཡིན་ལུགས་བཤད་མཁན་དག་ནི་རྨོངས་འཐོམ་གྱི་དབང་དུ་སོང་བ་ཙམ་ལས། དམངས་གཙོའི་ལམ་སྲོལ་ནང་ཐོབ་ཐང་དང་འགན་འཁུར་ཟུང་དུ་འབྲེལ་དགོས་པ་ལས། སྨྲ་བརྗོད་རང་དབང་ཞེས་པ་ནི་ས་མཚམས་མེད་པ་ཞིག་མིན་སྐོར་སོགས་བཀོད་འདུག ཁོང་ཚོས་དྷ་ས་མདོ་སྨད་མི་མང་གི་ངོས་འཛིན་དང་གསལ་བཤད་དོན་ཚན་ ༦ འཁོད་ཡོད་པ་ནི། ༡༽ བདེ་སྲུང་ལས་ཁུངས་ཀྱིས་ཀླུ་མཁར་བྱམས་ཆབ་སྲིད་བཙོན་ཟུར་དངོས་ཡིན་མིན་ཞིབ་འཇུག་ཁ་གསལ་མ་བྱུང་བར་ཨོ་སི་ཊོ་ལེ་ཡར་མི་གཏོང་རྒྱུའི་རེ་ཞུ་འབུལ་རྒྱུ། ༢༽ དྷ་སའི་མདོ་སྨད་མི་མང་གིས་ཀླུ་མཁར་བྱམས་ལ་འབྲེལ་བ་མི་བྱེད་པར་རང་ཚུགས་བརྟན་སྲུང་བྱ་རྒྱུ། ༣༽ ཀླུ་མཁར་བྱམས་ཀྱིས་དབྱིན་ཇི་བ་ཞིག་སྐུ་གྲོངས་བར་རྩོམ་ཡིག་བྲིས་པ་དེ་ནི། རྒྱ་ནག་གཞུང་དང་དོལ་ཚོགས་ཀྱི་བྱ་སྤྱོད་དང་གཅིག་པ་ཡིན་སྟབས་མཐའ་གཅིག་ཏུ་ངོ་རྒོལ་བྱེད་རྒྱུ། ༤༽ ཀླུ་མཁར་བྱམས་ཀྱིས་རྩོམ་རིག་སྒྱུ་རྩལ་དང་སྙན་ངག་སོགས་ལ་ཁ་གཡར་ཏེ་ཁོ་རང་གི་ནག་ཉེས་འགེབ་ཐབས་དང་མི་མང་ལ་མགོ་སྐོར་གཏོང་ཐབས་བྱེད་བཞིན་པ་དེར་མཐའ་གཅིག་ཏུ་ངོ་རྒོལ་བྱེད་རྒྱུ། ༥༽ བོད་པའི་ཆོས་བརྒྱུད་ཁག་གི་བླ་ཆེན་དང་དགེ་བཤེས། རྩོམ་པ་པོ་ཚོས་ཀླུ་མཁར་བྱམས་ཀྱི་བཤད་སྤྱོད་ལ་དུས་ནས་དུས་སུ་གསལ་བཤད་ཡོང་བའི་འབོད་སྐུལ། ༦༽ དྷ་ས་ས་གནས་རྒྱུན་ལས་རྣམ་པས་དངོས་ཤུགས་བརྒྱུད་གསུམ་ནས་ཀླུ་མཁར་བྱམས་ལ་འབྲེལ་བ་མི་བྱེད་རྒྱུ་སོགས་བཀོད་འདུག
