བྱ་དེ་ཁྲུང་ཁྲུང་དཀར་མོ།
།གཤོག་རྩལ་ང་ལ་གཡོར་དང་།།

སྨན་པ་ཚེ་དབང་ཡེ་ཤེས་སྤེན་པ་ལགས་ཀྱིས་དངོས་བྱུང་གནས་ཚུལ་ཞིག་གཞི་རྩར་བཟུང་སྟེ་བྲིས་པའི་དབྱིན་ཡིག་གི་སྒྲུང་དེབ་ཅིག་དེ་རིང་དབུ་འབྱེད་གནང་སོང་། དེ་རིང་ཕྱི་དྲོར་བོད་ཀྱི་དཔེ་མཛོད་ཚོགས་ཁང་དུ། བོད་ཀྱི་གཤག་བཅོས་སྨན་པ་ཚེ་དབང་ཡེ་ཤེས་སྤེན་པ་ལགས་ཀྱིས་དབྱིན་ཡིག་ནང་བྲིས་པའི་༼བྱ་དེ་ཁྲུང་ཁྲུང་དཀར་མོ། །གཤོག་རྩལ་ང་ལ་གཡོར་དང་།།༽ <<White Carne, Lend me your wing>>ཞེས་པའི་སྒྲུང་དེབ་ཅིག་དབུ་འབྱེད་གནང་སོང་། དཔེ་དེབ་དབུ་འབྱེད་མཛད་སྒོར་བོད་མིའི་སྒྲིག་འཛུགས་ཀྱི་སྲིད་སྐྱོང་བློ་བཟང་སེང་གེ་མཆོག་དང་སྲིད་བྱུས་ཉམས་ཞིབ་ཁང་གི་འགན་འཛིན་ཐུབ་བསྟན་བསམ་འཕེལ་ལགས། རྩོམ་པ་པོའི་བུ་མོ་སྤེན་པ་ལྷ་མོ་ལགས་གཙོས་པའི་རྩོམ་པ་པོ་དང་གངས་སྐྱིད་ཁུལ་གྱི་ལས་བྱེད་ཁག་ཅིག་བཅས་འཛོམ་འདུག སྤེན་པ་ལྷ་མོ་ལགས་ཀྱིས་ཁོང་མོའི་ཕ་ཚེ་དབང་ཡེ་ཤེས་སྤེན་པ་ལགས་ངོ་སྤྲོད་གནང་པ་ལྟར་ན། སྨན་པ་ཚེ་དབང་ཡེ་ཤེས་སྤེན་པ་ལགས་ནི་ ༡༩༣༢ ལོར་རྒྱང་རྩེར་སྐུ་འཁྲུངས་ཤིང་ ༢༠༡༡ ལོའི་ཟླ་ ༡༡ ཚེས་ ༢༦ ཉིན་སྐུ་གྲོངས་ཡོད་པ་དང་། ཁོང་གིས་སྔོན་ཆད་རྒྱ་གར་དར་རྒྱས་གླིང་དུ་སློབ་གཉེར་གནང་བ་དང་། ཕྱིས་སུ་ལོན་ཌོན་གཙུག་ལག་སློབ་ཆེན་དུ་ཕེབས་ནས་ཕྱི་ལུགས་གཤག་བཅོས་རབ་འབྱམས་པ་ (MBBS) ཞེས་པའི་ཐོབ་ཐང་བླངས་ཡོད་པ་མ་ཟད། ལོན་ཌོན་གཙུག་ལག་སློབ་ཆེན་གྱི་གཤག་བཅོས་ཚན་པས་ཁོང་ལ་ (Hallet Prize) ཞེས་པའི་གཟེངས་རྟགས་ཤིག་ཀྱང་གནང་ཡོད་འདུག ཚེ་དབང་ཡེ་ཤེས་སྤེན་པ་ལགས་ཀྱིས་དེ་སྔ་དབྱིན་ཡིག་ནང་ཕྱག་དེབ་གཉིས་བྲིས་ཡོད་ཅིང་། ༡༩༥༧ ལོར་ཁོང་གི་མཛད་རྣམ་ཕྱག་དེབ་ <<Young Days in Tibet>>ཞེས་པ་དང་།  ༡༩༦༦ ལོར་སྒྲུང་དེབ་ <<Idols on the Path>>ཞེས་པ་བྲིས་ཡོད་འདུག ཐེངས་འདི་ནི་ཁོང་གི་དཔེ་དེབ་གསུམ་པ་དེ་ཡིན་ཞིང་། ཤོག་ངོས་ ༤༦༨ ལས་གྲུབ་པའི་སྒྲུང་དེབ་དེའི་ནང་དོན་གཙོ་བོ་ཁམས་ཉག་རོང་ས་ཆ་ནས་བྱུང་བའི་གྲོགས་པོའི་མཛའ་བརྩེའི་སྐོར་གྱི་གཏམ་བརྒྱུད་ཅིག་གཞིར་བཟུང་སྟེ་བྲིས་ཡོད་པ་དང་། དེ་ནི་ཨ་རི་ནས་ཡེ་ཤུ་ཆོས་སྤེལ་མཁན་གཉིས་བོད་དུ་ཕྱིན་ཏེ། ཉག་རོང་ས་ཆའི་དཔོན་པོའི་བུ་དང་ཡེ་ཤུ་ཆོས་སྤེལ་མཁན་གྱི་བུ་གཉིས་རྩེད་གྲོགས་ཆགས་པ་དང་། སྐབས་དེར་རྒྱ་མིས་བོད་ལ་བཙན་འཛུལ་བྱས་པའི་སྐོར་སོགས་དཀྱུས་གཅིག་ཏུ་བྲིས་ཡོད་འདུག རྒྱལ་དབང་སྐུ་ཕྲེང་བདུན་པ་ཚང་དབྱངས་རྒྱལ་མཚོ་མམཆོག་ཁམས་ལི་ཐང་ནས་སྐུ་འཁྲུངས་ཡོད་པ་དང་། རྩེད་གྲོགས་ཀྱི་གཏམ་བརྒྱུད་དེའང་ཁམས་ཕྱོགས་ནས་བྱུང་ཡོད་སྟབས། སྒྲུང་དེབ་ཀྱི་མིང་ལ་བྱ་དེ་ཁྲུང་ཁྲུང་དཀར་མོ་གཤོག་རྩལ་ང་ལ་གཡོར་དང་ཅེས་བཏགས་པ་ཡིན་འདུག མཇུག་ཏུ། སྲིད་བྱུས་ཉམས་ཞིབ་ཁང་གི་འགན་འཛིན་ཐུབ་བསྟན་བསམ་འཕེལ་ལགས། རྩོམ་པ་པོའི་བུ་མོ་སྤེན་པ་ལྷ་མོ་ལགས་དང་སྙན་ངག་པ་བུ་ཆུང་བསོད་ནམས་ལགས། སྒྲུང་དེབ་དེ་སྒྲ་འཇུག་གནང་མཁན་ལྕམ་སུ་མེ་ཊ་སི་ཧ་གྷལ་ (Summit Sehgal)ལགས་བཅས་ཀྱིས་སྒྲུང་དེབ་དེའི་སྐོར་ལ་གྲོང་གླེང་གནང་སོང་། གསར་འགོད་པ། པདྨ་རྒྱལ།
