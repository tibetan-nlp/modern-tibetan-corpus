རུལ་སུངས་བ་འབུམ་ ༣ ལ་ཉེས་ཆད་བཅད།

  རྒྱ་ནག་དམར་ཤོག་ཚོགས་པས་ན་ནིང་གཅིག་པུར་རྒྱལ་ནང་གི་ལྐོག་ཟ་རུལ་སུངས་མཁན་གཞུང་འབྲེལ་དང་གཞུང་འབྲེལ་མ་ཡིན་པ་བཅས་ཀྱི་མི་འབུམ་ ༣ ཙམ་ལ་ཉེས་ཆད་ལྕི་ཡང་སྣ་ཚོགས་བཅད་ཡོད་ལུགས་བསྒྲགས་འདུག   ཨེ་ཕེ་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཁ་སང་རེས་གཟའ་ཉི་མའི་ཉིན་རྒྱ་ནག་དམར་ཤོག་ཚོགས་པས་ཕྱི་ལོ་ ༢༠༡༥ ལོར་རྒྱལ་ནང་གི་ལྐོག་ཟ་རུལ་སུངས་མཁན་མི་གྲངས་ ༣༠༠,༠༠༠ ཉེ་བར་ཉེས་ཆད་བཅད་ལུགས་བསྒྲགས་ཡོད་འདུག རྒྱ་ནག་གི་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱིས་རྒྱ་ནག་གཞུང་འབྲེལ་བཟོ་ལས་ཐོན་གཉེར་ཁང་ཁག་དང་ས་གནས་སྲིད་གཞུང་ཁག དམག་ཁུལ་དང་དམར་ཤོག་ཚོགས་པ་ནང་ཁུལ་བཅས་སུ་གཞི་རྒྱ་ཆེན་པོའི་སྒོ་ནས་ལྐོག་ཟ་རུལ་སུངས་གཙང་བཤེར་སྲིད་ཇུས་ལག་བསྟར་བྱས་ཏེ་དོགས་ཡོད་དཔོན་རིགས་ཚོར་ཚ་འདེད་མུ་མཐུད་གཏོང་གི་ཡོད་པ་དང་། རྒྱ་ནག་དབུས་སྒྲིག་བཤེར་ལྷན་ཁང་ (CCDI) གི་བསྒྲགས་གཏམ་དག་གི་ནང་། སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱི་ལྐོག་ཟ་རུལ་སུངས་གཙང་བཤེར་སྲིད་ཇུས་ཀྱི་བྱེད་ཐབས་དང་དངོས་ཡོད་གནས་ཚུལ་ཕྱི་བསྒྲགས་བྱེད་ཀྱི་མེད་པར་ངོས་འཛིན་བྱེད་ཀྱིན་ཡོད།   རྒྱ་ནག་དམར་ཤོག་དབུས་ལྟ་ཞིབ་ཚན་པའི་གནས་ཚུལ་ལྟར་ན། རྒྱ་ནང་གི་ལྐོག་ཟ་རུལ་སུངས་མཁན་ ༢༠༠,༠༠༠ ཉེས་ཆད་ཡངས་མོ་བཅད་ཡོད་པ་དང་། ལྐོག་ཟ་རུལ་སུངས་མཁན་ ༨༢,༠༠༠ ཙམ་ལ་དཔོན་གནས་ནས་ཕྱིར་ཕུད་དང་ཉེས་ཆད་ལྕི་མོ་བཅད་ཡོད་འདུག་ཀྱང་། ལྐོག་ཟ་རུལ་སུངས་ལས་འགུལ་ཁྲོད་དུ་དཔོན་རིགས་སྟོང་ཕྲག་མང་པོར་མི་སྒེར་གྱི་ཡིག་ཆ་ ༡༥༦༧ ལྷག་ལ་ཞིབ་བཤེར་དང་འདྲི་གཅོད་ནན་མོ་བྱས་ཀྱང་། དོན་དངོས་སུ་ཁོ་ཚོས་ནག་ཉེས་ཅི་བསགས་ཀྱི་བདེན་དཔང་ར་སྤྲོད་ཐུབ་མེད་ནའང་། དཔོན་རིགས་དེ་དག་དཔོན་གནས་ནས་ཕྱིར་ཕུད་དང་འཛིན་བཟུང་བཙོན་འཇུག་བྱས་ཡོད་པ་རེད།   ཞི་ཅིན་ཕིན་གྱི་ལྐོག་ཟ་རུལ་སུངས་གཙང་བཤེར་ལས་འགུལ་ཁྲོད་ད་ལྟའི་བར་ཞིབ་བཤེར་བཙོན་འཇུག་བྱས་པའི་དཔོན་རིགས་མཐོ་ཤོས་དེ་ཀྲུའུ་ཡོང་ཁང་ཡིན་པ་དང་། ཁོ་རྒྱ་ནག་དབུས་ཆབ་སྲིད་ལྷན་ཁང་གི་གཙོ་འཛིན་དང་དམར་ཤོག་ཚོགས་པའི་ཚོགས་མི་ཟུར་བ་བཅས་ཡིན་ཞིང་། ཕྱི་ལོ་ ༢༠༡༥ ལོར་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་ནག་ཉེས་ཐོག་ཚེ་བཙོན་གྱི་ཁྲིམས་ཐག་བཅད་ཡོད་པ་རེད།   གསར་འགོད་པ། པད་རྒྱལ།
