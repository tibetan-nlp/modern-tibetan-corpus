ལྷ་ས་ནས་གཞིས་རྩེ་བར་ལྕཊ་ལམ་བཟོས་ཡོད།

ལྷ་ས་ནས་གཞིས་རྩེ་བར་གྱི་ལྕགས་ལམ་བཟོ་སྐྲུན་ལེགས་འགྲུབ་བྱུང་སྟེ་ཚོད་ལྟ་ཡང་བྱས་ཡོད་པ་མ་ཟད། ཟླ་རྗེས་མའི་ནང་མེ་འཁོར་དངོས་སུ་ཤར་གཏོང་བྱ་རྒྱུ་ཡིན་སྐོར་བསྒྲགས་འདུག དེ་རིང་རྒྱ་ནག་གཞུང་འབྲེལ་གསར་ལམ་ཁག་ཏུ་གནས་ཚུལ་ཐོན་པར། ཉེ་ལམ་ལྷ་ས་ནས་གཞིས་རྩེ་བར་གྱི་ལྕགས་ལམ་གསར་སྐྲུན་ཟིན་ཏེ་ཟླ་འདིའི་ཚེས་ ༢༢ ཉིན་ཚོད་ལྟ་བྱས་ཡོད་པ་མ་ཟད། ཟླ་རྗེས་མའི་ནང་མེ་འཁོར་དངོས་སུ་ཤར་གཏོང་བྱ་རྒྱུ་ཡིན་སྐོར་བསྒྲགས་འདུག་ལ། གནས་ཚུལ་ནང་མུ་མཐུད་དུ། ༢༠༡༦ ལོར་གཞིས་རྩེ་ནས་རྒྱ་གར་དང་བལ་ཡུལ། འབྲུག་བཅས་ཀྱི་ས་མཚམས་སུ་མེ་འཁོར་ལྕགས་ལམ་བཟོ་སྐྲུན་བྱེད་མགོ་ཚུགས་ཏེ་ ༢༠༢༠ ལོའི་ནང་ཚུད་དུ་ཤར་གཏོང་ཐུབ་པ་བྱ་རྒྱུ་ཡིན་སྐོར་ཡང་བསྒྲགས་འདུག སྲིད་དོན་དབྱེ་ཞིབ་པ་ཚོས། སང་ཉིན་རྒྱ་གར་གྱི་ཕྱི་སྲིད་བློན་ཆེན་ལྕམ་སུ་ཞི་མཱ (Sushma Swaraj)བལ་ཡུལ་ནང་བསྐྱོད་དེ་བལ་ཡུལ་དུ་གློག་ཁང་གསར་རྒྱག་སོགས་ལས་གཞི་ཁག་ཅིག་གི་ཐོག་གྲོས་མོལ་བྱེད་རྒྱུ་ཡིན་པ་དང་། ཟླ་རྗེས་མར་སྲིད་བློན་ནཱ་རེན་ཌ་མུ་དྷེ་བལ་ཡུལ་ནང་བསྐྱོད་འཆར་ཡོད་པ་བཅས་ལ་བརྟེན་ཏེ། དེ་རིང་རྒྱ་ནག་གིས་ས་མཚམས་ཁུལ་དུ་ལྕགས་ལམ་རྒྱ་སྐྱེད་གཏོང་འཆར་ཡོད་པ་ཕྱིར་བསྒྲགས་བྱས་པ་ཞིག་ཡིན་བརྗོད་འདུག བོད་རང་སྐྱོང་ལྗོངས་མེ་འཁོར་འགྲིམ་འགྲུལ་ལས་ཁང་གི་ངེས་སྟོན་པ་གཞོན་པ་ཡང་ཡུ་ལིན་ (YangYulin)ཟེར་བ་དེས། རྒྱ་ནག་གི་གོ་ལའི་དུས་བབ་ ( Global Times)ཞེས་པའི་གསར་འགོད་པར། ལྷ་ས་ནས་གཞིས་རྩེ་བར་ལྕགས་ལམ་གསར་སྐྲུན་ཟིན་ཏེ། ཟླ་རྗེས་མའི་ནང་བོད་མིའི་ཆོས་ཕྱོགས་ཀྱི་བླ་ཆེན་གཉིས་པ་བཞུགས་ཡུལ་གྲོང་ཁྱེར་ནང་དངོས་སུ་ཤར་གཏོང་བྱ་རྒྱུ་ཡིན་པ་མ་ཟད། ༢༠༡༦ ལོར་མཚོ་བོད་ལྕགས་ལམ་གྱི་སྣེ་གཉིས་སུ་གྱེས་ཏེ་སྣེ་གཅིག་བལ་ཡུལ་དང་སྣེ་གཞན་དེ་རྒྱ་གར་དང་འབྲུག་གི་ས་མཚམསས་བར་མཐུད་དེ། ༢༠༢༠ ལོར་ཁྱིམ་མཚེས་རྒྱལ་ཁབ་དེ་གཉིས་ཀྱི་ས་མཚམས་བར་མེ་འཁོར་ཤར་གཏོང་ཐུབ་པ་བྱ་རྒྱུ་ཡིན། ཞེས་དང་། ཁོས་རྒྱ་ནག་མི་དམངས་ཉིན་རེའི་གསར་ཤོག་སྟེང་། གོར་མོ་ནས་ལྷ་ས་བར་གྱི་མཚོ་བོད་ལྕགས་ལམ་བཟོ་སྐྲུན་བྱེད་སྐབས་ཁོར་ཡུག་ལ་གནོད་པ་མི་འབྱུང་ཐབས་ལ་འབད་བརྩོན་བྱས་པ་ལྟར། ད་རེབ་ང་ཚོས་ལྷ་ས་ནས་གཞིས་ཀ་རྩེ་བར་གྱི་ལྕགས་ལམ་གསར་བཟོའི་སྐབས་སུ་གང་ཐུབ་ཀྱིས་ཁོར་ཡུག་ལ་མི་གནོད་པ་བྱས་ཡོད་སྟབས། མ་འོངས་པར་ལྕགས་ལམ་འདིས་བོད་ལྷོ་ནུབ་ཁུལ་དུ་གནས་སྡོད་བྱེད་མཁན་གྱི་བོད་རིགས་འབུམ་བདུན་ལྷག་གི་འགྲིམ་འགྲུལ་གྱི་དཀའ་ངལ་སེལ་རྒྱུར་ཕན་ཐོགས་ཆེན་པོ་འབྱུང་ངེས་ཡིན། ཞེས་སོགས་བརྗོད་འདུག ༢༠༠༦ ལོར་རྒྱ་གཞུང་གིས་མཐོ་ཚད་ལ་རྒྱ་མཚོའི་ངོས་ནས་མི་ཌར་ ༥༠༠༠ ལྷག་ཡོད་པའི་གོར་མོ་ནས་ལྷ་ས་བར་གྱི་ ལྕགས་ལམ་སྒོ་ཕྱེ་ཡོད་པ་དང་། ད་རེབ་མཐོ་ཚད་ལ་རྒྱ་མཚོའི་ངོས་ནས་མི་ཊར་ ༣༦༠༠ ནས་ ༤༠༠༠ བར་དང་། རིང་ཚད་ལ་ཀི་ལོ་མི་ཊར་ ༢༥༣ དང་བབས་ཚུགས་ ༡༣ ཙམ་ཡོད་པའི་ལྷ་ས་ནས་གཞིས་རྩེ་བར་གྱི་ལྕགས་ལམ་བཟོ་སྐྲུན་བྱས་ཡོད་ཅིང་། ལྷ་ས་ནས་གཞིས་རྩེ་བར་གྱི་མེ་འཁོར་ལྕགས་ལམ་དེ་ཆུ་ཚོད་གཅིག་ལ་ཆ་སྙོམས་སྤྱི་ལེ་ ༡༢༠ ཙམ་རྒྱུག་ཐུབ་ཀྱི་ཡོད་འདུག འདི་ནི་རྒྱ་ནག་གཞུང་གི་ལོ་ལྔའི་འཆར་གཞི་བཅུ་གཉིས་པའི་གྲས་ནས་བོད་རང་སྐྱོང་ལྗོངས་ཁོངས་ཀྱི་ལས་འཆར་གལ་ཆེ་ཤོས་ཤིག་ཆགས་ཡོད་འདུག ཉེ་ལམ་རྒྱ་གར་སྲིད་བློན་ནཱ་རེན་ཌ་མུ་དྷེ་མཆོག་གིས་ཁྱིམ་མཚེས་རྒྱལ་ཁབ་ཁག་དང་འབྲེལ་ལམ་ཡར་རྒྱས་སུ་བཏང་སྟེ་རྒྱ་གར་གྱི་མཐུན་གྲོགས་སུ་འགུག་ཐབས་བྱེད་ཀྱི་ཡོད་པ་དང་། ལྷག་པར་དུ་ཉེ་ལམ་འབྲུག་ནང་ཕེབས་ཏེ་དམག་སྦྱོང་དང་ཚོང་འབྲེལ་ཤུགས་ཆེ་རུ་གཏོང་རྒྱུར་གྲོས་མཐུན་བྱུང་ཡོད། ཆབ་སྲིད་དབྱེ་ཞིབ་པ་ཚོས། བོད་མི་དང་བོད་དོན་རྒྱབ་སྐྱོར་བ། དེ་བཞིན་རྒྱལ་སྤྱིའི་ཚོགས་སྡེ་མང་པོས་སྐྱོན་བརྗོད་བྱེད་མིན་ལ་མ་ལྟོས་པར། རྒྱ་ནག་གིས་འགྲོ་སོང་ཕོན་ཆེན་བཏང་སྟེ་མཚོ་བོད་ལྕགས་ལམ་དེ་གཞིས་རྩེ་བར་མཐུད་ཡོད་པ་དང་། མ་འོངས་པར་བལ་ཡུལ་དང་འབྲུག རྒྱ་གར་སོགས་ཁྱིམ་མཚེས་རྒྱལ་ཁབ་ཁག་གི་ས་མཚམས་བར་སླེབས་ཐབས་བྱེད་དོན་གཙོ་བོ། ས་ཁུལ་དེ་དག་བོད་དང་ས་འབྲེལ་ཆགས་ཡོད་པ་མ་ཟད་ཨེ་ཤ་ཡའི་གནད་ས་དག་ཆགས་ཡོད་སྟབས། རྒྱ་གར་དང་བལ་ཡུལ། འབར་མ། འབྲུག་བཅས་ལ་རྒྱ་ནག་གི་ཐབས་ཇུས་བཀོད་སྒྲིག་རྒྱ་སྐྱེད་གཏོང་ཆེད་ཡིན་པའི་ཚོད་དཔག་བྱེད་བཞིན་ཡོད་འདུག གསར་འགོད་པ། དོན་གྲུབ།
