སྲིད་གཞུང་གི་བྱ་སྤྱོད་འདྲ་མཉམ་མ་རེད།

བླ་བྲང་བསང་ཆུ་རྫོང་མི་མང་སྲིད་གཞུང་གིས་མང་ཚོགས་ལ་འདྲ་མཉམ་དང་དྲང་བདེན་མིན་པའི་སྲིད་བྱུས་ལག་བསྟར་བྱས་པར་དེ་རིང་ས་གནས་མི་མང་གིས་ངོ་རྒོལ་བྱེད་བཞིན་ཡོད་འདུག ད་ལྟ་རིང་བོད་ནས་གནས་ཚུལ་འབྱོར་གསལ་ལྟར་ན། དེ་རིང་སྔ་དྲོའི་བོད་ནང་གི་ཆུ་ཚོད་ ༡༠ ཙམ་ནས་བཟུང་། ཀན་སུའུ་ཞིང་ཆེན་ཀན་ལྷོ་ཁུལ་བསང་ཆུ་རྫོང་ཐོག་ཏུ་མི་མང་གིས་སྲིད་གཞུང་ལ་ངོ་རྒོལ་བྱེད་བཞིན་ཡོད་འདུག དེ་ནི་ལོ་འགའི་གོང་བླ་བྲང་བསང་ཆུ་རྫོང་གི་ཉེ་འཁོར་ཞིག་ཏུ་སྲིད་གཞུང་གིས་མང་ཚོགས་ཀྱི་ལག་ནས་ས་ཆ་ཉོས་བའི་སྔ་ཕྱིར་གོང་ཚད་བར་ཁྱད་ཆེན་པོ་བྱུང་ཡོད་སྟབས། ས་གནས་མང་ཚོགས་ཀྱིས་སྲིད་གཞུང་གིས་མང་ཚོགས་ལ་འདྲ་མཉམ་དང་དྲང་བདེན་བྱས་མེད་པར་ངོ་རྒོལ་བྱེད་བཞིན་ཡོད་འདུག ལོ་འགའི་གོང་ལ་ས་དེར་འོ་མའི་བཟོ་གྲྭ་ཞིག་འཛུགས་སྐབས་ས་ཆ་སྤྱི་ལེ་གྲུ་བཞི་མ་གཅིག་ལ་རྒྱ་སྒོར་ཁྲི་ ༦ ལས་སྤྲད་མེད་པ་དང་། དེའི་རྗེས་ས་ཁུལ་དེར་སློབ་འབྲིང་ཞིག་འཛུགས་རྒྱུར་སྲིད་གཞུང་གིས་ཡང་བསྐྱར་མང་ཚོགས་ཀྱི་ས་ཆ་ཉོ་སྐབས་སྤྱི་ལེ་གྲུ་བཞི་མ་གཅིག་ལ་རྒྱ་སྒོར་ཁྲི་ ༡༨ སྤྲད་ཡོད་སྟབས། མང་ཚོགས་ཀྱིས་ས་གཅིག་གི་ཐོག་ཏུ་སྲིད་ཇུས་གཉིས་ལག་བསྟར་བྱས་པ་དེ་སྲིད་གཞུང་གི་བྱེད་སྟངས་དྲང་བདེན་མིན་པའི་ཞུ་གཏུག་ཡང་ཡང་བྱས་ཀྱང་གཞུང་གིས་ཡ་ལན་མ་སྤྲད་སྟབས། འདས་པའི་ཟླ་བ་གཅིག་ཙམ་གྱི་གོང་ནས་བཟུང་ས་བདག་སྡེ་བ་(མ་ལེན་ཐང་།)ཡི་མང་ཚོགས་ཀྱིས་རྫོང་མི་དམངས་སྲིད་གཞུང་སྒོ་ཁར་རྒྱུན་སྡོད་ཀྱིས་ངོ་རྒོལ་གནང་ཡོད་འདུག དེ་རིང་ཞིང་ཆེན་གྱི་དཔོན་རིགས་མཐོ་གྲས་ཤིག་བསང་ཆུ་རྫོང་དུ་ཡོང་གི་ཡོད་སྟབས། མང་ཚོགས་ཧ་ཅང་ཕོན་ཆེན་པོ་འཛོམས་ཏེ་ཞི་བའི་ངོ་རྒོལ་ལས་འགུལ་སྤེལ་སྐབས། རྫོང་གི་ཉེན་རྟོག་པ་དང་དྲག་ཆས་དཔུང་དམག་མང་པོས་བཀག་འགོག་བྱས་ཡོད་པ་དང་། སྐབས་དེར་མང་ཚོགས་སྔར་ལས་མང་དུ་ཕྱིན་ཏེ་རྫོང་མི་མང་སྲིད་གཞུང་སྒོ་ཁ་སོགས་སྐད་འབོད་ངོ་རྒོལ་ཤུགས་ཆེན་བྱེད་བཞིན་ཡོད་འདུག སྲིད་གཞུང་གི་ཉེན་རྟོག་པ་དང་དམག་མི་ཚོས་བྱེད་ཐབས་འདྲ་མིན་ཐོག་ནས་མང་ཚོགས་བཀག་འགོག་བྱས་ཡོད་པ་དང་། མང་ཚོགས་ཀྱིས་ལག་ཏུ་འཕྲེད་དར་དང་འབོད་ཚིག་འདྲ་མིན་འཁོད་པའི་ཡིག་བྱང་བཟུང་སྟེ་ངོ་རྒོལ་བྱེད་བཞིན་ཡོད་འདུག
