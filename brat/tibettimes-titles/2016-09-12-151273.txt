ཐའེ་ཝན་སྲིད་གཞུང་གསར་པར་འབོད་སྐུལ་བྱས།

དེ་རིང་ཐའེ་ཝན་གྱི་ཡུལ་སྐོར་ལས་རིགས་གཉེར་མཁན་སོགས་མི་གྲངས་ཁྲི་ ༢ མ་ཟིན་ཙམ་འདུ་འཛོམས་བྱས་ཏེ་ཚེ་ཡེན་ཝུན་སྲིད་གཞུང་གསར་པས་ཁོ་ཚོར་ལས་ཀ་མེད་པའི་དཀའ་ངལ་སེལ་ཐབས་དགོས་པའི་འབོད་སྐུལ་ལས་འགུལ་སྤེལ་ཡོད་འདུག དེ་རིང་ཐའེ་ཝན་གྱི་རྒྱལ་ས་ཐའེ་པའེ་རུ། ཡུལ་སྐོར་ལས་རིགས་གཉེར་མཁན་ཁེ་ལས་བ་སོགས་མི་གྲངས་ཁྲི་ ༢ མ་ཟིན་ཙམ་འདུ་འཛོམས་བྱས་ཏེ། ང་ཚོ་འཚོ་གནས་དགོས་པས་ལས་ཀ་དགོས་པ་དང་། ལྟོ་གོས་གཉིས་ཀ་འཛོམས་དགོས་ཟེར་བ་སོགས་ཀྱི་འབོད་ཚིག་བསྒྲགས་ཏེ། ཚེ་ཡེན་ཝུན་སྲིད་གཞུང་གསར་པས་ལས་མེད་ཚོའི་དཀའ་ངལ་སེལ་ཐབས་དགོས་པའི་འབོད་སྐུལ་ལས་འགུལ་སྤེལ་ཡོད་པ་དང་། ཕྱི་དྲོར་ཆར་བའི་ནང་ལས་འགུལ་བ་ཚོ་ཧྭ་ཧྲན་རིག་རྩལ་སྤྱི་གླིང་དུ་འཛོམས་ཏེ་ཞུ་འབོད་བྱས་ཡོད་འདུག བརྒྱུད་ལམ་དུ་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཐེངས་འདིའི་ལས་འགུལ་ནང་ཡུལ་སྐོར་ལས་རིགས་དང་འབྲེལ་བའི་ཁེ་ལས་ཆེ་ཁག་ ༥ མཉམ་ཞུགས་བྱས་ཡོད་པ་དང་། ལས་འགུལ་བ་ཞིག་གིས་ད་ལྟ་ལས་ཀ་ཤོར་ཏེ་ང་ལ་ལས་ཀ་མེད་སྟབས། ཁྲོམ་གཞུང་དུ་ཐོན་ཡོད་པ་དང་། ཚེ་ཡེན་ཝུན་སྲིད་གཞུང་གིས་ང་ཚོ་སྐྱོབ་པའི་རེ་བ་ཡོད་ཅེས་བརྗོད་འདུག རྒྱ་ནག་ཤིན་ཧྭ་དྲ་བའི་ཐོག་ཏུ་ཐེངས་འདིའི་ལས་འགུལ་སྣེ་འཁྲིད་པས་ཆབ་སྲིད་ཀྱི་འབོད་ཚིག་བཀག་འགོག་བྱས་ཡོད་ལ། ཆབ་སྲིད་མི་སྣ་ལས་འགུལ་ནང་ཞུགས་པར་དགའ་བསུ་མེད་སྐོར་བརྗོད་ཡོད་ཀྱང་། འདུ་འཛོམས་ཁྲོད་ལས་འགུལ་བས་མཚོ་འགག་གཉིས་ཁྱིམ་ཚང་གཅིག་ཡིན་ཟེར་བའི་འབོད་ཚིག་བྱང་བུ་འཁྱེར་ཡོད་སྐོར་བརྗོད་འདུག རྒྱ་ནག་ཐའེ་ཝན་དྲ་བ་ཞེས་པའི་ཐོག ཐའེ་ཝན་སྲིད་གཞུང་གསར་པས་སྲིད་དབང་བཟུང་རྗེས་མཚོ་འགག་གཉིས་བར་གྱི་ལངས་ཕྱོགས་རབ་རུབ་ཏུ་གྱུར་ཏེ་ཕྱོགས་གཉིས་ཀྱི་འབྲེལ་བ་ཆེས་སྡུག་ས་ཞིག་ཏུ་ལྷུང་ཡོད་པ་དང་། ཐའེ་ཝན་ནང་ཡོད་པའི་རྒྱ་ནག་གི་ཁེ་ལས་ཚོགས་པ་ཚོར་འགོག་རྐྱེན་ཐེབས་པར་བརྟེན། འདི་ལོའི་ཕྱི་ཟླ་ ༥ པ་ནས་བཟུང་རྒྱ་ནག་སྐམ་ས་ནས་ཡོང་མཁན་ཇེ་ཉུང་ནས་ཇེ་ཉུང་དུ་གྱུར་ཡོད་པ་དང་། ཕྱི་ཟླ་ ༨ པའི་ནང་འདས་ལོ་དང་བསྡུར་ན་བརྒྱ་ཆ་ ༥༥ མར་ཆགས་ཡོད་ལ། དགུན་དུས་སུ་དེ་ལས་ཀྱང་ཉུང་དུ་འགྲོ་སྲིད་སྟབས། ཡུལ་སྐོར་ལས་རིགས་གཉེར་མཁན་ཁེ་ལས་བ་དང་ལས་མི་ཚོས་རང་ཉིད་སྐྱོབ་ཆེད་ཞུ་འབོད་ལས་འགུལ་འདི་འདྲ་སྤེལ་བཞིན་ཡོད་སྐོར་བརྗོད་འདུག ཐའེ་ཝན་ལས་དོན་ལྷན་ཁང་གི་མགོ་འཁྲིད་གཞོན་པ་ཆུ་ཁྲུའུ་ཀྲིན་གྱིས། ཐའེ་ཝན་ནང་རྒྱ་ནག་སྐམ་ས་ནས་ཡོང་མཁན་ཇེ་ཉུང་དུ་ཕྱིན་པ་དེས་ཐའེ་ཝན་གྱི་ལས་རིགས་ཁག་ཅིག་ལ་ཤུགས་རྐྱེན་ཆེན་པོ་ཐེབས་ཡོད་པ་དང་། མུ་མཐུད་རྒྱ་ནག་གི་ཡུལ་སྐོར་བ་ཇེ་ཉུང་དུ་ཕྱིན་པ་ཡིན་ན་འདི་ལོའི་ཕྱི་ཟླ་ ༡༠ པའི་རྗེས་ཐའེ་ཝན་ཡོངས་ཀྱི་ཟ་ཁང་ཚོང་ཁང་དང་ཡུལ་སྐོར་སྣེ་ཤན་ཁང་། ས་ཁང་ཚོང་གཉེར་ཁང་སོགས་ཁེ་ལས་ཁང་མང་པོར་དཀའ་ངལ་འཕྲད་རྒྱུ་ཡིན་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
