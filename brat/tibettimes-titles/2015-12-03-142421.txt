ཨ་རིའི་ནང་དམར་གསོད་དོན་རྐྱེན་བྱུང་།

ཨ་རིའི་མངའ་སྡེ་ཁ་ལི་ཧྥོར་ནི་ཡའི་གྲོང་ཁྱེར་སན་བྷི་ནར་ཌི་ནོ་ཡི་ན་གཞོན་དབང་པོ་སྐྱོན་ཅན་སྨན་བཅོས་བདག་སྐྱོང་ཁང་ཞིག་ཏུ་མེ་མདས་བརྒྱབ་སྟེ་མི་ ༡༤ དམར་གསོད་བཏང་ཡོད་འདུག   ཕྱི་ཚེས་ ༢ ཉིན་གྱི་སྔ་དྲོ་ས་གནས་ཀྱི་ཆུ་ཚོད་ ༡༡ ཐོག ཨ་རིའི་ཁ་ལི་ཧྥོར་ནི་ཡའི་(California)ལྷོ་ཕྱོགས་གྲོང་ཁྱེར་སན་བྷིར་ནར་ཌི་ནོ་(San Bernardino)རུ་ཡོད་པའི་ན་གཞོན་དབང་པོ་སྐྱོན་ཅན་སྨན་བཅོས་བདག་སྐྱོང་ཁང་ཞིག་ཏུ། གློ་བུར་དུ་མེ་མདས་བརྒྱབ་སྟེ་མི་གྲངས་ ༡༤ དམར་གསོད་དང་ཉུང་མཐར་མི་ ༡༧ ལ་རྨས་སྐྱོན་བཟོས་ཡོད་འདུག   སན་བྷིར་ནར་ཌི་ནོ་གྲོང་ཁྱེར་ཉེན་རྟོག་ཁང་གིས། ཉུང་མཐར་དོགས་ཡོད་མེ་མདའ་རྒྱག་མཁན་གསུམ་ཡུལ་དངོས་ནས་བྲོས་ཕྱིན་ཡོད་པ་དང་། ཁོ་ཚོས་དྲག་ཆས་རིགས་མང་པོ་འཁུར་ཡོད་པ་མ་ཟད། མདའ་འགོག་སྲུང་ཆས་ཀྱང་གྱོན་ཡོད་སྲིད་ལ། ཉེན་རྟོག་ཕྱོགས་ནས་ཉེས་ཅན་རྩད་གཅོད་བྱེད་སྐབས། དོན་རྐྱེན་བྱུང་ཡུལ་ནས་བྲོས་ཕྱིན་པའི་དོགས་ཡོད་རླངས་འཁོར་ཞིག་ལ་རྗེས་འདེད་བཏང་སྟེ་བཀོག་འགོག་བྱེད་སྐབས། ཉེན་རྟོག་པ་དང་དོགས་ཡོད་ཉེས་ཅན་བར་མེ་མདས་རྒྱག་རེས་བྱས་ཏེ། དོགས་ཡོད་ཉེས་ཅན་ ༢ ཡུལ་དངོས་སུ་བསད་པ་དང་། མི་གཅིག་འཛིན་བཟུང་བྱས་ཡོད་ལ། རླངས་འཁོར་ནང་གི་དོགས་ཡོད་དངོས་རྫས་ལ་ཞིབ་བཤེར་བྱེད་བཞིན་པའི་སྒང་ཡིན་སྐོར་བརྗོད་འདུག   ཨ་རིའི་མཉམ་འབྲེལ་ཞིབ་བཤེར་ཁང་གི་འགན་འཛིན་ལས་རོགས་ཌཱ་ཝེ་ཌི་བྷོ་ཌི་ཆི་(David Bowdich)ལགས་ཀྱིས་དམར་གསོད་དོན་རྐྱེན་དེ་འཇིགས་སྐུལ་རིང་ལུགས་པས་སྤེལ་བ་ཞིག་ཡིན་སྲིད་ཀྱང་། ད་ལྟའི་ཆར་ངེས་གཏན་མེད་སྐོར་བརྗོད་འདུག་ལ། མིང་མ་བརྗོད་པའི་ཁྲིམས་འཛིན་མི་སྣ་ཞིག་གིས་མེ་མདའ་རྒྱག་མཁན་གཅིག་སྨན་བཅོས་བདག་སྐྱོང་ཁང་དེའི་ལས་བྱེད་ཅིག་ཡིན་ཞིང་། ལས་བྱེད་རྣམས་བང་སྒྲིག་སྐབས་ལས་རོགས་ཤིག་དང་རྩོད་པ་ཤོར་ནས་ཕྱིར་ཐོན་སོང་བ་དང་། ཕྱིས་སུ་མི་གཅིག་གཉིས་ཤིག་ཁྲིད་ཡོང་སྟེ་གང་བྱུང་མེ་མདའ་བརྒྱབ་ཡོད་སྐོར་བརྗོད་འདུག   དམར་གསོད་དོན་རྐྱེན་བྱུང་རྗེས། སྲིད་འཛིན་ཨོ་བྷ་མས་ཨ་རིའི་ནང་གི་མེ་མདའི་དམར་གསོད་དོན་རྐྱེན་དེ་འཛམ་གླིང་གི་ཡུལ་ལུང་གང་དུའང་བྱུང་གི་མེད་པའི་གནས་སྟངས་ཤིག་ཏུ་གྱུར་ཡོད་སྟབས། ངེས་པར་དུ་ཨ་རིའི་ནང་བདེ་འཇགས་ཡོང་ཐབས་བྱེད་དགོས་སྐོར་བརྗོད་འདུག   སྤྱིར་བཏང་སན་བྷིར་ནར་ཌི་ནོ་གྲོང་ཁྱེར་ནང་མི་གྲངས་ཁྲི་ ༢༡ ལྷག་ལས་མེད་ཀྱང་། འདས་པའི་ལོ་བཅུ་གྲངས་ནང་གྲོང་ཁྱེར་དེའི་ནང་ལཱ་ཏེན་ཨ་མེ་རུ་ཁ་དང་ཨེ་ཤ་ཡ། དཀྱིལ་ཤར་ཡུལ་གྲུ་ཁག་ནས་སྐྱབས་བཅོལ་བ་ཧ་ཅང་མང་པོ་འབྱོར་ཡོད་སྟབས། ད་ལྟའི་ཆར་མི་གྲངས་འབུམ་ ༣༨ ལྷག་ཡོད་པའི་ཨ་རིའི་མི་འབོར་ཆེ་ཤོས་ཀྱི་གྲོང་ཁྱེར་ལོ་སི་ཨན་ཇེ་ལི་སིའི་(Los Angeles)མི་འབོར་དང་འདྲ་པོ་ཆགས་ཀྱི་ཡོད་འདུག   གསར་འགོད་པ། པདྨ་མཚོ།
