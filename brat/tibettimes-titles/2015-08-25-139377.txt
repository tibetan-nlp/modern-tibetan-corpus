རྡོ་རྗེ་སྒྲོལ་མ་ལཊ་བཙན་ཁྲིད།

ཉེ་སྔོན་རྔ་བ་རྫོང་ཁྲོམ་གཞུང་དུ་བུ་མོ་རྡོ་རྗེ་སྒྲོལ་མ་ཞུ་བ་ཞིག་གིས་ཞི་རྒོལ་གནང་སྟེ་ཉེན་རྟོག་པས་བཙན་ཁྲིད་བྱས་ཡོད་འདུག དེ་རིང་བོད་ནང་ནས་འདི་གར་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༠ ཉིན་གྱི་བོད་ནང་གི་ཞོགས་པའི་ཆུ་ཚོད་ ༨།༣༠ ཡས་མས་སུ། རྔ་བ་རྫོང་ཐོག་གི་གྲོང་ཁྱེར་དཀྱིལ་ལམ་ཞེས་པ་རུ་བོད་རིགས་བུ་མོ་རྡོ་རྗེ་སྒྲོལ་མ་ཞུ་བ་ཞིག་གིས་ཞི་རྒོལ་སྐད་འབོད་གནང་ཡོད་པ་དང་། དེ་མ་ཐག་ཉེན་རྟོག་པས་ཁོང་མོ་བཙན་ཁྲིད་བྱས་ཡོད་འདུག སྐབས་དེར་ལས་ཁུངས་ལ་འགྲོ་བཞིན་ཡོད་པའི་དངོས་སུ་མཐོང་མཁན་གྱི་བོད་རིགས་བུད་མེད་ཅིག་གིས། བུ་མོ་ཆུང་ཆུང་དེས་ཞི་རྒོལ་བྱེད་སྐབས་ཅུང་སྔ་བོ་ཡིན་པས་ཁྲོམ་གཞུང་དུ་མི་མང་པོ་མི་འདུག་ལ། ཁོང་མོས་ཞི་རྒོལ་གནང་བ་དངོས་སུ་མཐོང་མཁན་ཡང་ཉུང་ཉུང་རེད། ཁོང་མོས་ཞི་རྒོལ་བྱས་ནས་ཅུང་ཙམ་ལས་མ་འགོར་བར་ཉེན་རྟོག་པ་ ༡༠ གྲངས་ཤིག་བསླེབས་ཏེ་བཙན་ཁྲིད་བྱས་སོང་། དེ་ནས་བཟུང་ཁོང་མོ་གང་དུ་ཡོད་མེད་སོགས་ཅི་ཡང་གོ་རྒྱུ་མི་འདུག ཅེས་བརྗོད་ཡོད་པ་རེད། ཡང་གནས་ཚུལ་མཁོ་སྤྲོད་པས། གནས་ཚུལ་འདི་ཉིན་ཤས་སྔོན་ནས་སྤྱི་ཚོགས་བརྒྱུད་ལམ་ཁག་ཅིག་ཏུ་བསླེབས་སོང་། ཡིན་ནའང་། སྐབས་དེར་བུ་མོ་དེའི་མིང་སོགས་གནས་ཚུལ་གསལ་བོ་མི་འདུག ཁ་སང་ཞིག་ནས་བཟུང་མིང་སོགས་གནས་ཚུལ་གང་འཚམ་གསལ་བོ་ཆགས་འདུག་ཅེས་བརྗོད་བྱུང་། རྡོ་རྗེ་སྒྲོལ་མ་ལགས་ནི་རྔ་བ་རྫོང་རྨེའུ་རུ་མ་རུ་ཆེན་གཉིས་པའི་རྡོ་རབ་ཚང་གི་བུ་མོ་ཡིན་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
