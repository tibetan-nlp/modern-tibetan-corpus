ཕྱི་ཕྱོགས་ཀྱི་སྐད་འཕྲིན་ཚོགས་པའི་ནང་བོད་མིའི་འོས་བསྡུའི་སྐོར་གྱི་སྐད་ཆ་བཤད་པའི་ཉེས་མིང་འོག་བོད་མགོ་ལོག་མཚོ་སྐྱ་རིང་ཆུ་མགོ་སྡེ་གསར་གྱི་བོད་མི་གསུམ་འཛིན་བཟུང་བྱས་ཡོད་འདུག ཁ་སང་ཚེས་ ༣༠ ཉིན་གྱི་བོད་ནང་གི་སྔ་དྲོའི་ཆུ་ཚོད་ ༡༠ པ་ཡས་མས་ལ། མཚོ་སྔོན་ཞིང་ཆེན་མགོ་ལོག་ཁུལ་རྨ་སྟོད་རྫོང་མཚོ་སྐྱ་རིང་ཡུལ་མཚོ་ཆུ་མགོ་སྡེ་གསར་སྡེ་བའི་མགོ་ཁྲིད་བཞོན་པ་བསམ་གྲུབ་ཟེར་བ་ཞིག་དང་རོང་ཤེར་ཟེར་བ་ཞིག དེ་བཞིན། ལྷ་སྒྲོན་ཟེར་བའི་བུད་མེད་ཅིག་བཅས་མི་ ༣ རྨ་སྟོད་རྫོང་གི་ཉེན་རྟོགས་པས་འཛིན་བཟུང་བཙན་འཁྲིད་བྱས་ཏེ། ད་ལྟ་རྫོང་མི་དམང་ཁྲིམས་ཁང་དུ་བཀག་ཉར་བྱས་ཡོད་ཅིང་། ནང་མིར་ཐུག་འཕྲད་བྱེད་བཅུག་མི་འདུག ཁོང་ཚོ་འཛིན་བཟུང་བྱེད་དགོས་དོན་ཁ་གསལ་མེད་ཀྱང་། ས་གནས་མང་ཚོགས་ནང་ཁོང་ཅག་གིས་བོད་ནང་གི་གནས་ཚུལ་ཕྱི་གྱར་བྱས་པ་དང་། ཕྱི་ཕྱོགས་ཀྱི་སྐད་འཕྲིན་ཚོགས་པ་འདྲ་མིན་ནང་ ༢༠༡༦ ལོའི་བཙན་བྱོལ་བོད་མིའི་འོས་བསྡུའི་སྐོར་བཤད་པ་སོགས་ལ་བརྟེན་ཏེ། རྒྱ་ནག་གཞུང་གིས་ཆབ་སྲིད་ཉེས་འཛུགས་བྱས་པ་ཡིན་ཚུལ་ཞིག་ཤོད་བཞིན་ཡོད་འདུག བསམ་འགྲུབ་ནི་ད་ལྟ་རང་ལོ་ ༤༠ ཡས་མས་ཡིན་པ་དང་། ཁོང་དེ་སྔ་ལོ་མང་རིང་སྐྱ་རིང་ཡུལ་མཚོ་རུ་ཁག་དྲུག་པའི་མགོ་ཁྲིད་གནང་ཡོད་པ་མ་ཟད། དེང་སྐབས་རུ་ཁག་དང་པོ་གསུམ་པ་ལྔ་བ་སོགས་ཀྱི་རུ་སྤོ་དུད་ཁྱིམ་ ༡༥༠ ཡོད་པའི་མགོ་ཁྲིད་གཞོན་པའང་ཡིན་འདུག ལྷ་སྒྲོན་ནི་རྣམ་རྒྱལ་ཚང་ཟེར་བའི་མི་ཚང་ཞིག་གི་བུ་མོ་ཡིན་པ་དང་། མོ་ལ་ལོ་ ༡༡ ཅན་གྱི་བུ་མོ་དར་དཀར་ཟེར་བ་ཞིག་དང་། རང་ལོ་ ༨ ཡིན་པའི་བུ་ཚེ་རྒྱམ་ལོ་བཅས་བྱིས་པ་གཉིས་ཡོད་འདུག རོང་ཤར་ནི་ད་ལྟ་རང་ལོ་ ༢༩ ཡིན་པ་དང་ཁོང་གི་བཟའ་ཟླའི་མིང་ལ་གཡང་སྐྱིད་ཟེར་ཀྱི་ཡོད་འདུག རྒྱ་གར་ལྷོ་ཕྱོགས་ཀྱི་གསར་འགོད་པ་དོན་གྲུབ་བཀྲ་ཤིས།
