ཡུ་གུར་ནང་ཨག་ཚོམ་དང་གདོང་རས་བཀག་སྡོམ།

༢༠༡༤ ཟླ་ ༤ ཚེས་ ༡ ཉིན་བྷི་བྷི་སིས་སྤེལ་བའི་གནས་ཚུལ། རྒྱ་ནག་ནུབ་རྒྱུད་ཤིན་ཅང་ས་ཁུལ་དུ་དམ་དྲག་སྲིད་ཇུས་གསར་བ་ཞིག་ཁྱབ་བསྒྲགས་བྱས་ཡོད་པའི་ནང་ཁ་ཆེ་ཐལ་སྐྱོད་རིང་ལུགས་པར་རྡུང་རྡེག་ཚ་ནན་གྱི་ལས་འགུལ་སྤེལ་རྒྱུ་ཡིན་ལུགས་འཁོད་ཡོད་པ་རེད། རྒྱ་ནག་གཞུང་གི་དམ་དྲག་སྲིད་ཇུས་དེའི་ནང་ཁ་ཆེ་སྐྱེས་པ་ཚོས་རྒྱ་ནག་ཧན་མི་རིགས་དང་མི་འདྲ་བའི་ཨག་ཚོམ་རིང་པོ་འཇོག་མི་ཆོག་པ་དང་། ཁ་ཆེ་བུད་མེད་ཚོས་མི་མང་འདུས་སྡོད་བྱེད་ཡུལ་དག་ཏུ་ཁ་ཆེ་བའི་མགོ་དཀྲིས་དང་ལ་བ་ནགས་པོ་རིང་བོ་དེ་གྱོན་མི་ཆོག་པ། དེ་བཞིན། ཡུ་གུར་མི་མང་གིས་རྒྱ་ནག་གཞུང་འབྲེལ་བརྙན་འཕྲིན་ཚན་པར་ངེས་པར་དུ་བལྟ་དགོས་ལུགས་སོགས་འཁོད་ཡོད་འདུག ཤིན་ཅང་ས་ཁུལ་ནི་ཡུ་གུར་མི་རིགས་ཀྱི་ཕ་ས་ཕ་ཡུལ་ཡིན་ཞིང་། ཡུ་གུར་བ་ནི་གནའ་བོའི་ཁ་ཆེ་མི་རིགས་ཤིག་ལས་མཆེད་ཡོང་བ་རེད། ཡུ་གུར་མི་མང་གིས་དུས་རྒྱུན་ཁོང་ཚོར་རྒྱ་ནག་གཞུང་དང་མི་མང་གིས་གི་རིགས་དབྱེ་འབྱེད་ཀྱི་དཀའ་ངལ་ཡོད་སྐོར་རྗོད་ཀྱིན་ཡོད། ཉེ་བའི་ལོ་ཤས་རིང་ཤིན་ཅང་ས་ཁུལ་དུ་ཁྲག་དྲི་བྲོ་བའི་དྲག་ཟིང་རྒྱུན་མི་ཆད་པར་བྱུང་དང་འབྱུང་བཞིན་ཡོད་ཅིང་། རྒྱ་ནག་གཞུང་ཕྱོགས་ནས་དྲག་ཟིང་དེ་དག་ཁ་ཆེ་ཐལ་སྐྱོད་རིང་ལུགས་པ་དང་ཁ་བྲལ་རིང་ལུགས་པ་ཚོས་བཟོས་པ་ཡིན་སྐོར་རྗོད་ཀྱི་ཡོད། ཡིན་ནའང་རྒྱལ་སྤྱིའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ཚོགས་སྡེ་ཁག་གིས་ཤིན་ཅང་ས་ཁུལ་གྱི་ཟིང་འཁྲུག་དེ་དག་ནི་རྒྱ་ནག་གཞུང་གི་དྲག་ཆས་ཉེན་རྟོག་པ་ཚོས་དྲག་གནོན་བྱས་པ་ལས་བྱུང་བ་ཡིན་སྐོར་བཤད་ཡོད། ༡. པེ་ཅིན་གཞུང་གི་ཐོ་བས་ཤིན་ཅང་ས་ཁུལ་གྱི་ཤིང་འབྲས་གཅོག ༢. རྒྱ་ནག་གིས་ཤིན་ཅང་མི་མང་གི་ཕྱི་སྤྱོད་ལག་འཁྱེར་བཙན་འཕྲོག་བྱེད། ༣. དོན་ངོ་མར་རྒྱ་ནག་གཞུང་གིས་འཇིགས་སྐུལ་དྲག་སྤྱོད་པར་དྲག་གནོན་བྱེད་ཀྱི་ཡོད་དམ། རྒྱ་ནག་གཞུང་གིས་ཐེངས་འདིར་གསར་དུ་བསྒྲགས་པའི་དྲག་གནོན་སྲིད་ཇུས་དེ་དང་གཅིག་མཚུངས་དེ་སྔོན་ནས་ལག་བསྟར་བྱས་དང་བྱེས་བཞིན་ཡོད་ལ། བདུན་ཕྲག་འདིའི་ནང་དྲག་གནོན་སྲིད་ཇུས་དེ་དག་ཁྲིམས་མཐུན་ཡིན་ལུགས་བསྒྲགས་ཡོད་པ་རེད། རོའུ་ཊར་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། རྒྱ་ནག་གི་དྲག་གནོན་སྲིད་ཇུས་གསར་བ་དེའི་ནང་གཤམ་གྱི་གནད་དོན་ཁག་འཁོད་ཡོད་འདུག ༡. ཡུ་གུར་བའི་བྱིས་པ་ཚོ་སྲིད་གཞུང་གི་སློབ་གྲྭ་ཁག་ནང་གཏོང་མི་ཆོག ༢. ཤིན་ཅང་ནང་འཆར་ལྡན་བུ་སྐྱེས་བཅོས་བསྒྱུར་དེ་ལག་བསྟར་བྱ་མི་ཆོག ༣. གཞུང་གིས་ཐག་གཅོད་གཏན་འབེབས་བྱས་པའི་ཁྲིམས་མཐུན་ཡིག་ཆར་གཏོར་བརླག་གཏོང་མི་ཆོག ༤. གཉེན་སྒྲིག་ལ་ཁ་གཡར་བའི་ཆོས་ལུགས་བྱེད་སྒོ་རིགས་སྤེལ་མི་ཆོག ཅེས་སོགས་བཀོད་ཡོད་འདུག དམ་དྲག་སྲིད་ཇུས་དེའི་ནང་ད་དུང་རླངས་འཁོར་ས་ཚིགས་དང་གནམ་ཐང་སོགས་མི་མང་འདུས་སྡོད་ཁུལ་དུ་ཁ་ཆེ་བའི་མགོ་དཀྲིས་དང་། ལུས་ཧྲིལ་བོ་གཡོགས་པའི་གྱོན་ཆས་རིགས་གྱོན་མཁན་མཐོང་ཚེ་ཉེས་ཆད་གཅོད་རྒྱུ་ཡིན་ཞིང་། སྲིད་ཇུས་དེ་ཞིན་ཅང་མི་དམངས་འཐུས་མིས་ངོས་ལེན་བྱས་ཏེ་ཤིན་ཅང་ས་གནས་གཞུང་གི་དྲ་ཚིགས་སྟེང་སྤེལ་ཡོད་འདུག དེ་སྔོན་རྒྱ་གཞུང་གིས་ཤིན་ཅང་ལ་དམ་དྲག་སྲིད་ཇུས་ཤིག་གཏན་འབེབས་བྱས་པའི་ནང་ཡུ་གུར་མི་མང་གི་ཕྱི་སྐྱོད་ལག་འཁྱེར་བཀག་སྡོམ་བྱས་ཡོད། ཤིན་ཅང་དང་ཡུ་གར། ༡. ཡུ་གུར་མི་རིགས་ཀྱི་མཆེད་ཁོངས་ནི་ཏིར་ཀིའི་ཁ་ཆེ་མི་རིགས་རེད། ༢. ཤིན་ཅང་ས་ཁུལ་དུ་ཡུ་གུར་མི་རིགས་བརྒྱ་ཆ་ ༤༥ དང་ཧན་མི་རིགས་བརྒྱ་ཆ་ ༥༠ ཆགས་ཡོད། ༣. ཤར་ཊིར་ཀི་སི་ཐན་ནང་ ༡༩༤༩ ལོ་ནས་རྒྱ་ནག་གཞུང་གིས་ཤིན་ཅང་ཞེས་དབང་བསྒྱུར་བྱས་ཡོད། ༤. སྐབས་དེ་ནས་བཟུང་རྒྱ་ནག་ཧན་མི་རིགས་འབོར་ཆེན་ཤིན་ཅང་ས་ཁུལ་དུ་གནས་སྤོ་བྱས་ཡོད། ༥. ཡུ་གུར་མི་མང་གིས་གྲངས་མང་ཧན་མི་རིགས་ཁྲོད་གྲངས་ཉུང་ཡུ་གུར་མི་རིགས་ཀྱི་ཡུལ་སྲོལ་གོམས་གཤིས་དང་ཆོས་ལུགས་རིག་གནས་ཉམས་ཆག་ཏུ་འགྲོ་སྲིད་པའི་སེམས་ཚབ་བྱེད་བཞིན་ཡོད། གསར་འགོད་པ། པདྨ་རྒྱལ་གྱིས་བསྒྱུར་སྒྲིག་བྱས།
