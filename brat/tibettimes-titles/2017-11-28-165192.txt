རྒྱ་ནག་དམག་སྤྱི་ཆེན་མོས་རང་སྲོག་བཅད།

ལྐོག་ཟ་རུལ་སུངས་ཀྱི་དོགས་གཞི་ཡོད་པའི་རྒྱ་ནག་དབུས་དམག་དོན་ལྷན་ཁང་ཆབ་སྲིད་ཚན་པའི་འགན་འཛིན་དང་དམག་སྤྱི་ཆེན་མོ་ཀྲང་ཡང་གིས་སྐེ་བཏགས་ནས་རང་ཤི་བརྒྱབས་འདུག དེ་རིང་རྒྱ་ནག་ལྷོ་རྒྱུད་ཞོགས་པའི་ཆ་འཕྲིན་གྱིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཁ་སང་རྒྱ་ནག་གི་གཞུང་འབྲེལ་གསར་ལམ་ཁག་ཏུ། ཟླ་འདིའི་ཕྱི་ཚེས་༢༣ ཉིན་གྱི་ཞོགས་པར་རྒྱ་ནག་དབུས་དམག་དོན་ལྷན་ཁང་གི་དམག་སྤྱི་ཆེན་མོ་ཀྲང་ཡང་ (Zhang Yang) གིས་སྐེ་བཏགས་ནས་རང་ཤི་བརྒྱབས་ཡོད་སྐོར་བསྒྲགས་འདུག རྒྱ་ནག་གཞུང་འབྲེལ་ཤིན་ཧྭ་གསར་ཁང་གིས་དམག་སྤྱི་ཀྲང་ཡང་ལོ་ ༢༠༡༦ ལོར་ལྐོག་ཟ་རུལ་སུངས་བྱས་ཡོད་པའི་དོགས་གཞིའི་འོག་འཛིན་བཟུང་བྱས་པའི་རྒྱ་ནག་མི་མང་བཅིངས་གྲོལ་དམག་གི་དམག་སྤྱི་ཆེན་མོ་ཟུར་པ་གོ་པའོ་ཞོང་ (Guo Boxiong) དང་། ཕྱི་ལོ་༢༠༡༤ ལོར་འཛིན་བཟུང་བྱས་པའི་རྒྱ་ནག་དབུས་དམག་དོན་ལྷན་ཁང་གི་གཙོ་འཛིན་གཞོན་པ་ཟུར་པ་ཞི་ཚེ་ཧུའུ་ (Xu Caihou) གཉིས་དང་འབྲེལ་བ་ཡོད་སྐོར་བརྗོད་ཡོད་པ་རེད། ཝེ་ཁེ་རིག་མཛོད་ཀྱི་གནས་ཚུལ་ལྟར་ན། འདི་ལོའི་ཕྱི་ཟླ་ ༨ པའི་ཚེས་ ༢༨ ཉིན་རྒྱ་ནག་དབུས་དམར་ཤོག་ལྷན་ཁང་གིས་ཀྲང་ཡང་གི་ཁྲིམས་འགལ་དོགས་གཞིར་ཞིབ་བཤེར་ཆོག་པའི་ཆོག་མཆན་སྤྲད་དེ། རང་ཁྱིམ་དུ་འདྲི་རྩད་ཞིབ་བཤེར་བྱས་པ་བརྒྱུད་ཁོས་ཁྲིམས་སྒྲིག་དང་འགལ་བའི་ནག་ཉེས་ཚབས་ཆེན་བསགས་ཡོད་པ་དང་། ལྐོག་རྔན་འབོར་ཆེན་བླངས་བ། སྒེར་གྱི་རྒྱུ་དངོས་མང་པོའི་འབྱུང་ཁུངས་གསལ་པོ་མེད་པ་སོགས་ཀྱི་ནག་ཉེས་ར་འཕྲོད་བྱུང་ཡོད་སྟབས། ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༣ ཉིན་གྱི་ཞོགས་པར་རང་ཁྱིམ་ནས་རང་ཤི་བརྒྱབས་ཡོད་འདུག དམག་སྤྱི་ཆེན་མོ་ཀྲང་ཡང་རང་ལོ་ ༦༦ ཡིན་པ་དང་། རྒྱ་ནག་བཅིངས་འགྲོལ་དམག་གི་རྒྱལ་ཡོངས་བདེ་སྲུང་གཙུག་ལག་སློབ་ཁང་ (PLA National Defense University) ནས་སློབ་མཐར་ཕྱིན་ཡོད་པ་དང་། རྒྱ་ནག་དབུས་དམར་ཤོག་སློབ་གྲྭའི་ (Central Party School) ནང་དུའང་སློབ་གཉེར་བྱས་མྱོང་ཡོད་འདུག ཕྱི་ལོ་ ༢༠༠༧ ནས་ རྒྱ་ནག་ཀོང་ཀྲུའུ་གྲོང་ཁྱེར་དམག་ཁུལ་ཆབ་སྲིད་ཁང་གི་འགན་འཛིན་དང་། ཕྱི་ལོ་ ༢༠༡༢ ཟླ་ ༡༠ ནས་བཟུང་རྒྱ་ནག་མི་མང་བཅིངས་གྲོལ་དམག་སྤྱི་ཁྱབ་ཁང་གི་ཆབ་སྲིད་ཚན་པའི་འགན་འཛིན་བྱས། ཕྱི་ལོ་ ༢༠༡༦ ཟླ་ ༡ ནས་བཟུང་རྒྱ་ནག་དབུས་དམར་ཤོག་དམག་དོན་ལྷན་ཁང་ཆབ་སྲིད་ལས་དོན་ཁང་གི་འགན་འཛིན་དུ་བསྐོས་ཡོད་འདུག སི་ཨན་ཨན་གསར་ཁང་གི་དཔྱད་ཞིབ་ལྟར་ན། ཕྱི་ལོ་༢༠༡༢ ལོ་ནས་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱིས་ལྐོག་ཟ་རུལ་སུངས་གཙང་བཟོའི་ལས་འགུལ་སྤེལ་བ་ནས་ད་བར་རྒྱ་ནག་མི་མང་ས་ཡ་༡།༤ ལྷག་ལ་ཞིབ་བཤེར་ཉེས་ཆད་ཕོག་ཡོད་པ་དང་། དེའི་ནང་རྒྱ་ནག་གི་མཐོ་རིམ་མགོ་འཁྲིད་༣༠༠ མ་ཟིན་ཙམ་དང་ཉུང་མཐར་དམག་དཔོན་སོགས་དམག་དོན་སྲི་ཞུའི་ལས་བྱེད་ ༡༣༠༠༠ ཙམ་ཡོད་འདུག སྲིད་འཛིན་ཞི་ཅིན་ཕིན་གྱིས་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་ལས་འགུལ་དེ་ནི་རང་ཉིད་ཀྱི་དབང་ཆ་རྒྱ་ནག་དམར་ཤོག་ཚོགས་པའི་ནང་སྲ་བརྟན་ཡོང་བ་དང་། རང་ལ་ཁ་གཏད་གཅོག་མཁན་ཚོ་རྩ་མེད་བཟོ་བའི་ཐབས་བྱུས་ཤིག་ཏུ་གནད་དོན་ཞིབ་དཔྱོད་པ་ཚོས་ངོས་འཛིན་བྱེད་བཞིན་ཡོད་པ་དང་། རྒྱ་ནག་དམར་ཤོག་ཚོགས་པའི་ཚོགས་ཆེན་ཐེངས་༡༩ པའི་ཐོག་སྲིད་འཛིན་ཞི་ཅིན་ཕིན་ལ་གདེང་འཇོག་ཆེན་པོ་ཐོབ་པ་བཞིན། མགོ་འཁྲིད་མའོ་ཙེ་ཏུང་གི་རྗེས་སུ་རྒྱ་ནག་གི་མགོ་འཁྲིད་སུའང་དེ་ལྟར་ཐོབ་མྱོང་མེད་པ་དང་། རྒྱ་ནག་གི་ཕྱི་ནང་ཀུན་ནས་ཞི་ཅིན་ཕིན་ལ་རྒྱ་ནག་གི་སྟག་རྒོད་ཅེས་བསྟོད་བསྔགས་གཏོང་གི་ཡོད་འདུག ཞི་ཅིན་ཕིན་གྱིས་ཕྱི་ལོ་ ༢༠༡༢ ལོར་སྲིད་དབང་བཟུང་བ་ནས་ལོ་ལྟར་སྲིད་གཞུང་གི་དཔོན་རིགས་རང་ཤི་བརྒྱབ་མཁན་གྱི་གྲངས་འབོར་ཇེ་མང་ནས་ཇེ་མང་དུ་སོང་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༡༥ ལོའི་རྒྱ་ནག་གཞུང་འབྲེལ་གྱི་བསྡོམས་རྩིས་ཕྱིར་བསྒྲགས་ནང་། ཕྱི་ལོ་ ༢༠༡༢ ལོར་མི་གྲངས་ ༡༢ ཀྱིས་རང་ཤི་བརྒྱབས་པའི་ནང་མི་ ༤ ལྐོག་ཟ་རུལ་སུངས་དང་ལྐོག་རྔན་སྤྲོད་ལེན་ལ་འབྲེལ་བ་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༡༣ ལོར་མི་ ༧ བྱུང་བའི་ནང་ ༢ ལྐོག་རྔན་རུལ་སུངས་ལ་འབྲེལ་བ་ཡོད་པ། ཕྱི་ལོ་ ༢༠༡༤ ལོར་ ༣༩ བྱུང་བའི་ནང་མི་ ༧ ལྐོག་རྔན་དང་ལྐོག་ཟ་རུལ་སུངས་ལ་འབྲེལ་བ་ཡོད་སྐོར་བརྗོད་ཡོད་པ་རེད། ཡང་ཞིབ་འཇུག་བསྟི་གནས་ཤིག་གི་བསྡོམས་རྩིས་དབྱེ་ཞིབ་ལྟར་ན། ཕྱི་ལོ་ ༢༠༡༥ ལོར་སྡེ་ཚན་གྱི་མགོ་འཁྲིད་གཞོན་པ་ཡན་གྱི་དཔོན་རིགས་ ༡༥༠༠ ཡིས་རང་ཤི་བརྒྱབས་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༡༦ ལོར་རྒྱ་ནག་སྡེ་ཚན་གྱི་མགོ་འཁྲིད་གཞོན་པ་ཡན་གྱི་དཔོན་རིགས་ ༡༧༠༠ ཡིས་རང་ཤི་བརྒྱབས་ཡོད་སྟབས། ལོ་རེ་བཞིན་རང་ཤི་བརྒྱབ་མཁན་གྱི་གྲངས་འབོར་འཕར་སྣོན་བྱུང་ཡོད་པ་མ་ཟད། རྒྱ་ནག་དམར་ཤོག་ཚོགས་པའི་ལོ་རྒྱུས་ནང་དཔོན་རིགས་རང་ཤིའི་གྲངས་འབོར་མང་ཤོས་སུ་ངོས་འཛིན་བྱེད་ཀྱི་ཡོད་པའི་ཕྱི་ལོ་ ༡༩༦༦ སྟེ། རིག་གནས་གསར་བརྗེ་ཆེན་པོའི་སྐབས་ཀྱི་རང་ཤི་གྲངས་འབོར་ ༡༣༠༠ ལས་ཀྱང་བརྒལ་སྟེ། དམར་ཤོག་ཚོགས་པའི་ལོ་རྒྱུས་ནང་སྲིད་གཞུང་གི་དཔོན་རིགས་རང་ཤིའི་གྲངས་འབོར་མང་ཤོས་དེ་ཆགས་ཡོད་འདུག གསར་འགོད་པ། ཉི་མ་ཚེ་རིང་།
