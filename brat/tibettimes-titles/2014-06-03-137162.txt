ཆབ་མདོ་ཚ་བ་དཔའ་ཤོད་དུ་དམ་དྲག

བོད་རང་སྐྱོང་ལྗོངས་ཆབ་མདོ་ས་ཁུལ་ཚ་བ་དཔའ་ཤོད་རྫོང་དུ་སྲིད་གཞུང་ལས་བྱེད་དང་དམག་མི། ལས་དོན་རུ་ཁག་མཉམ་རུབ་ཀྱིས་དམ་དྲག་བྱེད་ཀྱི་ཡོད་འདུག ཕ་ཡུལ་ཆ་མདོ་ནས་ཡིན་པའི་བོད་མི་རྒྱ་མཚོས་གསར་ལམ་དུ་གནས་ཚུལ་ངོ་སྤྲོད་གནང་དོན་ལྟར་ན། ཟླ་སྔོན་མའི་ཕྱི་ཚེས་ ༣༡ ཉིན་ནས་བཟུང་བོད་རང་སྐྱོང་ལྗོངས་ཆབ་མདོ་ས་ཁུལ་ཚ་བ་དཔའ་ཤོད་རྫོང་སྲིད་གཞུང་གི་ལས་བྱེད་དང་། ས་གནས་ཉེན་རྟོག་པ། ས་གནས་ལས་དོན་རུ་ཁག་མཉམ་འབྲེལ་ཐོག དགོན་སྡེ་ཁག་དང་གྲོང་སྡེ་ཁག་ཏུ་ཚོགས་འདུ་རིམ་པ་སྐོང་ཚོགས་གནང་སྟེ། ཕྱི་ཟླ་ ༦ དང་ ༧ བཅས་ཟླ་གཉིས་རིང་ཚ་བ་དཔའ་ཤོད་རྫོང་གི་བོད་མི་སྐྱ་སེར་རྒན་གཞོན་གཅིག་ཀྱང་ས་གནས་གཞན་དུ་འགྲོ་མི་ཆོག་པར་མ་ཟད། དམིགས་བསལ་ལྷ་ས་དང་སྟོད་མངའ་རིས། ཨ་མདོའི་ཕྱོཊ་བཅས་སུ་འགྲོ་མི་ཆོག་ཚུལ་དང་། ད་དུང་ས་གནས་བོད་མི་ཚོར་སྨན་ཁང་དུ་བསྟེན་གཏུག་བྱེད་དགོས་པའི་ནད་པ་དང་། ས་གནས་གཞན་གྱི་སློབ་གྲྭར་འགྲིམས་བཞིན་པའི་སློབ་མའི་ཁྱིམ་བདག་སོཊ་ས་གཞན་དུ་འགྲོ་དགོས་བྱུང་ཚེ་ས་གནས་སྲིད་གཞུང་དང་ས་གནས་ལས་དོན་རུ་ཁག ཉེན་རྟོག་པ་བཅས་ལ་ཆོག་མཆན་ཞུ་དགོས་པ་དང་། ཆོག་མཆན་མེད་པར་ཕྱིན་ཚེ་ཆབ་སྲིད་ཀྱི་ནག་ཉེས་འོག་བཀག་ཉར་བྱ་རྒྱུ་ཡིན་ཚུལ་སོཊ་བསྒྲགས་འདུག གནས་ཚུལ་མཁོ་སྤྲོད་པས་བརྗོད་པ་ལྟར་ན། འདི་ལོའི་ཕྱི་ཟླ་ ༤ ནས་ ༨ བར་ཚ་བ་དཔའ་ཤོད་རྫོང་སྲིད་གཞུང་གིས་ས་གནས་བོད་མི་གཅིག་ཀྱང་སྟོད་མངའ་རིས་ས་ཁུལ་དུ་གནས་མཇལ་འགྲོ་མི་ཆོག་པར་བཟོས་ཡོད་པར་ཟད། ཕྱི་ཟླ་ ༥ ཚེས་ ༣༡ ཉིན་ནས་བཟུང་རྫོང་ཁོངས་སུ་དམག་མི་དང་ཉེན་རྟོག་པ་སོགས་འགྲེམས་འཇོག་གིས་དམ་དྲག་ལྟ་རྟོག་བྱེད་ཀྱིན་ཡོད་པར་མ་ཟད། དམིགས་བསལ་ལྷ་སར་འགྲོ་མཁན་གྱི་རླངས་འཁོར་སོགས་ལ་སྔོག་བཤེར་བྱེད་ཀྱི་ཡོད་འདུག ད་དུང་དགོན་སྡེ་ཁག་ཏུ་ཆོས་ལུགས་ཀྱི་བྱེད་སྒོ་སྤེལ་རྒྱུར་དམ་དྲག་དང་། གྲྭ་བ་ཚོ་བསྔོ་སྨོན་མཆོད་འབུལ་དང་ཞབས་རིམ་སོགས་ཀྱི་ཆེད་དུ་འགྲོ་བར་ས་གནས་སྲིད་གཞུང་དང་ལས་དོན་རུ་ཁག ཉེན་རྟོག་པ་བཅས་ཀྱི་གམ་དུ་སོང་སྟེ་མི་དེའི་མིང་དང་ཆོས་འདོན་དགོས་པའི་རྒྱུ་མཚན། ཁག་ཐེག་ཡི་གེའི་ཐོག་མཛུབ་ཐེབ་བཅས་རྒྱག་དགོས་ཀྱི་ཡོད་འདུག འདི་ལོའི་ཕྱི་ཟླ་ ༥ ཚེས་ ༢༠ ཡས་མས་ནས་བཟུང་ཚ་བ་དཔའ་ཤོད་རྫོང་གི་བོད་མི་ཚོར་རླངས་འཁོར་དང་འཕྲུལ་རྟ་སོགས་ཀྱི་ནང་སྣུམ་ལྡུག་པར། ངེས་པར་དུ་ས་གནས་ལས་དོན་རུ་ཁག་དང་ས་གནས་སྲིད་གཞུང་། ས་གནས་ཉེན་རྟོག་ལས་ཁུངས་བཅས་སུ་ཐོབ་ཐང་འཛིན་ཡིག་དང་ཁྱིམ་ཐོ་བཅས་བརྒྱུད་ཆོག་མཆན་ཞུ་དགོས་པར་མ་ཟད། སྣུམ་ཉོ་དགོས་པའི་རྒྱུ་མཚན་གནས་ལུགས་གསལ་བཤད་བྱེད་དགོས་ཀྱི་ཡོད་པ་སོགས་བོད་མི་ཚོར་དམ་དྲག་དང་།  སྤྱིར་བཏང་རྟ་ཡི་ལོ་འདིར་བོད་མི་ཚོ་གངས་སྐོར་སོགས་སུ་འགྲོ་གྱིན་ཡོད་ཀྱང་། ད་ཆར་རྒྱ་གཞུང་གིས་དམ་དྲག་འོག་བོད་མི་གངས་སྐོར་དང་གནས་སྐོར་སོགས་སུ་འགྲོ་མ་ཐུབ་པར་ལྷག་ཡོད་འདུག ཡང་བོད་མིའི་སྒྲིག་འཛུཊ་ཀྱི་དྲ་རྒྱར་གནས་ཚུལ་བཀོད་པ་ལྟར་ན། ཟླ་སྔོན་མའི་ཕྱི་ཚེས་ ༣༡ ཉིན་ནས་བཟུང་རྒྱ་གཞུང་གིས་གངས་རིན་པོ་ཆེར་འགྲོ་མཁན་གྱི་གནས་བསྐོར་བ་དང་ཡུལ་སྐོར་སྤྲོ་འཆམ་པའི་གྲངས་འབོར་ཚོད་འཛིན་བྱེད་ཀྱིན་པར་མ་ཟད། བོད་རང་སྐྱོང་ལྗོངས་ཀྱི་ལས་བྱེད་སུ་འདྲ་ཞིག་ཡིན་ཡང་འདི་ལོའི་ཕྱི་ཟླ་ ༤ ནས་ ༩ བའི་བར་དགོངས་ཞུ་བྱེད་མི་ཆོག་པ་དང་། དགོངས་པ་ཞུས་ཚེ་ལས་གནས་ནས་གནས་དབྱུང་གཏོང་རྒྱུ་ཡིན་ཚུལ་བསྒྲགས་ཡོད་འདུག་ལ། ད་དུང་བོད་མི་ཚོར་ཕྱི་བསྐྱོད་ལག་འཁྱེར་གསར་པ་སྤྲོད་རྒྱུ་དང་དུས་ཚོད་འཕར་འགྱངས་མཚམས་འཇོག་དང་། དེ་སྔ་སྤྲད་ཟིན་པ་རྣམས་ཀྱི་ལག་འཁྱེར་ཕྱིར་སློག་བྱེད་དགོས་པའི་བཙན་བཀའ་གཏོང་གི་ཡོད་འདུག དམིགས་བསལ་འདི་ལོའི་ཕྱི་ཟླ་ ༧ པར་༧གོང་ས་མཆོག་གིས་ལ་དྭགས་སུ་གནང་འཆར་ཡོད་པའི་དུས་འཁོར་དབང་ཆེན་ལ་འགྲོ་མཁན་བྱུང་ཚེ། བོད་མི་དེའི་ཐེམ་ཐོ་གཞུང་བཞེས་གཏོང་རྒྱུ་ཡིན་ཚུལ་སོགས་གསལ་བསྒྲགས་གནང་ཡོད་འདུག གསར་འགོད་པ། དོན་གོ
