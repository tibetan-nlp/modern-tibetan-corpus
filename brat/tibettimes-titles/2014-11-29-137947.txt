སྐྱིད་གྲོང་འགག་སྒོ་ཕྱེ་རྒྱུ་ཡིན།

འདི་ལོའི་ཕྱི་ཟླ་ ༡༢ པའི་ནང་བོད་ལྗོངས་གཞིས་ཀ་རྩེ་སྐྱིད་གྲོང་འགག་སྒོ་བསྐྱར་དུ་སྒོ་ཕྱེ་རྒྱུ་ཡིན་འདུག རྒྱ་ནག་གིས་སྐམ་སའི་ཚོང་འགྲུལ་འགག་སྒོ་ཆེ་ཤོས་སུ་བརྩི་བའི་བོད་ལྗོངས་སྐྱིད་གྲོང་འགག་སྒོའི་ཚོང་འགྲུལ་བྱ་བ་ཕྱི་ལོ་ ༡༩༨༥ ཙམ་ནས་བཟུང་མཚམས་བཞག་སྟེ་ལོ་ངོ་ ༢༠ ཙམ་འགྲོ་བཞིན་ཡོད་ཀྱང་། རྒྱ་གཞུང་གི་བོད་ཕྱི་ཕྱོགས་སུ་སྒོ་མོ་འབྱེད་པའི་སྲིད་བྱུས་ཟེར་བའི་འོག འདི་ལོའི་ཕྱི་ཟླ་ ༡༢ པའི་ནང་བསྐྱར་དུ་སྒོ་ཕྱེ་རྒྱུ་ཡིན་པ་ཕྱིར་བསྒྲགས་བྱས་འདུག ཉེ་སྔོན་སྐྱིད་གྲོང་འགག་སྒོའི་འགན་འཛིན་ཝང་ལུང་ཟེར་བ་དེས་གསར་འགོད་པར། སྐྱིད་གྲོང་འགག་སྒོ་ནི་རྒྱ་ནག་གི་སྐམ་སའི་ཚོང་འགྲུལ་འགག་སྒོ་ཆེ་ཤོས་དེ་ཡིན་ལ། དེ་ནི་ཕྱི་ལོ་ ༡༩༦༢ ལོར་གསར་དུ་བཙུགས་ཤིང་། ཕྱི་ལོ་ ༡༩༧༨ ལོར་རྒྱལ་ཁབ་ཀྱི་རིམ་པ་དང་པོའི་སྐམ་སའི་ཚོང་འགྲུལ་འགག་སྒོར་གཏན་འཁེལ་ཡོད་ཀྱང་། དུས་རབས་གོང་མའི་ལོ་རབས་ ༨༠ དུས་མགོ་ནས་བཟུང་སྐྱིད་གྲོང་འགག་སྒོའི་ཚོང་འགྲུལ་བྱ་བ་ཉམས་དམས་སུ་ཕྱིན། ཕྱི་ལོ་ ༢༠༠༩ ལོའི་ལོ་སྨད་དུ་འགག་སྒོ་སླར་གསོའི་ལས་དོན་སྤེལ་མགོ་བརྩམས་ཏེ་ལེགས་འགྲུབ་བྱུང་ཡོད་ལ། དེས་སྐྱིད་གྲོང་འགག་སྒོའི་དེ་སྔའི་འཁྲུག་ཆ་དོད་པའི་རྣམ་པ་སླར་གསོ་བྱ་ཐུབ་སྐོར་བརྗོད་འདུག ཁ་སང་བོད་ལྗོངས་ཆ་འཕྲིན་དྲ་བའི་ཐོག་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། བོད་ཀྱི་ལས་དོན་སྐོར་གྱི་བཞུགས་མོལ་ཚོགས་འདུ་ཐེངས་ལྔ་པ་ཞེས་པའི་ཐོག བོད་ཕྱི་ཕྱོགས་སུ་སྒོ་མོ་འབྱེད་པའི་འཐབ་ཇུས་བཏོན་ཏེ། བོད་རང་སྐྱོང་ལྗོངས་ཀྱི་སྐྱིད་གྲོང་འགག་སྒོ་གཙོ་གནད་དུ་བཟུང་སྟེ་འཛུགས་སྐྲུན་བྱ་རྒྱུ་དང་། འགྲམ་འགག་སྒོ་བརྟན་པོའི་ངང་རིམ་སྤོར་དང་གྲོ་མོ་འགག་སྒོ་སླར་གསོ། དེ་བཞིན་སྤུ་ཧྲེང་དང་རི་འོག་བཅས་ཀྱི་འགག་སྒོ་རྣམས་རིམ་བཞིན་འཕེལ་རྒྱས་གཏོང་བའི་གཏན་འབེབས་བྱས་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
