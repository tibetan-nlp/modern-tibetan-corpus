དགེ་འདུན་པ་ཞིག་ཀློད་གྲོལ་བཏང་།

རྔ་བ་ཀིརྟི་དགོན་གྱི་གྲྭ་བློ་བཟང་རྒྱ་མཚོ་ལགས་ཀྱི་བཙོན་འཇུག་དུས་ཡུན་ལོ་གསུམ་ཚང་སྟེ་སི་ཁྲོན་མེན་ཡཱང་བཙོན་ཁང་ནས་ཀློད་གྲོལ་བཏང་ཡོད་འདུག འདི་གར་བོད་མི་ཞིག་གིས་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། དེ་རིང་སྔ་དྲོར་རྔ་བ་ཀིརྟི་དགོན་གྱི་དགེ་འདུན་པ་བློ་བཟང་རྒྱ་མཚོ་ལགས་ལོ་ངོ་ ༣ གྱི་བཙོན་འཇུག་དུས་ཡུན་ཚང་སྟེ། སི་ཁྲོན་མེན་ཡཱང་བཙོན་ཁང་ནས་ཀློད་གྲོལ་བཏང་ནས་རང་དགོན་དུ་འབྱོར་ཡོད་འདུག ཕྱི་ལོ་ ༢༠༡༤ ཟླ་ ༤ ཚེས་ ༢ ཉིན་དགེ་འདུན་པ་བློ་བཟང་རྒྱ་མཚོ་ལགས་རྔ་བ་རྫོང་མཁར་གྱི་གཞུང་ལམ་དུ་ཐོན་ཏེ། ལག་ཏུ་ཁོང་རང་ཉིད་ཀྱིས་བྲིས་པའི་བོད་ཀྱི་རྒྱལ་དར་ཞིག་བཟུང་སྟེ། ༼བོད་ལ་རང་དབང་དགོས། ༧རྒྱལ་བ་རིན་པོ་ཆེ་བོད་ལ་གདན་འདྲེན་ཞུ་དགོས།༽ཞེས་སྐད་འབོད་གནང་ཡོད་ཀྱང་། སྐབས་ཐོག་དེར་རྒྱ་གཞུང་ཉེན་རྟོག་པའི་ལག་ཏུ་མ་ཤོར་བར་ཕྱིར་དགོན་པའི་ནང་འབྱོར་ཡོད་པ་རེད། ཡིན་ནའང་། ལོ་དེའི་ཕྱི་ཟླ་༤ ཚེས་ ༡༥ ཉིན་ཉེན་རྟོག་པ་དགོན་པའི་ནང་དུ་ཡོང་སྟེ་ཁོང་བཙན་འཁྲིད་བྱས་ཡོད་པ་རེད། ལོ་དེའི་ཕྱི་ཟླ་ ༡༡ ཚེས་ ༧ ཉིན་རྔ་བ་རང་སྐྱོང་ཁུལ་གྱི་ལྟེ་གནས་འབར་ཁམས་འབྲིང་རིམ་མི་མང་ཁྲིམས་ཁང་གིས། རྒྱལ་ཁབ་ལ་ངོ་ལོག་གི་བྱ་སྤྱོད་སྤེལ་བའི་ཉེས་མིང་འོག་ཁྱིམ་མིར་ཁྲིམས་རྩོད་པ་གླ་རུ་མ་བཅུག་པར་ཁོང་ལ་ལོ་ ༣ བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་པ་དང་། སྐབས་དེར་རྔ་བ་ཀིརྟི་དགོན་པའི་གྲྭ་བློ་བཟང་བསྟན་པ་ཞུ་བ་ཞིག་ལའང་ལོ་ ༢ ཀྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག དགེ་འདུན་པ་བློ་བཟང་རྒྱ་མཚོ་ལགས་དགུང་ལོ་ ༢༣ ཡིན་ཞིང་། རྔ་བ་རྫོང་རྨེའུ་རུ་མ་ཡུལ་ཚོའི་རུ་ཆེན་གསུམ་པ་ནས་ཡིན་ལ། ཁོང་གི་ཕའི་མིང་ལ་བྷེ་ཚེ་དང་མའི་མིང་ལ་ཞི་ཆུང་ཟེར་བ་དང་། ཁོང་ཆུང་དུས་ནས་རྔ་བ་ཀིརྟི་དགོན་པར་གྲྭ་པ་བྱས་ཏེ་ཁོང་འཛིན་བཟུང་བྱེད་སྐབས་སྔོན་འགྲོའི་འཛིན་གྲྭ་དགུ་པ་ཡིན་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
