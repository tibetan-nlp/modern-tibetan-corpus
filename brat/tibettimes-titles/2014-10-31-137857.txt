ཁ་ཆེ་བ་སྟོང་ཕྲག་མང་པོས་ངོ་རྒོལ་བྱས།

ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༩ ཉིན་མཚོ་སྔོན་ཞིང་ཆེན་བ་ཡན་རྫོང་གི་ཁ་ཆེ་བ་སྟོང་ཕྲག་མང་པོས་རྫོང་སྲིད་གཞུང་ལ་ངོ་རྒོལ་ཁྲོམ་སྐོར་བྱས་འདུག རྒྱ་གཞུང་གི་མཁར་གྲོང་གསར་པ་འཛུགས་སྐྲུན་གྱི་ལས་གཞིའི་འོག འདི་ལོར་མཚོ་སྔོན་ཞིང་ཆེན་བ་ཡན་རྫོང་ཐོག་གི་ལས་ཁུངས་ཚང་མ་མཁར་གྲོང་གསར་པའི་ཁུལ་དུ་གནས་སྤོ་བྱས་རྐྱེན། རྫོང་མཁར་དུ་ལུས་པའི་མི་མང་གི་ལས་དོན་ལ་སྟབས་བདེ་མེད་པ་མ་ཟད། སྲིད་གཞུང་གི་དབུལ་སྐྱོར་དངུལ་གྱིས་མི་མང་གི་འཚོ་བར་ཕན་ཐོགས་མེད་པར་བརྟེན། ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༩ ཉིན་ས་གནས་ཁ་ཆེ་སྟོང་ཕྲག་མང་པོས་བ་ཡན་རྫོང་གནས་སྤོ་བར་རྒྱལ་སྲིད་སྤྱི་ཁབ་ཁང་གི་བཀའ་མངག་ཡོད་དམ། མཁར་གྲོང་གསར་པ་འཛུགས་སྐྲུན་གྱིས་མགོ་ཁྲིད་ཚོ་ཕྱུག་པོ་ཆགས་ཡོད་ཀྱང་། རྒྱལ་ཁབ་དང་མི་མང་ལ་གནོད་སྐྱོན་ཆེན་པོ་བཟོ་ཡི་འདུག ཅེས་པའི་འཕྲེད་དར་བཟུང་སྟེ་ངོ་རྒོལ་ཁྲོམ་སྐོར་བྱས་ཡོད་འདུག ཉིན་དེའི་ཕྱི་དྲོར་གསར་འགོད་པས་ཁ་པར་བརྒྱུད་བ་ཡན་རྫོང་སྲིད་གཞུང་གི་འབྲེལ་ཡོད་ལས་བྱེད་ཅིག་ལ་མི་མང་གིས་ཅིའི་ཕྱིར་ངོ་རྒོལ་བྱས་པ་ཡིན་དྲིས་པར། མི་མང་པོ་ཁྲོམ་གཞུང་དུ་འདུ་འཛོམས་ཀྱིས་འུར་ཟིང་བསླངས་བ་དེ་ནོར་འཁྲུལ་ཡིན་སྐོར་བརྗོད་ཡོད་པ་ཙམ་ལས། སྲིད་གཞུང་གིས་མི་མང་ལ་ལན་འདེབས་ཅི་ཞིག་བྱས་ཡོད་འགྲེལ་བརྗོད་བྱས་མེད་པ་རེད། ངོ་རྒོལ་ནང་ཞུགས་པའི་ཁ་ཆེ་བ་ཞིག་གིས་གསར་འགོད་པར། སྲིད་གཞུང་ནས་མི་མང་གིས་ཁྲོམ་སྐོར་བྱས་པ་དེ་ནོར་འཁྲུལ་རེད་ཟེར་གྱི་ཡོད་ཀྱང་ནོར་འཁྲུལ་གང་ཡང་མ་རེད། ད་ལྟ་རྫོང་སྲིད་གཞུང་གི་ལས་ཁུངས་ཚང་མ་རྫོང་མཁར་ནས་ཆུ་ཚོད་ ༡།༣༠ ཙམ་གྱི་སར་ཡོད་པའི་མཁར་གྲོང་གསར་པའི་ཁུལ་དུ་གནས་སྤོ་བྱས་ཡོད་སྟབས། མི་མང་གིས་ལས་ཁུངས་ཁག་དང་འབྲེལ་བའི་བྱ་བ་བསྒྲུབ་པར་སྟབས་བདེ་ཁྱོན་ནས་མི་འདུག ཅེས་བརྗོད་འདུག ཡང་ཁོང་གིས། མཁར་གྲོང་གསར་པའི་ནང་གནས་སྤོ་བྱེད་པའི་ལས་གཞི་དེ་མགོ་ཁྲིད་ཚོའི་ཆེད་དུ་ཡིན་པ་ལས་མི་མང་གི་ཆེད་དུ་མ་རེད། སྤྱིར་བཏང་བ་ཡན་རྫོང་རྒྱལ་ཁབ་རིམ་པའི་དབུལ་ཕོངས་རྫོང་གི་ཁུངས་སུ་བཅུག་སྟེ་རོགས་སྐྱོར་བྱེད་ཀྱི་ཡོད་ཀྱང་། དབུལ་སྐྱོར་གྱི་དངུལ་ཚང་མ་མཁར་གྲོང་གསར་པ་འཛུགས་སྐྲུན་བྱེད་པར་བཏང་སྟེ། ས་གནས་མི་མང་གི་འཚོ་བའི་དཀའ་ངལ་སེལ་བར་ཕན་ཐོགས་གང་ཡང་མི་འདུག བ་ཡན་རྫོང་ཁུལ་དུ་དཔལ་འབྱོར་གྱི་ཡོང་ཁུངས་ཧ་ཅང་ཞན་སྟབས། རྫོང་མཁར་གྱི་མི་ཚོ་ས་གནས་གཞན་དུ་ཞོར་ལས་ལ་སོང་ནས་འཚོ་བ་སྐྱེལ་བཞིན་ཡོད་པ་རེད། དེ་སྔོན་ས་གནས་མི་མང་གིས་རྫོང་སྲིད་གཞུང་ལ་ཐེངས་མང་པོར་དཀའ་སྡུག་ཞུས་ཀྱང་ཉན་འཇོག་བྱེད་ཀྱི་མི་འདུག ཅེས་བརྗོད་འདུག རྒྱ་ནག་རྒྱལ་སྲིད་སྤྱི་ཁྱབ་ཁང་དབུལ་སྐྱོར་ཡར་སྐུལ་སྣེ་ཁྲིད་ཚོགས་ཆུང་གིས་བསྒྲགས་པའི་གནས་ཚུལ་ལྟར་ན། དེང་སྐབས་རྒྱ་ནག་ནང་རྒྱལ་ཁབ་རིམ་པའི་དབུལ་ཕོངས་རྫོང་ ༥༩༢ ཡོད་པ་དང་། ལོ་རེར་དབུལ་ཕོངས་རྫོང་གཅིག་ལ་རྒྱ་སྒོར་ཁྲི་ ༥༠ ཙམ་གྱིས་རོགས་སྐྱོར་བྱེད་ཀྱི་ཡོད་འདུག འདས་ལོར་དབུལ་ཕོངས་རྫོང་ ༡༩ ནང་ཕྱི་ལོ་ ༢༠༡༠ ནས་ ༢༠༡༢ ལོའི་བར་གྱི་དབུལ་སྐྱོར་ལས་གཞིར་རྟོག་ཞིབ་བྱེད་སྐབས། ཁྱོན་དབུལ་ཕོངས་རྫོང་ ༡༩ ཡི་ནང་དབུལ་སྐྱོར་དངུལ་ལྐོག་ཟ་དང་བེད་སྤྱོད་ལོག་པར་བཏང་བའི་འབྲེལ་ཡོད་ལས་བྱེད་ ༡༢༣ ཐེར་འདོན་བྱས་ཡོད་པ་དང་། དེའི་ནང་ནས་མི་ ༢ ལ་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་པ་དང་། མི་ ༧༨ ཏང་ནང་ནས་ཕྱིར་འབུད། མི་ ༤༣ ལ་སློབ་གསོ་བཏང་ཡོད་པ་དང་། མུ་མཐུད་ཞིབ་བཤེར་དགོས་པའི་འབྲེལ་ཡོད་ལས་བྱེད་ ༢༤ ཡོད་སྐོར་དང་། དབུལ་སྐྱོར་དངུལ་ཁྱོན་དུང་ཕྱུར་ ༢.༡༤ ཙམ་ལྐོག་ཟ་རུལ་སུངས་དང་བེད་སྤྱོད་ལོག་པར་བཏང་ཡོད་ལ། ད་དུང་དངུལ་དུང་ཕྱུར་ ༡.༤༥ ལྷག་གི་ཉེས་དོན་ལ་བསྐྱར་དུ་ཞིབ་བཤེར་བྱེད་དགོས་སྐོར་བསྒྲགས་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
