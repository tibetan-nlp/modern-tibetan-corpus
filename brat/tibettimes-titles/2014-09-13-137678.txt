དབྱིན་ཇི་དང་རྒྱ་ནག་གི་ཚོང་འབྲེལ།

ཁ་སང་གཟའ་པ་སངས་ཉིན་དབྱིན་ཇི་དང་རྒྱ་ནག་གཉིས་དབྱིན་སྒོར་དུང་ཕྱུར་ ༢༤ ཚོང་འབྲེལ་གན་རྒྱ་ཞིག་བཞག་ཡོད་འདུག ཟླ་འདིའི་ཚེས་ ༡༢ ཉིན་དབྱིན་ཇིའི་རྒྱལ་ས་ལོན་ཌོན་དུ། དབྱིན་ཇིའི་དཔལ་འབྱོར་བློན་ཆེན་ཨོ་སུ་བྷོན་ (Osborne) ལགས་དང་། རྒྱ་ནག་དམར་ཤོག་དྲུང་ཆེ་གཞོན་པ་མཱ་ཁེ་གཉིས་ཀྱིས། རྒྱལ་ཁབ་གཉིས་ཀྱི་ནོར་སྲིད་དཔལ་འབྱོར་གླེང་མོལ་ཐེངས་དྲུག་པར་མགོ་འཛིན་གནང་ཡོད་པ་མ་ཟད། རྒྱལ་ཁབ་གཉིས་བར་ཚོང་འབྲེལ་ཐོག་དབྱིན་སྒོར་དུང་ཕྱུར་ ༢༤ ཡིས་གན་རྒྱ་བཞག་ཡོད་པ་དང་། མ་འོངས་པར་དབྱིན་ཇིའི་གཞུང་ནས་གུ་ཡངས་ཀྱི་སྲིད་ཇུས་འོག་རྒྱ་ནག་གི་ཡུལ་སྐོར་བར་ཐོངས་མཆན་སྤྲོད་རྒྱུ་དང་། དབྱིན་གཞུང་ནས་དབྱིན་ཇིའི་རྩོམ་པ་པོ་གྲགས་ཅན་ཤ་ཁུ་སུ་ཕེ་རེ་ (Shakespeare)ཡི་གསུང་རྩོམ་ཕྱོགས་བསྒྲིགས་རྒྱ་ཡིག་ནང་བསྒྱུར་རྒྱུའི་ལས་འཆར་ལ་མ་དངུལ་གྱི་རོགས་སྐྱོར་བཅས་གནང་རྒྱུའི་འཆར་གཞི་ཡོད་སྐོར་སོགས་བསྒྲགས་ཡོད་འདུག དབྱིན་ཇིའི་གཞུང་དང་རྒྱ་ནག་གི་ཚོང་འབྲེལ་གན་རྒྱའི་ནང་གཙོ་བོ། དབྱིན་སྒོར་དུང་ཕྱུར་དྲུག་ཁྲི་ཆིག་སྟོང་དྲུག་རྒྱའི་ ༦༡༦༠༠ ཡི་མ་ལེ་ཤི་ཡའི་སྣུམ་གྱི་ཐོན་ཁུངས་ཐོག་གི་མཉམ་འབྲེལ་ལས་དོན་དང་། དབྱིན་སྒོར་དུང་ཕྱུར་ ༢ འཛུགས་སྐྲུན་ཡར་རྒྱས་ཀྱི་ལས་གཞི། རྒྱ་ནག་ནང་རྒན་གསོ་ཁང་དང་ཆེད་ལས་སྦྱོང་བརྡར་གྱི་སློབ་གྲྭ་གསར་རྒྱག་བཅས་ཀྱི་སྐོར་ཚུད་ཡོད་པར་མ་ཟད། རྒྱ་ནག་གཞུང་འབྲེལ་གྱི་ཆོག་མཆན་ཡོད་པའི་ཡུལ་སྐོར་ལྗོངས་རྒྱུ་བ་ ༢༥༠༠༠ ཡོད་པའི་ཚོགས་པ་དབྱིན་ཇིའི་ནང་ལྗོངས་རྒྱུར་ཡོང་སྐབས། ཐོངས་མཆན་དང་རོགས་རམ་གནང་རྒྱུ་ཡིན་སྐོར་བརྗོད་ཡོད་འདུག་ལ། ལས་འཆར་དེ་ནི་དབྱིན་ཇིའི་ལྗོངས་རྒྱུ་ལས་ཁུངས་དང་ནང་སྲིད་ལས་ཁུངས་ཟུང་འབྲེལ་ཐོག དབྱིན་རྒྱ་གཉིས་བར་གྱི་དབྱིན་སྒོར་ཁྲི་ ༡༦༠ ཡུལ་སྐོར་ལྗོངས་རྒྱུའི་ཚོང་འབྲེལ་ལས་འཆར་ཞིག་ཡིན་འདུག ཐེངས་འདིའི་དབྱིན་རྒྱ་གཉིས་ཀྱི་ཚོང་འབྲེལ་དང་འབྲེལ་བའི་ལས་གཞིའི་ཐད། ཁ་སང་དབྱིན་ཇིའི་ནང་སྲིད་བློན་ཆེན་གྱིས་ཌར་རེ་ (Terry) ཡིས་གསལ་བསྒྲགས་ཤིག་སྤེལ་ཏེ། དབྱིན་ཇིའི་ཕྱོགས་ནས་དེང་ཕྱིན་རྒྱ་ནག་གི་ཡུལ་སྐོར་སྤྲོ་འཆམ་པའི་ཐོངས་མཆན་ཐོག་སྔར་ལས་གུ་ཡངས་བཏང་སྟེ། དབྱིན་ཇིའི་གནམ་ཐང་དུ་ཐོན་ཏེ་ཐོངས་མཆན་སྤྲོད་དགོས་པའི་རྒྱ་རིགས་ཚོར། ཐད་ཀར་ཡུ་རོབ་མནའ་འབྲེལ་གྱི་ཆོག་མཆན་ཡོད་པའི་ཐོངས་མཆན་ཡིག་ཆ་སྤྲད་དེ། གཞན་པའི་ཡིག་ཆའི་རིགས་གང་ཡང་མི་དགོས་པའི་མཐུན་རྐྱེན་བསྐྲུན་རྒྱུ་ཡིན་ཚུལ་བརྗོད་ཡོད་པ་མ་ཟད། མ་འོངས་པར་དབྱིན་གཞུང་གིས་དབྱིན་སྒོར་ཁྲི་ ༡༥༠ རྩོམ་པ་གྲགས་ཅན་ཤ་ཁུ་སུ་ཕེ་རེ་བཟོ་གྲྭར་རོགས་དངུལ་བཏང་སྟེ། རྩོམ་པ་པོ་ཁོང་གི་གསུང་རྩོམ་ཕྱོགས་བསྒྲིགས་རྒྱ་ཡིག་ནང་བསྒྱུར་རྒྱུ་དང་། ༢༠༡༦ ལོར་རྩོམ་པ་པོ་ཁོང་སྐུ་མྱ་ངན་ལས་འདས་ནས་ལོ་ ༤༠༠ འཁོར་བའི་དུས་དྲན་སྐབས། གཞུང་ངོས་ནས་འགྲོ་གྲོན་བཏང་སྟེ་རྒྱ་ནག་ནང་དམ་པ་ཁོང་གི་འཁྲབ་གཞུང་སོགས་གཟིགས་འབུལ་ཞུ་འཆར་ཡོད་སྐོར་བརྗོད་ཡོད་འདུག གསར་འགོད་པ། དོན་གོ
