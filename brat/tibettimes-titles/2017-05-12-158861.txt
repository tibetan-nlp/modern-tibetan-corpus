ངས་བོད་ཀྱི་དུས་བབ་ཚཊ་པར་བཀླཊ་ནས་ལོ་ངོ་ ༢༠
ཕྱིན་སོང་།

ཕྱི་ལོ་ ༡༩༩༦ ལོར་བོད་ཀྱི་དུས་བབ་གསར་ཁང་དབུ་འཛུགས་གནང་སྟེ་ད་ལྟ་བགྲང་བྱ་ ༢༠ ཧྲིལ་པོ་འཁོར་བ་དང་བསྟུན། དུས་བབ་ཚགས་པར་ཐོག་མ་བཏོན་པ་ནས་བཟུང་མངགས་ཉོ་གནང་མཁན་ཀློག་པ་པོ་རྙིང་ཤོས་རྒན་དཀོན་མཆོག་ལགས་སུ། ལོ་ངོ་ ༢༠ རིང་དུས་བབ་ཚགས་པར་བཀླགས་པའི་མཐོང་བ་དང་མ་འོངས་རེ་འདོད་སྐོར་བཅར་འདྲི་ཞུས། རྒན་དཀོན་མཆོག་ལགས་བོད་མངའ་རིས་ཕྱོགས་སུ་སྐྱེས་ཤིང་། ཕྱི་ལོ་ ༡༩༥༩ ལོར་རྒྱ་ནག་གཞུང་གིས་བོད་ས་ཡོངས་སུ་བཙན་བཟུང་བྱེད་སྐབས། དགུང་ལོ་ ༡༠ ཙམ་ལ་སོན་པའི་ཁོང་ཕ་མ་གཉིས་ཀྱིས་འཁྲིད་དེ་རྒྱ་གར་ནང་བཙན་བྱོལ་དུ་ཡོང་སྟེ། སིམ་ལ་གཏན་སློབ་དང་མ་སུ་རི་ནས་གཞི་རིམ་སློབ་གཉེར་གནང་། དེའི་རྗེས་རྒྱ་གར་གྱི་མཐོ་སློབ་(Hans Raj College)ནས་ལོ་རྒྱུས་རིག་པའི་ཐོག་རབ་འབྱམས་པ་ཐོན་ཡོད་པ་རེད།  འདས་པའི་ལོ་ངོ་ ༣༦ རིང་དགེ་རྒན་སོགས་ཀྱི་ཕྱག་ལས་གནང་སྟེ་ད་ལྟ་རྒན་ཡོལ་དུ་ཕེབས་འདུག་ཀྱང་། སིམ་ལ་ཇོ་ནང་དགོན་དུ་དབྱིན་ཡིག་དང་ཚན་རིག་གི་དགེ་རྒན་གནང་མུས་རེད་འདུག དྲི་བ། རྒན་ལགས་ཀྱིས་དུས་བབ་ཚགས་པར་ཐོག་མར་གནས་སྟངས་ག་འདྲ་ཞིག་གི་འོག་ནས་མངག་ཉོ་གནང་ཡོད་པ་མདོར་བསྡུས་ཤིག་ངོ་སྤྲོད་གནང་རོགས། ལན། ཕྱི་ལོ་ ༡༩༩༦ ལོར་ངས་སིམ་ལ་གཏན་སློབ་ཏུ་དགེ་རྒན་བྱས་ནས་བསྡད་ཡོད་པ་དང་། སྐབས་དེར་དབྱིན་ཡིག་ཚགས་པར་འབེལ་པོ་ཡོད་ཀྱང་བོད་ཡིག་གི་ཚགས་པར་དཀོན་པོ་ཞེ་དྲགས་ཡིན་ལ། གནས་ཚུལ་ག་རེ་ཡིན་ནའང་ཧ་གོ་རྒྱུ་ཁག་པོ་ཡོད་པ་རེད། དེ་རིང་སང་ཉིན་དྲ་རྒྱ་སོགས་ཀྱི་ཐོག་ནས་གནས་ཚུལ་དེ་ཚོ་གོ་སླ་བ་དང་ལྟ་བདེ་བོ་ཡོད་པ་རེད། ལོ་ངོ་ ༢༠ ཡར་སྔོན་དུ་ང་རང་ཚོའི་སྤྱི་ཚོགས་ནང་དུས་བབ་ཚགས་པར་གནས་ཚད་མཐོ་ཤོས་དེ་ཆགས་ཡོད་ལ། ང་ཚོ་དགེ་རྒན་ཡིན་པའི་ཆ་ནས་སློབ་ཕྲུག་ཚོར་སློབ་འཁྲིད་བྱེད་སྐབས་དུས་ཐོག་གི་གནས་ཚུལ་རྣམས་མར་བཤད་དགོས་ཀྱི་ཡོད་པ་དང་། དུས་བབ་ཀྱི་གནས་ཚུལ་དེ་ཚོ་བློ་ལ་འགྲོ་བོ་ཡོད་པར་བརྟེན། ཕྱི་ལོ་ ༡༩༩༦ ཟླ་ ༩ པའི་ནང་ནས་བཟུང་དུས་བབ་ཚགས་པར་མངག་ཉོ་བྱས་པ་ཡིན། དྲི་བ། དེང་སང་དྲ་ལམ་བརྒྱུད་དུས་ཐོག་ཏུ་འཛམ་གླིང་གི་གནས་ཚུལ་ཁག་ལ་ལྟ་ཉན་བྱེད་ཐུབ་ཀྱི་ཡོད་སྟབས། ཚགས་པར་ཀློག་མཁན་ཇེ་ཉུང་དུ་འགྲོ་བཞིན་ཡོད་པ་རེད། ཡིན་ནའང་། ཁྱེད་ཀྱིས་དྲ་ལམ་ནས་གནས་ཚུལ་མང་པོར་གཟིགས་ཀྱི་ཡོད་པའི་ཞོར་ལ་ཚགས་པར་ཡང་མུ་མཐུད་མངག་ཉོ་གནང་དགོས་པའི་རྒྱུ་མཚན་ག་རེ་ཡིན་ནམ། ལན། བོད་ཡིག་ཚགས་པར་གྱིས་ང་ཚོའི་དུས་སྐབས་སུ་ཕན་ཐོགས་པོ་ཞེ་དྲགས་བྱུང་སོང་ལ། ང་ཚོའི་དུས་སྐབས་ཙམ་མ་ཡིན་པར་ཁྱིམ་དུ་ཡོད་པའི་བྱིས་པ་ཚོས་ཀྱང་དོ་སྣང་འདྲ་པོ་སྤྲད་དེ་རྒྱུན་དུ་ཚགས་པར་ལ་ལྟ་བའི་གོམས་གཤིས་འཇགས་འདུག ཚགས་པར་ལ་མཚོན་ན་གཅིག་ནས་བརྡ་རྟགས་སོགས་དགོས་ཀྱི་མེད་དུས་ས་ཆ་ག་བ་ནས་ཀློག་པ་ཡིན་ནའང་ཀློག་ཆོག་ཆོག་དང་། གནས་ཚུལ་མང་པོ་ཧ་གོ་ཐུབ་ཀྱི་ཡོད་པ་མ་ཟད། དངོས་སུ་བོད་ཡིག་གི་དག་ཆ་རྣམས་མཐོང་རྒྱུ་ཡོད་དུས་བྱིས་པ་ཚོར་ཕན་ཐོགས་ཆེན་པོ་འདུག གཉིས་ནས་ང་རང་ཚོ་སོ་སོའི་སྐད་ཡིག་ཐོག་ཡོད་པའི་ཚགས་པར་དེ་ཚོ་གལ་ཆེན་པོ་ཞེ་དྲགས་ཆགས་ཡོད་ཀྱང་ང་ཚོར་ད་ལྟའང་བོད་ཡིག་གི་ཚགས་པར་ཉུང་ཉུང་ཞེ་དྲགས་རེད། ང་རང་ལ་མཚོན་ན་དུས་བབ་ཀྱི་གནས་ཚུལ་དེ་ཚོ་ཁ་གསལ་པོ་དང་དུས་ཐོག་ལ་ཏག་ཏག་གོ་བདེ་བོ་བྱས་ནས་འདོན་གནང་གི་འདུག་སྟབས། སོ་སོར་དགའ་མོས་ཡོད་པ་ལྟར་མུ་མཐུད་མངག་ཉོ་བྱེད་ཀྱི་ཡོད་ལ། ཉི་མ་རེ་རེའི་གནས་ཚུལ་རྣམས་ཤེས་རྟོགས་ཐུབ་པ་མ་ཟད། བོད་ཡིག་བོད་སྐད་ཡར་རྒྱས་སུ་གཏོང་བར་ཕན་ཐོགས་ཞེ་དྲགས་འདུག དྲི་བ། འདས་པའི་ལོ་ངོ་ ༢༠ རིང་ཁྱེད་ཀྱིས་བཀླགས་ཚར་བའི་དུས་བབ་ཀྱི་ཚགས་པར་རྣམས་ཉར་ཚགས་ག་འདྲ་གནང་ཡོད། ལན། ངའི་ནང་དུ་དཔེ་མཛོད་ཁང་ཆུང་ཆུང་ཞིག་ཡོད་པ་དེའི་ནང་དུ་ད་ལྟའི་བར་གྱི་ཚགས་པར་ཚང་མ་ཉར་བསྡད་ཡོད། གལ་སྲིད་རྣམ་པ་ཚོས་ལོ་དེ་འདྲ་ཞིག་གི་ཚགས་པར་དགོས་ཀྱི་འདུག་གསུང་ན་ཕུལ་ཆོག་ཆོག་ཡོད། སྐབས་རེར་སྔར་གྱི་ལོ་རྒྱུས་སོགས་དང་འབྲེལ་བའི་གནས་ཚུལ་དེ་འདྲ་བསླེབས་སྐབས་སྔོག་ཆོག་ཆོག་ཡོད་པས་ངས་དེ་འདྲ་ལ་བེད་སྤྱོད་གཏོང་བཞིན་ཡོད། ཞེ་དྲགས་ཞེ་དྲགས་བྱ་རྒྱུ་ཡོད་པ་མ་རེད་དེ་སོ་སོའི་དུས་སྐབས་སུ་བེད་སྤྱོད་བཏང་ཚར་བ་རེད་ལ། བུ་ཕྲུག་ཚོས་ཀྱང་བེད་སྤྱོད་བཏང་བ་ཡིན་ན་ཚང་མ་ཉར་བསྡད་ཡོད། ང་ལ་བུ་མོ་གཅིག་དང་བུ་གཅིག་ཡོད་ལ། བུ་མོས་བུ་རུ་ཝ་ལ་སཾ་བྷོ་ཊ་ཉིན་སློབ་ཏུ་དགེ་རྒན་གྱི་ཕྱག་ལས་གནང་བཞིན་ཡོད་པ་དང་། མོས་མཚམས་མཚམས་ལ་ཚགས་པར་ཀློག་གི་འདུག བུ་ཡིས་གློག་ཀླད་སྐོར་ལ་ཆེད་ལས་སློབ་སྦྱོང་གནང་བཞིན་ཡོད་པ་རེད། དྲི་བ། བོད་མི་མང་པོ་ཞིག་གིས་བོད་ཡིག་གི་ཚགས་པར་དང་དེབ་དེ་ཚོ་གཅིག་ནས་བོད་ཡིག་ཡིན་པ་དང་། གཉིས་ནས་༧གོང་ས་མཆོག་གི་དབུས་པའི་བོད་པའི་བླ་སྤྲུལ་རྣམ་པའི་སྐུ་པར་མང་པོ་བཀོད་འདུག་པར་བརྟེན། ལྟ་ཀློག་བྱས་རྗེས་འཇོག་ས་མ་རག་པར་བཙོག་ཤོར་འགྲོ་ཡི་འདུག སྡིག་པ་གསོག་བྱེད་རེད་ཉོ་ཡི་མིན་ཟེར་བ་དང་། དེ་བཞིན་ལྷམ་དང་གྱོན་གོས་སོགས་ཀྱི་ཐོག་བོད་ཡིག་བྲིས་ན་ཉེས་པ་ཆེན་པོ་རེད་ཅེས་རང་གིས་འཛེམས་ཤིང་གཞན་ལ་འགོག་གིན་ཡོད་པ་རེད། ཁྱེད་རང་དེའི་ཐད་དགོངས་ཚུལ་ག་རེ་ཡོད། ལན། ང་ཚོས་བསམ་བློ་དེ་འདྲ་བཏང་སྟེ་བོད་ཡིག་ལ་དོ་སྣང་མ་སྤྲད་པ་ཡིན་ན་ནོར་འཁྲུལ་རེད་བསམ་གྱི་འདུག གང་ཡིན་ཟེར་ན། ཅ་ལག་ག་འདྲ་ཞིག་ཡིན་ནའང་སོ་སོར་མཁོ་བ་ཞིག་ཡིན་ན་དེ་ལ་དཔག་པའི་འཇོག་ས་ཞིག་དང་གནས་ས་ཞིག་བཟོ་དགོས་ཀྱི་ཡོད་པ་རེད་མ་གཏོགས་འཇོག་ས་མི་འདུག གཅིག་བྱེད་འགྲོ་ཡི་འདུག གཉིས་བྱེད་འགྲོ་ཡི་འདུག་ལབ་ན་འགྲིག་པ་ཁྱོན་ནས་ཡོད་པ་མ་རེད། སྤྱིར་བཏང་བཀའ་བསྟན་འགྱུར་དེ་ཚོ་ཚང་མ་བོད་ཡིག་ཐོག་ཡོད་པས་ཡིག་འབྲུ་གཅིག་ཡིན་ནའང་ཆོས་དཀོན་མཆོག་གི་གྲས་སུ་བསླེབས་ཀྱི་ཡོད་སྟབས། བོད་ཡིག་ལ་འགོམ་པ་སོགས་བྱས་ན་སྡིག་པ་བསག་གི་རེད་ཅེས་པའི་བཀའ་མོལ་ཞིག་ཡོད་པ་རེད།  འོན་ཀྱང་། དེ་ལ་འབྲེལ་བ་བཏགས་ཏེ་བོད་ཡིག་ལ་སྣང་ཆུང་དང་བོད་ཡིག་ཁྱབ་སྤེལ་བྱེད་པར་དོ་སྣང་མ་སྤྲད་པ་དེ་ཚོ་ཡག་པོ་ཨེ་ཡོད་བསམ་གྱི་འདུག ༧སྐྱབས་མགོན་རིན་པོ་ཆེ་མཆོག་ནས་དུས་རྒྱུན་བོད་ཡིག་བོད་སྐད་ལ་དོ་སྣང་དགོས་པའི་བཀའ་སློབ་ཡང་ཡང་ཕེབས་ཀྱི་ཡོད་པ་ལྟར། ང་ཚོས་ད་དུང་བོད་ཡིག་བེད་སྤྱོད་དང་ཁྱབ་སྤེལ་བྱེད་སྟངས་མང་དུ་གཏོང་དགོས་པ་ཞིག་མཐོང་གི་འདུག ང་རང་ལ་མཚོན་ན་དཀའ་ངལ་དེ་རིགས་གང་ཡང་འཕྲད་ཀྱི་མི་འདུག དྲི་བ། འདས་པའི་ལོ་ངོ་ ༢༠ ལ་ཕྱིར་མིག་ཅིག་བལྟས་ཏེ་འདས་པ་དང་ད་ལྟ་གཉིས་བསྡུར་བ་ཡིན་ན། དུས་བབ་ཀྱི་གསར་འགྱུར་ལ་བརྗོད་བྱ་དོན་གྱི་གནད་འཛིན་སྟངས་དང་ཁ་ཕྱོགས། དེ་བཞིན་རྗོད་བྱེད་ཚིག་གི་འགྲོ་སྟངས་དང་ཐ་སྙད་བེད་སྤྱོད་བྱེད་སྟངས་སོགས་ཀྱི་ཐད་ལ་ཁྱད་པར་ཅི་འདུག ལན། ང་རང་བོད་ཡིག་ཐོག་ལ་ཧང་སང་བའི་ཡག་པོ་དེ་འདྲ་མེད་དུས་མཐའ་གཅིག་ཏུ་འདི་འདྲ་འདུག་ཅེས་ཞུ་རྒྱུ་མེད་ཀྱང་། སྔོན་མ་དང་བསྡུར་ན་ད་ལྟ་ཕྱོགས་གང་ཐད་ནས་ཇེ་དྲག་ཏུ་ཞེ་དྲགས་ཕྱིན་འདུག གང་ཡིན་ཟེར་ན། དེང་སང་བོད་ཀྱི་དུས་བབ་ཀྱིས་འཛམ་བུ་གླིང་དང་རྒྱ་ནག་སོགས་ཀྱི་གནས་ཚུལ་ཤུགས་ཆེན་པོ་འདོན་བཞིན་ཡོད་པ་དང་། གསར་འགྱུར་དང་རྩོམ་རིག་ཟེར་བ་ལྟ་བུའི་སྡེ་ཚན་དུ་དབྱེ་བ་ཕྱེས་ཏེ་འདོན་གྱི་ཡོད་པ་དེ་ཡག་པོ་འདུག་ལ། ལྷག་པར་དུ་ང་རང་ཚོའི་སྤྱི་ཚོགས་ནང་གནས་ཚུལ་དགོས་གལ་ཆེན་པོ་ཅན་འོས་བསྡུ་ལྟ་བུར་གནས་ཚུལ་དཔེ་ཡག་པོ་འདོན་གྱི་འདུག ཚིག་འགྲོས་ཀྱང་དཔེ་ཀློག་བདེ་བོ་འདུག  སྔོན་མ་བོད་ཡིག་གི་ཚགས་པར་མང་པོ་ཞིག་ཡོད་ཀྱང་། ད་ལྟ་དུས་བབ་དང་བང་ཆེན་གཉིས་མ་གཏོགས་ཕལ་ཆེར་མེད་པ་འདྲ། སྤྱིར་བཏང་དེང་སང་དྲ་ལམ་ཞེ་དྲགས་དར་ཁྱབ་ཆེན་པོ་ཕྱིན་ཡོད་པས་ཚགས་པར་རྗེས་ལུས་ཐེབས་ཏེ་དཀའ་ངལ་འཕྲད་ཀྱི་ཡོད་ཀྱང་། སྤྱི་ཚོགས་ལ་ག་ཚོད་བར་དུ་ཕན་གྱི་ཡོད་ན་དེ་ཚོད་དུ་ཞབས་ཞུ་ཞུ་ཐུབ་ན་དཔེ་ཡག་པོ་འདུག དྲི་བ། དུས་བབ་ཀྱི་ཐ་སྙད་རེ་ཟུང་བེད་སྤྱོད་བྱེད་སྟངས་ཐད་སྤྱི་ཚོགས་ནང་བསམ་ཚུལ་ཡོང་གི་ཡོད་པ་རེད། ཁྱེད་རང་དུས་བབ་ཀྱི་ཀློག་པ་པོ་རྒན་ཤོས་ཤིག་ཡིན་པའི་ཆ་ནས་དེའི་ཐད་དགོངས་ཚུལ་ག་རེ་ཡོང་གི་འདུག ལན། ཚགས་པར་གྱིས་གནས་ཚུལ་ཞིག་མི་ལ་གོ་བརྡ་སྤྲོད་ཀྱི་ཡོད་པས་གོ་བརྡ་ཡག་པོ་འཕྲོད་དགོས་པ་དེ་གལ་ཆེ་ཤོས་རེད།  ཚིག་གི་ཆ་ནས་སྐད་རིགས་གཞན་མ་འདྲིས་པར་བོད་སྐད་གཙང་མ་ཞིག་དགོས་ཀྱི་ཡོད་པ་རེད། གལ་སྲིད་སྐད་གཞན་པ་ཕྲ་ལ་ཕྲན་བུ་འདྲིས་ཀྱི་ཡོད་ནའང་གཙོ་བོ་ཚིག་གི་ནང་དོན་གཅིག་པ་ཆགས་དགོས་པ་དེ་གལ་ཆེ་ཤོས་རེད་བསམ་གྱི་འདུག དྲི་བ། ལོ་ངོ་ ༢༠ རིང་དུས་བབ་ཚགས་པར་བཀླགས་གནང་བའི་བརྒྱུད་རིམ་ཁྲོད་བག་ཆགས་གཟབ་ཤོས་ཀྱི་གནས་ཚུལ་གང་བྱུང་སོང་། ལན། ༧སྐྱབས་མགོན་རིན་པོ་ཆེ་ཕར་ཚུར་ཕེབས་པ་དང་ཆོས་འབྲེལ་བཀའ་སློབ་གནང་བའི་མཛད་རིམ་དེ་ཚོ་ངས་གལ་ཆེ་ཤོས་སུ་རྩི་ཡི་ཡོད་ལ། མཛད་རིམ་སྐོར་གྱི་གནས་ཚུལ་དེ་ཚོས་ཕན་ཞེ་དྲགས་ཐོགས་ཀྱི་འདུག དེ་བཞིན་འོས་བསྡུའི་སྐབས་སུ་བགྲོ་གླེང་བསྡུས་པ་དང་འོས་མིར་བཅར་འདྲི་ངོ་སྤྲོད་གནང་བའི་གནས་ཚུལ་དེ་ཚོ་ཕན་ཐོགས་ཆེན་པོ་འདུག དྲི་བ། ཡར་རྒྱས་ལ་མཐའ་མེད་ཟེར་བ་ལྟར་འབྱུང་འགྱུར་དུས་བབ་ཀྱི་ལས་དོན་ཐད་ཀློག་པ་པོ་ཞིག་གི་ངོས་ནས་རེ་བ་ག་རེ་འདོན་རྒྱུ་ཡོད། ལན། ཚགས་པར་ཐོག་གི་སྐུ་པར་དེ་ཚོ་ལོ་ངོ་ ༢༠ རྗེས་ལའང་ཚོན་མེད་(Black white)མུ་མཐུད་འདོན་གྱི་འདུག་པས་མ་འོངས་པར་ཚོན་ཕྲ་ཅན་འདོན་ཐབས་དང་། ཤོག་གྲངས་ཅུང་མང་དུ་བཏང་སྟེ་ཡིག་གཟུགས་ཏོག་ཙམ་ཆེ་རུ་གཏོང་ཐུབ་ན་ཡག་པོ་ཡོད་ས་རེད། གང་ཡིན་ཟེར་ན། བགྲེས་པོ་མང་པོ་ཞིག་གིས་དུས་བབ་ཀྱི་ཚགས་པར་ཀློག་གིན་ཡོད་སྟབས། ད་ལྟའི་ཡིག་གཟུགས་ཆུང་དྲག་ནས་ལྟ་ཀློག་བྱེད་པར་དཀའ་ལས་ཁག་པོ་འདུག་ཟེར་མཁན་མང་པོ་འདུག ངས་ནམ་རྒྱུན་དུས་བབ་ཚགས་པར་བཀླགས་རྗེས་ང་རང་གི་གྲོགས་པོ་དེ་ཚོར་ཀློགས་ཀློགས་ལབ་ནས་གཡར་གྱི་ཡོད།  དེ་ནས་གནས་ཚུལ་ག་རེ་སྤེལ་མིན་དེ་སྐབས་ཐོག་གནས་ཚུལ་ག་རེ་བྱུང་གི་ཡོད་མེད་ལ་བལྟ་རྒྱུ་ཡིན་པས་ངས་གང་ཡང་ཞུ་ཐུབ་ཀྱི་མི་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
