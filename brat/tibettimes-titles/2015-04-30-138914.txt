བོད་མིའི་ཁྲོམ་སར་མེ་སྐྱོན་བྱུང་།

ཉེ་ལམ་ཀེ་ར་ལཱ་མངའ་སྡེའི་ནང་རྟེན་གཞི་བྱས་པའི་བོད་མིའི་ཁྲོམ་སར་མེ་སྐྱོན་བྱུང་སྟེ་ཚོང་ཁང་གསུམ་གྱི་ཆ་དངོས་ཡོངས་རྫོགས་ཚིག་ཡོད་འདུག ཟླ་འདིའི་ཚེས་ ༡༧ ཉིན་རྒྱ་གར་ལྷོ་ཕྱོགས་ཀེ་ར་ལཱ་མངའ་སྡེའི་ཝར་ཁཱ་ལཱ་བྲག་རོང་འཁྲིས་ཀྱི་མཚོ་འགྲམ་ (varkala clliff beach) ཁུལ་དུ་རྟེན་གཞི་བྱས་པའི་བོད་མིའི་ཁྲོམ་སར་གློག་གི་རྒྱུ་ལམ་ཆད་དེ་མེ་སྐྱོན་བྱུང་སྟེ། ཚོང་ཁང་བདུན་ཡོད་ཁོངས་ནས་གསུམ་གྱི་ཚོང་ཟོག་ཡོངས་རྫོགས་ཚིག་ནས་ཁྱིམ་ཚང་གཅིག་ལ་ཉུང་མཐར་ཧིན་སྒོར་འབུམ་ ༤༠ ལྷག་གི་གྱོང་གུད་ཕོག་ཡོད་འདུག དེ་རིང་གསར་འགོད་པས་ཁ་པར་བརྒྱུད་གོད་ཆག་ཕོག་མཁན་དང་ཁྲོམ་སའི་སྤྱི་མི་ལྷག་པ་མཚམས་ཆོས་ལགས་སུ་བཅར་འདྲི་ཞུས་པར་ཁོས་མོས་འདི་ལྟར་བརྗོད་བྱུང་། དེ་སྔོན་ང་ཚོས་ཀེ་ར་ལཱའི་ཀོ་ཧྥི་ཁང་ (Kerala Coffee house) ཞེས་པའི་ནང་ཁྲོམ་སྟོན་གྱིན་ཡོད་ཀྱང་། ལོ་རེ་བཞིན་ཁྲོམ་སའི་རིན་གོང་འཕར་གྱི་ཡོད་སྟབས་ཁྲོམ་ས་བརྗེ་པོ་བརྒྱབ་ཏེ། ༢༠༠༥ ལོ་ནས་བཟུང་ཝར་ཁཱ་ལཱ་བྲག་རོང་འཁྲིས་ཀྱི་མཚོ་འགྲམ་ (varkala clliff beach) ཞེས་པར་ཁྲོམ་ས་བཟུང་ནས་བསྡད་པ་རེད། འདིའི་ནང་སྦེལ་ཀོབ་དང་ལ་དྭགས། བལ་ཡུལ་དང་ལྡེ་ར་ལྡུན་སོགས་ནས་ཡོང་བའི་ཚོང་ཁང་ ༧ ཡོད་ཀྱང་ཕྱི་སུ་ ༢ ཕྱི་རྒྱལ་དུ་ཕྱིན་ཤུལ་དུ་ ༥ བསྡད་ཡོད་པ་རེད། ཅེས་བརྗོད་བྱུང་། ལྷག་པ་མཚམས་ཆོས་ལགས་སུ་མེ་སྐྱོན་ཁྲོད་ཚོང་ཁང་ཚིག་སྐྱོན་གནས་སྟངས་དྲིས་པར་ཁོ་མོས། ཚོང་ཁང་གཉིས་ཀྱི་ཅ་ལག་མེ་སྐྱོན་མ་བྱུང་སྔོན་ལ་ས་ཆ་གཞན་ལ་བསྡུ་གསོག་བརྒྱབ་ནས་ནང་ལ་འཐོན་རྩིས་བྱེད་བཞིན་ཡོད་སྟབས། ཐབ་ཚང་གི་ཅ་དངོས་མ་གཏོགས་མེས་ཚིག་ཡོད་པ་མ་རེད། ནང་མི་གཞན་གསུམ་གྱི་ཅ་དངོས་ཡོངས་རྫོགས་རྩ་མེད་ཆགས་ཚར་བ་རེད། ང་རང་ཉིན་དེར་ནང་ལ་སྤུན་མཆེད་སླེབ་ཡོད་སྟབས་སྔ་དྲོའི་ཆུ་ཚོད་ ༨ ཡོལ་ཙམ་ལ་ཚོང་ཁང་སྒོ་བརྒྱབ་ནས་ཁོ་ཚོ་དང་མཉམ་དུ་ཁྲོམ་ལ་ཕྱིན་པ་ཡིན། ང་ཚོ་ཁྲོམ་ནས་ཆུ་ཚོད་ ༡༠།༣༠ ཙམ་ལ་ཕྱིར་ནང་ལ་ལོག་ཡོང་སྐབས། ཚོང་ཁང་ལ་མེ་ཤོར་ནས་ཉེ་འཁོར་གྱི་མི་ཚོས་མེ་གསོད་ཐབས་བྱེད་བཞིན་འདུག་ཀྱང་། ཅ་དངོས་ཡོངས་རྫོགས་ཚིག་ནས་ཐལ་བ་ཆགས་འདུག ད་རེས་ང་རང་གི་ཚོང་ཁང་གཅིག་པུར་ཆ་བཞག་ན་ཧིན་སྒོར་འབུམ་ ༤༠ ནས་ ༥༠ ཙམ་ཞིག་གི་གྱོང་བརྒྱབ་སོང་། ཞེས་བརྗོད་བྱུང་། གསར་འགོད་པས་མེ་སྐྱོན་བྱུང་རྗེས་མངའ་སྡེའི་གཞུང་དང་བོད་མིའི་སྒྲིག་འཛུགས་སོགས་ནས་རོགས་རམ་གནང་ཡོད་མེད་དྲིས་པར་ཁོ་མོས། མངའ་སྡེའི་གཞུང་གིས་མེ་སྐྱོན་བྱུང་བའི་བོད་མིའི་ཚོང་ཁང་གསུམ་ལ་ཁྱོན་བསྡོམས་ཧིན་སྒོར་ཁྲི་གཅིག་གནང་བྱུང་བ་ལས་གཞན་རོགས་རམ་གང་ཡང་གནང་མ་སོང་། ད་ལྟ་ལྷོ་སྤྱི་མཆོག་ས་གནས་སུ་ཕེབས་ནས་གོད་ཆག་བྱུང་བ་ཚོ་དང་ཚོགས་འདུ་འཚོགས་བཞིན་ཡོད་སྟབས། བོད་མིའི་སྒྲིག་འཛུགས་ངོས་ནས་རོགས་རམ་ག་ཚོད་གནང་གི་ཡོད་མེད་ཞུ་ཐུབ་ཀྱི་མིན་འདུག མདང་དགོང་ལྷོ་སྤྱི་མཆོག་གིས་གང་ཐུབ་རོགས་རམ་གནང་གི་ཡིན་གསུངས་སོང་། ཞེས་བརྗོད་བྱུང་། དེ་རིང་ཉིན་རྒྱབ་གསར་འགོད་པས་ནང་སྲིད་དྲུང་ཆེ་འཁོར་ལྷ་ཚང་བསོད་ནམས་སྟོབས་རྒྱལ་ལགས་སུ་བཅར་འདྲི་ཞུས་པར་ཁོང་གིས་འདི་ལྟར་གསུངས་སོང་། རྩ་བའི་ཀེ་ར་ལཱ་ཁུལ་ལ་ཚོང་པ་བདུན་བརྒྱད་ཅིག་ཡོད་ས་རེད། ཁོ་ཚོའི་ཚོང་མང་ཆེ་བ་ཕྱི་རྒྱལ་གྱི་མིར་བཙོང་རྒྱུའི་རྡོའི་རིགས་ཡིན་ས་རེད། ཉེ་ཆར་ང་ཚོར་དུད་ཚང་ཁ་ཤས་ལ་གྱོང་ཆེན་པོ་ཕོག་སོང་ཟེར་ནས་ཡར་སྙན་སེང་བྱུང་སོང་། བྱུང་བ་གཞིར་བཟུང་ལམ་སེང་བྷིང་ལོར་གྱི་ལྷོ་སྤྱི་ས་གནས་རང་ལ་ཕེབས་ནས་དངོས་ཡོད་དཀའ་ངལ་གང་འཕྲད་འདུག གྱོང་གུད་ག་ཚོད་ཕོག་འདུག དེ་ཚོའི་ཐོག་ལ་ཞིབ་འཇུག་བྱོས་ཟེར་ནས་བཀའ་མངགས་བྱས་ཡོད་པ་རེད། ད་ལྟ་ང་ཚོར་གནས་ཚུལ་ཆ་ཚང་བ་ཞིག་འབྱོར་མ་སོང་ཡང་། དུད་ཚང་བདུན་ནང་ནས་གསུམ་ལ་གྱོང་གུད་ཆེན་པོ་ཕོག་ཡོད་པའི་གནས་ཚུལ་ཞིག་འདུག ལྷོ་སྤྱི་ས་གནས་ལ་སྐོར་སྐྱོད་ཕྱིན་པའི་སྙན་ཐོ་ཡར་འབྱོར་རྗེས། གྱོང་གུད་ལ་དཔག་པའི་རོགས་པ་ག་ཚོད་དགོས་ཀྱི་འདུག གོད་ཆག་ཕོག་མཁན་ལ་རོགས་རམ་གནང་ཕྱོགས་ཀྱི་སྒྲིག་གཞི་ཞིག་འདིར་ཡོད་པ་རེད། གྱོང་གུད་གྱི་དངུལ་འབོར་ལྟ་བུ་ཞིག་གནང་ཐུབ་རྒྱུ་ཁག་པོ་ཡིན་ཡང་། ང་ཚོས་དེ་སྔ་ནས་གང་ཐུབ་ཅི་ཐུབ་གནང་དང་གནང་གི་ཡོད་པ་གཞིར་བཟུང་ད་རེབ་ཀྱང་གང་ཡོང་ཡོང་གནང་གི་རེད། ཅེས་གསུངས་བྱུང་། གསར་འགོད་པ། དོན་གྲུབ།
