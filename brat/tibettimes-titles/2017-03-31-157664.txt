སྲིད་འཛིན་ཟུར་པར་ལོ་ ༡༠
ཡན་གྱི་ཁྲིམས་ཆད་ཕོག་སྲིད།

དེ་རིང་ཞོགས་པར་ལྷོ་ཀོ་རི་ཡའི་སྲིད་འཛིན་ཟུར་པ་ལྕམ་ཕར་ཅིན་ཧའེ་ལགས་འཛིན་བཟུང་བྱས་ཏེ་བཀག་ཉར་ཁང་ཞིག་ཏུ་བཅུག་ཡོད་འདུག དེ་རིང་ཞོགས་པར་ལྷོ་ཀོ་རི་ཡའི་ཞིབ་བཤེར་ཁང་གིས་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་དོགས་གཞིའི་འོག སྲིད་འཛིན་ཟུར་པ་ལྕམ་ཕར་ཅིན་ཧའེ་ལགས་འཛིན་བཟུང་བྱས་ཏེ་རྒྱལ་ས་སཱོལ་(Seoul )གྱི་བཀག་ཉར་ཁང་ཞིག་ཏུ་བཅུག་ཡོད་པ་དང་། ལྕམ་ཕར་ཅིན་ཧའེ་ལགས་ནི་ལྷོ་ཀོ་རི་ཡའི་ལོ་རྒྱུས་ཁྲོད་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་ནག་ཉེས་འོག་འཛིན་བཟུང་བྱས་པའི་སྲིད་འཛིན་གསུམ་པ་དེ་ཆགས་ཡོད་འདུག ཉེ་སྔོན་ནས་དབུས་ཁྲིམས་ཁང་གིས་ལྕམ་ཕར་ཅིན་ཧའེ་ལགས་ཀྱིས་ལྐོག་རྔན་བླངས་བ་དང་། དབང་ཆ་ལོག་སྤྱོད། སྲིད་གཞུང་གི་གསང་བ་ཕྱིར་གྱར་བ་སོགས་ཀྱི་དོགས་གཞིའི་འོག་འཛིན་བཟུང་དགོས་པའི་བཀའ་རྒྱ་བསྒྲགས་ཡོད་ཀྱང་། དངོས་སུ་འཛིན་བཟུང་དགོས་པའི་ཐག་གཅོད་དེ་ཁ་སང་ཆུ་ཚོད་ ༩ ཙམ་རིང་ཁྲིམས་ཁང་དུ་འདྲི་གཅོད་བྱས་པ་བརྒྱུད་ཐག་གཅོད་གནང་ཡོད་པ་དང་། ཞིབ་བཤེར་ཁང་ནས་དངོས་སུ་གཏུག་བཤེར་མ་བྱས་གོང་མང་ན་ཉིན་གྲངས་ ༢༠ རིང་བཀག་ཉར་འོག་འཇོག་སྲིད་སྐོར་བརྗོད་འདུག ལྷོ་ཀོ་རི་ཡའི་གསར་ལམ་ཁག་ཏུ་གནས་ཚུལ་ཐོན་པ་ལྟར་ན། ལྕམ་ཕར་ཅིན་ཧའེ་ལ་ནག་ཉེས་མི་འདྲ་བ་ ༡༣ བགྲང་གི་ཡོད་པའི་ནང་ནས་ཚབས་ཆེ་ཤོས་ལྐོག་རྔན་འབོར་ཆེན་བླངས་བའི་དོགས་གཞི་དེ་ཡིན་ལ།  ལྷོ་ཀོ་རི་ཡའི་བཅའ་ཁྲིམས་ནང་ལྐོག་རྔན་ཀོ་སྒོར་དུང་ཕྱུག་གཅིག་ལས་བརྒལ་ཚེ། ལོ་ ༡༠ ཡན་ནམ་ཡང་ན་ཚེ་བཙོན་གྱི་ཁྲིམས་ཆད་ཕོག་གི་ཡོད་སྟབས། ལྕམ་ཕར་ཅིན་ཧའེ་ལགས་ལ་ལོ་ངོ་ ༡༠ ཡན་གྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་གཅོད་སྲིད་པའི་ཚོད་དཔག་བྱེད་ཀྱི་ཡོད་འདུག ལྕམ་ཕར་ཅིན་ཧའེ་ལགས་ནི་ལྷོ་ཀོ་རི་ཡའི་ལོ་རྒྱུས་ནང་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་ནག་ཉེས་འོག་འཛིན་བཟུང་བྱས་པའི་སྲིད་འཛིན་གསུམ་པ་དེ་ཆགས་ཡོད་པ་དང་། ཕྱི་ལོ་ ༡༩༨༨ ལོར་སྲིད་འཛིན་སྐབས་ལྔ་པ་ཆོན་ཌུ་ཧན་(Chun Doo-hwan)ལགས་ཀྱིས་རོ་ཊིའུ་(Roh Tae-woo)ལགས་ལ་ཞི་འཇགས་ངང་སྲིད་དབང་རྩིས་སྤྲོད་གནང་ཡོད་ཀྱང་། སྲིད་དབང་རྩིས་སྤྲོད་བྱས་ཏེ་ཟླ་གཅིག་ཙམ་ཡང་མ་འགོར་བར་ཆོན་ཌུ་ཧན་ལགས་ཀྱི་དབང་ཆ་བེད་སྤྱད་དེ་ཁོའི་ཁྱིམ་ཚང་གིས་ལྐོག་རྔན་འབོར་ཆེན་བླངས་བ་དང་། རྒྱལ་ཁབ་ལ་རྒྱབ་གཏད་པའི་གནད་དོན་སོགས་རིམ་གྱིས་ལྷང་ལ་བུད་ཡོད་པ་མ་ཟད། ཆོན་ཌུ་ཧན་ལགས་དང་རོ་ཊིའུ་གཉིས་བར་ལྐོག་རྔན་སྤྲོད་ལེན་གྱི་འབྲེལ་བ་ཡོད་པ་དང་། རོ་ཊིའུ་ཡིས་ལྐོག་རྔན་དུང་ཕྱུར་མང་པོ་བླངས་བའི་ནག་ཉེས་བཅས་ར་འཕྲོད་པར་བརྟེན། ཕྱི་ལོ་ ༡༩༩༦ ཟླ་ ༨ པའི་ནང་ཆོན་ཌུ་ཧན་ལགས་ལ་དུས་ཡུན་ཕྱིར་འགྱངས་ཀྱིས་སྲོག་ཁྲིམས་བཅད་པ་དང་ཕྱིས་སུ་ཚེ་བཙོན་དུ་བསྒྱུར། རོ་ཊིའུ་ལ་ལོ་ ༡༧ གྱི་ཁྲིམས་ཐག་བཅད་ཡོད་པ་རེད།  ཡིན་ནའང་། བཙོན་ཁང་དུ་ལོ་གཅིག་ཙམ་ལས་མ་ཕྱིན་པར་ཕྱི་ལོ་ ༡༩༩༧ ཟླ་ ༡༢ པའི་ནང་ཁོ་གཉིས་ཀློད་གྲོལ་བཏང་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
