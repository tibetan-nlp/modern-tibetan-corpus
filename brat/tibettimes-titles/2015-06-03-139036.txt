མི་ཁྲི་ ༡༠༠ ལྷག་རེ་ཐ་མག་གིས་བསད།

རྒྱ་ནག་ནང་ཐ་མག་སྤྱོད་པའི་གནས་སྟངས་ཧ་ཅང་ཚབས་ཆེན་ཡིན་པས་ཟླ་འདིའི་ཕྱི་ཚེས་ ༡ ནས་བཟུང་པེ་ཅིན་གྲོང་ཁྱེར་ནང་ཐ་མག་བཀག་སྡོམ་ཤུགས་ཆེ་རུ་བཏང་ཡོད་འདུག པེ་ཅིན་གྲོང་ཁྱེར་འཕྲོད་བསྟེན་རྟོག་ཞིབ་ཁང་གིས་༼པེ་ཅིན་གྲོང་ཁྱེར་གྱི་ཐ་མག་བཀག་སྡོམ་སྲོལ་ཡིག༽ཅེས་པ་ཞིག་ཡོངས་བསྒྲགས་བྱས་ཏེ། ཟླ་འདིའི་ཕྱི་ཚེས་ ༡ ཉིན་ནས་བཟུང་ཐ་མག་བཀག་སྡོམ་སྲོལ་ཡིག་དེ་དང་འགལ་བའི་མི་སྒེར་རེར་མཐོ་ཤོས་རྒྱ་སྒོར་ ༢༠༠ ཡིས་ཉེས་ཆད་དང་། སྡེ་ཁག་ཅིག་ལ་མཐོ་ཤོས་རྒྱ་སྒོར་ཁྲི་ ༡ གིས་ཉེས་ཆད་གཅོད་རྒྱུ་ཡིན་ལུགས་བརྗོད་ཡོད་པ་མ་ཟད། པེ་ཅིན་གྲོང་ཁྱེར་ཡོངས་ཀྱི་ལས་ཁུངས་ཁག་དང་སློབ་གྲྭ་ཆེ་འབྲིང་ཆུང་གསུམ། སྤྱི་གླིང་། དེ་བཞིན། རིག་གནས་སྲུང་སྐྱོབ་ལྟེ་གནས་དང་འགྲིམ་འགྲུལ་ས་ཚིགས། ཟ་ཁང་སོགས་སུ་ཐ་མག་བཀག་སྡོམ་གྱི་བརྡ་ཁྱབ་འབྱར་ཡོད་པ་མ་ཟད། པེ་ཅིན་རྒྱལ་སྤྱིའི་གནམ་ཐང་གི་ཐ་མག་ཆེད་སྤྱོད་ཁང་གསུམ་སྒོ་བརྒྱབ་པ་དང་། འདི་ལོའི་ཕྱི་ཟླ་ ༥ ཚེས་ ༡༠ ནས་བཟུང་ཐ་མག་སྡེབ་ཚོང་ཁང་དག་གི་ད་ལྟའི་ཁྲལ་བརྒྱ་ཆ་ ༥ ནས་བརྒྱ་ཆ་ ༡༡ ལ་འཕར་གཏན་འཁེལ་ཡོད་སྟབས། ད་ལོ་ནི་རྒྱ་ནག་གི་ལོ་རྒྱུས་ཐོག་ཐ་མག་ལ་བཀག་སྡོམ་བྱས་པ་ཤུགས་ཆེ་ཤོས་ཤིག་ཆགས་ཡོད་སྐོར་རྗོད་ཀྱི་འདུག ཕྱི་ཟླ་ ༥ ཚེས་ ༣༡ སྟེ་འཛམ་གླིང་ཐ་མག་འགོག་པའི་ཉིན་མོར་རྒྱ་ནག་ནད་རིགས་སྔོན་འགོག་ལྟེ་གནས་ཀྱིས། ༼ཐ་མག་འགོག་པའི་བདེ་ཐང་སློབ་གསོའི་གལ་ཆེའི་ཆ་འཕྲིན།༽ཞེས་པའི་བསྒྲགས་ཡིག་ཅིག་སྤེལ་བའི་ནང་། རྒྱ་ནག་ནི་འཛམ་གླིང་ཐོག་ཐ་མག་བཟོ་སྐྲུན་གྱི་གྲངས་འབོར་མང་ཤོས་དང་བྲིན་ཤོས་ཀྱི་རྒྱལ་ཁབ་ཆགས་ཡོད་པ་དང་། ད་ལྟ་རྒྱ་ནག་ནང་མི་དུང་ཕྱུར་ ༣ ལྷག་གིས་ཐ་མག་འཐེན་གྱི་ཡོད་ལ། པེ་ཅིན་གྲོང་ཁྱེར་གཅིག་པུའི་ནང་མི་ཁྲི་ ༤༡༩ ཡིས་ཐ་མག་སྤྱོད་ཀྱི་ཡོད་སྟབས། ལོ་རེར་མི་ཁྲི་ ༡༠༠ ལྷག་རེ་ཐ་མག་གི་རྐྱེན་གྱིས་ནད་རིགས་སྣ་ཚོགས་བྱུང་སྟེ་དུས་མིན་འདས་གྲོངས་སུ་འགྲོ་བཞིན་ཡོད་སྐོར་བརྗོད་ཡོད་སྟབས། འཛམ་གླིང་འཕྲོད་བསྟེན་ལྷན་ཚོགས་ཀྱིས། གལ་སྲིད་རྒྱ་ནག་ནང་གོམས་གཤིས་ངན་པ་མུ་མཐུད་རྒྱུན་འཁྱོངས་བྱས་པ་ཡིན་ན། ཕྱི་ལོ་ ༢༠༥༠ ལོར་ལོ་རེའི་འདས་གྲོངས་ཀྱི་གྲངས་འབོར་ཁྲི་ ༣༠༠ ལ་འཕར་སྲིད་པའི་ཉེན་བརྡ་བཏང་ཡོད་པ་རེད། ཕྱི་ལོ་ ༢༠༠༣ ལོར་ཇེ་ནི་ཝ་(Geneva)རུ་ཚོགས་པའི་འཛམ་གླིང་འཕྲོད་བསྟེན་ལྷན་ཚོགས་སྐབས་ ༥༦ པའི་ཐོག ༼ཐ་མག་ཚོད་འཛིན་གྱི་སྒྲོམ་གཞིའི་སྤྱི་ཆིངས༽ཟེར་བར་རྒྱལ་ཁབ་མང་པོས་གྲོས་མཐུན་མིང་རྟགས་བཀོད་པ་དང་། ཕྱི་ལོ་ ༢༠༠༣ ཟླ་ ༡༡ ཚེས་ ༡༠ ཉིན་རྒྱ་ནག་གཞུང་གིས་ཀྱང་སྤྱི་ཆིངས་དེར་མིང་རྟགས་བཀོད་དེ། གྲོང་ཁྱེར་ཁག་གི་ནང་ཐ་མག་བཀག་སྡོམ་གྱི་ལས་གཞི་གང་འཚམ་སྤེལ་ཡོད་ཀྱང་། གྲུབ་འབྲས་ཐོན་རྒྱུ་ལས་ལྡོག་སྟེ་གནས་སྟངས་ཇེ་སྡུག་ཇེ་ཞན་དུ་འགྲོ་བཞིན་ཡོད་པ་མ་ཟད། ཕྱི་ལོ་ ༢༠༠༤ ནས་ ༢༠༡༤ བར་རྒྱ་ནག་ནང་ཐ་མག་ཐོན་སྐྱེན་ཚད་གཞི་བརྒྱ་ཆ་ ༣༩ ལ་འཕར་ཏེ། འདས་པའི་ལོ་ ༡༡ ནང་ཐ་མག་རྐང་དུང་ཕྱུར་ ༢༤༩༩༨༠ ཐོན་སྐྱེད་བྱས་ཡོད་འདུག ཐ་མག་བཀག་སྡོམ་གྱི་སྲོལ་ཡིག་དེ་བསྒྲགས་རྗེས། གནད་དོན་དབྱེ་ཞིབ་པ་ཚོས་ཐ་མག་བཀག་སྡོམ་གྱི་བྱ་བ་གཙོ་བོ་ནི། སྒྲིག་ཁྲིམས་བཟོ་རྒྱུ་ཁོ་ན་མིན་པར་སྒྲིག་ཁྲིམས་དངོས་སུ་ལག་བསྟར་བྱ་རྒྱུ་དེ་ཡིན་པས། དུས་ཡུན་རིང་པོའི་འཆར་གཞི་བསྒྲིག་སྟེ་ལྟ་ཞིབ་དང་དྲིལ་བསྒྲགས་གང་མང་བྱ་དགོས་པ་མ་ཟད། ཐ་མག་འཐེན་མཁན་ལ་ཉེས་ཆད་ལྕི་མོ་གཅོད་དགོས་ཚུལ་དང་། ཡང་ཐ་མག་གི་རིན་གོང་དང་ཚོང་འདྲེན་གྱི་ཁྲལ་མཐོ་རུ་གཏོང་རྒྱུ་དེ་ཐ་མག་བཀག་སྡོམ་བྱེད་པའི་ཐབས་ལམ་ཡག་ཤོས་ཡིན་སྐོར་རྗོད་ཀྱི་འདུག་ལ། ཡང་ཁ་ཤས་ཀྱིས་སྲིད་གཞུང་གིས་བདེ་ཐང་ལ་ཁ་གཡར་བ་ཙམ་མ་གཏོགས་དོན་གྱིས་ཐ་མག་གི་རིན་གོང་དང་ཁྲལ་མཐོ་རུ་བཏང་སྟེ། སྒོར་མོར་དམིགས་པ་ཞིག་ཡིན་པའི་སྐྱོན་བརྗོད་ཀྱང་བྱེད་ཀྱི་ཡོད། གསར་འགོད་པ། པདྨ་མཚོ།
