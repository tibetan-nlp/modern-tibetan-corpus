བསོད་ནམས་མགོན་པོ་ཀློད་གྲོལ་བཏང་།

  ཟི་ཁྲོན་དཀར་མཛེས་ཁུལ་སེར་ཤུལ་རྫོང་། རྫ་ཆུ་ཁ་རྫ་དབོན་པོ་དགོན་གྱི་གྲྭ་བསོད་ནམས་མགོན་པོ་ལགས་ཉིན་ཤས་གོང་ཀློད་གྲོལ་བཏང་སྟེ་ཕྱིར་ནང་དུ་འབྱོར་ཡོད་པ་ཤེས་རྟོགས་བྱུང་ཡོད་ཀྱང་། ཁོང་ཚེས་གྲངས་ག་དུས་ཀློད་གྲོལ་བཏང་བ་དང་ད་ལྟའི་གཟུགས་གཞིའི་གནས་སྟངས་སོགས་ཤེས་རྟོགས་མི་འདུག གྲྭ་བསོད་ནམས་མགོན་པོ་ལགས་ནི་ ༢༠༡༢ ཟླ་ ༡༢ པའི་ཚེས་ ༡༢ གྱི་ཞོགས་པ་ཆུ་ཚོད་བརྒྱད་པ་ཡས་མས་སུ་དབོན་པོ་དགོན་ནང་ནས་འཛིན་བཟུང་བྱས་འདུག་ལ། ཁོང་དང་མཉམ་དུ་དབོན་པོ་དགོན་རྨང་གཞིའི་རིག་གནས་སློབ་གླིང་གི་སློབ་མ་བསོད་ནམས་ཆོས་དར་དང་། བློ་བཟང་མཁས་གྲུབ། བློ་བཟང་ནོར་བུ། བློ་བཟང་དབྱིག་གཉེན། བསྟན་འཛིན་དགེ་འདུན། བཅས་འཛིན་བཟུང་བྱས་རྗེས་ཟླ་མང་པོའི་རིང་གར་སོང་ཆ་མེད་དུ་གྱུར་ཏེ། ༢༠༡༣ ཟླ་ ༩ ཚེས་ ༡༡ ཉིན་བསོད་ནམས་ནམས་མགོན་པོ་དང་། བསོད་ནམས་ཆོས་དར་གཉིས་ལ་ལོ་བཞི་རེའི་ཁྲིམས་ཆད་དང་། གཞན་རྣམས་རིམ་པས་ཀློད་གྲོལ་ཐོབ་ཡོད། ཁོང་རྣམས་འཛིན་བཟུང་བྱེད་དགོས་པའི་ཉེས་འཛུགས་གཙོ་བོ་ནི་ ༢༠༡༢ ཟླ་ ༢ པའི་ནང་དུ་དགོན་པའི་ཉེ་མཁོར་ཡུལ་ཚོའི་སློབ་རའི་ནང་གི་རྒྱ་ནག་གི་དར་དམར་མར་ཕབ་ཏེ་བོད་ཀྱི་དར་སེང་བསྒྲེངས་ནས་སྦྱར་ཡིག་དང་འགྲེམས་སྤེལ་བྱས་པར་བརྟེན་དམག་མི་འབོལ་ཆེན་ཡོང་སྟེ་ཟླ་གཅིག་ལྷག་རིང་སུན་བཙིར་འཇིགས་སྐུལ་བྱས་ཡོད་ཀྱང་སྐབས་དེར་མི་སུ་ཡང་འཛིན་བཟུང་བྱེད་ཐུབ་མེད་པ་དང་།   ལོ་དེའི་ཕྱི་ཟླ་ ༩ བའི་ནང་བསླར་ཡང་དར་དམར་མར་ཕབ་སྟེ་སེང་དར་བསྒྲེངས་ནས་ཡིག་ཆ་མང་པོ་གང་སར་བཀྲམས། ཟླ་དེའི་ཚེས་ ༡༥ ཡི་མཚན་མོར་དམག་མི་མང་པོ་འབྱོར་ནས་ཡང་བསྐྱར་སུན་བཙིར་བཟོ་མགོ་བརྩམས་ཏེ། ༢༠༡༢ ཟླ་ ༡༠ ཚེས་ ༡༥ ནས་ ༢༤ བར་དུ་གྲྭ་ཆོས་དར་དང་། སྐྱ་པེ། མི་འཁྲུགས་སོགས་སེར་སྐྱ་ ༢༧ ཡས་མས་ཤིག་འཛིན་བཟུང་བྱས་ཏེ། རྡུང་རྡེག་ཚད་མེད་བཏང་རྗེས་ཆོས་དར་དང་སྐྱ་པེ། མི་འཁྲུགས་གསུམ་ཟླ་ངོ་མང་པོའི་རང་གར་སོང་ཆ་མེད་དུ་གྱུར་བ་དང་། གཞན་རྣམས་རིམ་པས་ཀློད་གྲོལ་ཐོབ་ཡོད། དེ་བཞིན། གོང་གསལ་ལྟར་ ༢༠༡༢ ཟླ་ ༡༢ ཚེས་ ༡༢ ཉིན་བསོད་ནམས་མགོན་པོ་སོགས་འཛིན་བཟུང་བྱས་པའི་གནས་ཚུལ་བྱུང་། མཇུག་མཐར་ཆབ་སྲིད་གནད་དོན་དེ་དག་དང་འབྲེལ་ཡོད་ཉེས་འཛུགས་ཀྱིས་ ༢༠༡༣ ཟླ་ ༩ ཚེས་ ༩ གྲྭ་ཆོས་དར་ལ་ཁྲིམས་ཆད་ལོ་གཅིག་དང་སྐྱ་པེ་དང་། མི་འཁྲུགས་གཉིས་ལ་ལོ་བཞིའི་ཆབ་སྲིད་ཁེ་དབང་མེད་པ་བཟོས་པ་མ་ཟད། འབྲེལ་ཡོད་ཚོར་ཆབ་སྲིད་ལས་དོན་ལ་མི་ཞུགས་པའི་འགན་ཁུར་བཅུགས་ཏེ་ཀློད་གྲོལ་བཏང་ཡོད། བསོད་ནམས་མགོན་པོ་དང་། བསོད་ནམས་ཆོས་དར་གཉིས་ལ་ལོ་ ༤ རེའི་ཁྲིམས་ཐག་བཅད་ཡོད་ཀྱང་བསོད་ནམས་ཆོས་དར་ལ་ཁྲིམས་ཆད་ལོ་བཞི་ཚང་ཡོད་ནའང་ད་དུང་ཀློད་གྲོལ་བཏང་མེད།   སྦེལ་ཀོབ་ས་གནས་གསར་འགོད་པ་དོན་འགྲུབ་བཀྲ་ཤིས།
