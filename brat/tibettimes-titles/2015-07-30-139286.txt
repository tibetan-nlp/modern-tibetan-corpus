རྩོམ་པ་པོ་ཞིག་ཀློད་གྲོལ་བཏང་།

ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༤ ཉིན་ཚ་བ་དཔའ་ཤོད་རྫོང་ནས་ཡིན་པའི་གཞོན་སྐྱེས་རྩོམ་པ་པོ་གངས་ཅན་པ་གུ་དྲག་ལགས་ཀློད་གྲོལ་བཏང་ཡོད་ཀྱང་མུ་མཐུད་ལྟ་རྟོག་ནན་པོ་བྱེད་བཞིན་ཡོད་འདུག འདི་གར་གྲྭ་བྱང་ཆུབ་སེམས་པ་ལགས་ཀྱིས་གནས་ཚུལ་མཁོ་སྤྲོད་གནང་བ་ལྟར་ན། གཞོན་སྐྱེས་རྩོམ་པ་པོ་བསོད་ནམས་ཚེ་འཕེལ་ལམ་སྨྱུག་མིང་གངས་ཅན་པ་གུ་དྲག་ལགས། ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༤ ཉིན་ཉ་རི་བཙོན་ཁང་ནས་ཀློད་གྲོལ་བཏང་ཡོད་ཀྱང་། ད་རེས་ཁོང་གིས་ཁྲིམས་འགལ་ནག་ཉེས་བཟོས་པ་ཐེངས་གཉིས་པ་ཡིན་ལུགས་བརྗོད་དེ། དིང་རི་རྫོང་ཁོངས་ནས་བརྒལ་ཏེ་ས་གནས་གཞན་ལ་གནས་སྐབས་རིང་འགྲོ་མི་ཆོག་པའི་བཀའ་རྒྱ་བཏང་ཡོད་པ་མ་ཟད། ཁོང་གི་ཁ་པར་དང་ཐོབ་ཐང་ལག་ཁྱེར་སོགས་ཕྱིར་སྤྲད་མེད་པ་དང་། ཉིན་རེ་བཞིན་རྫོང་ཉེན་རྟོག་ཁང་ལ་མི་སུ་དང་འབྲེལ་བ་བྱས་པ་དང་ལས་ཀ་ཅི་བྱས། ས་ཆ་གང་དུ་ཕྱིན་སོགས་ཡར་ཞུ་ཐོ་འགོད་བྱེད་དགོས་ཀྱི་ཡོད་འདུག ཁོང་གིས་ལ་སྟོད་དིང་རི་རྫོང་ཤེལ་དཀར་གྲོང་རྡལ་དུ་ཚོང་ལས་གཉེར་ནས་བསྡད་ཡོད་པ་དང་། འདི་ལོའི་ཕྱི་ཟླ་ ༦ ཚེས་ ༢༩ ཉིན་གྱི་བོད་ནང་གི་ཕྱི་དྲོའི་ཆུ་ཚོད་ ༦ པ་ཙམ་ལ་རྒྱ་གཞུང་ཉེན་རྟོག་པས་ཁོང་གློ་བུར་དུ་འཛིན་བཟུང་བྱས་ཡོད་པ་མ་ཟད། མི་རིགས་བར་དཀྲུག་ཤིང་འགལ་ཟླ་སློང་བའི་རྩོམ་ཡིག་བྲིས་པ་དང་། ༧གོང་ས་མཆོག་དང་སྲིད་སྐྱོང་བློ་བཟང་སེང་གེ་མཆོག་སོགས་ཀྱི་གསུང་བཤད་ཉར་ཚགས་བྱས་པ། ༧གོང་ས་མཆོག་གི་སྐུ་པར་གང་སར་སྤྲད་ནས་སྤྱི་ཚོགས་ཀྱི་བདེ་འཇགས་ལ་གནོད་སྐྱོན་བཟོས་པ་སོགས་ཀྱི་ཉེས་མིང་འོག དུས་ཡུན་ཉིན་ ༢༠ རིང་གཞིས་ཀ་རྩེའི་ཉ་རི་བཙོན་ཁང་དུ་བཀག་ཉར་བྱས་ཡོད་པའི་ཁར། རྒྱ་སྒོར་ ༡༥༠༠ ཡིས་ཉེས་ཆད་བཅད་ཡོད་འདུག དེ་བཞིན་ཕྱི་ལོ་ ༢༠༠༩ ལོར་གཞིས་ཀ་རྩེ་ནས་རྒྱ་གཞུང་ཉེན་རྟོག་པས་ཁོང་འཛིན་བཟུང་བྱས་ཏེ་ཟླ་ ༩ ལྷག་རིང་ཉ་རི་བཙོན་ཁང་དུ་བཀག་ཉར་བྱས་ཡོད་པ་དང་། ཕྱིས་སུ་རྒྱལ་ཁབ་ཁ་བྲལ་གཏོང་བར་ངན་སྐུལ་བྱས་པའི་ཉེས་མིང་འོག ལོ་གཉིས་ཀྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་དེ་ངལ་རྩོལ་བསྒྱུར་བཀོད་ཁང་ཞིག་ཏུ་བཅུག་ཡོད་པ་དང་། ཕྱི་ལོ་ ༢༠༡༢ ལོའི་ལོ་སྟོད་ཙམ་ལ་གཞི་ནས་ཀློད་གྲོལ་ཐོབ་ཡོད་འདུག དེ་སྔོན་ཁོང་གིས་བྲིས་པའི་སྙན་ངག་ཅིག་ཏུ། དབྱངས་གསལ་སོ་བཞི་མ་བརྗེད་གངས་ཅན་པ།། ཕ་སྐད་གཙང་མ་མ་བརྗེད་གངས་ཅན་པ།། རང་ཚུགས་རང་གིས་མ་ཤོར་གངས་ཅན་པ།། ཕ་ས་གཞན་ལ་མ་བཙོང་གངས་ཅན་པ།། ཕ་ཡུལ་རྒྱང་ལ་མ་སྐྱུར་གངས་ཅན་པ།། རིགས་རྒྱུད་གཞན་ལ་མ་བསྲེས་གངས་ཅན་པ།། རང་རིགས་ལ་རྒྱ་མ་ཤོར་གངས་ཅན་པ།། ཞེས་བྲིས་ཡོད་པ་སོགས། བོད་མི་རིགས་ཀྱི་ངོ་བོ་སྲུང་སྐྱོབ་དང་སྐད་ཡིག་རིག་གཞུང་གཅེས་འཛིན་དགོས་པའི་སྐོར་གྱི་རྩོམ་ཡིག་མང་པོ་བྲིས་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
