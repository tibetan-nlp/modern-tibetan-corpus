འཇིགས་མེད་མགོ་རིལ་སྨན་ཁང་དུ་ཁྲིད་འདུག

འཇིགས་མེད་མགོ་རིལ་ལགས་ད་ལོའི་ཕྱི་ཟླ་ ༩ པའི་ནང་ལོ་ལྔའི་བཙོན་འཇུག་དུས་ཡུན་རྫོགས་ཀྱི་ཡོད་ཀྱང་ཉེ་ལམ་ཞིག་ནས་ནང་མིར་ཐུག་འཕྲད་བྱེད་བཅུག་མེད་པ་མ་ཟད་ཟླ་འདིའི་ཚེས་མགོ་ཞིག་ལ་སྨན་ཁང་དུ་འཁྲིད་ཡོད་འདུག   བོད་ནས་གནས་ཚུལ་འབྱོར་བ་ལྟར་ན། བླ་བྲང་རྒྱུད་སྟོད་གྲྭ་ཚང་གི་དགེ་འདུན་པ་འཇིགས་མེད་རྒྱ་མཚོའམ་ཡོངས་གྲགས་འཇིགས་མེད་མགོ་རིལ་ལགས་སུ་རྒྱ་ནག་གཞུང་གིས་ཁྲིམས་རྩོད་པ་བརྒྱུད་མ་བཅུག་པར་ལོ་ ༥ ཡི་ཁྲིམས་ཆད་བཅད་ཡོད་པ་ནི་ཡོངས་གྲགས་ཞིག་ཡིན། ཁྲིམས་ཆད་བཅད་རྗེས་ཟླ་རེ་ལ་ཐེངས་རེར་ནང་མི་ཐུགས་འཕྲད་དུ་འགྲོ་ཆོག་ནའང་། ཐུག་འཕྲད་སྐབས་ཟ་མ་བསྐྱལ་འཇུག་གི་མེད་པ་མ་ཟད་ཁ་པར་བརྒྱུད་སྐར་ཆ་ ༢༠ རེ་མ་གཏོགས་སྐད་ཆའང་བཤད་འཇུག་གི་མེད། དེ་ལྟ་ནའང་། ཕྱི་ལོ་ ༢༠༡༦ ཕྱི་ཟླ་ ༢ པའི་ནང་ནང་མི་ཐུག་འཕྲད་བྱེད་བཅུག་མེད་སྟབས། ༢༠༡༦ ཕྱི་ཟླ་ ༡ པོའི་ཚེས་ ༢༥ ཉིན་གྱི་ཐུག་འཕྲད་ནི་ནང་མི་ཐུག་འཕྲད་མཐའ་མ་དེ་ཆགས་ཡོད་འདུག ༢༠༡༦ ཕྱི་ཟླ་ ༣ པའི་ཟླ་མགོར་ཁྲིམས་ཁང་མི་སྣས་ཨ་ཁུ་འཇིགས་མེད་ད་ལྟ་སྨན་ཁང་ནང་(兰州大厦平监狱医院 ཞེས་པ་བཙོན་མའི་སྨན་ཁང་དེ་ཡིན་པ་རེད)ཁྲིད་ཡོད་པའི་གནས་ཚུལ་ཞིག་ནང་མི་ལ་ཁ་པར་བརྒྱུད་ནས་བརྡ་འབྱོར། ནང་མིས་ཁོང་ལ་ན་ཚ་ཅི་ཡོད་དྲིས་པ་ན། གནས་སྟངས་ཛ་དྲག་མིན་པས་སེམས་འཁྲལ་མི་དགོས་ཚུལ་བཤད་འདུག ཡིན་ནའང་། ཕྱི་ ༡ པོའི་ཟླ་མཇུག་ལ་ནང་མི་ཐུག་འཕྲད་སྐབས་ཨ་ཁུ་འཇིགས་མེད་རང་གིས། ང་སྨན་ཁང་ལ་ཁྲིད་རྒྱུ་ཡིན་པའི་སྐད་ཆ་ཞིག་འདུག་པས་དེ་ལ་དོ་སྣང་གྱིས་ཞེས་དངོས་སུ་གསུངས་འདུག་པ་དང་། དེ་དུས་ན་ཚ་འདི་ཡོད་ཟེར་བ་བཤད་རྒྱུ་མེད་པས་ནང་མི་ཚོས་ཁོང་བཙོན་གྲོལ་ཐོབ་ལ་ཉེ་བའི་སྐབས་འདིར་སྨན་ཁང་ལ་ཁྲིད་པར་བློ་དོགས་དང་སེམས་འཁྲལ་འཕར་མ་བྱེད་ཀྱིན་ཡོད་འདུག   སྤྱིར་བཏང་ངལ་རྩོལ་བའི་ལས་ཀའི་སྐར་མ་བསགས་ནས་བཙོན་འཇུག་གི་དུས་ཡུན་ཉུང་འཕྲི་བྱེད་ལུགས་ཤིག་ཡོད་པ་ལྟར། ལག་ལེན་དགོས་ལུགས་ཞེས་ཨ་ཁུ་འཇིགས་མེད་ཀྱིས་ཁྲིམས་ཁང་མི་སྣར་ཞུས་ཀྱང་དེ་ལ་ཉན་འཇོག་བྱས་མི་འདུག གལ་སྲིད་དེ་ལག་ལེན་བྱས་ཚེ་ ༢༠༡༦ ཕྱི་ཟླ་ ༢ པའི་ནང་བཙོན་གྲོལ་དང་། དེ་ལྟར་མ་བྱུང་ནའང་ཕྱི་ལོ་ ༢༠༡༦ ཕྱི་ཟླ་ ༩ པའི་ནང་ལོ་ལྔའི་ཁྲིམས་ཆད་ཀྱི་དུས་ཡུན་ཡོངས་སུ་རྫོགས་འགྲོ་ཡི་ཡོད།   ཨ་ཁུ་འཇིགས་མེད་ཁོང་ ༢༠༡༡ ཟླ་ ༨ ཚེས་ ༢༠ ཉིན་འཛིན་བཟུང་བྱས་པ་ནས་བཟུང་ལོ་གསུམ་རིང་ཁྲིམས་ཆད་མ་བཅད་པར་སྨན་ཁང་སོགས་སུ་གསང་བའི་སྒོ་ནས་བཀག་སྡོམ་བྱས་པ་དང་། ༢༠༡༤ ཟླ་ ༨ པའི་ནང་དངོས་སུ་ལོ་ ༥ ཁྲིམས་ཐག་བཅད་ཡོད་པ་རེད། དེ་ནི་ཁོང་བཙོན་འཇུག་བྱེད་ཐེངས་ ༤ པ་དེ་ཆགས་ཡོད་འདུག ཞེས་ཨ་ཁུ་འཇིགས་མེད་ཀྱི་དགེ་ཕྲུག་ཕྲན་དགེ་འདུན་ཕུན་ཚོགས་ལགས་ཀྱིས་གནས་ཚུལ་བཏང་འབྱོར་བྱུང་།
