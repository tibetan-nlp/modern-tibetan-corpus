
རྒྱལ་བའི་སྐུ་ཕྱྭ་གཙོ་བོ་དེ་ཁོང་གི་ཐུགས་དགོངས་དྭངས་ཐབས་བྱ་རྒྱུ་དེ་ཡིན།

བོད་རྒྱལ་ཡོངས་མང་གཙོ་ཚོགས་པའི་ཚོགས་ཆེན་དབུ་འཛུགས་སྟེང་བཀའ་ཟུར་ཀིརྟི་རིན་པོ་ཆེ་མཆོག་གིས་ད་ལྟའི་ང་ཚོའི་སྲིད་སྐྱོང་ག་འདྲ་ཞིག་ཡིན་ཡང་འོས་ཤོག་ཁྲི་ཁ་ཤས་ཀྱི་སྐད་ཆ་རེད། ༧རྒྱལ་བ་རིན་པོ་ཆེ་ལ་མི་འབུམ་ཕྲག་མང་པོས་ཡིད་ཆེས་དང་བློས་འགེལ་བྱས་ཡོད་པ་རེད། དེ་ལ་འབབ་པའི་འགན་ཞིག་དང་ནུས་པ་ཞིག་ཀྱང་ཡོད་པ་རེད་ཅེས་སོགས་གསུངས་སོང་། དེ་རིང་སྔ་དྲོའི་ཆུ་ཚོད་ ༩ བའི་སྟེང་། བཞུགས་སྒར་དྷ་ས་ནོར་བུ་གསོལ་མགྲོན་ཁང་དུ། བོད་རྒྱལ་ཡོངས་མང་གཙོ་ཚོགས་པའི་ཉིན་གསུམ་རིང་གི་ཚོགས་ཆེན་ཐེངས་བདུན་པ་དབུ་འཛུགས་གནང་སོང་། སྐུ་མགྲོན་གཙོ་བོར་བཀའ་ཟུར་ཀིརྟི་རིན་པོ་ཆེ་མཆོག་དང་། དམིགས་བསལ་སྐུ་མགྲོན་དུ་བོད་ཀྱི་གཞོན་ནུ་ལྷན་ཚོགས་ཀྱི་དྲུང་ཆེ་བཀྲ་ཤིས་ལམ་བཟང་ལགས་དང་ཧི་མ་ལ་ཡ་རིག་གཞུང་ལྟེ་གནས་ཁང་གི་འགན་འཛིན་བློ་བཟང་ཕུན་ཚོགས་ལགས་བཅས་ཕེབས་འདུག ཉིན་གསུམ་རིང་གི་ཚོགས་འདུའི་སྟེང་། མང་གཙོ་ཚོགས་པའི་སྒྲིག་གཞི་བསྐྱར་བཅོས་དང་། མ་འོངས་ལས་དོན་བྱེད་ཕྱོགས་སོགས་གྲོས་གཞི་ཁག་བཅུ་ཡོད་པ་དང་། ཚེས་ ༢༦ ཉིན་སྐབས་བདུན་པའི་རྒྱུན་ལས་གསར་འདེམས་གནང་རྒྱུ་ཡིན་འདུག་ལ། ཐེངས་འདིའི་ཚོགས་ཐོག་ཏུ་མང་གཙོ་ཚོགས་པའི་ས་གནས་ཁག་ ༢༠ ནས་ཚོགས་བཅར་བ་ ༤༠ ཙམ་ཕེབས་ཡོད་འདུག ཚོགས་འདུ་དབུ་འཛུགས་སྟེང་། བོད་རྒྱལ་ཡོངས་མང་གཙོ་ཚོགས་པའི་ཚོགས་གཙོ་དགེ་ལེགས་འཇམ་དབྱངས་ལགས་ཀྱིས་ཚོགས་པའི་ལོ་རྒྱུས་དང་ཐེངས་འདིའི་ཚོགས་ཆེན་ངོ་སྤྲོད་གནང་རྗེས། གཞོན་ལྷན་ཆེ་བཀྲ་ཤིས་ལམ་བཟང་ལགས་ཀྱིས་གཏམ་བཤད་གནང་དོན། བཙན་བྱོལ་སྤྱི་ཚོགས་འདི་ད་ཆར་དམངས་གཙོའི་ལམ་དུ་སྐྱོད་བཞིན་པ་ལྟར། འཛམ་གླིང་གི་བཙན་བྱོལ་བའི་སྤྱི་ཚོགས་གཞན་དང་མ་འདྲ་བ་ཞིག་ཆགས་བསྡད་ཡོད་པ་མ་ཟད། དམངས་གཙོའི་གནས་བབ་ཀྱི་ཐོག་ནས་དམངས་གཙོའི་ཀ་བ་བཞི་ཆ་ཡོངས་སུ་ཚང་བའི་ཐོག་ནས་འགྲོ་བཞིན་ཡོད་པ་ཞིག་ཆགས་ཡོད་པ་རེད། འོན་ཀྱང་། དམངས་གཙོའི་ཀ་བ་བཞི་ཡོད་པ་ཙམ་མ་ཡིན་པར་དམངས་གཙོ་ཀ་བ་བཞི་འཛིན་མཁན་གང་ཟག་དེ་གལ་ཆེན་པོ་ཆགས་ཡོད་པ་དང་། ང་ཚོའི་བོད་མིའི་སྒྲིག་འཛུགས་འདི་གྲོས་ཚོགས་ལམ་ལུགས་ཐོག་འགྲོ་བཞིན་ཡོད་སྟབས། གྲོས་ཚོགས་འཐུས་མི་དེ་ཧ་ཅང་གལ་ཆེན་པོ་ཆགས་ཀྱི་ཡོད་པ་ལྟར། གྲོས་ཚོགས་འཐུས་མི་ཚད་ལྡན་ཞིག་འདེམས་རྒྱུ་ཧ་ཅང་གལ་ཆེན་པོ་ཡིན་པ་དང་། གྲོས་ཚོགས་འཐུས་མི་ཚད་ལྡན་ཡོང་ཐུབ་མིན་མི་མང་ལ་ཐུག་ཡོད་པ་རེད། མི་མང་གིས་སྤྱི་འཐུས་ཚད་ལྡན་ཞིག་འདེམས་པའི་གོ་སྐབས་སྐྲུན་ཐུབ་མིན་བོད་རྒྱལ་ཡོངས་མང་གཙོ་ཚོགས་པ་སོགས་གཞུང་འབྲེལ་མིན་པའི་ཚོགས་པ་ཁག་ཆགས་བསྡད་ཡོད་པ་རེད། མ་འོངས་པར་འདི་ལས་ཚགས་ཚུད་དང་ཡར་རྒྱས་གཏང་ཕྱོགས་སུ་འབད་འབུངས་གནང་རྒྱུ་གལ་ཆེན་པོ་རེད། ཅེས་སོགས་བརྗོད་སོང་། ཧི་མ་ལ་ཡའི་རིག་གཞུང་ལྟེ་གནས་ཁང་གི་འགན་འཛིན་བློ་བཟང་ཕུན་ཚོགས་ལགས་ཀྱིས། ངས་འདིར་དོན་ཚན་གཉིས་ཀྱི་སྒང་ལ་ཞུ་འདོད་བྱུང་། དོན་ཚན་དང་པོ་དེ་སྤྱིར་བཏང་བཙན་བྱོལ་སྤྱི་ཚོགས་ནང་ཚོགས་པ་འདྲ་མིན་མང་ཐག་ཆོད་ཅིག་ཡོད་ནའང་། བོད་རྒྱལ་ཡོངས་མང་གཙོ་ཚོགས་འདི་༧རྒྱལ་བ་རིན་པོ་ཆེའི་དགོངས་གཞི་གཞིར་བཟུང་གིས་བཙུགས་པ་ཞིག་ཡིན་སྟབས། ངས་དེ་རྩ་ཆེན་པོ་ཞེ་དྲགས་མཐོང་གི་ཡོད་ལ། ཚོགས་པ་འདི་ང་ཚོའི་ཆོས་ལུགས་དང་ཆོལ་ཁ་ཚང་མར་དགོས་པའི་ཚོགས་པ་ཞིག་རེད། ༧རྒྱལ་བ་རིན་པོ་ཆེ་མཆོག་གིས་ང་ཚོར་དམངས་གཙོ་གནང་ཡོད་པ་དེ་ང་ཚོས་བདག་པོ་ཡག་པོ་ཞིག་རྒྱག་ཐུབ་རྒྱུ་དེ་ང་ཚོའི་ལས་འགན་རེད། ད་ལྟའི་བར་ང་ཚོས་དམངས་གཙོ་ལག་ལེན་བསྟར་བའི་བརྒྱུད་རིམ་ཁྲོད་དམངས་གཙོའི་ཕྱག་ལས་གནང་སྟངས་ཧ་ལས་པ་ཡ་མཚན་པོ་དེ་འདྲ་བྱུང་ཡོད་པ་རེད་ལ། སྐབས་རེར་ང་ཚོ་ཚང་མ་དངངས་སྐྲག་སྐྱེ་དགོས་པ་དེ་འདྲ་ཡང་བྱུང་ཡོད་པ་རེད། དེ་ལ་ཚང་མས་ཐུགས་འགན་བཞེས་དགོས་པ་དང་། ལྷག་པར་དུ་རྒྱལ་ཡོངས་མང་གཙོ་ཚོགས་པའི་ངོས་ནས་དེ་འདྲའི་རིགས་མི་ཡོང་བའི་ཆད་དུ་ཕྱག་ལས་གནང་རྒྱུའི་འགན་ཆེན་པོ་ཡོད་པ་རེད། ཚོགས་པ་འདིའི་ནང་ཆོས་ལུགས་ཆོལ་ཁ་དབྱེར་མེད་ཀྱི་ཚོགས་པ་ཞིག་རེད་ལ། ཚོགས་པ་འདིའི་ལས་འགན་ཡང་ཆེན་པོ་རེད། བོད་མི་མང་སྤྱི་འཐུས་ཟེར་བ་དེ་རྐུབ་སྟེགས་སྟེང་ལངས་ཐུབ་པ་ཙམ་མིན་པར། མི་མང་ལ་ཉན་ཤེས་མཁན་དང་ཕར་ལབ་ཐུབ་མཁན། གོ་བ་ལེན་ཤེས་མཁན། ཕྱག་ལས་གནང་བར་སེམས་ཤུགས་དང་ལྷག་བསམ་རྣམ་དག་ཡིན་པ་ཅིག་བསླེབ་ཐུབ་པའི་ཆེད་དུ་བོད་རྒྱལ་ཡོངས་མང་གཙོ་ཚོགས་པར་ལས་འགན་ཆེན་པོ་ཡོད་པ་རེད་བསམ་གྱི་འདུག མི་མང་ཟེར་བ་དགོངས་འཆར་མེད་པའི་ར་ལུག་ལྟ་བུ་རེད། ལུག་རྫིས་གང་ལ་འཁྲིད་འཁྲིད་ཡིན་སྟབས། ཁ་བདེ་པོ་ཚོའི་ཁ་རྗེས་སུ་རྒྱུག་འགྲོ་བཞག་གི་རེད། ད་ལྟའི་བར་ང་ཚོའི་སྤྱི་འཐུས་ནང་སྤྱི་འཐུས་ཆོ་མེད་པ་དེ་འདྲ་སླེབས་ཡོད་པ་རེད། ང་ཚོའི་མི་མང་གིས་རྒྱབ་ལོག་ནས་ཁྱད་མཚར་བོ་དེ་འདྲ་ལབ། མདུན་དུ་ཐུག་དུས་བཀྲ་ཤིས་བདེ་ལེགས་ཙམ་མ་གཏོགས་ལབ་ཐུབ་ཀྱི་མི་འདུག དེ་ང་ཚོ་བོད་པའི་ཞན་ཆ་རེད། འོས་ཤོག་ལྡུག་དུས་འོས་མི་སུ་ལ་ལྡུག་དགོས་མིན་ཧ་མ་གོ་བར་གཞན་གྱིས་གང་ལ་ལྡུག་གི་ཡོད་ན་དེ་ལ་ལྡུག་མཁན་མང་པོ་རེད། དེ་དག་ཐོག་ལ་བློ་གསལ་ཕུལ་དགོས་པ་རྒྱལ་ཡོངས་མང་གཙོ་ཚོགས་པའི་ལས་འགན་གྱི་ལྟེ་བ་རེད། མི་མང་ལ་ལྟོ་ཡག་པོ་རག་གི་ཡོད་མེད་དང་སློབ་གྲྭར་ཡག་པོ་འགྲོ་ཐུབ་ཀྱི་ཡོད་མེད་རྒྱལ་ཡོངས་མང་གཙོ་ཚོགས་པའི་ལས་འགན་མ་རེད། ལས་འགན་གྱི་ལྟེ་བ་དེ་རྒྱལ་བ་ཡིད་བཞིན་ནོར་བུ་མཆོག་གི་དགོངས་གཞི་གང་ཡིན་པ་དེ་མ་ནོར་བར་བསྒྲུབ་རྒྱུ་དང་། གཉིས་པ་དེ་རྒྱལ་བ་ཡིད་བཞིན་ནོར་བུས་གསོལ་རས་སུ་གནང་བའི་དམངས་གཙོ་འདི་དུས་གཏན་དུ་ཡག་པོ་ཞིག་ཡོང་ཆེད་དང་། ལམ་ཁ་ནོར་བ་ཞིག་ཏུ་མི་འགྲོ་བ་དང་འགྲོ་རུ་མི་འཇུག་རྒྱུར་བོད་ཆོལ་ཁ་གསུམ་ལ་འགན་ཡོད་པ་རེད་ལ། ང་ཚོ་ཕྱི་སྡོད་བོད་མི་ལས་ཀྱང་གལ་ཆེ་བ་དེ་བོད་ནང་གི་བོད་མི་ཚོ་རེད། ཁོང་ཚོས་རྒྱ་གར་དུ་༧རྒྱལ་བ་རིན་པོ་ཆེ་བཞུགས་ཡོད་པ་དང་བོད་གཞུང་ཡོད་པ་རེད། འདི་གར་ཅིག་གནང་གི་རེད་ཟེར་ནས་ཉིན་མཚན་མེད་པར་དྷ་རམ་ས་ལའི་ཕྱོགས་ལ་རེ་བ་བརྒྱབ་ནས་བསྡད་ཡོད་སྟབས། ང་ཚོས་འདི་གར་ཅིག་བྱས་ནས་སྟོན་དགོས་ཀྱི་འདུག་མ་གཏོགས། འདི་ལྟར་ཉི་མ་འཁྱོལ་ཐབས་བྱས་ནས་བསྡད་ན་འགྲིག་གི་ཡོད་པ་མ་རེད། དེ་ང་ཚོ་ཚང་མར་འགན་ཡོད་པ་རེད། ཅེས་བརྗོད་སོང་། ཁོང་གིས་མུ་མཐུད་དུ་གསུངས་དོན། སོ་སོ་སྔོན་ཆད་གཞིས་མགོ་བྱས་པའི་སྐབས་སུ་༧རྒྱལ་བ་རིན་པོ་ཆེ་དང་བོད་གཞུང་ལ་སྐྱོན་བརྗོད་བྱེད་མཁན་དེ་དག་ལ་ཁ་གཏད་བཅག་སྟེ་རྒྱ་གར་ནང་སྡོད་ཐུབ་མེད་པ་ཞིག་བཟོས་པ་ཡིན། དེ་ང་རང་གིས་ཅིག་བྱས་བཞག་ཡོད་ཟེར་གྱི་མེད། ལས་ཀ་ག་རེ་བྱས་ཀྱང་མཐར་སྐྱེལ་དགོས་པ་རེད་ཟེར་གྱིན་ཡོད། ༧རྒྱལ་བ་རིན་པོ་ཆེར་དམའ་འབེབས་བྱེད་མཁན་དང་། ༧རྒྱལ་བ་རིན་པོ་ཆེའི་བཀའ་དང་འགལ་ནས་ལས་ཀ་གཞན་པ་བྱེད་རྒྱུ་ཡིན་ཟེར་མཁན་ཚོར་ང་རང་མོས་མཐུན་རྩ་བ་ནས་མེད། ༧སྐྱབས་དགོན་རིན་པོ་ཆེའི་ཆེད་དུ་གང་ལ་མཆོངས་དགོས་ཀྱང་མཆོངས་ཆོག་ཆོག་ཡིན། ལས་ཀ་ག་རེ་ཡིན་ནའང་བྱེད་མ་ཐུབ་པ་དེ་འདྲ་མིན། ཁྱད་མཚར་བོ་དེ་འདྲ་ལ་བཟོ་ལྟ་བཟོ་དགོས་ན་ང་ལ་ཡོད་དཱ། དེ་རིང་ངས་བཀའ་ཟུར་ཀིརྟི་རིན་པོ་ཆེ་མཆོག་དང་ཚོགས་གཙོ་རྣམ་པ་ཚོ་ཡོད་ས་ནས་ཞུ་བཞིན་ཡོད། ཅེས་སོགས་གསུངས་སོང་། དེའི་རྗེས་བཀའ་ཟུར་ཀིརྟི་རིན་པོ་ཆེ་མཆོག་གིས་གསུངས་བཤད་གནང་དོན། བོད་རྒྱལ་ཡོངས་མང་གཙོ་ཚོགས་པ་འདི་བོད་ཀྱི་ཆབ་སྲིད་ཚོགས་པ་དང་པོ་དེ་ཡིན་ཟེར་གྱི་ཡོད་པ་རེད། ཆབ་སྲིད་ཚོགས་པ་དེ་བཙུགས་ནས་ད་ལྟའི་བར་དུས་ཚོད་གང་ལ་གང་འཚམ་ཕྱིན་ཡོད་ཀྱང་ཡར་རྒྱས་ཞེ་དྲགས་འགྲོ་ཐུབ་ཡོད་པ་མ་རེད། ཚོགས་པ་འདི་ཡར་ལངས་ཐུབ་ཆེད་དང་། བོད་མི་ཚོས་ལས་དོན་སྤེལ་ཐུབ་པའི་ཆེད་དུ་ཡར་རྒྱས་གཏང་དགོས་པའི་ཐོག་ལ་ཚང་མས་ཤུགས་རྒྱག་དགོས་པ་གལ་ཆེན་པོ་རེད། ཚོགས་མི་ག་ཚོད་མང་བ་དང་ཚོགས་པ་ག་ཚོད་རྒྱ་ཆེ་རུ་སོང་ན། བོད་གཞུང་དམངས་གཉིས་ལ་ནུས་པ་ཆེན་པོ་འདོན་ཐུབ་ཀྱི་རེད་མ་གཏོགས། ལས་འགུལ་སྤེལ་བ་ཙམ་གྱིས་ནུས་པ་ཞེ་དྲགས་འཐོན་རྒྱུ་ཁག་པོ་རེད། དེར་བརྟེན། གཙོ་ཆེས་ཤོས་དེ་ཚོགས་མི་རྒྱ་ཆེན་པོ་བཟོས་དགོས་པ་དང་། བོད་རྒྱལ་ཡོངས་ཟེར་བའི་མིང་དེ་བཏེགས་ཡོད་སྟབས། བོད་རྒྱལ་ཡོངས་ལ་ཁྱབ་ཐུབ་པའི་མི་མང་འབོར་ཆེན་ལ་བདག་པོ་གནང་དགོས་པ་གལ་ཆེན་པ་རེད། ཅེས་གསུངས་སོང་། རིན་པོ་ཆེ་མཆོག་གིས་མུ་མཐུད་དུ་གསུངས་དོན། ཚོགས་པ་འདི་ལ་དམངས་གཙོ་ཞེས་པའི་མིང་དེ་ཐོགས་བསྡད་ཡོད་དུས། མང་ཚོགས་ལ་གཙོ་བོ་དམངས་གཙོའི་ཤེས་ཡོན་ཁྱབ་སྤེལ་གནང་དགོས་པ་དེ་ལས་འགན་ཆེས་ཤོས་ཆགས་ཡོད། ད་ཆར་ང་ཚོ་ཚང་མས་མཁྱེན་གསལ་ལྟར། སྐབས་བཅོ་བའི་འོས་བསྡུའི་སྐབས་ལ་༧རྒྱལ་བ་རིན་པོ་ཆེ་ཐུགས་ཕམ་གྱི་གནས་ཚུལ་བྱུང་ཡོད་པ་རེད། ཐུགས་ཕམ་གྱི་གནས་ཚུལ་བྱུང་རྐྱེན་གཙོ་བོ་དེ་དམངས་གཙོའི་ཤེས་ཡོན་མེད་པའི་ཐོག་ནས་ཡོང་གི་ཡོད་པ་རེད་མ་གཏོགས། ཆེད་དུ་༧རྒྱལ་བ་རིན་པོ་ཆེའི་ཐུགས་དགོངས་དཀྲུག་པའི་ལས་དོན་བྱས་བཞག་པ་ཧ་ལམ་ཡོད་ཀྱི་མ་རེད། གཙོ་བོ་དེ་དམངས་གཙོའི་ཤེས་ཡོན་མེད་པ་དང་ལས་ཀ་བྱེད་སྟངས་ཐོག་ནས་ནོར་འཁྲུལ་མང་པོ་ཞིག་རང་གཞན་གཉིས་ཐོག་ནས་ཡོང་གི་ཡོད་པ་རེད། དེར་བརྟེན། མ་འོངས་པར་བོད་རྒྱ་ཆེའི་མི་མང་ནང་དམངས་གཙོའི་ཤེས་ཡོན་སྤེལ་བའི་ཐོག་ལ་རྣམ་པ་ཚོས་ཤུགས་བསྐྱོན་པ་ཡིན་ན། མ་འོངས་པའི་འོས་བསྡུའི་སྐབས་སུ་མི་མང་གི་ངོས་ནས་འོས་བསྡུ་ཕྱོགས་སུ་མ་ལྷུང་བ་ཞིག་ཡོང་ཐབས་ལ་བོད་མི་ཚང་མས་འགན་འཁུར་དགོས་པ་རེད་ལ། རྣམ་པ་ཚོས་ངོས་ནས་ཀྱང་ཐབས་ཤེས་གནང་དགོས་པ་རེད། ཁྱེད་རང་ཚོའི་གྲོས་གཞིའི་ནང་དུ་༧རྒྱལ་བ་རིན་པོ་ཆེའི་སྐུ་ཕྱྭའི་སྐོར་ཞིག་འཁོད་འདུག སྐུ་ཕྱྭ་གཙོ་བོ་དེ་༧རྒྱལ་བ་རིན་པོ་ཆེའི་ཐུགས་དགོངས་དྭངས་པ་བྱ་དགོས་ཀྱི་ཡོད་པ་དེ་རེད། དེང་སྐབས་ཁག་ཅིག་གིས་དམངས་གཙོ་ཡིན་ཟེར་ནས་གང་བྱུང་མང་བྱུང་ཤོད་ཀྱི་འདུག དེ་ཚོར་ཚོགས་པ་ཁག་ཡིན་ན་རེད། མི་སྒེར་བ་ཡིན་ན་རེད། དེ་དག་གི་བཤད་སྟངས་སྤྱོད་སྟངས་ལ་དོ་སྣང་སྤྲོད་རྒྱུ་གལ་ཆེན་པོ་རེད། ༧གོང་ས་མཆོག་གིས་བཀའ་བཀྱོན་གནང་བ་རེད་ཟེར་ནས་གང་བྱུང་མིང་བྱུང་བཤད་པ་དང་། གང་བྱུང་མང་བྱུང་འབྲི་བ་དེ་འདྲ་འདུག ༧རྒྱལ་བ་རིན་པོ་ཆེ་ལ་དེ་འདྲ་གསུང་རྒྱུའི་ཐོབ་ཐང་ཆ་ཚང་བོད་མི་མང་གིས་ཕུལ་བཞག་ཡོད་པ་རེད། ད་ལྟའི་ང་ཚོའི་སྲིད་སྐྱོང་ག་འདྲ་ཞིག་ཡིན་ནའང་འོས་ཤོག་ཁྲི་ཁ་ཤས་ཀྱི་སྐད་ཆ་རེད། ༧རྒྱལ་བ་རིན་པོ་ཆེ་ལ་མི་འབུམ་ཕྲག་མང་པོས་ཡིད་ཆེས་དང་བློས་འགེལ་བྱས་ཡོད་པ་རེད། དེ་ལ་འབབ་པའི་འགན་ཞིག་དང་ནུས་པ་ཞིག་ཡོད་པ་རེད། འགན་དེའི་ཐོག་ནས་བཀའ་བཀྱོན་དང་ཚིག་རྩུབ་པོ་གནང་གི་ཡདོ་པ་རེད། ཅེས་གསུངས་སོང་། གསར་འགོད་པ། པདྨ་རྒྱལ།
