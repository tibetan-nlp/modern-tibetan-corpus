ཤ་མོ།
བྱང་བློ་བཟང་ཀུན་མཁྱེན།

༡
“མདང་ནུབ་མཚན་གང་པོར་ཆར་བ་བབས་སོང་བས། ཡང་ཤ་མོ་མང་པོ་ཞིག་སྐྱེས་ཡོད་ངེས། ད་ཡར་ལོངས་ཤ་མོ་སྒྲུག་ཏུ་འགྲོ”
“ཤ་མོ་སྒྲུག་ཏུ་འགྲོ”ནམ་གསལ་པོ་ཞིག་ལངས་མེད་པས།རྒྱང་རིང་གི་ཡུལ་ལྗོངས་ཚང་མ་རབ་རིབ་ཙམ་ལས་མི་མཐོང་ཡང་།ཐར་ཐོར་དུ་གནས་ཡོད་པའི་རི་གྲོང་ཆུང་ཆུང་འདིའི་དུད་ཁྱིམ་དག་གི་སྒོ་ནས་མི་མང་པོ་ཞིག་ཐོན་ཏེ་གྲོང་ཚོའི་རྒྱབ་མདུན་གྱི་སྤང་རི་དག་གི་སྟེང་དུ་ཤོར་འགྲོ།
༢
“ཉེ་ཆར་རྫོང་ཐོག་ཏུ་ཤ་མོ་རྒྱ་མ་གང་རེར་སྒོར་བརྒྱ་ལྷག་སྟེར་གྱི་འདུག”འདི་ནི་ཉིན་ཤས་སྔོན་ཤང་ཐོག་ཚོགས་འདུར་ཕྱིན་པའི་གྲོང་ཚོའི་ཀྲུའུ་རིན་ལགས་ཀྱི་སྐད་ཆ་རེད། ཉིན་འདི་གའི་རིང་རི་གྲོང་ཆུང་ཆུང་འདིར་སྐད་ཆ་དེས་གྲགས་འགུལ་མི་ཆུང་བ་ཞིག་བཞག་སོང་།དེ་ནི་ནམ་རྒྱུན་ཚོགས་འདུའི་དགོངས་དོན་ཞིག་ཁྱབ་བསྒྲགས་བྱས་པ་ལས་ཀྱང་གྲགས་འགུལ་ཆེ་ཞིང་ལག་བསྟར་མཁས་པར་འདུག གྲགས་འགུལ་དེ་དང་བརྒྱུད་ནས་ཡུལ་མི་མང་ཆེ་བས་རང་འགུལ་གྱི་ལས་འགུལ་ཆེན་པོ་ཞིག་ཀྱང་སྤེལ་སོང་། ད་ནི་དུད་ཁྱིམ་མང་ཆེ་བའི་ཁྱིམ་མི་ཚང་མ་ཤ་མོ་སྒྲུག་པའི་ཚ་རླབས་ཤིག་ཕྱུར་ཡོད།གྲོང་ཆུང་འདིའི་ནང་གི་རྒན་པོ་རབས་ཆད་བཀྲ་རྣམ་དང་ཕྱུག་བདག་ཉི་མ་ཚང་མ་གཏོགས་དུད་ཁྱིམ་ཚང་མའི་རྒན་དར་གཞོན་གསུམ་གྱིས་ཤ་མོ་སྒྲུག་གིན་ཡོད། ཤ་མོ་སྒྲུག་རྒྱུ་ནི་ཡུལ་མི་མང་ཆེ་བའི་ལས་འགན་ཡིན། ཤ་མོ་སྒྲུག་རྒྱུ་ནི་ཡུལ་མི་མང་ཆེ་བའི་དགའ་ཕྱོགས་ཀྱང་ཡིན། ཤ་མོ་ལོས་སྒྲུག ཤ་མོ་མང་པོ་སྒྲུག་ཐུབ་ཚེ་དངུལ་ཡོད་ངེས། དངུལ་ཡོད་ཚེ་་་་་་་ཧ་ཧ་་་་གྲོང་མི་རེ་རེའི་གདོང་ལ་འཛུམ་ཆུང་མི་འདྲ་བ་རེ་མངོན་ཡོང་།གྲོང་མི་རེ་རེའི་ཁོག་ལ་འཆར་གཞི་མི་འདྲ་བ་རེ་དྲན་ཡོང་།
༣
གྲོང་ཆུང་འདིའི་མི་ཚོས་ཤ་མོ་སྒྲུག་མགོ་བརྩམས་བྱུང་།ནམ་རྒྱུན་ཕྱུགས་ཟོག་དང་རྫི་བོ་དག་གི་རྐང་རྗེས་ལས་འཇོག་མ་མྱོང་བའི་གྲོང་ཆུང་རྒྱབ་ཀྱི་སྤང་རི་དེར་ད་ནི་ལོ་ལོན་དང་བྱིས་པ་བརྩིས་པའི་གྲོང་མི་ཕལ་ཆེ་བ་སླེབས་འདུག ནམ་རྒྱུན་ཕྱུགས་ཟོག་གི་རྗེས་ནས་གོག་མ་ནུར་གྱིས་ཕྱུགས་རྫི་འགྲོ་འདོད་ལ་མི་འདོད་མཁན་ཚོ་ཡང་ད་ནི་ཕྱུགས་ཟོག་སྔོན་ལ་སླེབས་འདུག ལོས་ཡིན་ཏེ་ཕྱུགས་ཟོག་གི་སྔོན་དུ་ཕྱིན་ཏེ་ཤ་མོ་མ་སྒྲུག་ཚེ་ཕྱུགས་ཟོག་གི་མཇུག་ན་ཕྱུགས་ཟོག་གི་ཁ་ལམ་དུ་ལྷག་པའི་ཤ་མོ་ཞིག་ཕལ་ཆེར་རྙེད་དཀའ།ཉིན་འདི་གའི་རིང་གྲོང་མི་ཚོར་ལས་འགན་གཅིག་ཡོད་པ་ནི་ཤ་མོ་སྒྲུག་རྒྱུ་དེའོ།ཞོགས་རེ་རེ་བཞིན་གྲོང་མི་དག་ཁྱིམ་ནས་དོན་ཏེ་རི་ལ་འགྲོ།དགོང་རེ་རེ་བཞིན་གྲོང་མི་དག་རི་ལས་བབས་ནས་ཁྱིམ་ལ་ལོག ནམ་རྒྱུན་དགོང་མོར་གོ་ཁ་ནས་ཚོད་མ་ཚ་སོབ་སོབ་རེ་རེ་འཐུང་བཞིན་རྒན་ཁོག་དག་གི་སྒྲུང་ལ་ཉན་སྲོལ་ཡོད་ཀྱང་ད་ནི་དགོང་མོ་སླེབས་ཚེ་ཁྱིམ་མི་ཚང་མས་རང་རང་གིས་སྒྲུག་པའི་ཤ་མོ་བརྩིས་ནས་སུས་ཉིན་གཅིག་གི་ཡོང་འབབ་མང་བར་བལྟ།ནམ་རྒྱུན་ལྷིང་འཇགས་དང་འགྲོས་དལ་བའི་གྲོང་མི་ཚོའི་འཚོ་བ་ད་ནི་ཡོངས་སུ་འགྱུར་འདུག ཉིན་མོ་དག་ནི་བྲེལ་འཚུབ་འཚུབ།དབུགས་ཧལ་ཧལ། མཚན་མོ་དག་ནི་མཛུབ་ཀེར་ཀེར།ལབ་ཤིག་ཤིག
༤
ཕྱི་དྲོ་དེར་ཤང་ཐོག་ན་ཚོང་ཙག་ཙིག་གཅིག་རྒྱབ་ནས་འཚོ་ཁ་འཕྱིད་ཐབས་བྱེད་བཞིན་པའི་ཟླ་བ་གྲོང་ཆུང་འདིའི་ནོར་དགའ་ཚང་དུ་སླེབས་སོང་།དེ་དུས་ཉིན་དགུང་ཡོལ་ཙམ་ཞིག་རེད།ཟླ་བས་སྦག་སྦག་རྙིང་སོན་ཞིག་ཡོད་པ་དེ་བཞོན་ཏེ་སྣུམ་སྒོ་ཅུང་གློད་ནས་ཧར་སྒྲ་དང་བཅས་ཐར་ཐོར་དུ་གནས་ཡོད་པའི་དུད་ཁྱིམ་དག་གི་བར་ནས་ཐལ་འཚུབ་ཕྱུར་རེ་གྲོང་སྟོད་ཀྱི་དེའུ་འབུར་སྟེང་དུ་གནས་ཡོད་པའི་ནོར་དགའ་ཚང་གི་ཕྱོགས་སུ་ཕྱིན།ནམ་རྒྱུན་ཡིན་ན་ཁོ་གྲོང་གསེང་ནས་ཧར་སྒྲ་དང་བཅས་འགྲོ་དུས་བྱིས་པ་མང་པོ་ཞིག་ཀྱང་ཁོའི་རྗེས་འབྲངས་ནས་ནོར་དགའ་ཚང་དུ་སླེབས་ཡོང་།དེ་ནི་བྱིས་པ་དག་ཁོར་དགའ་བ་མ་ཡིན་པར་ཁོས་མངར་ཆ་ཏོག་ཙམ་རེ་འཚོང་རྒྱུ་འཁྱེར་ཡོང་བས་ཡིན་ལ།སྐབས་འགར་ཁོ་སེམས་པ་སྐྱིད་དུས་བྱེ་རིལ་རེ་གཉིས་ནོར་དགའ་ཚང་གི་སྒོ་འགྲམ་ནས་རེ་སྙེག་ཀྱི་མིག་ཟུང་ཁོ་ལ་ཅེར་ཡོད་པའི་གྲོང་ཆུང་འདིའི་བྱིས་པ་ཚོར་སྤྲོད་ངེས།འོན་ཀྱང་དེ་རིང་ཁོ་ནོར་དགའ་ཚང་དུ་སླེབས་ཀྱང་བྱིས་པ་གཅིག་ཀྱང་ཁོའི་རྗེས་སུ་འབྲངས་མཁན་མི་འདུག ཟླ་བ་ནོར་དགའ་ཚང་དུ་སླེབས་དུས་ཁྱིམ་དུ་ནོར་དགའ་གཅིག་པུ་ལས་མེད་ཅིང་།ནོར་དགའ་ནི་ཤ་མོ་རས་ཁུག་གསུམ་མཉམ་དུ་བྱོས་ནས་དབྱེ་བ་འབྱེད་བཞིན་འདུག ཟླ་བ་ནང་དུ་འཛུལ་མ་ཐག་ནོར་དགས“ངའི་ཚོང་ཤག་ཡོང་ངམ།་དེ་རིང་ཁྱོད་ཚུར་མ་ཡོང་ན་སང་ཉིན་ང་ཕར་ཁྱོད་ཡོད་སར་འགྲོ་རྩིས་ཡོད།”

ཆར་བ་མུ་མཐུད་འབབ་ཀྱིན་འདུག ད་ལོ་ངོ་མ་ཡ་མཚན་ཞིག་རེད།དབྱར་མཇུག་ཡོལ་ནས་སྟོན་ཁ་སླེབས་ཀྱང་ཤ་མོ་ད་དུང་སྐྱེ་ཀྱིན་འདུག འོན་ཀྱང་ཉིན་འདི་གའི་རིང་ཡུལ་མི་ཚོས་ཤ་མོང་སྒྲུག་མི་ཕོད་པས།ཡང་དབྱར་འགོ་ཚུགས་སྐབས་དང་འདྲ་བར་སྤང་ཡོངས་ཤ་མོ་གིས་ཁེངས་ཡོང་།རྫི་བོ་དག་གིས་རྫི་ལམ་ནས་ཤ་མོ་རེ་མཐོང་ཡང་ལུས་པོ་སྒུར་ནས་སྒྲུག་མི་ཕོད།ཁོ་ཚོའི་སེམས་ངོ་ན་ཤ་མོ་ཞིག་སྒྲུག་ཚེ་རང་ལ་ཆག་སྒོ་ཞིག་སྒྲུག་ཡོང་བའི་སྣང་བ་སྟེར་ཡོང་།རྣམ་རྟོག་ཅན་ལ་འདྲེ་དགུ་འཁོར་ཞེས་པ་ནི་ཅིས་མི་བདེན།ད་ནི་གྲོང་མི་ཚང་མས་ཤ་མོ་ལ་ཡིད་གཏིང་ནས་ཆད་སོང་།དེ་ལས་ལྡོག་སྟེ་ཤ་མོ་ནི་གནོད་ཀྱི་བསྐྱལ་བ་ལྟ་བུ་ཞིག་བརྩི་ཞིང་།ཤ་མོ་ཡོད་སར་འདྲེ་འཁོར་ཡོད་པ་ལྟ་བུའི་འདུ་ཤེས་ཤིག་སྐྱེས་འོང་ལ་འདྲེ་འཁོར་ཡོད་པས་རང་ལ་མི་དགེ་བ་མང་པོ་འབྱུང་ངེས་པར་སྙམ།ད་ལྟ་ཤ་མོ་གསར་དུ་སྐྱེས་པ་དག་མ་བསྒྲུག་ལ་བསྒྲུགས་ཟིན་པ་དག་ཀྱང་གཡོ་འགུལ་མ་བཏང་བར་འཇམ་ཅག་གེར་ཟླ་ཕྱེད་འགོར་སོང་། ད་ནི་དུད་ཁྱིམ་ཚོའི་ཁྱིམ་སྦུག་ཏུ་ཉར་ཡོད་པའི་ཤ་མོ་དག་རུལ་བཞིན་འདུག ཁྱིམ་ཚང་མང་ཆེ་བའི་ནང་འཛུལ་མ་ཐག་རུལ་དྲི་སྣ་ལ་གཟན་ལ་རེ་ཐུལ་ཡོང་། ཡང་གྲོང་མི་ཚོས་རང་འགུལ་ངང་ལས་འགུལ་ཞིག་སྤེལ་བཞིན་འདུག དེ་ནི་དེ་སྔོན་ནས་བསྒྲུགས་ཟིན་པའི་ཤ་མོ་དག་ཕྱི་རུ་འདོར་རྒྱུ་དེ་རེད།ཉིན་འདི་དག་གི་རིང་གྲོང་ཚོའི་རྒྱབ་མདུན་གྱི་སྤང་རི་དག་གི་སྟེང་ཤ་མོ་ཕོན་པོ་ཕོན་པོ་བྱོས་ཡོད་པ་མཐོང་ཐུབ་ལ།གྲོང་ཚོའི་ཉེ་རྒྱུད་ཀྱི་ཀོང་དོང་དང་ཐལ་སྤུང་འགྲམ་ལ་ཤ་མོས་གང་འདུག ཤ་མོ་བསྒྲུགས་སོང་།ཤ་མོ་གཡུགས་སོང་།སྔར་བསྒྲུགས་པ་དག་ད་ལྟ་ཕྱིར་གཡུགས་བཞིན་འདུག འོན་ཀྱང་ནོར་དགའ་ཁོ་ནས་ཕྱི་ཚུལ་སྟོན་བྱེད་དུ་ཤ་མོང་སྒྱེ་མོ་བཞི་གཡུགས་པ་ལས་ཁྱིམ་སྦུག་ཏུ་སྒྱེ་མོ་དོ་ཉར་འདུག དབྱར་དུས་གཅིག་རིང་ཚ་ཚ་འུར་འུར་གྱིས་སྒྲུག་པའི་ཤ་མོ་དག་ཉིན་གཅིག་ཀྱང་མགོར་བར་ཐག་གཅོད་བྱས་ཚར་སོང་།ཤ་མོ་དག་དབྱུགས་ཚར་རྗེས་གྲོང་མི་ཚོས་སེམས་ནད་ཀྱང་རིམ་བཞིན་ཡལ་འགྲོ ཡང་སྔར་རྒྱུན་བཞིན་ཞི་བབ་ལྷོད་འཇགས་ཀྱི་འཚོ་བ་ཞིག་སྐྱེལ་འགོ་བརྩམས་བྱུང་།ད་ནི་ཤ་མོ་སྒྲུག་མཁན་མེད་ལ་ཤ་མོ་བསྒྲུགས་ཟིན་པ་རྣམས་ཀྱང་གཙང་ཕྱགས་བྱས་ཟིན་པས་གྲོང་མི་རྒན་གྲས་དག་སེམས་པ་བདེ་ལ་བབས་འདུག འོན་ཀྱང་།བྱིས་པ་དག་གིས་མི་རྒན་ཚོས་མི་མཐོང་སར་ཤ་མོ་སྒྲུག་ནས་ཟ་བཞིན་འདུག གོ་ཐོས་སུ་གསར་བུ་ཁ་ཤས་ཤིག་གིས་ཀྱང་རྫི་ཞོར་དུ་རི་མགོར་ཤ་མོ་བསྲེགས་ནས་ཟ་གི་འདུག་ཟེར། དེ་དག་ནི་བཙོངས་རྒྱུའི་ཆེད་དུ་མིན་པར་ཉུང་ཉུང་གི་ཁ་ཟས་ཙམ་ཡིན་པས། སྔར་ལྟ་བུའི་གྲགས་འགུལ་མེད་ལ་རྙོག་དྲ་ཅི་ཡང་མེད།
