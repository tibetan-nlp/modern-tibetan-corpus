ལྷ་བྱམས་རྒྱལ་གྱི་ལྷུག་རྩོམ་གསར་པ། ཕྱིར་ལོག་མི་ཐུབ་པའི་མི།
ལྷ་བྱམས་རྒྱལ།


ཕ་ཡུལ་དང་གྱེས་ནས་ལོ་ངོ་ཉི་ཤུ་འགོར་ལ་ཉེ་བའི་སྐབས་འདིར། ངས་གཞི་ནས་རང་ཉིད་ཕྱིར་ལོག་ཐུབ་མིན་གྱི་སྐོར་ལ་གཟབ་ནན་གྱིས་བསམ་བློ་གཏོང་དགོས་བྱུང་།
བོད་ཀྱི་གཏམ་དཔེ་ཞིག་ཏུ“བྱ་རྒས་ཚང་དང་མི་རྒས་ཡུལ”ཟེར་སྲོལ་ཞིག་ཡོད། ད་ལྟའི་ང་ནི་རྒས་ནས་ཚད་དེ་འདྲ་ཞིག་ཏུ་ལྷུང་མེད་ཀྱང་། མཛུབ་མོ་ནང་དུ་བཀུག་སྟེ་རྩིས་ཤིག་བརྒྱབ་ན་ད་ནི་བྱེས་ནས་འཚོ་བ་རོལ་བའི་ལོ་ཟླའི་གྲངས་ཀ་དེ་ཕ་ཡུལ་ནས་འཚར་ལོངས་བྱུང་བའི་དུས་ཚོད་ལས་རིང་དུ་བརྒལ་བཞིན་ཡོད་པས། མཚམས་རེར་རྒྱུ་མཚན་ཅི་ཡང་མེད་པར་རང་ཉིད་རིམ་གྱིས་ཕ་ཡུལ་དང་བར་ཐག་རིང་ནས་རིང་དུ་འགྲོ་བཞིན་པའི་ཚོར་བ་ཞིག་ངང་གིས་འབྱུང་། ཚོར་བ་དེ་ནི་སྙིང་ལ་ཟུག་གཟེར་བསྐྱེད་པའི་ཚོར་བ་ཞིག་སྟེ། དེ་ནི་རང་དང་ཤིན་ཏུ་འདྲིས་ཤིང་སྙིང་ཉེ་བའི་མི་གང་ཞིག་ལ་ཁྲེལ་མེད་ཀྱིས་གྱེས་ནས་རིང་དུ་འགྲོ་བ་དང་མཚུངས། སྤྱིར་ང་ནི་ཡུལ་གདུང་ཆེ་བ་ཞིག་ཡིན། མི་ལོ་སུམ་ཅུ་ལྷག་གི་གོང་རོལ་ཏུ་དཔལ་དོན་གྲུབ་རྒྱལ་མཆོག་གིས་སྙན་ཚིག་ཅིག་ཏུ“ང་ནི་ལྷ་ཡུལ་ལྟ་བུའི་རྒྱལ་ས་ན་གནས་ཀྱང་།། ཕ་ཡུལ་དྲན་པའི་འདུན་པ་འགོག་ཐབས་རང་མི་འདུག”ཞེས་གསུངས་བ་བཞིན། པེ་ཅིན་དུ་འཚོ་བའི་དུས་ཚོད་ཀྱི་གོ་རིམ་གང་དུ་ཡང་རང་ཉིད་ཀྱི་སྐྱེས་ས་སྐྱེས་ཡུལ་དེ་ཡིད་དུ་ཡང་ཡང་འཁོར་ཡོང་ལ། ལོ་རེར་གོ་སྐབས་དང་ཁ་གཡོར་ས་གང་རུང་རེ་བཙལ་ནས་ཐེངས་ཁ་ཤས་ལ་ཅིས་ཀྱང་ལོག་དགོས། དེ་ལྟར་ཐེངས་རེ་གཉིས་ལ་ཕྱིར་ལོག་ནས་དུས་ཡུན་ཐུང་ཐུང་གི་རིང་ལ་ཕ་ཡུལ་དུ་འཁྱུག་ཙམ་བྱས་པ་ནི་གནས་སྐབས་ཀྱི་ཟུག་འཇོམས་སྨན་རྫས་དང་མཚུངས་པར་རེ་ཞིག་ལ་སེམས་ཀྱི་དྲན་གདུང་དེ་སེལ་བར་བྱེད་ཐུབ། ཡིན་ན་ཡང་། བྱེས་ནས་འཚོ་བ་རོལ་བའི་དུས་ཚོད་རིང་ནས་རིང་དུ་འགྲོ་བ་དང་བསྟུན་ནས་ཕྱིར་ལོག་མི་ཐུབ་པའི་སེམས་ནད་ཅིག་སྐྲན་ནད་བཞིན་སེམས་ཀྱི་བར་གསེང་ཞིག་ནས་མངོན་ཡོང་། སྐབས་དེ་དག་ཏུ། ངས་གཞི་ནས་མི་ཚེ་འདིར་ནམ་ཞིག་ལ་རང་ཉིད་དུས་གཏན་དུ་ཕ་ཡུལ་དུ་ཕྱིར་ལོག་ཐུབ་མིན་སྐོར་ལ་གཟབ་ནན་གྱིས་བསམ་བློ་གཏོང་བཞིན་ཡོད།
ཉིན་ཁ་ཤས་ཀྱི་སྔོན་ལ། ངས་ཅེ་ཁེའི་རྩོམ་པ་པོ་གྲགས་ཅན་རྨི་ལན་ཁུང་ངྷི་ར་（Milan Kundera）ཡིས་༢༠༠༤ལོར་བརྩམས་པའི་སྒྲུང་རིང«ཤེས་རྨོངས»（L’ignorance）ཞེས་པ་དེར་ལྷོད་ལྷོད་ཀྱིས་བལྟས་པ་ཡིན། འདི་ནི་ངས་དེབ་འདིར་ཐེངས་གཉིས་པར་བལྟས་པ་དེ་ཡིན། ཐེངས་དང་པོར་ལྟ་ཀློག་བྱེད་དུས་ང་རང་སློབ་ཆེན་གྱི་སློབ་མ་ཞིག་ཡིན་ཞིང་། དེ་དུས་པེ་ཅིན་དུ་ཡོང་ནས་ལོ་ཁ་ཤས་ལས་འགོར་མེད། ལོ་བཅུ་བཞིའི་རྗེས་ནས་ཡང་བསྐྱར་དེབ་འདིར་ལྟ་སྐབས་ངས་གཞི་ནས་རང་གིས་སྐབས་དེར་དེབ་འདིའི་ནང་དོན་ལ་གོ་བ་ཅི་ཡང་ལོན་མེད་པ་རྟོགས། སྒྲུང་གཏམ་འདིའི་ནང་དུ་རྩོམ་པ་པོས་གྱེས་བྲལ་དང་ཕྱིར་ལོག་པའི་གཏམ་རྒྱུད་ཅིག་བྲིས་ཡོད་ཅིང་། མཇུག་མཐར་ཀློག་མཁན་གྱི་སེམས་སུ་ཕྱིར་ཕ་ཡུལ་དུ་ལོག་མི་ཐུབ་པའི་མི་རེ་གཉིས་ཀྱི་སྣང་བརྙན་གསར་བཞེངས་བྱས་འདུག རྩོམ་མཁན་གྱིས་ཡུལ་དུས་གཉིས་ཆ་ནས་ཕ་ཡུལ་དང་རིང་དུ་གྱེས་ཏེ་ཕྱིར་ལོག་མི་ཐུབ་པའི་མི་དེ་རིགས་ལ་ཡུལ་གྱར་བ་ཞེས་བོས་འདུག་ཅིང་། རྩོམ་ནང་དུ “ཡུལ་གྱར་བའི་རྨི་ལམ་ནི་དུས་རབས་ཉི་ཤུ་བའི་དུས་མཇུག་གི་ཆེས་ངོ་མཚར་བའི་སྣང་ཚུལ་ཞིག་རེད”ཅེས་གསུངས་འདུག ད་ལྟ། ངས་རང་གིས་ལོ་གང་མང་གི་སྔོན་དུ་སྒྲུང་གཏམ་འདི་ལ་གོ་བ་མ་ལོན་པར་གུ་ཡངས་སྟེར་ཐུབ། དེ་དུས་ཀྱི་ང་ནི་ད་དུང་ཡུལ་གྱར་བ་ཞིག་ཏུ་གྱུར་མེད་པས། ཡུལ་གྱར་བ་བྲིས་པའི་སྒྲུང་གཏམ་དེར་གོ་བ་ཇི་ལྟར་ལོན། མི་ལོ་གང་མང་ཧྲིལ་གྱིས་ཕྱིན་སོང་། ད་ལྟ་རང་ཉིད་ཡུལ་གྱར་བ་ཞིག་ཡིན་པའི་ཚོར་བ་ཡང་ཇེ་གསལ་ཏུ་གྱུར་ཅིང་རྩོམ་མཁན་འདིའི་སྒྲུང་གཏམ་འདི་ནི་རང་གི་འཚོ་བའི་དགའ་སྐྱོ་ཤར་པའི་མེ་ལོང་ལྟ་བུ་ཞིག་ཏུ་གྱུར་འདུག་པའི་ལན་ཆ་བྱུང་བས། གོ་བ་ཟབ་མོ་ལོན་སོང་ཞེས་བཤད་ཆོག་པ་འདྲ། ཡུལ་གྱར་བ་ནི་ཕྱིར་ལོག་མི་ཐུབ་པའི་མི་སྟེ། རང་གི་སྐྱེས་ས་སྐྱེས་ཡུལ་དུ་ཕྱིར་ལོག་མི་ཐུབ་པར་མི་ཚེ་བྱེས་ནས་སྐྱེལ་དགོས་པའི་མི་དེ་རིགས་ཡིན་ཞིང་། དེ་ནི་ཤ་ཁྲག་གདོས་བཅོས་ཀྱི་གཟུགས་པོ་འདི་ཕྱིར་ལོག་མི་ཐུབ་པ་ཁོ་ན་བཤད་བཞིན་མེད་པར། དེ་བས་ཀྱང་སེམས་ཟེར་བའི་དགའ་སྐྱོ་ཆགས་སྡང་གི་བདག་ཉིད་ཅན་འདི་ཕྱིར་ལོག་མི་ཐུབ་པ་དེ་འདྲ་ཞིག་ཡིན་སྲིད།
ངས་བསམས་ན། “ཡུལ་གྱར་བའི་རྨི་ལམ”དེ་དུས་རབས་ཉེར་གཅིག་པའི་ནང་དུ་ད་དུང་རྨིས་མཁན་ཡོད་སྲིད་ཅིང་། སྔོན་དང་བསྡུར་ན་མང་དུ་འགྲོ་བཞིན་ཡོད་ལས་ཆེ། འགྲོ་བ་མིའི་རིགས་གོ་ལ་རིལ་མོ་འདིའི་སྒང་ནས་བདེ་སྐྱིད་འཚོལ་བའི་ལམ་དུ་ཆས་པ་ནས་བཟུང་། འཕེལ་འགྱུར་གྱི་གོ་རིམ་གང་དུ་ཡང་ལམ་འགྲོ་བ་ཞིག་གི་ངང་ཚུལ་བཟུང་ཡོད་ཅིང་། དེ་ཡང་འཚོལ་སྙེག་དང་རྒྱང་སྐྱོད། གསར་གཏོད་སོགས་ཀྱི་ལས་ཡང་ཡང་བརྩམས་ཏེ་སོར་ན་གནས་པར་ཡིད་མི་ཚིམ་པར་གྱུར་ཡོད། ལུས་དང་སེམས་གང་ཡིན་ཡང་གང་ཞིག་དང་བྲལ་ཏེ་ཅི་ཞིག་ལ་སྙེག་པར་འདོད་པ་ནི་མིའི་རིགས་ཀྱི་རྩ་ལམ་དུ་རྒྱུ་བའི་ཟུངས་ཁྲག་གི་ཕྲ་ཕུང་རེར་སིམ་པའི་གཞི་རྒྱུ་དེ་ཡིན་སྲིད། དེ་བས། ལོ་རྒྱུས་སུ་རང་གི་སྐྱེས་ས་སྐྱེས་ཡུལ་དང་བྲལ་ནས་བྱེས་སུ་འཁྱམ་ཅིང་ལྡོམ་མཁན་ནི་བགྲང་མེད་ཅིག་བྱུང་ཡོད་ལས་ཆེ་ཞིང་། མི་མང་པོ་ཞིག་ཡུལ་གྱར་བར་གྱུར་འགྲོ་བ་འདི་ཡང་ང་ཚོ་མི་རང་རྒྱུད་ཀྱི་ཁྱད་གཤིས་ཤིག་ཡིན་ངེས། ཡིན་ན་ཡང་། གནའ་བོའི་དུས་ཀྱི་སྲིད་འབྱོར་རིག་གསུམ་སོགས་ཀྱི་སྤྱི་ཚོགས་ཀྱི་གནས་བབ་ད་ལྟ་དང་བསྡུར་ན་ཧེ་བག་ཡོད་ལ། ལྷག་པར་དུ་འགྲིམ་འགྲུལ་སོགས་སྟབས་བདེ་བོ་མེད་རྐྱེན། ཉེ་རབས་ཚུན་ཆད་ནང་བཞིན་གྲངས་འབོར་ཆེན་པོའི་ཕ་ཡུལ་དང་བྲལ་མཁན་གྱི་དཔུང་ཚོགས་ཤིག་གྲུབ་མེད་སྲིད། གཞི་རྒྱ་ཆེ་བའི་ཡུལ་གྱར་བའི་དཔུང་ཚོགས་ནི་ཉེ་རབས་ནས་འགོ་བརྩམས་ཏེ་གྲུབ་འགོ་ཚུགས་ཤིང་། དུས་རབས་གཅིག་ལྷག་གི་དུས་ཚོད་ནང་དུ་རང་གི་སྐྱེས་ས་སྐྱེས་ཡུལ་དང་བྲལ་ནས་གཞན་ཡུལ་དུ་འཚོ་བ་རོལ་མཁན་མང་དུ་འགྲོ་ཞིང་། འདི་ཡང་འགྲོ་བ་མིའི་རིགས་ཀྱི་སྣང་ཚུལ་ཁྱད་མཚར་བོ་ཞིག་ཏུ་ཆགས་བཞིན་ཡོད་པ་རེད། འདི་འདྲའི་རྒྱབ་ལྗོངས་འོག་ཏུ། ངས་རང་ཉིད་ཀྱང་ཤེས་མེད་ཚོར་མེད་དུ་རང་ཡུལ་བསྐྱུར་ཏེ་གཞན་ཡུལ་དུ་འཚོ་བ་རོལ་མཁན་ཞིག་ཏུ་གྱུར་བ་རྟོགས་ཤིང་། རྣམ་ཀུན་རང་གིས་རང་ལ་གོང་གསལ་གྱི་རྒྱབ་ལྗོངས་ཆེན་པོ་དེ་གནས་ལུགས་སུ་བགྲངས་བཞིན་སྣང་བ་གསོ་བའི་གླུ་རེ་བླངས་ནས་བསྡད་པ་ཡིན།
ཡུལ་གྱར་བ་རྣམས་ཀྱི་སེམས་སུ་དུས་ནམ་ཡང་རྨི་ལམ་ཞིག་སྐུང་ཡོད་སྲིད་དེ། དེ་ནི་རང་ཉིད་དུས་ནམ་ཞིག་ལ་ཕ་ཡུལ་དུ་ལོག་པའི་རྣམ་པ་དེ་ཡིན། ཡིན་ན་ཡང་རྨི་ལམ་དུ་ཡང་ཡང་ཤར་བའི་ཕ་ཡུལ་ནི་མངོན་སུམ་ཁྲོད་དུས་ནམ་ཡང་སླེབས་མི་སྲིད་པའི་རྒྱང་རིང་དུ་གྱུར་འགྲོ་སྟེ། དེ་ནི་ཡུལ་གྱར་བ་དག་གིས་ཅིས་ཀྱང་གདོང་ལེན་བྱེད་དགོས་པའི་སྐྱོ་གར་དེ་ཡིན། པེ་ཅིན་དུ་སྡོད་ཡུན་རིང་བ་དང་བསྟུན་ནས། ལུས་སེམས་གཉིས་པོ་གྲོང་ཁྱེར་ཆེན་པོ་འདིའི་སྣོད་བཅུད་ལ་རིམ་གྱིས་གོམས་ཤིང་འདྲིས་པར་གྱུར་པ་དང་། འདི་གའི་སྲང་ལམ་དང་། གླིང་ག མི་ཚོགས། མཚན་ལྗོངས། མཁའ་རླུང་། དེ་མིན་ཟས་རིགས་སོགས་གཟུགས་ཅན་གྱི་དངོས་པོ་དེ་དག་དང་། གཞན་འདི་གའི་སྒྲིག་ལམ་དང་། འཚོ་བའི་བྲེལ་འཚུབ། བྲེལ་འཚུབ་རྗེས་ཀྱི་ངལ་དུབ། ཐ་ན་ཁོར་ཡུག་གཅིག་གིས་བསྐྱེད་པའི་མི་ཚོགས་ཀྱི་གཤིས་སུ་གྲུབ་པའི་འདྲ་མཚུངས་ལྟ་བུའི་འགྲོ་ལུགས་འདུག་ལུགས་སོགས་གཟུགས་འདས་དེ་དག་ལ། མི་ལོ་གང་མང་གི་གཙུབ་བརྡར་དང་འགོག་རྒོལ་གྱི་རྗེས་ནས་རིམ་གྱིས་དང་ལེན་དང་དེ་རང་ལ་གོམས་ཤིང་འདྲིས་པར་གྱུར། མི་ནི་ཅི་འདྲའི་ཁྱད་མཚར་བའི་སྲོག་ཆགས་ཤིག་རེད་ཨང་། ཡང་ཅི་འདྲའི་འགལ་ཟླའི་ཕུང་བོར་གྲུབ་པ་ཞིག་རེད་ཨང་། ངའི་མི་ཚེའི་སྟོད་ཀྱི་འཚོ་བ་རོལ་སའི་ཕ་ཡུལ་ཏེ། སོས་ཀའི་ལྗང་མྱུག་སྔོ་རིག་རིག་ཏུ་འབུས་པའི་ཞིང་གྲོང་གི་ཁོ་ར་ཁོར་ཡུག་དང་། རི་ར་བ་ཁང་ཁྱིམ། དེ་མིན་དབྱར་གྱི་ཁུ་བྱུག་ལྷང་ལྷང་གྲགས་པའི་སྤྲོ་སྣང་། བསིལ་རླུང་གི་ཁྲོད་དུ་གྲོ་ཞིང་ལ་ལྗང་མདོག་གི་རླབས་རིས་སྐྱོམ་ཞིང་སྐྱེས་དངོས་ཀྱི་དྲི་མ་ཐུ་ལུ་ལུ་བྱེད་པའི་མཁའ་རླུང་། ད་དུང་སྟོན་གྱི་མཐོན་མཐིང་གི་ནམ་མཁའ་དང་། ནོར་གྱི་ངུར་སྒྲ་དང་རྟ་ཡི་འཚེར་སྐད། ལུག་གི་འབའ་སྒྲ། ཁྱི་ཡི་ཟུག་སྐད། བྱ་ཕོའི་གྲགས་སྐད་སོགས། དགུན་གྱི་ཁ་བས་གཡོགས་པའི་དཀར་བོང་བོང་གི་འཇིག་རྟེན། དེ་མིན་ཡུལ་མི་དག་སྒྱིད་ལུག་ངང་ག་ལེར་ལམ་དུ་བསྐྱོད་པའི་ལྷོད་ཉམས་དང་། ལམ་ནས་འཕྲད་སྐབས་ཀྱི་འཚམས་འདྲི། ཁེངས་པ་ཆུང་བའི་འཛུམ་མདངས་དང་། བསང་དུད་ཀྱི་དྲི་མ་སོགས་མདོར་ན་ཕྱིར་དྲན་དུ་ལུས་པ་དེ་དག དུས་ལ་ངེས་བ་མེད་པར་ང་རང་གོམས་ཤིང་འདྲིས་པར་གྱུར་བའི་པེ་ཅིན་གྱི་འཚོ་བའི་ནང་དུ་འཚང་ཀ་བརྒྱབ་ཡོང་སྟེ། ངས་རྣམ་ཀུན་མིག་མདུན་གྱི་འཚོ་བའི་སྣོད་བཅུད་ལས་བྱུང་བའི་སུན་སྣང་དང་ཡི་ཆད་ཀྱི་སེམས་པ་དེ། ཕ་ཡུལ་དྲན་པའི་ཕྱིར་དྲན་གྱི་ཀློང་དུ་ལོག་པར་བྱས་ནས། དེ་ནས་རང་ལ་སེམས་གསོ་ཞིག་དང་དབུགས་དབྱུང་ཞིག་ཐོབ་པར་བྱེད་བཞིན་ཡོད། འདི་ལྟར་མི་གཅིག་ལ་ངོས་གཉིས་ཆགས་པའི་གནས་སྟངས་འདི་ནི་ཡུལ་གྱར་བ་ཀུན་ལ་ཐུན་མོང་གིས་ཡོད་པའི་ཁྱད་ཆོས་ཤིག་ཡིན་སྲིད། ང་རང་པེ་ཅིན་ལ་གོམས་ཤིང་འདྲིས་པའི་འཚོ་བ་རོལ་མཁན་ཞིག་དང་ཕ་ཡུལ་ལ་འཁྲེང་ཞིང་གདུང་པའི་ཡུལ་གྱར་བ་ཞིག་སྟེ། ངོས་གཉིས་ཅན་གྱི་མི་ཞིག་ཏུ་གྱུར་ཅིང་། མིག་སྔའི་འཚོ་བ་རོལ་སའི་གྲོང་ཁྱེར་འདི་ནི་ངས་ཅིས་ཀྱང་གདོང་གཏད་ནས་དང་ལེན་བྱེད་དགོས་པ་ཞིག་དང་། རྒྱང་རིང་གི་སྐྱེས་ས་སྐྱེས་ཡུལ་དེ་ནི་ངའི་སེམས་ཀྱི་ཟབ་ས་ཞིག་ཏུ་ཉར་ཡོད་པའི་ངལ་གསོ་ས་ཞིག་ཏུ་གྱུར་ཡོད། ཡང་པེ་ཅིན་ནི་ད་ལྟ་དང་མ་འོངས་བར་གྱུར་ཡོད་ལ། ཕ་ཡུལ་ནི་འདས་པ་དང་རྒྱང་རིང་དུ་གྱུར་ཡོད་པར་མ་ཟད། པེ་ཅིན་ནི་མདུན་ཕྱོགས་དང་། ཕ་ཡུལ་ནི་རྒྱབ་ཕྱོགས་སུ་གྱུར་ཡོད། འདི་ལྟ་བུའི་འགལ་འདུ་ལས་ང་ཡང་ཡུལ་གྱར་བ་ཕལ་མོ་ཆེ་དང་འདྲ་བར་ཕྱོགས་གཉིས་ཀའི་གང་ལ་ཡང་མི་གཏོགས་པའི་ཕུང་གསུམ་པ་ཞིག་ཏུ་གྱུར་ཡོད་ལ། གནས་མལ་དང་ལོག་མལ་གང་དུ་ཡང་ཁྱིམ་གྱི་འདུ་ཤེས་རྙེད་དཀའ་བར་གྱུར་ཅིང་། མཐའ་མར་ནམ་མཁའི་སྤྲིན་པ་བཞིན་འཕྱོ་ཞིང་སྐྱོད་པའི་སྣང་བ་ཞིག་གིས་ཡོངས་སུ་བཟུང་ཡོད།
ཡིན་ན་ཡང་། དོན་དངོས་སུ་ངས་ད་ལྟའི་ཡོད་ཚད་བློས་བཏང་ནས་ཕྱིར་ལོག་ཐུབ་བམ། ཡང་ན་ཕྱིར་ལོག་འདོད་ཡོད་དམ། ངས་ཕ་མ་སྤུན་མཆེད་ལ་བཅངས་པའི་རུས་གདུང་དང་། སྐྱེས་ས་སྐྱེས་ཡུལ་ལ་བཅངས་པའི་ཡུལ་གདུང་། གངས་རིའི་མི་རིགས་ལ་བཅངས་པའི་རིགས་གདུང་བཅས་བསྡེབས་གཅིག་ཏུ་སྤུངས་རྗེས། དེའི་ལྗིད་ཚད་ཀྱིས་ངའི་འཕྱོ་ཞིང་བསྐྱོད་པའི་ཚེ་སྲོག་དེ་མ་མཐའ་ཡང་ཡུལ་ཁམས་ཀྱི་ངོས་ནས་དེ་དག་གི་ཉེ་འགྲམ་ཞིག་ཏུ་སླར་ཡང་འགུགས་ནུས་མིན་གྱི་སྐོར་ལ། ད་ལྟའི་བར་དུ་གཟབ་ནན་གྱིས་བསམ་བློ་བཏང་མྱོང་མེད་པ་འདྲ། མི་ལོ་ཉི་ཤུར་ཉེ་བའི་བྱེས་ཀྱི་འཚོ་བའི་ནང་དུ། གོང་གི་དྲི་བ་དེ་དག་ལ་ང་རང་ཅི་ནུས་ཀྱིས་གཡོལ་ཅི་ཐུབ་བྱེད་བཞིན་ཡོད། ངས་དྲི་བ་དེ་དག་ལ་གདོང་གཏད་འདོད་མི་འདུག དྲི་བ་དེ་དག་ནི་ལྷམ་མཐིལ་དུ་ལུས་པའི་རྡོ་ཧྲུག་ཅིག་དང་མཚུངས་པར་ང་ལམ་དུ་བསྐྱོད་དུས་ན་ཟུག་བཏང་ཡོང་། འདི་ལྟར་འགལ་འདུའི་ཁྲོད་ནས་ལོ་དང་ཟླ་བ་གཅིག་ཐོག་གཉིས་བརྩེགས་བྱས་ཤིང་། དུས་ཚོད་ཀྱི་ཆག་གྲུམ་མང་ནས་མང་དུ་སྤུངས་རྗེས་ང་རང་རིམ་གྱིས་ལྷོད་ལ་བབས་པའམ་ཡང་ན་མ་མཐའ་ཡང་ཕྱི་ངོས་ནས་ལྷོད་ཉམས་ཤིག་རྒྱས་ཡོད། ད་ལྟ། ངས་རང་ལ་ཕྱིར་ལོག་འདོད་ཡོད་མེད་འདྲི་བ་དེར་དོན་སྙིང་མེད་པ་རྟོགས་ཤིང་། དེ་བས་ཀྱང་ཕྱིར་ལོག་ཐུབ་མིན་ལ་བསམ་བློ་གཏོང་དགོས་བྱུང་། འདི་ཡང་ང་ལྟ་བུའི་ཡུལ་གྱར་བ་དག་གིས་མི་ལོ་གང་མང་གི་རྗེས་སུ་ཅིས་ཀྱང་གདོང་གཏད་དགོས་པའི་གནད་དོན་ཆེ་ཤོས་དེ་ཡིན་སྲིད།
«ཤེས་རྨོངས»སུ་རྨི་ལན་ཁུང་ངྷི་ར་ཡིས་བློ་སེམས་ཞིབ་མོས་བཞེངས་སྐྲུན་བྱས་པའི་མི་སྣ་གཙོ་བོ་དབྱི་ལན་ནྰ་མི་ལོ་ཉི་ཤུའི་རྗེས་ནས་ཕ་ཡུལ་ཅེ་ཁེར་ལོག ཅེ་ཁེ་ནི་ཉེ་རབས་ལོ་རྒྱུས་སུ་གོད་ཆག་མང་བའི་རྒྱལ་ཁབ་ཅིག་ཡིན་ཏེ། སྤྱི་ལོ་༡༩༣༨ལོར་ཆིངས་ཡིག་གཅིག་གིས་འཇར་མན་གྱི་དབང་འོག་ཏུ་ཚུད་ཅིང་། སྤྱི་ལོ་༡༩༤༥ལོར་སུའུ་ལེན་གྱི་རོགས་སྐྱོར་འོག་སྒེར་ཚུགས་ཐོབ་ཀྱང་། ལོ་གསུམ་གྱི་རྗེས་སུ་སྔ་མའི་མ་རྩ་རིང་ལུགས་ཀྱི་ལམ་ལུགས་བསྒྱུར་ནས་གུང་ཁྲན་རིང་ལུགས་ཀྱི་སྲིད་དབང་བཙུགས། དེ་ནས་ལོ་སུམ་ཅུ་ལྷག་གི་རྗེས་ཏེ་སྤྱི་ལོ་༡༩༨༩ལོར་སུའུ་ལེན་གྱི་ཤུགས་རྐྱེན་འོག་ནས་ཐར་ཏེ་མ་རྩ་རིང་ལུགས་ཀྱི་ལམ་ལུགས་སླར་གསོ་བྱས། སྒྲུང་གཏམ་དུ་དབྱི་ལན་ནྰ་ནི་གུང་ཁྲན་རིང་ལུགས་ཀྱི་ལམ་ལུགས་ལག་བསྟར་བྱས་བའི་ཐོག་མའི་ལོ་ཟླ་དེའི་ནང་དུ་རང་གི་མེས་རྒྱལ་དང་ཁ་བྲལ་ནས་ཧྥ་རན་སི་ལ་ཡུལ་བྱོལ་དུ་ཕྱིན་པ་ཞིག་ཡིན་ལ། ཅེ་ཁེ་སུའུ་ལེན་གྱི་དབང་ཤུགས་འོག་ནས་ཐར་བའི་ལོ་རེ་གཉིས་ཀྱི་རྗེས་ནས་ཕྱིར་ཕ་ཡུལ་དུ་ལོག་མཁན་ཞིག་ཡིན། རྩོམ་པ་པོས་མི་སྣ་གཙོ་བོ་ལོ་ངོ་ཉི་ཤུ་ལྷག་གི་རྗེས་ནས་ཕྱིར་ལོག་ཚུལ་དེ་བྲི་དུས། དབྱི་ལན་ནྰ་དང་རང་གི་ཆུང་དུས་ཀྱི་གྲོགས་མོ་ཁ་ཤས་དགྱེས་འཛོམས་བྱས་པའི་རྣམ་པ་ཞིག་བྲིས་ཡོད་པ་འདི་ལྟ་སྟེ། “ཐོག་མར། ཁོ་མོ་ཅག་གིས་མོ་རང་གི་ཕྱི་རྒྱལ་གྱི་འཚོ་བའི་བརྒྱུད་རིམ་དེའི་སྐོར་ལ་ཁ་བརྡ་ཅི་ཡང་མི་བྱེད་ཅིང་། མོ་རང་གི་ལོ་ངོ་ཉི་ཤུ་ལྷག་གི་འཚོ་བ་དེ་དེ་ལྟར་གཏུབས་ནས་མེད་པར་བྱས་སོང་། སྐབས་འདིར། ཁོ་མོ་ཅག་གིས་འདི་ལྟ་བུའི་ཚ་འདྲི་ཞིག་བརྒྱུད་ནས་རང་གིས་གནའ་སྔ་མོ་ཞིག་ཏུ་བསྐྱུར་ཡོད་པའི་འདས་པ་དེ་དང་ད་ལྟའི་འཚོ་བ་གཉིས་ཟུང་འབྲེལ་བྱ་རྩིས་བྱེད། དེ་ནི་ལག་པའི་གྲུ་མོ་བཅད་ནས་ལག་མགོ་ཐད་ཀར་ལག་པའི་དཔུང་བར་སྦྱོར་བའམ་ཡང་ན་རྐང་པའི་བྱིན་པ་བཅད་ནས་རྐང་མགོ་ཐད་ཀར་དུ་པུས་མོར་སྦྱོར་རྩིས་བྱེད་པ་དང་མཚུངས”ཞེས་བྲིས་ཡོད། དཔེ་འཇོག་འདི་ལས་རྩོམ་པ་པོས་མི་སྣ་གཙོ་བོ་དབྱི་ལན་ནྰ་ད་ནི་རང་གི་མེས་རྒྱལ་དུ་ཕྱིར་ལོག་མི་ཐུབ་པའི་དཀའ་ཚེགས་དེ་མངོན་བསྒྲགས་བྱས་འདུག དོན་དམ་དུ། བྱེ་བྲག་གི་དོན་དག་ལ་ཁྱད་པར་ཡོད་པ་མ་གཏོགས། ཕ་ཡུལ་དང་རིང་དུ་བྲལ་ནས་དུས་ཡུན་རིང་བོ་སོང་བའི་ཡུལ་གྱར་བ་དག་ལ་ཐུན་མོང་གི་དཀའ་ངལ་ཞིག་ཡོད་པ་ཕལ་ཆེར་འདྲ་འདྲ་ཡིན་པ་འདྲ།
ང་རང་༡༩༩༩ལོར་ཕ་ཡུལ་དང་གྱེས་ཏེ་པེ་ཅིན་དུ་འགྲོ་དུས། ང་སྐྱེས་ས་སྐྱེས་ཡུལ་གྱི་ལུང་བ་དེའི་འཚོ་བའི་གོ་རིམ་གང་དུ་ཡང་དེང་རབས་རང་བཞིན་ཅན་གྱི་བྱ་དངོས་ཤིན་ཏུ་ཉུང་ཞིང་། མི་རྣམས་དུས་དང་རྣམ་པ་ཀུན་ཏུ་བག་ཕེབས་པོར་གནས་པ་དང་། དཔལ་འབྱོར་གྱི་ཁེ་ཕན་དང་ཉོ་ཚོང་གི་རྒྱལ་ཕམ་སོགས་ད་དུང་འཚོ་བའི་ནང་དུ་མངོན་གསལ་གྱིས་སླེབས་མེད། དེ་བས་ཀྱང་བུད་མེད་རྣམས་ཀྱིས་བོད་ལྭ་གྱོན་ནས་སྐྲར་ལན་བུ་སླེ་ཞིང་། ཡོ་འབོག་གི་ཤུན་པགས་བཀོག་ནས་སྦངས་པའི་གཤེར་ཁུ་འདག་འདག་དེས་སྐྲ་བཀྲུ། སྐྱེས་པ་རྣམས་ཀྱི་མགོ་ལ་རྒྱ་ཞྭ་སེར་པོ་རེ་གྱོན་ནས་འགྲོ་ཞིང་། ཞྭ་གྱོན་མེད་པར་མགོ་རྗེན་དུ་འགྲོ་བ་ནི་གཉེན་ཉེ་འདས་གྲོངས་སུ་གྱུར་ནས་སྡུག་འོག་ཏུ་ཡོད་མཁན་དེ་དག་ཡིན། ཡིན་ན་ཡང་། ང་རང་གྱེས་པའི་ལོ་ངོ་ཉི་ཤུ་ལྷག་གི་རིང་དུ། མཐའ་འཁོབ་ཀྱི་རི་གྲོང་འདིར་འཕོ་འགྱུར་ཆེན་པོ་བྱུང་འདུག ད་ལྟ་ང་རང་ཕྱིར་ཕ་ཡུལ་ལ་འཁོར་ཞག་ཏུ་བསྐྱོད་དུས། ཨར་འདམ་གྱི་གཞུང་ལམ་དུ་རླངས་འཁོར་ཆེ་ཆུང་ཟམ་མེད་དུ་རྒྱུགས་ཤིང་དུང་སྒྲ་བར་མེད་དུ་སྒྲོག་པ་དང་། གྲོང་ཚོའི་སྲང་ལམ་དུ་སྐྲ་ཡི་ཚོས་མདོག་བསྒྱུར་ནས་ཐུར་ལུག་ཏུ་འཕྱོ་བའི་གཞོན་ནུ་མ་མང་པོ་མཐོང་རྒྱུ་ཡོད་ལ། དེར་མ་ཟད་པེ་ཅིན་གྱི་སྲང་ལམ་དུ་འཕྲད་སྲིད་པའི་ན་ཆུང་དེ་དག་གི་དཔེ་གསར་བཟོ་ལྟ་ཅན་གྱི་གཞོན་ནུ་ཡང་མང་པོར་འཕྲད་ཡོང་། ཁོ་ཚོས་ལག་ཏུ་ཀུ་ཤུ་རྟགས་ཅན་གྱི་ལག་འཁྱེར་ཁ་པར་བཟུང་ནས་ཨམ་ཅོག་ཏུ་ཉན་བྱེད་སྦུ་གུ་རེ་དྲངས་ནས་འགྲོ། མཚམས་རེར་ང་གྲོང་ཚོའི་ནང་དུ་སྡོད་དུས་ཕྱི་ཚུལ་ཁོ་ནའི་ངོས་ནས་བལྟས་ན་སྣང་བ་ཞིག་ལ་རང་ཉིད་ད་དུང་པེ་ཅིན་དུ་ཡོད་པ་འདྲ་ཞིང་། གལ་སྲིད་པེ་ཅིན་མི་ཞིག་ཡིན་ཚེ་ད་ནི་ངའི་སྐྱེས་ས་སྐྱེས་ཡུལ་གྱི་ཕ་ཡུལ་འདིར་སླེབས་འདུག་ཅིང་། ད་དུང་སླེབས་ནས་རིང་དུ་འགོར་ཡོད་པའི་བཟོ་ལྟ་ཐོན་འདུག མ་གཞིར། མི་ལོ་ཉི་ཤུ་ལྷག་གི་དུས་ཚོད་ཀྱི་གོ་རིམ་ཏེ། ང་ཕ་ཡུལ་དུ་མེད་པའི་དུས་ཚོད་ནང་དུ་རྒྱང་རིང་མཐའ་འཁོབ་ཀྱི་རི་གྲོང་འདིས་སྦྲུལ་གྱིས་སྦྲུལ་ཤུན་བརྗེ་བ་ནང་བཞིན་སོ་སོར་འགྱུར་ལྡོག་ཅི་ཞིག་བཏང་པའམ་ཡང་ན་འགྱུར་ལྡོག་ཅི་ཞིག་ཐེབས་པ་ཡིན་ནམ། འདི་ལྟ་བུའི་འགྱུར་ལྡོག་ལ་ལེགས་ཉེས་དགེ་སྐྱོན་གྱི་བརྡར་ཤ་ཅི་འདྲ་ཞིག་གཅོད་དགོས་སམ། ངས་དངོས་ནས་མི་རྟོགས་ཏེ། ང་ཚུལ་འདི་ལ་རྨོངས་པ་ནི་མིའི་རིགས་ཀྱིས་ད་དུང་ཡང་ལན་མ་རྙེད་པར་ལུས་ཡོད་པའི་དོགས་གནས་ཁ་ཤས་ལ་རྨོངས་པ་དང་མཚུངས། ཡིན་ན་ཡང་ངས་གསལ་བོར་རྟོགས་པ་ནི་ལོ་ངོ་ཉི་ཤུའི་རྗེས་ཀྱི་ཕ་ཡུལ་དེ་ནི་པེ་ཅིན་དུ་ངའི་ཡིད་འཛིན་དུ་ཡང་ཡང་འཆར་བའི་ཕ་ཡུལ་དེ་མ་ཡིན་ཏེ། ངའི་སེམས་ནང་གི་ཕ་ཡུལ་ནི་ང་རང་དང་ཐོག་ལམ་ལ་ཆས་སའི་ཕ་ཡུལ་དེ་ཡིན་ཞིང་། དེ་ནི་ལོ་ངོ་ཉི་ཤུའི་སྔོན་གྱི་ཕ་ཡུལ་དེ་ཡིན། ལོ་ངོ་ཉི་ཤུ་ལྷག་གི་རིང་ལ། རི་གྲོང་འདིས་བརྒྱུད་པའི་འཚོ་བའི་འགྱུར་ལྡོག་ནི་ང་ལ་ཆ་རྒྱུས་གསལ་བོ་མེད། ང་རང་ཕ་ཡུལ་གྱི་ཆེས་གནད་འགག་གི་ལོ་རྒྱུས་ཀྱི་འགྱུར་ལྡོག་འདིའི་གོ་རིམ་ནང་ནས་ཕྱིར་བཤོལ་བྱེད་མཁན་ཞིག་ཏུ་གྱུར་ཡོད་ལ། རིག་གནས་འཕོ་འགྱུར་གྱི་གར་སྟེགས་སུ་སླེབས་ཐུབ་མེད་པའི་འཁྲབ་སྟོན་པ་དེར་གྱུར་ཡོད། དེ་བས། ལོ་ངོ་ཉི་ཤུའི་རྗེས་ནས་ང་རང་ཁོ་ཚོའི་གསེབ་ཏུ་བསྲེས་ཏེ། ཁོ་ཚོ་དང་གདངས་གཅིག་པའི་གླུ་རེ་ལེན་པའམ་སྟངས་སྟབས་གཅིག་པའི་གར་རེ་འཁྲབ་པ་ལ། ད་ནི་དཀའ་ཁག་ཆེན་པོ་ཞིག་བྱུང་སོང་།
ཡུལ་གྱར་བ་སུ་ཡིན་ཡང་སེམས་ཁོངས་སུ་གཅེས་པར་ཉར་བའི་ལོག་མལ་དེ་ནི་དང་ཐོག་རང་ཉིད་ལམ་ལ་ཆས་དུས་ཀྱི་རྣམ་པ་དེ་མ་འགྱུར་བར་སོར་ན་གནས་ཡོད་ན་སྙམ་ངེས། ཡིན་ན་ཡང་ཡོད་ཚད་མི་རྟག་པའི་ཆོས་ཉིད་ཀྱི་གཟུགས་སུ་གྲུབ་པའི་འཇིག་རྟེན་འདིར་མི་འགྱུར་པར་སོར་ན་གནས་པའི་བྱ་དངོས་ཤིག་ག་ལ་ཡོད་ཨང་། དེ་བས་ཀྱང་བྱ་དངོས་དེ་དུས་ཚོད་ཀྱི་གཙང་བོས་རིང་དུ་ཁུར་རྗེས་ནམ་ཡང་མི་འགྱུར་བར་སོར་ནས་ཁྱེད་ལ་སྒུག་མཁན་ཞིག་དེ་བས་ཀྱང་ཡོད་མི་སྲིད། ཕ་ཡུལ་གྱི་འགྱུར་ལྡོག་གིས་ང་ལ་ཕ་ཡུལ་ལ་རྒྱུས་མེད་པའི་སྣང་བ་ཞིག་ཡང་དང་བསྐྱར་དུ་སྟེར་བ་དང་། ད་ནི་ཕྱིར་དྲན་ཁྲོད་ཀྱི་སོས་དལ་བག་ཕེབས་ཀྱི་རྣམ་པ་དེར་ཕྱིར་ལོག་མི་ཐུབ་པར་གྱུར་ལ། འདྲིས་ཡོད་ཡོད་འདྲ་བའི་ཡུལ་མི་ཚོའི་གདོང་གི་བཅོས་མིན་གྱི་འཛུམ་མདངས་དེར་ཡང་འཕྲད་མི་ཐུབ་པ་འདྲ། དེར་མ་ཟད། ང་རང་ཉིད་ལོ་ངོ་ཉི་ཤུ་ལྷག་ལ་བྱེས་སུ་འཁྱམ་པའི་གོ་རིམ་ཁྲོད་རིམ་གྱིས་གཤིས་སུ་གྲུབ་པའི་དགའ་ཁ་འཁྱེར་སོ་དང་། འདུ་ཤེས་འཛིན་སྟངས། འགྲོ་སྟངས་འདུག་སྟངས། དེ་མིན་ད་དུང་རིན་ཐང་ལྟ་ཚུལ་དང་། མཛེས་དཔྱོད་ལྟ་ཚུལ་སོགས་ནི་ང་རང་ཕྱིར་ལོག་པའི་འགོག་རྐྱེན་དུ་གྱུར་ཡོད་པ་འདྲ། མདོར་ན། ང་བྱེས་སུ་ལུས་ཤིང་བྱེས་ཀྱི་འཚོ་བར་ཇི་ལྟར་གོམས་པ་དང་། འདྲིས་པ། ལོབས་པར་གྱུར་ན། ད་ནི་དེ་ལྟར་ཕ་ཡུལ་ལ་བློ་མཐུན་པའི་ལག་འཇུ་དང་། བར་ཐག་མེད་པའི་འཐམ་འཁྱུད། འགྲེལ་བཤད་མི་དགོས་པའི་བགྲོ་གླེང་སོགས་ཡོང་དཀའ་བ་འདྲ།
དངོས་ཡོད་ཀྱི་གནས་ཚུལ་ནི་འདི་ལྟར་སེམས་ལ་གཟན་པ་ཞིག་ཡིན་སྲིད་ཀྱང་། ང་པེ་ཅིན་དུ་སྡོད་ཡུན་རིང་པ་དང་བསྟུན་ནས་ཕ་ཡུལ་དུ་ལོག་འདོད་པའི་སེམས་འདུན་དེར་འགྱུར་བ་ནམ་ཡང་བྱུང་མེད། སེམས་ནང་ནས་ཕ་ཡུལ་དུ་ལོག་འདོད་ཀྱང་དོན་དངོས་ཁྲོད་ཕ་ཡུལ་དུ་ལོག་མི་ཐུབ་པའི་གནས་ཚུལ་འདིའི་རིམ་གྱིས་ངའི་མི་ཚེའི་ནང་གི་འགལ་བ་གཙོ་བོ་དེ་གྲུབ་ཡོད་ལ། འདི་ལྟ་བུའི་འགལ་འདུའི་ཁྲོད་ངའི་ཚད་ཡོད་པའི་མི་ཚེའི་དུས་ཚོད་འདི་ཡང་འདི་ལྟར་རིམ་གྱིས་ཟད་བཞིན་ཡོད། འདི་ནི་མཐར་ཐུག་ན་ངའི་ལས་དབང་ཡིན་ངེས་ཏེ། ང་དང་ཆ་འདྲ་བའི་ཡུལ་གྱར་བ་དག་གིས་དུས་ནམ་ཡང་གདོང་གཏད་དགོས་པའི་གཡོལ་མེད་ཀྱི་ལས་དབང་དེ་ཡིན་སྲིད། ཡུལ་གྱར་བ་དང་ཕ་ཡུལ་གྱི་འབྲེལ་བ་ནི་ཤོག་བྱ་དང་ཤོག་བྱ་གཏོང་མཁན་གྱི་འབྲེལ་བ་དང་མཚུངས་ཤིང་། ཤོག་བྱ་བར་སྣང་དུ་ཇི་ལྟར་མཐོ་རུ་འཕུར་ཀྱང་སྲད་བུས་བཏགས་ཡོད་ཅིང་། མཐོ་ནས་མཐོ་རུ་འཕུར་བའི་ཨུ་ཚུགས་དེ་ནི་ཕ་ཡུལ་དང་རིང་ནས་རིང་དུ་གྱེས་པའི་རྣམ་པ་ཞིག་ཡིན་ངེས། ཕ་ཡུལ་ནི་ཤོག་བྱ་གཏོང་མཁན་དེ་དང་འདྲ་བའི་ངོས་ནས་ཀྱང་ཤོག་བྱ་དེ་ཇི་ལྟར་རིང་དུ་འཕུར་ཀྱང་འདོགས་ཐག་དེ་མ་ཆད་ན་བསམ་ཡོད་སྲིད། དེ་བས། ཡུལ་གྱར་བ་དང་ཕ་ཡུལ་གཉིས་ཀྱི་བར་གྱི་སྦྲེལ་ཐག་ལྟ་བུའི་སྲད་བུ་དེ་ནི་ཅི་འདྲའི་ངོ་མཚར་བ་ཞིག་རེད་ཨང་། སྲད་བུ་དེ་ལ་ཡུལ་གདུང་ཞེས་བོས་ཆོག་གམ།
ང་ལ་མཚོན་ན་ཡང་། རང་ཉིད་དུས་ནམ་ཡང་ཕལ་ཆེར་ཕ་ཡུལ་དུ་ལོག་མི་ཐུབ་པ་རྟོགས་ཡོད་ན་ཡང་། ངས་རང་ཉིད་དུས་ནམ་ཡང་ཕ་ཡུལ་དུ་ཕྱིར་ལོག་པའི་ལམ་དུ་འགྲོ་བཞིན་ཡོད་པ་ཚོར། པེ་ཅིན་གྱི་འཚོ་བ་ནི་མདུན་ཕྱོགས་ཡིན་ལ་མ་འོངས་པ་ཡིན་ན་ཡང་། ང་ནི་དུས་ནམ་ཡང་ཁ་ཕྱིར་ལ་འཁོར་ནས་འགྲོ་མཁན་ཞིག་དང་འདས་པའི་ཕྱོགས་སུ་འགྲོ་མཁན་ཞིག་ཡིན་སྲིད། རྒྱུ་མཚན་ནི། དང་ཐོག་ཕ་ཡུལ་དང་གྱེས་དོན་ནི་རྒྱང་རིང་བསྙེགས་པ་དང་མ་འོངས་བ་སྙེགས་པ་སོགས་གང་ཡིན་རུང་། ད་ལྟ་ངའི་སེམས་སུ་ང་རང་མི་ཚེ་འདིར་ཕལ་ཆེར་ཕ་ཡུལ་གྱི་མིག་རྒྱང་གི་འོག་ནས་ཐར་མི་ཐུབ་ལ། ཕ་ཡུལ་གྱི་ལག་ཏུ་བཟུང་ཡོད་པའི་སྲད་བུ་དེས་ངའི་ཚེ་སྲོག་གི་འགྲུལ་བཞུད་འདིའི་རིང་ཚད་ཐག་བཅད་ཡོད་པ་འདྲ།
«ཤེས་རྨོངས»ཀྱི་མཇུག་ཏུ་རྩོམ་པ་པོས་འདི་ལྟར་བྲིས་ཡོད་དེ། “སློབ་ཆེན་གྱི་སྐབས་སུ། གོ་ལ་གཞན་ཞིག་ལ་འགྲུལ་བཞུད་བྱ་འདོད་ཀྱི་ཕུགས་འདུན་དེས་དུས་ནམ་ཡང་ཁོ་མོར་བསླུ་བྲིད་བྱེད་བཞིན་ཡོད། གལ་སྲིད་སྟོང་ཁམས་ཀྱི་འཇིག་རྟེན་ཆེན་པོ་འདི་ནས་ཐར་ཏེ། ཐག་རིང་ཐག་རིང་ཞིག་ལ་བྲོས་ནས། ཚེ་སྲོག་གི་རྣམ་པ་སོགས་ད་ལྟའི་མཐོང་ཡུལ་དུ་གྱུར་བ་འདི་དང་རྩ་བ་ནས་མི་འདྲ་བའི་འཇིག་རྟེན་ཞིག་སྟེ། ཤ་ཁྲག་གདོས་བཅོས་ཀྱི་ལུས་པོ་འདིར་ཡང་བརྟེན་མི་དགོས་པའི་ས་ཆ་ཞིག་ལ་འབྲོ་ཐུབ་ཚེ། དེ་ནི་ཅི་འདྲའི་བདེ་སྐྱིད་དང་ལྡན་པ་ཞིག་ཡིན་ངེས་ཨང་། འོན་ཀྱང་། ངོ་མཚར་བའི་མེ་ཤུགས་འཕུར་མདའ་གང་མང་ཞིག་འཕངས་ན་ཡང་། མི་ནི་སྟོང་ཁམས་ཆེན་པོ་འདི་ནས་ཐག་རིང་ཞིག་ལ་འགྲོ་ཐུབ་མེད། མི་ཡི་ཡུན་ཐུང་ལ་འཁྱུག་ཙམ་གྱིས་ཟད་འགྲོ་བའི་ཚེ་སྲོག་གིས་མགོ་ཐོག་གི་ནམ་མཁའ་དེ་ཀླད་ཁེབས་ནག་པོ་ཞིག་ཏུ་བསྒྱུར་འགྲོ་བ་དང་། མི་ཡི་མགོ་བོ་དུས་ནམ་ཡང་དེ་ལ་ཐོགས་ནས་གས་ཧྲུག་ཏུ་གྱུར་ཏེ། ཡང་བསྐྱར་ཚེ་སྲོག་ཡོད་ཚད་ཀྱིས་ཁ་ཆེ་གདངས་ནས་ཅི་ཞིག་ཟ་བར་རྩོམ་པའམ་ཡང་ཅི་ཞིག་གི་ཁ་ཟས་སུ་འགྱུར་ངེས་པའི་ས་ཡི་གོ་ལ་འདིའི་སྟེང་དུ་ལྷུང་འགྲོ” ཟེར། འདི་ལས་རྨི་ལན་ཁུང་དྷི་ར་ཡིས་ཡུལ་གྱར་བ་དག་གི་ལས་དབང་ལ་སྤྱི་བསྡོམས་ཤིག་བྱས་ཡོད་ངེས་ཏེ། དེ་ནི་ང་ཚོ་ཇི་ལྟར་རིང་དུ་གྱེས་ཀྱང་ད་ལྟར་རིང་དུ་འགྲོ་མི་ཐུབ། གལ་སྲིད་གནས་ལུགས་འདི་བདེན་པ་ཡིན་ན། ངས་ཀྱང་རང་གི་མི་ཚེ་འདི་རྒྱང་དུ་གྱེས་མི་ཐུབ་པའི་མི་ཚེ་ཞིག་ཡིན་ལ། དེ་བས་ཀྱང་ཕྱིར་ལོག་མི་ཐུབ་པའི་མི་ཚེ་ཞིག་ཡིན་པར་ངོས་འཛིན་བྱས་ཆོག་པ་འདྲ།
གང་ལྟར་ཀྱང་། གལ་སྲིད་སྐྱེ་བ་ཕྱི་མ་ཞིག་ཡོད་ཚེ། ངས་རང་ཉིད་སྔར་བཞིན་ཕ་ཡུལ་བོད་ནས་སྐྱེས་པའི་སྨོན་ལམ་འདེབས་བཞིན་ཡོད།

༢༠༡༨ལོའི་ཟླ་༣ཚེས༢༩ཉིན་པེ་ཅིན་ནས།


རྩོམ་སྒྲིག་པ། གངས་དགའ།
(མཆོད་མེའི་པར་དབང་གཞན་གྱིས་དགོལ་མི་རུང)

རྩོམ་ཁུངས། ངེད་དྲ་བ།
