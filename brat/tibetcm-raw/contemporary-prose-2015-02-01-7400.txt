འཛམ་གླིང་གི་ཞི་བདེ་དང་བཟོད་སྒོམ།
དབུ་རུ་དོན་འདུར།

ཞི་བདེ་ཞེས་པའི་མིང་ཚིག་དེ་ནི་འཛམ་གླིང་ཐོག་ས་ཁོངས་མི་འདྲ་བ་དང་། རྒྱལ་ཁབ་མི་འདྲ་བ། མི་རིགས་དང་སྐད་ཡིག་མི་གཅིག་པ། ཆོས་ལུགས་དད་མོས་མི་འདྲ་བ། འཚོ་བའི་དབུལ་ཕྱུག་དང་གོམས་སྲོལ་མི་འདྲ་བ་སོགས་སྐྱེ་བོ་ཡོད་ཚད་ཀྱི་སེམས་ནང་གི་འབོད་སྒྲ་ཡིན་པ་མ་ཟད། ནམ་མཁར་འཕུར་བའི་འདབ་ཆགས་དང་། ས་འོག་ཏུ་འཛུལ་བའི་འབུ་སྲིན་དང་། ས་སྟེང་དུ་གནས་པའི་དུད་འགྲོ ཆུ་ནང་དུ་འཚོ་བའི་ཉ་རིགས་སོགས་འགྲོ་བ་སེམས་ཅན་ཀུན་གྱི་འབོད་སྒྲ་ཡིན་པ་རེད། ཡིན་ན་ཡང་། མི་དང་དུད་འགྲོའི་བར་ལ་ཤེས་རབ་ཀྱི་རྣོ་དབལ་དང་། སྐད་ཆ་ཡོད་མེད་ཀྱི་ཁྱད་པར་བྱུང་བ་དང་། མི་དང་མིའི་བར་ལ་དབུལ་ཕྱུག་དང་བཙན་གཉོམ། དྲག་པོ་དང་ཉམས་ཆུང་། མང་བ་དང་ཉུང་བ་སོགས་ཀྱི་ཁྱད་པར་བྱུང་བ། ད་དུང་མི་རིགས་དང་རྒྱལ་ཁབ།ས་ཁོངས་སོགས་ལེགས་ཞན་བཟང་ངན་གྱི་ཁྱད་པར་སྣ་ཚོགས་བྱུང་བས་དེ་དག་གི་བར་ལ་འཐབ་རྩོད་དང་དམག་འཁྲུག་ཡང་ཡང་བྱུང་བ་ལས་ཞི་བདེ་དང་བསྟན་ལྷིང་ཞིག་ཧ་ཅང་ཁག་པོ་རེད། ཨ། མི་དང་དུད་འགྲོའི་བར་ལ་རྣམ་དཔྱོད་དང་སྐད་ཆའི་ཁྱད་པར་བྱུང་བ་དེས་མིས་དུད་འགྲོ་གང་བྱུང་དུ་བཀོལ་སྤྱོད་བྱས་སྟེ་མི་འཁྱོག་པའི་འཁེལ་ བཀལ། མི་འགྲོ་བའི་ལམ་ལ་དེད། མི་རུང་བའི་སྡུག་ལས་སྣ་ཚོགས་བྱེད་དུ་བཅུག་ནས་ཁོ་ཚོ་གསོན་པ་ལ་རང་དབང་མེད་ཅིང་ཁོ་ཚོ་བསད་ནས་ཤ་ཁྲག་གི་ཁ་ཟས་བྱེད་པས་མི་དང་དུད་འགྲོའི་བར་ལ་ཞི་བདེ་ཡོང་ཐབས་ བྲལ། ལས་ཀྱི་རྣམ་པར་སྨིན་ནས་ང་ཚོ་ལ་ཡང་ནད་རིགས་སྣ་ཚོགས་ཀྱི་བཙན་འཛུལ་བྱེད་འགོ་ཚུགས་ཡོད། ང་ཚོས་རང་ཉིད་ཀྱི་བདེ་ཐབས་ཁོ་ན་བསམ་ནས་སེམས་ཅན་མང་པོའི་རང་བྱུང་གི་ཁྱིམ་གཞི་ལ་གཏོར་སྐྱོན་བཏང་། སྔོ་ལྗང་གི་རྩྭ་ཐང་ས་རྒོད་དུ་གྱུར་བ་དང་། རང་བྱུང་གི་རི་བྲག་འགས་གཏོར་བྱས་པས་རྩ་ཆེའི་སྲོག་ཆགས་མང་པོ་རྩ་མེད་ལ་སོང་ཟིན་པ་དང་། སྟག་གཟིག་དོམ་དྲེད་སོགས་ང་ཚོའི་མིག་ལམ་ནས་རིམ་པས་ཡལ་གིན་འདུག ང་ཚོས་ཟས་ཅོག་གྲུ་བཞིའི་མཐའ་ནས་བསྐོར་ཏེ་སྤྲིའུ་གསོན་པོའི་ཀླད་ཞོ་ལྡག་དུས་དང་། བརྩེ་བའི་བེའུ་ཤུལ་དུ་བཞག་ནས་འབྲི་མོ་ཨ་མའི་ཚེ་སྲོག་བཀུམ་དུས། ཡ་ང་དང་སྙིང་རྗེ་ཟེར་བ་དྲན་མྱོང་ངམ། ང་ཚོ་ཤི་རྒྱུ་ལ་སྐྲག་པ་ནང་བཞིན་ཁོ་ཚོ་ཡང་ཤི་རྒྱུ་ལ་སྐྲག་ཅིང་ང་ཚོ་ལ་ན་ཟུག་ཡོད་ པ་ལྟར་ཁོ་ཚོ་ལ་ཡང་ན་ཟུག་གཏོང་རྒྱུ་ཡོད། ང་ཚོ་བུ་ཕྲུག་ལ་བརྩེ་བ་ནང་བཞིན་ཁོ་ཚོ་ཡང་བུ་ཕྲུག་ལ་བརྩེ་བོ་ཡོད། ཡིན་ན་ཡང་ཁོ་ཚོས་ཁ་གྲགས་ནས་ན་ཟུག་དང་སྡུག་བསྔལ་བརྗོད་མི་ཤེས་ལ་ཁོ་ཚོ་ཡར་ལངས་ནས་ང་ཚོ་ལ་ཚུར་བརྒོལ་བྱེད་པའི་སྙིང་སྟོབས་མེད་པ་དང་། ཐབས་ཤེས་མི་ཤེས་པ་ དེས་ང་ཚོས་ད་བར་དུ་ཁོ་ཚོ་ལ་རང་འགུལ་གྱིས་བསམ་གཞིགས་བྱེད་མྱོང་ མེད་ལ་སྲུང་སྐྱོབས་དང་། བྱམས་སྐྱོང་། ཡ་ང་དང་སྙིང་རྗེ་སོགས་བྱེད་མྱོང་མེད། སྡུག་སྤྱོད་དང་དམར་གསོད་ཀྱི་མདུན་དུ་ཁོ་ཚོ་ལ་ཞི་བདེ་ཡོང་སྲིད་དམ། ཆུ་མ་བདེ་ན་ཉ་མི་བདེ་བའི་དཔེ་ལྟར་ང་ཚོ་ལ་ཡང་ཞི་བདེ་ཡོང་གི་མེད་ལ་ཡོང་ཡང་མི་སྲིད། ཨི། མི་དང་མིའི་བར་ལ་ཁྱད་པར་བྱུང་བ་ནས་བཟུང་ཕན་ཚུན་བསམ་གཞིགས་ དང་བཟོད་སྒོམ་བྱེད་རེས་མེད་པར་དབང་ཆེན་གྱི་དབང་ཆུང་གནོན་པ་དང་བཙན་པོས་ཉམ་ཆུང་ཟོས་པ། མང་བས་ཉུང་ངུ་བཏུལ་བ་སོགས་ལོ་རྒྱུས་ཀྱི་གླེགས་བམ་ཁྲོད་དུ་རང་རིགས་ཀྱི་དཔའ་བོ་དང་དཔའ་མོ་ཡིན་ན་གཞན་རིགས་གསོད་པའི་ལག་དམར་བ་ཡིན་ཞིང་གཞན་རིགས་ཀྱི་དཔའ་བོ་དཔའ་མོ་ཟེར་བ་དེ་དག་ཀྱང་རང་རིགས་བརླག་པའི་ཤེད་མ་རེད་ཨང་། བོད་མཐོ་སྒང་གི་ཡུལ་གྲུ་ནས་མཐའ་རྒྱ་མཚོའི་གླིང་ཕྲན་བར་གྱི་མི་རིགས་འདྲ་མིན་སྣ་ཚོགས་པའི་བར་ལ་དམག་འཁྲུག་དང་འཐབ་རྩོད་མེད་པ་གཅིག་ཀྱང་མེད་ལ་འཐབ་རྩོད་མ་བྱས་པར་ཉམས་ཆུང་མཐའ་ནས་བསྐྱངས། བཙན་པོ་མགོ་ནས་གནོན། དྲང་པོའི་རྒྱབ་ལ་ལངས། བདེན་པའི་ཕྱོགས་ལ་བུད་པ་ག་ཚོད་ཡོད་དམ། ཨུ། ས་ཆ་བཅད་བཟུང་བྱས་ནས་ཚོགས་པ་དང་ཚོགས་པའི་བར་ལ་དམག་འཁྲུག་བྱུང་བ་དང་། ཡུལ་ཚོ་དང་ལུང་པ་བྱུང་ནས་ས་ཁུལ་དང་ས་ཁུལ་བར་ལ་དམག་འཁྲུག་བྱུང་། མ་རྒྱུད་ཚོ་པ་ནས་ཕ་རྒྱུད་ཚོ་པར་འཕོ་འགྱུར་བྱུང་བ་སོགས་སྤྱི་ཚོགས་ཀྱི་བརྡབ་བསིགས་སྣ་ཚོགས་ཁྲོད་དུ་རང་གཅེས་ གཞན་གནོད་ཀྱི་བྱ་བ་མ་བརྩམས་པའི་སྣ་མེད་ཅིང་། ཁོག་ཡངས་དཀྱེལ་ཆེ་ཟེར་བ་ཕར་བཞག་ནང་ཁུལ་དུ་ཡང་མཐུན་སྒྲིལ་དང་རོགས་རེས་མེད་ པར་ཕན་ཚུན་གསོད་རེས་དང་འཛིང་རེས་ཀྱི་རླབས་རྒྱུན་དགུང་ལ་འཕྱུར་ མྱོང་། ཕན་ཚུན་འཕྲོག་རྩོད་དང་གསོད་རེས་འཛིང་རེས་བྱས་ནས་ཕ་མ་མེད་པའི་དྭ་ཕྲུག་མི་ཡུལ་དུ་བསྐྱུར་མྱོང་ལ་ཁྱིམ་གཞི་མེད་པའི་སྤྲང་པོ་དང་དབུལ་པོ་ཚོ་བདག་མེད་ལྟོགས་ཤི་ཐེབས་པ་ཡང་བརྗོད་ཀྱི་ག་ལ་ལངས། མི་ཤི་ཁྱིམ་སྟོང་བྱས་པ་དང་། ས་སྟེང་རོ་ཡིས་ཁེངས་ཤིང་ཁྲག་གི་ཆུ་མོ་རྒྱུགས་པ་ཡང་ཐེངས་མ་གཅིག་གཉིས་ག་ལ་ཡིན། ཨེ། ས་ཁོངས་དང་ས་མཚམས་ཀྱི་དབྱེ་བ་བྱུང་ནས་རྒྱལ་ཁབ་བྱུང་ཞིང་རྒྱལ་ཁབ་ལ་རྒྱལ་པོ་དང་མི་སེར། དཔོན་པོ་དང་གཡོག་པོ། གཞིས་བདག་དང་བྲན་གཡོག་སོགས་མི་ཡིན་པ་འདྲ་ཡང་ཐོབ་ཐང་དང་དབང་ཆ་མི་གཅིག་ལ་འཚོ་གོས་དང་ལོངས་སྤྱོད་མི་འདྲ་བའི་བང་རིམ་སྣ་ཚོགས་ཆགས། བང་རིམ་མཐོ་ས་ནས་དམའ་སར་བརྙས་བཅོས་དང་སྡུག་ལས་བཀུར་ཞིང་བསམ་གཞིགས་དང་ཡ་ང་སྙིང་རྗེ་མེད་ལ་རོགས་རམ་དང་བྱམས་སྐྱོང་དེ་བས་བྱེད་མྱོང་མེད། རྒྱལ་པོས་གཡོག་པོ་ལ་ཅ་དངོས་ཀྱི་མཐོང་ལས་མེད་ལ་གཞིས་བདག་གིས་བྲན་གཡོག་ལ་ཕྱུགས་ཟོག་གི་མཐོང་ལས་མེད། མི་རིགས་དཀར་པོས་མི་རིགས་ནག་པོ་སྒོ་ཕྱུགས་བཞིན་བཀོལ་མྱོང་ལ་མི་རིགས་ཆེན་པོས་མི་རིགས་ཆུང་ངུ་ཁ་ཟས་བཞིན་ཁྱུར་མིད་བཏང་། མི་ཆེན་ཕྱུག་པོས་དབུལ་ཕོངས་རྒྱུ་ནོར་ལྟར་བདག་པོ་ཡང་རྒྱག་མྱོང་ལ་རྐང་ འོག་གི་འབུ་སྲིན་བཞིན་རྡོག་བརྫིས་ཀྱང་གཏོང་མྱོང་ཡོད། གང་ལྟར་འཁྲུག་རྩོད་དང་འཛིང་རེས་ཁོ་ན་ལས་སུས་ཀྱང་སུ་ལ་བཟོད་སྒོམ་དང་བསམ་གཞིགས་བྱེད་མྱོང་མེད། དེར་བརྟེན་ཁ་ནས་ཞི་བདེ་དང་ཞི་བདེ་ཞེས་ག་ཚོད་གླེང་ཡང་ཞི་བདེ་དང་ང་ཚོའི་བར་ཐག་ཧ་ཅང་རིང་། ཨུ། དུས་ཚོད་དང་ལོ་རབས་མང་པོ་ཞིག་ཕར་ལ་འདས་རྗེས་མི་རྣམས་ཀྱིས་ཞི་བདེ་འཕེལ་རྒྱས་ཟེར་བ་འཛམ་གླིང་མི་དམངས་ཀྱིས་དོ་ཁུར་བྱེད་པའི་གནད་དོན་ཆེན་པོ་གཉིས་ཡིན་པ་ངོས་འཛིན་བྱས་ཀྱང་ས་ཁོངས་དང་། བདག་དབང་གི་དཀའ་རྙོག་མདུན་དུ་སུས་ཀྱང་བཟོད་སྒོམ་དང་བསམ་གཞིགས་བྱེད་མྱོང་མེད། བཟོད་སྒོམ་དང་བསམ་གཞིགས་བྱེད་རེས་མེད་ན་ཁ་རྩོད་དང་འཛིང་རེས་སོགས་ཡོང་རྒྱུ་ནི་སུས་ཀྱང་མངོན་དཔག་བྱེད་ཐུབ་པ་ཞིག་རེད། གང་ལྟར་མི་རིགས་དང་རྒྱལ་ཁོངས་ཀྱི་དབྱེ་བ་མེད་པར་ཚང་མ་ཞི་བདེ་ལ་དགའ་ན་ངེས་པར་དུ་ཛ་དྲག་གི་དུས་སུ་བློ་མནལ་ལ་ཕབ་སྟེ་མི་ཡོད་ཚད་ལ་བསམ་བློ་གཏོང་དགོས་པ་དང་། དམག་འཁྲུག་མེད་པའི་ཐོག་ནས་ཕན་ཚུན་གྲོས་མོལ་དང་གནད་དོན་ཐག་གཅོད་ཐུབ་པ་བྱས་ན་ཤིན་ཏུ་ལེགས། ཨེ། དུས་རབས་ཉི་ཤུ་རྩ་གཅིག་པའི་དེ་རིང་གི་དུས་སུ་ཡང་དམངས་གཙོ་དང་རང་དབང་གི་ཕྱི་གོས་གྱོན་ནས་མི་དམངས་སྡུག་ལ་སྦྱར་བའི་རྣམ་པ་ ཡོད་ལ་ས་ཁོངས་དང་རྒྱ་མཚོའི་བདག་དབང་ལ་བསམ་ཏེ་དམག་སྦྱོང་ཙམ་ལ་ཁ་གཡར་ནས་མཚོན་ཆ་དང་སྟོབས་ཤུགས་ཕན་ཚུན་ངོམས་རེས་བྱེད་པའི་གནས་ཚུལ་ཡང་མེད་པ་མ་རེད། འགོ་འཛིན་དང་འགོ་འཛིན་བར་ལ་འཆམ་མཐུན་མ་བྱུང་བར་ཕན་ཚུན་ཁ་རྩོད་བྱེད་པ་དང་། ཕྱོགས་ཁག་དང་རྒྱལ་ཁབ་བར་ལ་ཡང་ཕན་ཚུན་རྩོད་པ་བྱེད་པ་ནི་མཐར་ཐུག་དམག་འཁྲུག་ཡོང་རྒྱུའི་སྔ་ལྟས་རེད། ད་ལྟ་འཛམ་གླིང་རྒྱལ་ཁབ་དང་མི་རིགས་གང་འདྲ་ཞིག་གི་བར་ལ་འགལ་བ་ཡོད་ན་དེ་ལ་གྲོས་མོལ་དང་ཕན་ཚུན་བསམ་གཞིགས་ཀྱིས་གནད་དོན་ཡང་དག་པར་ཐག་གཅོད་གནང་ཐུབ་ན་ལེགས་ཀྱང་ཐག་གཅོད་མ་ཐུབ་པར་ད་ལྟ་དམག་འཁྲུག་བྱེད་དང་བྱེད་བཞིན་པའི་རྒྱལ་ཁབ་མང་པོ་ཡོད། མི་ཚང་མ་བཟོད་སྒོམ་བྱེད་རྒྱུའི་བསམ་བློ་མེད་པའམ་ཡང་ཕྱོགས་གཅིག་གིས་བཟོད་སྒོམ་བྱས་ཀྱང་ཕྱོགས་གཅིག་གིས་བཟོད་སྒོམ་བྱེད་ཀྱི་མེད་པར་མཐའ་མ་དམག་འཁྲུག་ལ་འགྱུར་གྱི་ཡོད། ཨོ། ང་ཚོ་ཞི་བདེ་ལ་དགའམ། གལ་ཏེ་ང་ཚོ་ཞི་བདེ་ལ་དགའ་བ་བདེན་ན་ང་ཚོས་ཕན་ཚུན་བསམ་གཞིགས་དང་ཛ་དྲག་གི་དུས་སུ་ཁེ་ཕན་ཆུང་ངུ་བློས་ གཏོང་བྱས་ནས་བཟོད་སྒོམ་གྱིས་དམག་འཁྲུག་མི་ཡོང་བ་བྱེད་དགོས། ཕན་ཚུན་གྲོས་སྡུར་བྱས་ན་གནས་ཚུལ་ཆེན་པོ་ཅི་ཞིག་ཡོད་ཀྱང་ང་ཚོའི་ཚེ་སྲོག་ལས་ལྷག་པ་གཅིག་ཀྱང་མེད། ཚེ་སྲོག་གིས་མཆོད་པ་འཕུལ་ནས་ལེན་རིན་ཡོད་པའི་དོན་དག་ཅི་ཞིག་ཡོད། དེས་ན་ང་ཚོས་ཚེ་སྲོག་གྲངས་མེད་སྐད་ཅིག་ལ་མེད་པ་བཟོ་བའི་དམག་འཁྲུག་མཐའ་དག་མཚམས་འཇོག་བྱེད་དགོས་པ་ལས་མུ་མཐུད་གསོད་རེས་བྱེད་མི་རུང་། སུ་ལ་ཡང་རྒྱུ་མཚན་རེ་མེད་པར་དམག་འཁྲུག་བྱེད་མི་སྲིད། འོན་ཀྱང་རྒྱུ་མཚན་རེ་ལ་ཕན་ཚུན་གྲོས་སྡུར་བྱེད་རྒྱུ་མེད་པ་ག་ལ་ཡིན་སྲིད་དམ། ངས་འཛམ་གླིང་མི་དམངས་ལ་འབོད་སྐུལ་ཞུ། དམག་འཁྲུག་ནི་སྡུག་བསྔལ་གྱི་འབྱུང་གཞི་ཡིན་པ་དང་། ཞི་བདེ་ནི་ཀུན་ལ་བདེ་སྐྱེད་ཐོབ་པའི་དགོས་འདོད་ཀུན་འབྱུང་ཡིན། དེ་རིང་ནས་བཟུང་འཛམ་གླིང་གི་འགོ་ཁྲིད་གཙོ་བོ་ནས་སྣེ་ཁྲིད་དེ་བཟོད་སྒོམ་ཞེས་པའི་ མིང་ཚིག་དེ་འཛམ་གླིང་ཞི་བདེའི་དར་ཆ་ལ་འཛིན་རོགས། ཚང་མས་བཟོད་སྒོམ་བྱེད་རོགས། ཛ་དྲག་དང་ཉེན་ཁའི་དུས་སུ་སེམས་པ་མནལ་ལ་ཕབ་ནས་བཟོད་པ་སྒོམ་དང་། དེའི་ཕན་ཡོན་ནི་རྩ་ཆེའི་རྒྱུ་ནོར་གང་གིས་ཀྱང་དོ་མི་ཐུབ།
(མཆོད་མེའི་པར་དབང་གཞན་གྱིས་དགོལ་མི་རུང)

རྩོམ་ཁུངས། ངེད་དྲ་བ།
