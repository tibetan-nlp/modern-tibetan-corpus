ཨ་མ་གཡུ་སྒྲོན་གྱི་གཅིན།
རྣོ་སྦྲེང་།

ནམ་གུང་གི་ཆུ་ཚོད་གཅིག་ལ་ཐོན། ཇོ་ཁང་གི་མདུན་དུ་ཕྱག་འཚལ་མཁན་གྱི་མི་རེ་གཉིས་ཡོད་མོད། སྤྱིའི་ཆ་ནས་བཤད་ན། ད་ནི་ཉིན་མོར་མི་ཚོགས་ཀྱིས་འགད་ལ་བྱེད་པའི་ཇོ་ཁང་དང་བར་སྐོར་ལམ་ཧ་ཅང་ལྷིང་ཞིང་འཇགས་པའི་དུས་ཡིན་པ་འདྲ། ང་མཚན་མོ་དེར་གྲོགས་པོ་ཞིག་གི་མགྲོན་འབོད་ནང་བཞིན་བདེ་སྐྱིད་ལམ་གྱི་ཁོ་ཧྥེ་ཁང་ཞིག་གི་ནང་དུ་སོང་ནས་ཡུན་རིང་འགོར། ང་སྡོད་སའི་མགྲོན་ཁང་ནི་བར་སྐོར་ལམ་གྱི་ཉེ་འགྲམ་སྟེ་ཁྲོམ་གཟིགས་སྒང་དུ་འབོད་པའི་སྲང་ལམ་དུ་ཡོད་པས། ལམ་བར་དུ། ང་ལ་མཚན་ལྗོངས་ཀྱི་ཇོ་ཁང་དང་བར་སྐོར་ལམ། ཡང་ན་དབོན་ཞང་རྡོ་རིང་སོགས་པར་ཞིག་རྒྱག་འདོད་ཀྱི་ཚོར་བ་སྐོར་ལོག་ཅིག་གིས་དེད་དེ་གཙུག་ལག་ཁང་གི་ཕྱོགས་སུ་བསྐྱོད་པ་ཡིན། ང་རང་མོ་ཊ་ལས་བབས་ཏེ་ཇོ་ཁང་གི་མདུན་ཕྱོགས་ཀྱི་ལམ་ཆེན་དེད་ནས་སོང་། “ཅི་བྱེད་དུ་འགྲོ་བ་ཡིན།” “ང་སྡོད་སའི་མགྲོན་ཁང་ཁྲོམ་གཟིགས་སྒང་གི་འགྲམ་ན་ཡོད་པས་འདི་ནས་སོང་ཚེ་ཐག་ཉེ།” “ཁུག་མ་ཞིབ་བཤེར་འཕྲུལ་ཆས་སྟེང་ལ་འཇོག་རོགས།”དུས་ཚོད་ཉེར་བཞིའི་རིང་སྲུང་ཡོད་པའི་ཇོ་ཁང་མདུན་ཕྱོགས་ཀྱི་སོ་སྲུང་བས་དེ་སྐད་ལབ་བྱུང་། “ཁུག་མའི་ནང་གི་དེ་ཅི་ཞིག་ཡིན།” “ཇ་དམར་ཡིན།” ཁོས་སྨིན་མ་གཅིག་ཏུ་བསྡུས་ཤིང་སྣ་གཉེར་གཉེར་གྱིས་སྣུམ་ཙམ་བྱས་རྗེས་ཇ་དམར་ཕྱིར་ཁུག་མའི་ནང་དུ་འཇོག་བཞིན“སོང་ཆོག”ཟེར། ངས་གོམ་པ་ཇེ་བྲེལ་ལ་བཏང་སྟེ་ཇོ་ཁང་གི་མདུན་ཕྱོགས་སུ་སོང་། དམར་ཆེམ་ཆེམ་གྱི་གློག་འོད་འོག་ཏུ་མི་རེ་གཉིས་ལས་ཅི་ཡང་མཐོང་རྒྱུ་མི་འདུག ངས་ཟུར་སྣེ་མི་འདྲ་བ་ནས་པར་འགའ་རེ་བརྡབས་ཤིང་། པར་མིག་བསྐོར་ཐེངས་ཤིག་གིས་ཇོ་ཁང་ལྕགས་རིའི་འཁྲིས་ན་ཙོག་ནས་བསྡད་ཡོད་པའི་ཨ་མ་རྒན་མོ་དེ་མཐོང་བྱུང་། ཁོ་མོས་སློག་པ་མཐུག་པོ་ཞིག་གྱོན་ནས་ལག་གི་མ་ཎི་འཁོར་ལོ་ཆད་མེད་དུ་བསྐོར་ཞིང་། ཁ་ཡང་མུར་མུར་གྱིས་ཤིན་ཏུ་བྱང་ལོས་ཡོད་སྙམ་པའི་ཁ་བཏོན་ཞིག་ཆགས་མེད་དུ་ཟློ། ང་རང་ཁོ་མོའི་གམ་དུ་སོང་། ཁོ་མོས་ང་ལ་བལྟ་ཙམ་བྱས་རྗེས་ཅི་ཡང་མི་སྨྲ། “ཨ་མ་ལགས། རང་གང་གི་ཡིན་པས།” “ང་རང་ཨ་མདོ་ནག་ཆུ་ནས་ཡིན། ངས་ཁྱེད་རང་རྒྱ་མི་ཞིག་རེད་སྙམ་བྱུང་།” “རང་གིས་འདི་ན་ག་རེ་གནང་གི་ཡོད།” “ངས་སང་ཉིན་གྱི་ཇོ་མཇལ་ཐོག་མ་ལ་སྒུག་ཡོད།” “ད་ལྟ་དུས་ཚོད་ཞེ་དྲག་སྔ་པོ་འདུག་པས། ཕྱིར་སོང་ནས་ངལ་གསོས་ན་མི་དགའ་འམ།” “མིན། དུས་ཚོད་གསུམ་བཞིའི་སྟེང་ནས་བཟུང་ཇོ་བོར་མཇལ་མཁན་གྱི་མིས་འཚང་ཁ་རྒྱབ་འགྲོ” “འོ་རེད”ངས་དེ་ལྟར་ལན་སྟབས་བདེ་ཞིག་བཏབ་རྗེས་ཁོ་མོའི་འཁྲིས་ནས་ཙོག་སྟེ་སྟོང་བའི་དབྱིངས་སུ་སིམ། ལྟ་ཕྱོགས་ཤིག་གིས་ཨ་མ་རྒན་མོ་ལགས་ཀྱི་སྐྲ་ལོ་དཀར་པོ་དང་། རྒས་གཉེར་བརྩེགས་མ། ད་དུང་དྭངས་གཙང་གིས་ཁྱབ་པའི་འཛུམ་མདངས་སོགས་ཀྱིས་ང་ལ་ཞེ་བདེ་བག་ལྷོད་ཀྱི་ཚོར་བ་སྟེར་ཞིང་། སྐབས་དེར་ཇོ་མཇལ་ཐོག་མ་ལ་རེ་བའི་ཨ་མ་རྒན་མོ་ལགས་ཀྱི་དད་པ་ལ་གཏན་ཚིགས་ཀྱིས་འགྲེལ་རྒྱག་ས་མི་འགྲོ་བ་འདྲ་སྟེ། མིའི་རིགས་ཀྱིས་དགའ་ཁ་འཁྱེར་སོ་མི་འདྲ་བ་དང་། ཚོར་བའི་ཕྲ་ཕུང་རེ་རེ་ཡང་ཁྱོན་ནས་མི་འདྲ་བར། དད་པ་དེ་ཡང་སྒྱུ་རྩལ་ཞིག་གམ་སྨྲ་བསམ་བརྗོད་འདས་ཀྱི་ཚོར་བ་ཞིག་ལགས་པས། མི་སུ་ཡིན་རུང་ཚོར་བ་བདེ་བ་འདོད་པ་དང་ཚོར་བ་སྡུག་བསྔལ་མི་འདོད་པ་ནི་སྤྱི་མཐུན་དུ་གྲུབ་ཡོད་པ་བརྒྱ་ཁ་གཅིག་གྲགས་ཡིན་པ་ཐག་གིས་ཆོད། དེ་བས་ཁོ་མོའི་ཚོར་བ་བདེ་བ་དེ་ལ་སུས་ཀྱང་གཏན་ཚིགས་དང་རྒྱ་མའི་སྟེང་ལ་འཇོག་པ་ནོར་འཁྲུལ་ཞིག་ཡིན་པ་ཚོར། ང་དེ་ལྟར་བསམ་གཞིག་གི་བར་སྣང་ན་ངེས་མེད་དུ་གཡོ་བའི་ཁ་བ་བཞིན་འཕུར་དུས། ཨ་མ་རྒན་མོ་ལགས་ཀྱིས་མ་ཎི་བསྐོར་ལོ་བསྐོར་མཚམས་བཞག་སྟེ། “གཅེས་ཕྲུག ཁྱེད་རང་གང་ནས་ཡིན། ལྷ་སར་ཡོང་དོན་ཅི་ཡིན་ཟེར” “ང་རང་མཚོ་སྔོན་རེབ་གོང་གི་ཡིན། ལྷ་ས་ལ་ལྗོངས་རྒྱུར་ཡོང་་་་་་་ལྷ་ས་ལ་མཇལ་སྐོར་དུ་ཡོང་བ་ཡིན།” “ཨོ་རེད། ཚེ་ཐག་རིང་རྒྱུ་བོ”ཞེས་ངེད་གཉིས་ཀྱིས་ཁ་བརྡའི་ཀློང་བརྡོལ་མགོ་བརྩམས། ལྷ་སའི་ཟླ་འོད་འོག་ཏུ། ཨ་མ་གཡུ་སྒྲོན་དང་ངེད་གཉིས་དེ་ལྟར་ངོ་ཤེས་སུ་གྱུར་བ་ཡིན། ངས་རང་ཉིད་ཀྱི་འདས་མ་ད་གསུམ་གྱི་ལོ་རྒྱུས་དང་དཀའ་སྡུག་མང་པོ་ཞིག་ཁོ་མོར་བཤད་པ་ནང་བཞིན། ཁོ་མོས་ཀྱང་གཙུག་ལག་གི་མདུན་ནས། ད་རེས་ལྷ་ས་ལ་ཡོང་དོན་དང་། ཆུང་དུས་ཀྱི་དཀའ་ངལ། ཁྱིམ་ཚང་གི་འཚོ་བ་སོགས་དགའ་སྐྱོ་གཉིས་ལྡན་གྱི་གཏམ་རྒྱུད་མང་པོ་རིམ་བཞིན་ལབ་བྱུང་། གལ་སྲིད་ད་རུང་དུས་ཚོད་ཡོད་པའི་དབང་དུ་བཏང་ཚེ། ཁོ་མོའི་ཁོག་ན་ཉིན་ཞག་བཅོ་བརྒྱད་ལ་ཡང་གླེང་མི་ཚར་བའི་ངོ་མཚར་གྱི་གཏམ་རྒྱུད་མང་པོ་ཡོད་པ་འདྲ་མོད། དེ་དག་ལས་ཀྱང་ཁོ་མོའི་གཅིན་དང་འབྲེལ་བའི་དངོས་ཡོད་ཀྱི་གཏམ་རྒྱུད་དེས་ང་རང་རྨང་ནས་ཧང་སངས་པར་བྱས་སོང་། ཁོ་མོའི་མིང་ལ་ཨ་མ་གཡུ་སྒྲོན་ཟེར། གོང་ནས་མཚམས་སྦྱོར་བྱས་ཟིན་པ་ནང་བཞིན་ཁོ་མོའི་ཕ་ཡུལ་ནི་ཨ་མདོ་ནག་ཆུ་རེད། ཁོ་མོས་བཤད་ན། “ཁོ་ཚོ་ནི་འབྲོག་ལས་རྐྱང་རྐྱང་ལ་བརྟེན་ནས་རྩྭ་ཐང་ན་འཚོ་བ་རོལ་བའི་འབྲོག་པ་གཞན་དག་དང་འདྲ་མཚུངས་ཡིན”ཟེར། ཁོ་མོ་བཅུ་དྲུག་ལང་ཚོའི་མེ་ཏོག་མངོན་པར་བཀྲ་བའི་དུས་སུ་ཕ་མའི་མངགས་བཀོད་བཞིན་མཐུ་སྟོབས་ཀྱི་རྗེས་སུ་ཡོང་བ་ཡིན་ཞིང་། བརྩེ་དུང་ཟེར་བ་དང་གཉེན་སྒྲིག་རང་དབང་ཟེར་བ་ཁྱོན་ནས་མི་ཤེས་པའི་སྐབས་ཐོག་དེའི་འབྲོག་མོ་གཞན་དག་ཚོ་དང་འདྲ་བའི་ཁོ་མོ། ཚེ་འདིར་མཐུ་སྟོབས་ཀྱི་སྐྱ་ཡར་གྱུར་བ་ནི་ཐོག་མར་རང་གི་བསོད་ནམས་ཤིག་ཏུ་འདོད་མོད། “ཐ་མར་ངའི་ལས་སྣང་མ་དག་པས་་་་”ཞེས་སྐད་ཆའི་འཕྲོ་མཐུད་མ་ཐུབ་པར་ཡུན་རིང་ལ་ལུས། “མཐུ་སྟོབས་ནི་ངེད་ཚོའི་འབྲོག་སྡེ་ལ་མཚོན་ནས་བཤད་ན་ཡག་ཕོ་སྒོ་དང་ལག་ལས་ཀ་གཙོ་བོ་ནི་ཕ་མ་ལ་བསྙེན་བཀུར་ཆེ་ཞིང་། བཟའ་ཟླར་གཅེས་ཤིང་བྱམས་པའི་ཁྱོ་ཀ་བཟང་པོ་ཞིག་ཡིན། ང་རང་ཁོའི་རྗེས་ལ་ཡོང་བའི་ལོ་གཉིས་པར་མངལ་ཆགས་ཤིང་། ཁྱིམ་མི་ཡོངས་ཀྱིས་ངའི་ཁོག་གི་བྱིས་པར་རེ་སྒུག་ཆེན་པོ་བྱས་ནས་བསྡད་མོད། མ་རུངས་བདུད་ཀྱི་གཡབ་མོས་ངའི་གཅེས་ཕྲུག་མངལ་ལས་ལྷུང་བ་དང་སྐད་ཅིག་དུས་མཉམ་དུ་ཁོའི་ཚེ་སྲོག་འཕྲོག་སོང”ཞེས་ཁོ་མོའི་མིག་མཐའ་ནས་མཆི་མ་འཐོར་ཞིང་། ཕྲེང་བ་བཟུང་ཡོད་པའི་ལག་གཡོན་པས་མིག་ཆུ་འཕྱི་ཙམ་བྱས་རྗེས་ཁ་བརྡའི་ལྷག་འཕྲོ་མུ་མཐུད་དུ་གླེང་སོང་། དེ་ནས་བཟུང་ང་ལ་མཐུ་སྟོབས་ཀྱི་བྱིས་པ་འཁོར་མ་མྱོང་། ངེད་གཉིས་ཡར་དཀོན་མཆོག་ལ་གསོལ་བ་བཏབ་ཅིང་། བར་ལྷ་ཀླུ་ལ་བསང་མཆོད་བྱས་ན་ཡང་། སྐྱེས་དམན་ང་ལ་ཁོའི་ཤ་ཁྲག་ལས་མཆེད་པའི་རིགས་རྒྱུད་འཛིན་མཁན་ཞིག་བྱུང་མ་སོང་། དེ་ལྟར་ཉིན་ནས་ཟླ་དང་། ཟླ་ནས་ལོ་མཐུད་དེ་ང་རང་ཤེས་མེད་ཚོར་མེད་ཀྱིས་མོ་ལོ་ཉེར་བརྒྱད་ལ་སོན་ཞིང་། སྐབས་དེར་མཐུ་སྟོབས་ངེད་གཉིས་ལ་ཐབས་བཀོད་གཞན་ཅི་ཡང་མེད་པས་བུ་སྐལ་ཞིག་བླངས་པ་ཡིན། ལོ་དེའི་དབྱར་ཁར། ང་ཚོའི་ཚེ་འདི་ཕྱི་གཉིས་ཀྱི་སྐྱབས་གནས་བླ་མ་མཁའ་ལྡིང་རིན་པོ་ཆེ་ངེད་ཚང་ལ་ཇ་བཞེས་སུ་ཕེབས་ཤིང་མོ་རྩིས་ཡང་ཡང་བྱས་མཐར། “ས་བདག་གིས་གནོད་པ་བྱེད་བཞིན་ཡོད་པས་བཅའ་བ་རིམ་གྲོ་ཤིན་ཏུ་ལྕི”གསུངས། མཐུ་སྟོབས་ཀྱིས་པུས་མོའི་སླ་ང་ས་ལ་བཙུགས་ཏེ་རེ་བ་བྱས་པ་བརྒྱུད། ཉིན་གཉིས་པ་ནས་བཟུང་ཁོང་རིན་པོ་ཆེ་ངེད་ཚང་ལ་རིམ་གྲོ་སྒྲུབ་ཏུ་ཕེབས། ཉིན་མོ་དེར་མཐུ་སྟོབས་ཀྱིས“ང་དེ་རིང་ནོར་ལུག་གི་རྗེས་སུ་འགྲོ་དགོས། ཁྱེད་ཀྱིས་རིན་པོ་ཆེ་ཚང་ལ་སྐུ་གཡོག་ཞབས་ཞུ་ལེགས་པོ་སྒྲུབ་རོགས”ཞེས་བཤད་རྗེས་ཕྱིན་སོང་ལ། བླ་མ་རིན་པོ་ཆེ་ཚང་གིས“གསོལ་ཐབ་པ”ཡང་ཕྱིར་དགོན་པའི་ནང་དུ་དཔེ་ཆ་ཞིག་གི་རྗེས་ལ་མངགས་སོང་། ཉིན་མོ་དེར་ཚ་གདུག་ཤིན་ཏུ་ཆེ། སྦྲ་ནག་གི་དཀར་ཁུང་ལས་འོད་ཐིགས་ཐབ་རྩ་ལ་འཕྲོས་ཤིང་། སྦྲང་མའི་ཝང་སྒྲ་དང་རིན་པོ་ཆེ་ཚང་གི་འདོན་སྐད་ལས་གཞན་ཡོད་ཚད་ལྷིང་ཞིང་འཇགས། དཔྲལ་བ་ན་རྔུལ་ཆུ་འོད་ཀྱིས་འཚེར་གྱིན་པའི་ཁོང་གིས་འདོན་མཚམས་བཞག་སྟེ། “གཡུ་སྒྲོན། ང་ལ་ཆུ་ཞིག་བླུགས་དང”ཟེར། ངས་དཀར་ཡོལ་འབྲུག་རིས་མ་དེ་གཙང་མར་བཀྲུས་ཤིང་ཤུགས་རྡུལ་བྱས་རྗེས་ཁོང་ལ་ཆུ་འཁྱག་མཆོད་པ་ཡིན། ཁོང་གིས་ཆུ་འཁྱག་ལེན་ཞོར་ངའི་ལག་པ་བསྡམས་ཏེ་ཁོང་གི་མདུན་ནས་སྡོད་དང་གསུངས། ངས་ཅི་ཡང་སྨྲ་མི་ཐུབ་པར་ཁོང་གི་སྔུན་ནས་གུམ་བཙོག་བྱས་ཏེ་ཐོད་པ་ཁོང་གི་ཞབས་ལ་གཏུགས་ནས་བསྡད་པ་ན། ཁོང་གིས“ཁྱེད་ཀྱི་ལུས་ལ་མི་གཙང་བའི་སྡིག་སྒྲིབ་མཐུག་པོ་ཡོད་པས། ངས་ཁྱེད་ལ་ཞབས་ཁྲུས་ཤིག་གནང་དགོས”གསུངས་བྱུང་། ངས“ཞབས་ཁྲུས”ཟེར་བ་དེ་ཅི་ཡིན་མི་ཤེས་མོད“ལེགས་སོ”ཞུས་པ་ཡིན། ཉིན་མོ་དེར་ཉིན་མ་དཀར་ངོགས་ནས་ཁོང་གིས་ང་ལ་འཇིག་རྟེན་ཕོ་མོའི་བར་གྱི་དོན་དག་དེ་ལས་སོང་། ཁོང་གིས་གསུང་ན་དེ་ནི“རིམ་གྲོ”ཡིན་ཟེར་ལ། མཐུ་སྟོབས་ཀྱིས་ཀྱང་དོན་དེ་ཧ་གོ་ཡོད་གསུངས་བྱུང་། དེ་ལྟར་ཟླ་གཉིས་གསུམ་ལྷག་གི་རིང་ལ་ཁོང་རིན་པོ་ཆེ་ངེད་ཚང་ལ་ཉིན་འགའ་རེའི་ནང་རིམ་གྲོ་སྒྲུབ་ཏུ་ཕེབས་ཤིང་། ཁོང་རིམ་གྲོ་སྒྲུབ་ཏུ་ཕེབས་ཐེངས་རེ་ལ། མཐུ་སྟོབས་ནོར་ལུག་གི་རྗེས་ལ་བུད་འགྲོ་ཞིང་། གསོལ་ཐབ་པ་ཕྱིར་དགོན་ནང་དུ་བུད་འགྲོ། ཁོང་གིས་ང་ལ་རིམ་གྲོ་སྒྲུབ་པའི་ཉིན་དང་པོར། “དེ་རིང་ནས་བཟུང་ཁྱེད་ཀྱིས་གཅིན་ས་གཙང་ས་དང་རི་མཐོ་ས་ལ་གཏོང་དགོས”གསུངས། བླ་མ་ཐུགས་རྗེ་ཅན་ལ་ཐུགས་བསྐྱེད་སྨོན་ལམ་ཡོད་སྲིད་པས་ངས་ཀྱང་ཁོང་གི་བཀའ་གསུང་བཞིན། རང་གི་གཅིན་ནམ་རྒྱུན་ངེད་ཚང་གི་རི་བསང་ཕུད་སའི་དར་ལྕོག་དེའི་འགྲམ་ལ་བསྐྱལ་བ་ཡིན། ཡུན་རིང་མ་འགོར་བར་ཉི་མ་འདྲ་བའི་བླ་མའི་ཐུགས་བསྐྱེད་སྨོན་ལམ་གྱི་དབང་གིས“ང་ལ་མངལ་འཁོར་ཞིང་། ཟླ་དགུ་ཞག་བཅུ་འགོར་བའི་རྗེས་ནས་ད་ལྟའི་དགེ་འདུན་ལེགས་བཤད་སྐྱེས་པ་ཡིན”ཟེར། འདིར་ཀློག་པ་པོ་དག་ལ་ཞུ་དགོས་པ་ཞིག་ལ། དགེ་འདུན་ལེགས་བཤད་ནི་བླ་མ་མཁའ་ལྡིང་རིན་པོ་ཆེས་རིམ་གྲོའི་ནུས་པ་ལས་ཐོན་པའི་མཐུ་སྟོབས་དང་ཨ་མ་གཡུ་སྒྲོན་གཉིས་ཀྱི་བུ་ཆུང་ཡིན་ཞིང་། ཁོ་ལོ་བཅུ་གཅིག་ལ་སོན་པའི་བར་དེར། བླ་མ་ཚང་གིས་ཁོ་ནི་མ་འོངས་པ་ན་བསྟན་པ་དང་འགྲོ་བ་ལ་ཕན་ཐོགས་པའི་སྐྱེས་བུ་རླབས་ཆེན་ཞིག་ཡོང་རྒྱུའི་ལུང་བསྟན་གནང་ནས་རབ་བྱུང་སྒྲུབ་ཏུ་བཅུག ལོ་ངོ་དྲུག་བདུན་ལྷག་གི་རིང་ལ་ཁོང་གི་གམ་དུ་ཁ་བཏོན་དགེ་སྦྱོར་གྱི་སྐོར་སློབ་ཏུ་བཅུག་ཅིང་། ཕྱིས་སུ་ལྷ་སའི་སེ་ར་དགོན་ལ་བསྐྱལ་བ་རེད་འདུག ཨ་མ་གཡུ་སྒྲོན“ལྷ་སར་ཡོང་དོན་ཡང་གཅིག་ནས་དགེ་འདུན་ལེགས་བཤད་ལ་བལྟ་རུ་འོང་བ་ཡིན་ཞིང་། གཉིས་ནས་ཇོ་བོ་ཞལ་མཇལ་ཐོག་མ་བཅུ་གསུམ་དང་ཕྱག་སྐོར་ཆ་ཚང་ཞིག་ལེན་པའི་ཕྱིར་དུ་འོང་བ”ཡིན་ཟེར། ཁོ་མོས་བཤད་ན། “དེ་ནི་བླ་མ་མཁའ་ལྡིང་རིན་པོ་ཆེ་ཚང་གིས་སྐབས་དེར་ང་ལ་གནང་བའི་བཀའ་གདམས་ཡིན་པས་ད་རེས་ཁོང་གི་བཅའ་བ་འགྲུབ་དགོས”ཟེར། “འོ་ན། བླ་མ་རིན་པོ་ཆེ་ད་ལྟ་གང་ན་བཞུགས་སམ”ངས་བསྲན་མ་ཚུགས་པར་ཁོ་མོའི་སྐད་ཆའི་མཚམས་བཅད་ནས་དྲིས་པ་ན། ཁོ་མོས་ལག་པ་གཡས་པས་མ་ཎི་བསྐོར་ཞིང་གཡོན་པས་ཕྲེང་བ་ཉུལ་བཞིན“ཁོང་སྐུ་མྱ་ངན་ལས་འདས་སོང”ཟེར། ཁོ་མོས་བཤད་ན། མོ་རང་ལོ་སོ་བརྒྱད་ལ་སོན་པའི་དགུན་ཁ་ཞིག་ལ་མཁའ་ལྡིང་རིན་པོ་ཆེ་ཁོ་ཚང་ལ་རིམ་གྲོ་སྒྲུབ་ཏུ་ཕེབས་མྱོང་ཞིང་། ཉིན་མོ་དེ་ནི་ངས་རང་གི་གཅིན་རི་བསང་ཕུད་སའི་དར་ལྕོག་གི་འགྲམ་ལ་བསྐྱལ་བའི་ཆེས་མཐའ་མཇུག་གི་ཉིན་མོ་ཡིན། ཉིན་མོ་དེ་ནས་བཟུང་ངས་ཁོང་ཚང་ཐེངས་གཅིག་ལ་ཡང་མཐོང་མ་མྱོང་། “ང་ཚོ་ལ་ཁོང་གི་བསོད་ནམས་གང་ན་ཡོད”ཟེར། ང་ལ་སྐབས་དེའི་ཁོ་མོའི་ཁྱོ་ཀ་མཐུ་སྟོབས་ཀྱི་སེམས་ཁམས་སོགས་དྲི་བ་མང་པོ་ཞིག་ངག་གི་སྒོ་ཁར་འཚང་ཁ་རྒྱག་ན་ཡང་། ཁོ་མོས་མ་ཎི་སྐོར་ཞོར་དུ་སྐད་ཆའི་འཕྲོ་གླེང་བྱུང་། བླ་མ་མཁའ་ལྡིང་རིན་པོ་ཆེ་ཚང་ནི་ང་ཚོའི་ཚེ་འདི་ཕྱི་གཉིས་ཀྱི་སྐྱབས་གནས་ཡིན། རུ་སྡེ་བ་གཉིས་ཀྱི་ནོར་བུ་ཡིན། ཕྱིས་སུ་ངས་གོ་ཐོས་ལྟར་ན། “རུ་གོང་མའི་སྡེ་དཔོན་གྱི་ཆུང་མ་ལ་ཡང་མངལ་མ་ཆགས་པས། ཁོང་གིས་རིམ་གྲོ་སྒྲུབ་པའི་མཐུ་ལ་བརྟེན་ནས་བུ་མོ་ཡག་མ་མེ་ཏོག་འདྲ་ཞིག་སྐྱེས་འདུག”ཟེར། ཁོང་ཚང་ཕེབས་སྔ་བ་ཡང་ངེད་ཅག་གི་མ་དག་པའི་དུག་གྲིབ་ཕོག་པས་ཡིན་ན་ཡང་ཐང་། “བླ་མ་བཀའ་དྲིན་ཅན་ཁོང་ཨང”ཞེས་མིག་ཟུང་བཙུམས་བྱུང་། སྐབས་དེར་བར་སྐོར་ལམ་དུ་མཇལ་བ་ཇེ་མང་ལ་སོང་ཞིང་། ཇོ་མཇལ་ཞུ་མཁན་ཚོས་རང་འགུལ་གྱིས་པང་སྒྲིག་གིན་འདུག་ལ། བསྐོར་ཕྱག་བྱེད་མཁན་ཚོ་ཡང་རིམ་བཞིན་ཇེ་མང་ནས་ཇེ་མང་དུ་གྱུར། “ཨ་མ་གཡུ་སྒྲོན་ལགས། ང་རང་འགྲོ་རྒྱུ་ཡིན།” “ཁྱེད་ཀྱིས་ཇོ་མཇལ་མི་བྱེད་དམ།” “ངའི་ཁོག་པ་གཤལ་རྩིས་ཡོད་པ་འདྲ”ཞེས་ཁོ་མོར་རྫུན་གཏམ་ཞིག་སྨྲས་རྗེས་ཕྱིར་མགྲོན་ཁང་ལ་ཆས། ཁོ་མོའི་གཏམ་རྒྱུད་ལ་ཉན་པར་མགོ་འཁོར་ནས་ནམ་སྨད་ཀྱི་ཆུ་ཚོད་བཞི་དང་ཕྱེད་ཀ་ལ་བསླེབས་འདུག ངས་བར་སྐོར་ལམ་བརྒྱུད་ནས་ཕྱིར་མགྲོན་ཁང་ལ་ལོག་པའི་ལམ་བར་དུ། ཁ་བཏགས་འཚོང་མཁན་གྱི་བུ་མོ་ཆུང་ཆུང་ཞིག་བརྒྱུགས་འོང་ནས“རྒན་ལགས། ཁ་བཏགས་ཤིག་ཉོས་ཨང”ཟེར། ལྷ་སའི་ལྭ་བ་གྱོན་པའི་ཨ་ཅག་ཅིག་བརྒྱུགས་འོང་ནས“རྒན་ལགས། དར་ལྕོག་ཅིག་ཉོས་ཨང”ཟེར། ངའི་ཚོར་འདུ་སྣ་ཁ་སུམ་བརྒྱ་དྲུག་ཅུ་ཡོད་དོ་ཅོག་ཨ་མ་གཡུ་སྒྲོན་གྱི་གཅིན་པའི་སྟེང་ལ་འཁོར་ཡོད་པས། ཁོ་ཚོར་ཁ་ཡ་བྱེད་འདོད་ཀྱི་སེམས་ཁམས་ཤིག་ཁྱོན་ནས་མ་བྱུང་། མགྲོན་ཁང་དུ་ཐོན་དུས་ཆུ་ཚོད་ལྔ་བར་བསླེབས་འདུག ང་ལ་མཚོན་ན་མལ་དུ་མ་ལོག་པའི་ཡར་སྔོན་དུ་ཁྲུས་རྒྱག་པའི་སྲོལ་ཡོད་མོད། ཆད་སྣང་དྲག་པོ་ཞིག་གིས་ཉོབ་ཡོད་པས་གོན་པ་དག་ཡར་འཕེན་མར་འཕེན་བྱས་རྗེས་ཉལ་བ་ཡིན།
