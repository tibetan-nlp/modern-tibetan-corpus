མདོ་བའི་བུ་མོ།
བྱ་མདོ་མར་ལོ།

དེ་ནས་བཟུང་། ངའི་དུས་ཚོད་ཀྱི་མཚེའུ་ངོགས་སུ། གཏམ་རྒྱུད་དེའི་ཡལ་ག་ནས་སྨིན་ཐག་ཆོད་པའི་ལོ་འདབ་སྐམ་པོ་གཉིས་ཟགས་འོང་།དེ་ནི་མི་རྟག་པའི་ཆོས་ཉིད་ལྡང་བའི་ང་རོ་ཞིག་ཡིན་ནའང་།ཆུ་རུ་ཞུགས་པའི་ལོ་ཟླ་དེ་རྒྱང་ལ་བཞུར་བ་ནས་བཟུང་། ངས་དགོད་སྒྲའི་ངོགས་ཀྱི་ཉིན་མཚན་ཞིག་གི་འདས་སོང་སྲུང་ནས་སྡོད་དགོས་བྱུང་། ཁེར་རྐྱང་དུ་སྡོད་དགོས་པ་དང་ཁེར་རྐྱང་གིས་བཞུད་དགོས་པ་འདི་ཆོས་ཉིད་ཅིག་དང་འདྲ་བས། དང་ཐོག་འཇིག་རྟེན་འདི་རུ་འདུ་འཛོམས་ཀྱི་བྱུང་རིམ་ཞིག་ལ་དབུ་ཁྱུད་བྲིས་པ་ཅིའི་ཕྱིར་ཡིན་ནམ། མི་རུ་སྐྱེས་པ་ནས་བཟུང་། དགའ་བའི་རི་མོ་འཕེལ་བའི་བརྒྱུད་རིམ་ ཞིག་གི་སྣེ་རུ་སྡུག་གི་ཚོན་རྩི་ཐུར་ལ་བསྣར་ཞིང་།ཞི་བའི་སྤྲིན་པ་འཁྲིགས་པའི་མཛེས་སྡུག་ཅིག་གི་མཐའ་ལ་ཁྲོ་བའི་འབྲུག་སྐད་རབ་ཏུ་ལྡིར་བའི་ཆོས་ཉིད་འདི་ཡང་སྙིང་སུ་ཞིག་གི་དགོས་མཁོ་ཡིན་ནམ། མི་ལོ་ཉི་ཤུའི་སྲད་པུར་བསྟར་པའི་གཏམ་རྒྱུད་ཀྱི་ཕྲེང་རྡོག་ཅིག་སྟེ།བདག་འཛིན་གྱི་ཞགས་པས་དམ་དུ་བཅིངས་སྙིང་འདོད་ པའི་མཛེས་སྡུག་ཁྱོད། དེ་ནས་བཟུང་དུས་ཚོད་ཀྱིས་བསླེས་པའི་ལས་དབང་གི་རྩི་སྐུད་ནག་པོ་ཞིག་གི་སྣེ་རུ་བརྒྱུས་པས། ངས་ཐོབ་ཤོར་གྱི་སོ་ཁ་འཐབ་སའི་སྲིད་པའི་གྲུ་ག་འདི་རུ། ཁེར་རྐྱང་དང་གདུང་སེམས།སྡུག་བསྔལ་དང་ཞེ་འགྱོད་ཅེས་པའི་ཐ་སྙད་སྐྱུར་བོ་མང་པོ་ཞིག་ངོ་ཤེས་ཤིང་། ཕུང་བོ་འདིའི་ཁུད་དུ་རྒྱུ་བའི་ཟུངས་ཁྲག་གི་དྲོད་ཁོལ་གྱིས་སྐྱུར་ཞིང་བསྐ་བའི་ཐ་ སྙད་ཀྱི་ཁེར་རྐྱང་དང་གདུང་སེམས་ཆེན་པོ་དག་ལ་ལག་འཇུ་བྱེད་དགོས་བྱུང་། དེ་ནས་བཟུང་། བར་སྣང་གི་སྤྲིན་པ་ལས་ཁ་ཆར་བརྡོལ་རྒྱུ་ཟད་ཅིང་། ས་སྟེང་གི་རྒྱ་མཚོ་ལས་བརླན་ གཤེར་འཛད་ལ།རི་རབ་དང་ལྗོན་ཤིང་ཐམས་ཅད་ལ་དབྱར་ཁ་སླེབས་མ་བྱུང་། དེ་ནི་སྨྲ་བ་འཇགས་ཤིང་རང་ཉིད་བོར་བའི་ཉམ་ང་བའི་མི་ཞིག་གི་ཁེར་རྐྱང་དང་ཅི་ འདྲའི་མཚུངས་ཨང་། དེ་ནས་བཟུང་། མྱི་ཁའི་འཐོར་རླུང་ལ་མངོན་སེམས་འཛེམ་མེད་ཀྱི་ལག་པ་གཉིས་སྐྱེས་འོངས་ཤིང་། ངའི་འཚོ་བའི་ཉིན་མཚན་གྱི་གདགས་སྲིབ་གཉིས་ལ་ན་ཟུག་དང་ཞེ་འགྲས་ཀྱི་མུན་པ་བསྐྱིལ་སོང་།མྱི་ཁས་འཚོས་པའི་ངའི་ལུས་སེམས་ཀྱི་རི་ལུང་ཐང་གསུམ་དུ་མུན་རུབ་ཅིང་།མུན་པའི་ཁྲོད་ཀྱི་གྲང་ངར་ནི་ཁྱོད་ལ་སྤྲུལ་ཏེ་ངའི་ལུས་སེམས་ལ་འཁྱག་སིབ་ སིབ་ཀྱིས་གླན་ཡོང་། རེ་ཆད་ཀྱི་སེམས་ཁར་ཉུལ་འོངས་པའི་ ཚོར་བ་སྐྱུར་པོ་དག་ན་ཟུག་དང་འདྲེས་ ཤིང་།ན་ཟུག་གི་ཁུད་དུ་གཅུད་པའི་ཁྱོད་ ཕྱོགས་ལ་བདག་གི་འཁྲེང་ཞེན་ཡོངས་སུ་བཞུར་སོང་། ཡང་ན། དེ་ནས་བཟུང་། ལམ་ཕྲན་གཅིག་གི་སྣེ་རུ་འབྱོར་བའི་འུ་གཉིས་ཀྱིས་དྲོད་འཇམ་གྱི་མལ་སར་གཏོར་མྱོང་བའི་ཞེ་འདང་དག་ལ་བསྡུ་གསོག་རྒྱག་དགོས་ཤིང་།ལག་ཟུང་སྦྲེལ་བའི་ཧྲག་བར་ནས་འབུས་མྱོང་བའི་བརྩེ་བའི་མེ་ ཏོག་གི་འདབ་ལོ་དག་ས་རུ་གཏོར་དགོས་སོང་། ཡང་ན། དེ་ནས་བཟུང་། ཕྱིན་ཆད།འདུ་འབྲལ་གྱི་ཕྱིར་དུ་གཏོང་ དགོས་པའི་དགའ་འབོད་དང་སྐྱོ་སྣང་ཡོད་ཚད་ལ་བཙན་གྱིས་བཏང་སྙོམས་ཀྱི་བཞི་ཤད་བཀོད་པས།ཁྱོད་རང་ཁྱོད་ཀྱི་ལམ་ ཕྲན་ལ་ཞུགས་ཤིང་། ང་རང་ང་ཡི་གྲིབ་མ་དང་འགྲོགས་ཏེ་གདུང་སེམས་ཀྱི་ལ་མོར་འགོས།མི་ཚེ་མཇུག་རྫོགས་དགོས་ཕྱིར་འགོ་རྩོམ་དགོས་པ་ནི་སུ་ཞིག་གི་ནོར་འཁྲུལ་ ཡིན་མིན་མི་གསལ་རུང་།འཇིག་རྟེན་དཀར་ངོགས་སུ་འཚོ་དགོས་པའི་ཚེ་སྲོག་ཐམས་ཅད་ལ་ངོ་བོ་ཉིད་ཀྱིས་གནས་པའི་རང་ བཞིན་ཞིག་དཀོན་ཕྱིར།དབུགས་ཀྱི་རྒྱུ་བ་མ་ཟད་པའི་སྔ་རོལ་ཏུ།ངས་ཁྱོད་རང་ མཚན་མོ་གཞན་ཞིག་གི་ཁ་ཟས་སུ་གྱུར་ ཡོད་པ་མཐོང་སོང་ངོ་།།

(མཆོད་མེའི་པར་དབང་གཞན་གྱིས་དགོལ་མི་རུང)

རྩོམ་ཁུངས། ངེད་དྲ་བ།
