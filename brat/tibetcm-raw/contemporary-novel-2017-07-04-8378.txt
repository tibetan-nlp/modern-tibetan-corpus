ཀླད་ཀོར་གྱི་བརྩམས་སྒྲུང་གསར་བ། དཔྱིད་ཀ་འཁོར་བ།
ཀླད་ཀོར།(སྨར་ཁམས)

ཟླ་བ་བཀྲ་ཤིས་སྡོད་ཁང་གི་སྒོ་ནས་ཕྱི་ལ་ཐོན་མ་ཐག་འཇིག་རྟེན་གྱི་ཡོད་ཚད་དཀར་ལྷང་ལྷང་དུ་གྱུར་ཡོད་པ་མཐོང་བ་དང་ཆང་དུག་ཕོག་ཡོད་པའི་མགོ་བོ་དེའང་ཅུང་གསལ་པོར་གྱུར། མདང་དགོང་ཁོ་དང་གྲོགས་པོ་ཁ་ཤས‘མཚོ་ཆེན་པོ’ཞེས་པའི་ན་མ་ཁང་དུ་དུས་ལྟར་འཛོགས་པ་དྲན་རྒྱུ་ཡོད་ཀྱང་མཚན་མོར་ན་མ་ཁང་ནས་སྡོད་གནས་སུ་ཅི་ལྟར་ལོག་པ་མགོ་བོ་ལག་པས་བརྡུངས་ཀྱང་ཅི་ཡང་དྲན་མ་ཐུབ། མུན་ཁྲོད་དུ་ཁ་བ་འབབ་པའི་ཚུལ་ནི་དེ་བས་ཀྱང་དྲན་རྒྱུ་མི་འདུག་པས་ཞོགས་པར་ཁ་བ་གློ་བུར་མཐོང་མ་ཐག་རྨི་ལམ་ནས་རྨི་ལམ་གཞན་ཞིག་ཏུ་ལྷུང་སོང་བའི་སྣང་བ་ཤར། ཁོ་རང་ཁ་བའི་དྭངས་གཙང་ཁྲོད་ནས་རྡོག་སྒྲ་གྲ་གྲ་ཞེས་གྲག་བཞིན་ལས་ཁུངས་སུ་འགྲོ་བའི་ལམ་བར་དུ་རང་ཉིད་ཀྱིས་ད་ལྟ་རོལ་བཞིན་པའི་འཚོ་བར་ཕྱི་ལྟ་ཞིག་བྱེད་དགོས་བྱུང་། དུས་རྒྱུན་ཡིན་ན་ཁོའི་སེམས་ནང་དོན་དག་མང་པོ་ཞིག་འཚང་ཡོད་པས་རང་ཉིད་ཀྱི་འཚོ་བར་ལ་ཕྱི་དྲན་བྱ་རྒྱུ་ཕར་ཞོག སྐབས་སྐབས་སུ་ཆང་དུག་ཕོག་ནས་གཟེར་བཞིན་པའི་མགོའི་ན་ཚ་དེའང་བརྗེད་འགྲོ་གི་ཡོད། ཡིན་ཡང་། དེ་རིང་ཁོས་ལས་ཁུངས་ནང་གི་དོན་དག་དང་མགོའི་ན་ཟུག་ཕར་བཞག་ནས་རང་ཉིད་ཀྱི་རོལ་བཞིན་པའི་འཚོ་བ་འདི་ཅི་འདྲ་ཞིག་ཡིན་ནམ་སྙམ། ཁོ་ལ་མཚོན་ན་ལས་ཁུངས་ཀྱི་བྱ་བ་རྣམས་ལ་འཆར་གཞི་གང་ཡང་མེད་མོད། ཁོས་ཁ་སང་ངམ་གཟའ་འཁོར་འགའི་སྔོན་དུ་བྱུང་བའི་སྙིང་བོ་ཅི་ཡང་མེད་པའི་ལས་ཁུངས་ཀྱི་དོན་དག་ཁ་ཤས་ཡང་ཡང་དུ་དྲན་འོང་ཞིང་རིམ་གྱིས་རྡོག་ཁྲེས་སུ་གྲུར། ཉིན་ཞིག་གི་མཚན་མོར་ཁོས་རྒྱལ་སྒོ་ཕྱེ་ནས་བཞག་ཡོད་དུས་ལས་ཁུངས་ཀྱི་དཔོན་པོས“ཆུ་ཚོད་བཅུ་པར་ཡིན་གཅིག་མིན་གཉིས་ལས་ཁུངས་ཀྱི་རྒྱལ་སྒོ་དེ་རྒྱག་དགོས་ནི་རེད”ཅེས་བཤད་དུས་ཀྱི་རྣམ་འགྱུར་དེ་དྲན། ཉི་མ་གཉིས་འདས་རྗེས་སུ་རྣམ་འགྱུར་ཕལ་བ་ཞིག་ཡིན་མོད། གཟའ་འཁོར་ཁ་ཤས་ཀྱི་རྗེས་སུ་རྣམ་འགྱུར་དེ་སླར་དྲན་སྐབས“ཆུ་ཚོད་བཅུ་པར་ཟིན་ཡང་ཁྱོས་ད་དུང་སྒོ་མི་རྒྱག་དོན་ཅི་ཡིན”ཞེས་པའི་དོན་དག་ལྷག་མ་ཞིག་གོ་སོང་། དྲང་པོར་བཤད་ན་ཁོ་ལ་ལས་ཀ་ཞིག་ཡོད་ཀྱང་དོན་དག་གང་ཡང་མེད། ཁོ་རང་དང་པོ་ལས་ཀར་བཞུགས་དུས་དཔོན་པོས་གློག་ཀླད་ནང་དུ་རྒྱ་ཡིག་འབྲི་རུ་བཅུག ཁོས་དངངས་སྐྲག་ཆེན་པོས་གློག་ཀླད་དེ་འཛུབ་མོས་ཆུ་ཚོད་ཕྱེད་ཙམ་བརྡུངས་ཀྱང་ཡིག་འབྲུ་ཁ་ཤས་ལས་འབྲི་ཐུབ་མ་སོང་། དཔོན་པོས་ཚུལ་དེ་མཐོང་ནས“ཁྱོས་ཅི་ཡང་མི་ཤེས་ན་སྒོ་སྲུང་བྱོས”ཟེར། སྒོ་སྲུང་བའི་ཟླ་བ་ཁ་ཤས་ཀྱི་རིང་ཁོ་ལ་དོན་དག་གང་ཡང་མེད། སྒོ་ཉིན་གཅིག་སྲུང་ཚར་དུས་ངལ་གསོ་ཉིན་གསུམ་ཡོད། ཉིན་གསུམ་དེ་བེད་སྤྱད་ནས་གྲོགས་པོ་མང་པོ་དང་མཉམ་དུ་ཇ་ཆང་ལ་རོལ། ཁོས་ཇ་ཆང་ལ་རོལ་བའིི་དུས་སུ་ལས་ཁུངས་ཀྱི་སྒོ་ཁའི་ཁང་པ་ཆུང་ཆུང་དེ་ནང་ཙོག་པོར་བསྡད་རྒྱུ་ནི་དོན་དག་ཅིག་ཡིན་པའི་སྣང་བ་ཤར། དེའི་ཕྱིར་ཆང་ཁང‘མཚོ་ཆེན་པོ’ཡི་ནང་དུ་གྲོགས་པོས་ཆང་ཕོར་བཀྱགས་ཡོང་དུས“ད་ངས་འཐུང་མི་ཉན། སང་ཉིན་ལས་ཀ་ཡོད”ཅེས་བཤད་ཀྱི་ཡོད། ཡིན་ཡང་། སང་ཉིན་གྱི་ལས་ཀ་ནི་སྒོ་ཁའི་ཁང་པ་ཆུང་ཆུང་དེའི་ནང་དུ་ཙོག་ནས་སྒོ་ནས་ཡར་ཡོང་བའི་མི་རྣམས་ཀྱི་རྣམ་འགྱུར་དང་མོ་ཊའི་ཚོང་རྟགས་ལ་ལྟ་བཞིན་ཉི་མ་གཏོང་ཐབས་བྱ་རྒྱུ་ལས་གཞན་དོན་དག་ཅི་ཡང་མེད། དོན་དག་ཅིག་ངོ་མ་བགྲང་དགོས་ན་ལམ་ནོར་ཤོར་ཏེ་ལས་ཁུངས་ཀྱི་སྒོ་ནས་ཡར་འཛུལ་ཡོང་བའི་འབྲོག་པ་དེ་ཚོ་མར་བདས་དགོས་པ་དེ་རེད། པུའུ་ཀྲང་གིས་ཁོ་ལ“སྒོ་ནས་མི་ཆ་མེད་ཅིག་ཡར་འཛུལ་ཡོང་ཚེ་ཐོབ་ཐང་ལག་ཁྱེར་སྟོན་དུ་བཅུག་དགོས། ཁྱོད་ཀྱི་ལས་ཀ་ནི་མི་དེ་དག་གི་མིང་ཐོ་གསལ་བོར་འགོད་དགོས། གཞན་ཡང་ཁལ་མ་དང་ཕག་འདྲ་ཡར་འཛུལ་ཡོང་ན་མར་བདའ་དགོས”ཅེས་ཁོ་ལ་ཤོག་གུ་དང་ཞ་སྨྱུག ཞ་སྨྱུག་གཞོག་བྱེད་ཀྱི་གྲི་ཆུང་མང་པོ་སྤྲད་བྱུང་། ཁོ་རང་ནི་ལས་མི་གསར་པ་ཞིག་ཡིན་པས་སྒོ་ནས་ཡར་འཛུལ་ཡོང་མཁན་ཐམས་ཅད་མི་ཆ་མེད་དག་དག་རེད། དེའི་ཕྱིར་ཁོས་ཤོག་གུ་དག་སྐྱག་པ་ཕྱིས་ས་བྱེད་ཅིང་གྲི་ཆུང་ཆུང་དེ་དག་གིས་ཞ་སྨྱུག་ཡང་ཡང་དུ་བཞོགས་ནས་ཟད་དུ་བཅུག་ཏེ་དུས་ཚོད་གཏོང་ཅི་ཐུབ་བྱས། སྐབས་རེར་ཁོས་ལམ་ནོར་ཤོར་བའི་འབྲོག་པ་རྣམས་མར་བདས་རྗེས་སུ་ཁྱི་ཕག་རེ་གཉིས་ཅི་ཞིག་འཚོལ་ཉམས་ཀྱིས་སྒོ་ནས་ཐོན་འོང་ནའང་ཁོས་དེ་དག་ལ་ཁ་ཡ་མི་བྱེད་པར་ཡར་འཛུལ་དུ་བཅུག ཁོ་ཚོ་ཐེངས་མང་པོ་ཡར་འཛུལ་འོང་ནས་གོམས་ལོག་ཏུ་སོང་བ་འདྲ་སྟེ་ད་ནི་ཁོ་ལ་སྐྲག་སྣང་ཅི་ཡང་མེད་པར་མགོ་བོ་ཡར་བཀྱགས་ནས་འཛུལ་འོང་། ཁྱི་ཕག་དེ་ཚོའི་རྣམ་འགྱུར་ནི་ལས་ཁུངས་གི་དཔོན་པོ་དག་གི་རྣམ་འགྱུར་དང་ཧ་ཅང་འདྲ་བས་ཁོས་རྒྱ་ཕག་དཀར་པོ་དེ་ལ“པུའུ་ཀྲང་ཕག་ལགས”ཅེས་འབོད། ཁྱི་མོ་འཚབ་ཤིག་ཤིག་དེ་ལ“དྲུང་ཡིག་ཁྱི་ལགས”ཅེས་འབོད། ཡིན་ཡང་དེ་ཚོས་ཀྱང་ཁོ་ལ་ལན་ཅི་ཡང་མི་བརྒྱབ་པར་རང་རང་གི་དམིགས་སར་ཕྱིན་འགྲོ ད་དུང་ཁྱི་ཡིན་ན་ཕག་ཡིན་ན་གཅིག་ནད་ཕོག་ནས་ལུས་ཀྱི་སྤུ་ཡོད་ཚད་བྱི་འདུག དེའི་ལུས་པོ་ཕག་དང་འདྲ་ཡང་མཆུ་ཏོ་ཁྱི་དང་འདྲ། ཁོས་དེ་ལ་མིང་ཅི་ཞིག་བཏགས་པར་ཡང་ཡང་དུ་དྲན་མོད་ཀྱང་འོས་ཤིང་འཚམ་པའི་མིང་ཞིག་རྙེད་མ་སོང་། ཁོ་རང་ཁ་བའི་ཁྲོད་དུ་བསྐྱོད་བཞིན་འདི་ལྟར་དྲན་ཞིང་རང་ཉིད་ཀྱི་འཚོ་བ་རོལ་སྟངས་འདི་འདྲ་བ་ལ་སྙིང་བོ་ཅི་ཞིག་ཡོད་དམ་སྙམ། དུས་རྒྱུན་ཁོ་རང་ཆང་འཐུང་བར་མི་འགྲོ་བར་སྟོང་ལྷང་ལྷང་གི་ཁང་པ་ནང་ཁེར་པོར་བསྡད་དུས་ཀྱང་ཁོམ་པ་ཞིག་བྱུང་མ་མྱོང་། སྐད་འཕྲིན་གྱི་ནང་དུ་ཁོ་ལ་དགའ་རོགས་གསུམ་ཡོད་ཅིང་ཁོ་ཚོའི་དྲི་བ་རྙོག་ཟིང་ཅན་དག་ལ་ལན་བརྒྱབ་ཏེ་བརྩེ་དུང་གི་གནད་དོན་ཆུང་ཆུང་ཞིག་ལའང་་འགྲེལ་བཤད་ཡུན་རིང་བྱེད་དགོས། ཁོ་ཚོས“ང་ཁྱོད་ལ་དགའ”ཟེར་ཡོང་དུས་ཁོས་ཀྱང“ང་ཁྱོད་ལ་དགའ”ཞེས་ལན་རྒྱག་དགོས། དེ་འདྲའི་འཚོ་བ་ལ་སྙིང་བོ་ཅི་ཞིག་ཡོད་དམ། ཁོ་རང“ང་ཁྱོད་ལ་དགའ”ཟེར་མཁན་གྱི་བུ་མོ་དེ་ཚོར་རབ་ཏུ་སུན། ཁོ་རང་ད་བར་བུ་མོ་སུ་ལའང་དགའ་མ་མྱོང་། ཡང་གཅིག་བཤད་ན་མི་སུ་ལའང་དགའ་མ་མྱོང་། ཁོ་རང་ཆུང་དུས་ནས་ཁེར་རྐྱང་གི་འདུ་ཤེས་ཟབ་མོ་ལྡན་ཞིང་ངོ་མི་ཤེས་པ་དང་རྒྱུས་མེད་པའི་མི་དང་མཉམ་དུ་བསྡད་དུས་དངངས་སྐྲག་གིས་ལུས་སེམས་འདར་འོང་། བྱིས་དུས་ནས་ཁོ་རང་ཁེར་འགྲོ་ཁེར་འདུག་གི་འཚོ་བ་ལ་ཡོངས་སུ་གོམས་ཡོད། རང་ཉིད་ཀྱི་སྡུག་བསྔལ་དག་མི་སུ་ཞིག་གི་རོགས་རམ་གྱི་འོག་ཏུ་སེལ་ཐུབ་པ་ཞིག་མིན་པས་ཡིན་གཅིག་མིན་གཉིས་རང་ཁེར་པོས་གདོང་གཏོད་དགོས་པར་འདོད། ཡིན་ཡང་ཁོ་ལ་ཆང་གྲོགས་ཤིན་ཏུ་མང་། ཁོ་རང་ཆང་གྲོགས་ཚོ་ཆང་གིས་བད་རྗེས་བཤད་པའི་སྐད་ཆ་མཛེས་པོ་དེ་ཚོ་ལ་སུན་གྱིན་མེད། སྐད་དེ་དག་ནི་ཞོགས་ཀྱི་ཟིལ་བ་དང་འདྲ་བར་ཆང་དྲི་ཁ་བའི་མཁའ་དབུགས་ནས་ཁད་ཀྱིས་ཡལ་འགྲོ། ཁ་བའི་ཁྲོད་དུ་ཁོས་དུས་ད་ལྟ་བའི་རང་ཉིད་ནི་མངོན་སུམ་གྱིས་འགྲུབ་པའི་བདེན་པ་ཉག་གཅིག་དེ་ཡིན་པའི་སྣང་བ་སྐྱེས། ཆུང་དུས་ཁོས་འདི་ལྟར་འཇིག་རྟེན་དང་རང་ཉིད་གཉིས་ཀ་དྭངས་གསལ་དུ་མཐོང་གི་ཡོད། ལོ་ན་ཇེ་ཆེར་སོང་བ་དང་འཇིག་རྟེན་འདི་ནི་མག་མོག་ཏུ་གྱུར་སོང་། དུས་ད་ལྟ་བའི་རང་ཉིད་ཀྱང་འདས་མ་འོངས་ཀྱི་རང་ཉིད་དང་འདྲ་བར་མག་མོག་རབ་རིབ་ཏུ་སྣང་། འཇིག་རྟེན་ནི་སྟོང་པ་དང་ཁོ་རང་ཉིད་ཀྱི་ཀླད་པ་ཡང་ཤ་རྡོག་སྟོང་པ་ཞིག་ལས་མ་མདས་སོ་སྙམ། ཡིན་ཡང་དེ་རིང་གི་ཁ་བའི་ཁྲོད་དུ་དུས་ད་ལྟ་བའི་རང་ཉིད་ནི་འཇིག་རྟེན་འདིའི་ནང་གི་བདེན་པ་ཉག་གཅིག་དེ་ཡིན་པའི་ངེས་ཤེས་རྙེད་ཅིང་ད་ནས་བཟུང་སྟོང་པའི་ངོ་བོ་ཅན་གྱི་འདོད་པའི་རྗེས་སུ་མི་འགྲོའོ་སྙམ། སྔོན་མར་ཁོས་འདི་ལྟར་མང་དུ་བསམ་མྱོང་ཡང་དེ་རིང་གི་དམ་བཅའ་འདི་ཁ་བས་ཇེ་ལྗེད་དུ་བཏང་སོང་བའི་སྣང་བ་ཤར། དུས་རྒྱུན་གླ་གཏོང་རླངས་འཁོར་གྱི་ཁེངས་པའི་གཞུང་ལམ་དེ་དེ་རིང་སྟོང་ལྷང་ལྷང་ངང་ཁ་བ་གྱོན་ཏེ་ཉལ་འདུག ཁོ་རང་མཛེས་སྡུག་གི་ཁ་བའི་ཁྲོད་དུ་འདི་ལྟར་དྲན་སྣང་གིས་སྟས་ཏེ་དགའ་ལྷང་ལྷང་གིས་བསྐྱོད་པའི་དུས་སུ་ཟུར་ག་ཞིག་ཏུ་མི་མང་པོའི་ཟང་ཟིང་བྱེད་པའི་རྣམ་པས་ཁོ་རང་ལྷ་ཡུལ་ནས་མི་ཡུལ་དུ་དྲངས་སོང་། ཁ་བས་མི་ཡུལ་འདི་ལྷ་ཡུལ་དུ་བསྒྱུར་མི་ཐུབ། མི་རྣམས་སྔར་བཞིན་ཁྲག་རྣག་གི་བར་དུ་ཚོར་ཡོད། མི་ཡུལ་གྱི་ཟང་ཟིང་གི་ཁོ་རང་ཁ་བའི་མཛེས་སྡུག་ལས་སད་དུ་བཅུག་ཅིང་མི་ཡུལ་དུ་སླར་ཡང་དོན་དག་ཅི་ཞིག་འབྱུང་སོང་སྙམ་པའི་མཚར་ནང་གིས་ཁོ་ཡང་མི་མང་ཚོགས་སར་སོང་། མི་མང་པོ་ཞིག་གིས་དངོས་པོ་ཅི་ཞིག་ལ་ལྟ་བཞིན་གཏམ་གླེང་བྱེད་བཞིན་འདུག ཁོ་རང་མི་མང་པོའི་དཀྱིལ་དུ་སླེབས་ཅིང་ཁོ་ཚོས་གླེང་བྱའི་བྱ་དངོས་དེའང་མིག་ལམ་དུ་ཤར། མི་ཞིག་ཁ་བའི་ཁྲོད་དུ་འགྱེལ་འདུག ཁོའི་ཉེ་འགྲམ་གྱི་ཁ་བ་ཡོད་ཚད་ཁྲག་གིས་དམར། ཁོས་ད་བར་མའི་ཕུང་པོ་མཐོང་མ་མྱོང་། ཡིན་ཡང་ད་ལྟ་མཐོང་བའི་མིའི་ཕུང་པོ་དེ་བལྟས་མི་བཟོད་པ་ཞིག་རེད། མི་དེའི་ངོ་གདོང་ཡོད་ཚད་ཁྲག་གིས་དམར་འདུག དེའི་ཐོག་ཏུ་ཁ་བ་འབབ་པ་དང་འཁྱགས་ནས་ཆབ་རོམ་ཆགས་ཅིང་མིག་དང་། སྣ། རྣ། མཆུ་དང་བཅས་པ་ཁྲག་གིས་སྦགས་པའི་ཆབ་རོམ་གྱིས་བསྒྲིབས་ཅིང་ལུས་སྨད་ཡོངས་རྫོགས་ཁ་བའི་ནང་དུ་ནུབ་འདུག ཁོས་ཐེངས་དང་པོར་མཐོང་བའི་མིའི་རོ་ནི་འདི་འདྲ་བལྟས་མི་བཟོད་པ་ཞིག་ཡིན་པས་མཐོང་མ་ཐག་ནས་སྐྱུག་པ་ཁད་ཀྱིས་ཁོག་སྨད་ནས་མགྲིན་པར་འཚང་འོང་། ཁོ་རང་དེ་མུར་གཞུང་ལམ་གྱི་ཕྱོགས་སུ་བརྒྱུགས། ཁོ་རང་གཞུང་ལམ་ཁར་སླེབས་དུས་མགྲིན་པའི་ནང་གི་སྐྱུག་པ་དག་ཁོག་སྨད་དུ་མར་ལྷུང་བ་འདྲ་སྟེ་ཅི་ལྟར་བསྐྱུགས་ཀྱང་ཁ་ཆུ་མ་གཏོགས་ཅི་ཡང་འོང་རྒྱུ་མི་འདུག འཆི་བ་ནི་འདི་འདྲའི་འཇིགས་སུ་རུང་བ་ཞིག་རེད་ཨང་སྙམ་བཞིན་རང་དབང་མེད་པར་རྐང་པ་འདར་བྱུང་། མི་རོ་དེས་ཁོ་ལ་གློག་བརྙན་ཁ་ཤས་དྲན་དུ་བཅུག་ཅིང་དུས་ཡུན་རིང་པོར་གཏིང་བརྗེད་དུ་སོང་བའི་རང་ཉིད་ཀྱི་ཆེད་ལས་ཀྱང་དྲན་དུ་བཅུག ཁོ་རང་ནི་པེ་ཅིང་གི་ཉེན་རྟོག་སློབ་ཆེན་ནས་ཐོན་པ་ཡིན། སློབ་ཆེན་ལོ་བཞིའི་རིམ་དགེ་རྒན་ཚོས་རྟག་པར“ཉེན་རྟོག་པའི་ལས་ཀ་ནི་མི་གསོན་པོ་བཟུང་རྒྱུ་དེ་མིན་པར་མི་ཤི་བོ་དང་འབྲེལ་བ་བྱ་རྒྱུ་དེ་ཡིན”ཞེས་བཤད་ཀྱིན་ཡོད། ཁོས་སྐད་ཆ་འདིའི་གོ་བ་ལོན་གྱི་མེད་ཀྱང་སློབ་ཆེན་ལོ་བཞི་ནི་འགྱིག་གིས་བཟོ་བའི་མའི་ལུས་ཕུང་དང་རུས་པ་ལ་འབྲེལ་བ་བྱས་ནས་རིམ་གྱིས་རྫོགས། ཁོའི་ཆེད་ལས་ནི་མྱུལ་ཞིབ་ཡིན། དགེ་རྒན་གྱིས་བཤད་པ་བལྟས་ན་ཕུལ་བྱུང་གི་མྱུལ་ཞིབ་པ་ཞིག་བྱེད་དགོས་ན་ངེས་པར་དུ་མིའི་ཕུང་པོ་ལ་རྒྱུས་མངའ་ཡོད་དགོས་ཟེར། ཁོ་རང་འགྱིག་གིས་བཟོ་བའི་མིའིི་དབང་པོ་དང་རུས་པ་དཀར་པོ་རྣམས་ལ་ཤིན་ཏུ་སུན་ཡང་སྐར་གྲངས་དང་བདེ་བླག་ངང་སློབ་གྲྭ་ནས་ཐོན་པའི་ཆེད་དུ་འགྱིག་གི་བཟོས་པའི་མིའི་དབང་པོ་དག་གི་མིང་དང་ཆ་ཤས། བྱེད་ནུས་སོགས་ནན་ཏན་ངང་མི་སྦྱོང་རང་སྦྱོང་རེད། མ་བསམ་ས་ནས་ཁོ་རང་སློབ་གྲྭ་ནས་ཐོན་དུས་དེའི་བསྐོར་གྱི་ཤེས་བྱ་ཕུན་སུམ་ཚོགས་པོ་ཡོད་པར་གྱུར། ཡིན་ཡང་རྒྱ་ཡིག་གི་རྩོམ་སྒྲིག་པུའུ་ཞིག་ཏུ་ལས་ཀ་བགོས་པས་རང་གི་ཆེད་ལས་དང་དེའི་བསྐོར་གྱི་ཤེས་བྱ་རྣམས་ཐག་རིང་བའི་རྨི་ལམ་འདྲ་བོར་གྱུར། ཁོ་ལ་ལས་ཀ་ཅི་ཡང་མེད་ཚེ་མྱུལ་ཞིབ་པའི་གློག་བརྙན་ལ་བལྟ་རྒྱུ་དགའ་ཞིང་དེའི་ནང་དུ་མིག་ཤེལ་ནག་པོ་བརྒྱབ་ཅིང་ལས་ཀར་བྱང་ཆ་ལྡན་པའི་མྱུལ་ཞིབ་པ་དེ་ཚོ་ལ་ཡིད་སྨོན་གྱིན་ཡོད། སྐབས་ཤིག་ལ་ཁོའི་དགའ་རོགས་ཡིན་ཟེར་བའི་སྐད་འཕྲིན་སྒང་གི་བུ་མོ་ཞིག་གིས“རྗེས་མར་ལས་ཀ་ཅི་ཞིག་བྱེད་འདོད་ཡོད”ཅེས་དྲིས་ཡོང་དུས་ཁོས“མྱུལ་ཞིབ་པ་ཞིག་བྱེད་འདོད་ཡོད”ཅེས་ལན་བཏབ། ཁོ་མོས“དེ་ན་ཁྱོད་རང་མི་གསོད་རྒྱུར་དགའ་པོ་ཡོད་པ་རེད”ཅེས་བཤད་མྱོང་། ཁོས“ཉེན་རྟོག་པ་དང་མྱུལ་ཞིབ་པ་ནི་མི་གསོད་མཁན་ཡིན་ནམ”ཞེས་དྲིས་ཀྱང་མོ་རང་དགོད་པ་ལས་ལན་ཅི་ཡང་བརྒྱབ་མ་བྱུང་། བུ་མོ་ཟེར་བ་དེ་གསང་བ་ལྐོག་གྱུར་གྱི་སེམས་ཅན་ཞིག་རེད་ཨང་སྙམ། ཁ་བ་ནང་གིི་མིི་རོ་ཁྲག་གིས་སྦགས་པ་དེ་མཐོང་བའི་དངངས་སྐྲག་རིམ་གྱིས་ཡལ་འགྲོ་བ་དང་རང་ཉིད་ཀྱི་ཆེད་ལས་ཀྱི་ཤེས་བྱ་དག་སླར་དྲན་བྱུང་། དེ་རིང་གི་ཤི་བོ་དེ་ལ་སྙིང་རྗེ་བ་ཡིན་ནའང་འདྲ། ཁོས་ཐབས་ཤེས་དུ་མ་དང་བསྟུན་ནས་མི་གསོད་མཁན་གྱི་ལག་དམར་དག་ཁྲིམས་ལ་སྦྱར་བའང་སྲོག་ཐོག་ཏུ་བཏང་བ་ནི་དོན་སྙིང་ལྡན་པའི་བྱ་བ་ཞིག་རེད་བསམས། ཁོས་མི་གསོད་མཁན་དེ་བཟུང་རྗེས་དེའང་གསོོད་པ་ཡིན་ན་དེ་རིང་བཞིན་དངངས་སྐྲག་དང་སྙིང་ན་བ་ཞིག་ཡོང་མི་སྲིད། གཅིག་བྱེད་དུས་སེམས་པ་དགའ་ནའང་ཐང་སྙམ་གྱིན་མདུན་དུ་བསྐྱོད་ཚེ་ཁོ་རང་ནི་རང་ཉིད་ནི་གློག་བརྙན་ནང་གི་མྱུལ་ཞིབ་པ་དང་གཅིག་འདྲ་ཡིན་པའི་སྣང་བ་ཤར། ཡིན་ཡང་དེ་རིང་རླུང་བུ་ཕྲན་ཙམ་ཡང་མི་འདུག་པས་གློག་བརྙན་ནང་གི་མྱུལ་པ་ནང་བཞིན་ཁོའི་སྟོད་ཐུང་རླུང་གིས་བསྐྱོད་ཀྱིན་མི་འདུག ཁོས་མི་རོ་དེ་རིག་རྗེས་རྐང་འོག་གི་ཁ་བ་སྟུག་པོ་དེའང་རེ་ཞིག་བརྗེད་སོང་ལ་ཐེངས་དང་པོར་མི་སྤྱི་ཅུས་ཀྱི་ཅུས་ཀྲང་ཡིན་པའི་རང་གི་ཞང་བོ་དྲན་ཞིང་ཞང་བོས“ལས་ཀ་ཅི་ཞིག་བྱེད་འདོད་ན་ང་ལ་བརྡ་ཞིག་རྒྱོབ”ཟེར་བའི་སྐད་དེ་དྲན། སྐབས་དེར་ཁོ་རང་ལས་ཀ་བགོས་མ་ཐག་ཡིན་པས་རང་གི་རྩོལ་བར་བརྟེན་ནས་ཞང་བོར་ངུ་ནོམ་མི་དགོས་པ་ཞིག་ཏུ་བཟོ་དགོས་བསམས། ལས་ཀ་བགོས་མ་ཐག་གློག་ཀླད་དུ་རྒྱ་ཡིག་བཏབ་མ་ཐུབ་པས་དཔོན་པོས་ཁོ་རང་ལས་ཁུངས་ཀྱི་སྒོ་སྲུང་དུ་བཞག ད་ལྟ་ཁོས་སྒོ་ཉིན་གཅིག་སྲུང་ན་ངལ་གསོ་ཉིན་གསུམ་ཡོད་ལ་ལས་ཁུངས་ཀྱི་ཚོགས་འདུ་འདྲ་མི་འདྲ་ཡོད་ཚད་འགྲོ་མི་དགོས་ཀྱང་མི་གཞན་གྱིས་ཁོའི་ལས་ཀ་དེ་ཐབས་སྡུག་ཅིག་ཏུ་རྩི་བཞིན་ཡོད་པ་འདྲ། ཉིན་ཞིག་གི་མཚན་མོར་ཁོས་ལྕགས་སྒོ་བརྒྱབ་རྗེས་གཉིད་དུ་ཤོར་སོང་བ་ན་པུའུ་ཀྲང་གཞོན་པས་མོ་ཊ་བཏང་ནས་ཡར་འཛུལ་མ་ཐུབ་པས་ཨེན་ཙམ་མིན་ན་ཁོ་ལ་འགྲམ་ལྕག་ཅིག་བརྒྱབ་བྱུང་། ཁོས“དགོངས་པ་མ་ཚོམས། དགོངས་པ་མ་ཚོམས”ཡང་ཡང་ཟེར་བ་ལས་གཞན་ཐབས་གང་ཡང་མེད་སོང་། དུས་དེ་ནས་བཟུང་ཁོས་རང་ཉིད་ནི་མིའི་ཕྱེད་ཀ་དང་ལས་ཁུངས་ནང་གི་ལས་མི་མཐའ་མ་ཞིག་ཡིན་པའི་ངེས་ཤེས་སྐྱེས། པུའུ་ཀྲང་གིས་མུ་ཊ་བཏང་ནས་འགྲོ་དུས་རྗེས་འདེད་མཁན་མུ་ཊ་བཞི་ལྔ་ཙམ་ཡོད་ལ་པུའུ་ཀྲང་རྐང་ཐང་ལ་འགྲོ་དུས་རྗེས་འདེད་མཁན་མི་ཉི་ཤུ་ཙམ་འདུག ཞིང་འབྲོག་གི་མི་དམངས་རྣམས་ལས་ཁུངས་ནང་འཛིན་འདྲ་བཟོ་བར་ཡོང་དུས་ཉེས་པ་ཅི་ཞིག་བསགས་པ་བཞིན་ལྕེ་ནར་ནར་གྱིས་ཡར་ཡོང་བཞིན་འདུག་ལ་ལས་མི་རྣམས་ཀྱི་མི་དམངས་རྣམས་ལ་འདྲི་མི་དགོས་པའི་སྐད་ཆ་མང་པོ་ཞིག་འདྲི་བཞིན་མཆིས། ཁོས་བལྟས་ན་དེ་ནི་དཔོན་ཉམས་སྟོན་ཚུལ་ཞིག་ནས་གཞན་སྙིང་བོ་གང་ཡང་མི་འདུག འདི་ལྟར་ཁོ་རང་སྤྱི་ཚོགས་སུ་བཞུགས་པ་ནས་བཟུང་སྤྱི་ཚོགས་ནང་དུ་མིའི་རིམ་པ་མང་པོ་ཞིག་ཡོད་པ་ཤེས་ཅིང་མི་ཀུན་ནུས་ཤུགས་ཡོད་རྒུས་ཡར་ཡར་འཛེག་བཞིན་པར་རྟོགས། སྒོ་སྲུང་བར་འབད་བརྩོན་དང་ཤེས་བྱ་ཆེ་མི་མཁོ་བས་ཡར་ཡར་འཛེགས་པའི་ཐབས་ཤེས་ཞིག་མི་འདུག ཁོས་སྔ་མོ་ཞིག་ནས་ལས་ཁུངས་འདི་དང་བྲལ་ཏེ་ལས་ཀ་དེ་མིན་ཞིག་འཚོལ་བའི་བསམ་ཚུལ་སྔར་མོ་ནས་ཡོད་མོད། ད་ལྟའི་རང་དབང་དེ་འདོར་འདོད་མེད་པས་བསམ་ཚུལ་དེ་ལག་ལེན་བྱས་མ་ཐུབ། སྲང་ལམ་སྟོང་པའི་ངོས་ཀྱི་ཁ་བའི་ཁྲོད་དུ་རྟ་མཆོག་རྟགས་ཅན་གྱི་མོ་ཊ་ཞིག་ཚུར་བརྒྱུགས་བྱུང་བ་དང་ཁོའི་འགྲམ་དུ་ཤགས་སེར་བཞག སྒེའུ་ཁུང་ཧྲག་སྒྲ་དང་བཅས་མར་འཐེན་འོང་བས་ཤེལ་སྒོ་ནག་པོའི་རྒྱབ་ཏུ་མི་ཞིག་གི་འཛུམ་མདངས་མངོན་འོང་། ཁོས་མོ་ཊ་མཐོང་མ་ཐག་ཁ་ལོ་བ་ངོ་ཤེས་བྱུང་། རྫོང་ཁ་འདི་རུ་རྟ་མཆོག་རྟགས་ཅན་གྱི་མུ་ཊ་གཏོང་མཁན་ཁོའི་ཆུང་དུས་ཀྱི་རྩེད་གྲོགས་ཚེ་བཟང་མ་གཏོགས་མེད། ཁོ་རང་སློབ་གྲྭ་ཆེན་མོར་མ་འགྲོ་གོང་ནས་ཚེ་བཟང་སྦྱིན་བདག་ཆེན་མོར་གྱུར་ཟིན། ཚེ་བཟང་གིས་ཚོང་དག་ཁེ་ཆེན་པོ་ཡོད་པ་དག་དག་ཡིན། དཔེ་ན་གླ་རྩི་དང་ཤྭ་རྭ། དུམ་གྱི་རྐང་ལག འབར་མ་ནས་ལྐོག་འདྲེན་བྱེད་ཡོང་བའི་དུག་རྫས་སྣ་ཚོགས་བཅས་རེད། ཟླ་བ་བཀྲ་ཤིས་སློབ་གྲྭ་ཆེན་མོར་འགྲོ་སྐབས་ཚེ་བཟང་གིས“སློབ་གྲྭ་ཆེན་མོ་ནི་བྲན་གཤིས་སློབ་ས་ཡིན་པས་དེ་ནས་ཐོན་པ་ཐམས་ཅད་ཀླད་སྐྱོན་ཅུང་ཙམ་ཡོད། སྤྱི་ཚོགས་འདིའི་ནང་དུ་ཀླད་པ་ཆེ་བས་མི་ཕན་སྙིང་ཆེ་དགོས”ཅེས་བཤད་མྱོང་། དེ་རིང་ཡང་སྦྱིན་བདག་ཆེན་པོ་དེས་སྒེའུ་ཁུང་གི་ཤེལ་སྒོ་ནག་པོ་འོག་ནས་ཁོ་ལ་འཛུམ་བྱུང་། འཛུམ་མདངས་དེ་འཛུམ་མདངས་རྣལ་མ་ཞིག་མ་རེད། དེའི་ནང་དུ་ཟུར་ཟའི་ཤད་ཆེན་པོར་འདྲེས་ཅིང་འཛུམ་པའི་སྐབས་སུ་ཐོན་ཡོང་བའི་གསེར་སོའི་འོད་དེ་ཁ་བས་ཆེ་ཆེར་བཏང་འདུག “སྒོ་སྲུང་ལགས། ཡང་འཆར་ཡན་གྱི་ཀློང་དུ་བཞུགས་ཡོད་དམ”ཟེར། རང་ལ་སྒོ་སྲུང་ཟེར་བར་ཤིན་ཏུ་མི་སྙན་པ་ཞིག་འདོད་ཀྱང་ཚེ་བཟང་གི་རླངས་འཁོར་གྱི་སྒོ་ཕྱེ་བྱུང་བས་ཁོ་རང་ཡང་ནང་དུ་མི་འཛུལ་ཀ་མེད་བྱུང་། རླངས་འཁོར་གེ་ས་གསོལ་ཁང་གི་སྒོ་འགྲམ་བཞག་པ་སྟེ་ཚེ་བཟང་རླངས་འཁོར་ལས་བབས་མ་ཐག་རྐང་པ་ཁ་བ་ནང་ནུབ་སོང་། ཚེ་བཟང་གིས་ཅི་མི་སྙམ་པའི་ངང་ནས་རས་དམར་པོ་འཐེན་ཡོད་པའི་རྐོ་སྐེས་ཀྱི་ཐོག་ཏུ་རྐང་པ་རྡེབ་ཙམ་རྡེབ་ཙམ་བྱེད་རྗེས་གསོལ་ཁང་གི་སྒོ་ནང་འཛུལ། ཁོ་ཡང་དེ་བཞིན་བྱས་ནས་འཛུལ། ཚེ་བཟང་གིས་སྒོ་འགྲམ་གྱི་ཞབས་ཞུ་མ་གཟུགས་རིང་དེ་མཐོང་མ་ཐག“བུ་མོ་ཀྲང་ལ་ཞོགས་པ་བདེ་ལེགས”ཟེར་བཞིན་མགོ་བོ་སྒུར་ཙམ་བྱས་སོང་། དེ་ནི་ཁོས་བརྙན་འཕྲིན་ནང་གི་མི་ཆེན་ཚོ་ལ་ལད་མོ་བྱས་པ་ཡིན་ས་རེད། ཡིན་ཡང་བུ་མོ་ཀྲང་ལགས་དགོད་ཙམ་བྱེད་པ་ལས་ཁོ་ལ་ལན་ཅི་ཡང་མ་བརྒྱབ། ཡར་ཙམ་སོང་ན་མི་རྒྱགས་པ་ཞིག་འདུག་སྟེགས་ཆུང་ཆུང་ཞིག་གི་ཐོག་ཏུ་ཙོག་འདུག ཁོའི་སྤྱི་མགོ་ཕྱེད་ཙམ་འདུག་སྟེགས་ཀྱི་ཕྱིར་ཐོན་ཡོད་པས་བལྟས་པ་ཙམ་གྱིས་ཤིན་ཏུ་མི་བདེ་བའི་སྣང་བ་ཞིག་སྐྱེས་མོད་ཀྱང་ཁོ་རང་ནི་སྣང་དོགས་སྤུ་ཙམ་ཡང་མེད་པར་བསམ་གཞིག་གི་ཀློང་དུ་འཁོར་འདུག ཚེ་བསམ་གྱིས་དེ་ལ“སྦྱིན་བདག་ཝང་ལགས། དེ་རིང་ཚོང་ག་འདྲ་འདུག”ཅེས་དྲིས། “ཨེན་ཙམ། ཨེན་ཙམ”མི་རྒྱགས་པ་དེས་འདི་ལྟར་བཤད་པ་ལས་མགོ་བོ་ཡར་བཀྱགས་ཙམ་ཡང་བྱས་མ་སོང་། ཚེ་བཟང་གིས་མི་ངོ་ཤེས་མི་ཤེས་ཡོད་ཚད་ལ་ཁ་གྲག་པ་དེ་ཁོ་རང་ལྟ་རྒྱ་ཆེན་པོ་ཡོད་པ་དང་འབྲེལ་བ་རྒྱ་ཆེ་ཡོད་པའི་ཚུལ་ངོམ་བཞིན་པ་སྣང་། ཁོ་གཉིས་ཀྱིས་ཞོགས་ཟས་ཟ་སྐབས། ཚེ་བཟང་གིས་གསར་དུ་ཉོས་བའི་ཀུ་ཤུ་བདུན་པ་དེ་རྩེད་ནས་བསྡད་པ་ལས་ཟས་ལ་ལག་པ་རེག་ཙམ་ཡང་བྱས་མ་སོང་། ཁོའི་ཀུ་ཤུ་བདུན་པ་རྟགས་ཅན་གྱི་ཁ་པར་དེར་ཕྱི་ཤུན་དང་ལོང་ཐག་སོགས་དཀར་ཁྲ་དམར་ཁྲ་མང་པོས་བརྒྱན་ནས་བྱིས་པ་ཞིག་གི་རྩེད་ཆས་ལྟར་བཟོས་འདུག ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་སྣང་ཚུལ་དེ་དག་མཐོང་བས་ཟས་ཀྱི་དང་ག་ཡང་ཆད། ཚེ་བཟང་གིས་སྣང་ཚུལ་དེ་མཐོང་བས“ཅི་རེད། ཕོགས་ལེན་ཡོང་དུས་ཟས་མི་འགྲོ་བར་གྱུར་བ་མ་རེད་དམ”ཟེར། སྒོ་སྲུང་ཞིག་ལ་མཚོན་ན་ཟླ་གཅིག་ལ་བདུན་སྟོང་ཡོད་པ་དེ་མང་པོ་རེད་ཟེར་ཆོག་མོད། ཡིན་ཡང་ཚོང་དཔོན་ལ་མཚོན་ན་དེ་ནི་བྱིས་པ་ཞིག་ལ་སྔན་པ་སྤྲོད་བྱེད་ཙམ་ལས་མ་འདས། ཁོ་གཉིས་ཁ་བྲལ་ཁར་ཚོང་དཔོན་ནས“དེ་རིང་ཁ་བ་བབས་པས་སྣང་བ་སྐྱིད། ངེད་གྲོགས་པོ་གཉིས་ཆང་འཐུང་བར་འགྲོ་ན་ག་འདྲ་རེད”ཟེར། ཁོས“དེ་རིང་ང་ལ་ལས་ཀ་ཡོད”ཅེས་བཤད་མོད་ཚེ་བཟང་གིས“ཁྱོས་ལས་ཀ་ལས་ཀ་ཟེར་བ་དེ་སྒོ་སྲུང་རྒྱུ་ལ་ཟེར་བ་ཡིན་ནམ”ཟེར། ཁོས་སྐད་ཆ་དེ་ལ་ཁ་གཡར་མ་བྱས་པར་ཁ་བའི་ཁྲོད་དུ་རྟོག་སྒྲ་ཧྲག་ཧྲག་གྲག་བཞིན་ཚེ་བཟང་དང་རྒྱང་ཇེ་རིང་དུ་སོང་། དེ་སྐབས་ཚོང་དཔོན་གྱིས་ཁོའི་རྒྱབ་གཟུགས་ལ་ལྟ་བཞིན་དགོད་ཀྱིན་པ་ཚོར། ཁོ་རང་ལས་ཁུངས་ཀྱི་ཕྱོགས་སུ་མ་འགྲོ་བར་ལྡོག་ནས་མི་སྤྱི་ཅུས་ཡོད་སར་ཕྱིན། མི་སྤྱི་ཅུས་ཀྱི་ཅུས་ཀྲང་གཞུང་ལས་ཁང་དུ་ཞང་བོ་ནི་རྩིས་འཁོར་ཆེན་པོ་ཞིག་གི་ངོས་སུ་ཧད་འདུག ཁོ་ལ་ལྟ་ཙམ་ཡང་མ་བྱེད་པར“ཐིག་རྒྱབ་དགོས་པ་ཨེ་རེད། འཛིན་ཁུར་ཤོག”ཟེར། ཁོས“ཞང་བོ་ལགས། ང་འཛིན་བཟོ་རུ་ཡོང་བ་མིན། ང་ལ་དོན་དག་ཅིག་ཡོད”ཅེས་བཤད། ཞང་བོས་མིག་རྩིས་འཁོར་དང་བྲལ་བ་དང་ཁོ་ལ་ལྟ་ཡག་ཅིག་བྱས་ཏེ“འོ། ཟླ་བ་བཀྲ་ཤིས་རེད་འདུག་གཱ སྡོད་དང་། སྡོད་དང”ཟེར་རྗེས་སླར་ཡང་གློག་ཀླད་ཀྱི་ངོས་སུ་མགོ་འཁོར་སོང་། ཁོས་རང་གི་འདུན་པ་དག་ཡུན་རིང་ཞིག་ལ་བཤད་པའི་རིམ་ཞང་བོ་ནི་སྔ་བཞིན་རྩིས་འཁོར་ལ་མགོ་འཁོར་ནས་འདུག ཁོས་བསྒུགས་བསྒུགས་པའི་མཐར་ཞང་བོས་ཁོའི་ཕྱོགས་སུ་ལྟ་ཙམ་བྱས་ཏེ“ཁྱོད་རང་སྒོ་སྲུང་བྱེད་འདོད་མི་འདུག་ཟེར་བ་ཡིན་ནམ”ཟེར། ཁོས“ཡིན་ལགས། ཡིན་ལགས”ཟེར་དགོས་བྱུང་། ཞང་པོས་ཤོག་ལེབ་ཅིག་གི་ཐོག་ཏུ་ཡི་གེ་འགའ་མགྱོགས་ཤིག་ཤིག་གིས་བྲིས་རྗེས་ཁོ་ལ་སྤྲད་བྱུང་། ཁོས་ལན་གཅིག་བལྟས་པ་ན་འཁྱུག་ཡིག་རེད་འདུག་པས་ཡང་སྙིང་ཅི་ཞིག་བྲིས་ཡོད་པ་ཤེས་མ་ཐུབ། ཡི་གེ་དེ་ཚོ་འབྲི་མཁན་ཞང་བོ་རང་ཉིད་ཀྱིས་བཀླགས་ན་མ་གཏོགས་སུས་ཀྱང་ཀློག་ཐུབ་ས་མ་རེད། ཁོས་ཡི་གེའིི་འགོ་བརྗོད་ལ་ཡུན་རིང་ཞིག་ཅེར་རྗེས་ཚོད་ཀྱིས་འདི་ནི་ང་སྤྲོད་ཡི་གེ་ཞིག་ཡིན་ས་རེད་བསམས། ཁོས“ཞང་བོ་ལགས། ད་ང་འགྲོ་གི་ཡིན”ཟེར་ནས་སྒོ་ཁར་སོང་བ་ནས་ཞང་བོས“དཔེ་བསགས་སོང”ཞེས་སྒྲོག་ཙིའི་ཐོག་ཏུ་ཁུ་ཚུར་ཞུས་བྱུང་། ཁོས་དོན་དག་ཅི་ཞིག་བྱུང་བ་ཡིན་ནམ་སྙམ་གྱིན་ཞང་བོའི་འགྲམ་དུ་སོང་ནས“དོན་དག་ཅི་ཞིག་བྱུང་སོང”ཞེས་དྲིས། ཞང་བོས“ཕམ་སོང་། ཕམ་སོང”ཟེར། ཁོས་གློག་ཀླད་ཀྱི་ངོས་སུ་ལྟ་ཡག་ཅིག་བྱེད་དུས་ཞང་པོས་ཤོག་བག་རྩེད་བཞིན་ཡོད་པ་ཤེས། ངོ་སྤྲོད་ཡི་གེ་འབྲི་བའི་བར་མཚམས་སུ་ཁོ་རང་ཕང་སོང་བ་ཡིན་འདུག བློ་བདེ་བག་ཕེབས་ངང་ལས་ཁུངས་ཀྱི་ཕྱོགས་སུ་ལོག ཕྱི་རོལ་དུ་ཉི་མ་ཐོན་འདུག་ལ་རླངས་པ་སྲབ་མོ་ཞིག་སྲང་ལམ་གྱི་ཐོག་ཏུ་གཡོ་བ་མེད་པར་འཁྲིགས་འདུག དེའི་ཁྲོད་དུ་བསྐྱོད་ཚེ་ལུས་ལ་དྲོན་མོ་ཡོད་ཀྱང་རླུང་བུ་ཞིག་ལྡང་ན་ཅི་མ་རུང་སྙམ་འོང་། ལས་ཁུངས་ཀྱི་སྒོ་ཁར་སྔོན་མ་ཁོའི་རྩོམ་སྒྲིག་པུའུ་ཡི་ལས་རོགས་ཚེ་རིང་ངོ་གདོང་ཡང་དམར་པོར་གྱུར་ནས་འགྲེང་འདུག དང་པོ་ཁོས་དེ་ནི་ཁ་བའི་གྲང་ངར་གྱིས་དམར་བ་འདོད་མོད། ཉེ་འགྲམ་དུ་སླེབས་དུས“དུས་ཚོད་དུ་རེད་འདུག ཁྱོད་རང་སྐྱག་པ་གང་དུ་ཟ་འགྲོ་བ་ཡིན”ཞེས་སྡིགས་བྱུང་བས་ཚེ་རིང་གིས་རང་ལ་དེ་འདྲ་བཤད་ཡོང་བར་བསམ་པའི་གཏིང་ལ་མེད། ཚེ་རིང་ཁོ་གཉིས་རྩོམ་སྒྲིག་པུའུ་རུ་མཉམ་དུ་ལས་ཀ་བགོས་ཡོང་བ་ཡིན། སྐབས་དེར་ཚེ་རིང་གིས་ཁོ་ལ“འ་བུ་ཟླ་བ་བཀྲ་ཤིས”ཅེས་སྙིང་ཉེ་ཉེ་ངང་ནས་འབོད་འོང་། རྗེས་སུ་ཁོ་རང་སྒོ་སྲུང་གྱུར་ཅིང་ཚེ་རིང་ནི་རྩོམ་སྒྲིག་པུའུ་རུ་དབང་ཆ་ཅུང་ཡོད་པའི་མི་ཞིག་ཏུ་གྱུར། སྤྱི་ཚོགས་འདིའི་ནང་དུ་དབང་ཆེན་པོ་ཡོད་མཁན་ཚོས་དབང་ཆུང་ཆུང་ལས་མེད་པ་རྣམས་མནན་ཞིང་དབང་ཆ་ཆུང་ཙམ་ཡོད་མཁན་ཚོས་དབང་ཆ་མེད་པ་རྣམས་མནན་པའི་སྐེས་རིམ་བཙན་པོ་ཞིག་ཏུ་འགྲུབ་མོད། ཚེ་རིང་ཡང་འདི་ལྟར་སྐེས་རིམ་དེའི་ནང་ལ་ལས་སླ་པོའི་ངང་ནས་ཚུད་སོང་བ་ཁོའི་བློ་ཡུལ་དུའང་མེད། དེ་རིང་ཚེ་རིང་གིས་དབང་ཆ་ཆུང་ཆུང་དེ་སྤྱད་ནས་ཁོ་ལ་རྫིག་རྫིག་བྱེད་ཡོང་དུས་སྐྱོ་བ་ཞིག་དང་མཉམ་དུ་རང་ཉིད་ནི་མིའི་ཕྱེད་ཀར་གྱུར་ཟིན་པའི་སྣང་བ་སྐྱེས་པ་དང“པུའུ་ཀྲང་གིས་ང་འཚོལ་བཞིན་འདུག”ཅེས་རྫུན་བཤད་རྗེས་སྒོ་སྲུང་མ་བསྡད་པར་པུའུ་ཀྲང་གི་གཞུང་ལས་ཁང་ཡོད་སར་སོང་། པུའུ་ཀྲང་གིས“འོ། ཟླ་བ་བཀྲ་ཤིས་སླེབས་ཡོང་ངམ”ཟེར། ཁོས་རང་ཉིད་ཡོང་དོན་རྣམས་ཞིབ་པར་བཤད་པ་ན་པུའུ་ཀྲང་གིས“ཅི་རེད། སྒོ་སྲུང་མི་བསམ་པ་རེད་དམ། དེ་འདྲ་ཡིན་ན་ངས་ཁྱོད་ལ་ལས་ཀ་གཞན་ཞིག་བགོ་ཆོག”ཅེས་བཤད་བྱུང་། “དེ་འདྲ་གང་ནས་ཡིན། ངའི་ཆེད་ལས་ནི་ཉེན་རྟོག་མྱུལ་ཞིབ་ཡིན་པ་དང་ང་ལ་དེའི་བསྐོར་གྱི་ཤེས་བྱ་ཕུན་སུམ་ཚོགས་པོ་ཡོད། ཤེས་བྱ་དེ་དག་བེད་སྤྱོད་བྱེད་མ་ཐུབ་ན་འཕྲོ་བརླག་རེད། ག་ལེར་བརྗེད་འགྲོ་གི”ཅེས་བཤད་པར་པུའུ་ཀྲང་གིས་ཀྱང་ཕམ་པའི་རྣམ་འགྱུར་ཞིག་སྟོན་བཞིན་འཛིན་ཞིག་གི་ཐོག་ཏུ་ཡི་གེ་འགའ་བྲིས་རྗེས་ཐིག་བརྒྱབ་སོང་། ཁོས་པུའུ་ཀྲང་གི་གདོང་ལ་ཞིབ་ལྟ་ཞིག་བྱེད་དུས་ཕམ་པའི་རྣམ་འགྱུར་དེ་དེ་འདྲའི་ཆེན་པོ་ཞིག་མིན་པ་མ་ཟད། ལས་ཀའི་གོམས་གཤིས་ཀྱིས་རྣམ་འགྱུར་དེ་འདྲ་ཞིག་སྟོན་ཁུལ་བྱེད་བཞིན་འདུག་ཅེས་ཟེར་ནའང་འགྲིག་པ་འདུག ཁོས་འཛིན་ཡིག་མང་པོ་ཁ་སྐོང་རྗེས་མི་སྤྱི་ཅུས་ཀྱི་ཞང་བོའི་སར་བསྐྱལ་བ་དང་ཕྱི་དྲོར་སྤྱི་བདེ་ཅུས་སུ་འགྲོ་རྒྱུ་བྱས། ལས་ཀ་སྤོས་པ་ནི་འདི་འདྲའི་ལས་སླ་པོ་ཡིན་པ་ནི་གློག་ཀླད་ངོས་སུ་ཤོག་བག་རྩེད་བཞིན་པའི་ཞང་བོའི་དྲིན་ཏེ་སྔོན་མར་གྲོགས་པོ་ཞིག་གིས་ལས་གནས་སྤོས་བར་ལོ་གསུམ་ཙམ་འགོར་སོང་ཟེར་ནས་མགོ་ན་བཞིན་པ་མཐོང་མྱོང་། ཕྱི་དྲོར་ངོ་སྤྲོད་ཡི་གེ་ཁུར་ནས་སྤྱི་བདེ་ཅུས་ཀྲང་སར་སོང་། སྤྱི་བདེ་ཅུས་ཀྲང་གིས་ངོ་སྤྲོད་ཡི་གེ་ལ་ཐེངས་ཁ་ཤས་བལྟས་རྗེས“འོ་ཁྱོད་རང་བློ་བཟང་ཅུས་ཀྲང་གི་ནང་མི་ཡིན་ནམ”ཞེས་དྲིས། དེ་ནི་ཡི་གེ་ལ་བལྟས་ནས་འབྲི་མཁན་ངོ་ཤེས་སོང་བ་འདྲ། “བློ་བཟང་ཅུས་ཀྲང་ངའི་ཞང་བོ་ཡིན”ཞེས་པ་ན“ད་ལྟའི་རྫོང་སྤྱི་བདེ་ཅུས་ཀྱི་ཉེན་རྟོག་པ་རྣམས་རིག་གནས་དམའ་བ་རྐྱང་རྐྱང་རེད། ཁྱོད་འདྲ་རིག་གནས་ཡོད་མཁན་མཁོ། དང་པོ་ཁྱོད་རང་ཏུའུ་གསུམ་པའི་ཏུའུ་ཀྲང་བྱོས། རྗེས་སུ་དལ་བ་དལ་བར་ལྟ་ཆོག་རེད”ཟེར་བ་དང་ཉེན་རྟོག་པ་གསུམ་འབོད་ནས་སྡོད་ཁང་རྩིས་སྤྲད་བྱེད་དུ་བཅུག ཏུའུ་གསུམ་པ་ལ་རུ་ཁག་གཉིས་ཡོད། རུ་ཁག་གཅིག་གིས་མུ་ཊ་ཁྲ་ཁྲ་གཏོང་བཞིན་ཉིན་མཚན་མེད་པར་རྫོང་ཁ་བདེ་སྲུང་བྱེད་དགོས། ཁོ་རང་ཏུའུ་ཀྲང་བསྐོས་པའི་ཉིན་དང་པོའི་མཚན་མོར་བདེ་སྲུང་རུ་ཁག་གིས་བྱ་བ་ཇི་ལྟར་བྱེད་བཞིན་པར་བལྟ་བ་འགྲོ་དུས་རུ་ཁག་དེའི་ཉེན་རྟོག་པ་རྣམས་ཀྱིས་རླངས་འཁོར་ཁྲ་ཁྲའི་གློག་བཀར་ནས་སྲང་ལམ་གྱི་གཡས་གཡོན་དུ་བཞག་རྗེས་རང་ཉིད་ཁང་པའི་ནང་འཛུལ་ཏེ་ཆང་འཐུང་བཞིན་ཡོད་པ་རིག་པས་ཁྲོང་ཁྲོ་ཆེན་པོ་ལངས་མོད་ཁོ་ཚོས་ཁོ་མཐོང་མ་ཐག“ཏུའུ་ཀྲང་ཟླ་བ་བཀྲ་ཤིས་ལགས”ཅེས་སྙིང་ཉེ་ཉེ་བྱེད་བཞིན་འདུག་པས་མགྲིན་པ་ནང་ཐོན་ཡོད་པའི་དམའ་འབེབས་ཀྱི་ཚིག་རྣམས་ཁོག་པའི་ནང་མིད་དགོས་བྱུང་། “རྗེས་མར་གཟབ་དགོས”ཟེར་ནས་ཕྱིར་ལོག ཡུལ་དུ་ཁ་པར་བཏང་དུས་ཁ་པར་ལེན་མཁན་ནི་ཨ་མ་རེད་འདུག ཁོས་རང་ཉིད་ལས་གནས་སྤོ་ཚུལ་བཤད་པ་ན་ཨ་མས“ཉེན་རྟོག་པ་ཟེར་བ་དེ་མི་གསོད་མཁན་མིན་ནམ། དེ་བྱས་ནས་ཅི་བྱ”ཞེས་བཤད་བྱུང་། “རྒད་མོ་ཁ་བདེ། ཁྱོས་ཅི་ཤེས། ཕྲུག་ཅི་བྱེད་བསམས་ན་དེ་བྱེད་དུ་བཅུག”ཨ་ཕས་སྐད་ཆེན་པོར་ཨ་མ་ལ་སྡིགས་པ་ཐོས། ཁ་པར་བརྒྱབ་རྗེས་དགོད་རྒྱུ་ཞིག་བྱུང་ལ་ཉིན་ཁ་ཤས་ཀྱི་རྗེས་སུ་སླར་ཡང་ཁྱིམ་ནང་ཁ་པར་ཞིག་གཏོང་དུས་ཨ་མས“བྱིས་པ་ཚོས་པྲ་ཆལ་འདྲ་བྱེད་དུས་ངས‘སང་ཉིན་ཨ་ཁུས་མེ་མདའ་ཁུར་ནས་ལོག་ཡོང་གི་རེད’ཟེར་དུས་སྐྲག་གི་འདུག”ཅེས་བཤད་བྱུང་། སྐབས་དེ་དུས་ཀྱི་ཨ་མའི་སྐད་གདངས་ནང་དུ་དགའ་བ་ཞིག་འདྲེས་ཡོད་པ་གོ “ཏུའུ་ཀྲང་ལགས། དེ་རིང་འདྲི་གཅོད་ཁང་དུ་མི་གསོད་མཁན་ཞིག་ལ་འདྲི་གཅོད་བྱེད་དགོས་པས་ཁྱོད་རང་མགྱོགས་པོར་ཕེབས་དགོས”ཅེས་པའི་ཁ་པར་གྱིས་ཁོ་རང་གཉིད་ལས་དཀྲོགས་སོང་བས་ཞོགས་ཟས་ཀྱང་ཟ་ཁོམ་མེད་པར་འདྲི་གཅོད་ཁང་དུ་རྒྱུགས་སོང་། འདྲི་གཅོད་ཁང་དུ་མི་གསོད་མཁན་ཡིན་ཟེར་བ་དེ་གཅར་རྡུང་བྱས་ནས་དྲན་པའང་མི་གསལ་བར་བཟོས་འདུག སྐད་ཆ་ཅི་འདྲ་ཞིག་དྲིས་ཀྱང“ངས་མི་གསོད་མེད”ཅེས་ཡང་ཡང་ཟློས་པ་ལས་གཞན་ཅི་ཡང་བཤད་མི་ཐུབ་པ་འདུག་པས་འདྲི་གཅོད་བྱ་རྒྱུ་ཟེར་བ་དེ་མི་ཞིག་གི་ཁ་ནས་ཁྲག་འདྲེས་པའི་ཁ་ཆུ་ཡང་ཡང་དུ་་བཞུར་བར་བལྟ་རྒྱུ་དེ་ཡིན་འདུག ཟླ་བ་བཀྲ་ཤིས་ཀྱིས“ཁྱོད་ཀྱི་མི་ལ་ཅི་ཟེར”ཟེར་དུས་མི་དེས“ངས་མི་གསོད་མེད”ཟེར། ཟླ་བ་བཀྲ་ཤིས་ཀྱིས“ཁྱོད་རང་ས་ཆ་གང་གི་མི་ཡིན”ཟེར་དུས་མི་དེས“ངས་མི་གསོད་མེད”ཟེར་བྱུང་བས་ཟླ་བ་བཀྲ་ཤིས་དང་མཉམ་དུ་ཡོད་པའི་ཉེན་རྟོག་པ་ཞིག་ཁྲོས་ནས“ཏུའུ་ཀྲང་གིས་ཁྱོད་ལ་སྐད་ཆ་འདྲི་བཞིན་པ་མ་གོའམ། སྐད་ཆ་ཡག་པོ་ཤོད”ཅེས་བརྡུང་གྲབས་བྱེད་དུས་ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་མི་དེ་བཀག་ནས“ད་དུང་བརྡུངས་ན་ཁོའི་སྲོག་ཀྱང་ཤོར་ས་རེད”ཅེས་བཤད་ཅིང་མི་དེ་ལ་ཁོ་མགྱོགས་པོར་སྨན་བཅོས་བྱེད་དགོས་པ་དང་དྲན་པ་གསལ་དུས་གཞི་ནས་འདྲི་གཅོད་བྱ་རྒྱུ་བྱས། ཞོགས་པའི་ལས་ཀ་གྲོལ་ནས་ཉལ་ཁང་དུ་སླེབས་མ་ཐག “མི་དེས་ཏུའུ་ཀྲང་ལ་སྐད་ཆ་བཤད་དགོས་ཟེར་གི”ཞེས་པའི་ཁ་པར་སླེབས་པས་ཟླ་བ་བཀྲ་ཤིས་སླར་ལས་ཁུངས་སུ་ལོག་དགོས་བྱུང་། ཁོ་རང་འདྲི་གཅོད་ཁང་དུ་འཛུལ་དུས་ཅི་ཡང་མ་དྲིས་ཀྱང་མི་དེས”ངས་མི་གསོད་མེད”ཅེས་པའིི་སྐད་ཆ་དེ་ཡང་ཡང་ཟློ་བཞིན་འདུག ཏུའུ་ཀྲང་ཟླ་བ་བཀྲ་ཤིས་ཀྱིས“ཁྱོད་ཀྱི་སྐད་ཆ་དལ་བར་ཤོད། འཚབ་དགོས་དོན་མེད། དང་པོར་ཁྱོད་རང་ཇི་ལྟར་འཛིན་བཟུང་བྱེད་ཚུལ་ཤོད”ཅེས་པར་མི་དེས་བསམ་བློ་ཡོན་རིང་ཞིག་གཏོང་རྗེས“ང་ནི་རྒྱུ་དང་དབང་ཅི་ཡང་མེད་པའི་འཁྱམས་པོ་ཞིག་ཡིན། ང་རང་རྫོང་ཁ་འདི་བརྒྱུད་དེ་ཁ་བ་དཀར་པོ་མཇལ་བ་འགྲོ་བར་ཡིན། ཡིན་ཡང་ཉེ་འགྲམ་གྱི་ཉེན་རྟོག་ཞིབ་བཤེར་ས་ཚིགས་ཤིག་བརྒྱུད་སྐབས་ཉེན་རྟོག་པ་ཁ་ཤས་ཀྱིས་ངའི་ལམ་བཀག་བྱུང་། ཁོ་ཚོར་ཐོབ་ཐང་དཔང་ཡིག་དགོས་ཟེར། ཡིན་ཡང་ངས’’ང་འདྲ་གནས་ངེས་མེད་དུ་འཁྱམས་མཁན་ཞིག་ལ་ཐོབ་ཐང་དཔང་ཡིག་ཅིག་ག་ན་ཡོད’ཅེས་བཤད་པར་ཁོ་ཚོའི་ནང་གི་དཔོན་འདྲ་བ་ཞིག་གིས་ང་ལ་སྡིགས་བྱུང་། ངས་ཀྱང་ཁོ་ལ་ངར་ངར་བྱས་པ་ཡིན། མ་བཟང་ས་ཞིག་ནས་ཁོ་རང་ཅུས་ཀྲང་གཞོན་པ་ཡིན་འདུག་པས་ང་རང་དང་པོར་ཐོབ་ཐང་དཔང་ཡིག་མེད་པར་ཁ་གཡར་ནས་འཛིན་བཟུང་བྱས་སོང་། ཁ་སང་ཕྱི་དྲོ་ཅི་ཡིན་འདི་ཡིན་མེད་པར་ཁ་བ་འབབ་ཉིན་གྱི་མི་གཉིས་གསོད་མཁན་དེ་ང་རང་རེད་ཅེས་མ་ཉན་ཁ་གཡོག་བྱས་བྱུང”ཞེས་དལ་བར་བཤད་པར་ཏུའུ་ཀྲང་ནས“ཉེན་རྟོག་པས་གཙོ་བོར་དཔང་རྟགས་ལ་བསྟེན་ནས་མི་བཟུང་དགོས། དེ་བཞིན་ཁྱོད་ལ་ཡང་དཔང་རྟགས་དགོས། གཙོ་བོ་ཁ་བ་འབབ་པའི་མཚན་མོར་ཁྱོད་རང་གང་དུ་ཡོད་པ་བདེན་དཔང་བྱོས”ཅེས་བཤད་པར“ཁ་བ་འབབ་པའི་མཚན་མོར་ང་རང་རྫོང་ཁར་བསླེབ་མེད། དེ་ཉིན་ཉེ་འགྲམ་གྱི་དུད་ཚང་ཞིག་ཏུ་གནས་ཚང་གཡར་བ་ཡིན”ཞེས་ལབ། “དུད་ཚང་དེ་གང་དུ་ཡོད་པ་ད་ལྟ་དྲན་རྒྱུ་ཨེ་འདུག” “ལོས་ཡོད” ཁ་སང་མཚན་གང་གྲང་རླུང་ལྡང་བས་ཕྱི་རོལ་གྱི་ཁ་བ་ཡོད་ཚད་ཆབ་རོམ་དུ་གྱུར་འདུག་ལ་ཁང་པའི་སྒོ་ནས་མར་ཐོན་ཡོང་བའི་མི་རྣམས་ཀྱིས་རླངས་པ་དཀར་ཐོལ་ཐོལ་དུ་སྐྱུག་བཞིན་འདུག ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་སྤྱི་བདེ་རླངས་འཁོར་དུ་མི་གསོད་མཁན་རེད་ཟེར་བ་དེ་ཁུར་ནས་དུད་ཚང་དེ་འཚོལ་དུ་སོང་། ལམ་འགྲམ་གྱི་ཞིབ་བཤེར་ས་ཚིགས་འགྲམ་དུ་སླེབས་མ་ཐག་མི་དེས“ང་འཛིན་བཟུང་བྱེད་ས་འདི་ག་རེད”ཅེས་བཤད་བྱུང་བས་ཁོས་རླངས་འཁོར་ཉེ་འགྲམ་དུ་བཞག་རྗེས་མི་དེ་ཞིབ་བཤེར་ས་ཚིགས་ཀྱི་ཁང་པ་ཆུང་ཆུང་དེ་ནང་འཁྲིད། དེ་ཡོད་ཀྱི་ཉེན་རྟོག་པ་རྣམས་ལ་སྐད་ཆ་དྲིས་པས་ཁོ་ཚོས་མི་དེ་ཚེས་བཅུ་བཞིའི་ཉིན་ཕྱེད་ཙམ་ལ་འཛིན་བཟུང་བྱས་པ་ཡིན་ཟེར། བཟུང་དོན་ནི་མི་དེ་ལ་ཐོབ་ཐང་ལག་འཁྱེར་མེད་པས་ཟེར་ཡང་དོན་དུ་མི་དེས་སྐབས་དེར་དེ་གར་ཡོད་པའི་ཅུས་ཀྲང་གཞོན་པར་ཁ་ལན་བརྒྱབ་པས་ཡིན་འདུག དེ་ལྟར་ཡིན་ན་ཁ་བ་འབབ་དུས་ནི་ཚེས་བཅུ་གསུམ་པའི་མཚན་མོ་ཡིན་དང་མི་དེ་རྫོང་ཁར་ཡོང་དུས་ནི་ཚེས་བཅུ་བཞི་པའི་ཉིན་ཕྱེད་ལ་ཡིན་པས་ཁོས་མི་གསོད་པ་དེ་བདེན་པ་འགྲུབ་མི་ཐུབ། ཏུའུ་ཀྲང་ནས་ཉེན་རྟོག་པ་དེ་དག་གི་སྐད་ཆ་དག་སྟབས་བདེའི་ཟིན་བྲིས་ཕབ་རྗེས་སླར་མི་དེ་ཁུར་ཏེ་དུད་ཚང་དེ་འཚོལ་དུ་སོང་། ཞིབ་བཤེར་ས་ཚིགས་ནས་ལེ་བར་གསུམ་ཙམ་སོང་ན་དུད་ཚང་དེ་སར་སླེབས་སོང་། དུད་ཚང་དེའི་ཕ་རྒན་གྱིས་མི་དེའི་ཁྲག་གི་ཁེངས་པའི་ངོ་གདོང་མཐོང་མ་ཐག“ཨ་ཁ། ཁྱོད་ལ་ཅི་བྱུང”ཞེས་དྲིས་སོང་། མི་དེས་ལན་ཇི་བཞིན་བཏབ་པས“སྙིགས་དུས་ཟེར་བ་འདི་མི་ངན་ཚོ་བྱ་བ་ལམ་འགྲོ ཆོས་བྱེད་མཁན་ཚོ་གནས་མཇལ་དུ་སོང་བ་ལ་ཡང་བར་ཆད་འདྲ་མིན”ཞེས་ལག་པ་སྙིང་ག་བཞག་ནས་གསོལ་བ་འདེབས་པ་ལྟར་བྱས། ཏུའུ་ཀྲང་ནས་སྐད་ཆ་རེ་རེ་བཞིན་དྲིས་པར་རྒད་པོས་ཀྱང་མི་དེ་ཚེས་བཅུ་གསུམ་གྱི་མཚན་མོར་གནས་ཚང་བསྡད་རབས་དང་དེ་ཉིན་ཁ་བ་ཆེན་པོ་འབབ་རབས། ཕྱི་ཉིན་ཉི་མ་ཐོན་དུས་མི་དེ་རྫོང་ཁར་འགྲོ་རབས་སོགས་མི་དེས་འདྲི་གཅོད་བྱེད་སྐབས་བཤད་པའི་སྐད་ཆ་རྣམས་ཁ་གཏུགས་པས་ཏུའུ་ཀྲང་གིས་སྐད་ཆ་དེ་དག་སྒྲ་ཕབ་རྗེས་ཕྱིར་སྤྱི་བདེ་ཅུས་སུ་ལོག ལམ་བར་དུ་ཁོ་གཉིས་ཟ་ཁང་ཆུང་ཆུང་ཞིག་ཏུ་འཛུལ། ཟ་ཁང་ཆུང་ཡང་དེའི་ནང་དུ་མི་མང་པོ་འཚང་འདུག “ཁྱོས་ཅི་བཟའ”ཟླ་བ་བཀྲ་ཤིས་ཀྱིིས་འདི་ལྟར་དྲིས་པར་རོགས་པས“ཁྱོས་ཅི་བཟས་ན་དེ་ཡིན”ཞེས་བཤད་བྱུང་། མཐར་ཁོ་གཉིས་ཀྱིས་ཐུག་པ་འཐུང་། ཐུག་པ་དཀར་ཡོལ་དེ་ཤིན་ཏུ་ཆེ་བས་ཟླ་བ་བཀྲ་ཤིས་ཕྱེད་ཙམ་ལས་འཐུང་མ་ཐུབ་མོད། རོགས་པས་ཐུག་པ་དཀར་ཡོལ་གང་བོ་གཅིག་ཀྱང་མ་ལུས་པར་ཟས་སོང་། ཟླ་བ་བཀྲ་ཤིས་ཀྱིས“སྦྱིན་བདག་ལགས། ད་དུང་ཕོར་གང་དགོས”ཅེས་འབོད། དེའང་དེ་བཞིན་གཅིག་ཀྱང་མ་ལུས་པ་ཟས་སོང་བས་སླར་ཡང་ཕོར་གང་ཉོ་དགོས་བྱུང་། རོགས་པས་ཐུག་པ་དཀར་ཡོལ་གསུམ་པ་དེ་འཐུང་བཞིན་པའི་སྐབས་དེར་ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་ཆང་ཁང་གི་གང་སར་མིག་ཉུལ་བྱས། གཡོན་ཕྱོགས་ཀྱི་བོད་མི་རྒྱགས་པ་དེས་རིན་མི་དགོས་པའི་ཇ་དྭངས་མོ་དེ་བར་མེད་དུ་འཐུང་བཞིན་འདུག ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་ཁོའི་ཇ་འཐུང་ཚུལ་ལྟ་བཞིན་པ་ཁོས་ཚོར་སོང་བ་འདྲ། དོགས་མི་བདེ་བའི་ངང་ནས་ཡར་ལངས་བྱུང་ལ་འགྲོ་ཁར་སྒྲོག་ཙིའི་ཐོག་ཏུ་ཡོད་པའི་སྣ་ལུད་འབྱིད་བྱེད་ཀྱིས་ཤོག་བུ་དེ་ཡང་ཡང་འཐེན་ནས་ཁུར་སོང་། དེ་ཉིན་ཕྱི་དྲོ་ནས་མཚན་མོ་བར་མིང་ལ《ཚེ་བཅུ་གསུམ་པའི་ཁ་བའི་མི་གསོད་དོན་རྐྱེན་གྱི་སྙན་ཞུ》ཞེས་པའི་སྙན་ཞུ་རིང་པོ་ཞིག་བྲིས་ནས་ཕྱི་ཉིན་ཚེས་བཅུ་དྲུག་པའི་ཞོགས་པར་ཅུས་ཀྲང་གི་གཞུང་ལས་ཁང་དུ་སོང་སྟེ་ཅུས་ཀྲང་ལ་ཕུལ། ཡིན་ཡང་ཁོས་ཅུས་ཀྲང་གཞོན་པའི་ཁྲིམས་མེད་ཀྱི་བྱ་སྤྱོད་དག་བྲིས་མེད། ཆུ་ཚོད་གཉིས་ཀྱི་རྗེས་སུ་ཅུས་ཀྲང་གིས་ཁོ་རང་གཞུང་ལས་ཁང་དུ་ཡོང་དགོས་པའི་སྐད་བཏང་བྱུང་། ཅུས་ཀྲང་ནས་ཁོ་ལ་བསྟོད་བསྔགས་ཡང་ཡང་བྱས་པ་དང་མི་འཁྱམས་པོ་དེ་དངུལ་ཁ་ཤས་སྤྲད་ནས་བཙོན་གྲོལ་བྱེད་དགོས་པའི་བཀའ་ཕབ། མི་དེ་ཁ་བྲལ་ཁར་ཏུའུ་ཀྲང་གི་ལག་པ་འཇུས་ནས“དྲན་ཅན་པོ”ཞེས་ངུ་ལ་ཁད་བྱེད་བཞིན་འདུག ཏུའུ་ཀྲང་ཟླ་བ་བཀྲ་ཤིས་ནས“ཁྱོད་རང་རྫོང་ཁར་ཉིན་ཁ་ཤས་ལུས་པོ་གསོ་རྗེས་སོང་ན་དགའ”ཞེས་བཤད་ཀྱང་མི་དེས“དྲིན་ཅན་པོ། ང་བསྡད་མི་ཉན། ད་ཅི་མགྱོགས་གང་མགྱོགས་ཀྱིས་ཁ་བ་དཀར་པོའི་ཕྱོགས་སུ་ལོག་འགྲོ ཁ་བ་དཀར་པོ་མཇལ་ཐུབ་སོང་ན་ངས་ཁྱོད་ཀྱི་བསམ་པ་ལྷུན་གྱི་འགྲུབ་པའི་སྨོན་འདུན་ཞུས་ཆོག”ཅེས་བཤད་རྗེས་དལ་བས་སྲང་མི་ཚོགས་ཀྱི་ཁྲོད་དུ་ཕྱིན་སོང་། “ཁ་བ་དཀར་པོ་མཇལ་ཐུབ་སོང་ནས་ངས་ཁྱོད་ཀྱི་བསམ་པ་ལྷུན་གྱིས་འགྲུབ་པའི་སྨོན་འདུན་ཞུ་ཆོག”ཅེས་པའི་སྐད་ཆ་དེ་ཏུའུ་ཀྲང་གི་རྣ་བའི་ནང་ཡང་ཡང་འཁོར་བ་དང་རང་གིས་ལས་ཀ་བཟང་པོ་ཞིག་བྱས་ཐུབ་སོང་བར་དགའ་སྤྲོ་ཚད་མེད་སྐྱེས། ལས་ཁུངས་སུ་ཟང་ཟིང་བྱེད་བཞིན་འདུག ཞིབ་པར་དྲིས་པ་ན་དེ་ནི་འདས་པོའི་ནང་མི་ཚོས་ཕུང་པོ་ཡུལ་དུ་འདྲེན་གྲབས་བྱེད་ཀྱིན་ཡོད་པ་རེད། ཏུའུ་ཀྲང་ཟླ་བ་བཀྲ་ཤིས་ནས་དེ་མ་ཐག་མི་གསོད་ཀྱི་ལག་དམར་མི་རྙེད་པའི་བར་དུ་ཕུང་པོ་ཡུལ་དུ་འདྲེན་མི་ཆོག་པའི་བཀའ་ཕབཔ་དང་ནང་མི་ཚོས“ཕུང་པོས་སྐད་ཆ་ནི་བཤད་མི་ཤེས། སྤྱི་བདེ་ཅུས་སུ་འདི་ལྟར་བཞག་ན་ཕན་པ་ཅི་ཡོད། དེ་ལས་ཆོ་ག་འདྲ་སྤེལ་ཏེ་པུར་སྲེག་བྱེད་བཞག་ན་འདས་པོར་ཕན། ནང་མི་ཚོ་ཡང་སེམས་བདེ”ཞེས་བཤད་བྱུང་། ཁོས“ད་ལྟ་མི་གསོད་མཁན་སུ་ཡིན་པ་དེ་ཕུང་པོས་གསལ་བཤད་བྱེད་མི་ཐུབ་ན་སུས་ཀྱང་ལག་དམར་བཟུང་ཐུབ་ཀྱི་མ་རེད”ཅེས་བཤད་པ་མ་ཟད་ཕུང་པོ་ལ་བརྟག་དཔྱད་ཞིབ་པོར་བྱས་ན་མི་གསོད་ཀྱི་ལག་དམར་བཟུང་བར་ཕན་ཐོགས་ཆེན་པོ་ཡོད་ཚུལ་གསལ་བཤད་བྱས། ཁོས་སླར་ཡང“མི་གསོད་ཀྱི་ལག་དམར་བཟུང་བར་ཕུང་པོ་མཁོ་བ་མ་ཟད་ཁྱོད་ཚོ་ནང་མི་ཚོས་ཀྱང་རོགས་རམ་བྱེད་དགོས”ཅེས་བཤད། ཁོས་ཕུང་པོ་གཉིས་ལ་རགས་ཙམ་ཞིག་བལྟས་པ་བརྒྱུད། དེ་གཉིས་ཀྱི་མགྲིན་པ་གྲིས་བཅད་པས་ཤི་བ་ཡིན་འདུག་ལ་མགྲིན་པའི་ཐོག་གི་རྨ་ཁ་ལ་ཁྱད་པ་ཆེན་པོ་མི་འདུག་པ་གཞིགས་ན་གཉིས་ཀ་མི་གཅིག་གྲི་གཅིག་གིས་བསད་པར་ར་སྤྲོད་ཐུབ། ཞི་བདེ་ལམ་གཡོན་གྱི་གོམ་པ་བཅུའི་མཚམས་སུ་བསད་པའི་མི་དེ་ནི་ཕོ་ལོ་བཞི་བཅུ་ཡིན་ཞིང་མིང་ལ་བློ་གྲོས་སུ་འབོད། ཁོ་ནི་ཏུའུ་ཀྲང་ཟླ་བ་བཀྲ་ཤིས་ཆང་དུག་གིས་མགོ་བོ་ན་བའི་ཞོགས་པ་དེར་མཐོང་མྱོང་བའི་ཁ་བའི་ཁྲོད་ཀྱི་ཤི་བོ་དེ་ཡིན་འདུག ནང་མི་རྣམས་ཀྱིས་བཤད་པ་བལྟས་ན་མཚན་མོ་དེར་ཁོ་རང་ཕྱི་རོལ་དུ་ཆང་འཐུང་བར་འགྲོ་ནས་ཕྱི་མ་ལོག་པ་རེད་ཟེར། གཅིག་ཤོས་ནི་མོ་ལོ་ཉི་ལྔ་ལས་མེད་པའི་མིང་ལ་བཀྲ་མཚོ་ཟེར་བའི་བུ་མོ་ཞིག་རེད། ཁྱིམ་མི་རྣམས་ཀྱིས་བཤད་པ་བལྟས་ན་མོ་རང་གྲོགས་མོ་ས་ནས་ཁྱིམ་དུ་ལོག་པའི་ལམ་བར་དུ་བསད་པ་ཡིན་འདུག འདས་པོ་དེ་གཉིས་ཀྱི་ཁྱིམ་མི་བར་གཉེན་ཉེའི་འབྲེལ་བ་གང་ཡང་མེད་པ་མ་ཟད་སྔོན་མར་ཕུན་ཚུན་ཐུག་མྱོང་ཡང་མེད་པ་རེད། ཟླ་བ་བཀྲ་ཤིས་ནས་ཁྱིམ་མི་རྣམས་ཀྱི་སྐད་ཆ་རྣམས་སྒྲ་རུ་ཕབ་པ་དང་ཟིན་བྲིས་ཀྱང་བྲིས། ཁོ་རང་དེ་མ་ཐག་ཡུལ་དངོས་སུ་སོང་སྟེ་བརྟག་དཔྱད་བྱས། བློ་གྲོས་ནི་ཞི་བདེ་ལམ་ནས་གཡོན་དུ་རྨིད་བདུན་གྱི་མཚམས་སུ་བསད་པ་ཡིན་འདུག་ལ་བཀྲ་མཚོ་ནི་ཞི་བདེ་ལམ་འགྲམ་གྱི་སྒོ་རྩིས་ཚོང་ཁང་གི་སྒོ་འགྲམ་དུ་བསད་པ་ཡིན་འདུག དེ་གཉིས་བར་རྨིད་ཉི་ཤུ་རྩ་གསུམ་གྱི་བར་ཐག་ལས་མི་འདུག སྒོ་རྩིས་ཚོང་ཁང་གི་སྒོ་འགྲམ་གྱི་ཁ་བ་དག་འཁྱགས་ཟིན་པས་རྗེས་ཙམ་ཡང་ལྷག་མེད། བློ་གྲོས་གསོད་ཡུལ་དུ་ཁྲག་དང་ཕུང་པོའི་རྗེས་རང་སར་ཡོད་ཀྱང་རྐང་རྗེས་མང་པོ་ཕབ་འདུག ཁོས་རྐང་རྗེས་ཁྱད་པ་ཅན་བཅུ་གདམས་རྗེས་དེ་དག་གི་རིང་ཚད་གཞལ་བ་དང་རི་མོའི་ཁྱད་ཆོས་པར་ལེན་ནས་རྐང་རྗེས་ཀྱི་བདག་པོ་དག་ཉིན་གཅིག་ལ་འཚོལ་ཡང་མི་བཞི་ལས་མ་རྙེད། གསུམ་ནི་ལྟད་མོ་བ་དང་གཅིག་ནི་ཉེན་རྟོག་པ་ཡིན་འདུག ཁོས་རྐང་རྗེས་ལ་བསྟེན་ནས་ལག་དམར་འཚོལ་བའི་ཐབས་ལམ་དེ་དེ་ག་ཙམ་མི་བཞག་ཐབས་མེད་བྱུང་། ཁོས་མི་གསོད་ཡུལ་ལ་བདག་པོ་རྒྱག་མི་ཤེས་པའི་རྫོང་ཁའི་ཉེན་རྟོག་པ་དག་ལ་དམའ་འབེབས་ཚད་མེད་ཅིག་བྱས་བསམས་ཀྱང་ཡང་སྙིང་སུ་ཞིག་ལ་བྱེད་དགོས་པའང་མི་ཤེས་པ་འདུག ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་སླར་ཡང་ཕུང་པོ་ལ་བརྟག་དཔྱད་བྱེད་འགོ་རྩོམ། བརྟག་དཔྱད་ཞིབ་པར་བྱེད་པའི་འབྲས་བུ་དང་པོ་དེ་ནི་འདས་པོ་བློ་གྲོས་ཀྱི་སྐེ་སྡོང་ཐོག་ཏུ་ཡོད་པའི་ཁྲག་གི་མཛུབ་རྗེས་གསལ་པོ་དེ་རེད། མཛུབ་རྗེས་དེ་གློག་ཀླད་དུ་ཕབ་རྗེས་འདས་པོ་རང་ཉིད་ཀྱི་མཛུབ་རྗེས་དང་སྡུར་སྐབས་གཅིག་མཚུངས་རེད་མི་འདུག་ལ་གཞན་ཡང་ཕུང་པོ་སྤྱི་བདེ་ཅུས་སུ་འདྲེན་མཁན་གྱི་ཉེན་རྟོག་པ་རྣམས་ཀྱི་མཛུབ་རྗེས་དང་ཡང་གཅིག་མཚུངས་རེད་མི་འདུག་པས་མཛུབ་རྗེས་དེ་མི་གསོད་ལག་དམར་གྱི་མཛུབ་རྗེས་ཡིན་པ་ཁོ་ཐག་རེད། ཁོས་ཉིན་གཅིག་བེད་སྤྱད་ནས་གློག་ཀླད་ནང་བླུགས་ཡོད་པའི་རྫོང་ཁ་འདིའི་མངའ་ཁུངས་ཀྱི་མཛུབ་རྗེས་ཀུན་དང་བསྡུར་ཡང་འདྲ་པོ་གཅིག་ཀྱང་མི་འདུག མཛུབ་རྗེེས་ལ་བསྟེན་ནས་ལག་དམར་འཚོལ་བའོ་ཐབས་ལམ་དེ་དེ་ཙམ་གྱིས་བཞག་དགོས་བྱུང་། ཁོས་འདས་པོ་བཀྲ་མཚོའི་ཕུང་པོ་ལ་བརྟག་དཔྱད་བྱེད་སྐབས་མོའི་གྲོད་པ་ཅུང་འབུར་ཡོད་པ་མཐོང་། ཉིན་དེའི་ཕྱི་དྲོར་ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་སྨན་པའི་ལྭ་བ་དཀར་ལ་རིང་བ་དེ་འདྲ་ཞིག་གྱོན་ཏེ་སྤྱི་བདེ་ཅུས་ནང་འཛུལ་ཡོང་དུས་ངོ་ཤེས་རྣམས་ཀྱིས“ཏུའུ་ཀྲང་ལགས། དེ་རིང་ཁྱོད་ཨེམ་རྗེ་བྱས་པ་ཨེ་ཡིན”ཞེས་དྲིས་བྱུང་། ཁོས་སྐད་ཆ་དེ་དག་ལ་ཁ་ཡ་མ་བྱེད་པར་རོ་བཞག་ཡོད་པའི་ཁང་པ་དེའི་ལྡེ་མིག་ཕྱེ་ནས་ནང་དུ་འཛུལ། སྐབས་དེར་ཕྱི་རོལ་གྱི་ཉེན་རྟོག་པ་ཁ་ཤས་ཀྱིས“ང་ཚོའི་ཏུའུ་ཀྲང་གིས་མི་ཤི་རོ་ལ་སྨན་བཅོས་བྱེད་བཞིན་འདུག”ཅེས་འཆད་བཞིན་པ་གོ ཆུ་ཚོད་གཅིག་གི་རྗེས་ནས་ཟླ་བ་བཀྲ་ཤིས་ཕྱི་ལ་ཐོན་ཡོང་བར་ཉེན་རྟོག་པ་མང་པོས་མཐོང་ཡང་ཡོད་ཚད་ཧད་ནས་བསྡད་པ་ལས་རྩེད་དགའི་གཏམ་སྨྲ་ཕོད་མཁན་གཅིག་ཀྱང་མི་འདུག ཟླ་བ་བཀྲ་ཤིས་ཀྱི་ལྭ་བ་དཀར་པོ་དེ་ཁྲག་དམར་གྱིས་ཡོངས་སུ་སྦགས་འདུག ཅུས་ཀྲང་གིས་ཀྱང་ཕྱི་རོལ་གྱི་རྒྱུན་ལྡན་མེད་པའི་སྣང་ཚུལ་དེ་གོ་སོང་བ་འདྲ། གཞུང་ལས་ཁང་ནས་ཐོན་ཡོང་བ་དང་ཚུལ་དེ་མཐོང་བས་ཧ་ལས་ནས“ཁྱོས་ཡང་སྙིང་ཅི་ཞིག་བྱེད་པ་ཡིན”ཞེས་དྲིས། “མོའི་གྲོད་པ་ནང་ཕྲུ་གུ་འདུག”ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་འདི་ལྟར་བཤད། “གྲོད་པ་ནང་གི་ཕྲུ་གུ་དང་མི་གསོད་ཀྱི་ལག་དམར་བར་འབྲེལ་བ་ཅི་ཞིག་ཡོད་དམ” “འབྲེལ་བ་ཡོད་ནའང་ཐང” “སུས་ཁྱོད་ལ་འདས་པོའི་ཕུང་པོ་བཤའ་བའི་ཆོག་མཚན་སྤྲད་བྱུང་། ངས་སྤྲོད་བྱུང་ངམ། ཁྱིམ་བདག་གིས་སྤྲོད་བྱུང་ངམ།” ཟླ་བ་བཀྲ་ཤིས་ལ་སྐད་ཆ་གང་ཡང་མེད་པ་ལུས་སོང་། “ཉེན་རྟོག་པ་ཟེར་ན་ལག་དམར་བཟུང་མཁན་རེད། ཕུང་པོ་བཤའ་མཁན་མ་རེད། མགྱོགས་པོར་སོང་ལ་ལག་དམར་བཟུང་ནས་ཤོག”ཟེར། ཅུས་ཀྲང་གི་སྐད་ཆ་དེས་ཁོ་ལ་ཁོང་ཁྲོ་ཚད་མེད་ཅིག་ལངས་སུ་བཅུག ཁོ་རང་ཁོང་ཁྲོ་དེ་ལས་སད་དུས་སྤྱི་བདེ་ཅུས་དང་བྲལ་ནས་ཕྱི་རོལ་གྱི་སྲང་ལམ་ཐོག་ཏུ་ཐོན་འདུག སྲང་གི་མི་ཚོགས་དག་འཇིག་རྟེན་འདིའི་ནང་དུ་དོན་དག་ཅི་ཡང་བྱུང་མེད་པའང་འབྱུང་མི་སྲིད་པ་བཞིན་དལ་བྲེལ་མེད་པའི་འཚོ་བ་རོལ་བཞིན་འདུག ཁོ་རང་གློ་བུར་མི་ཚོགས་དེ་དག་གི་ཁྲོད་དུ་ཐོན་མ་ཐག རའི་ཁྱུ་རུ་ལུག་གཅིག་ཐོན་པ་བཞིན་མི་ཀུན་མཚར་སྣང་གིས་ཁོ་ལ་ལྟ་བཞིན་ཚོར། ཁོས་ཀྱང་རང་ཉིད་ལན་གཅིག་བལྟས་པས་ཨེམ་རྗེའི་སྟོད་ཐུང་དཀར་པོ་དེ་ཁྲག་གི་དམར་ཡོད་པ་རིག དེ་མུར་ལྭ་བ་དཀར་པོ་དེ་ཕུད་ནས་ལམ་འགྲམ་གྱི་གད་སྙིགས་བླུགས་སྒམ་དུ་བརྫངས། ལྭ་བ་དཀར་པོ་དེའི་ནང་གི་གོས་ལ་ཁྲག་དམར་སིམ་མི་འདུག་པས་ཁོ་རང་ཡང་སྲང་གི་མི་ཚོགས་དང་འདྲེས་སོང་། ཁོང་ཁྲོ་ལས་སད་དུས་སླར་ཡང་ཁ་བའི་ཁྲོད་དུ་ཤི་སོང་བའི་མི་དེའི་ཚུགས་ཀ་དྲན་བྱུང་། ཁོས་འདས་པོ་བཀྲ་མཚོ་ལ་ཕྲུ་གུ་འཁོར་ཡོད་པ་དེ་མི་གསོད་ཀྱི་ལག་དམར་དེ་སུ་ཡིན་ཤེས་བྱེད་ཀྱི་བདེན་དཔང་ཞིག་ཡིན་ནའང་ཐང་སྙམ། དེ་མ་ཐག་འདས་པོའི་ཁྱིམ་མི་ལ་ཁ་པར་བརྒྱབ། “ང་ཏུའུ་ཀྲང་ཟླ་བ་བཀྲ་ཤིས་ཡིན”ཟེར་དུས་ཕ་རོལ་པོས“མི་གསོད་ཀྱི་ལག་དམར་བཟུང་ཐུབ་སོང་བ་ཨེ་རེད”ཅེས་དྲིས་བྱུང་། ཁོས་འདས་པོ་བཀྲ་མཚོ་ལ་ཕྲུ་གུ་འཁོར་ཡོད་པའི་གནས་ཚུལ་དག་བཤད་མ་ཕོད་ལ་ཁོས་འདས་པོ་བཤའ་ཚུལ་ནི་དེ་བས་ཀྱང་བཤད་ག་ལ་ཕོད། ཁོས་བཀྲ་མཚོ་འདས་ཉིན་གྱི་མཚན་མོར་འཁྱམ་དུ་འགྲོ་སའི་གྲོགས་མོ་དེའི་ཁ་པར་ཨང་གྲངས་དྲིས། ཕ་རོལ་པོས་ཁ་པར་གྱི་ཨང་གྲངས་བཤད་པ་དང་ཆབས་ཅིག“མོ་དང་འབྲེལ་བ་ཡོད་ནི་མ་རེད”ཅེས་བཤད་བྱུང་། ཁོས་འདས་པོ་བཀྲ་མཚོ་གྲོགས་མོ་དེ་ལ་ཁ་པར་བཏང་ནས་རང་ཉིད་ངོ་སྤྲོད་བྱེད་རྗེས་མེ་ཏོག་ཇ་ཁང་དུ་ཐེངས་གཅིག་ཐུག་དགོས་པའི་རེ་བ་སྟོན། མེ་ཏོག་ཇ་ཁང་དུ་མི་མང་པོའི་ཟང་ཟིང་བྱེད་བཞིན་འདུག ཁང་པའི་ནང་དུ་འཛུམ་མ་ཐག་ནས་ཇ་བསྲུབས་མའི་དྲི་མ་ཞིམ་པོ་ཞིག་སྣ་ལམ་དུ་འཐུས་འོང་། ཏུའུ་ཀྲང་ཟླ་བ་བཀྲ་ཤིས་ཟུར་གྱི་འདུག་ས་སྟོང་པ་ཞིག་ཏུ་ཙོག་རྗེས་ཞབས་ཞུ་མ་འབོད་ནས“ཇ་བསྲུབས་མ་ཚ་དམ་གང་དགོས”ཅེས་བཤད། ཇ་ཚ་དམ་གང་ལང་སང་ཁུར་བྱུང་། ཁོས་ཇ་འཐུང་བཞིན་འཐུང་བཞིན་འདས་པོ་བཀྲ་མཚོའི་གྲོགས་མོ་ལ་བསྒུགས། སྐར་མ་བཅུ་ལྔ་ཙམ་འདས་རྗེས་སྒོ་ནས་བུ་མོ་ལོ་ཆུང་ཞིག་ཐོན་བྱུང་ལ་མོས་ཅི་ཞིག་བོར་བ་ལྟར་བུའི་རྣམ་འགྱུར་གྱིས་ཇ་ཁང་གི་གང་སར་མིག་ཉུལ་བྱེད་བཞིན་འདུག ཁོས་ལག་པ་ཡར་བཀྱངས་དང་མོ་རང་ཁོ་སར་སླེབས་བྱུང་། “དགོངས་པ་མ་ཚོམས། ང་རང་རྗེས་སུ་ལུས་སོང”ཟེར། “སྐྱོན་མེད། ཁྱོད་ཀྱི་ཚུ་ཚོད་དེ་རྩྭ་ཐང་གི་ཆུ་ཚོད་ཡིན་པ་རེད”ཅེས་བཤད་རྗེས་རང་ཉིད་ཧ་ཧ་དགོད་ཀྱང་མོ་རང་ངུ་ནག་ནས་བསྡད་པ་ལས་དགོད་ཙམ་ཡང་བྱས་མ་སོང་། ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་གཞི་ནས་རང་ཉིད་ཉེན་རྟོག་པ་ཞིག་ཡིན་པ་དྲན་ཞིང་གདོང་ལ་ཉེན་རྟོག་པའི་རྣམ་འགྱུར་ཞིག་བསྡུས་དང“དེ་ཉིན་འདས་པོ་བཀྲ་མཚོ་ཅིའི་ཕྱིར་ཁྱོད་སར་ཡོང་བ་རེད། དུས་ནམ་ཞིག་ལ་ཁ་བྲལ་བ་རེད”ཅེས་དྲིས། “མོ་རང་མོ་རང་ཉིན་དེ་འཁྱམས་ཡོང་བ་རེད། ངའི་སྡོད་ཁང་ནས་ཐོན་དུས་མཚན་མོའི་ཆུ་ཚོད་ཆུ་ཚོད་བཅུ་གཅིག་བཅུ་གཉིས་ཙམ་ཡིན”མོས་ངག་དིག་བཞིན་འདི་ལྟར་བཤད་རྗེས་ཇ་ཕོར་བསླངས་ནས་འཐུང་ཁོལ་བྱས་ཀྱང་ལག་པ་འདར་བཞིན་འདུག་པས་འཐུང་ཐུབ་མ་སོང་། ཁོས་དལ་བར“ཁྱོས་བཀྲ་མཚོ་ལ་ཕྲུ་གུ་འཁོར་ཡོད་པ་ཤེས་སམ”ཞེས་དྲིས། “ཤེས” “མོ་རང་གི་ནང་མི་ཚོས་ཨེ་ཤེས” “ཤེས་གི་མ་རེད” “ཁྱོས་ཅི་ལྟར་ཤེས་པ་ཡིན། ཕྲུ་གུ་དེ་སུ་དང་འབྲེལ་བ་ཡོད” “བཀྲ་མཚོ་ལ་བློ་གྲོས་བསྟན་པ་ཟེར་བའི་དགའ་རོགས་ཤིག་ཡོད། ཁོ་གཉིས་འགྲོགས་ནས་ཟླ་གཅིག་གི་རྗེས་སུ་བློ་གྲོས་བསྟན་པས་མོ་དང་ལུས་འབྲེལ་བྱེད་དགོས་པའི་རེ་བ་སྟོན་པ་རེད། ཡིན་ཡང་བཀྲ་མཚོས་མོས་མཐུན་བྱས་མི་འདུག ཉིན་ཞིག་ས་སྲོས་ཀྱི་སྐབས་སུ་ཁོ་གཉིས་འཛོམས་ནས་ཁ་བྲལ་ཁར་བློ་གྲོས་བསྟན་པས‘ཉིན་དེ་དག་གི་རིམ་ང་རང་འདོད་པས་བཟི་ནས་བསྲན་མི་ཐུབ་གི ད་ཁྱོས་དང་ལེན་མ་བྱས་ཚེ་ང་གཉིས་སོ་སོ་བྲལ་བ་ལས་འོས་མེད’ཟེར་བར་བཀྲ་མཚོ་ནས་དང་ལེན་བྱེད་འདུག བཀྲ་མཚོས་ང་ལ‘བུ་མོ་ཚོ་ས་སྲོས་ཀྱི་སྐབས་སུ་ཕོ་གསར་དང་འཕྲད་མི་ཉན་གི འདི་དག་ཡོད་ཚད་དུས་ཀྱིས་ལན’ཞེས་བཤད་བྱུང”མོས་སྣང་དོགས་མེད་པར་འདི་ལྟར་བཤད་རྗེས་ཇ་ཕོར་ཡར་བླངས་ཏེ་འཐུང་ཙམ་འཐུང་ཙམ་བྱས། ད་ནི་ལག་པ་འདར་གྱིན་མི་འདུག ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་བུ་མོ་འདིས་སྐད་ཆ་འདི་འདྲ་ཞིག་བཤད་ཡོང་བསམ་པའི་གཏིང་ལའང་མེད་པས་ངོ་གདོང་ཡང་དམར་པོར་གྱུར་ཅིང“དེ་ནས་ཕྲུ་གུ་ཅི་ལྟར་འཁོར་བ་རེད”ཅེས་སྟབས་སྲབས་པོས་དྲིས། “ཉིན་དེར་ཁོ་གཉིས་མགྲོན་ཁང་སྟབས་བདེ་ཞིག་ནང་འཛུལ་ཞིང་བཀྲ་མཚོས’སྲུང་ཤུགས་གྱོན་རྒྱུ་ཨེ་ཡོད’ཅེས་འདྲི་དུས་བློ་གྲོས་བསྟན་པས‘སྲུང་ཤུགས་མེད། ངས་ནང་དུ་ཤོར་མི་ཡོང’ཞེས་དམ་བཅས་ཀྱང་མཇུག་མཐར་ནང་དུ་ཤོར་འདུག”བུ་མོ་དེས་འདི་ལྟར་བདེན་ཅོག་ཅོག་གིས་བཤད་ཡོང་དུས་ད་ནི་བུ་མོ་དེ་ནི་ཉེན་རྟོག་པ་དང་ཟླ་བ་བཀྲ་ཤིས་ནི་སྐད་ཆ་འདྲི་ཡུལ་དེ་ཡིན་པའི་སྣང་བ་སྐྱེས། ཟླ་བ་བཀྲ་ཤིས་ལ་འདྲི་རྒྱུ་ཅི་ཡང་མི་འདུག ཁོ་གཉིས་སྐད་ཆ་ཅི་ཡང་མེད་པར་ཇ་བསྲུབས་མ་འཐུང་བཞིན་བསྡད་དུས་མི་གཞན་གྱིས་བལྟས་ན་ཁོ་གཉིས་ནི་ད་སོ་མ་གསར་དུ་འགྲོགས་པའི་དགའ་རོགས་ཆ་གཅིག་དང་འདྲ། ཁ་བྲལ་ཁར་མོ་ལ་བློ་གྲོས་བསྟན་པའི་ཁ་པར་ཨང་གྲངས་དྲིས། མོས“སྐད་འཕྲིན་ཡོད”ཟེར་ནས་སྐད་འཕྲིན་གི་ཨང་གྲངས་བསྟན་བྱུང་། ཁོས་བློ་གྲོས་བསྟན་པའི་སྐད་འཕྲིན་མནན་རྗེས་ཀྱི་ཆུ་ཚོད་གཅིག་གི་རྗེས་སུ་དང་ལེན་བྱས་སོང་། དེ་དུས་ཁོ་རང་བུ་མོ་དེ་དང་བྲལ་ནས་ཕྱི་རོལ་དུ་ཐོན་ཡོད། བློ་གྲོས་བསྟན་པའི་སྐད་འཕྲིན་དུ་ཟླ་བ་བཀྲ་ཤིས་ཁོ་རང་ནི་ཉེན་རྟོག་པ་ཡིན་པ་དང་ཁྱོད་ལ་སྐད་ཆ་ཁ་ཤས་འདྲི་རྒྱུ་ཡོད་ཅེས་བཤད་ནས་ཐུག་རྒྱུ་བྱས། བློ་གྲོས་བསྟན་པ་ནི་ལྷའི་བུ་འདྲ་བུ་ཞིག་འདུག ཉེན་རྟོག་པ་ལ་སྐྲག་སྣང་ཅི་ཡང་མེད་པར་སྐད་ཆ་མ་འདྲི་གོང་ནས“ཨ་ཁ། མིའི་འདོད་པ་ཟེར་བ་དེ། ཨ་ཁ་ཁ། མིའི་འདོད་པ”ཞེས་ཡང་ཟློ་བཞིན་འདུག་པས་སྐད་ཆ་འདྲི་རྒྱུ་ཅང་མེད་སོང་། བཀྲ་མཚོས་དུས་ཚོད་ལ་ཁ་གཡར་བ་དང་བློ་གྲོས་བསྟན་པས་མིའི་འདོད་པ་ལ་ཁ་གཡར་བ་གཉིས་ལ་འདྲ་ས་ཞིག་འདུག་སྙམ། མཐར་བློ་གྲོས་བསྟན་པ་ལ་ཤོག་དཀར་གྱི་ཐོག་ཏུ་མཛུབ་ཐིག་ཅིག་བརྒྱབ་ཏུ་བཅུག་རྗེས་སྤྱི་བདེ་ཅུས་སུ་ལོག སྤྱི་བདེ་ཅུས་ཀྱི་གློག་ཀླད་ནང་མཛུབ་ཐིག་དེ་བླུགས་ཏེ་འདས་པོའི་སྐེ་སྡོང་དུ་ཡོད་པའི་མཛུབ་ཐིག་གསལ་པོ་དེ་དང་བསྡུར་ཡང་ཆ་འགྲིག་གི་མི་འདུག ཁོས་མི་གསོད་ལག་དམར་གྱི་མཛུབ་ཐིག་དེ་ལ་ཡུན་རིང་ཞིག་ཅེར་ཡང་མཛུབ་ཐིག་དེས་བདག་པོ་སུ་ཡིན་བཤད་མི་སྲིད་པ་འདྲ། ནམ་ཡང་འགྱུར་བ་མེད་པ་གློག་ཀླད་དུ་འཇམ་ཐིང་ངེར་གནས་ཀྱང་དེའི་བདག་པོ་འཚོལ་མཁན་གྱི་མིག་ལ་ན་ཟུག་ཚད་མེད་སྟེར། ཁོ་རང་གཞུང་ལས་ཁང་ནས་ཐོན་པ་དང་འཇམ་ཐིང་ཐིང་རེད་འདུག་པ་ན་ལས་ཀ་གྲོལ་འདུག་ཨང་སྙམ་སྐབས། “ཟླ་བ་བཀྲ་ཤིས། ཟླ་བ་བཀྲ་ཤིས”མི་ཞིག་གིས་འདི་ལྟར་འབོད་ཡོང་བས་ཟླ་བ་བཀྲ་ཤིས་ཟེར་བའི་མིང་སྔོན་དུ་ཏུའུ་ཀྲང་ཞེས་མི་སྣོན་པ་འདི་སུ་ཡིན་ནམ་སྙམ་བཞིན་ཕྱི་ལྟ་ཞིག་བྱས་པ་ན་ཅུས་ཀྲང་རེད་འདུག ཅུས་ཀྲང་ཁོའི་འགྲམ་དུ་སླེབས་མ་ཐག་བྱམས་བརྩེ་ལྡན་པའི་ངང་ནས“རང་གིས་རང་ལ་གཅིག་ལྟོས་དང་། རང་གི་ལུས་པོ་སྒང་བུའི་དབུགས་ཤོར་བ་བཞིན་དུ་འགྱུར་འདུག”ཅེས་བཤད། ཁོས་སྐད་ཆ་དེ་ཉན་རྗེས་ཅུས་ཀྲང་གི་བྱམས་བརྩེ་དེ་ནི་རྫུན་མ་ཞིག་རེད་ཨང་དེའི་ནང་དུ་དུག་ཅིག་སྦས་ཡོད་པ་འདྲ་སྙམ་གྱི་ལན་མི་བཏབ་པ་ཡོད་དུས་ཅུས་ཀྲང་གིས་མུ་མཐུད་དུ“ངས་ཁྱོད་ལ་དགོངས་པ་ཉིན་གཉིས་སྤྲོད་ཆོག ལུས་སེམས་གཉིས་ཀ་ངལ་ཡག་པོ་ཞིག་གསོས”ཟེར། ཟླ་བ་བཀྲ་ཤིས་ཀྱིས“ལགས་སོ”ཟེར་ཏེ་གོམ་པ་མགྱོགས་ཙམ་སྤོས་ནས་ཅུས་ཀྲང་གི་མདུན་དུ་སོང་། “འོ། ད་དུང་ཡང་དོན་ཞིག་འདུག ཁྱོད་རང་སྲང་དུ་སོང་ནས་བདེ་སྲུང་རུ་ཁག་གིས་ལས་ཀ་ཅི་འདྲ་བྱེད་བཞིན་འདུག་པ་གཅིག་ལྟོས་དང”ཅུས་ཀྲང་གིས་རྒྱབ་ནས་འདི་ལྟར་འབོད། ཁོས་སླར་ཡང“ལགས་སོ”ཟེར་རྗེས་སྲང་ལམ་གྱི་ཕྱོགས་སུ་སོང་། ལམ་འགྲམ་གྱི་བདེ་སྲུང་ས་ཚིགས་ཤིག་གིི་ཁང་པའི་ནང་དུ་འཛུལ་བ་ན་ནང་དུ་ཡོད་པའི་ཉེན་རྟོག་པ་ཀུན་གྱིས་ཟང་ཟིང་བྱེད་བྱུང་། ཁ་ཤས་ཀྱིས་སྦི་རག་ཤེལ་དམ་དག་སྒྲོག་ཙིའི་འོག་ཏུ་བཞག་པ་དང་ལ་ལས་ཤོག་བག་རྣམས་པར་ཁུག་ནང་བཅུག་སོང་། ཡིན་ཡང་ཟླ་བ་བཀྲ་ཤིས་ཀྱི་དོ་སྣང་བྱ་ཡུལ་ནི་དེ་དག་མིན་པར་སྒྲོག་ཙི་གཞན་ཞིག་ཏུ་ཡོད་པའི་ཤ་རྭ་གསར་པ་ཆ་གཅིག་དེ་རེད། ཤ་རྭ་ལས་ཁྲག་ཟགས་བཞིན་པ་མཐོང་བས“འདི་ཅི་རེད”ཅེས་དྲིས། ཉེན་རྟོག་པ་ཁ་བདེ་ཞིག་གིས“དེ་རིང་ང་ཚོས་ཤ་རྭ་འཚོང་མཁན་ཞིག་བཟུང་ཐུབ་སོང”ཟེར། ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་ཁོའི་རྣམ་འགྱུར་དང་སྐད་གདངས་ལས་རྫུན་ལབ་བཞིན་པ་ཤེས་ཏེ“རྫུན་མ་ལབ”ཅེས་སྐད་ཆེན་པོས་སྡིགས་པས་ཁ་བདེ་དེའི་མཛུམ་མདངས་ཡལ་ཞིང་བསྐྲགས་ནས་ཕྱིར་ནུར་བྱས་སོང་། ཁོ་ཚོའི་ནང་གིས་ཙུག་ཀྲང་དེས་དེ་རིང་ཏུའུ་ཀྲང་ངོ་མ་ཁྲོས་ཡོད་པ་ཤེས་པ་འདྲ། ཏུའུ་ཀྲང་ཁྲིད་ནས་བདེ་སྲུང་ས་ཚིགས་ཀྱིས་ཁང་པ་ཆུང་ཆུང་དེ་དང་བྲལ་ཏེ་ཕྱི་རོལ་དུ་ཐོན། པར་ཁུག་ནས་ཀྲུང་ཧྭ་རྟགས་ཅན་གྱི་ཐ་མག་བག་ཆ་གཅིག་བླངས་ཏེ་དུ་རྐང་གཅིག་ཏུའུ་ཀྲང་ཟླ་བ་བཀྲ་ཤིས་ལ་སྦྱིན་ཞིང་མེ་སྒྲོན། ཁོ་རང་གིས་ཀྱང་དུ་རྐང་གཅིག་མཐེན་སོང་། དུས་ཅུང་ཙམ་འདས་རྗེས་ཏུའུ་ཀྲང་གི་ཁོང་ཁྲོ་དལ་བར་ཡལ་འགྲོ་བ་དང་བསྟུན་ནས“ཏུའུ་ཀྲང་ལགས། ཁྱེད་ཀྱིས་ཤེས་གསལ་རེད། ང་ཚོ་ཁྱོད་དང་མི་འདྲ། ང་ཚོ་ལ་རིག་གནས་གང་ཡང་མེད། ང་ཚོ་ནི་སྤྱི་ཚོགས་ཐོག་གི་ལས་མེད་མཁྱམས་པོ་ཁ་ཤས་ཡར་བསྡུས་ནས་ཉེན་རྟོག་པ་བཟོ་བ་དེ་འདྲ་རེད། གནས་ལུགས་དང་ཁྲིམས་མི་ཤེས་པས་ཁྱོད་ཀྱི་གོ་ཡང་གནང་རོགས”ཟེར། “ཡང་སྙིང་ཤ་རྭ་དེ་ཅི་ལྟར་བྱུང་སོང” “ཏུའུ་ཀྲང་ལགས། ངས་དྲང་པོ་བཤད་ན་ཁྱོས་ང་ཚོ་ལ་གོ་ཡང་གནང་ཆོག་གམ” “ཆོག་གི”ཁོའི་ཁ་ནས་འདི་ལྟར་ཤོར། “ཏུའུ་ཀྲང་ལགས། ཁྱེད་ཀྱིས་ཤེས་གསལ་རེད། མེ་མདའ་རྣམས་གོང་རིམ་གྱིས་ཡར་སྡུད་བཞིན་ཡོད་རེད། ད་ལོ་བྱིས་གྲོངས་ཤང་དུ་མེ་མདའ་འབོར་ཆེན་ཞིག་བསྡུ་རྒྱུ་བྱུང་བས་ཁ་སང་ང་ཚོ་ལེན་དུ་ཤོག་ཟེར། མེ་མདའ་བཞི་བཅུ་ཙམ་དང་མདེའུ་གྲངས་མེད་འདུག གོང་རིམ་དུ་ཡར་སྤྲོད་པའི་འཛིན་དུ་མེ་མདའ་བཞི་བཅུ་ལས་གཞན་མདེའུ་རྣམས་འགོད་མི་འདུག་པས་ང་ཚོས་ལམ་བར་མེ་མདའ་རྒྱག་བཞིན་རྩེད་མོ་རྩེ་བཞིན་ཡོང་བ་ཡིན། ཏག་ཏག་ནགས་ཚལ་ཞིག་བརྒྱུད་སྐབས་སུ་མི་གཅིག་གིས‘མོ་ཊ་དེ་ག་བཞག་དང་རི་དགས་འདུག’ཟེར། མོ་ཊ་བཞག་མ་ཐག་བསམ་གྲུབ་ཀྱིས་མེ་མདའ་བརྒྱབ་བྱུང་ལ‘ཕོག་ཐལ། ཕོག་ཐལ’ཟེར་བྱུང་། མེ་མདའ་དེ་ཤ་བའི་མགོ་བོ་ལ་ཕོག་འདུག གོམ་པ་ཁ་ཤས་བྲོས་རྗེས་དེ་གར་ཤི་འདུག ཁ་ཤས་ཀྱིས་དེང་སང་ཤ་རྭ་ལ་གོང་ཆེན་པོ་རག་གི་རེད་ཟེར་བས་ང་ཚོས་ཤ་རྭ་བཅད་དེ་ཁུར་ཡོང་བ་ཡིན”ཞེས་བཤད་རྗེས་ཏུའུ་ཀྲང་གི་རྣམ་འགྱུར་ལ་བལྟས་བྱུང་། ཏུའུ་ཀྲང་ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་སྐད་ཆ་དེ་ཐོས་རྗེས་ཁོང་ཁྲོ་ཚད་མེད་ཅིག་ལངས་ནའང་གཅིག་ནས་སྔོན་དུ་གོ་ཡང་གནང་བའི་ཁ་ཆད་བྱས་ཟིན་པ་དང་གཉིས་ནས་གོང་རིམ་ལ་ཡར་ཞུས་ན་རང་ཉིད་ཀྱི་མགོ་ཐོག་ཏུའང་གྱོང་ཞིག་ཡོང་རྒྱུ་རེད་སྙམ། ཁོ་རང་བདེ་སྲུང་ས་ཚིགས་ཀྱི་ཁང་ཆུང་དུ་འཛུལ་ཏེ“ཁྱོད་ཚོ་བཙོན་ཁང་དུ་འགྲོ་བསམ་གི་འདུག་གམ”ཞེས་སྐད་ཤུགས་ཆེན་པོས་སྡིགས་རྗེས་སྲང་ལམ་ནས་སྤྱི་བདེ་ཅུས་ཀྱི་ཕྱོགས་སུ་བསྐྱོད། ཁོ་རང་སྒོ་སྲུང་བསྡད་སྐབས་ཀྱང་ཉེན་རྟོག་པ་དེ་ཚོ་ལ་རྒྱུས་ཅུང་ཙམ་ཡོད། ཁོ་ཚོར་དོན་དག་གང་ཡང་མེད་ཀྱང་སྤྱི་བདེའི་མོ་ཊ་ཁྲ་ཁྲ་དེ་བཏང་ནས་ཤགས་སེར་སྒོ་ནས་ཡར་འཛུལ་ཡོང་ལ་ཤགས་སེར་མར་འཕུར་འགྲོ འདི་ལྟར་མོ་ཊ་ཁྲ་ཁྲ་དེ་བཏང་ནས་ལས་ཁུངས་དང་ས་ཚིགས་གང་འདྲ་ཞིག་ཏུ་ཐོན་ཡང་འགོག་མཁན་མེད། མོ་ཊ་ཁྲ་ཁྲ་དེའི་ནང་དུ་ཁྲིམས་མེད་ཀྱི་དངོས་པོ་ཅི་ཞིག་བཅུག་ཡོད་ཀྱང་སྐྲག་དགོས་དོན་མེད། ཁོས་གནད་ཚུལ་འདི་ཡར་ཞུ་བྱེད་ཆོག་ཀྱང་ད་ལྟ་ཁོས་སྤྱི་ཚོགས་འདི་ལ་རྒྱུས་ལོན་ཡོད་པས་ཁོས་གླེན་རྟགས་དེ་རིགས་བསྟན་མི་སྲིད། གལ་སྲིད་གནད་དོན་འདྲ་བྱུང་ན་གནད་དོན་བྱུང་སའི་དཔོན་ཆུང་བ་དེས་འགན་ཁུར་དགོས་པ་ལས་དཔོན་ཆེ་བ་དག་གིས་འགན་ཁུར་མི་སྲིད། གནད་དོན་དེ་ཡར་ཞུ་བྱས་སོང་ན་འགན་འཁུར་མཁན་ནི་ཁོ་རང་ཉིད་ཡིན་པ་ལས་ཅུས་ཀྲང་སོགས་ཡིན་མི་སྲིད། འགན་འདེད་འོང་དུས་ང་ལ་ཉེས་པ་མེད་ཟེར་ས་ཡོད་མ་རེད། དེང་གི་སྤྱི་ཚོགས་ཀྱི་ནང་དུ་མི་ཞིག་ཉེས་པ་ཅན་དུ་བཟོ་བ་ལས་སླ་མོ་རེད། དེ་ནི་མར་ཕོད་མི་ཐུབ་པའི་ཞྭ་མོ་ནག་པོ་ཞིག་བསྐོན་པས་ཆོག ཁོ་རང་སྤྱི་བདེ་ཅུས་ནང་གི་སྡོད་སར་སླེབས་སྐབས་ཁོ་རང་སྒོ་སྲུང་ཡིན་དུས་ཀྱི་སྡོད་ས་དེ་དྲན་བྱུང་། ད་ལྟ་སྒོ་སྲུང་ཡིན་དུས་ཀྱི་ཁང་པ་དེ་དང་བྱ་བ་རྣམས་ལ་ཕྱི་དྲན་ཞིག་བྱས་ན་དེ་དག་ནི་ལོ་ངོ་དུ་མ་ཞིག་འདས་ཟིན་པའི་སྣང་བ་ཤར། སྒོ་སྲུང་ཡིན་དུས་སྐྱག་པ་གཏོང་དུས་ཀྱང་ཤུག་སྒོམ་དགོས་ཀྱི་མེད། ད་ལྟ་ཟ་མ་བཟའ་དུས་ཀྱང་མགྱོགས་ཤིག་ཤིག་གིས་ཟ་དགོས། ཁོ་རང་ཁང་པ་འདིའི་ནང་གི་ཧྲ་ཧྥ་ཐོག་ཏུ་ཙོག་དུས་ཁང་པ་འདི་ནི་དེ་འདྲའི་རྒྱ་ཆེ་ལ་དེ་འདྲའི་སྟོང་བའི་སྣང་བ་ཤར་བ་དང་གད་སྙིགས་ཀྱིས་གང་ཡོད་པའི་སྒོ་སྲུང་ཡིན་དུས་ཀྱི་ཁང་པ་དེ་ནང་ལོག་པའི་འདོད་པ་སྐྱེས་ལ་སྔོན་མའི་ཆང་གྲོགས་དག་དྲན་ཞིང་ནག་འཚོང་རྒྱག་མཁན་གྱི་སྦྱིན་བདག་གམ་གྲོགས་པོ་ཚེ་བཟང་ཡང་དྲན་བྱུང་། ཉིན་འདི་གར་རྫོང་གི་སྲང་ལམ་ཐོག་ཏུ་རྟ་མཆོག་རྟགས་ཅན་གྱི་མོ་ཊ་དེ་མཐོང་རྒྱུ་མི་འདུག་ལ་དེའི་བདག་པོ་འཛུམ་དུས་ཀྱི་གསེར་སོ་དང་གསེར་སོ་ལས་འཕྲོས་པའི་ཟུར་ཟའི་འོད་དང་བཅས་པའང་མཐོང་རྒྱུ་མི་འདུག ཞིབ་པར་བསམས་ན་ད་ལྟ་ཁོ་གཉིས་ནི་དགྲ་བོར་གྱུར་ཟིན། གཅིག་ནི་ནག་འཚོང་རྒྱག་མཁན་ཞིག་དང་གཅིག་ཤོས་ནི་གྲོགས་པོས་ནག་འཚོང་རྒྱག་བཞིན་པ་ཤེས་མཁན་གྱི་ཉེན་རྟོག་པ་ཞིག རྗེས་མར་ཁོ་གཉིས་ཐུགས་སྐབས“གྲོགས་པོ་ཆང་འཐུང་བར་འགྲོ”ཟེར་དགོས་སམ་ཡང་ན“གྲོགས་པོ་དགོངས་པ་མ་ཚོམས། ཁྱོད་རང་བཙོན་ནང་འཛུལ་རོགས”ཟེར་དགོས། ཁོས་ཤེས་མ་སོང་། མ་འོངས་པ་ཟེར་བ་དེ་ནམ་ཡང་མི་རིག་སའི་མདུན་དུ་གབ་ཡོད་པ་ན་སུས་ཀྱང་ཚོད་དཔག་བྱས་མི་ཐུབ། ཅུས་ཀྲང་ནས་ཁོ་ལ་དགོངས་པ་ཞག་མ་གཉིས་གནང་ཡང་དོན་དུ’ཁ་བའི་ནང་གི་མི་གསོད་དོན་རྐྱེན’གྱི་ཐག་གཅོད་བྱེད་པའི་དབང་ཆ་དེ་མི་གཞན་ཞིག་གི་ལག་ནང་དུ་བཞག་བསམ་པ་མ་ཡིན་ནམ། ཁོས་འདི་ལྟར་བསམ་གྱིན་ཕྱི་རོལ་དུ་ཐོན་ཞིང་ལྷ་སའི་སྦི་རག་དམ་བེ་གསུམ་ཉོས་ཏེ་ཁོ་རང་སྒོ་སྲུང་ཡིན་དུས་ཀྱི་སྡོད་སའི་ཕྱོགས་སུ་ཕྱིན། ལྡེ་མིག་ཟྭ་ནང་བཅུག་ཏེ་སྒོ་ཕྱེ་མ་ཐག མིག་ལམ་དུ་མངོན་ཡོང་བ་ནི་མི་སུས་ཀྱང་སྒུལ་མེད་པའི་གད་སྙིགས་དང་གད་སྙིགས་ཁྲོད་ཀྱི་བཟའ་འཕྲོ་ལུས་པའི་ཕོར་བ་སོགས་རེད། མཐར་ན་དེ་སྔོན་ཧྲ་ཧྥ་རྙིང་ཧྲུལ་དེའི་ཐོག་ཏུ་བུ་མོ་ཆ་མེད་ཅིག་དང་འདོད་པ་སྤྱོད་པའི་དྲི་མ་དེའང་ད་དུང་མ་ཡལ་བར་མཁའ་དབུགས་ཀྱི་ཁྲོད་དུ་འཁྲིགས་ཡོད་པ་ཚོར། ཁོས་བུ་མོ་གཞན་ཞིག་ཀྱང་ཧྲ་ཧྥ་དེའི་ཐོག་ཙོག་ཏུ་བཅུག་མྱོང་། བུ་མོ་དེས“འདིི་ཅིའི་དྲི་མ་རེད”ཅེས་དྲིས། ཁོས“འདི་ནི་སྐྱེས་པ་ཕོའི་དྲི་མ་རེད་ཡ”ཞེས་ཟེར་མྱོང་། གད་སྙིགས་ཁྲོད་ཀྱི་དངོས་པོ་དག་རིན་ཐང་ཅི་ཡང་མི་འདུག ཁོས་དེ་དག་རེ་རེ་བཞིན་ཡར་བླངས་རྗེས་སླར་གད་སྙིགས་ཀྱི་ཁྲོད་དུ་མར་གཡུགས། ཉལ་ཁྲིའིི་གཡས་གཡོན་དུ་ལྷམ་ཆ་གསུམ་འདུག ལྷམ་ཆ་གསུམ་གྱི་གཡས་གཡོན་དུ་ཨུ་མུ་སུ་མང་པོ་འདུག གལ་སྲིད་ཁོ་རང་ནི་ཚེ་བཟང་ཡིན་ན་ཨུ་མུ་སུ་དག་ཡར་བླངས་ནས་དྲི་མ་སྣུམ་ཙམ་སྣུམ་ཙམ་བྱེད་ངེས་རེད། དེ་སྔོན་ཟླ་བ་བཀྲ་ཤིས་དང་ཚེ་བཟང་གཉིས་འགྲོན་ཁང་སྐར་ལྔ་ཅན་ཞིག་ཏུ་ཞག་སྡོད་བྱེད་དུས་ཚེ་བཟང་གིས་ཨུ་མུ་སུ་ཡ་གཅིག་ཕུད་རྗེས་དྲི་མ་སྣུམ་ཙམ་སྣུམ་ཙམ་བྱས་ཏེ“ཅི་ཡིན་ན་རྐང་པའི་དྲི་མ་ཇེ་ངན་དུ་འགྱུར་བཞིན་འདུག”ཅེས་ཟེར། ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་ཚུལ་དེ་མཐོང་བས་ཞེན་པ་ལོག་ནས“སློབ་འབྲེང་གི་སྐབས་སུ་ཡོད་པའི་ཁྱོད་ཀྱི་སྲོལ་བཟང་དེ་ད་ལྟ་སྦྱིན་བདག་ལ་འགྱུར་ཡང་དོར་མི་འདུག་ཨང”ཞེས་ཟུར་ཟ་བྱས་མོད། ཚེ་བཟང་ནི་སྣང་དོགས་མེད་པར་ཨུ་མུ་སུ་ཡ་གཞན་དེའང་སྣུམ་ཙམ་སྣུམ་ཙམ་བྱས་སོང་། དེ་རིང་རང་ཉིད་ཀྱི་ཉལ་ཁྲི་འགྲམ་དུ་ཨུ་མུ་སུ་མང་པོ་མཐོང་བས་ཚེ་བཟང་དྲན་བྱུང་། ཡིན་ཡང་རང་ཉིད་ནི་ཚེ་བཟང་མིན་པས་ཨུ་མུ་སུའི་དྲི་མ་སྣུམ་མི་སྲིད། དེ་དག་རེ་ཞིག་སྐྱོར་བཞག་པ་དང་མལ་ཁྲིའི་ཐོག་གི་མལ་གོས་ཡར་འཐེན་པ་ན་རུལ་དྲི་ངན་པ་ཞིག་སྣ་ནང་འཛུལ་བྱུང་ལ་འོག་ཏུ་PLAYBOYརྟགས་ཅན་གྱི་བྱ་སྤུ་སྟོད་ཐུང་ཞིག་ཐོན་བྱུང་། དེ་ནི་དངུལ་ཕོགས་དང་པོ་སྤྲོད་ཡོང་དུས་སུ་ཁོས་སྒོར་བརྒྱད་བརྒྱ་སྤྲད་ནས་ཉོས་པ་རེད། ཁོས་ཞིང་གྲོང་ནས་ཡོང་བའི་ངོ་ཤེས་དག་ལ“ངས་བྱ་སྤུ་སྟོད་ཐུང་འདི་ལ་སྒོ་བརྒྱད་བརྒྱ་སྤྲད་པ་ཡིན”ཞེས་ངོམ་ངོམ་བྱས་མྱོང་། ཡིན་ན་ཚེ་བཟང་གིས་བྱ་སྤུ་སྟོད་ཐུང་འདིར་ལ་ལག་པས་རེག་ཙམ་རེག་ཙམ་བྱེད་རྗེས“རྫུན་མ་རེད་འདུག”ཅེས་ལྟ་ཚུལ་བརྗོད་སོང་། ཁོ་རང་སྐད་ཆ་དེ་ལ་ཡིད་རྩ་བ་ནས་མགུ་མ་སོང་། དེ་ནི་བྱ་སྤུ་སྟོད་ཐུང་དེ་རྫུན་ཡིན་པས་མ་ཡིན་པ་སྟེ་མི་མང་པོའི་ཁྲོད་དུ་རང་ཉིད་ཀྱི་ལ་རྒྱ་བརླགས་སོང་བས་རེད། PLAYBOYརྟགས་ཅན་གྱི་བྱ་སྤུ་སྟོད་ཐུང་གི་གང་སར་སྐྱུག་པའི་ཁེངས་འདུག དེ་སྔོན་ཁོ་རང་ཆང་གིས་ཇི་ལྟར་བཟི་ཡང་སྟོད་ཐུང་གོང་མཐོ་པོ་དེ་ལ་གཅེས་སྤྲད་བྱེད་ཀྱི་ཡོད་མོད། ཡིན་ཡང་ཁ་བ་འབབ་པའི་མཚན་མོར་ཁོ་རང་ཆང་གིས་བད་ནས་ཡོད་ཚད་བརྗེད་འདུག པར་ཁུག་གཡས་པའི་ནང་དུ་སྒོར་ཟུར་ལྔ་ལས་ཅི་ཡང་མི་འདུག ཁོས་རེ་བ་ཐམས་ཅད་པར་ཁུག་གཡོན་པར་བརྐྱངས་མོད། པར་ཁུག་གཡོན་པའི་ནང་དུ་ཐང་ཆུའི་ཕྱི་སྐོགས་ཤོག་གུ་ཞིག་ལས་ཅི་ཡང་མི་འདུག བྲང་ངོས་ཀྱི་པར་ཁུག་ལ་ལག་ཉུལ་ཞིག་བྱེད་དུས་དེའི་ནང་དུ་དངོས་པོ་རྡོག་རྡོག་ཅིག་ཡོད་པ་ཚོར། ཁོས་དེ་ཡར་བླངས་པ་ན་དེ་ནི་ཁྲག་དམར་གྱིས་ཁེངས་པའི་གྲི་ཐུང་ཐུང་ཞིག་ཡིན་འདུག གྲི་ཐུང་ཐུང་དེ་ནི་ཁོ་རང་སྒོ་སྲུང་ཡིན་དུས་པུའུ་ཀྲང་གིས“སྒོ་ནས་མི་ཆ་མེད་ཅིག་ཡར་འཛུལ་ཡོང་ཚེ་ཐོབ་ཐང་ལག་ཁྱེར་སྟོན་དུ་བཅུག་དགོས། ཁྱོས་དེ་དག་གི་མིང་ཐོ་གསལ་བོར་འགོད་དགོས”ཅེས་ཁོ་ལ་ཤོག་གུ་དང་ཞ་སྨྱུག ཞ་སྨྱུག་གཞོག་བྱེད་ཀྱི་གྲི་ཆུང་མང་པོ་སྤྲད་མྱོང་། པར་ཁུག་ནང་གི་ཁྲག་གི་ཁེངས་པའི་གྲི་ཆུང་དེའང་པུའུ་ཀྲང་གིས་ཁོ་ལ་སྤྲད་མྱོང་བའི་གྲི་ཆུང་དེ་འདྲ་ཞིག་རེད། ཁོས་ཁྲག་གི་ཁེངས་པའི་གྲི་ཆུང་དེ་མཐོང་མ་ཐག་དངངས་སྐྲག་གང་ཡང་བྱུང་མ་སོང་། དེ་ལས་ལྡོག་ཏེ་ཁོས་ཡུན་རིང་ཞིག་ལ་འཚོལ་བཞིན་པའི་མི་གསོད་ཀྱི་ལག་དམར་དེ་གློ་བུར་རྙེད་སོང་བ་ལྟ་བུའི་དགའ་སྣང་ཡ་མཚན་ཞིག་སྐྱེས་བྱུང་། ཁོས་ལག་ཤུག་དཀར་པོ་ཞིག་གྱོན་ནས་མཛུབ་རྗེས་མི་ཕོག་པར་བྱས་ཏེ་ལག་རས་ཀྱིས་ཁྲག་གི་ཁེངས་པའི་གྲི་ཆུང་ཆུང་དེ་ཡང་ཡང་ཕྱིས་རྗེས། སྟོད་ཐུང་གི་པར་ཁུག་ནང་བཅུག་ནས་ཕྱི་རོལ་དུ་ཐོན། ཁོ་རང་སྒོ་སྲུང་ཡིན་དུས་ཀྱི་གོམས་གཤིས་ལྟར་མི་རིང་བའི་སའི‘མཚོ་ཆེན་པོ’ཞེས་པའི་ཆང་ཁང་གི་ཕྱོགས་སུ་སོང་། ཆང་ཁང’མཚོ་ཆེན་པོ’རུ་ནུབ་གླིང་གི་ཆང་རག་དཀར་ཁྲ་དམར་ཁྲ་མང་པོ་ཡོད་ཅིང་ཁོས་དེ་ཚོ་རེ་རེ་བཞིན་འཐུང་ནས་གང་ཞིམ་གྱི་ཚོད་ལྟ་བྱས་མྱོང་། ཁོས་ཆང་ཤེལ་དམ་རེ་ཉོས་ཚེ་སྦྱིན་བདག་བཟོད་པས་ཤེལ་དམ་དེ་ཡར་གཡུགས་མར་གཡུགས་ཡང་ཡང་བྱེད་རྗེས་ཁ་ཕྱེ་བ་དང་ཁོ་ལ་སྤྲོད་སྲིད། སྦྱིན་བདག་ནས“ཆང་ཡང་ཡང་དུ་གཡུགས་རྗེས་དྲི་མའང་འདྲ་གི་མ་རེད”ཟེར་བས་ཁོས་ཀྱང“ཁྱོད་རང་ནི་ཨ་རག་གི་ཆེད་མཁས་རེད”ཅེས་ཟེར་བར་སྦྱིན་བདག་བཟོད་པ་ཤིན་ཏུ་དགའ་སོང་། སྦྱིན་བདག་བཟོད་པའི་དགའ་བ་མ་ཡལ་བའི་གོང་དུ་བཟོད་པ་དང་བཟོ་པ་གཉིས་སྒྲ་གདངས་འདྲ་བས། སྒོ་སྲུང་ཟླ་བ་བཀྲ་ཤིས་ཀྱིས“ཨ་རོགས་བཟོ་པ། ཁྱོད་ཆང་དམ་ཡར་གཡུགས་མར་གཡུགས་བྱེད་ཤེས་པ་ལས་གཞན་ཅི་བཟོ་ཤེས”ཅེས་དྲིས། སྦྱིན་བདག་བཟོད་པས“ངས་མི་བཟོ་ཤེས”ཟེར་བས་དེ་ཡོད་ཀྱི་མི་རྣམས་འུར་རེ་དགོད་བྱུང་། སྒོ་སྲུང་གིས་དགོད་ཁ་བསྲན་ནས“དེ་ན་ང་ལ་བུ་མོ་སྙིང་ལ་རྗེ་མོ་ཞིག་བཟོ་རོགས”ཟེར་བར་བཟོད་པས“ཁྱོད་ཀྱི་དགའ་རོགས་ང་ལ་མཚན་གང་གཡོར་དང་། ངས་ཁྱོད་ལ་བུ་མོ་ཞིག་བཟོ་ཆོག”ཅེས་པས་ཟླ་བ་བཀྲ་ཤིས་ཀྱང་རང་དབང་མེད་པར་བགད། ཆང་ཁང‘མཚོ་ཆེན་པོ’རུ་ཆང་དྲི་སྣ་མང་ཁ་ཞིང་མིག་གིས་བལྟས་ན་དེ་དག་ནི་དུ་སྤྲིན་རབ་རིབ་ཅིག་ཏུ་གྱུར་ནས་ཡལ་བ་མེད་པར་མཁའ་དབུགས་སུ་འཁྲིགས་ནས་འདུག དེའི་ཁྲོད་དུ་སེམས་ནད་མེད་པར་ཙོག་པའི་སྣང་བ་དེས་ཁོའི་ཁེར་རྐྱང་གི་ལོ་ཟླ་ལ་བདེ་སྐྱིད་སྩོལ་མྱོང་། ཉེན་རྟོག་པ་བྱེད་རྗེས་ཁོས་བདེ་སྐྱིད་དེའི་རིགས་ནི་འཁྲུལ་སྣང་ངོ་སྙམ། ཁོ་རང་སྒོ་ནས་ཐོན་པ་མཐོང་ཚེ་སྦྱིན་བདག་བཟོད་པས“སྒོ་སྲུང་ལགས་ཕེབས་སམ། ཉིན་མང་པོ་ཞིག་ལ་ཁྱོད་རང་མཐོང་རྒྱུ་མི་འདུག”ཟེར་རྗེས་རང་གི་མགོ་བོར་ལག་པས་བརྡུངས་ཏེ“ངའི་དྲན་པ་འདི། ད་ལྟ་སྒོ་སྲུང་ག་ལ་ཡིན་ཏུའུ་ཀྲང་རེད་ཡ། ཏུའུ་ཀྲང་ཟླ་བ་བཀྲ་ཤིས་ལགས། ཅི་ཞིག་འཐུང་བསམ་དོ། རྒུན་ཆང་དམར་པོ་ཨེ་ཡིན”ཟེར། དུས་རྒྱུན་ཡིན་ན་ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་ཁོ་ལ་མཚར་གཏམ་མང་པོ་ཞིག་བཤད་སྲིད་མོད། དེ་རིང་ངོ་ནག་ནས་འདུག་ལ“དེ་རིང་ཆང་མི་འཐུང་། ང་ལ་དོན་དག་ཡོད”ཅེས་བཤད། “ཆང་ཁང་དུ་འོང་རྗེས་ཆང་མི་འཐུང་མཁན་སུ་ཞིག་ཡོད”ཟེར་བཞིན་ཤེལ་དམ་དམར་པོ་ཞིག་ཡར་གཡུགས་མར་གཡུགས་བྱས་ཏེ་ཁ་དབྱེ་གྲབས་བྱེད་སྐབས་གློ་བུར་ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་ངོ་གདོང་རྒྱུན་ལྡན་མེད་པར་ནག་ཡོད་པ་མཐོང་བས་ཤེལ་དམ་རང་སར་བཞག “དེ་ན་རྒུན་འབྲུམ་གྱི་ཁུ་བ་མཆོད”ཟེར་བཞིན་ཤེལ་དམ་གཞན་ཞིག་ཁ་ཕྱེ་སྟེ་པེ་ཙིའི་ནང་དུ་བླུགས་སོང་། ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་པེ་ཙེ་དེ་ཕར་ཙམ་འཕུལ་པ་དང་ཉེན་རྟོག་པའི་ཚུགས་ཀ་དང་བཅས“ངས་ཁྱོད་ལ་དོན་དག་ཞིག་འདྲི་རྒྱུ་ཡོད”ཅེས་བཤད། སྦྱིན་བདག་བཟོད་པས་ཡར་གཡགས་མར་གཡུགས་བྱེད་བཞིན་པའི་ཤེལ་དམ་དེ་གཡུགས་མཚམས་བཞག་ཏེ་ཞབས་ཞུའི་སྒྲོག་ཙེ་ལ་ལུས་པོ་བསྙེས་བ་དང“ཅི་ཞིག”ཅེས་དྲིས་བྱུང་། “ཚེས་བཅུ་གསུམ་གྱི་མཚན་མོར་སུ་ཞིག་ཆང་གིས་བད་ནས་དྲན་པ་མི་གསལ་བར་འགྱུར་འདུག” “འོ། ཁྱོས་ཁ་བ་འབབ་པའི་མཚན་མོ་དེར་ཟེར་བ་ཡིན་ནམ” ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་རྒུན་ཆང་ཁུ་བའི་ཁེངས་པའི་པེ་ཙེ་དེ་ཚུར་བླངས་ནས་འཐུང་ཙམ་འཐུང་ཙམ་བྱས། “འོ། ཁ་བ་འབབ་པའི་མཚན་མོ་དེར་ཁྱོད་རང་ཆང་གིས་བད་ནས་དྲན་པ་མི་གསལ་བར་འགྱུར་འདུག ཆང་ཁང་ནང་དུ་ཁྱོད་དང་ཕོ་གསར་ཁ་ཤས་བར་ཁ་རྩོད་ཤོར་སོང་། ངས་བར་བྱས་པ་མ་གཏོགས་རྒྱག་རེས་ཤོར་ངེས་རེད། ཁྱོད་རང་ཆང་ཁང་ནས་ཐོན་ཁར‘དེ་རིང་སུ་ཞིག་དང་འཕྲད་ན་དེ་དམྱལ་བར་བརྫང་རྒྱུ་ཡིན’ཞེས་ཀི་བརྒྱབ་བྱུང” སྐད་ཆ་དེ་ཉན་རྗེས་ཟླ་བ་བཀྲ་ཤིས་རབ་ཏུ་ཁྲོས་བྱུང“ངས་ཁྱོད་ལ་དེ་ཉིན་ང་རང་ཅི་ལྟར་བཟི་འདུག་ཅེས་འདྲི་བ་མིན”ཅེས་པེ་ཙེ་སྒྲོག་ཙེའི་ཐོག་ཏུ་ཐག་སྒྲ་དང་བཅས་བཞག་པས་དེ་ཡོད་ཀྱི་མི་ཚང་མས་ཁོ་གཉིས་ཀྱི་ཕྱོགས་སུ་བལྟས། སྦྱིན་བདག་བཟོད་པས་ས་འཛུམ་མུལ་མུལ་ངང“དེ་རིང་ངའི་གྲོགས་པོ་བཟི་འདུག”ཅེས་གསལ་བཤད་བྱས། སྦྱིན་བདག་བཟོད་པས་ད་ལྟའི་ཏུའུ་ཀྲང་ཟླ་བ་བཀྲ་ཤིས་དང་དེ་སྔོན་གྱི་སྒོ་སྲུང་ཟླ་བ་བཀྲ་ཤིས་གཉིས་མི་འདྲ་བར་རྟོགས་ནས“བཟི་བོ་ཆོས་རྒྱལ་ཆང་ཁང་ནས་ཡར་འཛུལ་ཡོང་དུས་ཀྱང་བཟི་ནས་དྲན་པ་མི་གསལ། ལྟོས་དང་དེ་རིང་ཡང་བཟི་འདུག”ཅེས་བཤད་རྗེས་ཆང་ཁང་གི་ཟུར་ཞིག་ཏུ་བལྟས། བཟི་བོ་ཆོས་རྒྱལ་སྒྲོག་ཙི་ཞིག་གི་ཐོག་ཏུ་ཁ་ཆུ་ཡང་ཤོར་བཞིན་ཉལ་འདུག སྒྲོག་ཙིའི་ཟུར་གྱི་ཨ་རག་དཀར་པོ་ཕྱེད་ཙམ་ཡོད་པའི་ཤེལ་དམ་དེ་མར་ལྷུང་ལ་ཁད་འདུག ཟླ་བ་བཀྲ་ཤིས་ཀྱང་བཟི་བོ་མང་པོའི་ཁྲོད་དུ་ཙོག་ནས་རྒུན་འབྲུམ་ཁུ་བ་འཐུང་བཞིན་སླར་ཁ་བ་འབབ་པའི་མཚན་མོར་དེ་དྲན། དེ་ཉིན་ཁོས་རྒུན་ཆང་ཤེལ་དམ་ཆེན་པོ་ཞིག་ཉོས་པ་དང་ཤེལ་དམ་གྱི་ངོས་སུ་ལྟ་དུས་3%ཞེས་དྲིས་འདུག་པས་ཁོས་དེ་ནི་ཆང་གི་ངར་ཚད་ཀྱི་ཏུའུ་ལ་ངོས་འཛིན། ཁོས་ཏུའུ་གསུམ་ལས་མེད་ན་ངར་ཚད་སྦི་རག་དང་གཅིག་པ་ཡིན་འདུག་སྙམ་བཞིན་ཚད་མེད་དུ་འཐུང་། ད་ལྟ་ཕྱི་དྲན་ཞིག་བྱས་ན་3%དེ་ནི་རྒུན་འབྲུམ་འདྲེས་ཚད་ཡིན་པ་དང་སྤྱི་རྒུན་ཆང་ཕལ་ཆེར་ཏུའུ་བཅུ་ནས་ཉི་ཤུ་ཙམ་ཡོད་པ་རེད། ཁོས་རྒུན་ཆང་འདི་བསྟུད་མ་ལན་ཁ་ཤས་འཐུང་བ་དང་རིམ་གྱིས་དྲན་པ་མི་གསལ་བར་གྱུར། དེ་ཉིན་གྱི་ཉེས་པ་ཡོད་ཚད་རྒུན་ཆང་ལ་རགས་ལས་པ་དང་ལྷག་ཏུ་གོ་ནོར་ཤོར་སླ་བའི་ཨང་ཀི3%ལ་རགས་ལས་འདུག་པས་རང་ཉིད་ཆང་བད་ནས་དྲན་པ་ཤོར་བའམ་དེ་རྗེས་སུ་ཉེས་པ་བསགས་པ་ལའང་ཁག་མི་འདུག་སྙམ། དེ་ནི་ཁོས་ལྔ་ལམ་དུ་བརྫངས་པའི་འདས་པོ་བཀྲ་ཚོས་ཉེས་པ་ཡོད་ཚད་ས་སྲོས་ལ་ཁ་གཡར་བ་དང་མོའི་དགའ་རོགས་བློ་གྲོས་བསྟན་པས་མིའི་འདོད་པ་ལ་ཁ་གཡར་མྱོང་བ་བཞིན། དེ་རིང་ཁོས་ཉེས་པ་ཡོད་ཚད་ཤེལ་དམ་ངོས་ཀྱི་ཨང་ཀི་3%ལ་ཁ་གཡར་སོང་། ཁོས་འདི་ལྟར་བསམ་སྐབས་བཟི་བོ་ཆོས་རྒྱལ་གཉིད་ལས་སད་པ་དང་སྒྲོག་ཙེའི་ཟུར་དུ་ཡོད་པའི་ཤེལ་དམ་དེ་ལྷུང་ལ་ཁད་ལྷུང་ལ་ཁད་བྱེད། ཡིན་ཡང་ཤེལ་དམ་དེ་ཁོའི་ལུས་ཀྱི་ཆ་ཤས་ཤིག་ཡིན་པ་བཞིན་ལག་པ་བརྐྱངས་ཐེངས་གཅིག་གིས་བཟུང་བ་དང་ཡར་བཀྱགས་ཏེ་ཨ་རག་དཀར་པོ་འགྲིན་པའི་ནང་དུ་བླུགས་སོང་། དེ་ནི་མི་སྐོམ་ཡོད་པ་ཞིག་གིས་ཆུ་འཐུང་བ་དང་ཨ་ན་མ་ན་རེད། ཟླ་བ་བཀྲ་ཤིས་ཀྱི་ཉེན་རྟོག་པའི་གཟབ་ནན་གྱི་མིག་ཟུང་གིས་བཟི་བོ་ཆོས་རྒྱལ་གྱི་འགུལ་བསྐྱོད་ཆུང་ངུ་ཙམ་ལའང་དོ་སྣང་བྱེད་བཞིན་དུས་ཡུན་རིང་ཙམ་ཞིག་འདས་པ་དང་བཟི་བོ་ཆོས་རྒྱལ་གྱིས་འགུལ་བསྐྱོད་ཆེན་པོ་ཞིག་བྱེད་བྱུང་། དེ་ནི་ཁོ་རང་ཡར་ལངས་ཏེ་ཆང་འཐུང་མཁན་མང་པོའི་དཀྱིལ་ནས་ཚུར་འོང་བ་དེ་རེད། ཁོ་རང་དེ་མུར་ཟུར་གྱི་གསང་སྤྱོད་ཕྱོགས་སུ་བསྐྱོད་སོང་བས་ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་ད་འགྲིག་སོང་སྙམ་བཞིན་ཁོ་རང་དེད་དེ་སོང་། ཟླ་བ་བཀྲ་ཤིས་གསང་སྤྱོད་ཀྱི་སྒོ་ཁར་ཐ་མག་རྐང་གཅིག་ལ་མེ་སྒྲོན་རྗེས་དེ་བརྔུབས་ནས་འགྲིང་བསྡད། ཡིན་ཡང་བཟི་བོ་དེ་གསང་སྤྱོད་ནས་མར་ཐོན་རྒྱུ་མི་འདུག ཁོས་དུ་རྐང་གཅིག་དེ་མ་རྫོགས་བར་བརྔུབས་རྗེས་ནང་དུ་འཛུལ་གྲབས་བྱེད་དུས་སྒོ་ཧྲ་སེ་ཕྱེ་བྱུང་བ་དང་མཉམ་དུ་གཅིན་དྲི་སྣ་ལ་གཟན་པ་ཞིག་དྲོ་འོང་། ཁོས་དེ་མ་ཐག་ཏུ་ལག་ཤུག་གྱོན་ཡོད་པའི་ལག་པས་པར་ཁུག་ནང་དུ་གྲི་ཐུང་དེ་ལེན་པ་རྩོམ་མོད། བཟི་བོ་ཆོས་རྒྱལ་གྱི་སྣམ་བུ་དངལྷམ་ཡོད་ཚད་གཅིན་གྱི་བརླན་ཡོད་པ་མཐོང་མ་ཐག་ཅུང་ཙམ་ཧད་དགོས་བྱུང་། བཟི་བོ་དལ་བར་ཁོ་རང་བརྒྱུད་ནས་འགྲོ་བ་དང་བསྟུན་ནས་གྲི་ཐུང་དེ་བཟི་བོའི་སྟོད་ཐུང་པར་ཁུག་ནང་དུ་བཅུག ཟླ་བ་བཀྲ་ཤིས་གསང་སྤྱོད་དུ་འཛུལ་རྗེས་གཅིན་གཏོང་ཁུལ་བྱས་ཀྱང་གཅིན་ཐིགས་གཅིག་ཀྱང་འོང་རྒྱུ་མི་འདུག གཅིན་དྲི་སྟུག་པོ་དེའི་ཁྲོད་དུ་ཡུན་རིང་བསྡད་མི་བཟོད་པས་སྣམ་བུ་ཡར་འཐེན་ཙམ་བྱེད་རྗེས་གསང་སྤྱོད་ནས་མར་ཐོན། བཟི་བོ་སྦྱིན་པ་གོམ་པ་ཕྱ་ཕྱོར་གྱིས་མི་མང་པོའི་ཁྲོད་ནས་ཕར་འགྲོ་བཞིན་འདུག མི་རྣམས་མཚར་སྣང་གིས་བཟི་བོ་སྦྱིན་ལ་ལྟ་བཞིན་འདུག ལ་ལ་དགོད་བཞིན་འདུག་ལ་ཁ་ཤས་ཀྱིས་ལག་པས་སྣ་བཀག་འདུག མི་ཚོགས་ཀྱི་ཁྲོད་དུ་བཟི་བོ་ངོ་ཤེས་མཁན་ཞིག་གིས་སྐད་བརྒྱབ་བྱུང“ཨ་རོགས་སྦྱིན་པ། ཁྱོད་རང་གསང་སྤྱོད་དུ་འགྲོ་བ་དེ་གཅིན་སྣམ་བུར་གཏོང་དུ་སོང་བ་ཡིན་ནམ”ཟེར་བས་མི་ཀུན་ཧ་ཧ་དགོད། སྦྱིན་པས་དེ་དག་ལ་ཁ་ཡ་ཅི་ཡང་མ་བྱས་པར་རང་གི་འདུག་སར་ཙོག་རྗེས་ཤེལ་དམ་ཁ་ལ་སྤྲད། ཆང་ཁམ་གང་མིད་རྗེས་ཐ་མག་བག་ཆག་གཅིག་གི་ནང་དུ་དུ་རྐང་གཅིག་ཡར་འཐེན་ནས་ཁ་ནང་བཅུག་སོང་། པར་ཁུག་ཏུ་ལག་པ་ཉུལ་བཞིན་གྲི་ཐུང་དེ་བླངས་བྱུང་། ཁོས་དེ་ནི་མེ་ཆ་ཡིན་ནམ་སྙམ་པ་འདྲ། ལྟ་ཡག་ཅིག་བྱེད་རྗེས་སླར་པར་ཁུག་ནང་མར་བཅུག མཐར་པར་ཁུག་ཡོད་ཚད་བསློགས་ཀྱང་མེ་ཆ་རྙེད་མ་སོང་། སླར་ཡང་སྒྲོག་ཙིའི་གཡས་གཡོན་དུ་ཅི་ཞིག་འཚོལ་ཉམས་བྱེད་པའི་སྐབས་དེར་ཏུའུ་ཀྲང་ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་ཁ་པར་ནང་༡༡༠མནན། “ལྦེ” “ང་རང་ཏུའུ་ཀྲང་ཟླ་བ་བཀྲ་ཤིས་ཡིན། ཁྱོད་ཚོ་མགྱོགས་པོར་ཆང་ཁང‘མཚོ་ཆེན་པོ’རུ་ཤོག མི་གསོད་མཁན་རྙེད་ཐལ” སྐར་ལྔའི་རྗེས་སུ་ཉེན་རྟོག་པ་མང་པོ་ཞིག་ཆང་ཁང་གི་སྒོ་ནས་བདེ་བདེ་སངས་སངས་འདྲ་བྱས་ནས་འཛུལ་ཡོང་། ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་དེ་ནི་ཁོའི་འགྲམ་དུ་འཁྲབ་སྟོན་འདྲ་བྱེད་བཞིན་པ་གསལ་པོ་རྟོགས་ཤིང་དཔོན་མེད་སར་ཁོ་ཚོར་སྟོབས་ཤུགས་ཅི་ཡང་མེད་པར་ལྷོད་ལྷོད་དུ་གྱུར་སྲིད་ལ་དཔོན་ཡོད་སར་ཁོ་ཚོ་གཉིད་ལས་སད་པ་ལྟར་བྱས་སྲིད། ཉེན་རྟོག་པ་རྣམས་འདི་ལྟར་ཐོན་ཡོང་དུས་ཆང་ཁང་འདི་འཇམ་ཐིང་ཐིང་དུ་གྱུར། ཁོ་ཚོ་ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་གཡས་གཡོན་དུ་འཚང་ནས་ཁོའི་བཀོད་འདོམས་ལ་བསྒུགས་འདུག “ཁོ་བཟུང་ནས་སྤྱི་བདེ་ཅུས་སུ་ཁྲིད་ཅིག”ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་བཟི་བོ་སྦྱིན་པར་འཛུབ་མོ་གཏོད་བཞིན་འདི་ལྟར་བཤད་མ་ཐག ཉེན་རྟོག་པ་རྣམས་མགྱོགས་ཤིག་ཤིག་ངང་བཟི་བོ་བཟུང་བ་དང་གྲིལ་མ་འདུད་ཀྱིས་མར་ཁྲིད་སོང་། བཟི་བོའི་ཁ་ནང་ཡོད་པའི་དུ་རྐང་དེ་ཁ་ཆུས་བརླན་འདུག་ལ་སྐྲག་པ་ཡིན་ནམ་གྲང་བས་ཡིན་གཅིན་གྱིས་བརླན་པའི་བརླ་སིག་སིག་འདར་བཞིན་འདུག “ང་ལ་ཅི་ཞིག་བྱུང་སོང་། ང་ལ་དོན་དག་ཅི་ཞིག་བྱུང་སོང་ངམ”བཟི་བོས་འདི་ལྟར་འབོད་པའི་སྐད་སྒྲའི་ཁྲོད་དུ་སྦྱིན་བདག་བཟོད་པའི་གདོང་ལ་གླེན་འཛུམ་ལྟས་ངན་པ་ཞིག་ཤར་ཡོད་པ་མཐོང་བས་ཟླ་བ་བཀྲ་ཤིས་ཀྱང་སྟབས་སྟབས་པོས་ཆང་ཁང་དང་བྲལ་དགོས་བྱུང་། ཕྱི་རོལ་གྱི་མུན་ཁྲོད་དུ་མི་ལ་སུན་སྣང་བཟོ་བར་བྱེད་པའི་ཉེན་རྟོག་པའི་མོ་ཊ་ལྔའི་གློག་འོད་སྔོན་པོ་ནི་སྙིང་གི་འཕར་འགྲོས་དང་མཉམ་དུ་ཡང་ཡང་འཚེར་བཞིན་འདུག ཟླ་བ་བཀྲ་ཤིས་ཀྱིས་གློག་འོད་དེ་མཐོང་མ་ཐག་སེམས་ལ་དགའ་བ་ཡ་མཚན་ཞིག་སྐྱེས་བྱུང་བ་དང“ཏུའུ་ཀྲང་ལགས། ཁྱེད་རང་མོ་ཊ་ནང་བཞུགས་རོགས”ཉེན་རྟོག་པ་ཞིག་གི་མོ་ཊ་རྗེས་མ་དེའི་སྒོ་ཕྱེ་བྱུང་། ཁོ་རང་ནང་དུ་འཛུལ་གྲབས་བྱེད་སྐབས་མོ་ཊའི་ཁ་ལོ་པས་འཛུམ་མུལ་མུལ་གྱིས་ཁོ་ལ་ལྟ་བཞིན་འདུག་པ་མཐོང་བས། སྣང་བ་ཞིག་ལ་རང་ཉིད་ཀྱིས་ཉེན་རྟོག་པའི་མོ་ཊ་གཏོང་སྐབས་ཉེན་རྟོག་པ་ངོ་མ་ཞིག་ཡིན་པ་དང་གཞན་གྱིས་ཉེན་རྟོག་པའི་མོ་ཊ་བཏང་ནས་རང་ཉིད་ཁུར་འགྲོ་དུས་རང་ཉིད་ནི་བཟུང་མ་ཐག་པའིི་ཉེས་ཅན་ཞིག་ཡིན་པའི་སྣང་བ་ཤར། ཁོས་ཉེན་རྟོག་པའི་མོ་ཊ་ནང་བཅུག་ཟིན་པའི་མགོ་བོ་དེ་མར་འཐེན་པ་དང་སྒོ་ཧྲག་གེ་ཡར་བརྒྱབ། “ཁྱོད་ཚོ་སྔོན་ལ་རྒྱུགས། ང་རྗེས་སུ་སླེབས་ཡོང” མོ་ཊ་ལྔ་པོ་ཏུང་ཏུང་གི་སྒྲ་དང་བཅས་མཚན་གྱི་སྲང་ལམ་སྟོང་པ་རུ་བྱ་འཕུར་བ་ལྟར་བརྒྱུགས་སོང་། ཁོས་པར་ཁུག་བསློགས་ནས་ཅི་ཞིག་འཚོལ་ཉམས་བྱས་ཀྱང་ལག་ཤུགས་དཀར་པོ་ཞིག་ལས་རྙེད་རྒྱུ་མི་འདུག དེ་མུར་དབུགས་རིང་ཞིག་གཏོང་བ་དང་།. ཁ་ནས་ཐོན་མ་ཐག་པའི་དབུགས་དཀར་པོ་དེ་མུན་གྱི་ནམ་འཁར་རུ་ཕྱོར་བཞིན་པ་མཐོང་འོང་། ཁོས་རང་ཉིད་ཀྱིས་སྐྱུགས་པའི་རླངས་པ་དཀར་པོ་ལ་ལྟ་ཡག་ཅིག་བྱེད་དུས་དེའི་ཁྲོད་དུ་སྐར་མ་ཆུང་ཆུང་ཞིག་འདར་བཞིན་འདུག སྐར་མ་དེ་ངའི་དོན་དུ་འདར་བ་ཡིན་ནམ། ས་གཞི་ཆེན་པོའི་ཟུར་ཀུན་དུ་མུན་ནག་ལས་ཁ་བ་སྐར་མ་ཙམ་ཡང་མཐོང་རྒྱུ་མེད་པ་འདི་ཅི་ཡིན་ནམ། དཔྱིད་ཀ་འཁོར་བ་ཡིན་ནམ་སྙམ་མོད་འདྲི་ལན་ཡོད་ཚད་ཁོའི་སེམས་ཁུང་ནས་ཤོར་ཏེ་ཐག་རིང་དུ་བྲོས་སོང་།
རྩོམ་སྒྲིག་པ། གངས་དགའ།

(མཆོད་མེའི་པར་དབང་གཞན་གྱིས་དགོལ་མི་རུང)

རྩོམ་ཁུངས། ངེད་དྲ་བ།
