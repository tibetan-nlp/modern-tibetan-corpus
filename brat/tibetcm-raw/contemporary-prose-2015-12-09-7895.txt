རྒྱུད་མངས་མཚོ་མོའི་ཟིན་བྲིས།
དཀར་ཁུང་།༼ ཉི་ཧོང ༽

ངའི་ཁང་བའི་ཉེ་ས་ན་མཚོ་ཆུང་ཆུང་ཞིག་ཡོད། ཁོམ་ལོང་རེ་བྱུང་ན་རྒྱུན་དུ་མཚོ་ཁར་ཕྱིན་པ་ཡིན། མཚོ་དེའི་མིང་ལ་རྒྱུད་མངས་མཚོ་མོ་①ཟེར་ལ། ཧ་ལམ་ལོ་ངོ་ཁྲི་བཞི་བརྒྱའི་ཡར་སྔོན་དུ་གྲུབ་ཅིང་། ཉི་ཧོང་གི་སྐམ་སའི་ནང་རོལ་དུ་ཡོད་པའི་ཆེས་ཆེ་བའི་མཚོ་རེད། རྒྱ་ཁྱོན་ལ་སྤྱི་ལེ་གྲུ་བཞི་མ་670.4ཡོད་ཅིང་། རང་རེའི་མཚོ་སྔོན་པོའི་དྲུག་ཆའི་གཅིག་ཙམ་རེད། མཚོ་དེ་ནི་ཉི་ཧོང་གི་རྒྱལ་ས་རྙིང་བ་ཁྱོའོ་ཐོ་དང་གྲོང་ཁྱེར་ཆེ་བ་ཨོ་ས་ཁ་གཙོར་བྱས་པའི་ཉི་ཧོང་ནུབ་རྒྱུད་ཀྱི་མི་དུང་ཕྱུར་གཅིག་དང་བྱེ་བ་བཞིའི་འཐུང་ཆུ་ཡིན་ལ། ནང་དུ་འཚོ་བའི་ཉ་རིགས་བཞི་བཅུ་ལྷག་དང་། ཆུ་བྱ་ཁྲི་དྲུག་ལྷག་གི་འཚོ་གནས་ཀྱི་གཞི་རྟེན་དུ་གྱུར་ཡོད། ཉིན་ཤས་སྔོན། ཉི་ཧོང་གི་རིག་གནས་ངོ་སྤྲོད་ཅེས་པའི་སློབ་ཁྲིད་ཀྱི་ནང་དོན་ལྟར། ང་ཚོ་ཕྱི་འགྲིམས་སློབ་མ་སྐོར་ཞིག སྐྱེ་དངོས་དང་རྒྱ་མཚོར་ཞིབ་འཇུག་བྱེད་ཀྱིན་པའི་དགེ་རྒན་གཉིས་ཀྱི་སྣེ་ཤན་འོག ཆེ་ཆུང་རན་པའི་སློབ་གྲྭའི་ཆེད་སྤྱོད་ཀྱི་མཚོ་གྲུ་ཞིག་ལ་བསྡད་ནས། མཚོ་ཐོག་ལ་ལྟ་སྐོར་དུ་བསྐྱོད་པ་ཡིན། མཚོ་དེའི་ནང་དུའང་གླིང་ཕྲན་ཆུང་ངུ་མང་དུ་མཆིས་ཀྱང་། རེབ་འདིར་མཚོ་མཐའ་ནས་བརྩམས་ཏེ་སྐར་མ་ཉི་ཤུར་བགྲོད་དགོས་པའི་གླིང་ཕྲན་ཆུང་ངུ་ཞིག་གི་སྟེང་དུ་ཕྱིན་པ་ཡིན། གླིང་ཕྲན་དེའི་ཆེ་ཆུང་ནི་སློབ་གྲྭའི་རྩལ་སྦྱངས་ར་བའི་རྒྱ་ཁྱོན་ལས་འགོངས་མི་འདུག འོན་ཀྱང་། འཛིངས་ཤིང་སྟུག་པའི་ནགས་ཚལ་དང་། དཀྱིལ་དུ་བཞེངས་ཡོད་པའི་ལྷ་ཁང་སོགས་ཀྱི་བཀོད་པས་མཐོང་བ་ཙམ་གྱིས་ཡིད་དབང་མེད་དུ་བཀུག་འགྲོ ལྷ་ཁང་དེའི་ནང་རྒྱུན་པར་ཇོ་མོ་ཞིག་བཞུགས་ཡོད་དེ། དེ་གར་མཆོད་བཤམས་ཀྱི་ལས་བགྱི་བཞིན་ཡོད་ཚུལ་ཐོས་བྱུང་ཡང་། ཉིན་དེར་མ་མཐོང་ངོ་། ཤར་ཕྱོགས་ན་བྲག་ལོག་ཅིག་ཡོད་དེ། མཐོ་ཚད་ལ་མི་ཚད་བཞི་ལོངས་པ་ཙམ་རེད། སྟེང་དུ་རྒྱ་ཡིག་འགའ་ཤས་བྲིས་ཡོད་མོད། རྒན་ལགས་ནས་ཀྱང་དོན་དགྲོལ་མ་ཐུབ་པར་རེ་ཞིག་ལུས། སློབ་མ་དགྱ་ནག་པ་འགའ་དེ་ལ་གླན་ནས་པར་བརྡབས་པ་མཐོང་བས། དེར་ཡོད་པའི་དགེ་རྒན་དང་ཨ་རི་བ་ཁ་ཤས་གད་མོ་དགོད་ཀྱིན་འདུག ཕྱིར་ཡོང་ཁར་རྒན་རྣམ་པ་གཉིས་ནས་མཚོ་ངོས་ནས་ཐུར་དུ་རྨིད་སུམ་ཅུའི་མཚམས་ཀྱི་ཆུ་བླངས་ནས་ང་ཚོར་བླུད་བྱུང་། ཆུ་དེ་ཧ་ཅང་འཁྱགས་པས། རྒན་གཉིས་ཀྱིས་ཀྱང་མགྲོན་བསུ་ཆེན་པོ་ཞིག་བྱས་པ་ལྟ་བུའི་ཉམས་སྟོན་ཀྱིན་འདུག རྒན་ལགས་ནས་ངོ་སྤྲོད་གནང་བ་ལྟར་ན། ལོ་ཤས་སྔོན་ལ་མཚོ་འདིར་ཡང་རང་རེའི་མཚོ་སྔོན་པོ་ནང་བཞིན་བཙོག་བསྒྱུར་ཚབས་ཆེན་བྱས་པས། སྲིད་གཞུང་ནས་དག་སེལ་གྱི་བྱ་བའི་མགོ་བརྩམས་ཏེ། མཚོའི་ནང་དུ་དག་སེལ་ཆས་མང་པོ་བཞག་ཡོད་ལ། མཚོའི་དྭངས་གཙང་གི་ཆ་ལ་དམིགས་ནས། དེའི་སྐོར་གྱི་ཤེས་བྱ་མང་དག་ཅིག་མང་ཚོགས་ལ་ངོ་སྤྲོད་བྱེད་ཀྱིན་ཡོད། ལྷག་དོན་དེའི་མཐའ་སྐོར་དུ་འཚར་ལོངས་བྱུང་བའི་བུས་པ་ཞིག་ཡིན་ན། སློབ་ཆུང་ལོ་རིམ་ལྔ་བའི་སྐབས་སུ་ངེས་པར་ཟླ་གཅིག་གི་རིང་ལ་སློབ་གྲྭའི་མཚོ་གྲུར་བསྡད་ནས་མཚོ་ཐོག་ཏུ་བསྐྱོད་དེ། མཚོ་མོའི་ལོ་རྒྱུས་དང་། བཙོག་སྒྱུར་གནས་ཚུལ། དག་སེལ་གྱི་གལ་ཆེན་རང་བཞིན་སོགས་ཀྱི་སློབ་ཁྲིད་ལ་མཉན་དགོས་ཚུལ་གསུངས་བྱུང་ལ། མཚོ་མོའི་དག་སེལ་དང་མཐའ་སྐོར་གྱི་མིའི་སེམས་ལ་“དག་སེལ”བྱས་པ་བརྒྱུད། ད་ལྟ་མཚོ་མོའི་དག་གཙང་གི་ཚད་95%ཡན་ལ་སོན་ཡོད་ཅེས་ངོམ་སོ་བྱས་འོངས། མིའི་དབུགས་ཀྱི་འབྱིན་རྔུབ་ལྟ་བུའི་རླབས་སྒྲ་དང་མཉམ་དུ། ཐེངས་འདིའི་ང་ཚོའི་མཚོ་ཐོག་གི་འགྲུལ་བཞུད་མཇུག་བསྒྲིལ་སོང་ཡང་། ངས་བསྐྱར་དུ་མཚོ་ལ་ཚེ་སྲོག་ཡོད་པའི་བདེན་ཁ་རྙེད་སོང་ལ། མཚོ་མོར་ཚེ་སྲོག་ཡོད་རྐྱེན། ནང་དུ་འཚོ་བའི་སྲོག་ཆགས་གྲངས་མེད་ཅིག་དང་། མཐའ་རུ་བསྐོར་བའི་མི་གྲངས་རྫོགས་མེད་ཅིག་གསོ་ཐུབ་པ་བྱུང་བ་རེད་སྙམ་བྱུང་། ༦ ༨ ནམ་སྟོད་ལ་བྲིས། མཆན། ①རྒྱུད་མངས་མཚོ་མོ། 琵琶湖 ཟེར་བའི་ཉི་ཧོང་གི་སྐམ་སའི་ནང་རོལ་དུ་ཡོད་པའི་མཚོ་ཞིག

(མཆོད་མེའི་པར་དབང་གཞན་གྱིས་དགོལ་མི་རུང)

རྩོམ་ཁུངས། ངེད་དྲ་བ།
