ལྷ་བྱམས་རྒྱལ་གྱི་རྩོམ་གསར་བ། ངས་འབོར་བཞིན་པའི་ཀློག་པ་པོ།
ལྷ་བྱམས་རྒྱལ།


པེ་ཅིང་དུ། ང་རྣམ་ཀུན་དུས་ཚོད་མི་འདུག་ཁོམ་སྐབས་མི་འདུག་ཟེར་མཁན་གྱི་མི་འབའ་ཞིག་ལ་འཕྲད་ཡོང་། པེ་ཅིང་གི་ལམ་ནང་དུ་ཡང་མི་རྣམས་གཅིག་རྗེས་གཅིག་གིས་བཅད་དེ་མགྱོགས་པོར་འགྲོ། རླངས་འཁོར་རྣམས་ཀྱིས་ཀྱང་དུང་སྒྲ་བར་མེད་དུ་བསྒྲགས་ཏེ་མདུན་ལམ་འགོག་མཁན་ཡོད་དོ་ཅོག་ལ་བཟོད་བསྲན་མི་ཚུགས་པའི་རྣམ་འགྱུར་སྟོན་པ་དང་། ང་ཁོར་ཡུག་དེ་ནས་མགྱོགས་ཚད་ཅིག་གིས་མགོ་ཡུ་འཁོར་བར་བྱས་བྱུང་། དུས་ཚོད་འདི་ལྟར་གཡང་དུ་མཆོངས་བ་ལྟར་མགྱོགས་དུས། ངས་རང་ཉིད་ཀྱིས་མང་བོ་ཞིག་འབོར་བཞིན་ཡོད་པ་ཚོར་བྱུང་སྟེ། ལང་ཚོ་དང་། སྤོབས་པ། ཕུགས་འདུན། དེ་མིན་ངོ་ཤེས་གསར་བ་ནས་རྙིང་བ་དང་། རྙིང་བ་ནས་རྒྱུས་མེད་དུ་འགྱུར་བཞིན་པའི་མི་དེ་དག འདི་ཐམས་ཅད་དུས་ཚོད་ཀྱིས་སྟོར་ཏེ་འཛིན་ཐབས་མ་བྱུང་ཞིང་། ཅི་ཞིག་བོར་སྟོར་བྱུང་བའི་ཕངས་སེམས་ནི་འཚོ་བའི་ནང་བློ་བདེ་བག་ཕེབས་མི་ཡོང་བའི་རྒྱུ་རུ་གྱུར་ཡོད། ཡིན་ན་ཡང་འདི་ཡོད་ཚད་ལ་ང་ཆུང་ཕྲུ་གུའི་དུས་ནས་ཡིད་ལ་གཤགས་པའི་གནས་ལུགས་ཤིག་གིས་སེམས་གསོ་བྱ་ཐུབ་སྟེ། དེ་ནི་ཆོས་ཐམས་ཅད་མི་རྟག་པའོ་ཟེར་བ་དེ་ཡིན་དེ། དེ་དེ་བཞིན་ཡིན་པའི་གནས་ལུགས་དེས་མ་མཐའ་ཡིན་ཡང་ངས་དུས་ཚོད་ཀྱི་འགྲོས་མགྱོགས་མོའི་ཁྲོད་ནས་དུས་དང་རྣམ་པ་ཀུན་ཏུ་འཁང་ར་དང་རེ་ཆད་ཁོ་ན་བྱེད་མི་སྲིད། དེ་ལྟར་ལགས་མོད། ང་ལ་མཚོན་ན་ཆོས་ཐམས་ཅད་མི་རྟག་པའོ་ཟེར་བའི་གནས་ལུགས་དེས་ཀྱང་ཡིད་བདེ་ལ་ཕབ་མི་ཐུབ་པ་ནི་ངས་འབོར་བཞིན་པའི་ཀློག་པ་པོ་དག་ཡིན་ཏེ། ཁོ་ཚོ་དུས་ཚོད་ཀྱི་ཀློང་ནས་བོར་སྟོར་བྱུང་བ་ནི་ང་ལ་མཚོན་ན་ཆེས་ཕངས་སེམས་ཆེ་བ་ཞིག་ཏུ་གྱུར་ཡོད། ངས་ད་ལྟའི་བར་དུ་སྒྲུང་གཏམ་མང་པོ་ཞིག་བྲིས་མྱོང་། སྒྲུང་གཏམ་འདི་དག་ནི་ངའི་མི་ལོ་བཅུ་ཕྲག་གི་དུས་ཚོད་རླུང་ལ་བསྐུར་མ་མྱོང་བའི་དན་རྟགས་ཙམ་ཡིན་ཞིང་། ཕྱི་ཤུན་ལྟ་བུའི་རྒྱུ་མཚན་དེ་དག་ཕར་བཤུས་ཚེ་ཤུལ་དུ་ལྷག་པ་ཡང་དོན་སྙིང་འདི་ཁོ་ན་ལས་མེད་པར་སྣང་སྟེ། བཤད་ཤེས་ཀྱིས་སྤྲོས་བརྗོད་ཅི་འདྲ་བྱས་ན་ཡང་བབ་ཆགས་པའི་བདེན་པ་ཞིག་ལྷག་ཡོད་པ་ནི་སྒྲུང་གཏམ་དེ་དག་ང་རང་སྒེར་གྱི་ངོས་ནས་བལྟས་ན་འདས་ཟིན་པའི་དུས་ཚོད་རླུང་ལ་མ་བསྐུར་བའི་དན་རྟགས་ཙམ་ལས་མ་འདས། ང་ཚོ་མི་སུ་ཡིན་རུང་དུས་ཚོད་ཟེར་བའི་གཟུགས་སྒྲ་དྲི་རོ་རེག་བྱས་སྟོང་བའི་བཞུར་འགྲོ་བ་འདི་ལ་ཐབས་ཤེས་རྩ་བ་ནས་མི་འདུག སྔོན་ཆད་དེ་འདྲ་ཡིན་པར་མ་ཟད། ད་ལྟ་ཡང་དེ་འདྲ་ཡིན་ལ། མ་འོངས་པར་ཡང་དེ་འདྲ་ཡིན། གང་ལྟར་ཀྱང་ཡུལ་དུས་ཟེར་བ་ནིི་ཚེ་སྲོག་ལ་ཕྱེས་པའི་བར་སྟོང་དང་གོ་སྐབས་ཤིག་ཡིན་ན། ང་ཚོས་ཡུལ་ཁམས་གྱི་བར་སྟོང་གི་བར་ཐག་ཐབས་བརྒྱ་ཇུས་སྟོང་གིས་ཇེ་ཉེར་བཏང་ཆོག་ཀྱང་། ཚེ་སྲོག་གི་རྒྱུན་བསྲིངས་བའི་གོ་སྐབས་ཕྲ་མོ་འདི་མི་འགྱུར་བརྟན་པར་གནས་ཐུབ་པར་ཐབས་ཤེས་རྩ་ནས་ཡོད་པ་མ་རེད། མཚམས་ཤིག་ནས་བཟུང་། ངས་འགྲོ་བ་མིའི་རིགས་ཀྱི་རྨོངས་འཐོམ་དང་ན་ཟུག་ཡོད་ཚད་དུས་ཚོད་ཟེར་བའི་བཞུར་འགྲོ་བ་འདི་ལ་ཁ་གཏད་པ་ལས་བྱུང་བའི་གནས་ལུགས་དེ་ངང་གིས་རྟོགས་ཡོང་བ་དང་། དེ་ནས་བཟུང་། ང་དུས་ཚོད་ལ་རྫི་གསིང་རབ་ཏུ་རྣོ་བའི་མི་ཞིག་ཏུ་གྱུར་ཅིང་། ཁ་ནས་དུས་ཚོད་དུས་ཚོད་ཟེར་མཁན་གྱི་མི་ཞིག་ཏུ་གྱུར། ངས་པེ་ཅིང་གི་མི་དེ་དག་གི་ངག་ནས་ང་ལ་དུས་ཚོད་མི་འདུག་ཁོམ་སྐབས་མི་འདུག་ཟེར་བ་ཐོས་དུས། ཁོ་ཚོས་དུས་ཚོད་སྐར་ཆ་གཅིག་ཡིན་རུང་བེད་སྤྱོད་གང་ལེགས་བྱས་ཏེ་རང་རང་གི་དམིགས་བྱའི་དོན་ཅི་ཞིག་སྒྲུབ་བཞིན་ཡོད་པ་རེད་སྙམ། དེ་ལྟར་སྙམ་དུས་རང་ཉིད་ཀྱི་འཚོ་བ་དྲན་ཡོང་། འདས་ཟིན་པའི་ལོ་བཅུ་ཕྲག་ཙམ་གྱི་རིང་ལ། ངས་སྒྲུང་གཏམ་མང་པོ་ཞིག་བྲིས་མྱོང་། དེ་དུས་ངས་ངག་ནས་དུས་ཚོད་མི་འདུག་དུས་ཚོད་མི་འདུག་ཅེས་གཞན་ལ་བཤད་མྱོང་བ་མི་དྲན་མོད། འོན་ཀྱང་ད་ལྟ་བསམས་ན་དེ་དུས་དུས་ཚོད་ཀྱི་བཞུར་རྒྱུན་མང་པོ་ཞིག་གི་ཁྲོད་དུ་ངས་སྒྲུང་གཏམ་འབྲི་བཞིན་ཡོད་ལས་ཆེ་སྟེ། དེ་མིན་ཚེ་ད་ལྟའི་བར་དུ་ངས་ཡིག་འབྲུ་འདི་འདྲ་མང་པོ་ཞིག་བསགས་ག་ལ་ཐུབ་ཨང་། རྒྱུ་མཚན་འདི་འདྲ་ཞིག་ལ་གཞི་བཅོལ་ནས་བཤད་ན། དུས་ཚོད་ཟེར་བའི་བཞུར་འགྲོ་བ་འདི་ལ་ཐབས་ཤེས་རྩ་བ་མེད་ཀྱང་། ངས་རང་ཚོད་ཀྱིས་དེ་འཛིན་པར་ཐབས་ཤེས་ཤིག་བཀོལ་མྱོང་བ་དང་། དེ་ནི་སྒྲུང་གཏམ་འབྲི་བ་དེ་རེད། ཞིབ་ཏུ་བསམས་ན། དུས་ཚོད་ཀྱི་ཆེད་དུ་དུས་ཚོད་འཛིན་པར་བྱེད་པ་ལ་དགེ་མཚན་ཅི་ཡང་མེད་པར་མཐོང་སྟེ། དུས་ཚོད་སྐར་མ་ཁ་ཤས་ཙམ་ཟིན་པ་ལ་ཡང་དོན་སྙིང་ཟེར་བ་ཞིག་ཐོན་པར་བྱེད་པ་དེ་ནི་གནད་དམ་པ་དེ་ཡིན་ངེས། ད་ལྟའི་བར་དུ་ངས་མྱངས་བའི་སློབ་གསོ་དང་མི་ཚེའི་ཉམས་མྱོང་ལྟར་ན། དུས་ཚོད་ལ་དོན་སྙིང་ཡོད་པར་བྱེད་པ་ནི་དུས་ཚོད་བདེ་སྤྱད་དེ་རང་ཉིད་མ་ཡིན་པའི་གཞན་ཟེར་བ་ཞིག་ལ་ཕན་བདེ་བསྐྲུན་པར་བྱེད་པ་དེ་ཡིན། གནད་འདི་ལ་ང་ལ་ཡིད་ཆེས་བརྟན་པོ་སྐྱེས་ཡོད་ཅིང་། ཕལ་ཆེར་ཚེ་ཐོག་འདི་ནས་གནས་ལུགས་འདི་བདེན་པར་སྒོམ་ནས་འགྱུར་བ་གཏོང་མི་སྲིད་པ་འདྲ། གནས་ལུགས་འདི་སེམས་ལ་སིམ་པ་ནས་བཟུང་། ངས་རང་གིས་སྒྲུང་གཏམ་འབྲི་བ་དེ་རང་གིས་དུས་ཚོད་རླུང་ལ་མ་བསྐུར་བའི་དན་རྟགས་ཙམ་ཁོ་ན་ཡིན་པར་མི་འདོད་དེ། སྒྲུང་གཏམ་འབྲི་བ་དེ་སྒྲུང་གཏམ་བཀླགས་མཁན་དེ་དག་གི་ཆེད་དུ་ཡིན་པ་དང་། ངའི་སྒྲུང་གཏམ་གྱིས་མ་མཐའ་ཡིན་ཡང་གནས་སྐབས་ཤིག་གི་རིང་ལ་ཀློག་མཁན་གང་ཞིག་གི་སེམས་ལ་བག་ཆགས་བཞག་ཡོད་པར་ཡིད་ཆེས་སྐྱེས་བྱུང་། དེ་བས། འདས་ཟིན་པའི་དུས་ཚོད་ནང་དུ། ང་བཞུར་འགྲོ་བའི་དུས་ཚོད་ཀྱི་གཙང་བོའི་འགྲམ་ནས་བསྡད་དེ། དོན་སྙིང་ཟེར་བའི་གསེར་ཉ་ལྟ་བུ་ཞིག་བཟུང་ཡོད་པར་འདོད་ཅིང་། དེ་ནི་ཆོས་ཐམས་ཅད་མི་རྟག་པའོ་ཟེར་བའི་སེམས་གསོ་དེ་འདྲ་ཞིག་མ་ཡིན་པར། དུས་ཚོད་ལ་དམ་འཛིན་བྱས་ཤིང་དམ་འཛིན་ལས་དོན་སྙིང་རྙེད་པའི་སེམས་གསོ་ཡག་པོ་ཞིག་ཡིན། སྔོན་ཆད། བོད་ཡིག་ཀློག་མཁན་མང་བོ་ཞིག་གིས་ང་ནི་ཁྱེད་ཀྱི་ཀློག་པ་པོ་ཡིན་ཟེར། ཁ་ཤས་ཤིག་གིས་ཀུ་རེའི་ཚུལ་དུ་ང་ནི་ཁྱེད་ཀྱི་སྒྲུང་གཏམ་བཀླགས་ནས་འཚར་ལོངས་བྱུང་བ་ཡིན་ཟེར། གང་ལྟར་ཀྱང་སྣང་བ་ཞིག་ལ་སྔོན་ཆད་ངས་བྲིས་བ་ཐམས་ཅད་ལ་ཀློག་པ་པོ་ཟེར་བ་ཞིག་ཡོད་པ་འདྲ་ལ། དེར་མ་ཟད་གྲངས་ཀ་ཡང་མང་བོ་ཞིག་ཡོད་པ་འདྲ། ཡིན་ན་ཡང་སྔར་བཞིན་དུས་ཚོད་གླེང་དགོས་བྱུང་སྟེ། དུས་ཚོད་ཟེར་བ་འདི་མི་སྡོད་མདུན་དུ་འགྲོ་བའི་འགྲོས་དང་བསྟུན་ནས། སྔོན་ཆད་སྐད་ཆ་དེ་རིགས་ལབ་མཁན་དག་གིས་ད་ལྟ་ང་ལ་དེ་ལྟར་བཤད་མི་ཡོང་། ཁོ་ཚོ་ང་དང་འཕྲད་ཡོང་དུས་སྒྲུང་གཏམ་དང ་འབྲེལ་བ་མེད་པའི་སྐད་ཆ་ལབ་ཡོང་། ངས་མི་དེ་རིགས་ལ་ཞིབ་ཏུ་བརྟགས་དུས། གཞི་ནས་དུས་ཚོད་ཀྱིས་ཁོ་ཚོའི་ས་ནས་འཕྲོག་སོང་བའི་སྟོང་ཆ་མང་པོ་མཐོང་བྱུང་སྟེ། དེ་དག་ལས་ཆེས་མངོན་པར་གསལ་བ་ནི་ན་ཚོད་སྟེ། དང་ཐོག་གི་གཞོན་ནུའམ་གཞོན་ནུ་མ་དེ་ད་ནི་དར་མར་གྱུར་འདུག་པ་དང་། དང་ཐོག་གི་གཡའ་དག་པའི་འཛུམ་རིས་ཅན་དེ་ད་ནི་ཐོད་པར་གཉེར་མདུད་ཅན་དུ་གྱུར་འདུག དེ་ལས་ཀྱང་། དུས་ཚོད་ཀྱིས་ཁོ་ཚོའི་ཟུར་ཁ་དེ་དག་མཉེན་འཇམ་དུ་བརྡར་འདུག དེ་ལྟ་བུའི་ངའི་སྔོན་ཆད་ཀྱི་ཀློག་པ་པོ་དག་ལོ་ངོ་གང་མང་གི་རྗེས་ནས་ང་དང་འཕྲད་དུས། ངས་གཞི་ནས་དུས་ཚོད་ཀྱི་མགྱོགས་ཚད་ཁྲོད་ངས་ཀློག་པ་པོ་སྐོར་ཅིག་འབོར་བཞིན་པ་ཚོར་བྱུང་། སྤྱིར་མི་ཞིག་གིས་འཚོ་བའི་ཁྲོད་ཀློག་འདོན་གྱི་གོམས་སྲོལ་བསྐྱངས་མིན་དེ་ནི་མི་སྒེར་གྱི་དོན་ཞིག་ཏུ་མཐོང་སྲིད་ཀྱང་། ངས་མཐའ་གཅིག་ཏུ་དེ་ནི་མི་ཞིག་ལ་མཚོན་ན་མེད་དུ་མི་རུང་བའི་གོམས་སྲོལ་ཅིག་ཏུ་འདོད་བཞིན་ཡོད། དཔེ་བཀླགས་ལ་དགའ་བ་དང་དཔེ་ཆའི་ནང་གི་མཐོང་ཐོས་མྱོང་གསུམ་སྦྲང་མས་མེ་ཏོག་གི་རྩི་བཅུད་གཞིབ་པ་ལྟར་རང་རྒྱུད་ལ་འགོས་པར་བྱེད་པ་ནི་མཛེས་སྡུག་ཡིད་དུ་འོང་བའི་ལས་ཀ་ཞིག་ཡིན་པར་ངེས་ཏེ། མི་གཅིག་གི་མི་ཚེ་ཟེར་བ་འདི་ཇི་ལྟར་རིང་རུང་མགོ་མཇུག་གི་ཚད་བཀག་ཅན་ཞིག་ཡིན་པས་དེ་རང་འདོད་དང་བསྟུན་ནས་རིང་དུ་བསྟུངས་ཐབས་བྲལ་ན་ཡང་། དེའི་ཞེང་ཚད་ཆེ་རུ་བསྐྱེད་ཐུབ་པ་ནི་ངེས་གཏན་ཡིན་པས། དཔེ་ཀློག་གིས་དེ་ལྟ་བུའི་ཞེང་ཚད་བསྐྱེད་པར་རམ་འདེགས་ཀྱི་ནུས་པ་ཐོན་ཐུབ་པ་ཡང་གདོན་མི་ཟ། དེར་བརྟེན། མི་རང་སྒེར་ཡིན་ན་ཡང་འདྲ། མི་རིགས་ཡིན་ན་ཡང་འདྲ། རྒྱལ་ཁབ་ཡིན་ན་ཡང་འདྲ། དཔེ་ཀློག་གི་གོམས་སྲོལ་ཡོད་པ་ཅན་དེ་ནི་རིག་གནས་དར་ཡོད་པའི་རྟགས་མཚན་ཞིག་ཏུ་རྩིས་ཆོག གནད་འདི་འདྲ་ཞིག་གི་ངོས་ནས་བལྟས་ན་རང་ཅག་ཁ་བ་རི་བའི་མི་རྣམས་ཀྱིས་དུས་རྒྱུན་གྱི་ཟང་ཟིང་གི་འཚོ་བའི་བར་གསེང་ནས་ལྷོད་འཇགས་ཀྱི་གོ་སྐབས་ཤིག་སྤྱད་དེ་རང་འགུལ་གྱིས་དཔེ་ཆ་ལག་ཏུ་ལེན་མཁན་ཉུང་ཚོད་དུ་གདའ་བ་ང་ཚོ་རང་ཉིད་ཀྱིས་སོ་སོར་རང་རྟོག་ཅིག་བྱོས་དང་གསལ་ལེར་ཤེས་ཡོང་བ་རེད། དུས་རབས་ཀྱི་རྒྱབ་ལྗོངས་སམ་གནས་བབ་འདི་འདྲ་ཞིག་གི་འོག་ནས་ངས་ཀློག་པ་པོ་འབོར་བཞིན་ཡོད་པར་ཡང་རྒྱུ་མཚན་མེད་པ་ཞིག་ཅི་ལ་ཡིན་ཨང་། དེ་བས་ཀྱང་ཞིབ་ཏུ་དཔྱད་ན། རྩོམ་རིག་བརྩམས་ཆོས་དང་ལྷག་པར་དུ་སྒྲུང་གཏམ་ལ་མཚོན་ན། ངེད་ཅག་ལ་ཐད་ཀར་དུ་སྤྱི་ཚོགས་སྤྱི་ཚོགས་འཚོ་བའི་ནང་འཚོ་གནས་ཀྱི་གནད་དོན་ཐག་གཅོད་པར་བྱེད་པར་རོགས་འདེགས་ཐོན་མི་ཐུབ་ཅིང་། དེང་དུས་ཀྱི་ཚན་རྩལ་དང་དངོས་པའི་ལག་རྩལ་གྱི་ཐབས་ལམ་མཛུབ་སྟོན་བྱེད་པ་སོགས་ཀྱི་འཇོན་ནུས་ཀྱང་ཅི་ཡང་མི་ལྡན་པས། དེ་ལ་ཉེ་བར་བཅར་མཁན་ད་ནི་སུ་ཞིག་ཡོད་ཨང་། མཚམས་རེར་སྒྲུང་གཏམ་རྩོམ་པ་པོ་ཞིག་གི་ངོས་ནས་ངས་ཀྱང་རང་རྟོག་ཡང་ཡང་བྱེད་བཞིན་ཡོད། ཅིའི་ཕྱིར་ཡར་སྔོན་གྱི་ཀློག་པ་པོ་དེ་དག་གིས་དུས་ཚོད་ཀྱི་བཞུར་རྒྱུན་ཁྲོད་ནས་སྒྲུང་གཏམ་ཀློག་འདོན་གྱི་གོམས་སྲོལ་དེ་རླུང་ལ་བསྐུར་བ་ཡིན་ནམ། ཞིབ་ཏུ་གཞིགས་ན། སྒྲུང་གཏམ་ཀློག་འདོན་བྱ་བ་ནི་མི་ཞིག་གི་སེམས་ཁམས་དང་འཚོ་བའི་རྣམ་འགྱུར་འཛིན་སྟངས་ཤིག་ལ་འབྲེལ་བ་དམ་ཟབ་ཡོད་པ་འདྲ། སྤྱིར་རྩོམ་རིག་བརྩམས་ཆོས་དང་ལྷག་པར་དུ་སྒྲུང་གཏམ་བཀླགས་པར་དགའ་མཁན་དེ་དག་ནི་མི་ཚེ་འདིའི་ཚད་ངེས་ཅན་གྱི་ཡུན་ཚད་བསྲིངས་བའི་གོ་རིམ་ཁྲོད་ནས་འཚོ་བ་ལ་ནང་སེམས་ཀྱིས་ཚོར་སྣང་བྱེད་པར་དང་དོད་ནན་མོ་བྱེད་མཁན་འབའ་ཞིག་ཡིན་སྲིད་དེ། མི་དེ་རིགས་ཀྱིས་ཕྱིའི་དངོས་བོའི་ལོངས་སྤྱོད་ལས་ནང་སེམས་ཀྱི་ལོངས་སྤྱོད་ལ་དོ་གལ་བྱ་བཞིན་ཡོད་ངེས། ཁོ་ཚོར་མཚོན་ན་རྙོག་ཟིང་གི་སྤྱི་ཚོགས་འཚོ་བ་ཁྲོད་ནས་མཚམས་རེར་ཟུར་དུ་བཀྱེད་སྒྲུང་གཏམ་གྱི་འཇིག་རྟེན་ལ་མྱུལ་བ་དང་། དེ་ནས་སྒྲུང་གཏམ་ནང་གི་འཇིག་རྟེན་དེས་བསྐྱེད་པའི་སེམས་ཚོར་དང་སད་རྟོགས་ཀྱི་ཚོར་བ་དེ་དག་ལ་བརྟེན་ནས་མངོན་སུམ་འཇིག་རྟེན་གྱི་གནད་དོན་དེ་དག་ལ་ཐག་གཅོད་བྱེད་པའི་རྣམ་འགྱུར་ཅིག་བྱུང་བཞིན་ཡོད་ངེས། ཚུལ་འདི་ལས་ཤེས་ཐུབ་པ་ནི། སྒྲུང་གཏམ་ཟེར་བ་དེ་མཐར་གཏུགས་ན་གཏམ་རྒྱུད་ཀྱི་བྱུང་རིམ་ཙམ་མ་ཡིན་པར་གཏམ་རྒྱུད་ལ་བརྟེན་ནས་འཇིག་རྟེན་འཚོ་བའི་ནང་གི་སད་རྟོགས་ཤིག་བརྒྱུད་སྒྲོག་བྱེད་པ་དེ་ཡིན་ངེས། གང་ལྟར་ཀྱང་། སྒྲུང་གཏམ་བཀླགས་པར་དགའ་བ་ནི་འཚོ་བའི་རྣམ་འགྱུར་འཛིན་སྟངས་ཤིག་ལ་རག་ལས་ཏེ། གཞོན་སོ་དཀར་གྱི་དུས་སུ་ང་ཚོར་འཚོ་བའི་ཉམས་མྱོང་མང་པོ་མེད་པས་སྒྲུང་གཏམ་ནང་གི་འཇིག་རྟེན་དང་འཚོ་བ་དེ་ལ་ཡིད་ཤོར་སླ་བ་ཞིང་། རིམ་གྱིས་ལོ་རྒས་བར་གྱུར་ནས་འཚོ་བའི་རྙོག་ཟིང་སྣ་མང་བརྒྱུད་དུས། མ་གཞིར་རང་ཉིད་སྒྲུང་གཏམ་ཞིག་ཏུ་གྱུར་ཡོད་པ་ལྟ་བུ་ཡིན་པས། ད་ནི་སྒྲུང་གཏམ་ནང་གི་ཞིང་ཁམས་དེ་ལ་གཡེང་འདོད་ཀྱི་སྤྲོ་སྣང་མེད་པར་ཆགས་པ་ཡང་ཆོས་ཉིད་ཅིག་ཡིན་པ་འདྲ། མཚམས་འདི་ནས་དུས་ཚོད་ཟེར་བ་འདི་སླར་ཡང་གླེང་བར་འདོད་དེ། ཀློག་པ་པོ་ནས་ཀློག་པ་པོ་མ་ཡིན་པར་གྱུར་བ་འདི་ཡང་དུས་ཚོད་ཀྱི་ཆོ་འཕྲུལ་ཙམ་ལས་མ་འདས། འགྲོས་མགྱོགས་པའི་དུས་ཚོད་ཀྱི་བང་ཁ་ནས་ང་ཚོ་ཐམས་ཅད་རྒས་སླ་བའི་མི་རབས་ཤིག་ཏུ་གྱུར་ཅིང་། གཡའ་དག་པ་དང་། སྤྲོ་སིང་ངེའི་སེམས་ཁམས་ཡོད་ཚད་ཅི་ཞིག་གི་རྗེས་སུ་ལུས་དོགས་ཀྱི་ཧུར་བརྩོན་དང་སེམས་ནད་ཀྱིས་ཤུལ་མེད་དུ་བཟོས་སོང་། དེ་བས་ན་ཡར་སྔོན་གྱི་སློབ་རྭ་ནས་རང་རང་གི་རྨི་ལམ་ཉིན་ཐོར་ཕབ་བཞིན་པའི་སློབ་འབྲིང་སློབ་མ་དེ་ཚོ། སྤོབས་སེམས་དང་ཡིད་ཆེས་ཀྱིས་ཁེངས་བའི་སློབ་ཆེན་པ་དེ་དག ད་ནི་དུས་ཚོད་ཀྱི་འགྲོས་དང་བསྟུན་ནས་ནར་སོན་ཅིང་སྨིན་པར་གྱུར་བ་དང་། འཚོ་བའི་གྲང་རེག་གིས་འཁྱག་འདར་ཡང་ཡང་ཐེབས་པ་བརྒྱུད་ནས་དུས་ད་ལྟར་ལག་ན་དཔེ་ཆ་མེད། སེམས་ན་ཕུགས་སེམས་མེད། ངོ་ན་འཛུམ་རིས་མེད། དེ་ལས་ལྡོག་སྟེ་ཉིན་མཚན་ཀུན་ཏུ་སེམས་ན་སྒོར་མོ་དང་། ངོ་སོ། ཐོབ་དང་གོ་ས། དབང་དང་འབྱོར་བ་ཅན་འབའ་ཞིག་ཏུ་གྱུར་ཡོད་དུས། ངས་ཡར་སྔོན་གྱི་སྙིང་ཉེ་བའི་ཀློག་པ་པོ་དེ་དག་འབོར་བཞིན་པའི་དོན་དངོས་འདི་ལ་ཡང་སྐྱོ་གར་གྱི་ཁ་དོག་གིས་བཀླུབས་ཡོད་ངེས། ༢༠༡༥ལོའི་སྤྱི་ཟླ་༥བའི་ཚེས་༢༩ཉིན་པེ་ཅིང་ནས།
(མཆོད་མེའི་པར་དབང་གཞན་གྱིས་དགོལ་མི་རུང)


རྩོམ་ཁུངས། ངེད་དྲ་བ།
