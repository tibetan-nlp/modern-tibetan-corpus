ཁྲུང་ཁྲུང་དཀར་མོ།(ཞར་བཀོད་གཅིག་ཡོད)
ཁུ་བྱུག་སྔོན་མོ།(རྨ་ལྷོ)


ཁྲུང་ཁྲུང་ཡིན་ན་བསམས་བྱུང།། འཕུར་རྒྱུ་བྱུང་ན་བསམས་བྱུང་།། བརྩེ་བ་ཟབ་ན་བསམས་བྱུང་།། བར་ཆད་མེད་ན་བསམས་བྱུང་།། ___སྔོན་འགྲོའི་རོལ་དབྱངས།
ཁྲུང་ཁྲུང་དཀར་མོ་ནི་གཡུང་ཞིང་སྲུན་ལ་གཤིས་ཀ་འཇམ་པའི་བྱ་ཞིག་སྟེ། ལུས་ཀྱི་སྤུ་མདོག་དུང་ལྟར་དཀར་ཞིང་འོད་འཁྱུག་པ། མགྲིན་པ་དང་རྐང་པ་རིང་ཞིང་། གཟུགས་ཆེ་ཞིང་མཆུ་ལེབ་ཅིང་རིང་བས། བྱ་རིགས་གཞན་ལས་གཟུགས་གཞི་དོ་མཉམ་ཞིང་སྙིང་རྗེ་བོ་ཞིག་ཡིན། ངག་རྒྱུན་ལྟར་ན། ཁྲུང་ཁྲུང་དཀར་མོའི་མིང་ལ་བྱ་ཚེ་རིང་ཞེས་ཀྱང་འབོད་ལ། ཚེ་རིང་རྣམ་དྲུག་གི་ཡ་གྱལ་ཡིན་པར་སྨྲ། འཚོ་སྡོད་ཀྱི་གནས་མལ་ནི་གཅིག་ཏུ་མ་ངེས་ཏེ། དབྱར་དུས་སུ་བྱང་ཕྱོགས་ཀྱི་མཚོ་འགྲམ་སོགས་དང་དགུན་ཁར་ལྷོ་ཕྱོགས་སུ་ནམ་ཟླ་སྐྱེལ་བཞིན་ཡོད་པས། དུས་བྱའི་རིགས་སུ་གཏོགས་པ་དང་། ནམ་མཁར་འཕུར་དུས་མང་པོ་མཉམ་དུ་བསྟར་ཕྲེང་བསྒྲིགས་ཏེ་སྐད་སྙན་སྒྲོག་ཞོར་འཕུར་བཞིན་ཡོད། ཆུང་དུས་སུ། ངས་སྟོན་མཇུག་གི་དུས་ཚིགས་ཤིག་ལ་ཁྲུང་ཁྲུང་དཀར་མོ་བསྟར་ཕྲེང་བསྒྲིགས་ཏེ་ལྷོ་རུ་འཕུར་བ་མཐོང་རྗེས། 《ཁྲུང་ཁྲུང་ཡ། ཕྱིར་སླེབས་དུས་ངའི་དབྱར་གཞུང་ཁུར་ལ་ཤོག》ཅེས་པའི་སྙན་ངག་ཅིག་བྲིས་མྱོང་། ད་ལྟ། ཕྱིར་དྲན་ཞིག་བྱས་ཚེ་དེ་ནི་སེམས་ཀྱི་མེ་ལོང་ངོས་སུ་གསལ་ལེར་ཤར་ཡོང་། ཡིན་ནའང་། དེ་དུས་ལྷོ་ཕྱོགས་སུ་འཕུར་བའི་ཁྲུང་ཁྲུང་དཀར་མོ་བརྒྱུད་ནས་སེམས་ཀྱི་རེ་སྨོན་ཞིག་བརྗོད་ཡོད་པ་མ་གཏོགས་ཁྲུང་ཁྲུང་དཀར་མོའི་བརྩེ་བ་དང་གཤིས་ཀ་སོགས་ལ་ཆ་རྒྱུས་ཅི་ཡང་མེད། མི་སུ་ཞིག་ཡིན་ཡང་རྗེ་ཚངས་དབྱངས་པའི“བྱ་དེ་ཁྲུང་ཁྲུང་དཀར་མོ།། གཤོག་རྩལ་ང་ལ་གཡོར་དང་།། ཐག་རིང་རྒྱང་ལ་མི་འགྲོ།། ལི་ཐང་བསྐོར་ནས་སླེབས་ཡོང་།།”ཞེས་པའི་མགུར་ཚིག་དེ་ཆུང་དུས་ནས་ཐོས་ཡོད་པ་གདོན་མི་ཟ་སྟེ། དེས་ན་ཁྲུང་ཁྲུང་ཞེས་པའི་མིང་དེས་མི་སུ་ཞིག་ཡིན་ཡང་བག་ཆགས་མི་དམན་པ་ཞིག་བཞག་ཡོད་སྲིད་ལ། མི་སུ་ཞིག་ཡིན་ཡང་ཁྲུང་ཁྲུང་བཞིན་ནམ་མཁའ་ནས་འཕུར་འདོད་པའི་བློ་ཞིག་སྐྱེས་མྱོང་ཡོད་པའང་གདོན་མི་ཟའོ།། ང་ལ་མཚོན་ན། དེ་འདྲའི་བག་ཆགས་དང་འདོད་བློ་ནི་ཆུང་དུས་ནས་བཞག་ཅིང་སྐྱེས་ཡོད་པར་མ་ཟད། ཁྲུང་ཁྲུང་དཀར་མོ་ནི་ལྷ་དབང་རྒྱལ་བོའི་ལྷ་བྱ་དང་གེ་སར་རྒྱལ་བོའི་འཕྲིན་བྱ་ཡིན་རབས་སོགས་ཀྱང་རྣ་ཐོས་སུ་གྱུར་ཡོད་མོད། ཁྲུང་ཁྲུང་དཀར་མོ་ལ་སེམས་གཏིང་ནས་དགའ་བར་གྱུར་བའི་རྒྱུ་མཚན་ནི་དེ་དག་མ་ཡིན་ཏེ། གང་ཡིན་ཞེ་ན། དེ་ནི་ཁྲུང་ཁྲུང་དཀར་མོ་ལ་སྲོག་ཆགས་གཞན་གང་ལའང་མེད་པའི་བརྩེ་བ་ཡ་མཚན་ཞིག་ཡོད་པས་ཡིན། དེ་ལ་དངོས་གནས་རྣམ་པར་དག་པ། ཟོལ་དང་བྲལ་བ། འགྱུར་བ་མེད་པ་ཞེས་བརྗོད་ཀྱང་འབྱོར་ནས་མི་ཚད་ལྷག་ཡོད་ལ། བརྩེ་བ་ཞེས་པ་ལ་མཚན་གཞི་སྐྱོན་བྲལ་ཞིག་འཇོག་དགོས་ཚེ་ས་སྟེང་འདིར་ཁྲུང་ཁྲུང་དཀར་མོའི་བརྩེ་བ་ལས་འཇོག་རྒྱུས་ཕོངས་སོ་ཞེས་བདག་གིས་མགྲིན་པ་གཟེངས་སུ་བཏེགས་ནས་སྨྲ་བར་སྤོབས། ཁྲུང་ཁྲུང་དཀར་མོའི་བརྩེ་བ་ནི་དེ་འདྲ་ཞིག་ཡིན་པ་གང་ལས་རྟོགས་ཟེར་ན། སྔོན་གྱི་དུས་ཤིག་ལ་མི་ཞིག་གིས་སར་བབས་པའི་ཁྲུང་ཁྲུང་གི་ཁྱུ་ཚོགས་ཤིག་ཏུ་རྡོས་བརྒྱབ་སྟེ་ཁྲུང་ཁྲུང་ཕོ་ཞིག་བསད་པ་ན། ཁྲུང་ཁྲུང་མོ་ཞིག་ཁྲུང་ཁྲུང་ཕོ་དེའི་བེམ་པོ་དང་གཏན་ནས་ཁ་མི་འབྲལ་ཞིང་། ད་དུང་ལུས་བོ་ཁྲུང་ཁྲུང་ཕོ་དེར་བརླན་ནས་མིག་ནས་ཁྲག་ཡོང་རག་བར་ངུས་ངུས་ནས་མཐར་ཤི་བར་གྱུར་བའི་གནས་ཚུལ་ཞིག་གོ་མྱོང་མོད། དེ་དུས་ང་དེ་ལ་ཡིད་མ་ཆེས་པར་ལྷ་སྒྲུང་ལྟ་བུ་ཞིག་ཏུ་བརྩིས་པ་ཡིན། ཉེ་བའི་ལོ་འགའི་ཡར་སྔོན་གྱི་སྟོན་མཇུག་གི་དུས་ཤིག་ལ། ཁྲུང་ཁྲུང་དཀར་མོ་ཁྱུ་ཞིག་བསྟར་ཕྲེང་བསྒྲིགས་ནས་ལྷོ་རུ་འཕུར་སྐབས། སྟབས་མ་ལེགས་པར་ཁྲུང་ཁྲུང་མོ་ཞིག་གློག་སྐུད་ཅིག་ལ་གདོང་ཐུག་ནས་ཁྱིམ་ཚང་ཞིག་གི་ར་སྐོར་དུ་ལྷུང་བའི་ཚེ། ཁྱིམ་ཚང་དེའི་ཁྱིམ་མིས་ཁྱིམ་ནས་གསོ་བྱར་འཛིན་འདོད་ཀྱིས་ཁྲུང་ཁྲུང་མོ་དེའི་སྒྲོ་བྲེགས་ནས་འཕུར་རྩལ་ཉམས་པར་བྱས་པ་ན། ཁྲུང་ཁྲུང་ཕོ་བྱ་ཞིག་ཉིན་མཚན་འགར་ཁྱིམ་ཚང་དེའི་ར་ཁར་བབས་ཏེ་སྐད་བསྒྲགས་མོད། མཇུག་ཏུ་སར་ལྷུང་བའི་ཁྲུང་ཁྲུང་དེ་དང་མཉམ་དུ་བསྐྱོད་རྒྱུའི་ཐབས་ཟད་ནས་མི་འཕུར་ག་མེད་དུ་གྱུར། སང་ལོའི་དུས་དེར་ཡང་ཁྲུང་ཁྲུང་ཕོ་དེ་ཡང་བསྐྱར་ཁྱིམ་ཚང་དེའི་ར་ཁར་བབས་ནས་ཁྲུང་ཁྲུང་མོ་དེར་བལྟས་ཏེ་སྐད་སྒྲོག་བཞིན་ཡུན་རིང་ཞིག་ལ་བསྒུགས་མོད། སྒྲོ་བྲེགས་པའི་ཁྲུང་ཁྲུང་མོ་དེ་འཕུར་མ་ཐུབ་པས་རང་གཅིག་པུ་འཕུར་དགོས་པར་གྱུར། ཡང་རྗེས་མའི་ལོ་གསུམ་པའི་དུས་དེར་ཡང་ཁྲུང་ཁྲུང་ཕོ་དེ་ཁྱིམ་ཚང་དེའི་ར་ཁར་བབས་ནས་སྐད་སྒྲོག་བཞིན་རེ་སྒུག་བྱས་ཀྱང་ཁྲུང་ཁྲུང་མོ་དེ་འཕུར་མ་ཐུབ་པར་བརྟེན། མཐར་ར་སྐོར་དུ་བཅུག་པའི་ཁྲུང་ཁྲུང་མོ་དེའི་གམ་དུ་བབས་ནས། མཇིང་བ་ཕན་ཚུན་ལ་དཀྲིས་ནས་ཚེ་ཡི་འདུ་བྱེད་བཏང་བའི་གནས་ཚུལ་ཞིག་པར་དང་བཅས་ཏེ་མཐོང་བས། ང་རང་ཧང་སངས་པར་བྱས་ཤིང་སྔོན་ཆད་ཀྱི་གནས་ཚུལ་དེ་ཡང་བདེན་པ་ཡིན་པའི་ངེས་ཤེས་འདྲོངས་པར་གྱུར། ངེས་ཤེས་དེ་འདྲོངས་པར་གྱུར་རྗེས། ངས་ཏག་ཏག་དྲང་སྲོང་གྲོག་མཁར་གྱིས་རྔོན་པ་ཞིག་ལ་དམོད་བོར་བའི་ཚིག་ཅིག་དྲན་བྱུང་། དེ་ནི“རྔོན་པ་ཁྱོད་ནི་ནམ་དུ་ཡང་།། ཐར་བ་ཐོབ་པར་འགྱུར་མ་ཡིན།། དགའ་མགུར་རྩེན་པའི་འདབ་ལྡན་ཟུང་།། གཅིག་ནི་ཁྱོད་ཀྱིས་མངོན་པར་བསད།། ”ཅེས་པ་སྟེ། མ་གཞིར་དྲང་སྲོང་གྲོག་མཁར་ཞི་གནས་ལས་གྲོལ་སྐབས་ཐོག་མར་ནགས་ཚལ་དུ་འདབ་ལྡན་ཟུང་ཞིག་ཡོད་པའི་གཅིག་རྔོན་པ་ཞིག་གིས་མདས་བསད་པ་མཐོང་ནས་སྙིང་རྗེ་དང་ཁྲོ་བས་བསྐུལ་ཏེ་དམོད་དེ་འདྲ་བོར་བ་ནི་འདབ་ལྡན་དེ་གཉིས་ཀྱི་བརྩེ་བའི་སྡུག་བསྔལ་གྱིས་བསྐྱེད་པ་བསྙོན་དུ་མེད་དེ། དེ་ནས་བཟུང་། ངས་སྐྱོ་གདུང་བརྗོད་པའི་ཤོ་ལོ་ཀ་ཐོག་མ་དེར་གོ་བ་གཏིང་ཟབ་ཅིག་ལོན་བྱུང་སྙམ་པའི་ཚོར་བ་ཡང་ཡང་སྐྱེས། བསམ་ཡུལ་ལས་འདས་པ་ཞིག་ལ། ཉིན་འགའི་སྔོན་ལ་ཁྲུང་ཁྲུང་དཀར་མོ་ཁྱུ་ཞིག་ནམ་མཁར་འཕུར་དུས་གློ་བུར་དུ་བབས་པའི་སེར་རྡེའུ་ཞིག་གིས་ཁྲུང་ཁྲུང་ཞིག་གི་ལུས་པོ་རྨས་ཏེ་ངེད་ཚང་གི་ལྷས་སུ་ལྷུང་། དེ་འཕྲལ་ངས་ནི་ཁྲུང་ཁྲུང་དེའི་ལུས་པོར་སྨན་བཅོས་ཤིག་བྱས་རྗེས་བྱ་ཁང་ཆུང་ཆུང་ཞིག་བཟོས་ཏེ་དེའི་ནང་དུ་བཞག སྐབས་དེར་ཁྲུང་ཁྲུང་གཞན་ཞིག་གློ་བུར་དུ་འཕུར་ཡོང་ནས་སྐད་སྒྲོག་ཅིང་། དེ་ནས་ཉིན་གཉིས་གསུམ་ཞིག་ལ་ངེད་ཚང་གི་མགོ་ཐོག་གི་བར་སྣང་ནས་ལྡིང་བསྐོར་བྱས། དེའི་རྗེས་བྱ་ཁང་དེའི་ཉེ་སར་བབས་ནས་ཡར་རྒྱུག་མར་རྒྱུག་བྱེད་ཅིང་། མཐར་གང་ལའང་མི་འཛེམ་པར་བྱ་ཁང་དུ་འཛུལ་ཏེ་ཁྲུང་ཁྲུང་རྨས་མ་དེ་དང་མཉམ་དུ་བསྡད་བྱུང་། ཉིན་འགའི་རྗེས་ཏེ་ཁྲུང་ཁྲུང་རྨས་མ་དེ་དྲག་སྐྱེད་བྱུང་རྗེས་གཉིས་ཀ་ལྷན་དུ་གཤོག་འགྱུར་ལྡེམ་མེར་འཕུར་སོང་། དེ་ལྟར་འཕུར་དུས། ངས་སྔོན་ཆད་ཁྲུང་ཁྲུང་དཀར་མོ་ལ་བྲིས་པའི་སྙན་ངག་དེའི་ནང་གི“རེ་འདུན་ལྗང་ཁུ་ཞིག་གཤོག་རྩེར་བཏེགས་ནས་འཕུར་སོང་། དབྱར་གཞུང་དྲོན་མོ་ཞིག་སེམས་ལ་བསྒོམས་ནས་འཕུར་སོང”ཞེས་པའི་ཚིག་དེ་ཡིད་ལ་དྲན་བྱུང་ལ། དངོས་གནས་དེ་ལྟ་དེ་ལྟར་འཕུར་སོང་བའི་ཉམས་སྣང་ཞིག་བདག་ལ་བཅོས་མིན་དུ་བྱིན་སོང་། ང་དངོས་སུ་འཕྲད་པའི་ཁྲུང་ཁྲུང་དཀར་མོ་གཉིས་ཀྱི་དོན་རྐྱེན་དེས་ང་ལ་ལྷད་མེད་མཛེས་སྡུག་གི་བརྩེ་བའི་རྒྱལ་ཁམས་ཤིག་དངོས་སུ་མིག་མདུན་ནས་གསལ་ལེར་བསྟན་བྱུང་། དེ་ནི་འཇིག་རྟེན་མི་ཡུལ་ན་ལུས་པའི་ཆེས་རིན་ཐང་ཆེ་བའི་བརྩེ་བ་ཡི་རང་གཟུགས་རྣལ་མ་དེ་ཡིན་ལ། བརྩེ་བ་དེ་འདྲ་ཞིག་གི་ཚུགས་ཀ་ཙམ་ཡང་ངས་འགྲོ་བ་མིའི་འཚོ་བའི་ཁྲོད་ནས་གཏན་ནས་མཐོང་མ་མྱོང་། རེད་དེ། ཁྲུང་ཁྲུང་དཀར་མོའི་བརྩེ་བ་དང་བསྡུར་ན་རྒྱ་གར་གྱི་སློབ་དཔོན་དབྱུག་པ་ཅན་གྱིས་བྲིས་པའི“ཤི་བ་ཞེས་བྱ་ཕ་རོལ་ཏུ།། བདག་དང་འགྲོགས་ཕྱིར་བདག་འཆི་འདོད།། ཨ་བིནྟ་དེ་བདག་གིས་ནི།། སྐྱེ་འདི་ཉིད་ལ་ཇི་ལྟར་ཐོབ།།”ཅེས་པ་ནི་ཁ་བཤད་ཙམ་རེད་ལ། རང་རེའི་བོད་ཀྱི་མཛའ་གཞས་ཁྲོད་ཀྱི“ཆུང་ཆུང་ཆུང་ནས་འདྲིས་པའི།། ཆུང་འདྲིས་སྒོག་པ་དཀར་པོ།། ཕ་མས་སྒོག་ཤུན་བཤུས་ཀྱང་།། ང་གཉིས་འབྲལ་འདོད་མི་འདུག།”ཅེས་པའང་དེ་ལས་མ་འདས། ཚུལ་འཆོས་དང་བསླུ་བྲིད། རྒྱུ་ནོར་སོགས་ཀྱི་དྲི་མས་བསླད་པའི་ད་ལྟའི་སྤྱི་ཚོགས་ཀྱི་མཁའ་དབུགས་ཁྲོད། རང་རེའི་མི་དང་མིའི་བར་སྨོས་ཅི་དགོས། ཚེ་གཅིག་གི་བཟའ་ཟླའི་བར་དུའང་རྒྱུན་དུ་ཁ་འཁོན་དང་ལག་འཛིང་། ཐ་ན་ཕན་ཚུན་གསོད་རེས་བྱེད་པའང་མང་དུ་མཐོང་ཞིང་། ད་དུང་སྐྱ་ཡ་རང་གི་ཆེད་དུ་ཤི་ནའང་སེམས་འཁུར་ཉག་ཙམ་མེད་པར་དགའ་སྐྱིད་ལ་རོལ་བ། སྐྱ་ཡ་ནད་ཀྱིས་མནར་ཏེ་འཆི་ལ་ཉེ་ཡང་ཁ་ཡ་མི་བྱེད་པར་གཞན་ལ་འགྲོགས་པ་སོགས་ནི་ཇི་འདྲའི་ཞེ་མེར་ཞིང་སྐྱུག་བྲོ་བ་ལ་ཨང་། བདེན་མིན་ཅི་ཆ་ཡང་འཇིགས་ཡེར་བ་ཞིག་ལ་ཉེ་ལམ་གཏམ་འདི་འདྲ་ཞིག་ཀྱང་རྣ་ལམ་དུ་བརྒྱུད་བྱུང་སྟེ། བཟའ་ཟླ་ཞིག་གིས་རང་གི་སྐྱ་ཡ་བསད་རྗེས། བེམ་པོའི་ལུས་སྟེང་གི་པགས་པ་བཤུས་ཏེ། ཚིགས་མལ་རེ་རེ་ནས་སོ་སོར་ཕྲལ་རྗེས་རྡང་ལ་བཏགས་འདུག་སྐད། ཨ་ཙི། དེ་ལ་ཅི་བྱུང་བ་ཡིན་ནམ། འདྲེ་ཡིས་ཀྱང་དེ་ལྟར་བྱེད་མི་སྲིད། དེ་འདྲའི་མཐོང་ཐོས་ཀྱི་དབང་གིས་ད་ནི་བསམ་གྱིན་བསམ་གྱིན་ང་རང་མིའི་འཚོ་བ་ལ་ཡིད་གཏིང་ནས་ཆད་ཅིང་། རང་ཉིད་ཁྲུང་ཁྲུང་དཀར་མོ་ཞིག་ཡིན་ན་ཅི་མ་རུང་སྙམ་ཡོང་། 2017.7.17 དགུན་ཁ། མདང་དགོང་གི་རྨི་ལམ་དུ། དབྱར་གཞུང་གི་དུས་ཚིགས་ཤིག་ཐེངས་འགར་རྨིས་བྱུང་ནའང་། མངོན་སུམ་ནི་དེ་ལས་ལྡོག་སྟེ་ད་ནངས་གཉིད་ལས་སད་དུས། སྒེའུ་ཁུང་ལས་ཕྱི་རོལ་གྱི་ཁ་བ་མཐོང་བྱུང་། དེར་བརྟེན་ངས་ད་གཟོད་དགུན་ཁ་རང་གི་གན་དུ་སླེབས་པ་ཚོར། ང་ལ་མཚོན་ན། དགུན་ཁ་སླེབས་སྟངས་དེ་ནི་གློ་བུར་བ་ཞིག་རེད། ཅིའི་ཕྱིར་ཞེ་ན། ངའི་སེམས་པ་ནམ་རྒྱུན་སྟོན་ཁ་ནས་དབྱར་གཞུང་ལ་འགྲོ་བཞིན་ཡོད་པས་ཡིན། སྔོན་ཆད། ངས་ཁོ་མོར“ཁྱོད་རང་དཔྱིད་ཀ་ནས་མར་གོམ་པ་གང་སྤོས་དང་། ང་རང་སྟོན་ཁ་ནས་ཡར་གོམ་པ་གང་སྤོ་ཡོང་། འུ་གཉིས་དབྱར་གཞུང་ནས་འཕྲད”ཅེས་དུས་ཆད་བྱས་ཤིང་། དེ་ནི་སྙན་ངག་པ་ཚོས་ཚིག་གི་རྩེད་མཚར་དུ་བརྩི་བའི་སྙན་ངག་གི་རྒྱན་ལས་ཕྲ་མོའི་རྒྱན་ལྟ་བུ་ཞིག་མིན་ཏེ། དུས་ཆད་ཀྱི་དམ་བཅའ་ནི་ནང་སེམས་ཀྱི་འདུན་པ་ཐད་ཀར་དུ་བསྟན་ཡོད་པ་ལས་ཕྱི་ཡི་བརྡ་དང་རྣམ་འགྱུར་གྱིས་མཚོན་ཡོད་པ་མིན་ལ། དུས་ཆད་ཀྱི་ཡུན་ཡང་ཚེ་དང་མཉམ་པའི་ཡུན་རིང་གི་དུས་ཆད་ཅིག་ཡིན་པ་ལས་ཧ་ཅང་ཐུང་བའི་ཞག་གཅིག་གི་དུས་ཡུན་ཞིག་ཡེ་ནས་མིན། དུས་ཆད་དེ་ལྟར་ངེད་གཉིས་དབྱར་གཞུང་དུ་འཕྲད་པའི་ཚེ། ཏག་ཏག་དུས་ཚོད་ནི་མཚན་མོར་འཁེལ་ཞིང་རླུང་ཡང་ལྡང་བས། ང་ཁོ་མོར་གཏན་ནས་མཇལ་མ་ཐུབ། དེ་ནི་གོ་ནོར་ཞིག་ལ་འབྲེལ་བ་ཡོད་ལ། བར་ཆད་ཅིག་ཀྱང་ཡིན་ལས་ཆེ། དེ་ནས་བཟུང་། དབྱར་གཞུང་དེ་ནི་རྨི་ལམ་ཞིག་ཏུ་གྱུར་ལ། ཁོ་མོ་ཕྱིར་དཔྱིད་ཀར་ལོག་ཅིང་། ང་རང་ཕྱིར་སྟོན་ཁར་སྤྱུགས་པས། སེམས་ཀྱི་སྨྲེང་བ་ཡོད་ཚད་སྟོན་ལོའི་སྟེང་ལ་རིས་སུ་ཕབ། ཁྲུང་ཁྲུང་དཀར་མོ་ལ་དྲན་གདུང་གི་འཕྲིན་ཞིག་བསྐུར་ཡང་འཕྲིན་ལན་ཞིག་ཡེ་ནས་འབྱོར་རྒྱུ་མ་བྱུང་། སྟོན་མཇུག་གི་ཉམ་ངའི་ཁྲོད་ནས། ངས་ཟིལ་བ་དང་བཅས་ཏེ་བཞད་ཡོད་པའི་ཆེས་མཇུག་མཐའི་མེ་ཏོག་ཅིག་མཐོང་ཞིང་དེ་ཡིས་ཁོ་མོ་ངའི་གན་དུ་ཡོད་ཡོད་འདྲ་བའི་སྣང་བ་ཞིག་སྟེར་ནའང་། དེ་ཡི་འདབ་ལོ་སྔོན་པོའི་སྟེང་གི་ཟིལ་བ་ནི་གློ་རྡེག་ལ་ངའི་སེམས་ཀྱི་གཏིང་དུ་ལྷུང་ཞིང་། དེ་ནས་ཅང་མ་འགོར་བར་བལ་གྱིས་ཁྱེར་སོང་། སྐབས་དེའི་རང་གི་ཚོར་བ་ནི་སྙན་ངག་གི་འབྲི་དེབ་སྟེང་དུ་འདི་ལྟར་བྲིས་ཡོད་དེ། “དུས་ཚོད་འདི་ནི་སྟོན་ཟླ་ཐ་མ་སྟེ།། ལོ་མ་སེར་པོའི་ལས་དབང་ཡ་རེ་ང་།། ག་ཤེད་ཅིག་ཏུ་མེ་ཏོག་ཁེར་སྐྱེས་ཤིག། ད་དུང་ཟིལ་བ་དང་བཅས་ལྷམ་མེར་བཞད།། ཟིལ་བ་དེ་ནི་ཞོགས་པའི་ཉི་ཟེར་ཁྲོད།། དྭངས་གསལ་དྲི་མ་མེད་པར་འོད་དུ་འཚེར།། ད་ནི་གྲང་འཁྱག་བསླེབ་པར་ཐག་མི་རིང་།། སྔོ་ལྗང་ལ་ནི་ཚེ་ཚད་ཅི་ཙམ་མཆིས།། དེ་ལྟར་དྲན་པའི་མོད་ལ་ཟིལ་བ་ནི།། གློ་བུར་རླུང་ལྡོམ་ཞིག་གིས་སར་སྟོར་སོང་།། ཟིལ་བ་ས་ལ་ལྷུང་བའི་སྐད་ཅིག་མར།། སུ་ཡི་མིག་ཆུ་སེམས་ཀྱི་གཏིང་དུ་ལྷུང་།། ཟིལ་བ་ཆགས་སའི་ལོ་འདབ་སྔོན་པོའི་སྟེང་།། གར་ཡོང་མ་ངེས་བལ་ཞིག་དེ་མུར་ཆགས།། ”ཞེས་པ་དེའོ།། དགུན་ཁ་འདི་ལྟར་གློ་རྡེག་ལ་སླེབས་པས། ཁོ་མོ་དང་ངེད་གཉིས་ཀྱི་བར་ཐག་ནི་ཤིན་ཏུ་ནས་ཇེ་རིང་དུ་སོང་འདྲའི་ཚོར་སྣང་ཞིག་དྲག་ཏུ་སྟེར། གསལ་བོར་བཤད་ན། ད་ནི་ཁོ་མོ་དང་ངེད་གཉིས་ཀྱི་བར་དུ་དུས་ཚིགས་གཉིས་མཆིས་ཏེ། གཅིག་ནི་དབྱར་ཁ་དང་ཅིག་ཤོས་ནི་སྟོན་ཁའོ།། གལ་ཏེ་ཁོ་མོ་དཔྱིད་ཀ་བརྒལ་ནས་མར་ཡོང་ཞིང་། ང་ནི་དགུན་ཁ་བརྒལ་ནས་ཡར་སོང་ནའང་། དུས་ཚིགས་གཅིག་ལ་ཕན་ཚུན་འཕྲད་ཐབས་མི་འདུག་པས། ཕུགས་རེ་དེ་སྟོང་བསམ་ཞིག་ཏུ་མི་བརྩི་ག་མེད་རེད། གང་ལྟར། ངས་ཁོ་མོར་ནམ་རྒྱུན་བརྩེ་བའི་དབྱར་གཞུང་ཞིག་ཡོད་ཅིང་། སྟོན་གྱི་འབྲས་བཟང་ཞིག་ཀྱང་ཐོབ་སྟེ། དགུན་གྱི་སྐྱོ་ངལ་དང་བྲལ་བར་སྨོན་ལམ་ཡང་ཡང་འདེབས། ད་ནངས། སྒེའུ་ཁུང་ལས་ཕྱི་རོལ་དུ་ཁ་བ་བབས་པ་མཐོང་འཕྲལ། ང་རང་མྱུར་དུ་ཡར་ལངས་ཏེ་ཉེ་རྒྱང་ཀུན་ལ་ཞིབ་ཏུ་བལྟས་པ་ན། མཐོང་ལམ་གྱི་ཁོར་ཡུག་ཧྲིལ་བོ་ནི་དཀར་སིང་སིང་གི་འཇིག་རྟེན་ཞིག་ཏུ་གྱུར་འདུག་ཅིང་། སྐབས་དང་སྐབས་སུ་བཟོད་དཀའ་བའི་འཁྱག་རླུང་ཞིག་ཀྱང་གྲང་ཤུར་ཤུར་གྱིས་ལྡང་བར་བྱེད། སྣང་ཚུལ་དེ་མཐོང་ལམ་དུ་མངོན་པ་ན། ཁ་བས་གཡོགས་པའི་འཇིག་རྟེན་དེ་ནི་སུ་ཞིག་གི་སེམས་པ་ཡིན་པ་འདྲ་ལ། གྲང་ཤུར་ཤུར་གྱི་འཁྱག་རླུང་དེ་ནི་སུ་ཞིག་གི་ཤུགས་རིང་ཡིན་པ་འདྲ། ཡིན་ནའང་། ངས་བསམས་ན། མིའི་སེམས་ཀྱི་སྐྱོ་གདུང་ནི་ཕྱི་རོལ་ཡུལ་གྱི་སྣང་ཚུལ་གྱིས་ཇི་མ་ཇི་བཞིན་དུ་གཏན་ནས་མཚོན་ཐབས་མི་འདུག་སྙམ། དཀར་སིང་སིང་གི་ཁ་བའི་སྟེང་ནས། ང་རང་གང་ཞིག་ཏུ་འགྲོ་འདོད་ཀྱི་འདུན་པ་འགོག་མེད་ཅིག་སྐྱེས་ཤིང་། དེའི་མུར་གོམ་པའི་འདེགས་འཇོག་མདུན་དུ་བསྒྱུར། གོམ་པའི་འདེགས་འཇོག་དང་བསྟུན་ནས་ཐྲིག་ཐྲིག་གི་སྒྲ་དབྱངས་ཤིག་གིས་ངའི་མདུན་སྐྱོད་ཀྱི་གོམ་ཤུགས་ལ་རམ་བུ་འདེགས་ནའང་གོམ་འགྲོས་ནི་མཚམས་རེར་འཁྱོར་ཅིང་ལྗིད་ཏིག་ཏིག་ཏུ་འགྱུར་འགྲོ། དཀར་སིང་སིང་གི་ཁ་བའི་སྟེང་དུ། ངས་སྔོན་ཆད་ཅིག་ལ་སྙིང་ཆུང་ཆུང་གཉིས་ལྷན་དུ་སྦྲེལ་ཡོད་པའི་གཟུགས་རིས་ཤིག་བྲིས་མྱོང་། དེ་ནི་དོན་དུ་སེམས་ཀྱི་རུས་པར་བརྐོས་ཡོད་པའི་རྣམ་པ་ཙམ་ཕྱི་རུ་ཕབ་པ་ཞིག་ཡིན་ལ། དེ་དུས་ཀྱི་ཁ་བ་དཀར་པོ་ནི་སྔ་མོ་ནས་ཉི་མས་ཞུ་ཟིན་ནའང་། སེམས་ཀྱི་རུས་པར་བརྐོས་པའི་རི་མོ་ནི་དུས་གཏན་དུ་ཡལ་མི་སྲིད། ཡིན་ནའང་། རི་མོ་དེ་ནི་ད་ལྟ་སེམས་ཀྱི་རྨ་ཁར་གྱུར་ཡོད་ཅིང་ན་ཟུག་ཀྱང་རྒྱུན་དུ་སྐྱེ། ཁ་བའི་སྟེང་དུ། མི་དང་སྲོག་ཆགས་རེ་འགའི་རྐང་རྗེས་ཁ་ཤས་ཕར་དང་ཁ་ཤས་ཚུར་ལ་འཐེན་འདུག དེ་ནི་ཇི་འདྲའི་དོ་མི་མཉམ་པ་ཞིག་རེད་ཨང་། མི་དང་སྲོག་ཆགས་ཁ་ཤས་ཅིའི་ཕྱིར་ཕར་འགྲོ་དགོས་པ་ཡིན་ནམ། ཁ་ཤས་ཅིའི་ཕྱིར་ཚུར་ཡོང་དགོས་པ་ཡིན་ནམ། ཕན་ཚུན་མཉམ་དུ་འཛོམས་ནས་ཕྱོགས་གཅིག་ཏུ་སོང་ན་ཅིའི་ཕྱིར་མི་ཆོག དེ་ནི་དངོས་གནས་བསྒྱུར་དཀའ་བའི་ལས་དབང་ཞིག་ཡིན་ནམ། རྐང་རྗེས་དེ་དག་གིས་ང་ལ་དེ་འདྲའི་དགོས་མེད་ཀྱི་དྲི་བ་མང་པོ་ཞིག་འདོན་དུ་བཅུག དོན་དུ་དྲི་བ་དེ་དག་ནི་ཁོ་མོར་འདོན་དགོས་ཀྱང་ད་ནི་རང་བྱུང་ཁམས་ལ་འདོན་པ་ལས་ཐབས་གཞན་ཅི་ཞིག་མཆིས། ད་དུང་། རྐང་རྗེས་དེ་དག་གིས་ང་ལ་རྒྱལ་བ་ཚངས་དབྱངས་རྒྱ་མཚོ་དྲན་དུ་བཅུག་བྱུང་། ཞོགས་པ་ཞིག་ལ་རྒྱལ་བ་ཚངས་དབྱངས་རྒྱ་མཚོས་ཀྱང་རྐང་རྗེས་ཤིག་ལ་གཟིགས་ནས། “སྲོད་ལ་བྱམས་པ་བཙལ་བས།། ཐོ་རངས་ཁ་བ་བབས་བྱུང་།། གསང་དང་མ་གསང་མི་འདུག། ཞབས་རྗེས་གངས་ལ་བཞག་འདུག།”ཅེས་གསུང་དུས་ཀྱང་ཏག་ཏག་ད་ནངས་འདྲ་བའི་ཞོགས་པ་ཞིག་ཡིན་སྲིད། ཡིན་ནའང་། ཞོགས་པ་འདིའི་ཁ་བའི་ཁྲོད་དུ་ཁོ་མོའི་རྐང་རྗེས་ཕབ་ཡོད་ན་ཅི་མ་རུང་སྙམ་ཡང་དེ་ནི་གཏན་ནས་མི་སྲིད་དེ། ཁོ་མོ་ནི་དཔྱིད་ཀ་ན་ལུས་ཤིང་། ང་ལ་དགུན་ཁ་སླེབས་པས། ཁོ་མོའི་ཁོར་ཡུག་ལ་རླུང་བུ་ལྡང་བ་མ་གཏོགས་ཁ་བ་ཞིག་འབབ་མི་སྲིད། ཁ་བའི་འཇིག་རྟེན་གྱི་ག་ཤེད་ཅིག་ཏུ། ཤིན་ཏུ་སྙིང་རྗེ་བའི་བྱེའུ་ཆུང་གཉིས་ཀྱིས་ཀྱང་ཀྱུར་ཀྱུར་གྱི་གསུང་སྐད་འབྱིན་བཞིན་ཡར་འཕུར་མར་འཕུར་བྱེད་ཅིང་། མཚམས་རེར་ཁ་བའི་སྟེང་དུ་བབས་ཏེ་ཕྲ་ལ་མཉེན་པའི་སྡེར་མོས་ཡིད་དུ་འོང་བའི་རི་མོ་སྣ་ཚོགས་ཤིག་འབྲི་བར་བྱེད། དེ་ནི་དངོས་གནས་སྤྲོ་སྐྱིད་ལྡན་པ་ཞིག་རེད། ཡིན་ནའང་། ངས་ནི་དེ་ལ་ཡིད་སྨོན་ཙམ་བྱེད་པ་ལས་སྤྲོ་སྐྱིད་དེའི་ཆ་ཤས་ཙམ་ཡང་རང་ལ་དབང་ཐབས་མི་འདུག ཡུད་ཙམ་གྱི་རྗེས་སུ། ཤར་ཕྱོགས་ཀྱི་རི་རྩེར་འཕགས་པའི་ཉི་མ་འང་མངོན་ཆུང་གིས་རབ་རིབ་ནས་ཅུང་ཟད་ཇེ་གསལ་དང་ཇེ་དྲོར་གྱུར་ལ། བྱིས་པ་མང་པོ་ཞིག་ཀྱང་གང་ཞིག་ནས་འུར་ཆིལ་ལེར་ཡོང་སྟེ། ལ་ལས་ཁངས་རྩེད་བྱེད་ཅིང་། ལ་ལ་ཁངས་ཤུད་སྟེག དེ་ནས་ཚང་མ་མཉམ་གཅིག་ཏུ་འདུས་ཏེ་ཁ་བའི་ཁང་ཁྱིམ་ཞིག་ལས་བྱུང་བ་མ་ཟད། ད་དུང་དེ་རུ་ཁ་བའི་བུ་དང་བུ་མོ་ཞིག་དང་། དེའི་སྒོ་ཁར་ཁ་བའི་རྟ་དང་། ནོར། ལུག་བཅས་ཀྱང་ལས་བྱུང་། སྐབས་དེར། ང་ནི་ཅི་ཞིག་གིས་ཁྲིད་བཞིན་པ་དང་འདྲ་བར་ཁོ་ཚོས་ལས་པའི་ཁ་བའི་ཁང་ཁྱིམ་དུ་སོང་སྟེ། ཁ་བའི་བུ་དེ་ནི་རང་ཉིད་དང་ཁ་བའི་བུ་མོ་དེ་ནི་ཁོ་མོ་ཡིན་པར་རྟོག་བཟོ་བྱེད་ཀྱིན་དེ་གཉིས་ཀྱི་གན་དུ་ཁ་བའི་བུ་ཆུང་ཞིག་ཀྱང་ལས་དགོས་བྱུང་། ཁ་བའི་བུ་ཆུང་དེ་ལས་དུས། ངས་རང་གི་བློ་རྩེ་ཐམས་ཅད་དེའི་སྟེང་དུ་སྒྲིམ་སྟེ་ཧ་ཅང་ཞིབ་ཚགས་ཀྱིས་ལས་པ་ཡིན། ཁ་བའི་བུ་ཆུང་དེ་ལས་ཚར་རྗེས། ད་ནི་ཅི་ཞིག་གྲུབ་སོང་སྙམ་པའི་སེམས་བྱུང་གིས་ཅུང་ཟད་ལྷོད་ལ་བབས་པ་ན། ངས་རང་གི་ཉེ་གན་དུ་བུ་མོ་ཞིག་སླེབས་འདུག་པ་མཐོང་བྱུང་ལ། བུ་མོ་དེའི་གན་དུ་ད་དུང་བྱིས་པ་ཆུང་ཆུང་ཞིག་ཀྱང་འདུག མཐོང་སྣང་དེས་ང་རང་གློ་རྡེག་ལ་ལྐུགས་པ་ཞིག་ཏུ་བཏང་། ངས་ཅི་ཞིག་བཤད་བསམ་ཡང་ངག་གཏན་ནས་འཁྲོལ་རྒྱུ་མི་འདུག ཁོ་མོས་མིག་ཆུ་ཤམ་ཤམ་གྱིས“མཚན་མོ་དེའི་དབང་གིས། ངས་དབྱར་གཞུང་གཏན་ནས་མཐོང་མ་བྱུང་ལ། སྟོན་ཁར་ཡང་ཞེན་ཆགས་ཅི་ཡང་མི་འདུག ང་དཔྱིད་ཀ་ནས་ཐད་ཀར་དུ་དགུན་ཁར་ཡོང་བ་ཡིན།”ཞེས་སྨྲས་བྱུང་། ཁོ་མོས་དེ་ལྟར་སྨྲས་རྗེས། ངའི་སེམས་ཀྱི་ག་ཤེད་ཅིག་ནས་གློ་རྡེག་ལ་དྲོད་ཅིག་འཁྱིལ་ཡོང་། ཡིན་ནའང་། ཁོ་མོས་དེ་མ་རྟོགས་པ་འདྲ་སྟེ། མཚམས་རེར་ང་ལ་ལྟ་ཞིང་མཚམས་རེར་རང་གི་བྱིས་པ་ལ་ལྟ་བར་བྱེད།
