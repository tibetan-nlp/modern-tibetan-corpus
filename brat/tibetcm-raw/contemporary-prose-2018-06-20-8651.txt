ཚེ་རིང་དོན་གྲུབ་ཀྱི་ཐོལ་རྩོམ་གསར་པ། 《ལྷོ་བོད་ཟིན་ཐོ》
ཚེ་རིང་དོན་གྲུབ།


ལོ་ངོ་སུམ་ཅུའི་ཡར་སྔོན་ཏེ1987ལོའི་ཟླ་བརྒྱད་པར། བོད་ལྗོངས་རྩོམ་རིག་སྒྱུ་རྩལ་ལྷན་ཚོགས་དང《བོད་ཀྱི་རྩོམ་རིག་སྒྱུ་རྩལ》དུས་དེབ་ཁང་གིས་གཙོ་གཉེར་བྱས་པའི་བོད་ཡིག་རྩོམ་པ་པོའི་སྨྱུག་ཚོགས་ལྷ་སར་བསྡུས། ཕ་ཡུལ་ནས་ཟི་ལིང་བར་ལ་སྤྱི་སྤྱོད་རླངས་འཁོར་དུ་བསྡད་ནས་ཏག་ཏག་ཉིན་གཅིག་དང་། ཟི་ལིང་ནས་གོར་མོ་བར་ལ་ཤིང་གི་རྐུབ་སྟེགས་ཅན་གྱི་མེ་འཁོར་རྙིང་བ་ཞིག་ལ་བསྡད་དེ་ཉིན་གཅིག་ཞག་གཅིག དེ་ནས་ལྷ་ས་བར་ཉལ་ཁྲི་ཅན་གྱི་སྤྱི་སྤྱོད་རླངས་འཁོར་དུ་བསྡད་དེ་གྲང་ལྟོགས་གཉིས་ཀའི་ངང་ནས་གསོན་པོར་ལྷ་སར་འབྱོར་ཙམ་བྱུང་། ཡིན་ནའང་དེ་དུས་ཀྱི་རྩོམ་གྲོགས་ལ་ལས་ཕྱིར་དྲན་ནང་བྲིས་ཡོད་པ་བཞིན“དེ་དུས་ང་ཚོ་ཤ་དར་ཁྲག་དར་གྱི་ཕོ་གསར་མོ་གསར”ཡིན་པས་ན། དཀའ་སྡུག་མྱངས་པ་དང་ཐང་ཆད་པའི་སྣང་བ་མེད་ལ། དབྱངས་དབུགས་མི་འདང་བ་དང་ལ་དུག་ཕོག་ཟེར་བ་ནི་དེ་བས་ཀྱང་གོ་མ་མྱོང་བའི་ཚོད་རེད། རྗེས་སུ་གཞུང་གི་ལས་ཀ་དང་སྒེར་གྱི་ཡུལ་སྐོར་གྱི་དབང་གིས། ཤར་ཞི་བདེ་རྒྱ་མཚོའི་འགྲམ་དང་བྱང་སོག་པོའི་རྒྱལ་ཁབ་ཀྱི་ས་སྣེ་དང་ཨུ་རུ་སུའི་ས་མཚམས། ནུབ་ལི་ཡུལ་གྱི་བྱེ་ཐང་དང་གྲོང་ཁྱེར། ལྷོ་འཇང་ཡུལ་དང་ཧྭེ་ནན་གླིང་ཕྲན་སོགས་སུ་འགྲོ་མྱོང་མོད། འོན་ཀྱང་དུས་དང་རྣམ་པ་ཀུན་ཏུ་རྡོ་ཁབ་ལེན་ཇི་བཞིན་ངའི་སེམས་པ་འདྲེན་པར་བྱེད་པ་ནི་དབུས་གཙང་གཅིག་པོ་རེད། དེའི་ཕྱིར་མཚམས་རེར་ནང་མི་དང་མཚམས་རེར་གྲོགས་པོ། མཚམས་རེར་རང་སྒར་འབྲོག་གི་ལས་སྣེ་དང་མཉམ་དུ་ལོ་འགའ་རེའི་ནང་ལྷ་སར་ཐེངས་རེར་བསྐྱོད་དགོས་བྱུང་། ལྷག་པར་དུ་ལོ་དྲུག་བདུན་ཙམ་གྱི་ཡར་སྔོན་ནས་ནང་མི་རྒན་པ་དག་སྔ་གཞུག་ཏུ་འཇིག་རྟེན་གཞན་ལ་གཤེགས་པ་ནས་བསྟུད་མར་ལོ་རེར་ལྷ་སར་ཐེངས་རེར་འགྲོ་དགོས་བྱུང་བ་དང་། རྗེས་སུ་ཤི་ཆད་མ་བྱུང་ནའང་ལྷ་སར་འགྲོ་རྒྱུ་ནི་གོམས་སྲོལ་ཞིག་ཏུ་ཆགས།
ད་ལོའི་སྤྱི་ཟླ་ལྔ་པའི་ཚེས་མགོ་ནས་ངེད་བཟའ་མི་གཉིས་གྲོགས་པོ་སྟེང་ཞལ་དང་མཉམ་དུ་དབུས་གཙང་ལྷོ་ཕྱོགས་སུ་ཟླ་ཕྱེད་ལྷག་གི་ལྗོངས་རྒྱུར་སོང་བའི་ལམ་བར་དུ། ཡིད་སྨོན་འཆོར་བའི་དམངས་སྲོལ་གཙང་མ་དང་མིག་དབང་འཕྲོག་པའི་རི་ཆུ་ནགས་གསུམ་གྱིས་དགའ་བ་བླ་མེད་སྦྱིན་པའི་དུས་མཚུངས་སུ། ལམ་འགྲམ་གྱི་གད་སྙིགས་སྦགས་བཙོག་དང་ཚིག་སྐྱོན་ཡིག་ནོར་གྱིས་ཁེངས་པའི་སྒོ་བྱང་དང་ལམ་བྱང་གིས་ཕངས་སེམས་རབས་དང་རིམ་པ་བསྐྱེད་དེ་མཐར་གཏུགས་ན་བོད་རང་སྐྱོང་ལྗོངས་ཟེར་བས་བོད་ཀྱི་རིག་གནས་ལ་འདི་ལྟར་སྣང་ཆུང་བྱེད་དགོས་དོན་ཅི་ཡིན་དང་། བོད་ལྗོངས་ལ་མི་རིགས་སྐད་ཡིག་བྱ་བའི་ལས་ཁུངས་ཤིག་ཡོད་མེད། གལ་ཏེ་ཡོད་ན་དེ་གའི་མི་རྣམས་ལོང་བ་ཡིན་ནམ་ཡིག་རྨོངས་པ་ཡིན་མིན་ལ་དོགས་པ་མི་ཟ་ཀ་མེད་བྱུང་། སྤྱིར་ན་དེང་སྐབས་ང་ཚོས་སྐད་འཕྲིན་དུ་མཐོང་བའི“བོད་མི་སྡེབ་འཚོང”ལྟ་བུ་སོགས་རེ་རེ་བཞིན་བཀོལ་ན་འགྲིག་མོད། འོན་ཀྱང་བོད་མི་རང་གིས་རང་གི་སྐད་ཡིག་ལ་བརྙས་བཅོས་བྱེད་པའི་རྣམ་པ་འདི་འདྲ་ནི་སུས་ཀྱང་མི་མཐོང་བའི་སྨོན་ལམ་དང་བཅས་མ་བྲིས་སོ། །
དེ་ཡང་ལྷ་ས་ནས་ཉིང་ཁྲིའི་བར་ལ་སྤྱི་ལེ་བཞི་བརྒྱ་ཡོད་པར་མགྱོགས་ལམ་ཤར་གཏོང་ཐུབ་པ་དང་། མི་འགྱང་བར་ལྕགས་ལམ་ཡང་ཤར་གཏོང་ཐུབ་པས་ཧ་ཅང་སྟབས་བདེ་ཡིན་པའི་ཁར་སྤྱི་སྤྱོད་རླངས་འཁོར་དུ་བསྡད་ན་རིན་གོང་དམའ་མོ་ཡིན་ནའང་གྲོག་མ་བཞིན་མང་ཞིང་། མི་མང་འདུ་ས་དང་ལྷག་པར་དུ་ཟ་འཐུང་བྱེད་སར་སྐད་མགོ་མཐོན་པོས་འུར་སྐད་རྒྱག་པ་དང་། གང་སར་གད་སྙིགས་འདོར་བ་སོགས་ཀྱིས་འཛམ་གླིང་ཡུལ་གྲུ་སོ་སོའི་མི་རྣམས་ཧ་ཅང་སུན་པོ་བཟོ་བའི་ནང་ལོགས་ཀྱི་ཡུལ་སྐོར་བ་དང་མཉམ་དུ་འགྲོ་དགོས་པ་ནི་ངེས་གཏན་རེད། དེའི་ཕྱིར་ང་ཚོས་རིན་གོང་མཐོ་བའི་རླངས་འཁོར་ཆུང་ཆུང་གླས་ནས་མི་འགྲོ་ཀ་མེད་བྱུང་། ལམ་བར་དུ་ཁ་བ་ལྷབ་ལྷུབ་འབབ་པའི་དགུན་ཁ་དང་མེ་ཏོག་རྣམ་པར་བཞད་པའི་དབྱར་ཁ་བརྒྱུད་དེ་རླངས་འཁོར་ཁ་ལོ་བར་ཡང་བསྐྱར་གླ་ཆ་ཁ་སྣོན་བྱས་ནས་མགྱོགས་ལམ་དང་བར་ཐག་སྤྱི་ལེ་བཞི་བཅུ་ཡས་མས་ཡོད་པའི་བྲག་གསུམ་མཚོ་ལ་སོང་། གངས་རི་དང་། གཡའ་རི། རྫ་རི། ནགས་རིས་ཡོངས་སུ་བསྐོར་བའི་དཀྱིལ་ན་གཡུ་ཡི་མེ་ལོང་འདྲ་བའི་མཚེའུ་ཞིག་ཡོད་པ་དང་། མཚེའུའི་ཤར་སྣེའི་དཀྱིལ་ན་གསོམ་པ་དང་ཤུག་པ། བེ་ཤིང་། སྟག་མ་སོགས་ཚང་ཚིང་སྐྱེས་པའི་གླིང་ཕྲན་ཆུང་ངུ་ཞིག་གི་སྟེང་དུ་མཚོ་རྫོང་དགོན་ཞེས་པའི་བཀའ་བརྒྱུད་ཀྱི་དགོན་པ་ཆུང་ངུ་ཞིག་ཡོད་པ་བཅས་ནི་དྲན་ཤུགས་མཐའ་ཡས་པའི་སྒྱུ་རྩལ་པ་ཞིག་གི་པིར་རྩེ་ལས་ཐོན་པའི་སྣུམ་ཚོན་རི་མོ་ཞིག་དང་གཉིས་སུ་མ་མཆིས་པས་འདི་ལ་བྲག་གསུམ་མཚོ་ཟེར་བ་ལས་སུམ་རྩེན་མཚོ་ཟེར་ན་མི་འགྲིག་གམ་སྙམ་པའི་བག་ཆགས་སྟེར།
ཉིང་ཁྲིར་སླེབས་དུས་གྲོགས་པོ་ཞིག་གིས་རིན་མེད་དུ་སྡོད་གནས་བཀོད་སྒྲིག་བྱས་ཡོད་པར་མ་ཟད་དགོང་ཟས་ཀྱང་གྲ་སྒྲིག་བྱས་འདུག ཕྱི་ཉིན་བླ་མ་གླིང་ཞེས་པའི་རྙིང་དགོན་དེར་མཆོད་མཇལ་ཞུས་པར་གྲོགས་པོ་གཞན་ཞིག་གིས་སྔོན་ནས་འབྲེལ་གཏུག་བྱས་ཡོད་པའི་ཕ་ཡུལ་ནག་ཆུ་ནས་ཡིན་པའི་གྲྭ་བ་ཞིག་གིས་རྟེན་བཤད་རྒྱས་པ་གནང་མོད། འོན་ཀྱང་དགོན་སྡེ་ཁག་གི་ངོ་སྤྲོད་ནི་གདན་རབས་དང་གནས་ཡིག་སོགས་སུ་ཧ་ཅང་མང་བས་འདིར་བསྐྱར་ཟློས་བྱེད་པའི་དགོས་པ་མ་མཆིས། དེ་ནས་གནས་ཡུལ་ལུང་བ་ཞེས་པར་སོང་སྟེ་གདོད་མའི་ནགས་ཚལ་གྱི་མཛེས་ལྗོངས་ལ་ལྟ་སྐོར་བྱས། དེ་རུ་རྩེ་མོ་གནམ་ལ་ཟུག་པའི་ལྗོན་པ་ཚང་ཚིང་སྐྱེས་ཡོད་པ་དང་། རྒས་ནས་ས་ལ་འགྱེལ་འདུག་པའི་ལྗོན་པ་ཡང་ངའི་ཕ་ཡུལ་གྱི་རྩྭ་སྐྱ་བཞིན་དུ་མང་བ་མཐོང་ཚེ《དམ་པའི་གསུང་རབས》ཏུ་ཡོད་ཚད་ས་ནས་སྐྱེས་ཤིང་། ཡོད་ཚད་ས་ལ་ཐིམས་དགོས་ཅེས་པ་དབང་མེད་དུ་དྲན་ཡོང་། ཕྱི་ཉིན་མུ་མཐུད་དུ་རླངས་འཁོར་གླས་ཏེ་ཡར་ཀླུང་གཙང་པོའི་ཁུགས་སོ་ཆེན་མོར་ལྟ་སྐོར་དང་ནམ་ཡང་སྤྲིན་གྱིས་གཡོག་པའི་གནམ་ལྕགས་འབར་བས་ང་ལ་དགའ་བསུ་ཞུ་བ་བཞིན་སྐད་ཅིག་ཙམ་ལ་ཞལ་ཕྱེས་པར་རྒྱང་མཇལ་ཙམ་བྱས་རྗེས། མཚམས་རེར་ཁ་ཆར་འདྲེས་མ་དང་མཚམས་རེར་སྨུག་སྤྲིན་སྟུག་པོའི་ཁྲོད་དུ་རི་ཀླུང་མཐོན་པོ་དང་ཆུ་ཀླུང་དྭངས་མོ། ནགས་ཚལ་སྟུག་པོ་དང་སྤང་ཐང་བདེ་མོ་སོགས་ཧ་ཅང་ཉམས་དགའ་བའི་ཀླུ་ནང་མཛེས་ལྗོངས་བརྒྱུད་དེ་སྤོ་བོ་རྫོང་ལ་སོང་། ཉིང་ཁྲི་ནས་སྤོ་བོ་རྫོང་གི་བར་ལ་སྤྱི་ལེ་ཉིས་བརྒྱ་ལྷག་ལས་མེད་ནའང་མགྱོགས་ཚད་ཚོད་འཛིན་ཡོད་པས（བོད་ལྗོངས་ཀྱི་གཞུང་ལམ་ཚང་མར་མགྱོགས་ཚད་ཚོད་འཛིན་ཡོད་ལ། བཤད་ཚུལ་ལྟར་ན་ལོ་འགའི་སྔོན་བསྟུད་མར་འཁོར་གེགས་ཚབས་ཆེན་ཐེངས་འགའ་བྱུང་སྟེ་ནང་ལོགས་ཀྱི་ཡུལ་སྐོར་བ་མང་པོ་རྐྱེན་ལམ་དུ་གྱུར་པས་རེད་ཟེར）ཕལ་ཆེར་ས་རུབ་ལ་ཉེ་དུས་མ་གཏོགས་སླེབས་མ་ཐུབ། ཕྱིར་ལམ་སྤོ་བོ་བརྒྱུད་ཉིང་ཁྲི་ནས་ཞག་གཅིག་ལ་བསྡད་རྗེས་ལྷོ་ཁའི་ཕྱོགས་སུ་སོང་། ལམ་བར་སླེབས་རིའི་གནའ་བོའི་བང་སོའི་ཚོགས་ལ་འཁྱུག་ཙམ་བལྟས། སྔོན་ཆད་ཨི་ཐ་ལིའི་མཁས་པའི་དབང་པོ་G•ཏོའུ་ཆིའི《བོད་ཀྱི་གནའ་བརྟག》དང་མཚོ་སྔོན་ཏུའུ་ལན་རྫོང་དུ་ཡོད་པའི་གནའ་བོའི་བང་སོའི་སྐོར་གྱི་ཞིབ་འཇུག་སོགས་གནའ་བརྟག་དང་གནའ་བོའི་བང་སོའི་སྐོར་གྱི་དཔེ་ཆ་ཧ་ཅང་ཉུང་ཙམ་ལ་མ་གཏོགས་བལྟས་མ་མྱོང་བས་དེའི་སྐོར་གྱི་ཤེས་བྱ་ཅི་ཡང་མེད་མོད། འོན་ཀྱང་ཡ་མཚན་པ་ཞིག་ནི་ང་ལ་མཚོན་ན་གནའ་ཤུལ་གནའ་རྫས་ཀྱིས་འདས་མ་ད་གསུམ་དང་དངོས་ཡོད་གཟུགས་འདས་བཅས་ལ་མི་དྲན་དགུ་དྲན་བྱེད་པའི་འདུ་ཤེས་ཁྱད་པར་བ་ཞིག་སྟེར་གྱིན་འདུག ལྷག་སྒྲོན་གྱིས་བརྩམས་པའི《བོད་ཀྱི་གནའ་རབས་མཆིམས་ཀྱི་རུས་རྒྱུད་སྐོར་བརྗོད་པ》ཞེས་པའི་ནང་དུ“སླེབས་རིའི་བང་སོའི་ཚོགས་འདི་ཉིད་ནི་བོད་ཀྱི་ས་ཁུལ་ཁག་ལས་གནའ་རྫས་རྟོག་དཔྱོད་པས་རྙེད་པའི་གནའ་རབས་བང་སོའི་ཚོགས་ཀྱི་ནང་ནས་གཞི་རྒྱ་ཆེ་ཤོས་དང་། རྒྱ་མཚོའི་ངོས་ནས་མཐོ་ཚད་སྤྱི་ཁྲེ་སུམ་སྟོང་ཉིས་བརྒྱ་ཡན་ཟིན་གྱི་ཡོད། རི་དེའི་གཤམ་དུ་གནས་པའི་གྲོང་ཚོར་སླེབས་རི་གྲོང་ཚོ་ཞེས་རྗོད་པ་དང་། གྲོང་ཚོ་དེའི་ཤར་ངོས་སྤྱི་ལེ་ལྔ་ཙམ་གྱི་སར་སྐྱེམས་སྟོང་ཤང་གནས་ཡོད། སྐྱེམས་སྟོང་ཕུ་ཆུ་དེ་སླེབས་རི་བྱང་མའི་ལྷོ་འདབས་བརྒྱུད་ཤར་ནས་བྱང་ལ་ཁ་འགྱུར་ཏེ་ཡར་ཀླུང་གཙང་པོར་འདྲེས། བང་སོའི་ཚོགས་གཉིས་ཀྱི་རྒྱ་ཁྱོན་དང་བང་སོའི་གྲངས་འབོར། བཟོ་དབྱིབས། ཆེ་ཆུང་། མཐོ་ཚད་བཅས་དཔེ་རིས་འདྲ་པར་དང་མཉམ་དུ་ངོ་སྤྲོད་ཞིབ་མོ་བྱས་ཡོད་པར་མ་ཟད་ད་དུང་གནའ་བརྟག་རིག་པའི་མཁས་པ་བསོད་ནམས་དབང་འདུས་ཀྱི《གསར་དུ་རྙེད་པའི་གནའ་རབས་ཀྱི་བང་སོའི་ཚོགས》ཞེས་པའི་དཔྱད་རྩོམ་ནང་གི་སྤྱི་བསྡོམས་རང་བཞིན་གྱི་ཚིགས་དུམ་ཞིག་ཀྱང་ལུང་དྲངས་ཡོད་དེ། “སླེབས་རིའི་བང་སོའི་ཚོགས་དང་འཕྱོངས་རྒྱས་བཙན་པོའི་བང་སོའི་བཟོ་དབྱིབས་དང་སྐྲུན་ཐབས་སོགས་གང་ཅིར་གཞི་རྩའི་ཁྱད་པར་མེད། བང་སོ་མང་ཆེ་བ་སྟེང་ནས་འོག་ཏུ་བྲུས་ནས་དུར་རྫས་རྐུ་འཁྱེར་སོང་བའི་གནས་ཚུལ་ཡང་གཅིག་མཚུངས་ཡིན་པས། ཁུལ་དེ་གཉིས་ཀྱི་བང་སོ་འབངས་གྱེན་ལོག་མ་བྱུང་སྔོན་གྱི་སྤུ་རྒྱལ་བཙན་པོའི་དུས་སུ་བྱུང་བ་དང་། འབངས་གྱེན་ལོག་སྐབས་གཏོར་བཤིག་བྱུང་བ་མཐོང་ཐུབ།……སླེབས་རིའི་བང་སོ་ཚོགས་ལ་ཁྱོན་བསྡོམས་བང་སོ་ཉིས་བརྒྱར་ཉེ་བ་ཞིག་ཡོད་པ་དེ་རྣམས་ལོ་ངོ་སྟོང་ཕྲག་འགའི་ནང་རིམ་པར་བཏབ་ནས་ད་ལྟའི་གཞི་ཁྱོན་འདི་ཆགས། མ་ཟད་སླེབས་རིའི་བང་སོའི་ཚོགས་ལ་བཟོ་དབྱིབས་མི་འདྲ་བའི་བང་སོ་ཁ་ཤས་ཡོད་པས་བོད་ཁུལ་ནས་རྙེད་པའི་བང་སོ་གཞན་ཚང་མ་དང་མི་འདྲ་བའི་ཁྱད་པར་ཞིག་རེད།”
མདོར་ན་ས་རྡོ་ཤིང་གསུམ་གྱིས་བསྐྲུན་ཅིང་གཞི་རྒྱ་ཆེ་ལ་གྲངས་འབོར་མང་བའི་བང་སོའི་ཚོགས་དེ་མཐོང་དུས། ད་ལྟ་སྐྱེ་བོ་གང་མང་གིས་བླུན་རྨོངས་རྗེས་ལུས་དབུལ་ཕོངས་སུ་སེམས་པའི་བོད་འདི་ལོ་ངོ་སྟོང་ཕྲག་གི་ཡར་སྔོན་དུ་སྐབས་དེའི་ས་ཁུལ་གཞན་དག་དང་བསྡུར་ན་ཅི་འདྲའི་སྟོབས་འབྱོར་མངའ་ཞིང་ཤེས་རིག་ལྡན་པ་ཞིག་རེད་ཨང་སྙམ་བྱུང་།
ཉིན་འདིར་ད་དུང་རྒྱལ་དབང་བཅུ་གསུམ་པ་ཆེན་མོ་ཐུབ་བསྟན་རྒྱ་མཚོའི་འཁྲུངས་གཞིས་ལ་མཇལ་བའི་སྐལ་བ་ལྡན། འཁྲུངས་གཞིས་ནི་རིག་གསར་སྐབས་སུ་རྩ་མེད་བཟོས་ཤིང་། རིག་གསར་རྗེས་སུ་གསར་དུ་བཞེངས་པ་སྨོས་ཅི་དགོས། རྒྱལ་དབང་འདི་ནི་ཕལ་ཆེར་འཛམ་གླིང་ཧྲིལ་པོའི་གནས་ཚུལ་ཧ་ཅང་དཀའ་རྙོག་ཆེ་བའི་དུས་ལ་ཁེལ་ཞིང་། ཁོང་གིས་མ་རུང་ཁོག་བཅུག་གི་མན་ཆིང་དང་མེན་གོ་སྲིད་གཞུང་དང་དཔའ་ངར་ཞུམ་མེད་ཀྱིས་གདོང་གཏད་པ། ཁྱོན་ཡོངས་ནས་བོད་ཀྱི་ཆོས་སྲིད་ལ་བསྒྱུར་བཅོས་ཆེན་པོ་གནང་སྟེ་དམངས་གཙོ་དང་ཚན་རིག་གི་ཕྱོགས་སུ་སྐྱོད་པའི་ཐུགས་འདུན་མངའ་བ། ད་དུང་ཡང་བཟང་པོའི་རྒྱུད་པ་ཆད་མེད་ཟེར་བའི་དཔེ་ལྟར་གླང་མདུན་ཁྱིམ་རྒྱུད་འདིར་ང་ཚོར་ཧ་ཅང་ཆ་རྒྱས་ཡོད་པའི《གཙུག་གཡུ》རྩོམ་པ་པོ་གླང་མདུན་དཔལ་འབྱོར་ཡང་འཁྲུངས་མྱོང་བ་སོགས་ནི་ཚང་མས་མཁྱེན་གསལ་ཡིན་པས་འདིར་ཡིག་ཚོགས་སྤེལ་བའི་དགོས་པ་མེད་ལ། རྒྱལ་དབང་མཆོག་གི་འཁྲུངས་གཞིས་ལ་མཇལ་སྐབས་རང་དབང་མེད་པར་ཁོང་དང་འབྲེལ་བ་ཟབ་པའི་ཕྱི་གླིང་བ་གཉིས་དྲན་ཡོང་བའི་ནང་གི་གཅིག་ནི་ཨུ་རི་སིའི་ཁོངས་ཀྱི་སོག་ཡུལ་ཧ་རི་ཞི་སྦེར་ཞེས་པའི་ནང་ཚན་སྤོ་རད་ཐུ（དགེ་བཤེས་ཆོས་གྲགས་ཀྱི་ཕ་ཡུལ་ཡང་འདི་རེད）ཞེས་པར་འཁྲུངས་པའི་སོག་པོ་མཚན་ཞབས་པ་ངག་དབང་བློ་བཟང་ངམ་མཚན་གཞན་ལ་རྡོ་རྗེ་ཡིབ（1854—1938）ཅེས་པ་དེ་ཡིན་ལ། ཁོང་ལོ་བཅུ་དགུའི་སྟེང་འཁྲུངས་ཡུལ་དང་བྲལ་ནས་ལྷ་སར་སླེབས་ཏེ་འབྲས་སྤུངས་སྒོ་མང་གྲྭ་ཚང་གི་ཆོས་སྒོར་ཞུགས། ལྷ་སར་བཞུགས་རིང་ལོ་ངོ་བཅུའི་རིང་ལ་རྒྱལ་དབང་སྐུ་ཕྲེང་བཅུ་གསུམ་པའི་མཚན་ཞབས་མཛད། དུས་མཚུངས་སུ་ཡང་ཡང་རང་ཡུལ་དུ་ཕེབས་ཏེ་དགོན་པ་དུ་མ་གསར་བཞེངས་གནང་། ཁོང་ནི་ཆོས་ལུགས་མི་སྣ་ཙམ་མ་ཡིན་པར་སྐབས་དེའི་བོད་ཕྱོགས་ཀྱི་ཆབ་སྲིད་སྟེང་གི་ལོ་རྒྱུས་མི་སྣ་གལ་ཆེན་ཀྱང་ཡིན་ཏེ། ཁོང་རྒྱལ་དབང་མཆོག་གི་ཕོ་ཉའི་ཚུལ་གྱིས་སྐབས་དེའི་ཨུ་རི་སིའི་རྒྱལ་པོར་དངོས་སུ་མཇལ་ཏེ་ཁོང་གཉིས་བར་འབྲེལ་བ་འཛུགས་པའི་འབད་བརྩོན་བྱས། སྐབས་དེའི་རྒྱ་ནག་གི་གོང་མ་ཡོན་ཧྲི་ཁེ་རྒྱལ་དབང་མཆོག་ལ་ཡི་མ་རངས་པར་རྒྱལ་བ་རིན་པོ་ཆེའི་མཚན་གནས་ནས་བླངས་པ་ཁྱབ་བསྒྲགས་བྱས་པར། རྡོ་རྗེ་ཡིབ་ཀྱིས་རྒྱལ་བ་རིན་པོ་ཆེའི་མཚན་གནས་དེ་སྲིད་ཀྱི་གོ་གནས་ལྟར་སྟེར་འདོད་ན་སྟེར་བ་དང་ལེན་འདོད་ན་བླངས་ཆོག་པ་ཞིག་གཏན་ནས་མ་ཡིན་པའི་གནས་ལུགས་བརྗོད་དེ། ནུབ་ཕྱོགས་རྒྱལ་ཁབ་ཁག་ཏུ་ཡོན་ཧྲི་ཁེའི་དགོད་བྲོ་བའི་ཁྱབ་བསྒྲགས་དེ་ནུས་མེད་གཏོང་བར་ཤུགས་རྐྱེན་བཟོས་པ་དང་། 《བོད་སོག་ཕྱིན་ཆད་མཐུན་འགྱུར་གྱི་ཆིངས་ཡིག》འཇོག་པར་ཡང་ནུས་པ་བླ་ལྷག་བཏོན། རྡོ་རྗེ་ཡིབ་བོད་དང་སོག་པོ། བོད་དང་ཨུ་རི་སིའི་བར་ལ་འགྲོ་འདུར་རྒོད་པས་དབྱིན་ཇིས་ཁོང་ཨུ་རི་སིའི་གསང་ཉུལ་དུ་ངོས་འཛིན་གྱིས་ཁོང་གསོད་མཁན་ཡོད་ན་སྡིག་ཡོན་སྤྲད་རྒྱུའི་བརྡ་སྦྱོར་ཡང་སྤེལ། ཨུ་རི་སིའི་གུང་ཁྲན་ཏང་ལ་རྒྱལ་ཁ་ཐོབ་རྗེས་རྡོ་རྗེ་ཡིབ་བཙོན་ལ་བཟུང་ནས་དུས་འགྱང་གི་སྲོག་ཁྲིམས་བཅད་ཀྱང་གྲོགས་པོའི་འབྲེལ་བ་བརྒྱུད་ནས་སྲོག་ཁྲིམས་ལས་ཐར་བར་མ་ཟད་བཙོན་གྲོལ་ཐོབ། རྗེས་སུ་སི་ཏ་ལིན་གྱིས་མི་ཆོས་དང་བྲལ་བའི་སྒེར་གཅོད་དབང་སྒྱུར་བྱེད་སྐབས་ཡང་བསྐྱར་འཛིན་བཟུང་བྱས་ཏེ་བཙོན་ཁང་དུ་སྐུ་གྲོངས། 1990ལོར་ཉེས་མེད་དཀར་འདོན་བྱས། 2015ལོར་ཨ་རི་ནས་པར་སྐྲུན་བྱས་པའི《ལ་རྩེ》དུས་དེབ་ཏུ་རྡོ་རྗེ་ཡིབ་ཀྱི་ཚིགས་བཅད་རང་རྣམ་ཕྱག་བསྟར་མ་ཚོན་ལྡན་འདྲ་པར་གྱི་ལམ་ནས་པར་འདེབས་བྱས་ཡོད་ལ། དེའི་ནང་དུ་སྐབས་དེའི་བོད་ཀྱི་མཐོ་རིམ་མི་སྣ་མང་ཤོས“རྒྱ་ཆེན་ཡུལ་གྲུར་མ་བགྲོད་པས།། མཁྱེན་རྒྱ་ཆེ་ཡང་མཐོང་རྒྱ་ཆུང་།།”ཞེས་བྲིས་ཡོད་པ་ཅིས་མ་བདེན་སྙམ།
ཅིག་ཤོས་ནི་ཕ་ཡུལ་ཉི་ཧོང་ནས་ཡིན་པའི་ཐ་ད་ཐོའུ་ཀང（多田等观，1890—1967）སྟེ་ཚ་སྤྲུལ་རིན་པོ་ཆེས་ཉི་ཧོང་དུ་སློབ་གཉེར་གནང་སྐབས་ཀྱི་ཉི་ཡིག་དགེ་རྒན་དེ་ཡིན། ཁོང་གིས་ཀྱང་ཚ་སྤྲུལ་གྱི་མདུན་དུ་བོད་ཡིག་ཐོག་མར་སྦྱངས། 1912ལོར་རྒྱ་གར་དུ་རྒྱལ་དབང་བཅུ་གསུམ་པ་དང་མཇལ་ཅིང་། རྒྱལ་དབང་རིན་པོ་ཆེས་ཁོང་ལ“ཐུབ་བསྟན་རྒྱལ་མཚན”ཞེས་པའི་བོད་མིང་གནང་ཞིང་རྗེས་སུ་བོད་ལ་ཡོང་ཆོག་པའི་ལམ་ཡིག་ཀྱང་གནང་བས། 1913ལོར་རྒྱ་གར་དང་འབྲུག་ཡུལ་བརྒྱུད་དེ་ལྷ་སར་སླེབས་ནས་སེ་ར་དགོན་དུ་ལོ་ངོ་བཅུའི་རིང་སློབ་གཉེར་བྱས། དུས་མཚུངས་སུ་རྒྱལ་དབང་མཆོག་ལ་བོད་ཀྱི་ཁྲལ་སྡུད་ལམ་ལུགས་བསྒྱུར་བཅོས་བྱེད་པ་དང་། གསེར་གྱི་དངུལ་ལོར་བཟོ་བ། མཚོན་ཆ་ཉོ་སྒྲིག་བྱེད་པ། དུས་ལྟར་རྒྱལ་སྤྱིའི་གནས་ཚུལ་ཞུ་བ（ཁོ་པར་ཟླ་བ་གཉིས་གསུམ་རེའི་ནང་ལ་ཕ་ཡུལ་ནས་འཚོ་བའི་མཁོ་ཆས་དང་ཚགས་ཤོག་སོགས་བསྐུར་ཡོང་གིན་འདུག）སོགས་སྲི་ཞུ་བླ་མེད་བསྒྲུབས་པས། རྒྱལ་དབང་ཚང་ཡང་ཁོ་པར་ཧ་ཅང་ཐུགས་འདང་ཆེ་ཞིང་། ཕྱིར་ལོག་དུས་བཀའ་བསྟན་གཙོར་བྱས་པའི་གླེགས་བམ་དང་ལེགས་སྐྱེས་རྩ་ཆེན་གང་མང་གནང་བ་བཅས་རྒྱ་ཡིག་ཏུ་བསྒྱུར་ནས་པར་སྐྲུན་བྱས་ཡོད་པའི་ཁོང་གི《བོད་ནང་འཛུལ་བའི་ཟིན་ཐོ》（《入藏行记》）ཞེས་པ་སོགས་སུ་གསལ་ལ། ཕ་ཡུལ་དུ་ལོག་རྗེས་བཀའ་བསྟན་གྱི་དཀར་ཆག་བསྒྲིགས་པ་དང་བོད་ཡིག་གི་སློབ་མ་མང་དུ་བསྐྱངས་པའི་ཁྲོད་ཀྱི་འགའ་རེའི་སློབ་མ་དང་བུ་ཕྲུག་གིས་ད་ལྟ་ཉི་ཧོང་དུ་བོད་ཀྱི་ནང་ཆོས་དང་ལོ་རྒྱུས། རྩོམ་རིག་སོགས་ཉི་ཡིག་ཏུ་བསྒྱུར་པ་དང་། ཚིག་མཛོད་བསྒྲིགས་པ། དུས་དེབ་དཔར་བ་སོགས་བོད་ཀྱི་རིག་གནས་འཛིན་སྐྱོང་སྤེལ་གསུམ་ལ་གཞོལ་བཞིན་པ་དེ་རེད།
ཟངས་རི་མཁར་དམར་དང་འོལ་ཁ་ཆོས་ལུང་སོགས་ལ་ལོ་ངོ་བདུན་བརྒྱད་ཙམ་གྱི་ཡར་སྔོན་དུ་བླ་མ་ལཱ་ཁ་རིན་པོ་ཆེ་ཚང་དབུས་མཇལ་གནང་སྐབས། ང་རང་ལཱ་ཁ་དགོན་པའི་ལས་སྣེ་ཁག་དང་ངེད་ཡུལ་ཚོའི་གཞི་རིམ་ལས་བྱེད་རྣམས་དང་མཉམ་དུ་སྐུ་རྗེས་འབྲངས་མྱོང་ནའང་། བློ་ཡུལ་ལས་འདས་པ་ཞིག་ལ་ད་ཐེངས་སྔོན་ཆད་སོང་མ་མྱོང་བའི་ལམ་མ་ནོར། སྔོན་ཆད་སོང་མྱོང་བའི་ལམ་ནོར་ཏེ་ཡང་ཡང་འདྲི་དགོས་བྱུང་། ཡིད་འགུལ་ཐེབས་པ་ཞིག་ནི་ལམ་འགྲམ་གྱི་སྐྱེ་བོ་ཕོ་མོ་རྒན་གཞོན་སུ་ལ་དྲིས་ནའང་ངང་རྒྱུད་འཇམ་ཞིང་སུན་སྣང་མེད་པར་ལམ་སྟོན་གསལ་པོར་གནང་བས། ལམ་ལོག་ཆེར་འགྲོ་མ་དགོས་པར་འོལ་ཁ་ཆོས་ལུང་དུ་སླེབས། འདི་རུ་མགྲོན་ཁང་ཐོག་བརྩེགས་ཅན་གཉིས་བཟོས་ཡོད་པ་ལས་དགོན་པ་ལ་འགྱུར་ལྡོག་ཆེར་མེད་ལ། འགྱུར་ལྡོག་ཡོད་པ་ནི་དེ་དུས་ང་ཚོ་མཉམ་དུ་སོང་བའི་གཞི་རིམ་ལས་བྱེད་མང་ཤོས་ཀྱིས་ད་ལྟ་ངལ་གསོ་བྱས་ཟིན་པ་དང་འགའ་རེ་ཚེ་ལས་འདས་ཟིན་པ། གྲྭ་བ་འགའ་རེ་གྲྭ་ལོག་བྱས་ཟིན་པ་དེ་རེད།
ལྷ་སར་ལོག་རྗེས་ཆུང་འདྲིས་གྲོགས་པོ་ཞིག་གིས་རྒྱ་གར་ལུགས་ཀྱི་ཟ་ཁང་དུ་སྣེ་ལེན་གནང་བར་མ་ཟད་དངོས་སུ་རླངས་འཁོར་བསྐོར་ཏེ་ང་རང་ཡང་ཡང་འགྲོ་བསམས་ཀྱང་འགྲོ་མ་ཐུབ་པའི་བོད་ལྗོངས་ནོར་ནག་འགྲེམ་སྟོན་ཁང་ལ་དབུས་རོགས་དང་བཅས་ལྟ་སྐོར་དུ་ཁྲིད། ད་དུང་མ་བསམས་ས་ནས་གྲོགས་པོའི་ཁ་ནས་ནོར་ནག་འགྲེམ་སྟོན་ཁང་གི་ཉེ་འགྲམ་ན་ཁྲི་ལྡེ་སྲོང་བཙན་སྐབས་སུ་བཞེངས་པའི་སྐར་ཆུང་དགོན་པའང་ཡོད་པ་ཤེས་ཏེ་སྤྲོ་བ་རབ་ཏུ་འཕེལ། དེང་རབས་ཀྱི་ཁང་བརྩེགས་མཐོན་པོ་དག་གི་ཁྲོད་ན་ཧ་ཅང་མངོན་ཆུང་གི་ཚུལ་གྱིས་གནས་ཤིང་། འབག་སྐུ་གསར་བ་འགའ་མཆོད་པ་དང་བཅས་ལྷ་ཁང་དུ་བཞུགས། ལོ་ངོ་ཆིག་སྟོང་ལྷག་གི་ལོ་རྒྱུས་ཡོད་པའི་སྐར་ཆུང་རྡོ་རིང་ཞེས་པའི་སྐད་གྲགས་ཅན་དེ་སྐེད་ནས་བཅག་སྟེ་སྒོ་རུ་གཡུགས་འདུག་པའི་ངོས་ཀྱི་ཡི་གེ་ངོས་ཟིན་དཀའ་ནའང་དགེ་ཆོས་ཀྱི་ངོ་བཤུས་སུ“འཕྲུལ་གྱྀ་ལྷ་བཙན་པོ་ཁྲི་ལྡེ་སྲོང་བརྩན་གྱྀ་རིང་ལ།།དམ་པའྀ་ཆོས་ཡུན་དུ་བརྟན་པར་གཙྀགས་བརྣན་པ།། འཕྲུལ་གྱྀ་ལྷ་བཙན་པོ། མྱེས། ཁྲི་སྲོང་བརྩན་གྱྀ་རིང་ལ། སངས་རྒྱས་ཀྱྀ་ཆོས་མཛད་དེ།། ར་སའི་གཙུག་ལག་ཁང་ལས་སྩོགས་པ་བརྩགས་ཤིང་།། དཀོན་མཆོག་གསུམ་གྱྀ་རྟེན་བཙུགས་པ་དང་།……”ཞེས་པའི་བསམ་གཞིག་གི་རྟ་ཕོ་ལོ་ངོ་ཆིག་སྟོང་ལྷག་གི་ཡར་སྔོན་དུ་འདྲེན་པར་བྱེད་པའི་ཚིག་དུམ་འདིས་མཇུག་བསྡུ་བར་བྱའོ།། 2018ལོའི་ཟླ་ལྔ་བར།
