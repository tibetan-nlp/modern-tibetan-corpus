ཡེ་ཤུ་རྒད་པོ་བོད་ལ་སླེབས་པ།
ཨ་སྒང་ཡར་རྒྱས།

ཡེ་ཤུ་རྒད་པོ་བོད་ལ་སླེབས་སོང་། སྤྱིར་ཡེ་ཤུ་བོད་ལ་ཕེབས་དོན་འདི་གསར་འགྱུར་གློ་བུར་བ་ཞིག་ཡེ་ནས་མ་རེད། ། འདི་ནས་ཁྱེད་ལ་གཏམ་རྒྱུད་ཅིག་འཆད་འོང་། བོད་བཙན་པོ་སིལ་བུར་འཐོར་ཕྱིན་ཆོས་རྒྱལ་གྱི་གདུང་རྒྱུད་རིམ་པས་གུ་གེ་རུ་སྲིད་དབང་བཟུང་།ཧི་མ་ལ་ཡའི་རི་རྒྱུད་དང་གངས་རིན་པོ་ཆེའི་དཀྱིལ་དབུས་སུ་འཛམ་གླིང་ཐོག་ཆེས་མཐོ་བའི་སྲིད་ཀྱི་བ་དན་འཕྱར། རྒྱལ་རབས་འདི་མི་ལོ་བདུན་བརྒྱ་དང་རྒྱལ་པོ་༣༥ཡིས་རིམ་པར་བསྐྱངས། ཆོས་འབྱུང་དུ་འདི་ལ་བསྟན་པ་ཕྱི་དར་གྱི་དུས་སྐབས་བླ་ན་མེད་པར་བསྔགས། འོན་ཀྱང་འདི་འདྲའི་རྒྱལ་ཕྲན་ཞིག་གནམ་སའི་བར་ནས་བྱེ་རྡུལ་རླུང་གིས་སྤུར་སྤུར་དང་།སྨུག་བུན་ཉི་མས་དྭངས་དྭངས་བྱས་ཏེ་གློ་བུར་བོར་བརླགས་སོང་བ་རེད། ཤུལ་དུ་ལུས་པ་ནི་འཇིག་རྟེན་ལ་མ་རངས་སྡུག་ཡོས་འབོད་གིན་འདྲའི་མཁར་ཤུལ་འཐོར་རོ་སྐྱ་བེ་ལེ་ཞིག་དང་། བདེན་ལུགས་ཀྱིས་བཀྲེས་དེ་ལྟོགས་ཤུ་ཐེབས་སོང་འདྲའི་ས་ཕུག་ཁ་ཆེན་གདངས་པ་འགའ་ལས་ཅི་ཡང་ཡོད་མ་སོང་།། གུ་གེ་རྒྱལ་རབས་ཀྱི་སྟོབས་འབྱོར་མངའ་ཐང་། གུ་གེ་རྒྱལ་རབས་ཀྱི་སྐྱེ་རྒ་ན་འཆི། ད་དུང་ཡང་མངའ་རིས་ཀྱི་གད་པ་སྐྱ་མོ་དག་གིས་གྱེར་བའི་རླུང་སྒྲ་ཇི་བཞིན་དེ་ཡི་གསང་བ་རྟོགས་འགྲོ་མཁན་ཧ་ལམ་མེད་ཚོད་རེད། རེད་དེ། ཁྱེད་ཀྱང་གུ་གེའི་མཁར་ཆག་ལས་བཙན་པོའི་འདས་གྲོངས་ལ་ཐུགས་ཁྲལ་བཞེས་མཁན་དེ་ཡིན་ན། དུད་དྲེག་གིས་ཟིན་པའི་ལྡེབས་རིས་གོག་པོ་དང་བཟོ་ལུས་ཡ་མཚན་གྱི་ལྷ་རིས་ཁྲོ་བོ་དག་གི་ལྟག་རྒྱབ་ན་གསང་བ་གང་ཡོད་ཤེས་འདོད་སྐྱེ་མཁན་ཞིག་ཡིན་ན། ངས་གཏམ་རྒྱུད་གྱི་ནང་དོན་མུ་མཐུད་བཤད་པར་བྱའོ།། “ཡུལ་འདིའི་རྒྱལ་པོ་ནི་སྟོན་པ་ཡེ་ཤུའི་གསུང་ལ་ཐུགས་རབ་ཏུ་མཉེས། གུས་པ་དག་གིས་རྒྱལ་ཕྲན་འདི་ནས་ཡེ་ཤུའི་ཀྱི་གསུང་རབ་སྒྲོག་པ་ལ་མཐུན་རྐྱེན་གང་ལེགས་སྦྱར། སངས་རྒྱས་ལ་དད་པ་འཆང་བ་དག་ནི་མ་རངས་བའི་ཚུལ་དུ་ཡོད་ཀྱང་། རྒྱལ་པོ་ཡི་ཁྲིམས་ཆེན་པོས་ང་ཚོ་གཅེས་ཤིང་བསྲུངས།”འདི་ནི་སྤྱི་ལོ་ཆིག་སྟོང་དྲུག་བརྒྱ་བཞི་བཅུའི་ནང་ཞིག་ལ། ཨན་དོ་ངྷི་ཟེར་བའི་ཡེ་ཤུའི་ཆོས་སྤེལ་བ་ཞིག་གིས་གུ་གེའི་མཁར་གྱི་རྒྱ་ཕིབས་གསེར་ལམ་མེ་བ་དེའི་འོག་ཏུ་ཐག་རིང་གི་རོ་མ་རྒྱལ་ཁབ་ཀྱི་ཡེ་ཤུའི་ཆོས་ཚོགས་ཀྱི་འགོ་གཙོ་ལ་ཕུལ་བའི་འཕྲིན་ཡིག་ཞིག་གི་ནང་དོན་ཆ་ཤས་རེད། དེ་ན་ངས་ཨན་དོ་ངྷི་ཟེར་བའི་རྒད་པོ་འདིེའི་བྱུང་བ་ངོ་སྤྲོད་ཙམ་བྱེད། ། ཨན་དོ་ངྷི། ཁོ་ནི་བོད་པ་ཚོས་སྤྱན་རིས་གཟིགས་ཀྱི་གདུལ་ཞིང་དུ་ངོས་འཛིན་པའི་གངས་ཀྱི་ལུང་པ་འདིར་ཐོག་དང་པོར་དབྱུག་པ་རྒྱ་གྲམ་ཅན་ཞིག་འཁུར་ནས་ཐོན་འོང་མཁན་དེ་རེད།1550ལོར་ཕོར་ཐུ་གལ་གྱིས་རྒྱ་གར་གྱི་གནས་གོ་ཝ་བཙན་བཟུང་བྱས་ཤིང་། ཆོས་དཔོན་ཁུ་ལི་མིན་གྱིས་གོ་ཝ་ཆོས་ཚོགས་དབུ་བརྙེས་པ་ཁྱབ་སྒྲོག་མཛད། རྗེས་ཀྱི་1579ལོའི་ཟླ་11པར་ཕོར་ཐུ་གལ་གོ་ཝ་ཆོས་ཚོགས་ཧོར་གྱི་ཨ་ཁུ་བྷ་རྒྱལ་པོས་གདན་ཞུས་ལྟར་ཡེ་ཆོས་སྤེལ་བར་ཆིབས་སྒྱུར་གནང་། དེ་ནས་བཟུང་། ཡེ་ཤུའི་རྗེས་འཇུག་པ་དག་གིས་ཐད་གར་རྒྱ་བོད་ས་མཚམས་བརྒལ་དེ་ཡེ་ཤུའི་གདུལ་བྱ་བོད་དུ་སྐྱོང་བའི་ལས་ཀྱི་སྒོ་འཕར་དང་པོ་ཕྱེ་བཞག་པ་རེད།། 1642.3.20ཉིན་འདི་ལ། རྒྱ་གར་ནས་རྐང་པ་རྒྱབ་ཏེ་ཐོན་པའི་གངས་སྐོར་བ་རུ་འཕྲེང་ཞིག་གངས་རིན་པོ་ཆེའི་འདབས་ན་ཐང་ཆད་པའི་ཉམས་ཀྱིས་ག་ལེར་མདུན་དུ་སྐྱོད། དེའི་ཁྲོད་དུ་གསང་བའི་བྱ་བཞག་གཞན་ཞིག་ལ་ཧམ་པ་འཆང་དེ་ཡོང་མཁན་མི་འགའ་ཡོད་པ་སུས་ཀྱང་ཤེས་གི་མ་རེད། མི་དེ་དག་གིས་ལྭ་བའི་མཁྲིག་མར་དམ་པའིི་གསུང་རབ་ཆ་ཚང་རེ་ཉར་ཞིང་མགུལ་དུ་རྒྱ་གྲམ་དབྱིབས་ཀྱི་འདོགས་ཆ་རེ་རེ་བསྒྲོན་ཡོད། འདི་ནི་ཁོང་ཚོ་ཧི་མ་ལའི་རི་འདབས་ལ་བཅར་དོན་ངོ་མ་དེ་ཡིན། འདི་ནི་ཁོང་ཚོས་ལམ་གྱི་དཀའ་སྡུག་ལ་གདོང་ལེན་ཐུབ་པའི་རྒྱུ་རྐྱེན་ཆ་ཚང་དེ་ཡིན། དེ་ལྟར། ལོ་དེའི་ཟླ་བ་བརྒྱད་པའི་ཉིན་ཞིག་ལ། མི་དེ་ཚོའི་གདོང་བ་འགེབས་བྱེད་ཀྱི་རས་ལྷེ་ཡར་བཀྱགས་ནས་མགོ་བོ་གྱེན་ལ་ལྟ་དུས། ཁོང་ཚོ་གཞི་ནས་བཟོ་བཀོད་ངོ་མཚར་བའི་ཕོ་བྲང་རི་ལ་བརྩེགས་པ་ཞིག་གི་འདབས་ཁར་སླེབས་འདུག དེ་ནི་གུ་གེ་རེད། ད་ནི་ཁྱེད་ཀྱིས་མི་དེ་དག་སུ་ཡིན་ཧ་ལམ་རྟོགས་ཡོད་ངེས། རེད་དེ། དེ་ནི་ཆོས་དཔོན་ཨན་དོ་ངྷི་ཡིས་སྣེ་དྲངས་པའི་ཡེ་ཆོས་སྤེལ་མཁན་སྐོར་ཞིག་རེད། རྗེས་སུ་གུ་གེའི་མཚར་སྡུག་ནག་བསུབ་གཏོང་བར་ཐད་ཀའི་རྒྱུ་རྐྱེན་འདྲེན་ཡོང་བའི་ཕྱི་རྒྱལ་མགོ་སེར་འགའ་རེད། སྐབས་དེའི་གུ་གེ་རྒྱལ་པོས་ཐོག་མ་ནས་ཕྱི་འོང་མགྲོན་པོ་འདི་དག་ལ་སྣེ་ལེན་གཟབ་རྒྱས་དང་དགའ་གུས་དཔག་མེད་བྱས་ཤིང་། ཁོ་པ་ཅག་གིས་ཁྱེར་འོང་བའི་ཡེ་ཤུར་འབོད་པའི་དམ་པ་གང་དེའི་གསུང་རབ་ལ་ཡང་ཐུགས་མཉེས་པོ་བྱུང་། དེ་ལྟར། སྤྱིར་བཏང་གི་ཆོས་ལུགས་སྤེལ་སྟངས་ལྟར་ཐོག་མར་མི་སེར་ཕལ་བའི་སེམས་རྒྱུད་བརླབས་ནས་རིམ་པར་གོང་རིམ་ལ་བཅར་བ་དང་མི་འདྲ་བར། ཆོས་སྤེལ་གྱི་ཐེམ་པ་དང་པོ་རྒྱལ་པོའི་སྐུ་དྲུང་ནས་མགོ་བཙུགས་ཤིང་། ཡེ་ཤུ་པ་ཚོའི་མི་སེམས་ལ་ཉེ་བའི་གསུང་རབ་སྒྲོག་ལུགས་དང་ཕལ་བའི་འཚོ་བ་དང་འབྲེལ་བའི་ཆོས་སྐད་ཀྱིས་ནི་སྲིད་ཀྱི་ཀ་བར་ཆགས་པའི་རྒྱལ་པོ་ཁོང་ལ་ཡང་ཡེ་ཤུ་པ་ཚོ་བདེན་བདེན་འདྲ་བའི་སེམས་ཤེས་ཆགས། ལྷག་པར་ཡེ་ཤུའི་ཆོས་གཞུང་གི་འཕགས་མ་མྰ་ལི་ལ་བུ་ཡེ་ཤུ་འཁྲུངས་བའི་དཀའ་སྡུག་གི་གཏམ་རྒྱུད་གསོན་པོའི་རྒྱལ་ཡུམ་དང་བཅས་པ་ཡིད་འགུལ་ཐེབས་ཏེ་མིག་ནས་མཆི་མ་དབང་མེད་དུ་བཞུར་ཞིང་། ཐ་ན་སྐབས་དེར་ནང་བའི་ཆོས་སྒོར་ཞུགས་ཟིན་པའི་རྒྱལ་པོའི་ཁུ་བོ་དང་ནུ་བོ་གཉིས་ཀྱིས་ཀྱང་ཕྱི་རོལ་བའི་ཆོས་འདིར་འགོག་རྒོལ་ཅུང་ཙམ་མ་བྱས། དེ་ལྟར་དཀའ་ལས་ཁག་པོ་གང་ཡང་མ་མྱང་བར་ཨན་དོ་ངྷིའི་ཐོག་མའི་ཆོས་སྤེལ་གྱི་འཆར་གཞི་ཡོངས་སུ་འགྲུབ་ཞིང་།སེམས་སུ་མི་འགྱངས་བར་བོད་ཀྱི་ཡུལ་འདི་ནི་དཀོན་མཆོག་ཡེ་ཤུ་ལ་སྐྱབས་སུ་འགྲོ་བའི་མི་རིགས་ཤིག་ཏུ་དན་དན་འགྱུར་ཐུབ་ངེས་རེད་སྙམ། ཨན་དོ་ངྷིས་ད་དུང་བོད་ཡིག་ལ་བརྩོན་པ་ཧུར་གྱིས་འཇུས་ཏེ་ཟླ་འགའི་རིང་བོད་ཡིག་འབྲི་ཀློག་གི་རྨང་གཞི་ཡོད་པར་བྱས་ཤིང་། ག་ལེ་ག་ལེར་ཡེ་ཆོས་བོད་ཡིག་གིས་བྲིས་དེ་མི་སེར་ཕལ་བའི་ཁྲོད་དང་ནང་པའི་དགོན་པ་སོགས་སུ་བཀྲམས། ། རྒྱལ་པོའི་བཀའ་དང་ལེན་ཀྱིས་མངའ་འབངས་སྐྱ་སེར་ཀུན་གྱིས་གུ་གེ་རུ་ཡེ་ཤེས་ཆོས་ཁང་བཞེངས་ཤིང་། མིང་ལ་འཕགས་མའི་སྨོན་ལམ་ཞེས་སུ་བཏགས། འོན་ཀྱང་སྐབས་དེའི་གུ་གེ་ནི་སྟོབས་འབྱོར་ཆེས་ཆེ་བའི་དུས་ཡུན་ལས་ཡོལ་ཏེ་རླུང་རྟ་མར་རྒུད་ཀྱི་དུས་སུ་ཡོད་ཅིང་། ལྷག་པར་ཏུ་བྱང་ཕྱོགས་ཀྱི་ལ་དྭགས་བར་དུས་རྒྱུན་དམག་འཁྲུག་བྱུང་བས་སྟོབས་ཤུགས་ཉམས་རུད་དུ་གྱུར་ཡོད། མངའ་ཞབས་ཀྱི་ན་གཞོན་མང་ཤོས་ནི་ནང་པའི་དགོན་པར་གྲྭ་བར་ཕྱིན་པས་དམག་གིས་མི་འདང་བར་གྱུར་ཞིང་། དེ་འདྲའི་གནས་བབས་སྐྱོ་བོའི་འོག་རྒྱལ་པོ་ལྷག་ཏུ་ཡེ་ཆོས་ལ་མོས་པར་གྱུར། ཁོང་གིས་བཙན་ཐབས་ཀྱིས་གྲྭ་བ་ཕལ་མོ་ཆེ་སྐྱ་སར་ཕབ་པས། ནང་པའི་སྡེ་ལ་གློ་བུར་སྐྱོམ་འགུལ་ཚབ་ཆེན་བཟོས་ཤིང་། ཡེ་ཆོས་ཀྱི་ཤུགས་རྐྱེན་འོག་ནང་པ་རང་གི་གནས་སྟངས་ལ་ཉམས་ཉེས་ཆེན་པོ་ཤོར་གིན་པ་ཐོག་དང་པོར་ཚོར། མ་གཞི་རྒྱལ་པོའི་ཁུ་བོ་དང་ནུ་བོ་ཚང་མ་ནང་པའི་གྲྭ་བ་ཡིན་ཁར་ནང་པའི་སྟོབས་ཤུགས་ད་དུང་ཡང་ཧ་ཅང་ཆེན་པོ་ཡོད་ཅིང་། རྒྱལ་ཁབ་ཀྱི་དབང་འཛིན་པ་མང་ཤས་སེར་མོ་བ་ཡིན་པས་ཆོས་སྲིད་ཟུང་འབྲེལ་གྱི་ལམ་ཕྱོགས་འདིས་ཕྱོགས་གཉིས་ཀར་འཇིགས་སྣང་བསླངས། དེས་ཕྱིར་རྒྱལ་པོས་ཡེ་ཆོས་ཀྱི་སྟོབས་ཤུགས་སྤྱད་དེ་ནང་པའི་སྟོབས་ཤུགས་ཤུལ་མེད་དུ་བརླག་པའི་བསམ་བློ་བཅིངས་ཤིང། ཁོང་གིས་བཀའ་རྒྱ་བཏང་ནས་འབངས་མི་སྐྱ་སེར་རེ་རེས་ཡེ་ཆོས་སློབ་སྦྱོང་དགོས་པ་གཏན་ལ་ཕབ་པ་དང་ཆོས་སྤེལ་མཁན་ལ་བཀག་འགོག་ངོ་ལོག་གང་ཡང་བྱས་མི་ཆོག་པར་ཁྲིམས་བཅས། རྒྱལ་པོ་རང་གིས་ཀྱང་ཉིན་རེ་རེར་དམ་པའི་གསུང་རབ་ལ་ཐོས་བསམ་བྱས་པ་མ་ཟད། ཐ་ན་དཀོན་མཆོག་གི་བླ་ན་མེད་པའི་གཟུགས་འདས་ཀྱི་སྟོབས་ཤུགས་དག་ལ་ཡིད་ཆེས་ཤིང་རྨོངས་དད་སྐྱེ་འགོ་བརྩམས། རྒྱལ་པོས་ཡེ་ཆོས་ཀྱི་གནས་ལུགས་ཁྱེར་ནས་ནང་པ་དག་ལ་ངན་མཚང་སྣ་དགུ་བཏོན་ཞིང་། དམག་རྒྱག་ཁར་ནང་པ་ཚོས་བཅའ་བ་རིམ་གྲོ་སྒྲུབ་ལུགས་ཀྱང་ག་ལེར་ཡེ་ཤེས་ཆོས་པ་དག་གི་ལུགས་ལྟར་བྱས། ད་དུང་རྒྱལ་ཡུམ་གྱི་ཉེ་གནས་ཤིག་གི་ཆང་ས་རྒྱག་ལུགས་ཀྱང་ཡེ་ཤུ་པ་ཚོའི་ལུགས་ལྟར་བྱས་སྐད། དེ་འདྲའི་མྱོ་ཧམ་ཆེན་པོས་ཡེ་ཆོས་ལ་དགའ་གུས་བྱེད་སྟངས་འདིས་རྒྱལ་པོའི་ཁུ་བོ་དང་ནུ་བོ་གཉིས་ཧ་ཅང་ཡི་མུག་པར་གྱུར། ཁོང་གཉིས་ཀྱིས་རྒྱལ་པོ་སྲས་བཅས་དགོན་པར་གདན་དྲངས་ནས་བླ་མ་གང་མཁས་ཕྱན་དྲངས་ཏེ་ཟླ་བ་གཉིས་རིང་ཆོས་གསུང་དུ་བཅུགས་ཤིང་། ནང་བའི་བསྟན་པ་ལ་ཅིས་ཀྱང་སྐྱབས་སུ་འགྲོ་དགོས་ལུགས་ཀྱི་དྲན་སྐུལ་ཡང་ཡང་ཞུས། གནས་ཚུལ་དེས་ཡང་ཨན་དོ་ངྷིའི་སེམས་མི་བདེ་བར་བཟོས། ཁོས་ཐབས་ཤེས་སྣ་ཚོགས་ཀྱིས་རྒྱལ་པོར་བཅར་ཐབས་བྱས་ཤིང་། ཁུ་བོ་དང་ནུ་བོའི་སྦྱོར་བ་ཐམས་ཅད་ནི་གདོན་འདྲས་བྱས་པ་རང་ཡིན་པས་དཀོན་མཆོག་ལ་དད་པའི་བློ་ལ་ཅིས་ཀྱང་མི་གཡོ་བ་ཞིག་དགོས་ཞེས་གདམས། རྒྱལ་པོས་ཀྱང་ཨན་དོ་ངྷིི་དང་གྲོས་འཆམ་མེད་ཐོག་གནད་དོན་གང་ཡང་རང་དགར་ཐག་གིས་མི་གཅོད་པར་ཁག་ཐེག་བྱས། ངོ་བོའི་ཐོག་ནས་ཨན་དོ་ངྷི་ནི་སྲིད་ཀྱི་བློ་འདྲི་ཞུ་ས་ལྟ་བུར་གྱུར།། དེ་ལྟར། ཡུལ་བྱེའུ་ཡི་ས་རི་བྱེའུ་ཡིས་ཕུད་པའི་སྡུག་བསྔལ་དེ་རྒྱལ་པོའི་ཁུ་བོ་དང་ནུ་བོའི་མཚོན་པའི་ནང་པའི་སྡེ་ལ་བྱུང་ཞིང་། ཡི་ཆད་དང་ཁོང་ཁྲོ། ངོ་ལོག་གི་བསམ་བློ་ཞིག་ག་ལེ་ག་ལེར་ནང་པའི་སྡེ་ནས་རྒྱ་ཁྱབ་དུ་འཕེལ། ཉེན་ཁ་དེ་རྒྱལ་པོས་ཀྱང་མཐོང་ཆོས་སུ་གྱུར་བ་ལོས་ཡིན། ཁོང་གིས་དེ་མུར་རང་གི་ཁུ་བོ་དང་བཅས་པའི་སེར་མོ་པའི་དབང་ཤུགས་དང་མངའ་འབངས་མ་ལུས་གཞུང་བསྡུ་བྱས་ཤིང་། སེར་བོ་པའི་ཁ་གྲངས་ཀྱང་སྔ་མའི་སྟོང་ཕྲག་དྲུག་ནས་བརྒྱ་ཁ་ལོངས་ཙམ་ཞིག་ལ་ཉུང་དུ་བཏང་།། འོན་ཀྱང་། དེ་ཡིས་ནི་གནད་དོན་ཐག་གཅོད་ཡོང་རྒྱུ་ལྟ་ཞོག གནས་སྟངས་སྔར་བས་ཉེན་ཚབ་ཅན་དུ་བསྒྱུར།། དེ་ཡང་ཨན་དོ་ངྷི་ཕྱིར་རྒྱ་གར་གོ་ཝ་ལ་སོང་ཤུལ་དང་། རྒྱལ་པོ་སྐུ་སྙུངས་ནས་མལ་སར་ལྷུང་པའི་སྐབས་གླགས་ཤིག་ལ་གཞི་ནས་གུ་གེ་མཁར་འདབས་ཀྱི་གྲོང་ཁྱེར་ནས་ཀི་སྒྲ་དང་དུད་སྨུག་གིས་གནམ་ས་གཡོ་བར་བརྩམས། དེ་ནི་རྒྱལ་པོ་ཁོང་གིས་ཆེས་ཐོས་ན་མི་འདོད་པའི་འབངས་གྱེན་ལོག་གི་ངར་སྐད་རེད། གྱེན་ལོག་གི་མེ་རོ་ཐོག་མ་དེ་མཐའི་མཁར་གྲོང་ནས་མཆེད་འགོ་ཚུགས། ས་གནས་དཔོན་པོ་དང་སེར་མོ་བ་དག་གིས་རང་གི་ནུས་པ་དང་ཤུགས་རྐྱེན་ལ་བརྟེན་ནས་དམག་གྲ་སྒྲིགས་ཤིང་། དྲག་པོའི་ངོ་ལོག་གི་རྒོལ་འཛིང་ཐུ་མོ་དེས་གུ་གེའི་ཕོ་བྲང་ཡང་ཉིན་ཤས་དུད་ལོང་གིས་གཡོགས་སོང་། ལ་དྭགས་རྒྱལ་ཕྲན་གྱིས་གོ་སྐབས་དེ་ལ་བརྟེན་ནས་མྱུར་མོར་གུ་གེ་ལ་དམག་དྲངས་ཤིང་། ཕོ་བྲང་མཐའ་ནས་བསྐོར་བའི་གྱེན་ལོག་དམག་དང་མཉམ་པོར་ཤི་གོད་ཆེན་པོ་ཉོ་མ་དགོས་པར་གུ་གེ་ཆམ་ལ་ཕབ་སོང་། ལ་དྭགས་དམག་གིས་གུ་གེའི་མཁར་ནང་གི་སྲིད་ན་དཀོན་པའི་རིན་པོ་ཆེའི་རིགས་རྣམས་འཁུར་ཞོར་འདྲུད་ཞོར་གྱིས་བུད་སོང་ལ། ལྕགས་རི་སྟོབས་ཀྱིས་བསྒྱེལ། ལྡེབས་རིས་གྲི་ཡིས་གཤགས། ཁང་བཟང་མེ་ལ་བསྒོས། རྒྱལ་པོ་ཁྲིམས་ལ་དེད་སོང་། དེ་ལྟར། དམག་རྟའི་གོམ་སྒྲ་འཇགས་ཤིང་།རྡུལ་འཚུབ་ཁད་ཀྱིས་དྭངས་དུས། ཡེ་ཤུའི་ཆོས་པ་དག་ཀྱང་ཤུལ་མེད་དུ་བཟོས་སོང་མོད། དེ་ལས་ལྷག་པའི་རྒྱལ་རབས་གསེར་གྱི་རི་མོ་ཞིག་གནམ་སའི་བར་ནས་ཧྲིལ་གྱིས་ཕྱག་སོང་། འཇིག་རྟེན་སྐྱོབ་པའི་མགོན་དུ་བསྔགས་པའི་ཡེ་ཤུའི་རྒད་པོ་དེས། གུ་གེའི་མཁར་འདབས་ནས་ངའི་མེས་དབོན་གྱི་བླ་རུས་བཅགས་རྗེས། སླར་ཡང་ནོར་རྐྱལ་འཁུར་དེ་ཡུལ་ལ་ལོག་སོང་། ། གཏམ་རྒྱུད་འདི་དེ་ལྟར་མཇུག་རྫོགས་དགོས་པ་ཡིན་ནམ། མི་ལོ་བརྒྱ་ཕྲག་ཁ་ཤས་འགོར་བའི་དེ་རིང་། ང་ལ་ད་སོ་མར་ལྷ་ས་ནས་གྲོགས་པོ་ཞིག་གིས་ཡེ་ཤུའི་འཁྲུངས་སྐར་གྱི་ཤིས་ཚིག་ཅིག་སྐུར་ཡོང་། ཡང་ན། གཏམ་རྒྱུད་འདི་ད་དུང་མཇུག་མ་རྫོགས་པ་འདྲ། ། 2013.12.25ཡེ་ཤུའི་འཁྲུངས་སྐར་ཉིན་བྲིས། ཡེ་ཤུའི་རྗེས་འཇུག་པ་ཡོངས་ལ་འཁྲུངས་སྐར་བདེ་ལེགས། Merry Christmas
(མཆོད་མེའི་པར་དབང་གཞན་གྱིས་དགོལ་མི་རུང)

རྩོམ་ཁུངས། ངེད་དྲ་བ།
