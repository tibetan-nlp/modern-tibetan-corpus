འཁྱམ་རུ་གནམ་ལྷ་རྒྱལ།(པེ་ཅིང)


བུད་མེད་ཀྱིས་གཅེས་ཕྲུག་ཁྲིད་ནས་ཕྱི་རོལ་ལ་ཁམས་སངས་གོམ་བསྐྱོད་དུ་ཕྱིན། ཉི་མ་ནུབ་རིར་ཡོལ་ལ་ཉེ། གནམ་གཤིས་ཧ་ཅང་འཇམ། དུས་སྐབས་འདིར། བུད་མེད་དེ་དང་འདྲ་བའི་བུད་མེད་གཞན་དག་གིས་ཀྱང་རང་རང་གི་གཅེས་ཕྲུག་ཁྲིད་དེ་ཁམས་སངས་གོམ་བསྐྱོད་བྱེད་པར་དགའ། གཅེས་ཕྲུག་ཚོ་ཇི་འདྲའི་སྙིང་དུ་སྡུག་ཨང་། མ་གཞི་ཚེ་སྲོག་ནི་ཡིད་དུ་འོང་བ་ཞིག་ཡིན་ལ། ཡུན་གྱིས་ན་སྨིན་ཅིང་ནར་སོན་པའི་གོ་རིམ་ནི་མཛེས་སྡུག་ཅིག་དང་བདེ་སྐྱིད་ཅིག་ཅི་ལ་མིན།

“གལ་སྲིད་ང་འཕྱུག་སོང་ན། འུ་གཉིས་ཀྱིས་ཡང་བསྐྱར་བྱིས་པ་ཞིག་བཙའ།” བུད་མེད་ཀྱིས་དེ་ལྟར་བརྗོད།
ཁྱོ་གས་སླར་ཡང་བུད་མེད་ལ་སྡང་མིག་ཅིག་བསྟན།
སྡང་མིག་དེའི་ནང་དོན་ནི་གཞན་ཞིག་ཡིན་ཏེ། དེའང་བུད་མེད་ཀྱིས་གསལ་ལེར་རྟོགས་ཐུབ། མོས་སྐད་ཆ་དེ་ཐེངས་མང་པོར་བརྗོད་མྱོང་མོད། འོན་ཀྱང་ཁྱོ་ག་བྱིས་པ་བཙའ་རྒྱུར་མོས་འཐད་མིན་པའི་བསམ་ཚུལ་དེར་མོ་ཡང་འཐད་པ་བྱུང་ཡོད། ཡིན་ཡང་། ཁྱོ་གས་ནམ་རྒྱུན་གཅེས་ཕྲུག་ལ་“ཕར་སོང” “གདོན་ཕྲུག” “ལན་ཆགས་འདི”ཞེས་བརྗོད་པའི་སྐབས་ཐོག་དེར། བུད་མེད་ཀྱི་སེམས་པ་འཁྱག་ཤུར་ཤུར་བྱེད་ཅིང་། མོ་ལ་ཡང་བསྐྱར་བྱིས་པ་ཞིག་བཙའ་བའི་འདུན་པ་དྲག་ཏུ་ཁོལ་ཡོང་།
འོན་ཀྱང་། ཁྱོ་ག་ནི་གཅེས་ཕྲུག་ལ་ཇི་ལྟར་མི་དགའ་ཡང་། བུད་མེད་ཀྱིས་དེའི་ཕྱིར་དུ་ཁྱོ་ག་དང་ཁ་འཛིང་ཞེ་འཛིང་བྱེད་མི་སྲིད། སྐབས་རེར་མོས་འདི་ལྟར་གླེན་འདང་ཡང་རྒྱག་སྟེ། གལ་སྲིད་ཁྱོ་ག་རང་ཉིད་ནང་བཞིན་གཅེས་ཕྲུག་ལ་ཧ་ཅང་དགའ་ཞིང་དུང་ན། དེ་ནི་ཁོ་གཉིས་ཀྱི་བདེ་སྐྱིད་ཅིག་ཅི་ལ་མིན་ཨང་།
མཚན་མོ་དེར། ཁྱོ་གས་རང་གི་ལས་ཀ་གྲུབ་རྗེས་སྔ་མོ་ནས་གཉིད་དུ་ཡུར།
བུད་མེད་ཀྱིས་གཅེས་ཕྲུག་པང་དུ་བཟུང་ནས་ཡུན་རིང་ལ་གཉིད་མ་ཁུག མོས་གཅེས་ཕྲུག་གི་མགོ་བོར་བྱིལ་བྱིལ་བྱེད་བཞིན་ཟླ་འགའི་ཡར་སྔོན་དུ་རང་གིས་གཅེས་ཕྲུག་ཁྱེར་ཏེ་དགའ་ ལྷང་ལྷང་གིས་ཕྱིར་ཁྱིམ་དུ་སླེབས་པའི་དོན་དག་དེ་ཝ་ལེར་དྲན་བྱུང་།
“དེ་ཅི་ཞིག་རེད།”ཉིན་དེར་བུད་མེད་ཀྱིས་གཅེས་ཕྲུག་པང་སྟེ་ཁྱིམ་དུ་འབྱོར་མ་ཐག ཁྱོ་གས་མཚར་སྣང་དང་བཅས་དེ་ལྟར་དྲིས།
“ངས་ཉོས་པ་ཡིན།”
“དེ་ཉོས་ནས་ཅི་བྱ།” ཁྱོ་ག་ཁོང་ཁྲོ་ལངས་བྱུང་།
“མི་ཞིག་གིས་འཚོང་གིན་འདུག འདི་དྭ་ཕྲུག་ཅིག་རེད།”
“སྒོར་ག་ཚོད་བྱིན་ཏེ་ཉོས་པ་ཡིན།” ཁྱོ་གས་སྡིགས་མོ་གཏོང་འགོ་བརྩམས་བྱུང་།
བུད་མེད་ཀྱིས་ཁྱི་གུ་ཆུང་ཆུང་དེའི་མགོ་ལ་བྱིལ་བྱིལ་བྱེད་བཞིན་འདི་ལྟར་ལན་བཏབ།
“སྒོར་ལྔ་བཅུ།”
2018/6/1ཉིན་བཟོ་བཅོས་ཐེངས་གཉིས་པ་བརྒྱབ།

རྩོམ་སྒྲིག་པ། གངས་དགའ།
(མཆོད་མེའི་པར་དབང་གཞན་གྱིས་དགོལ་མི་རུང)


རྩོམ་ཁུངས། ངེད་དྲ་བ།
