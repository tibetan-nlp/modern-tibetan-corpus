ཕྱིར་དྲན་ཁྲོད་ཀྱི་མཚར་སྡུག་གི་ཕ་ཡུལ།
དུར་སྲི།

ང་རང་ཆུང་དུས་སུ་དབྱར་གཞུང་ལ་ཧ་ཅང་དགའ།རྒྱུ་མཚན་མདོ་དབུས་མཐོ་སྒང་ལ་མཚོན་ན།དབྱར་དུས་ནི་དུས་ཚིགས་ཀུན་པོ་ལས་ཆེས་ཁྱད་འཕགས་ཀྱི་དུས་ཚིགས་ཤིག་ཡིན། ད་དུང་མེ་ཏོག་མང་པོ་ནམ་མཁའི་སྐར་ཚོགས་མཁའ་ལ་འཆར་བ་ཇི་བཞིན་བཞད་སྲིད་ལ། མེ་ཏོག་མང་པོ་ཚོམ་བུར་ཚོམ་བུར་སྐྱེས་ནས་དེའི་སྟེང་དུ་ཉི་འོད་ཀྱིས་ཟེགས་མ་འཕྲོས་ཚེ། མིག་གིས་ཀྱང་བཟོད་དཀའ་བར་འགྱུར་འགྲོ། ད་དུང་རྫ་མགོ་རི་མགོ་དག་ན་རི་དྭགས་ཤ་བ་དགོ་བ་སོགས་ཁྱུ་ཁྱུ་བྱས་ནས་འགྲོ་བ་དང་། སྤང་ཐང་དང་རི་ཞོལ་དག་ཏུ་ནོར་ནག་དང་། ལུག་ཁྱུ། རྟ་ཕོ་སོགས་བག་ཕེབས་དང་ལྷིང་འཇགས་ངང་དལ་བུར་རྒྱུ་ལ། སྐབས་རེར་ཤིན་ཏུ་ལྷིང་ཞིང་འཇགས་པས། སྒོ་ཕྱུགས་ཀྱིས་རྩྭ་ཟ་བའི་སྒྲ་དང་རླུང་བུ་དལ་གྱི་ལྡང་བའི་སྒྲ་ཡང་ཐོས་ཐུབ།དེ་བས་དབྱར་དུས་ཀྱི་མདོ་དབུས་མཐོ་སྒང་ནི་འོད་སྣང་དང་དྲོད་ཁོལ་ལྡན་པའི་དུས་ཚིགས་ཤིག་ཡིན་ལ། བདེ་སྐྱིད་དང་དགོད་སྒྲའི་ཁེངས་བའི་ཞིང་ཁམས་ཤིག་ཀྱང་ཡིན། ང་རང་ཆུང་དུས་ནས་མེ་ཏོག་ལ་ཧ་ཅང་དགའ། བུ་མོ་ལྷ་སྐྱིད་ཀྱང་མེ་ཏོག་འཐུ་བར་ཧ་ཅང་དགའ། ངེད་ཚང་གི་ལྷས་ར་ནས་ཕར་སོང་ན་ཆུ་བོ་ཆུང་ཆུང་ཞིག་ཡོད།ཆུ་བོ་ཆུང་ཆུང་དེ་བརྒལ་ནས་གོམ་པ་བཅུ་ཕྲག་ལྷག་སོང་ན།དེའུ་འབུར་ཆུང་ངུ་ཞིག་ཡོད།དེའི་འགྲམ་ན་ལྷ་སྐྱིད་ཚང་ཡོད། ལྷ་སྐྱིད་ངེད་ཁག་གཉིས་ནི་ཁྱིམ་མཚེས་རྙིང་བ་ཡིན་པས། རྒྱུན་པར་འབྲེལ་འདྲིས་ཤིན་ཏུ་ཆེ་སྟེ། དེ་རིང་ནོར་ལུག་ཅི་ཞིག་ཤི་ན། ཤ་མཉམ་ཟ་དང་༼རྩྭོ༽མཉམ་ཕོ་ཡིན།དེར་བསྟུན་ལོ་རེའི་དབྱར་དུས་རེ་རེ་ལ། ལྷ་སྐྱིད་ངེད་གཉིས་རྩེད་ར་གཅིག་ཏུ་འཛོམས་པ་དང་། སྐབས་རེར་ཆུ་ཉུག་གྲམ་ཉུག་བྱས་ནས་ཉ་འཛིན་པ་དང་། ལན་རེར་རི་སྒང་དག་ཏུ་མེ་ཏོག་འཐུ་བར་འགྲོ་ཞིང་། མེ་ཏོག་གི་ཞྭ་མོ་དང་། མེ་ཏོག་གི་གདུ་བུ། མེ་ཏོག་གི་སྐེ་རྒྱན་སོགས་སླེ་བར་བྱེད་ལ། ཁོ་མོས་མེ་ཏོག་གི་མགུལ་རྒྱན་བཏགས་ནས། ཚེ་རིང་ང་ཨེ་ཡག་ཟེར་བའི་རྣམ་པ་ནི། ད་ལྟའང་ད་དུང་བདག་སེམས་ན་འཇིག་མེད་རྡོ་བརྐོས་རི་མོ་ཇི་བཞིན་གསལ་ལྷང་ངེར་ཡོད། ངེད་གཉིས་ནམ་རྒྱུན་བེའུ་རྫི་ཡིན། ཁོ་མོས་རྒྱུན་དུ་སློག་པ་རྙིང་རུལ་མཐའ་ཤན་གོས་ཅན་དམར་པོས་བརྒྱན་ཡོད་པ་ཞིག་གྱོན་པ་དང་། དེའི་ནང་དུ་ཚི་ལེན་དམར་སྨུག་ཅིག་གྱོན་ཡོད།རྒྱུན་དུ་གདོང་ལ་འཛུམ་གྱིས་ཁེངས་བ་དང་། མིག་ཟུང་ཆུང་ངུ་དེས་བདག་ལ་གཟབ་ནན་གྱིས་བལྟས་ཏེ་དགོད་པར་བྱེད། དེ་དུས་བདག་གིས་འཇིག་རྟེན་སྟེང་ན་ཆེས་མཛེས་སྡུག་ལྡན་པ་ནི།དབྱར་གཞུང་མ་ཡིན་ལ་ཁྲ་ཆིལ་དགུ་ཆིལ་དུ་བཞད་པའི་མེ་ཏོག་འདི་ཚོའང་མ་ཡིན་པར།ཁོ་མོའི་འཛུམ་མདངས་ཁོ་ན་དེ་རེད་སྙམ། དེ་ལྟར་ངེད་གཉིས་དགོད་སྒྲ་དང་འཛུམ་མདངས་ཁྲོད་མཚར་ཞིང་། བདེ་སྐྱིད་དང་མཛེས་སྡུག་ཁྲོད་སྐྱེས། ཡུན་ཐུང་གི་དབྱར་ཟླ་གསུམ་པོ་ནི་དེ་ལྟར་སྤྲོ་སྐྱིད་དང་འཛུམ་མདངས་ཁུར་ནས། མིག་རྡེབ་དབང་ཙམ་ལ་ཚར་སོང་། སང་ཉིན་ནི་ངེད་གཉིས་རང་རང་གི་བེའུ་དང་ལུག་ཁྱུ་དེད་ནས་ལམ་ཁ་གཞན་པ་གཉིས་ལ་བསྙེག་དགོས། ཡིན་ན་ཡང་དེ་རིང་ངེད་གཉིས་སྔར་བཞིན་རི་བོ་མཐོན་པོ་དེའི་ཞོལ་ན་མཉམ་དུ་འཛོམས་ཡོད། ཁོ་མོའི་འཛུམ་མདངས་སྔར་བཞིན་ཤིན་ཏུ་སྙིང་ན་སྡུག ཡིན་ནའང་དེ་རིང་ཁོ་མོའི་འཛུམ་མདངས་ཏེ།ནམ་རྒྱུན་དང་མི་འདྲ་བར་འཛུམ་མདངས་དེའི་གསེབ་ན་སྐྱོ་སྡུག་དང་ན་ཟུག་ཅིག་སྦས་ཨེ་འདུག་སྙམ་པའི་ཚོར་བ་ཞིག ང་ལ་ཤ་གསེང་རུས་གསེང་ནས་སྐྱེས་བྱུང་། དེ་ལྟར་ངེད་གཉིས་སྐད་ཆ་ཅི་ཡང་མ་བཤད་པར་གཅིག་གིས་གཅིག་ལ་ཧད་དེ་བསྡད། དུས་ཚོད་ཅི་ཙམ་འགོར་བ་མ་ཤེས་མོད། གློ་བུར་དུ་གནམ་ངོ་འཐིབ་ཅིང་ཆར་བ་བབས་བྱུང་། ཆར་བ་དྲག་ཏུ་མ་བབས་ན་འང་ངེད་གཉིས་ཕྱི་ནང་མེད་པར་བརླན་སོང་། རི་ཞོལ་དེར་ངེད་གཉིས་དམ་དུ་འཁྱུད་ནས་ཡུད་ཙམ་འགོར། ཆར་བ་འབབ་མཚམས་མ་བཞག་ནའང་ཉི་མ་སྔར་ལྟར་ཤར་བ་དང་།ཉི་མའི་འོད་ཟེར་ཀློང་ན་འཇའ་ཡག་པོ་ཞིག་ཀྱང་ཤར་བྱུང་། སྤྱིར་འཇའ་ཤར་བ་ནི་ཕྱི་རོལ་ཡུལ་གྱི་རང་བྱུང་གི་འཁོར་ཡུག་འཕོ་འགྱུར་བྱས་ནས་བྱུང་བ་ཞིག་ཡིན་ནའང་ང་ཚོའི་ཕ་ཡུལ་ལ་མཚོན་ན་འཇའ་ཤར་བ་ནི་རྟེན་འབྲེལ་ཧ་ཅང་ཡག་པོ་ཞིག་ཏུ་འཛིན་ཞིང་། ནམ་ཞིག་ལ་འཇའ་ཤར་ན་ཁྱིམ་མི་ཚང་མ་སྦྲ་སྒོར་བུད་དེ།དེའི་ཕྱོགས་ལ་བལྟས་ནས་༼ངོ་སོ་བཅུ་གསུམ་བསྟོད། འཇའ་འདི་ཅི་འདྲའི་ཡག་པ་ལ༽ཟེར་ཞིང་ཁྱིམ་མི་ཚང་མའི་ཞལ་ངོ་འཛུམ་གྱིས་ཁེངས་བར་བྱེད། དེར་མ་ཟད་ན་ནིང་ཞིག་ལ་ཡང་འཇའ་འདི་འདྲ་ཞིག་ཤར་མྱོང་ལ། འཇའ་སྣེ་དེ་ཁྱིམ་མཚེས་རྡོ་རྗེ་ཚང་ལ་ཟུག་པ་དང་། ཉིན་དེར་ཁྱིམ་མཚེས་ཚང་ལ་བུ་ཞིག་སྐྱེས་པས། རྟེན་འབྲེལ་ཤིན་ཏུ་ལེགས་ཏེ། བུ་དེ་དགེ་བཤེས་ཉི་མ་ཚང་གི་ཡང་སྤྲུལ་ཡིན་པར་ངོས་བཟུང་བ་རེད། གང་ལྟར་མཐོ་སྒང་ནས་འཇའ་ཤར་བ་ནི་བཀྲ་ཤིས་པ་ཐམས་ཅད་ཀྱི་ལྟས་བཟང་ཞིག་ཏུ་འཛིན་པས།ངེད་གཉིས་ཀྱང་དགའ་གྲོག་གྲོག་གིས་འཇའ་ལ་བལྟས། དེ་ནི་དངོས་གནས་ཕྱིར་དྲན་ཞིག་རེད། མཛེས་སྡུག་ལྡན་པའི་ཕྱིར་དྲན་ཞིག་རེད། ཕྱིར་ཉིན་ང་ཚོས་རང་རང་ས་ནས་དངོས་པོ་རྣམས་བསྡུ་བསོག་བརྒྱབ་ཅིང་། སྦྲ་ལྟེབས་ཤིང་ཕྱུག་དེད་ནས་རང་རང་ས་ནས་ལམ་ལ་ཆས་པ་ཡིན། ང་རང་ཨ་ཕའི་རྟ་འཕོངས་ན་ཡོད། སྤྱིར་ཡིན་ན་ང་རང་རུ་སྤར་བར་ཧ་ཅང་དགའ། རྟ་ཞོན་ནས་བྱ་དངོས་མང་པོ་ཞིག་ལ་བལྟ་བ་དང་།རི་ལུང་མང་པོ་ཞིག་བརྒྱུད་ནས་འགྲོ་དགོས་པས། སྤྲོ་སྣང་གནོན་མི་ཐུབ་ཞིག་ཡོད་ལ། ནམ་རྒྱུན་རུ་སྤར་བ་ལ་ངང་སྒུག་བྱེད་མྱོང་། ཡིན་ནའང་དེ་རིང་ང་རྒྱུན་ལྡན་ཞིག་མིན་ལ། མིག་གོང་ཆུང་ངུ་དེ།བུ་མོ་ཆུང་ཆུང་དེའི་རྒྱབ་གཟུགས་ལ་ཡང་ཡང་ཤོར་བ་དང་།ཁོ་ཚང་གི་ལུག་ཁྱུ་ལ་ཡང་ཡང་ཤོར། ཁོ་ཡང་ང་དང་འདྲ་བར་ཁོའི་ཨ་ཕའི་རྟ་འཕོངས་ན་ཡོད།ཁོ་མོའི་རྒྱབ་གཟུགས་ལ་བལྟས་ནས་བདག་གི་མིག་གོང་ལས་ཚོད་ཚོད་ལ་མིག་ཆུ་ལྡོག་པོ་འགའ་དྲིལ་བྱུང་། དེ་ནི་ཨ་མས་མ་རྡུང་བར་བཞུར་བའི་མིག་ཆུ་དང་པོ་དེ་ཡིན་སྲིད། ད་ནི་ཕལ་ཆེར་མི་ལོ་བཅུ་བཞི་འགོར་སོང་།ལོ་དེར་ངེད་ཚང་དབྱར་ས་ནས་སྟོན་སར་སྤར་བ་ནས་བཟུང་།ང་རང་དཔེ་ཁུག་ཆུང་ཆུང་ཞིག་ཁུར་ནས་སློབ་གྲྭར་ཆས་པ་དང་།དབྱང་གསལ་སོ་བཞིའི་གདངས་དབྱངས་ལ་ཐོག་དང་པོའི་ས་བོན་བཏབ།ངེད་ཚང་ཡང་དེའི་ཕྱིར་ལོ་རྒྱལ་ཁབ་རིམ་པ་ཁུལ་རིམ་པ་རྫོང་རིམ་པ་ནས་ཡིག་ཆ་འབྱོར་ཏེ།ད་ལོ་ནས་ང་ཚོའི་དབྱར་སའི་ལུང་མདོ་དེ་ན། ཆུ་མཛོད་ཅིག་བཟོ་སྐྲུན་བྱེད་དགོས་པ་དང་།ལུང་བའི་ནང་དུ་འགྲིམ་འགྲུལ་སྟབས་བདེ་བས་ཆེད་དུ།གཞུང་ལམ་ཞིག་ཀྱང་བཟོ་དགོས་པའི་བཀའ་བསྒྲགས། དང་པོ་ངེད་སྡེ་བའི་སྡེ་དཔོན་དང་ས་དེར་འཚོ་བའི་མང་ཚོགས་རྣམས་ཀྱིས་མི་དགའ་བའི་ནམ་འགྱུར་བསྟན་ན་ཡང་། རྗེར་མར་སྒོར་མང་ཙམ་བྱིན་པ་དང་གྲོང་བརྡལ་དུ་ཁང་བ་གསར་བ་རེ་བསྐྲུན་རྒྱུ་ཁེས་བླངས། ད་དུང་ཆུ་མཛོད་ཡོད་པའི་དགེ་མཚན་དང་།མེད་པའི་ཉེས་པ་སོགས་གནས་ལུགས་མང་པོ་བཤད་པ་དང་།གཞུང་ལམ་བསྐྲུན་ན་སྡེ་བའི་དཔལ་འབྱོར་འཕེལ་རྒྱས་འགྲོ་བ་ལ་ཕན་པ་ཆེ་བ་སོགས་གནས་ལུགས་རབ་དང་རིམ་པ་བཤད་པས། སྡེ་མི་རྣམས་ཀྱིས་མགོ་ལྕོགས་ལྡེམ་ལྡེམ་མི་བྱེད་ག་མེད་བྱུང་།དེ་བས་ང་རང་མཛེས་སྡུག་ལྡན་པའི་ཕ་ཡུལ་དང་བྲལ་ནས་མི་ལོ་བཅུ་བཞི་འགོར་སོང་། ངེད་གཉིས་མ་འཕྲད་པར་ཡང་ཏག་ཏག་མི་ངོ་བཅུ་བཞི་འགོར་སོང་། འོན་ཀྱང་། མི་ལོ་བཅུ་བཞིའི་རིང་ལ་མོ་སྔར་བཞིན་ངའི་ནང་སེམས་ན་མཛེས་སྡུག་གི་མེ་ཏོག་ལྟར་བཞད་ཡོད། དེ་དུས་ངས་ད་དུང་བར་ཐག་ནི་མཛེས་སྡུག་ཅིག་ཏུ་འདོད་མྱོང་ལ། མོ་ནི་འཇིག་རྟེན་གྱི་བདེ་སྐྱིད་མཐའ་དག་གི་ནང་སྙིང་དུ་འདོད་ཡོད། ད་ལོའི་དབྱར་གནང་སྐབས་ང་རང་ཕ་ཡུལ་ལ་ཆས་ཤིང་། གྲོང་བརྡལ་ཆུང་ངུ་དེའི་མཁའ་དབུག་རྫོབ་པོ་དང་། རླང་འཁོར། མི་ཚོགས་ཀྱི་འདུ་འཛི་སོགས་སུན་དྲགས་པའམ་མིག་ཆུ་ཐེངས་གཅིག་ལྷུང་མྱོང་བའི་མཛེས་སྡུག་གི་ཕྱིར་དྲན་དེའི་ཆེད་དུ།ང་རང་མཁོ་ཆས་རྣམས་བཟུང་ནས་ལམ་ལ་ཆས་པ་ཡིན། རླངས་འཁོར་ཆུ་ཚོད་གཅིག་ཙམ་བསྐྱོད་རྗེས་གཞི་ནས་ཕ་ཡུལ་ལ་འབྱོར་བྱུང་། བློ་ཡུལ་ལས་འདས་པ་ཞིག་ལ་ད་ལྟ་ངའི་ཕ་ཡུལ་ནི་མི་ལོ་བཅུ་ལྷག་གི་སྔོན་གྱི་ངའི་ཕ་ཡུལ་རེད་ཟེར་ན།ཡིད་ཆེས་དཀའ་བ་ཞིག་ཏུ་གྱུར་འདུག རྒྱུ་མཚན་རི་སྒང་དག་ཏུ་སྲོག་ཆགས་གཅིག་ཀྱང་མཐོང་དུ་མེད་ལ། རྩི་ཤིང་དང་མེ་ཏོག་ཕར་ཞོག ཐ་ན་ངལ་གསོ་བྱེད་སའི་ལྗང་མདོག་གི་ཡུལ་གྲུ་ཆུང་ཆུང་ཞིག་ཀྱང་ལར་ནས་མི་འདུག ལོ་དེར་ལྷ་སྐྱིད་ངེད་གཉིས་ཀྱི་ཕྱིར་དྲན་མཐའ་དག་ལུས་སའི་རི་ཞོལ་དེ་ཡང་། ད་ལྟ་གཞུང་ལམ་ཆེན་པོ་ཞིག་གིས་རྨེག་མེད་དུ་བཅོམ་ཟིན་པ་དང་། ཉིན་ངོས་ཀྱི་སྐེད་པ་རྫ་རི་ཡིན་ལ་རྩེ་མོ་གངས་ཀྱིས་བཟུང་བ། ཡུལ་མི་ཚོས་རི་འདིའི་མིང་ལ་ཨ་མྱེ་ཁྱུང་ཆེན་ཞེས་འབོད་པར་བྱེད་པ་དང་། ལོ་རེའི་ལོ་བསང་དང་རུ་བསང་།འཛམ་གླིང་སྤྱི་བསང་སོགས་ཀྱི་སྐབས། རང་སྡེ་བའི་ཡུལ་མི་ཡོད་ཚད་དང་ཕྱི་ཡོང་མགྲོན་པོ་མང་པོ་གནས་འདིར་མ་གྲོས་གཅིག་མཐུན་དུ་འཛོམས་ཤིང་། རི་འདི་དར་མཚོན་སྣ་ལྔས་བརྒྱན་པ་དང་། རྒན་པ་ཚོས་བསང་མཆོད་ཁ་སྨོན་དང་། གསར་བུ་རྣམས་ཀྱིས་ཀི་སྒྲའི་ཀློང་ན་རི་འདིའི་ངོ་སོ་དགུང་ལ་བསྟོད་ཡོད། མདོར་ན་ཨ་མྱེ་ཁྱུང་ཆེན་ནི་བསྟོད་ན་སྙིང་ཉེ་བ།བཀུར་ན་གཉེན་བྱེད་པ་ཞིག་ཡིན་པས། སྡེ་མི་རྣམ་གྱིས་ལོ་གཅིག་གི་སྒོ་ཕྱུགས་ཀྱི་འཕེལ་ཁ་དང་ལོ་ལེགས་ཀྱི་འབྲས་བུ་ཡོད་ཚད་ཀྱང་ཡུལ་ལྷ་ཁྱུང་ཆེན་ལ་གཏད་ཡོད། དེ་བས་རི་འདི་ནི་ཡུལ་མི་ཚོས་སེམས་ཀྱི་ཁེན་ས་དང་ཡིད་ཀྱི་བསྟེན་ས་རུ་འགྱུར་ཡོད་པ་སྨོས་མ་དགོས་པ་ཞིག་རེད། ཡིན་ན་ཡང་ཡུལ་མི་ཚོས་ཁོག་གི་སྙིང་དང་། དཔྲལ་བའི་མིག་ལྟ་བུའི་རྫ་རི་མཐོན་པོ་རྩེ་མོ་གནམ་ལ་ཟུག་པ་དེའི་དཀྱིལ་སྙིང་ལ་ཁེལ་ཨེ་འདུག་སྙམ་ས་དེ་ནས་ས་བརྐོ་རྡོ་སློག་བྱས་ཏེ།ནང་དུ་འདྲུད་འཁོར་ཆེ་བ་དང་ཟོང་འཁོར། ས་རྐོ་འཁོར་སོགས་ཅི་འདོད་དུ་འགྲོ་འོང་བྱེད་ཐུབ་པ་ཞིག་བཟོས་འདུག དེ་ནས་མར་བལྟས་ན་ལོ་བཅུའི་སྔོན་ལ་མཚོ་ཆུང་གཡུ་སྒྲོན་དུ་འབོད་པ། ཡབ་མེས་གོང་མ་ཚོའི་ངག་རྒྱུན་དུ་རྒྱ་བཟའ་བོད་ཡུལ་དུ་ཡོང་དུས་ཕ་ཡུལ་དང་ཕ་མ་དྲན་ནས་མིག་ཆུ་ལྷུང་བའི་ཤུལ་ཡིན་ཟེར་བ་དང་། ཡང་ཁ་ཅིག་གིས་འདི་ནི་ཧོར་གྱི་བློན་ཆེན་ཤན་པ་དམེ་རུ་རྩེ་སོགས་ཀྱིས་འབྲུག་མོ་གླིང་ནས་བཙན་ཁྲིད་བྱས་ཏེ།ཧོར་ཡུལ་བགྲོད་སྐབས་འབྲུག་མོས་ཕ་ཡུལ་དང་ཕ་མ་དྲན་པའི་སྐྱོ་གདུང་དང་། དཔའ་མཛངས་གཉིས་ལྡན་གྱི་རང་གི་ཁྱོ་བོ་དྲན་དེ་མ་བཟོད་པར། མིག་ཆུ་ལྷུང་བའི་ཤུལ་ཡིན་ཟེར།གང་ལྟར་མཚོ་སྔོན་མོ་གཡུ་ཡི་མེ་ལུང་འདྲ་བ་ཞིག་ཡོད་པ་དེ།ད་ལྟ་ཁ་དོག་ནག་པོ་བསྒྱུར་འདུག་ལ། ས་བརྐོ་ཆེད་མཁས་པ་དག་གིས་འདི་ལ་དུག་ཆུའམ་དུག་ཆུ་མཚོ་ནང་དུ་འདྲེས་པ་ལ་དུག་མཚོ་བྱ་སྐད། ཆུ་འདི་མི་དང་སེམས་ཅན་སྲོག་ལྡན་སུ་ཡི་ཁ་ལ་སོང་ན་མི་ཤི་གསོན་ནས་འགྲོ་བ་ནི་རྨི་ལམ་ཡིན་ཟེར། དེ་ནི་ངས་རྗེས་སུ་སློབ་གྲྭ་ནས་ཤེས་བ་ཡིན་ལ། རྫས་འགྱུར་དགེ་རྒན་གྱིས་ནན་བཤད་བྱེད་མྱོང་བའི་གནད་དོན་ཞིག་ཀྱང་རེད། དེ་ལྟར་དྲན་པའི་མོད་ལ་ང་རང་ལུང་ནང་དུ་གོམ་ཁ་བསྒྱུར་མ་འདོད་པར།ལུང་མདོ་ནས་མི་ལོ་བཅུ་བཞིའི་སྔོན་གྱི་མཛེས་སྡུག་གི་ཕ་ཡུལ་ལ་བལྟས། མི་ལོ་བཅུའི་བཞིའི་རྗེས་ཀྱི་འབག་བཙོག་གི་ཕ་ཡུལ་ལ་བལྟས།ད་དུང་སྙན་ངག་ཅིག་སྐད་གསེང་མཐོན་པོས་ཡང་ཡང་གྱེར། དེ་ནི་ཉེ་ལམ་བོད་ཀྱི་སྒྱུ་རྩལ་བ་ཆེན་པོ་ཞི་བདེ་ཉི་མ་ལགས་ཀྱིས་རྒྱ་ཡིག་གི་ལམ་ནས་བྲིས་པའི་སྙན་ངག༼རྩྭ་ཐང་གང་དུ་བུད་སོང༽ཞེས་པ་དེ་རེད། སྙན་ངག་དེ་གྱེར་གྱིན་གྱེར་གྱིན། མིག་ནས་མཆུ་མ་ལུད་ཅིང་། ངག་ནས་འོ་དོད་བྱུང་སྟེ། སྙན་ངག་གི་ཚིག་འབྲི་རེ་རེ་ལ་དབྱངས་ཀྱི་མཛེས་པ་བསྐྲུན་སོང་།དུས་ཚོད་ཅི་ཙམ་འགོར་བ་མ་ཤེས་མོད། གློ་བུར་དུ་གནམ་ངོ་འཐིབ་ཅིང་འབྲུག་ལྡུར་བ། གློག་དམར་འཁྱུག་པ། རླུང་ལྡང་བ་སོགས་དྲགས་ཆར་ཞིག་འབབ་རྒྱུ་ཡིན་པ་ཟོལ་གྱིས་སྟོན། ནམ་མཁའ་འདི་མི་ཞིག་ཡིན་པའི་དབང་དུ་བཏང་ན།ད་ལྟ་ཁོང་རབ་ཏུ་ཁྲོས་ནས་ངོ་ནག་ཅིང་མིག་རྩ་ཡང་དམར་པོ་འགྱུར་ཡོད་པ་འདྲ། ངས་དེ་ལྟར་བསམ་གྱིན་འཇིགས་སྐྲག་དང་སྔངས་འདར་ཁྲོང་ན་ཅི་ཡང་བྱེད་མ་ནུས་པར་རང་སར་ལུས། ངོ་མ་དུས་ཚོད་ནི་མིག་རྡེབ་དབང་ཙམ་ལའང་ངལ་གསོ་བྱེད་ཁོམ་མེད་པར་མུ་ནས་མུ་རུ་བརྒྱུག་པ་དང་།"ཚེ་མི་རྟག་ལ་ཁའི་ཉི་མ། རྒྱུ་མི་རྟག་རྩྭ་མགོའི་ཟིལ་བ"ཟེར་བ་ཇི་བཞིན་དེའི་ཞོར་ལ་འཇིག་རྟེན་གྱི་སྐྱེ་དངོས་མང་པོ་ཞིག་ཀྱང་།མེད་པ་ནས་ཡོད་པར་འགྱུར་བ།ཡོད་པ་ནས་སླར་ཡང་མེད་པར་འགྱུར་བ།མདོར་ན་མི་རེ་རེའི་ནང་སེམས་ན་མཛེས་སྡུག་གི་འཚོ་མྱོང་བའི་བྱ་དངོས་དང་།མི་ཚོགས་དག་ཀྱང་ལོ་རྒྱུས་དང་དུས་ཚོད་ཀྱི་གཙང་བོའི་ཀློང་དང་འགྲོགས་ནས་མེད་པར་འགྱུར་འགྲོ་ལ།དེ་དག་གིས་མི་སེམས་ལ་བཟོད་དཀའ་བའི་ན་ཟུག་དང་རྨ་ཁ་རེ་འཇོག་འགྲོ་བའང་ཆོས་ཉིད་རེད། ངས་དེ་ལྟར་དྲན་པའི་མོད་ལ།
ནོར་བུ་རང་ལ་ཡོད་དུས།། ནོར་བའི་ནོར་ཉམས་མ་ཆོད།། ནོར་བུ་གཞན་ལ་ཤོར་དུས།། སྙིང་རླུང་སྟོད་ལ་འཚངས་བྱུང་།།
ཟེར་བ་ལྟར་ཡིད་སེམས་བློང་བློང་བོར་གྱུར་བ་དང་།སྙིང་གི་ཕར་ཚད་སླར་ལས་ཇེ་མགྱོགས་བྱེད་ལ།མི་ལོ་བཅུ་བཞིའི་སྔོན་གྱི་མཛེས་སྡུག་གི་ཕ་ཡུལ་དང་གཡང་དཀར་ལུག་ཁྱུ། ནོར་ནག་གི་ངུར་སྒྲ། རྟ་ཕོའི་མཚེར་སྒྲ། བུ་མོ་ལྷ་སྐྱིད་ཀྱི་འཛུམ་མདངས་སོགས་དྲན་པ་དང་། ངེད་གཉིས་ཀྱི་མཇུག་མཐའི་ལག་འཇུ་དང་། ངེད་གཉིས་དམ་དུ་འཁྱུད་ནས་འཇའ་ལ་བལྟས་ཚུལ་གྱི་རྣམ་པ་སོགས་གློག་བརྙན་འཆར་བ་ཇི་བཞིན། ཡིད་ལ་ཡང་ཡང་དྲན་པ་དང་མིག་གོང་ནས་མཆི་མའི་རླབས་རྒྱུན་ནི་རེ་དོགས་དང་ཐེ་ཚོམ་མེད་པར་མུ་ནས་མུ་རུ་བཞུར་བྱུང་། ད་ནི་ཉི་མ་ཡང་འཁར་བར་ཁེན་པའི་རྒན་པོ་ཞིག་དང་འདྲ་བར་ག་ལེར་རི་རྒྱབ་ལ་བཞུད་སོང་། ང་རང་རི་ཞོལ་དེར་དར་ལྕོག་བཞིན་འགྲེང་ནས་བསམ་གཞིགས་བྱས། ངས་འདི་དྲན་ནམ་ཞིག་ང་ཡང་མགོ་སྐྲ་རླུང་གིས་ཁུར་བའི་རྒན་པོ་ཞིག་ཡིན། དེ་དུས་ངས་ཀྱང་ཕ་ཡུལ་རབ་ཏུ་དྲན་སྲིད་ལ། ཕ་ཡུལ་གྱི་རི་ལུང་ཐང་གསུམ་གྱི་མཛེས་སྡུག་དག་ཀྱང་ཡིད་ལ་ཡང་ཡང་དྲན་སྲིད། དེ་དུས་ངས་ཚ་བོ་དང་ཚ་མོ་རྣམས་ལ་ཕ་ཡུལ་གྱི་མཚར་སྡུག་གི་རྣམ་པ་དང་།ལྷ་སྐྱིད་ངེད་གཉིས་ཀྱི་ཆེས་དྭངས་གཙང་གི་བརྩེ་བ་དག་གཏམ་རྒྱུད་དུ་བཤད་སྲིད།ཡིན་ན་ཡང་དེ་དུས་དེ་དག་ཐམས་ཅད་མཛེས་སྡུག་གི་ལྷ་སྒྲུང་ཞིག་ཏུ་འགྱུར་ཡོད། ང་ཁོ་ནར་དབང་བའི་མཛེས་སྡུག་གི་ཕྱིར་དྲན་ཞིག་ཏུ་འགྱུར་ཡོད་།ཡིན་ན་ཡང་ངས་ཆེས་མཐའ་མཇུག་ཏུ་ད་དུང་ཡང་རང་གི་དུར་ས་སྔར་གྱི་ཕ་ཡུལ་བདབས་དགོས་པའི་རེ་བ་འདོན་སྲིད་ལ། ཤི་མིག་དུར་ཁུང་ནས་ཀྱང་རང་གི་ཕ་ཡུལ་བལྟ་བའི་དམ་བཅའ་འཇོག་སྲིད།
