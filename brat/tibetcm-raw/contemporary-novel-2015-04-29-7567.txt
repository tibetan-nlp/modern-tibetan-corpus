དབྱངས་ཅན་མའི་པ་ཕ་བཙལ་བ།
ཁ་བ་རྐང་གླིང་།༼ ལྷ་རྩེ ༽

རྨི་ལམ་དུ་ཟླ་བ་གསུམ་མཉམ་དུ་ཤར་བཞིན་གླིང་རྗེ་གེ་སར་ཡིན་ཟེར་བའི་མི་བུ་མགོ་མོ་ལྕགས་ཀྱི་བཏུམས་ཤིང་། སྐེད་པ་གླང་པོ་ཆེའི་གྲོད་པ་ཙམ་བྱས་པ་ཞིག་གིས་མོའི་མལ་ཆས་ནང་འཛུལ་ནས་འོ་མ་གཉིས་པོ་ལྕེ་རྨིད་གཉིས་ཙམ་བྱས་པ་ཞིག་གིས་ལྡག་བཞིན། མོའི་བརླ་ཡི་བར་གྱི་མོ་མཚན་ནི་ལག་པ་སྡེར་མོས་ཁེངས་པ་ཞིག་གིས་འཕུར་སྙེགས་བྱེད་པ་རྨིས་ན་ཡང་། དབྱངས་ཅན་མ་ནི་གཉིད་ལས་དངངས་སད་བྱེད་རྒྱུ་ཕར་ཞོག དེ་ལས་དེ་ནི་སུས་ཀྱང་ལོངས་སྤྱོད་བྱེད་ཐབས་བྲལ་བའི་བདེ་སྐྱིད་ཅིག་ཏུ་ངོམ་ནས་མུ་མཐུད་དུ་རྨི་ལམ་གྱི་དག་ཞིང་དུ་སྐྱོད། མི་རབས་འགའི་རྗེས་སུ་དབྱངས་ཅན་མ་གཞི་ནས་གཉིད་ལས་སད་དེ་རྨི་ལམ་སྐྱིད་པོ་དེར་དྲན་གསོ་བྱེད་ཁོམ་ཡང་མེད་པར། རང་གི་པ་ཕ་ལུག་གི་རྐང་པ་སྐྱེས་པ་དེ་འཚོལ་བར་ཕྱིན། ཕྱིན་ཕྱིན་པའི་ལམ་ཁ་ནི་ཇེ་ནག་ནས་ཇེ་ནག་ཏུ་སོང་བས། ངུ་སྐད་ཀྱི་དབྱངས་རྟ་སྙན་མོ་ཞིག་བཏོན་ནས་ཆགས་པས་མྱོས་པའི་སྐྱེས་པ་གཟུགས་མེད་ཀྱི་རྣམ་ཤེས་དེ་ཚོ་རང་གི་ཕྱོགས་སུ་བསྐོང་། མོས་རྣམ་ཤེས་ཚངས་པ་ཞེས་པའི་གཟུགས་ཀྱི་སྐྱེ་བོ་དེར་འཁྱུད་དེ་ཁྲག་དམར་རྫར་རྫར་དུ་འདོན་བཞིན་ཡིད་སྐྱོ་མདོག་ཁ་པོ་ཞིག་བསྟན་ཏེ་རང་གི་པ་ཕ་ལུག་གི་རྐང་པ་སྐྱེས་པ་དེ་འཚོལ་བའི་གྲོགས་སུ་བོས།། དེ་ལྟར་ཁོ་གཉིས་འདོད་ཆགས་ལོངས་སུ་སྤྱོད་ཀྱིན། ལམ་བར་དུ་ཐུག་པའི་གྲོང་སྡེ་ཤམ་བྷ་ལ་དང་། སྐབས་གསུམ་གནས་སོགས་སུ་འཚོ་གནས་བྱེད་པའི་མིའུ་ཆུང་ནམ་མཁར་འཕུར་མཁན་དག་ལ་བཀའ་འདྲི་བྱེད་ཅིང་བསྐྱོད། ཡིན་ནའང་། ཁོ་ཚོས་དབྱངས་ཅན་མའི་ནུ་མ་ཆེན་པོ་ཇོ་མོ་གླང་མ་འདྲ་བ་དེར་བལྟ་བ་ལས་ཁོ་མོའི་འདྲི་བར་ལན་འདེབས་མཁན་ཅིག་ཀྱང་མ་བྱུང་།ལོ་ངོ་ཉིས་བརྒྱ་ལྷག་ཙམ་ཕྱིན་རྗེས། བྱང་ཆུབ་སེམས་དཔའ་པ་ཞེས་པའི་གྲོང་ཁྱེར་ཆེན་པོ་ཞིག་ཏུ་སླེབས། དེ་ཉིན་སུ་ལ་ཡང་མ་བབས་པར་ཁོ་གཉིས་ཀྱི་ལུས་སྟེང་དུ་ཁ་བ་ནག་པོ་ཆེ་ཆུང་གཡག་མགོ་ཙམ་ཤུགས་ཀྱིས་བབས་པས་ཁོ་གཉིས་ལ་གྲོང་ཁྱེར་ཆེན་པོ་དེའི་ནང་དུ་གནས་ཚང་གཡོར་མཁན་མ་བྱུང་སྟབས། ནམ་མཁའ་ནས་བྱ་ཁྲུང་ཁྲུང་གཉིས་ལ་སྒོར་མོ་བརྒྱ་སྤྲད་དེ་རི་བོ་དེའི་རྒྱབ་ཀྱི་བྲག་ཕུག་ཞིག་ཏུ་བསྐྱལ། ཁོ་གཉིས་གྲང་ཟུག་གིས་ལུས་སྟེང་ཡོངས་འཁྱག་པ་རྒྱབ་སྟེ་སྒུལ་སྐྱོད་ཀྱང་བྱེད་མ་ཐུབ། སྐབས་དེ་དུས་ལྷ་ཡུལ་ཡིན་ཞེས་པའི་མི་ཞིག་གིས་ཁོ་གཉིས་མཉམ་དུ་བསྙལ་ཏེ། ཁོ་་རང་མྱུར་མོར་སྒོར་ཕྱིར་བུད་སྟབས། ཁོ་གཉིས་ཀྱི་བསམ་པར་འདི་ནི་གཤིན་རྗེས་འུ་གཉིས་ལ་གནང་བའི་བདེ་སྐྱིད་ཅིག་ཡིན་ངེས་བསམ་བཞིན་འདོད་ཆགས་ལ་རོལ། དེ་ཉིན་འདོད་ཆགས་ཀྱི་བདེ་སྐྱིད་དེས། མོས་རང་གི་པ་ཕ་འཚོལ་རྒྱུ་བརྗེད། འོན་ཀྱང་སང་ཉིན་མུན་རུབ་པ་དང་མོས་རང་གི་པ་ཕ་འཚོལ་རྒྱུའི་དོན་དག་དེ་ཡིད་ལ་དྲན་ནས། ལན་རེ་ས་རུ་སྒྱེལ་བཞིན་ལན་རེ་གནམ་ལ་འཕུར་གྱིན་དུ་མུ་མཐུད་ཤར་ཕྱོགས་སུ་བསྐྱོད། མོ་རང་གཅིག་པོར་ལོ་ངོ་སུམ་བརྒྱ་ཏག་ཏག་ཤར་ཕྱོགས་སུ་སྐྱོད་པས། མོ་ཐང་ཆད་ཐག་ཆོད་ནས་གཉིད་དུ་ཡུར། རྨི་ལམ་དུ་མོའི་ཕ་པས་རང་གཉིད་ཀྱིས་ངོ་གདོང་ལ་བྱིལ་བྱིལ་བྱེད་པ་རྨིས། རྨི་ལམ་དེ་ལོ་ངོ་བརྒྱ་ཏག་ ཏག་རྨིས། རྨི་ལམ་ལས་སད་དུས་རང་གི་པ་ཕ་རང་གི་མདུན་ན་དུང་ལྟར་དཀར་བའི་མགོ་སྐྲ་ས་རུ་དྲུད་བཞིན། ཚངས་སྲས་མ་ཟེར་བའི་ཤིན་ཏུ་ནས་མཛེས་ཤིང་མགོ་སྐྲ་ནས་རྐང་མཛུབ་བར་དུ་འོད་ཆེམ་ཆེམ་གྱི་བ་སོ་དང་སེར་ཐིང་ཐིང་གི་རག་གིས་རྒྱན་པའི་བུ་མོ་ལོངས་སུ་སྤྱད་ནས་འདུག་ཅིང་། མོ་རང་གད་སྙིག་བྱོལ་སའི་རླངས་འཁོར་ཞིག་གིས་ནང་དུ་བླུགས་ཏེ་ཁྲག་གི་འཁྱིལ་བའི་རྒྱ་མཚོ་ཆེན་པོ་ཞིག་གི་ནང་དུ་གཡུགས། དེ་ནས་གནས་དེ་རུ་མོའི་མི་ཚེའི་མཐའ་མཇུག་བསྐྱལ།།
(མཆོད་མེའི་པར་དབང་གཞན་གྱིས་དགོལ་མི་རུང)


རྩོམ་ཁུངས། ངེད་དྲ་བ།
