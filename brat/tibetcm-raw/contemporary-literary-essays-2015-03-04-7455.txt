སྒྲུང་རྩོམ་གྱི་རྒྱུ་ཆ་འཚོལ་སྡུད་བྱེད་སྟངས་དང་འབྲི་ཐབས།
བེ་རི་འཇིགས་མེད་དབང་རྒྱལ།

སྒྲུང་རྩོམ་གྱི་གོ་དོན་དང་རྣམ་གྲངས། སྒྲུང་རྩོམ་ནི་གནའ་དེང་གང་དུའང་མེད་ཐབས་མེད་པའི་བོད་ཀྱི་རྩོམ་རིག་གི་ཆ་ལག་གལ་ཆེ་རང་བཞིན་ལྡན་པ་ཞིག་ཡིན་ལ། མི་རིགས་རང་ཉིད་ཀྱི་སྤྱི་ཚོགས་ཚོར་བ་མཚོན་པའི་སྒྱུ་རྩལ་གྱི་ལྷུ་ལག་མེད་ཐབས་མེད་པ་ཞིག་ཀྱང་ཡིན། སྤྱིར་‘སྒྲུང་’གི་གོ་དོན་ཐད། བོད་རྒྱ་ཚིག་མཛོད་དུ། ‘སྔོན་བྱུང་རབ་བཏགས་ཀྱི་ལོ་རྒྱུས’། ཞེས་བྲིས་སྣང་ལ། སྒྲུང་མཁན་གྱི་གོ་དོན་ཡང་། ‘སྔོན་བྱུང་ལོ་རྒྱུས་བཤད་མཁན་དང་ལོ་རྒྱུས་གླུ་རུ་ལེན་མཁན་’ལ་ངོས་འཛིན་དང་། སྒྲུང་གཏམ་ནི། ‘སྔོན་བྱུང་གཏམ་རྒྱུན་’རང་ལ་ངོས་འཛིན་བྱས་འདུག ཡང་‘རབ་གཏགས་ཀྱི་ལོ་རྒྱུས་བློ་ལ་ཤར་བ་ལ་སྒྲུང་བབ་’ཅེས་ཟེར་བ་དང་། སྒྲུང་བཤད་པ་ནི། ‘སྔར་གྱི་གཏམ་རྒྱུན་བཤད་པའམ་། ཡང་ན་རྫུན་བཤད་པའི་དོན་དོན་ནམ་མིང་ཚབ་’ཏུ་བཀོལ་ཚུལ་འཁོད་པ་འདི་དག་ནི་སྒྲུང་གི་ངོ་གདོང་གི་གོ་དོན་གྱི་འགྲེལ་བརྗོད་ཙམ་ཡིན། དུང་དཀར་ཚིག་མཛོད་དུ། སྒྲུང་གི་གོ་དོན་བཤད་སྐབས། སྒྲུང་བོན་ཞེས་པ་རྒྱལ་པོ་གཉའ་ཁྲི་བཙན་པོ་ནས་དྲངས་པའི་རྒྱལ་རབས་དྲུག་པ་ཁྲི་ལྡེའུ་བཙན་གྱི་དུས་དབུ་རུ་ངོམ་ཤོད་བོན་ཞེས་བྱ་བར་བོན་གྱི་གཤེན་རབ་མི་བོ་ཆེ་བྱ་བ་ཞིག་བྱུང་བ་དེས་ཐོག་མར་བརྩམས་པ་རྣམས་ཡིན། འདི་ལ་གཞུང་གཞན་དུ་བརྡོལ་བོན་ཟེར། ཞེས་བྲིས་འདུག་པ་ནི། སྒྲུང་ལྡེའུ་བོན་གྱི་ནང་གི་སྒྲུང་ཞེས་པ་དེ་གང་ལས་བྱུང་བའི་ཁུངས་བསྟན་པ་སེམས་ལ། དེ་ཉིད་བརྡོལ་བོན་གྱི་གཞུང་ཡིན་པའི་ཁུངས་དང་རྒྱུ་མཚན་ནི་གཡུང་དྲུང་བོན་ལུགས་འཛིན་པའི་མཁས་པ་ཞིག་ལ་དྲིས་ན་མ་གཏོགས་སུས་ཤེས། སོག་པོ་དགེ་བཤེས་ཆོས་གྲགས་ཀྱི་བརྡ་དག་མིང་ཚིག་གསལ་བར། བོན་སྒྲུང་ནི། བོན་གྱི་ཆོས་འབྱུང་ལྟ་བུ་ཞེས་འགྲེལ་འདུག །གང་ལྟར་ཡང་། རྫུན་པའི་ལམ་ནས་བདེན་པའི་གནས་ལུགས་མཚོན་པ་ནི་སྒྲུང་གི་ངོ་གདོང་གི་རྒྱབ་དུ་ཡོད་པའི་གནས་ལུགས་ཀྱི་རིན་ཐང་དངོས་དེ་ཡིན་པས་རིན་ཐང་དེ་ཉིད་ངོ་གདོང་གི་རྒྱབ་ཏུ་གབ་སྡོད་ཡོད་པ་དེ་ཡང་མཁྱེན་དགོས་པའི་དོན་གནད་གལ་ཆེ་ཞིག་ཡིན། དེས་ན་རྫུན་པའི་ལམ་ནས་བདེན་པའི་གནས་ལུགས་ལམ་སྟོན་ནུས་པའི་གཏམ་རྒྱུད་དང་འབྲེལ་བའི་སྒྱུ་རྩལ་གྱི་བརྡ་ཐབས་སམ་རྩོམ་རིག་ཆ་ལག་ཅིག་ལ་སྒྲུང་རྩོམ་ཟེར་ན་ཆོག་སྙམ། འོན་ཀྱང་སྒྲུང་རྩོམ་ཡིན་ཚད་འདིའི་ཁྱབ་ཁོངས་སུ་འདུ་བའི་ངེས་པ་མེད་དེ། རྩོམ་པ་པོས་མ་འོངས་མངོན་གཟིགས་སམ་ཚོད་བགོམ་གྱིས་འབྱུང་འགྱུར་ལམ་སྟོན་དང་འབྲེལ་བ་ཡོད་པའི་གཏམ་ལ་སྒྱུ་རྩལ་གྱི་ཚོན་རྩི་བྱུགས་པའི་སྒྲུང་རྩོམ་ཡང་དེང་རབས་སྒྲུང་རྩོམ་སྨྲ་བ་མི་ཉུང་བ་ཞིག་གིས་བྲིས་འདུག་པ་མཐོང་དུ་ཡོད་པས་སོ།།ནང་གསེས་དབྱེ་བ་རགས་ཙམ་ཕྱེས་ན། གནའ་སྒྲུང༌། ལྷ་སྒྲུང༌། གླིང་སྒྲུང༌། བྱིས་སྒྲུང༌། དེང་སྒྲུང་བཅས། ལྔ་རུ་འབྱེད་ན་ཅུང་གོ་བདེ་བར་སྙམ་ལ། ཡང་སྒྲུང་རྩོམ་ལ་རྟོག་པ་འཆར་ཡན་གྱི་སྒྲུང་རྩོམ་དང་། སྒྱུ་འཕྲུལ་དངོས་ཡོད་ཀྱི་སྒྲུང་རྩོམ་གྱི་དབྱེ་སྒོ་གཉིས་སུ་ཕྱེས་ནའང་རུང་སྙམ། ནང་གསེས་ཞིབ་ཕྲའི་དབྱེ་བ་ནི་ཤིན་ཏུའང་མང་པོར་དབྱེ་རུང་བ་སེམས་སོ། ། གནའ་སྒྲུང་ནི། སྒྲུང་ལྡེ་བོན་གསུམ་གྱི་ནང་གི་མ་སངས་ཀྱི་སྒྲུང་ལྟ་བུ་སྟེ། དེང་སང་རོ་ལངས་གསེར་འགྱུར་གྱི་སྒྲུང་ཕལ་ཆེ་བ་རྣམས་སོ། ། ལྷ་སྒྲུང་ནི། གནའ་བོའི་སྤྱི་ཚོགས་ཁྲོད་ཀྱི་བསམ་བློའི་འཆར་སྒོ་དང་བསྟུན་ནས་སྐབས་དེའི་སྤྱི་ཚོགས་འཚོ་བའི་དགོས་མཁོ་དང་འཚམས་པའི་གནས་སྟངས་མང་པོ་ཞིག་ལྷ་འདྲེ་ལ་གཞི་བཅོལ་ནས་བཤད་པའི་སྲོལ་རྒྱུན་ཡིག་ཐོག་ཏུ་བཀོད་པ་ལ་ཟེར་བ་སྙམ། འདི་དང་གནའ་སྒྲུང་གཉིས་ཕན་ཚུན་གཅིག་པ་ལྟ་བུ་ཡང་ཤིན་ཏུ་མང་༌། གླིང་སྒྲུང་ནི། གླིང་གེ་སར་རྒྱལ་པོའི་གཏམ་བརྒྱུད་བཤད་སྒྲོས་འདྲ་མིན་ལ་གཞི་བཅོལ་ནས། མདོ་ཁམས་གང་ས་སྤྱི་ཚོགས་ཁོར་ཡུག་གི་འཚོ་བའི་རྒྱུད་རིམ་དང་སྲོལ་རྒྱུན་རིག་གཞུང་སྲོག་རྩར་གཟུང་ནས་བཅོས་སྒྲུང་འདྲ་མིན་བྲིས་པའི་རིག་ལ་ཟེར་ཆོག་སྙམ། བྱིས་སྒྲུང་ནི། གནའ་ནས་ད་ལྟའི་བར་བྱིས་པ་ཚོས་རང་གི་སེམས་ཀྱི་འཆར་སྒོ་གཞིར་བཟུང་ནས་ཕྱིའི་ཁོར་ཡུག་སོགས་ལ་འགྲེལ་བརྗོད་འདྲ་མིན་དང་འབྲེལ་བའི་ངག་སྒྲོས་དང༌། རྒན་རབས་ཚོས་བྱིས་པའི་སེམས་ཁམས་ལ་གཞི་བཅོལ་ནས་སྣོད་བཅུད་ཀྱི་འགྲེལ་བརྗོད་ཅི་རིགས་སུ་བཅོས་པའི་ངག་སྒྲོས་ཡིག་ཏུ་ཕབ་པའི་ཚིག་ལྷུག་གི་རིགས་རྣམས་ཡིན་ཆོག་སྙམ། དཔེ་ན། ཤིག་དང་ལྗི་བའི་བྱིས་སྒྲུང་ལྟ་བུས་མཚོན་ནུས་སྙམ། དུས་རབས་སྒྲུང་རྩོམ་ནི། བསམ་བློའི་ལྟ་གྲུབ་དུས་རབས་ཀྱི་རྗེས་བསྙེགས་པ་དང༌། བརྗོད་གཞིའི་སྙིང་དོན་ཚོར་བ་དང་ཆེས་འབྲེལ་བ། རྩོམ་གྱི་སྒྲོམ་གཞི་དེང་རབས་ཀྱི་ཚིག་སྦྱོར་དང་མཐུན་པ། བྱེད་ལས་སྤྱི་ཚོགས་ཁྲོད་ཀྱི་སྐྱོན་ཡོན་ཡོ་སྲོང་དང་། ངན་སྲོལ་ཐེར་འདོན། གོ་སྐོན་ར་སྤྲོད་ཀྱི་ནུས་པ་ལྡན་པ། ཚིག་གི་སྒྲོམ་ཀློག་ན་སྙན་ཞིང་གསོན་ཉམས་ལྡན་པ། དོན་གྱི་རོ་མྱོང་ན་སྤྱི་ཚོགས་ཚོར་བའི་གནས་བབས་གསལ་བ་ཞིག་ལ་ངོས་འཛིན་བྱས་ན་ཆོག་སྙམ། དཔེར་ན་དོན་གྲུབ་རྒྱལ་གྱི་སད་ཀྱིས་བཅོམ་པའི་མེ་ཏོག་དང༌། བཀྲ་ཤིས་དཔལ་ལྡན་གྱི་ཁྱིམ་ཚང་ཕལ་བའི་གཏམ་བརྒྱུད་ལྟ་བུས་མཚོན་ནུས་སྙམ།
སྒྲུང་རྩོམ་གྱི་རྒྱུ་ཆ་འཚོལ་སྡུད་བྱེད་སྟངས།
སྒྲུང་རྩོམ་ནི་ཚོར་བའི་ཟ་འཕྲུག་གིས་འགུལ་སྐྱོད་ཐེབས་པའི་སྙན་ཚིག་ཙམ་དུ་མི་སྡོད་པར་དེ་ལས་ཡོངས་སུ་ལས་བརྒལ་ཏེ། ཚོར་བའི་ཟ་འཕྲུག་དེ་དང་འབྲེལ་བའི་རབ་བཏགས་ཀྱི་ཡ་མཚན་པའི་གཏམ་རྒྱུད་གང་ཞིག་གསར་དུ་བསྐྲུན་ནས། གཏམ་རྒྱུད་དེའི་ནང་གི་མི་སྣ་དང་ཁོར་ཡུག རྒྱུ་རྐྱེན་དང་བྱུང་རིམ། རྙོག་འཛིངས་དང་མཇུག་འབྲས་བཅས། རྩོམ་པ་པོའི་བསམ་བློའི་ཀུན་རྟོག་གིས་བློས་བསླངས་དང་། དངོས་ཡོད་གནས་སྟངས་ཀྱི་རམ་འདེགས་ལ་བརྟེན་ནས་གསོན་ཉམས་ལྡན་པའི་སྒྱུ་རྩལ་གྱི་ངག་རིག་ལས། མགོ་མཐོང་བ་དང་མཇུག་ཀློག་འདོད་པའི་ཚོར་ཤུགས་རང་བཞིན་གྱིས་སླེབས་པ། ཀློག་གིན་ཀློག་གིན་རྩོམ་ཡིག་ནང་གི་བརྗོད་གཞི་མི་སྣའི་ཚོར་ཤུགས་དེ་དག་མི་མངོན་པའི་སྒོ་ནས་ཀློག་པ་པོ་རང་གི་ཚོར་བའི་ཐོག་ཏུ་ཤུགས་མ་ཐུབ་ཀྱིས་སླེབས་འོང་པ། ཚོར་ཤུགས་ཀྱི་རྒྱུད་རིམ་དེ་དང་འབྲེལ་ནས་ཀློག་པ་པོའི་སེམས་ཁམས་སུ་དགའ་སྐྱོ་ཁྲོ་ཆགས་ཀྱི་རྣམ་པ་ཅི་རིགས་སུ་འགྱུར་བའི་སྣེ་ནས་སྒྲུང་དེའི་དོན་སྙིང་གི་ལམ་སྟོན་གང་ཞིག་གནང་ཡོད་པ་དེ་ཆུ་སྐྱར་ཞི་ལ་རྐུ་ཆགས་བཞིན་ཀློག་པ་པོའི་སེམས་ངོགས་སུ་འཛུལ་ཐུབ་པ་ཞིག་དགོས། སྤྱིར་སྒྲུང་རྩོམ་འབྲི་བར་མེད་དུ་མི་རུང་བའི་ཆ་རྐྱེན་ཁག་གཅིག་ངོ་སྤྲོད་ཞུ་ན། ༡༽ བསམ་བློའི་ཐོག་ནས་བརྗོད་གཞི་བཀོད་སྒྲིག་བཟོ་སྟངས་དེ་ཡིན། རྩོམ་པ་པོ་རང་ཉིད་ཀྱིས་སྤྱི་ཚོགས་ཁྲོད་ཀྱི་སྐྱོན་ཡོན་གང་ཞིག་ཐེར་འདོན་བྱེད་འདོད་པ་དེ་ཉིད་སྒྲུང་གི་མི་སྣ་གང་འདྲ་ཞིག་དང་སྡེབ་ན་འཚམས་མིན་ཐག་གཅོད་བརྟན་པོ་བྱེད་དགོས། ༢༽ མི་སྣ་གང་ཞིག་བཀོད་སྒྲིག་བྱས་པ་དེ་དང་འཚམས་པའི་ཀུན་སྤྱོད་བཟོ་དབྱིབས་ངོ་སྤྲོད་ཀྱི་ཐོག་ནས་མི་སྣ་དེའི་བསམ་བློའི་རྒྱུད་རིམ་ཡང་རང་བཞིན་གྱིས་ཤེས་ནུས་པ་དགོས། ༣༽ མི་སྣ་དེ་དང་དེ་དག་ཕན་ཚུན་འགྲོ་འདུག་གི་བྱ་སྤྱོད་སྤེལ་སའི་ཁོར་ཡུག་གི་མཛེས་བཀོད་སོགས་ཀྱང་སྒྲུང་གི་མི་སྣའི་བསམ་བློ་དང་བསྟུན་ནས་བཀོད་སྒྲིག་ཞིབ་ཚགས་བྱེད་དགོས། ༤༽ སྒྲུང་གི་གྲུབ་ཆ་གཙོ་བོ་ནི་སྐད་ཆ་ཡིན། དེར་བརྟེན་སྐད་ཆའི་རྒྱུད་རིམ་ལ་རྩོམ་པ་པོའི་ཞེ་འདོད་ཀྱི་དམིགས་ཡུལ་གང་ཞིག་ཡོད་པ་དེ་མ་བརླགས་པའི་ཐོག་ནས་ཀློག་པ་པོའི་བསམ་བློར་དགའ་སྐྱོ་ཁྲོ་ཆགས་ཀྱི་རྣམ་པར་བསྒྱུར་ཐུབ་པའི་སྒྱུ་རྩལ་གྱི་ཉམས་ཐོན་ཐབས་བྱེད་དགོས། ༥༽ སྒྲུང་རྩོམ་གང་ཞིག་བྲིས་པ་དེའི་རྙོག་འཛིང་གི་རྒྱུད་རིམ་ཁྲོད་ནས་སྤྱི་ཚོགས་ནང་གི་ངན་པ་ཐེར་འདོན་དང་བཟང་པོ་ངོ་སོ་ཐོན་པ་སོགས། མདོར་ན་སྤྱི་ཚོགས་ཀྱི་ཁ་ཕྱོགས་ལ་ཕན་ཐོགས་པའི་བླང་བྱ་གང་ཞིག་ཀློག་པ་པོའི་བསམ་བློའི་ནང་མི་མངོན་པའི་སྒོ་ནས་ཆུ་སྐྱ་བྱི་ལ་རྐུན་ཆགས་བཞིན་འཛུལ་ཐུབ་པ་ཞིག་དགོས། (འགྲེལ་བརྗོད་ངག་ཐོག་ནས་ཞུ་རྒྱུ)།
སྒྲུང་རྩོམ་འབྲི་ཐབས།
སྒྲུང་རྩོམ་ནི་སྙན་ངག་དང་ལ་གཞས་ལྟ་བུའི་སྟབས་བདེའི་འབྲི་སྟངས་ཤིག་མིན་ལ། ལྷག་པར་དུས་རབས་ཀྱི་ཁྱད་ཆོས་ལྡན་པའི་སྒྲུང་རྩོམ་ནི་དེ་བས་ལས་སླ་བ་ཞིག་མིན། འདིར་སྒྲུང་རྩོམ་འབྲི་ཐབས་ཀྱི་དྲན་གསོ་ཕྱོགས་སྡོམ་རེ་གཉིས་ཙམ་ཞུ་ན། ༡༽ གཞན་གྱིས་བྲིས་པའི་སྒྲུང་རྩོམ་ཡང་ཡང་ཀློག་དགོས། དེས་རང་ཉིད་ཀྱིས་མ་ཤེས་པ་དང་མ་གོ་བའི་གནས་ལུགས་མང་པོ་ཞིག་རང་ཉིད་དངོས་སུ་སོང་ནས་དཀའ་སྤྱད་འབད་བརྩོན་མི་དགོས་པར་ཤེས་རྟོགས་ཐུབ་པ་མ་ཟད། ཚིག་སྦྱོར་གྱི་ཐོག་ནས་སྒྱུ་རྩལ་གྱི་ཞག་རྩི་ལྡན་པའི་མཚོན་སྟངས་ཀྱང་མང་པོ་ཞིག་ལས་སླ་བོའི་སྒོ་ནས་ཤེས་ཐུབ། ༢༽ རྒྱུན་ལྡན་ནས་མི་གཞན་གྱི་སྐད་ཆ་ལ་དངོས་ཤུགས་རྒྱུད་གསུམ་ནས་དོ་སྣང་ཤུགས་ཆེར་སྤྲོད་རྒྱུ་མ་ཟད། སྐད་ཆའི་རྒྱུད་རིམ་དང་། སྐད་ཆའི་བབ་དང་བསྟུན་ནས་བཤད་མཁན་གྱི་རྣམ་འགྱུར་ལའང་དོ་སྣང་ཤུགས་ཆེར་བྱེད་དགོས་ལ། ད་དུང་སྐད་ཆ་དེ་དག་གི་རྒྱུད་རིམ་དང་འབྲེལ་བ་ཡོད་པའི་གནས་ཚུལ་ལ་ཕུད་དེད་གཏོང་ནས་བརྟག་དཔྱད་ནན་པོ་གནང་དགོས། དེས་རང་ཉིད་ཀྱི་སྒྲུང་རྩོམ་གང་ཞིག་བྲིས་ནའང་། དེ་དག་དང་འབྲེལ་བའི་རྒྱུད་རིམ་དང་མི་སྣ་ཞིབ་འབྲི། སྒྲུང་གི་རྒྱུད་རིམ་ལ་ཞིབ་ཚགས་ཀྱི་རང་བཞིན་དང་གནས་ལུགས་གཏིང་ཟབ་ཏུ་གཏོང་པའི་ཁྱད་ཆོས་ལྡན་ཐུབ། ༣༽ ས་གནས་གང་དུ་སླེབས་ཀྱང་དེའི་གནམ་གཤིས་འགྱུར་འགྲོས་དང་། རི་ལུང་ཆགས་དབྱིབས། ཐ་ན་ཁང་པའི་ནང་གི་ནང་ཆས་སྒྲིགས་བཀོད་ཚུན་ཆད་རྒྱུན་ལྡན་ནས་དོ་སྣང་ནན་པོ་གནང་དགོས། དེས་རང་ཉིད་ཀྱིས་སྒྲུང་རྩོམ་གང་ཞིག་འབྲི་ནའང་། སྒྲུང་རྩོམ་དེ་དང་འབྲེལ་བའི་ཁོར་ཡུག་གམ་མཛེས་བཀོད་དམ་ཡང་ན་ཡུལ་ལྗོངས་བཀོད་སྒྲིག་གི་ཆ་ལག་ཐད་དཀའ་ཚེགས་མེད་པའི་སྒོ་ནས་འབྲི་ཐུབ། ༤༽ གོང་གསལ་ཆ་རྐྱེན་དེ་དག་ལ་དོ་སྣང་ཤུགས་ཆེ་དང་འབྲེལ་བའི་སྒོ་ནས་རྒྱུན་ལྡན་ཤོག་སྨྱུག་གཉིས་པོ་ཆོས་གྲོགས་དང་། ཡང་ན་དགའ་གྲོགས་ནང་བཞིན་རང་ལུས་དང་མི་འབྲལ་བར་མ་ཟད། ཆོས་གྲོགས་དང་དགའ་གྲོགས་ནང་བཞིནཉིན་རེ་ནས་ཉིན་རེར་ཕན་ཚུན་བཟང་ཕྱོགས་ཀྱི་འབྲེལ་བ་ཁ་ཇེ་དམ་དང་གཏིང་ཇེ་ཟབ་ཏུ་གཏོང་དགོས། དེས་རང་ཉིད་ཀྱིས་སྒྲུང་རྩོམ་གང་ཞིག་འབྲི་ནའང་། ཁ་བཤད་སྐམ་པོའི་སྒྲུང་མཁན་ཙམ་དུ་མི་འགྱུར་བར། རིག་པ་དཔེ་ཐོག་འགོད་པའི་ལག་ལེན་དངོས་ཀྱི་ཐོག་ལ་ཁ་ཡོད་ལག་ཡོད་ཀྱི་དོན་སྙིང་ལྡན་པའི་སྒྲུང་རྩོམ་འབྲི་མཁན་ཞིག་ཏུ་འགྱུར་ངེས། ཁ་བཤད་ཆུ་ཡི་ལྦུ་བ་དཀར་པོ་ཡིས།། བ་ཚྭའི་ཁྲོན་པ་སྟོང་ཕྲག་ཁེངས་ན་ཡང་།། རིག་པ་དཔེ་ཐོག་འགོད་པའི་ལག་ལེན་གྱི།། གསེར་གྱི་ཐིགས་པ་གཅིག་གི་དོ་ཟླ་མིན།།
(མཆོད་མེའི་པར་དབང་གཞན་གྱིས་དགོལ་མི་རུང)

རྩོམ་ཁུངས། ངེད་དྲ་བ།
