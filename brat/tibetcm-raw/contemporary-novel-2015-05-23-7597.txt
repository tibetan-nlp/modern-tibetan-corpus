གྲྭ་ལོག
སྤྱང་ལྟོགས་ཁྲིད་རྒོད།

ཞོགས་པའི་བསང་དྲི་དང་ལྷན་དུ་མུ་ཏིག་དགོན་པའི་ཨ་ཁུ་བསྟན་འཛིན་གྱིས་དུས་རྒྱུན་གྱི་ཁ་དོན་འདོན་ཞོར་མིག་གིས་ཅི་ཞིག་ལ་རེ་སྒུག་བྱེད་པ་ནང་བཞིན་དྲ་མ་ལས་རྒྱང་རིང་གི་རྨ་ཁུགས་སྡེ་བར་བལྟས་ཏེ་ལག་གི་དྲིལ་བུ་དཀྲོལ་གྱིན་འདུག དྲིལ་བུ་ཡི་སིལ་སྒྲ་སྙན་མོ་དེའང་སྐྱིལ་སེར་རླུང་དང་རྡེབ་ནས་རྨ་ཁུགས་སྡེ་བའི་ཕྱོགས་སུ་འཐིམ། “ཨ་ཁུ་སྐྱ་རྡོ་ང་སྒྲོལ་མ་ཡིན། ཁ་སང་སློབ་གྲྭ་གཏོར་བྱུང་། ཉིན་གཅིག་ལའང་སྒུག་མ་འདོད་པར་ཁྱོད་ལ་བལྟ་རུ་ཡོངས་བ་ཡིན།” ཞེས་སྒྲོལ་མ་དབུགས་འཚངས་འཚངས་ངང་ལག་ཏུ་སིལ་ཏོག་ཁུག་མ་གང་ཁྱེར་ནས་ནང་སྒོ་ལས་འཛུལ་ཡོང་། ཨ་ཁུ་བསྟན་འཛིན་གྱི་གདོང་མ་གཞི་ནས་དམར་ནག་ཅིག་ཡིན་ལ། སྒྲོལ་མའི་སྐད་ཆ་དེ་ཐོས་མ་ཐག་ཇེ་དམར་ནས་ཇེ་ནག་ཏུ་རྒྱུར་ཏེ་ལི་མདོག་ལྟ་བུ་ཞིག་ཏུ་གྱུར་བྱུང་། ཨ་ཁུ་བསྟན་འཛིན་དགུན་གྱི་ཁུ་བྱུག་བཞིན་དལ་ཅག་གེར་བསྡད་དེ་ལག་གི་དྲིལ་བུ་དེའང་དཀྲོལ་མཚམས་བཞག་ནས་སྙིང་གི་ལྡིང་ཚད་ཀྱང་སུ་ཞིག་གིས་རྐང་རྩེད་སྤོ་ལོ་ལ་རྡོག་རྫིས་བྱེད་པ་བཞིན་ལྷང་འཇགས་ཀྱི་ཁོར་ཡུག་དེ་ངང་གིས་དཀྲོག་བྱུང་། “དགའ་བསུ་བྱེད་མི་འདོད་དམ། དགའ་བསུ་མི་བྱེད་ན་ང་ཕྱིར་འགྲོ་རྒྱུ་ཡིན”ཞེས་སྒྲོལ་མའི་ཐེ་ཚོམ་ཅི་ཡང་མེད་པར་དེ་ལྟར་ལབ། ཨ་ཁུ་བསྟན་འཛིན་གྱིས་ངག་ལྐུགས་པའི་བྱིས་པ་བཞིན“མ་་་་་་རེད་་་་་་དགའ་་་་་བསུ་་་་་་ལོས་བྱེད”ཅེས་བཤད་ཞོར། གདོང་ལས་བཅོས་མིན་གྱི་འཛུམ་ཞིག་བསྟན། ཉི་མ་ནི་རི་རྩེ་ལས་ཤར་ཏེ་ཞོགས་པའི་སྨུག་སྤྲིན་དང་རྡེབ་ནས་མཚམས་སྤྲིན་དང་བཅས་མུ་ཏིག་དགོན་པར་ཤར་བྱུང་། “ངས་འདི་ནས་ཁྱོད་ལ་རེ་སྒུག་བྱས་ཏེ་རྒྱུན་དུ་རྨི་ལམ་དུའང་རྨིས་ཀྱིན་འདུག” “ཁྱོད་ཀྱི་སེམས་སུ་ང་ནི་སེམས་མེད་པའི་ཤིང་ཐོ་ཞིག་རེད་འདོད་དམ” “དེ་འདྲ་ཞིག་ག་ལ་ཡིན་སློབ་ཆེན་ནང་དུ་རྒྱུ་དང་འབྱོར་བ། ཕོ་སྒོ་སོགས་གང་ཐད་ནས་རྒྱས་པའི་གསར་བུ་མང་བོ་འདུག་པས” “ཁྱོས་དེ་ལྟར་བཤད་ན་ང་ནི་འཆལ་མོ་ཞིག་་་་་་་་་་་་་་་་” “་་་་་་་་་་་་་་་་་་” “་་་་་་་་་་་་་་་་་” “ས་རུབ་ལ་ཉེ་གི། ཁྱོད་ཡུལ་ལ་འགྲོ་རན་འདུག” “ཅི་རེད་འགྱོད་པ་སྐྱེས་ནི་མིན་ན། སྐབས་ཐོག་དེར་འཛུམ་མདངས་ཁྲོད་གཟན་ཤམ་ཡར་ལ་རྫི་མཁན་དེ་ཁྱོད་རང་མིན་ནམ།་་་་་་་་་་”ཅེས་བཤད་ཞོར་མུ་ཏིག་དགོན་པ་ནི་གྲམ་པའི་ཆུ་སྒྲ་དང་བསང་དྲིའི་ཁྲོད་ནས་མཚམས་ལན་རེར་བན་ཆུང་ཞིག་གི་འཁུན་སྒྲ་མོད་ལ། མཚམས་ལན་རེར་བུ་མོ་ཞིག་གི་ངུ་སྒྲ་མོད་པའི་སྐད་དབྱངས་དེ་དྲིལ་བུའི་སིལ་སྒྲ་དང་ལྷན་དུ་མཁའ་དབྱིངས་སུ་འཐིམས་སོ།།
(མཆོད་མེའི་པར་དབང་གཞན་གྱིས་དགོལ་མི་རུང)

རྩོམ་ཁུངས། ངེད་དྲ་བ།
