ས་དམར་གྲགས་པའི་སྒྲུང་ཐུང་། ཆང་མ་གྲུ་དཔོན།
ས་དམར་གྲགས་པ།


འཁྲུངས་ལྷ་ཡར་གསོལ་འགྲོ་སའི་ལམ་དེ་ནས། ང་ལ་ཏིག་ཙམ་བསྒུག་གནང་དང་། སེ་ར་ནས་ལྷ་སར་ཡོང་སའི་ཞིང་ཁ་དེ་ན་ངའི་ཨ་མའི་ཕྱག་ཕྲེང་ལུས་ཡོད། “བདེ་སྟོང་ལགས། ཁྱེད་རང་གིས། དེ་ལྟར་ཨ་སྟོང་བརྒྱབ་པ་ལྟར་མ་སྣང་། ང་མགྱོགས་པོར་ལྷ་སར་ཡོང་གི་ཡིན། རྫ་མ་དྭངས་དཀར་ཁ་ཡི། རྡོ་བའི་ཁ་ལེབ་དེ་གཡོར་མཁན་སུ་རེད། ”ཀོ་གྲུ་ཕར་ལ་དེད་ཀྱང་། རླུང་གིས་ཕྱིར་ལ་འཐེན་སོང་། ཐག་རྡང་སྟེང་གི་རྨི་ལམ་ལྟ་བུའི་ལས་ཀྱི་སྣམ་སྨིན་ཡོད་ཚད། ཆུ་ལ་དཔར་བའི་གྲིབ་གཟུགས་བཞིན། ང་ཡི་མིག་ལ་དཔར་ཞིང་། རྣ་ཡི་བུ་ཀར་འཛུལ་ཏེ་མགོ་ནང་ནས་ཀི་འདེབས། བཀོད་མ་དེ་ལས་བབས་པའི་བདག་གི་མིག་ཆུ་དེ་མ་སྐམ་སྔོན། ངས་རང་གི་རྐང་པ་ཞོན་ནས་ལྷ་སར་འགྲོ་ཡིན་ཡོད། “ཆང་མ་ཚང་གི་སྒོ་དར་མཐོང་ཨེ་བྱུང་། བདེ་སྟོང་ལགས།དེ་ན་ལུས་ཀྱི་ཤུན་སྐོགས་ལྟ་བུའི་ཁྲུང་རྐྱན་དང་། འབྲུ་ཡི་སྙིང་ཁུ་ཡོད། ལྷ་ལ་ཕུལ་བ་མ་རེད། ལྷ་ཡིས་གནང་བ་རེད། དེ་ལ་ཞེད་སྣང་གིས་ག་རེ་བྱེད་ཀ། ཆང་འཐུང་བ་ནི་རང་གིས་རང་ལ་འདྲི་གཅོད་བྱས་པ་དང་འདྲ་བས། སྙན་ངག་གི་ཀོ་གྲུ་དེ། ངས་ལྷ་གདོང་གྲུ་ཁ་ནས་བཅོལ་ཏེ། ལྷ་སར་ཆང་འཚོང་གི་ཡོང་ཡོད། བདེ་སྟོང་ལགས། ཁ་བ་མ་བབས་པའི་སྔོན་ལ་ཆང་རྒྱ་མ་གང་ངས་ཁྱོད་ལ་འཚོང་། ཁྱོད་ཀྱིས་རང་གི་དུས་ཚོད་བཏང་ནས། ལྷ་གདོང་གྲུ་ཁའི་ལམ་དེ་བཙལ་བར་སོང་། དུས་ཚོད་དེ་འདྲ་རིང་པོ་ཞིག་ཡོད་པའི་དབང་དུ་བྱས། ང་རང་བཟི་དགོས་པའང་མ་རེད། སྣང་དོགས་དེ་འདྲས་ག་རེ་བྱེད་ཀ། ཆང་འཚོང་མཁན་དེ་ཁྱོད་རང་གི་བློ་དྲི་སའི་ཤན་པ་དེ་རེད། ལྷ་ས་ནས་མར་ཚོང་བྱེད་མཁན་གྱི་ཨ་མདོ་བ་དེ་དག་མ་རེད།” ཆང་མ་བཏུང་པའི་སྔོན་དུ་ལག་པ་རང་གི་ལུས་པོར་ཕྱིས། ངག་ནས་བབས་པའི་ཁ་བ་དེར་མིག་གིས་རྭ་བརྩིགས། ལྷ་འདྲེ། “ངས་གཡུ་ཆུང་གྲུ་དཀར་དེ་ཁྲིམ་གཟིགས་སྒང་ནས་བོར་སོང་ཟེར་དགོས་པ་མ་རེད། གྲུ་དཔོན་གྱིས་ཆང་མ་འཚོང་བར་གཡུ་བཙོངས། ཁོ་བླ་གྱར་བ་རེད། བདེ་སྟོང་ལགས། གཏིང་ཁུང་ཆུ་མིག་ཟེར་བ་དེ་ཐག་རིང་མ་རེད། ལོ་རྒྱུས་ལྟ་བུའི་ཤོག་ལེ་རྙིང་པ་ཞིག་གི་སྟེང་འོག་གཉི་ག་ཡང་མ་རེད། ཆུ་ཧུབ་གང་རེད། མཐའ་ན་རྫོགས་མཐའ་མེད་པའི་ལས་ཀྱི་རྫིང་བུ་དེ་རེད། ”ཀོ་གྲུ་འདོགས་བྱེད་ཀྱི་ཐག་པའི་སྣེ་མོར། ཀླད་པའི་ནང་དུ་འཁྱིལ་བའི་རྒྱ་མཚོ་དེ་དང་། དེ་ལ་ས་གཏོར་བྱེད་ཀྱི་ལྕགས་ཁམ་སོགས་ཀོ་གྲུའི་སྣེ་ལ་འཇུས་ཏེ། རྒྱ་མཚོར་བགྲོད་པར་གསུངས་བྱུང་། འོན་ཏང་། ང་རང་གཅིག་པུས་ཆང་ཉོ་མཁན་ཞིག་ལ་བསྒུག། ལོ་གཅིག་གི་སྔོན་ལ། ངས་ཁོ་རང་རྒྱ་གར་ལམ་ནས་བསྒུག་པར་དམ་བཅའ་བཞག། དགོངས་དག། ལྷ་འདྲེ། བདེ་སྟོང་ལགས་ནི་ཚོང་པ་ཞིག་རེད། རུས་པ་ཕར་བཙོངས་ཚུར་ཉོ་བྱེད་མཁན་གྱི་ཚོང་པ་ཆུང་ཆུང་ཞིག་རེད། རྗེས་སུ་མགྲོན་ཁང་ཞིག་གི་སྦྱིན་བདག་ཏུ་གྱུར། ཁོང་ལ་སྐད་ཆ་བཤད་དུས་ལགས་མ་ཟེར་ཚེ། ཁོང་གི་མིག་ནས་མེ་འབར་ཞིང་ངག་ནས་རྟུག་དྲི་ཁ། ངས་ཁོང་ལ་དུས་རྒྱུན་བདེ་སྟོང་ལགས་ཅེས་འབོད་བཞིན་ཡོད། ལྷ་འདྲེ་ནི་ངའི་གྲོགས་པོ་ཞིག་གི་མིང་སྟེ། ཁོང་ནི་གྲུ་དཔོན་ཞིག་ཡིན་ལ། ད་ལྟ་གྲུ་གཏོང་གི་མེད། ཁོང་གིས་མ་གཞི་ནས་གྲུ་གཏོང་མྱོང་མེད། འོན་ཀྱང་ཁོང་ལ་ཀོ་གྲུ་ཆུང་ཆུང་ཞིག་ཡོད། ལག་པ་ཞིག་ལས་ཅུང་ཆེ་ལ། མི་རྒན་ཞིག་གི་ལྷམ་འདྲ་ཞིག་ཡོད། ཁོང་ནི་གྲུ་ལ་དགའ་བས་ང་ཚོས་ཁོང་ལ་ལྷ་འདྲེ་ཞེས་དུས་རྒྱུ་འབོད་བཞིན་མེད། གྲུ་དཔོན་ཞེས་འབོད་བཞིན་ཡོད། རྒན་པ་ཚོས་ལྷ་གདོང་གྲུ་ཁ་སྔོན་མ་ཁོ་པ་ཚང་གི་རེད་ཟེར་སྲོལ་ཡོད་ཅེས་བཤད་ཀྱི་ཡོད། ངས་ཞུས་པའི་ཆང་ཁང་དེ་ཕོ་བྲང་གི་ཞོལ་ན་ཡོད་པའི་སྲིད་བློན་དེ་བ་ཚང་གི་མ་རེད། བར་སྐོར་གྱི་རྒྱ་གར་ལམ་ན་ཡོད་པའི་སྒོ་ཟྭ་མེད་པའི་ཆང་མ་དེ་བ་ཚང་རེད། གླིང་ཚང་ཡོད་སའི་བར་དེར་རེད། ང་རང་སོང་མྱོང་མེད། མཚན་ལ་གནས་ཚང་གཡོར་ས་ཙམ་གྱི་དོད་དུ། ང་དེར་འགྲོ་དགོས་དོན་མེད་པར་འདོད། འོན་ཀྱང་ལྷ་འདྲེ་དེར་འགྲོ་རྒྱུ་དགའ་བས། ཆང་འཐུང་དུས་དེར་འགྲོ་དགོས། ང་མི་འགྲོ་བའི་རྐྱེན་གྱིས། ལྷ་འདྲེ་ང་ལ་དགའ་བོ་དེ་འདྲ་མེད་ཟེར་མོད། འོན་ཀྱང་ཆང་གིས་བཟི་རྗེས་ང་དུས་རྒྱུན་ཆང་ཁང་དེར་འགྲོ་ཡིན་ཡོད། འོན་ཀྱང་ངས་ང་སོང་མེད་ཟེར་གྱི་ཡོད། ཁོང་ཚོས་ང་སོང་ཡོད་མ་རེད་ཟེར་གྱི་ཡོད་རེད། ང་སོང་མ་མྱོང་། “འཁྲུངས་ལྷ་ཡར་གསོལ་གྱི་མཐའ་བཞི་རུ་གནས་བཅས་པའི་བྱེ་མའི་ཕུང་པོ་དག་རྡུལ་དུ་གྱུར་བ་ནས། དུ་བའི་སྣེ་མོ་ཕུ་ཐུང་དུ་འཛུལ། འཇིག་རྟེན་གྱི་འགྱུར་བ་མ་ལུས་ནུབ་ཀྱི་སྤྲིན་པ་སྤར་མོ་གང་དང་འདྲ་བར་རླུང་དང་ཆར་བ་གང་གིས་ཟས་པ་མ་ཡིན་པར། རང་བཞིན་གྱིས་མདུད་པ་གང་གི་སྣེ་མོ་དེ་གྲོལ་སོང་བ་ལྟར། རྐང་ནས་ལྷམ་ལྷུང་། དེ་ཡང་མིག་གིས་ཅེ་རེར་མཐོང་མ་སོང་། ཡུས་ཆེ་ཙམ་གྱིས་རྒྱལ་སོང་བ་ཙམ་མོ།།”ལྷ་འདྲེ་ལགས་ཀྱིས་ལྷ་གདོང་གྲུ་ཁ་ནས་ང་ལ་བསྐུར་བའོ་ཞེས་བྲིས་པའི་སྐད་འཕྲིན་ཞིག་ཉིན་དེར་ང་ལ་འབྱོར། ནུབ་ཁུལ་གྲོང་གསར་གྱི་ཆང་ཁང་ག་གེ་མོའི་ནང་གི་ལག་པ་ནག་པོ། སྐྲ་རིང་མོ། མིག་དམར་པ། ཆང་དྲི་འཚུབ་འཚུབ། ངག་ལྡབ་ལྡིབ། ལུས་འདར་འདར། སྣ་ཁུང་དུ་དུ་བ་ལྷོག་ལྷོག། འདི་ལྟར་ཚིག་གི་སྣེ་ཟིན་ཙམ་བྱས་ཏེ“ཆང་མ་ཆང་གིས་བཟི་སོང་། ཨ་ཕོ་ཁྱེད་ཅག་གང་དུ་འགྲོ། ངས་རང་གི་རུམ་ཁ་མེ་ཡིས་དྲོས་ཏེ་ལུས་ལ་ཆང་དྲི་ཞུགས་ཡོད། འདྲ་ཆགས་པོ་མ་གནང་དང་། ཨ་ཅག་གཞིས་ཀ་རྩེ་ནས་པཱ། ཉམས་བཟང་མི་འདུག་གམ། ཨ་ཕོ་ཚུར་ཤོག་དང་། ངའོ་ཉམས་པས་ཅི་བྱ། རྟུག་དྲི་ཚུབ་ཚུབ་ཀྱིས་པས་མགོའང་ན་བྱུང་མོད། བཟོ་ལྟ་ཡག་པོ་འདུག་གིས། ཆང་གིས་བཟི་ན་ཚང་མ་བརྗེད་འགྲོའོ།།”ཆང་མ་ལགས་ཀྱིས་གྲུ་དཔོན་ལགས་ལ་འདི་ལྟར་བཤད། “ཁྱེད་རང་པེ་ཅིང་ནུབ་ལམ་ནས་ལྷོ་ལམ་ལ་ཨེ་སོང་། བདེ་སྐྱིད་ལམ་ནས་ཅང་སོའུ་ལམ་ལ་ཨེ་སོང་། སྲང་ལམ་གཉིས་པར་ཨེ་སོང་། རྡོ་སེང་གེ་ལམ་ན་ལྷ་ས་ནས་གཞིས་རྩེ་དང་ལྷོ་གནས་ལ་འགྲོ་མཁན་གྱི་མོ་ཊ་ཡོད་རེད། དགའ་ལྡན་ལ་མ་སོང་ཚེ་ང་རང་ཡོད་སའི་ལྷ་གདོང་གྲུ་ཁར་ཡོང་མི་དགོས་པ། གསལ་པོར་མཁྱེན་པར་གྱིས། བུམ་པ་རིའི་དབུ་ལ་སྐྱེས་པའི་ལྕགས་ཀྱི་རལ་བ་དེའི་འོག་ན་ང་ཡོད་རེད། རླུང་གིས་དེད་པའི་བུ་མོ་དེའི་མཛེས་པ་དང་མི་སྡུག་པ་ཡོད་རེད། ” ཅེས་གྲུ་དཔོན་གྱིས་བདེ་སྟོང་ལགས་ལ་འདི་ལྟར་བཤད་སོང་། བདེ་སྟོང་ལགས་ཀྱིས་ཀླད་པའི་ནང་དུ་འཁོར་བ་ནི་ལྷ་གདོང་གྲུ་ཁའི་ཉེ་ས་ཞིག་ནས་གཡུ་དང་གཟི་སོགས་གང་མང་ཞིག་དེ་དུས་ཤིག་ལ་ཆུར་བསྐུར་བ་དེ་རེད། གྲུ་བས་འདི་ལྟར་བཤད་དོན་ཁོང་ཚང་ལ་དེ་དུས་རྒྱུ་ཆེན་པོ་ཡོད་ཟེར་འདོད་པ་དེ་རེད། ངེད་ཀྱི་ཁྱིམ་མིང་ལ་ལྷ་གདོང་གྲུ་ཁའི་དཔོན་ཚང་ཞུས་པ་དེ་རེད། ཆང་མ་དང་གྲུ་དཔོན་ངའི་ཨ་མ་ལགས་དང་ཕ་ལགས་རེད། ངའི་མིང་ལ་བུ་ཆུང་བ་བདེ་ལེགས་ཟེར་གྱིས། ངའི་མིང་ལ་བུ་ཆུང་བ་ཟེར་དོན་ངའི་ཕ་ལགས་དང་ཨ་མ་ལགས་མཉམ་དུ་སྡོད་པའི་སྐབས་སུ་ཁོང་ལ་མཛེས་མ་གཞན་ཞིག་སྐྱེས་ཡོད་ཅེས་ཨ་མའི་སྔོན་ནས་ཕ་ལགས་ར་བཟི་རྗེས་ཁ་ཤོར་བ་རེད། དེ་ནས་ཨ་མ་ལགས་ཀྱིས་ཀྱང་བོད་ལྗོངས་སློབ་ཆེན་གྱི་དགེ་རྒན་ཆེན་མོ་མིང་ལ་རཱ་ནཱ་མཱ་ཟེར་བའི་ཕྱི་གླིང་བ་ཞིག་ཡོད་པ་དེ་ངའི་ཕ་ལགས་རེད་ཅེས་ཕ་ལགས་ལ་ཞུས་པ་རེད། དེ་ནས་ངའི་ཕ་ལགས་ཀྱིས་བྱིས་པ་དེ་འདི་མ་རེད། དེ་སྐྱེས་མ་ཐག་མི་གཞན་ལ་སྤྲད་པ་ཡིན། བྱིས་པ་འདི་ཉིན་ལྟར་དབུལ་བོ་ཞིག་གིས་སྨན་ཁང་ནས་སྐྱེས་མ་ཐག་བསྐྱུར་ཏེ་བྲོས་སོང་བས་ངས་ཁྱེར་ཡོང་ནས་རང་གི་བུ་རེད་ཅེས་ཞུས་པ་ཡིན་ཞེས་བཤད། ང་རང་གིས་ཁོང་གཉིས་ལ་ཅི་ཡང་ཟེར་མ་མྱོང་།ཉིན་དེ་ནས་ངས་ཁོང་གཉིས་ལ་ཆང་མ་དང་གྲུ་དཔོན་ཞེས་བོས་པ་ཡིན། ཁོང་གཉིས་ཀ་དགའ་ཐག་ཆོད་སོང་། གོང་གི་སྲང་ལམ་དེ་དག་ན་ཆང་མ་དང་གྲུ་བ་གཉི་གར་གྲོགས་པོ་མང་པོ་ཡོད་རེད། དེ་དག་ནི་ཇོ་ཁང་སྒོ་ཁའི་གསེར་ཡིག་འབྲི་མཁན་དང་། ཁ་བཏགས་འཚོང་མཁན། མར་འཚོང་མཁན། ལྷམ་དབྱི་མཁན། གོས་ལྷམ་འཚོང་མཁན། མགྲོན་ཁང་སྦྱིན་བདག། གསེར་དངུལ་རྫུན་མ་སོགས་འཚོང་མཁན་སོགས་རེད། འོན་ཀྱང་ཆང་མ་དགའ་ས་དེ་གྲུ་དཔོན་མ་རེད། གྲུ་དཔོན་གྱི་གྲོགས་པོ་མགྲོན་ཁང་གི་སྦྱིན་བདག་བདེ་སྟོང་ལགས་རེེད། གྲུ་དཔོན་དགའ་སའི་བུ་མོ་དེ་བདེ་སྟོང་ལགས་ཀྱི་མགྲོན་ཁང་སྒོ་ཁའི་སྒོ་ཁྱིམ་དང་པོའི་ནང་གི་བུ་མོ་ཆོས་སྒྲོན་ལགས་རེད། ལོ་དེར་ངས་བུ་མོ་ཆོས་སྒྲོན་ལ་འོ་ཞིག་བྱས་པས་གྲུ་དཔོན་གྱིས་ང་ལ་གཤེ་གཤེ་མང་པོ་བཏང་སོང་། དེའི་ཕྱི་ཉིན་ངས་བུ་མོ་ཆོས་སྒྲོན་ཁྲིད་ནས་ལྷ་ས་ནས་ལྷོ་ཁར་བྲོས། ཉིན་དེ་ནས་ང་ལ་མི་གཞན་པས་ཁྱིམ་མིང་ལ་ཅི་ཟེར་ཅེས་དྲིས་དུས། ངས་ང་རང་ལྷ་གདོང་གྲུ་དཔོན་ཚང་གི་ཡིན་ཞེས་ཞུས་པ་ཡིན། ཁོང་ཚོས་ང་རང་སྐུ་དྲགས་རེད་ཟེར་གིས། ལོ་དེར་གྲུ་དཔོན་གྱིས་བང་ལ་ཁ་པར་སོགས་ཅི་ཡང་བཏང་མ་སོང་། རྗེས་ནས་གྲུ་དཔོན་་ལྷོ་ཁར་ཡོང་མཆོད་མཇལ་ཡོང་དུས་ངེད་གཉིས་འཕྲད། ཆང་མ་ཕྱི་རྒྱལ་བ་དེ་དང་ལྷན་དུ་པེ་ཅིང་ལ་བུད་སོང་ཟེར། གྲུ་དཔོན་ཁོང་གྲྭ་བ་ཐབ་ཆགས་ཤིག་བྱས་ཏེ་ཡར་མར་ལ་ཕྱག་སྐོར་འགྲོ་བཞིན་ཡོད་ཟེར། མཚན་མོ་དེར་གྲུ་དཔོན་གྱིས་ངའི་ཆུང་མ་ཆོས་སྒྲོན་མཐོང་པས་མཚན་ལ་ང་རང་སྐད་བཏང་ཏེ་ཆང་འཐུང་གི་འགྲོ་ཟེར། དེ་ནས་ངེད་གཉིས་ཆང་ཁང་དུ་སོང་། ང་རང་ར་བཟི་རྗེས་ཁོ་པས་ངའི་ཞྭ་མོ་ཁྱེར་ནས་རྒྱང་ལ་བུད་སོང་ཅེས་ཡི་གེ་ཞིག་བྲིས་ཏེ་ཁྲུང་ཞུ་མའི་ལག་ཏུ་བཞག་འདུག ང་རང་མགྱོགས་པོར་ཁྱིམ་དུ་ལོག་དུས། ཆོས་སྒྲོན་གྱིས་གྲྭ་བ་ཞིག་གི་གཟན་བཀྲུས་ཏེ་བསྡད་འདུག། ཁོ་མོས་ང་ལ་བཤད་རྒྱུར"གྲུ་དཔོན་གྱི་བུ། ཁྱེད་རང་འདི་ལྟར་བསྡད་མ་འདུག། མགྱོགས་པོར་ས་ཆ་ཞིག་ལ་འཚོང་རྒྱག་ཏུ་སོང་ཟེར།" ངས་རང་གི་སྟོ་གོས་ཚང་མ་ཤོག་སྒམ་ཞིག་ཏུ་བླུགས་ཏེ་ཁང་པ་གཞན་ཞིག་གཡོར་དུ་སོང་བ་ཡིན། ཞོར་དུ་ངས་གྲུ་དཔོན་གྱི་གཟན་ཤམ་ཁུག་གི་ཕྱག་ཕྲེང་དེ་ཁ་རོག་གེར་ཁྱེར་སོང་བ་ཡིན། དེ་ནི་ངའི་ཨ་མའི་ཡིན་པས་སོ།། ༢༠༡༦།༡༡།༡༣ཉིན་གཡོས་ཁང་དུ་གྲུབ།
རྩོམ་སྒྲིག་པ། དབང་མོ། (མཆོད་མེའི་པར་དབང་གཞན་གྱིས་དགོལ་མི་རུང)

རྩོམ་ཁུངས། ངེད་དྲ་བ།
