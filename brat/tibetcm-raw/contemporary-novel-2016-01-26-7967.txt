་་་་་་་ དགོང་མོ་དེར་ཟླ་བ་ལས་ངེད་གཉིས་ལ་ལྐོག་ལྟ་བྱེད་མཁན་ཞིག་ཡེ་ནས་མེད་པ་ལྟར། བག་ལྷོད་ངང་མཉམ་དུ་ལས་པའི་ཟ་མ་ཞིག་ལ་མཉམ་དུ་རོལ་བཞིན་བར་ཐག་ཅིག་ལ་འདང་རྒྱག་གིན་བསྡད། དགོང་མོ་དེ་ནི་སྒྲོལ་མ་དང་ཆེས་ཉེ་བའི་ཉིན་མོ་ཞིག་རེད་བསམ་གྱིན་བསྡད། ཟ་མ་ཟོས་ཚར་རྗེས། ངས་སྒྲོལ་མ་དང་མཉམ་དུ་སྒྲོལ་མ་འདོན་འགོ་བརྩམས། སྒྲོལ་མ་ཞབས་བྲོ་འཁྲབ་རྒྱུར་ཧ་ཅང་དགའ། དེ་ངས་ཤེས་ཡོད། འོན་ཀྱང་ཅི་འདྲ་ཞིག་འཁྲབ་མིན་མི་ཤེས། དེའི་རྗེས་ཀྱི་ཏག་ཏག་སྐར་ཚོགས་ཀྱིས་རུབ་པའི་དགོང་མོ་དེ་ལ། ང་དང་སྒྲོལ་མ། ད་དུང་མོའི་གྲོགས་མོ་འགའ་ཞིག་བཅས་ཆང་ཁང་དུ་སོང་བ་ཡིན། དགོང་མོར་དེར་ངས་ཆེས་ཐོག་མར་མོས་ཞབས་བྲོ་འཁྲབ་པ་མཐོང་བ་ཡིན། དགོང་མོ་དེར་ང་ལ་བཟི་ཁ་ཟུང་མའི་སྦྲུལ་ལྕེས་ཤི་མིན་རོ་མིན་ཞིག་ཏུ་བཟོས་སོང་། ཁོ་མོའི་འཇོ་སྒེག་གིས་བཟི་སོང་བའམ་ཆང་འཐུང་བས་བཟི་བ་གང་ཡིན་མ་ཤེས་པར་བྱ་རྨྱང་ཐེངས་ཤིག་དང་བཅས་ཏེ་གཉིད་ཀྱི་འགྲུལ་བཞུད་ཐེངས་གཅིག་ཏུ་གོམ་པ་སྤོས། ཕྱིར་གཉིད་ལས་སད་དུས་མཚན་གུང་གི་ཆུ་ཚོད་གསུམ་ཙམ་དུ་སླེབས་འདུག་པས། མགྲོན་ཁང་ཞིག་ཏུ་ཉལ་དུ་ཕྱིན་པ་ཡིན། མགྲོན་ཁང་ནས་ཅི་བྱས་སོང་བ་ཡིན་ཞེས་རང་གིས་རང་ལ་འདྲི་བ་ཞིག་དྲིས་མྱོང་ལ། ད་དུང་ཡང་འདྲི་བཞིན་ཡོད། མིན། ངེད་གཉིས་རུམ་བྱས་པ་བདེན། ཡིན་ཡང་ངས་ཅི་ཡང་བྱས་མྱོང་མེད། དེའང་བརྩེ་དུང་རྣལ་མ་ཞིག་དང་སེམས་པ་ལྷད་མེད་ཅིག་གི་སྔོན་འགྲོ་ཡིན། དོན་དུ་མོས་དེ་འདྲ་ཞིག་ཏུ་བསམ་ཡོད་མེད་མི་ཤེས། དུས་ཚོད་ནི་ལྕགས་ལམ་དུ་འགྲོས་སྙོམས་སུ་སྐྱོད་པའི་མེ་འཁོར་ཞིག་གི་འགྲོས་རེད། དུང་ཕྲུག་རི་སྒང་གི་སྤང་ཐང་དེའང་བསམ་དབང་དྲན་དབང་མེད་པར་སེར་མདོག་གི་གཞིས་ཀ་ཆུང་ཆུང་ཞིག་ཏུ་གྱུར་སོང་། ད་ནི་སློབ་སྐབས་ཤིག་ལ་བཞི་ཤད་འཇོག་པར་ཐོན་སོང་། དགོང་མོ་དེར་མོས་ང་ལ་ཕ་ཡུལ་དུ་འགྲོ་རྒྱུའི་སྐོར་ནས་ཁ་བརྡ་ཨེན་ཙམ་གླེང་བྱུང་། དེ་ནི་ང་ལ་མཚོན་ན་གྲང་ཤུར་ཤུར་གྱི་གྲང་རླུང་ཞིག་ངའི་བྲང་ཁར་སྣུན་པ་དང་མཚུངས། མཁའ་མཐའི་མཚམས་སྤྲིན་དང་གཤིབས་ནས་འོང་བའི་ཉི་ཟེར་དག་གིས་དབང་རིམ་གྱི་སྤྲིན་པ་གཤགས་ནས་དབྱིངས་སུ་འཕྲོས་པས་འཆར་ཀའི་མདངས་ཡོངས་སུ་དམར་པོར་གྱུར་འདུག སྐབས་དེར་ངས་མོ་རང་ལམ་འཇུག་བྱས་པ་ཡིན། ཁྱེད་འགྲོ་ཁར་ང་ལ་སྐད་ཆ་མང་བོ་ཞིག་མེད། གཅིག་ཡོད་པ་ནི་ཁྱེད་ང་དང་བྲལ་རྗེས། ཡང་ན་ཁྱེད་ཀྱིས་ཕ་ཡུལ་ནས་ང་ལ་འབྲེལ་བ་མང་ཙམ་གནང་རོགས།
