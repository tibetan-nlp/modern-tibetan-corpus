༢༠༡༠ ལོའི《སྦྲང་ཆར》དེབ་གཉིས་པར་སྙན་ངག“རྫུན་མ”བཀོད་འདུག
ཚེ་རིང་རྣམ་རྒྱལ།

སྣ་ཐིག ༢༠༡༠ ལོའི་ཟླ ༤ པའི་ཚེས ༡༤ ཉིན་ཡུལ་ཤུལ་ནས་ས་ཡུམ་བྱུང་མུར།དེར་འབྲེལ་གྱི་རྩོམ་རིག་བྱེད་སྒོ་གསར་བ་ཞིག་བྱུང་།སྙན་ངག་ནི་གཙོ་ཁྲིད་ནུས་པར་གྱུར་ཡོད།《གངས་རྒྱན་མེ་ཏོག》དང《སྦྲང་ཆར》《བོད་ཀྱི་རྩོམ་རིག་སྒྱུ་རྩལ》སོགས་ཀྱིས་ཆེད་སྒྲིག་དེབ་དང་ལེ་ཚན་གྱི་སྙན་ངག་པ་མང་པོའི་གདུང་བ་མངོན། རེད་དེ། 《སྦྲང་ཆར》དེབ་གཉིས་པར་རྐུན་འབྲི་བྱས་པའི་སྙན་ངག“རྫུན་མ”དེས། ཐེངས་འདིའི་ས་ཡོམ་སྙན་ངག་ལ་གྲིབ་ནག་བྱུགས་འདུག རྩོམ་པ་པོའི་གསང་བ། ཡུལ་ཤུལ་ས་ཡོམ་སྙན་ངག་གི་ཁྲོད་དུ《སྐྱེ་རྒུའི་མ་བུའི་བརྩེ་སེམས》 ཞེས་པ་དེ། ཁྲི་ཚང（汶川）གི་ས་ཡོམ་བྱུང་རྗེས་ཀྱི་སྙན་ངག་ཐོག་མ《གཅེས་ཕྲུག་ཨ་མའི་ལག་པར་དམ་པོར་འཇུས（孩子，快抓紧妈妈的手）》 ཞེས་པའི་སྙན་ངག་དང་གཅིག་འདྲར་སོང་འདུག རྒྱའི་སྙན་ངག་འདིའི་རྩོམ་པ་པོ་སུ་ཡིན་ཐད་དུ།ཀྲང་ཧྲུའུ་ཞིན（张书新）རེད་ཟེར་མཁན་དང་།སུའུ་ཧྲན་ཧྲིན་（苏善生）ཧུང་ཝེ་ཀོ（洪卫国）ལུའུ་ལུའུ（柳柳）སོགས་ཡིན་ཟེར་མཁན་བྱུང་བར་མ་ཟད།ཆེས་ཐོག་མར་དྲ་ཚིགས་གང་ཏུ་བུད་པ་དང་དུས་དེབ་གང་ཞིག་ཏུ་བཀོད་པ་ཡང་རྩད་གཅོད་གསལ་བོ་བྱུང་མེད། འོན་ཀྱང་སྙན་ངག་འདི་ཁྲི་ཚང་ས་ཡོམ་བྱུང་དུས་བྲིས་པ་བསྙོན་དུ་མེད་པར་མ་ཟད། རྩོམ་པ་པོ་བོད་རིགས་ཤིག་མིན་པ་ཡང་ཤེས་ཐུབ། ཡིན་ཡང་། ༢༠༡༠ ལོར་ཡུལ་ཤུལ་ས་ཡོམ་བྱུང་དུས། ཁ་བྱང་ལ་འགྱུར་བ་ཅུང་ཙམ་བཏང་བ་དང་། སྙན་ངག་གི་དོན་རྐྱེན་བྱུང་ས་ཡུལ་ཤུལ་དུ་བསྒྱུར་ཅིང་མཇུག་ཏུ་སྐད་ཆ་ཁ་ཤས་བསྣན་རྗེས་《སྦྲང་ཆར》གྱི་ངོས་ནས་མངོན་བྱུང་། རྐུན་འབྲི། རང་རྒྱལ་གྱི་ཁྲིམས་དཔྱོད་ལག་ལེན་ཁྲོད་དུ་རྐུན་འབྲི་བྱས་ཡོད་མེད་ཐག་གཅོད་ལ་རྩ་དོན་གཉིས་ཡོད་དེ།གཅིག་ནི་རྐུན་ཡུལ་གྱི་རྩོམ་ཆོས་ལ《བརྩམས་ཆོས་བདག་དབང》ཁྲིམས་ཀྱིས་སྲུང་སྐྱོབ་ཐོབ་མིན།གཉིས་སུ་རྐུས་རྩོམ“ལུགས་མཐུན་བེད་སྤྱོད”ལས་བརྒལ་ཡོད་མེད་དེ་ཡིན།འདིར་བརྗོད་པའི་ལུགས་མཐུན་བེད་སྤྱོད་ནི་གྲངས་ཀྱི་དབྱེ་མཚམས་ཤིག་ཏུ་ངོས་འཛིན་དགོས་ལ།རང་རྒྱལ་གྱི《དཔེ་དེབ་དུས་དེབ་སྲུང་སྐྱོབ་ཀྱི་ཁྲིམས་སྲོལ་ལག་བསྟར་སྲོལ་ཞིབ་ཚོད་ལྟ་མ》ཞེས་པའི་དོན་ཚན་བཅོ་ལྔ་བར།སྙན་ངག་སྐོར་གྱི་རྩོམ་རིག་ལུང་འདྲེན་བྱས་ན་རྐང་བ་བཞི་བཅུ་འམ་སྙན་ངག་ཧྲིལ་བོའི་བཞི་ཆའི་གཅིག་ལས་བརྒལ་མི་རུང་ཞེས་ཐག་གཅོད་བྱས་ཡོད།《སྐྱེ་དགུའི་མ་བུའི་བརྩེ་སེམས》ཞེས་པའི་སྙན་ངག་འདི།ཁྲི་ཚང་གི་ས་ཡོམ་དུས་སུ་བྱུང་བའི《གཅེས་ཕྲུག ཨ་མའི་ལག་པ་དམ་པོར་འཇུས》ཞེས་པའི་ནང་དོན་དང་བརྒྱ་ཆའི་གོ་ལྔ་ཡན་གཅིག་མཚུངས་གྱུར་འདུག་པས། འདི་ནི་ཡིག་རིགས་མི་མཐུན་ཡང་རྐུན་འབྲི་བྱས་ཡོད་པའི་ལས་དབང་སྒྱུར་དུ་མེད། ཀྲང་ཐེན་ཁན་གྱིས།《ཀྲུང་གོ་རུ་དཔྱད་རྩོམ་རྫུན་མ་འདི་འདྲའི་མང་བའི་རྒྱུ་མཚན་ཅི》ཞེས་པའི་ཁྲོད་དུ།དཔྱད་རྩོམ་རྫུན་མ་མང་པའི་རྒྱུ་མཚན་གཉིས་དྲངས་ཡོད་དེ།གཅིག་ནི་མི་གཞི་ཟེར་བ་མེད་པས།རྩོམ་རིག་ཅེས་པ་གང་།རིགས་གཞན་ནི་ཁོར་ཡུག་གིས་བསྐྱེད་པ་ཡིན་ཞེས་གསུངས་ཡོད།རང་རེའི་བོད་ཀྱི་རྩོམ་རིག་གི་ཁོར་ཡུག་ལ་མཚོན་ན།གཙོ་བོ《སྦྲང་ཆར》དང《གངས་རྒྱན་མེ་ཏོག》སོགས་དུས་དེབ་མཛུབ་མོས་བགྲངས་ནུས་པ་ཙམ་ལས་མེད།དེ་དག་གི་ངོས་སུ་རྩོམ་ཞིག་འགོད་རྒྱུ་ནི་རྒྱན་ཤོག་ཐོབ་པ་ལས་ཀྱང་དཀའ་བར་གྱུར།དེ་བས་ཐེངས་འདིའི་ཆེད་སྒྲིག་ནི་གསར་སྐྱེས་རྩོམ་པ་པོ་ཞིག་ལ་མཚོན་ན་གོ་སྐབས་དགའ་ཞིག་ཏུ་གྱུར་ཡོད་དོ། རྩོམ་ཆོས་བདག་དབང་། རྩོམ་ཆོས་ལ་ཁྲིམས་ཀྱིས་སྲུང་སྐྱོབ་ཐོབ་ཡོད་པས།གང་འདོད་འདོད་དུ་ལུང་འདྲེན་དང་བསྒྱུར་བཀོད།དུས་དེབ་ཏུ་བཀོད་ཆོག་པ་ཞིག་གཏན་ནས་མིན།བཀོད་ཀྱང་དེའི་རྩོམ་པ་པོ་དང་ཡོངས་ཁུངས་སོགས་གསལ་དགོས་པ་སྨྲོས་ཅི་དགོས། འོན་ཀྱང་།རང་རེའི་བོད་ཀྱི་རྩོམ་པ་པོའི་སྡེ་དང་རྩོམ་སྒྲིག་པའི་དཔུང་ཁག་ལ་རྩོམ་ཆོས་བདག་དབང་སྐོར་གྱི་འདུ་ཤེས་ཞན་པ་འདྲ།ད་ལོའི་ཟླ ༣ པའི་ཚེས ༢༣ ཉིན།ཁོ་བོས་བསྒྱུར་བའི་ཧ་བྷོལ་རྒྱང་ཤེལ་གྱིས་ལོ་འཁོར་ཉེ་ལྔར་བླངས་པའི་བར་སྣང་གི་པར་ཚོམས་ཞེས་པ་ཞིག་ཀྲུང་གོའི་བོད་ལྗོངས་དྲ་བར་བཀོད།ཚེས ༢༤ ཉིན་ཀྲུང་གོའི་བོད་སྐད་རླུང་འཕྲིན་དྲ་བ་ཞེས་པའི་འཕྲིན་སྟེགས་སུ་ཧ་བྷོལ་རྒྱང་ཤེལ་སྐོར་གྱི་པར་ཚོམས་དེ་མཐོང་བྱུང་།བདག་གིས་ཁོ་ཚོའི་འགྱུར་དེར་བལྟ་འདོད་ཆེན་པོས་བཀླགས་པར།རྒྱུས་ཡོད་ཀྱི་སྣང་བ་ཞིག་ངང་གིས་ཤར་བྱུང་།དེ་ནས་ཀྲུང་གོའི་བོད་ལྗོངས་དྲ་ངོས་སུ་ཡོད་པའི་རང་གི་འགྱུར་དེར་བལྟས་ཙ་ན་གཅིག་མཚུངས་སུ་སྣང་།གལ་ཏེ་ཀྲུང་གོའི་བོད་སྐད་རླུང་འཕྲིན་གྱི་ཁུངས་སོགས་བཀོད་ཡོད་ན་དེར་སྐྱོན་མི་འདུག ཡིད་ཕངས་པ་ཞིག་ལ་འགྱུར་དེའི་མགོ་མཇུག་བར་གསུམ་དུ་དེའི་སྐོར་གྱི་ཚིག་གཅིག་ཀྱང་མི་འདུག
