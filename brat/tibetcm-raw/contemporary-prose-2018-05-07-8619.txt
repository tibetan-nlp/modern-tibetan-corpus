སྙན་ངག ཁེར་རྐྱང་གི་སྐྱོ་ཡ་གཅིག་པུ།
གཙང་དྲག་ཆར།

མཚན་མོ་རེ་རེར། ངས་སྙན་ངག་གི་མཆོད་པ་བཤམས། ངས་རྒྱང་རིང་རྒྱང་རིང་ཞེས་པར་ལག་བརྡ་ཡང་བཏང་། སྲན་ཚུགས་ཀྱིས་འགྱིང་བའི་རྨི་ལམ་ལྗང་ཁུ་དེས། སྙན་ངག་གི་ཆང་སིལ་ཧུབ་ཀྱིས་འཐུང་ཞོར། འཇིག་རྟེན་འདིའི་སྐྱོ་ཡ་གཅིག་པུ། སྙན་ངག་ནི། བུ་མོ་ཞིག་གི་ལུས་དྲོད་དུ་སྒོམ་ཡོད། ཁ་ཐག་མི་རིང་བའི་ས་ནས་རོལ་དབྱངས་མཆེད་བྱུང་། དབྱངས་ཀྱི་འཇམ་ཤ་དེའི་ཕྲག་གོང་ནས། ས་སྲོས་ལ་བྱུགས་པའི་ཟླ་འོད་ཇི་བཞིན་རྨི་ལམ་བརྡལ་མྱོང་། ང་ཚོས་སྐྱེ་འཆི་དང་བར་དོ། ལྷ་ཡུལ། སེམས་ཀྱི་གསང་ཐིག་ལ་བསྐུངས་པའི་མཛེས་པ་མཐའ་དག ཚིག་ཕྲད་བྱེ་བའི་པང་ན་ཤེད་ཤུགས་ཀྱིས་སྦེད་དུས། ད་གཟོད་གྲོང་དང་གྲོང་ཁྱེར་གྱི་རུམ་ནས་མུན་པ་འཚུབ་བྱུང་། ཆར་སྨག་འཁྲིགས་བྱུང་། ཐོག་མ་ཉིད་ནས། ང་རང་ཁེར་རྐྱང་དུ་ཆས་ཟིན་པས། འདོད་ཡོན་གྱིས་སེམས་གཤིས་བསླུས་པའི་འཁོར་བའི་ཐང་ནས། ང་རང་སྔར་བཞིན་ཁེར་རྐྱང་དུ་ལོག་དགོས། ཁེར་རྐྱང་ཁེར་རྐྱང་དུ་ལོག་དགོས་རུང་། ཐེ་ཚོམ་དང་དམ་བཅའ། གཉོམ་དུང་དུང་གི་གོམ་ཆུང་དྲུད་དེ། སྙན་ངག་འདྲ་བོ་འགས་རྣམ་རྟོག་གི་སེམས་འགྱུ་ཡང་ཡང་དཀྲུགས་པ་ནས། ཐལ་མདོག་གི་ཚལ་གྲིའི་ངོས་ན་ཡིད་བརྙན་ཁྲི་འབུམ་འཚུབ་ལ། འདབ་མ་བྲུལ་བའི་གཞུང་རྟའི་རྩེ་མོར། ཚོར་བ་བརྒྱ་ཕྲག་གི་གླུ་གར་རེམ་གྱིས་རྩེན། ང་ཚོ། སྐད་ཅིག་མ་གྲངས་མེད་ལས་གྲུབ་ཅིང་། སྒྱུ་འཕྲུལ་གྱི་དར་མདུང་རྩེ་ན་ལྡེམ་ཤུགས་ཀྱིས་འཁྱིལ་བའི་འཁྲབ་ཆས་དག་དང་ཡང་ཧ་ཅང་མཚུངས་པར་སྣང་སྟེ། ཚོར་བའི་མྱོང་སྟོབས་ལས་འཁྲུངས་པའི་བསམ་གཞིག་གི་སྣེ་མོར། འཇའ་ཚོན་བཞིན་ཁྲ་ཆིལ་ལེར་མཚན་མཚོན་གྱི་ཐིག་རིས་དབར་ནས་འཁོར་བའི་འདུ་བྱེད་བཀྲ་དུས། ཐ་སྙད་ཀྱིས་མ་འདང་བའི། སྐད་ཆས་མ་འགྲོལ་བའི། སྨྲ་བསམ་བརྗོད་འདས་ཀྱི་རྣམ་རྟོག་དག ཚོར་འདྲེས་དང་འབྱམས་ཀློང་། རྟོག་པའི་རྐྱོང་སྐུམ་ལས་ལྷུག་པོར་དགྲོལ་ཆོག་ན། ཕྲ་ཕུང་རེ་རེ་ནས་འབར་བའི་ཚེ་སྲོག་གི་མངའ་ཐང་། ནམ་ཞིག་གི་ཚེ། ང་ཚོས་སྙན་ངག་གི་མགུལ་བར་དྲན་པས་འཁྱུད་དེ། དོག་མོའི་འཕྲང་ནས་གསོལ་འདེབས་ཡང་ཡང་བྱེད་ཆོག་ཨང་། པྲ་ཆལ་དང་མཆོང་སྐྱོད། ཧོལ་རྒྱུག་བཅས་ཀྱིས་བདག་རྒྱུད་ཉོན་སྒྲིབ་ཀྱི་བཙན་རྫོང་དཀྱིལ་ན་རེ་ཞིག་བྱིང་དུ་བཅུག་ཀྱང་། དོན་སྣང་དེའི་ཐད་ཚོར་ཕྲ་མོས། སྒེར་གྱི་རྟོག་པའི་རླབས་རྒྱུན་ཡོངས་སུ་དཀྲུགས་ཏེ། སྐྱོ་སྣང་རིང་ལུགས་ཀྱི་ས་ལམ་ཁྲོད་ནས་འཆར་ཡན་གྱི་གཟུགས་སྤྱི་མེ་རུ་སྦར་བས། འགྲོ་མིའི་རྟེན་འབྲེལ་གྱི་སྣང་ཚོར་ཡུལ་ལ་བཏགས་པ་ཙམ་གྱིས། ཁངས་སྐྱ་དང་རླང་དམར། ཐོག་སེར་གྱི་དབལ་ཁ་བཅས་ནི་འགུལ་ཤུགས་སུ་གྱུར་ཏེ། འགས་གཏོར་རེ་རེའི་བར་ཧྲག་ནས་རྙེད་པའི་རང་ཉིད་ངོ་མ་བོའི་དགའ་སྡང་དག སྙན་ངག་ཡིན་མིན་ཅི་རིགས་ཀྱི་པང་ན་མེར་གྱིས་སྦེད་དུས། ད་གཟོད་སྙན་ངག་འབྲི་བ་ནི་སུའི་ཆེད་དུ་མིན་པར། རང་ཉིད་ལ་སེམས་གསོ་དང་ཁེར་རྐྱང་གི་སྐྱོ་ཡ་གཅིག་པུ་དེ་ཡིན་པར་འདོད། འཆར་སྒོའི་རྫི་གསང་རྣོ་བ་དེས། ཡི་གེའི་གབ་སྦས་དང་ཐད་བརྗོད། མཚོན་བྱ་དང་མཚོན་བྱེད། ཚོར་འདུའི་གཡོ་སྐྱོམ་སོགས་ལས། "སྣང་ཚོར་རོ་མཉམ"དུ་གྱུར་པའི་བཟུང་འཛིན་གྱི་འཆར་ཤེལ་ངོས་སུ། སྲིད་པ་ཆགས་ཚུལ་ནས་བསྐལ་པ་སྟོང་ཚུལ། སྐྱེ་བ་ནས་འགག་པའི་བར་གྱི་གཏམ་རྒྱུད་མཐའ་དག་ལ་སྣང་བས་བརྡར་ཏེ། སྒེར་ཚུགས་ཀྱི་ཐབ་ཀའི་འོག་ནས་ཚོར་འདུའི་མེ་རླབས་འཕྱོ་བས། ཟུམ་པ་མེད་པའི་ཉམས་མྱོང་གི་སྨྲ་སྒོར། རྒྱུ་འབྲས་རྟེན་འབྲེལ་གྱིས་གྲུབ་པའི་སྙན་ངག་གི་མེ་ཏོག་ཆུ་ཟིལ་ཁྲོམ་ཁྲོམ་དུ་གདའ། ཆུ་ཟིལ་ཁྲོམ་ཁྲོམ་དུ། ང་ཚོ་སྦས་པའི་ཚུལ་གྱིས། ཐ་སྙད་ཀྱི་ལྷུམས་སུ་འཚོ་བའི་སྐྱ་རིས་བཏོད་ཀྱང་། སྐད་ངག་འདར་བའི་རྫི་གླུ་འཁྱག་པོ་བཞིན་ཚོར་བའི་མེ་དབལ་ནས་ཅུང་ཟད་མངོན་རྐྱེན། སྟོང་ཆེན་པོར་བྲག་ཅ་མཆེད་པའི་ངང་ཚུལ་ནི་ཅི་འདྲའི་སྐྱོ་ཨང་སྙམ། དེ་ལྟ་ནའང་། ངོས་ལྐོག་དང་ཟོལ་སྦྱོར་མ་ཡིན་པའི་སྙན་ངག་རེ་རེ་མཚན་མོའི་སྒྲ་བཤད་ཀྱི་ཚུལ་དུ་བྲིས་པའི་མོད། ཡིག་འབྲུའི་གསེབ་ནས་གཤར་འོང་བའི་རྟ་དམག བུ་མོ། མྱོས་བུམ། ཉི་ཟླ། ན་ཟུག་བཅས་ནི་དོན་ངོ་མར་རང་ཉིད་ཀྱི་གཟུངས་གཞུག་ཏུ་སྦྲུམ་ཡོད་ཕྱིར། ཚོར་དཔྱོད་ཀྱི་བློ་མིག་རྣོན་པོ་དེས། འཁོར་བའི་ཉིན་སྲིབ་ནས་རང་གི་གཏན་གྱི་ལོག་མལ་བཙལ་བས། "བཅད་དང་གྲོལ་བའི་ལུས་བྱེད་མཐའ་དག"ལས་མི་ཚེའི་དབུགས་རིང་གཏང་ཆོག་ན། རང་ཉིད་བཙལ་འཚོལ་དང་རང་ཉིད་སྒོམ་ཚུལ་ཡང་འདི་ཁོ་ནར་ལགས་པས། སྙན་ངག་ནི། མི་ཚེའི་ཁེར་རྐྱང་གི་སྐྱོ་ཡ་གཅིག་པུར་འདོད། ཁ་ཕོ་ཆེས། སྙན་ངག་ནི་ཅི་ཡང་ཡིན་པར་འདོད་རུང་། སྒོམ་ཚུལ་དང་བློས་བསླངས། སྒྲིག་གཞིའི་བཀྲོལ་སྟངས་ལས་ཤ་མར་བཞིན་ལྡད་ཆོག་ཆོག་དང་། ཐམས་ཅད་པོའི་བསམ་ཟས་སུ་འགྱུར་བའི་ངེས་པ་གཏན་ནས་མེད་ཁར། ཐ་སྙད་ཀྱི་གྲུབ་གྲོལ་ལས་སྐད་ཆའི་ཉིང་ཁུ་འཁྲུངས་ཤིང་། སྐད་ཆ་དེ་ཡང་རང་ཉིད་རྣལ་མ་བོའི་སེམས་འགྱུ་ཡིན་གཤིས། དོན་དམ་པར། སྙན་ངག་བྲིས་པ་ནི་རང་ཉིད་ཕྱིར་སྟོན་པ་དང་འདྲའོ་སྙམ། དེ་ལྟ་བས་ན། སྙན་ངག་པའི་ནང་སེམས་འཐུང་བའི་བྱེད་པོ་ཡང་ཀློག་པ་པོ་ཡིན་ལ། རྒྱུ་ཚོགས་དེ་དག་གི་སྔོན་འགྲོ་བྱས་པའི་རྐྱེན་འབྲས་ཀྱང་རང་ཉིད་ཁེར་རྐྱང་ལ་གོམས་སོང་བས་ཨེ་ཡིན་འདོད། ད་ནི། མི་རྗེ་དང་དབུ་བླ།
