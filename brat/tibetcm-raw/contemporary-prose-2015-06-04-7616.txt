སྒྲོལ་དབྱངས་དང་ངའི་མི་ཚེའི་ལ་མོ།
ཚྭ་རྡེའུ་ཚེ་དབང་འཇིགས་མེད།

དེ་རིང་ནི་སྤྱི་ལོ༢༠༡༥ལོའི་ཟླ་༣པའི་ཚེས་༨དེ།རྒྱལ་སྤྱིའི་བུད་མེད་དུས་ཆེན་ཡང་དེ་རིང་གི་ཉིན་མོ་འདི་ལ་འཁེལ་ཡོད། ཐོ་རེངས་ནས་ད་ཚོད་བར་དུ་རྒྱུན་མི་ཆད་པའི་པོད་ཕྲན་དང་འཕྲིན་ཕྲན་བར་གྱི་ཤིས་ཚིག་སྣ་ཚོགས་དང་། མིའི་རིགས་ཀྱི་བླ་ན་མེད་པའི་རང་དབང་དང་འཐབ་རྩོད་ཀྱི་བདེན་པ་མོད་པོར་རྗེས་དྲན་དང་ཕྱིར་རྟོག་བྱེད་པའི་བརྩམས་ཆོས་སྣ་ཚོགས་ལ་མིག་བཤེར་བྱེད་ཞོར། ངས་ཀྱང་དམིགས་བསལ་གྱི་ཉིན་མོ་འདི་ལ་རང་དབང་གི་ཁང་མིག་ཆུང་ཆུང་ཞིག་ན། དང་ཁོག་ནས་ཕྱུང་བའི་གསོང་བོའི་ཡིག་འབྲུ་ཁ་ཤས་བབ་ཆགས་ཀྱི་སྲད་པུར་བརྒྱུ་བཞིན་ཡོད། འདི་ནི་མི་ལོ་བཅུ་ཕྲག་ཅིག་གི་དུས་ཚོད་དང་སྐྲ་ལོ་རེ་རེ་ས་མཐར་ཟད་པའི་ཕྱིར་དྲན་གྱི་ཐོ་ཡིག་ཡིན་ལ། ཡང་གཅིག་བཤད་ན། སྒྲོལ་དབྱངས་ཏེ་ངའི་མི་ཚེའི་ལ་མོ་བརྒལ་རོགས་ཀྱི་ན་ཆུང་དེར་ཕུལ་བའི་དུས་ཆེན་གྱི་ལེགས་སྐྱེས་ཡིན། མི་ཚེ་ཡི་ལམ་ཁ་ཟེར་བ་ནི་ཏན་ཏན་དེ་འདྲ་ཞིག་ཏུ་སྲིད་དེ།ནར་སོན་གྱི་བརྗེད་བྱང་ཟེར་བའང་ཞིབ་བློས་གཞལ་ན་འདི་ལྟར་ཅི་ལ་མིན་ཨང་། དུས་དེ་དུས་ཤིག་ལ། ང་དངོས་ནས་རླུང་ཁར་ལྷུང་བའི་ལུ་གུ་ཞིག་དང་འདྲ། དངོས་ནས། འབྲོག་པ་ཚོས་ཤེས་སྲིད། རླུང་ཁར་བཞག་པའི་ལུ་གུ་ནི་དངོས་ནས་དགོད་བྲོ་བ་ཞིག་ཡིན། རྣ་ནང་དུ་རླུང་བུ་ཨེན་ཙམ་ལྡང་ཚེ་ས་ཕྱོགས་ངེས་མེད་ཏུ་བྲོས་ནས་ཟིན་པའི་ཐབས་མེད། སྐད་ཆར་ཤུན་ལྤགས་མ་བསྐོན་པར་བཤད་ན། དེ་དུས་ཀྱི་བྱ་ལས་རེ་འགའ་ནི་དེ་དུས་ལྟ་ཅི། ད་སྐབས་བསམ་ཚེའང་ངོ་གནོང་བ་ཞིག་རེད། དྲང་མོར་བཤད་ན།ལོ་ལྔ་དྲུག་ཅིག་གི་དུས་ཡུན་ནི་དངོས་ནས་ཧ་ཅང་མགྱོགས། བལྟས་བལྟས་རིག་རིག་ཏུ་ངའང་ཕོ་ལོ་ཉེར་གསུམ་ལ་བུད་སོང་། ལོ་དེར་སྒྲོལ་དབྱངས་ཟེར་བའི་བུ་མོ་ཞིག་དང་ཕྲད་པ་ནས། ངའི་མི་ཚེ་དང་རྡོག་སྣེའི་ཕྱོགས་ཀ་ལ་འཕོ་འགྱུར་ཆེན་པོ་ཞིག་བྱུང་ཡོད། དྲང་མོ་ཡིན། ཡང་ན་འདི་ནི་དཔའ་སར་ཟླ་བར་བཏགས་པའི་བུ་ཆུང་ཞིག་བདེ་སྐྱིད་ཀྱིས་ཐོབ་པས་ཡིན་ཡང་སྲིད ། གང་ལྟར། མིག་སྔར་གྱི་ང་ནི་འབྲོག་རིའི་རྩི་རྟིང་ནས་སུ་ཞིག་གི་ལྭ་འདབས་ལ་འཐིམས་ཟིན་པའི་བསིལ་ཆུ་ཆུང་ཆུང་ཞིག་སྟེ། འཐིམས་ཚུལ་དེའང་ཧ་ཅང་སྟབས་བདེ་ཞིག་དང་།ཡང་ན་ཧ་ཅང་ཕྱི་གསལ་ནང་གསལ་འབའ་ཞིག་ཏུ་ཆགས་ཡོད། ཡང་ན། འདི་དག་ཡོད་ཚད་མོས་ཀྱང་ཤེས་སྲིད། ལོ་དེར། གཞོན་ཚུགས་ཀྱི་སྣང་བ་སྐྲ་རྩེར་འཕུར་བའི་ཅིང་མེད་ང་དང་ཕྲད་མེད་རྒྱུ་ན། ཁྱེད་ཀྱི་མི་ཚེ་དང་འཚོལ་བསྙེག་གི་ལམ་ཁ་མཐའ་དག་ཅི་འདྲ་ཞིག་ཏུ་གྱུར་ཡོད་པ་སུས་ཤེས། དངོས་ནས་བཤད་ནའང་།ཉི་ཧོང་གི་སྐད་ཡིག་སྦྱངས་པའི་ལོ་ཟླ་རེ་རེ་དང་། སྒྱུ་རྩལ་གྱི་རེ་སྨོན་རུམ་པའི་བཤུལ་ལམ་མཐའ་དག་དྲན་ཐེངས་རེ་རེར། ངའི་ནང་སེམས་ཀྱི་ཟབ་ས་ཀུན་དུ་ཙག་རླུང་གི་སྤུ་གྲི་གཟེངས་ཏེ་འོང་མོད།ལོ་ངོ་འདི་དག་གི་རིང་ལ། ཁྱོད་ཀྱིས་ང་ལ་ལེ་བདའ་དང་འཁང་སེམས་ཅི་ཡང་བསྟན་མ་མྱོང་། མྱངས་ནས་འོང་བ་ནི་ངའི་ཚུགས་ཐུང་གི་རང་གཤིས་དང་དམ་ཚིག་གི་ར་ལག་གས་རྗེས་སུ་ཐོབ་པའི་སྡུག་ཡུས་དང་སྐྱོ་གདུང་གི་མིག་ཆུ་ཡིན། སྒྲོལ་དབྱངས། དེ་རིང་ངས་ཁྱེད་ལ་བུད་མེད་ཀྱི་ཐོབ་ཐང་དང་རང་བརྩི། སྙིང་སྟོབས་དང་སྒེར་ཚུགས་སོགས་ཀྱི་སྐད་ཆ་ཟོལ་མ་དེ་དག་དངོས་ནས་ལབ་མི་འདོད། གོ་སླ་ཞིང་བབ་ཆགས་པའི་ཡིག་འབྲུ་འདི་དག་གིས། དོ་ནུབ་ངས་ཁྱེད་རང་ཁོ་ནར་དབང་བའི་ནང་སེམས་ཀྱི་མཆོད་པ་ཞིག་བཤམས་འདོད། ཁྱེད་རང་ཁོ་ནར་འོས་པའི་རེ་སྨོན་གྱི་སྦྲ་ཁྱིམ་ཞིག་ཕུབ་འདོད། ཁུར་མོང་གི་འབྲུ་གུ་ལྟར་རླུང་གིས་ཁུར་ཞིང་ཐབ་ཟངས་ཀྱི་རྒྱ་སྲན་ལྟར་མཆོངས་ཤིང་རྩེན་པའི་བཤུལ་ལམ་མཐའ་དག དངོས་ནས་ཕྱིར་མིག་གི་ཆ་གར་ལྷུང་སྲོལ་མེད་ཀྱང་། ཁྱེད་དང་མ་ཕྲད་པའི་སྔ་རོལ་དུ་ངས་འཇིག་རྟེན་གྱི་སྣང་བ་དང་བྲོ་བའི་ཁ་དོག་ཚོར་མེད་པ་སུ་ལ་གསང་། སྒྲོལ་དབྱངས། དོ་ཞག་ངའི་ཉམ་ཆུང་གི་སེམས་པ་ཕྲ་མོ་ཁྱེད་ཀྱི་རྔུར་སྒྲ་ཡངས་མོའི་མཐིང་ཀ་རུ་ཡན་ཡན་འཕུར་བཞིན། ལོ་ངོ་བཅུ་ཕྲག་ཅིག་ལ་མྱངས་པའི་ས་མཐའི་སྡུག་བསྔལ་དང་ཆད་སྣང་མཐའ་དག ཁྱེད་ཀྱི་ཞིམ་སྣང་འབར་བའི་གཉིད་ཐུན་གྱི་གཏིང་རིམ་དུ་སྦས་བ་འདོད། ཧྲེང་ཧྲེང་དུ་འདུར་བའི་ལོ་ངོ་དེ་དག་གི་རྣམ་རྟོག་སྐྱུར་བོ་ག་ཤེད་ཅིག་ལ་གཡུག་རྗེས། ཁྱེད་ཉིད་ཁོ་ན། ངའི་བྱིས་དུས་ཀྱི་ནམ་མཁར་བཀལ་བའི་ཉི་གུར་དེ་ཡིན་པ་ཚོར་བྱུང་། ཁྱེད་ཉིད་ཁོ་ན། ངའི་རྒས་དུས་ཀྱི་སྔས་འགྲམ་བསྲུང་བའི་མར་མེ་དེ་ཡིན་པ་མཐོང་སོང་། ལོ་དེ་ནས་བཟུང་། མྱི་ཁ་དང་འཕྱ་སྨོད་ཀྱི་ཆར་རྡེའུ་ཅི་ཙམ་བབས་ཀྱང་།ཁྲག་དོག་དང་དངོས་ཡོད་ཀྱི་བཙུར་གནོན་ཅི་ཡང་གཟན་ཡང་།ཁྲེལ་གཞུང་དང་ཕྱི་ཐག་གི་མདུད་ཁ་ཅི་ཙམ་རལ་ཡང་། ང་དང་ལྷན་དུ་༼ སྐྱིད་ན་སྐྱ་ཀ་ཚང་མཉམ་དང་། སྡུག་ན་ཕོ་རོག་ནགས་མཉམ་ ༽བྱེད་མཁན་དེ་ཁྱེད་ཉིད་ཡིན།ཐོག་མའི་དམ་ཚིག་བསྒྲིགས་ཉིན་ནས་ན་ནིང་གཅེས་ལོ་སྐྱེས་ཡུན་བར་དུ། ཁྱེད་ཀྱིས་དངོས་ནས་དཀའ་ལས་མྱངས་སོང་། ཉེ་རིང་ཆགས་སྡང་གསལ་ཞིང་། ལས་འབྲས་མིག་ལྟར་བརྩི་ལ། ཁྲེལ་གཞུང་ཕྱི་ཐག་རིང་བའི་རང་གཤིས་སྲ་མོ་དེས་དབང་གིས། ངའི་རྒྱ་ཡན་གྱི་སེམས་པ་རྟ་རྒོད་འདྲ་བོའང་ཁྱེད་ཀྱི་སྤང་ཐང་ན་རང་དབང་གི་བསིལ་ཉལ་བྱས་ནས་ཡོད། ཞིབ་ཏུ་བསམ་ན། མི་གཤིས་ཀྱི་འགྱུར་ལྡོག་ལ་སྐྱུར་དྲི་ཁ་བའི་དེང་གི་ན་ཆུང་དག་གི་གྲས་ནས། གཉོམ་ཆུང་གི་ཁྱེའུ་ཞིག་སྨན་ཐུབ་ཤོག་ཁུག་གང་ལ་མ་བརྗེ་བར།བྱམས་སེམས་ཀྱི་ལག་པས་མི་ཡུལ་ལ་བསུ་མཁན་དེ་ཁྱེད་ཉིད་ཡིན། སྒྲོལ་དབྱངས། ནང་མོ་དེར་ངས་རང་གི་བུ་ཆུང་ལ་དཔའ་སར་ཟླ་བ་ཞེས་པའི་གླིང་གི་གཅེས་ཕྲུག་བཅུ་གསུམ་པོའི་བྱེ་བྲག་ཅིག་གི་མིང་བཏགས་ཡོད་ལ།ནང་མོ་དེ་ནས་བཟུང་།ངའང་གཅེས་ལོ་དཔའ་སར་དང་འདྲ་བར་རིམ་གྱིས་ནར་སོན་བཞིན་ཡོད་པས། ཁྱེད་ཀྱིས་སེམས་བདེ་གནང་རོགས། ཁྱེད་ཉིད་ཁོ་ན། ངའི་ལོ་ཟླའི་བྲང་ཁར་དཔྱངས་པའི་དངུལ་གྱི་གའུ་ཡིན། ཁྱེད་ཉིད་ཁོ་ན། ངའི་འཚོ་བའི་ངམ་ཁ་གཅོད་པའི་གསེར་གྱི་གཤོག་པ་ཡིན། ༢༠༡༥-༣-༨ཉིན་རང་ཤག་ཏུ་བྲིས།
(མཆོད་མེའི་པར་དབང་གཞན་གྱིས་དགོལ་མི་རུང)


རྩོམ་ཁུངས། ངེད་དྲ་བ།
