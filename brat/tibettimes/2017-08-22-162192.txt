ཨ་ཀྲུག་བློ་སྤོ་ཀློད་གྲོལ་བཏང་།

བོད་ཀྱི་ཆབ་སྲིད་བཙོན་ཟུར་རོང་རྒྱས་ཨ་གྲགས་ལགས་ཀྱི་ཚ་བོ་ཨ་ཀྲུག་བློ་སྤོ་ལགས་ཀྱི་བཙོན་འཇུག་དུས་ཡུན་ལོ་ ༡༠ ཚང་སྟེ་ཀློད་གྲོལ་བཏང་ཡོད་ཀྱང་ཁོང་གི་སྐུ་གཟུགས་གནས་སྟངས་ག་འདྲ་ཡིན་མིན་སོགས་ཤེས་རྟོགས་ཐུབ་མི་འདུག སྤྱི་འཐུས་ལྷ་རམས་པ་ཨ་ཀྲུག་ཚེ་བརྟན་ལགས་ཀྱིས་གནས་ཚུལ་མཁོ་གཏོང་གནང་བ་ལྟར་ན། ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༡ ཉིན་གྱི་མཚན་མོ་བོད་ནང་གི་ཆུ་ཚོད་ ༡༡ ཙམ་ཞིག་ལ། ཆབ་སྲིད་བཙོན་ཟུར་རོང་རྒྱས་ཨ་གྲགས་ལགས་ཀྱི་ཚ་བོ་དང་། ལི་ཐང་དགོན་ཆེན་གྱི་དབུ་ཟུར་ཨ་ཀྲུག་བློ་སྤོ་ལགས་ཀྱི་བཙོན་འཇུག་དུས་ཡུན་ལོ་ ༡༠ ཧྲིལ་པོ་ཚང་སྟེ་ཀློད་གྲོལ་བཏང་ཡོད་འདུག་ཀྱང་། ད་ལྟའི་ཆར་ཁོང་གི་སྐུ་གཟུགས་གནས་སྟངས་ག་འདྲ་ཡིན་མིན་སོགས་ཀྱི་གནས་ཚུལ་ཤེས་རྟོགས་ཐུབ་ཀྱི་མེད་པ་རེད། པྱི་ལོ་ ༢༠༠༧ ཟླ་ ༨ ཚེས་ ༡ ཉིན་གྱི་ལི་ཐང་བརྒྱད་གཅིག་རྟ་རྒྱུག་དུས་ཆེན་ཐོག ཆབ་སྲིད་བཙོན་ཟུར་རོང་རྒྱས་ཨ་གྲགས་ལགས་སྡིངས་ཆའི་སྟེང་དུ་བུད་དེ། བོད་རྒྱ་མི་མང་སྟོང་ཕྲག་མང་པོའི་མདུན་དུ་༸རྒྱལ་བ་རིན་པོ་ཆེ་བོད་ལ་གདན་འདྲེན་ཞུ་འཇུག་དགོས། ང་ཚོ་བོད་ལ་རང་དབང་དགོས། པཎ་ཆེན་ཆུང་སྲིད་རིན་པོ་ཆེ་དང་སྤྲུལ་སྐུ་བསྟན་འཛིན་བདེ་ལེགས་གཉིས་ཀྱི་གཙོས་ཆབ་སྲིད་གཙོན་པ་རྣམས་མྱུར་ཏུ་སློད་གྲོལ་དགོས། ཅེས་སྐད་འབོད་ངོ་རྒོལ་གནང་ཡོད་པ་དང་། དེ་མ་ཐག་ཏུ་རྒྱ་གཞུང་ཉེན་རྟོག་པས་ཁོང་དང་ཁོང་གི་ཚ་བོ་ཨ་ཀྲུག་བློ་སྦོའི་གཙོས་བོད་མི་ཁག་ཅིག་འཛིན་བཟུང་བྱས་ཡོད་པ་དང་། ཕྱིས་སུ་རྒྱལ་ཁབ་མགོ་གཏིང་སློག་པ་དང་། རྒྱལ་ཁབ་གྱི་གསང་བ་ཕྱིར་བསྒྲགས་བྱས་པ། རྒྱལ་ཁབ་ཀྱི་དཔལ་འབྱོར་ལ་གྱོང་གུད་བཏང་བ་སོགས་ཀྱི་ཉེས་མིང་བཏགས་ཏེ། ཨ་ཀྲུག་བློ་སྦོ་ལ་ལོ་ ༡༠ དང་། རོང་རྒྱས་ཨ་གྲགས་ལ་ལོ་ ༨། སྒྱུ་རྩལ་པ་འཇམ་དབྱངས་ཀུན་མཁྱེན་ལ་ལོ་ ༩། ཨ་ཀྲུག་སྐལ་བཟང་རྒྱ་མཚོ་ལ་ལོ་ ༥།  བློ་ཐོག་ལ་ལོ་ ༣། འཇམ་དབྱངས་བསྟན་འཛིན་ལ་ལོ་ ༣། བདེ་སྐྱིད་གྲོང་པ་བློ་བཟང་ཕུན་ཚོགས་ལ་ལོ་ ༡ དང་ཕྱེད་ཀ་བཅས་ཁྲིམས་ཐག་བཅད་ཡོད་པ་རེད། ཕྱིས་སུ་རིམ་བཞིན་བཙོན་འཇུག་དུས་ཡུན་ཚང་སྟེ་ཁོང་ཚོ་ཀློད་གྲོལ་བཏང་ཡོད་ཀྱང་། རོང་རྒྱས་ཨ་གྲགས་ལགས་བཙོན་ཁང་དུ་བཞུགས་སྐབས་གདུག་སྦྱོང་མནར་གཅོད་མང་པོ་བཏང་བའི་རྐྱེན་གྱིས། ཁོང་གི་མིག་གཅིག་གིས་མི་མཐོང་བ་དང་། རྣ་བ་གཅིག་གིས་མི་གོ་བར་གྱུར་བའི་གནས་ཚུལ་ཐོན་པར་བརྟེན། བོད་ཕྱི་ནང་གཉིས་ནས་ཁོང་མྱུར་ཏུ་ཀློད་གྲོལ་གཏོང་དགོས་པའི་ཞུ་འབོད་ལས་འགུལ་སྤེལ་དགོས་བྱུང་ཡོད་པ་ལྟར། དེ་སྔོན་ལོ་ ༣ བཙོན་འཇུག་བྱས་ཏེ་ཀློད་གྲོལ་ཐོབ་པའི་འཇམ་དབྱངས་བསྟན་འཛིན་ལགས་བསྐྱར་དུ་འཛིན་བཟུང་བྱས་ཏེ་ལོ་གསུམ་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་པ་སོགས། རོང་རྒྱས་ཨ་གྲགས་ལགས་ཀྱི་ཉེ་འབྲེལ་དང་ས་གནས་ཡུལ་མི་ཁག་ཅིག་ལའང་ཉེས་འཛུགས་ཁྲིམས་ཆད་བཅད་པའི་གནས་ཚུལ་ཐོན་ཡོད་པ་རེད། གསར་འགོད་པ། པདྨ་མཚོ།
