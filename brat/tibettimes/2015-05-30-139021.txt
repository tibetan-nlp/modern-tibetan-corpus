དགེ་རྒན་ཞིག་ལ་སྲོག་ཁྲིམས་བཅད།

ཀན་སུའུ་ཞིང་ཆེན་ཝུའུ་ཧྲན་རྫོང་ཁོངས་ཀྱི་སློབ་ཆུང་དགེ་རྒན་ཞིག་གིས་བུ་མོ་ཆུང་ཆུང་ ༢༦ ལ་བཙན་གཡེམ་བྱས་ཏེ་སྲོག་ཁྲིམས་བཅད་ཡོད་འདུག ཕྱི་ཚེས་ ༢༨ ཉིན་ཀན་སུའུ་ཆེས་མཐོའི་མི་དམངས་ཁྲིམས་ཁང་གིས་བཙན་གཡེམ་འཆལ་སྤྱོད་དང་འབྲེལ་བའི་ཉེས་དོན་ ༥ ཕྱིར་བསྒྲགས་བྱས་ཡོད་པའི་ནང་། ཀུན་སུའུ་ཝུའུ་ཧྲན་རྫོང་ཁོངས་ཀྱི་སློབ་ཆུང་ཞིག་གི་དགེ་རྒན་ལི་ཅི་ཧྲིན་ཟེར་བ་ཞིག་གིས། ཕྱི་ལོ་ ༢༠༡༡ ལོའི་ལོ་མགོ་ནས་ཕྱི་ལོ་ ༢༠༡༢ ཟླ་ ༦ ཚེས་ ༤ བར་དུ། ནར་མ་སོན་པའི་བུ་མོ་ཆུང་ཆུང་ ༢༦ ལ་བཙན་གཡེམ་དང་འཆལ་སྤྱོད་སྤེལ་བའི་ནག་ཉེས་འོག་སྲོག་ཁྲིམས་བཅད་ཡོད་འདུག  ལི་ཅི་ཧྲིན་ཟེར་བ་དེས་སློབ་ཕྲུག་བུ་མོ་དེ་དག་ལ་མགོ་སྐོར་བཏང་སྟེ། ཉལ་ཁང་དང་སློབ་ཁང་། གྲོང་ཚོའི་ཉེ་འཁོར་གྱི་ནགས་ཚལ་སོགས་སུ་ཁྲིད་ནས་བཙན་གཡེམ་དང་འཆལ་སྤྱོད་སྤེལ་ཡོད་པ་མ་ཟད། བུ་མོ་གཅིག་ལ་ཐེངས་མང་པོར་བཙན་གཡེམ་དང་། ཡང་ན་དུས་གཅིག་ལ་བུ་མོ་མང་པོར་བཙན་གཡེམ་བྱས་ཡོད་འདུག བུ་མོ་དེ་དག་ལོ་ན་ ༤ ནས་ ༡༡ བར་གྱི་རང་སྲུང་གི་ནུས་པ་ཞན་ཞིང་། རང་ལུས་རང་གཅེས་ཀྱི་འདུ་ཤེས་ནར་མ་སོན་པ་རྐྱང་རྐྱང་ཡིན་འདུག ཡང་ཉིན་དེའི་ཉེས་དོན་ཕྱིར་བསྒྲགས་ནང་། རྒྱ་ནག་ཧུའུ་པེ་ཞིང་ཆེན་པོ་ཐོ་གྲོང་ཁྱེར་གྱི་སློབ་འབྲིང་ཞིག་གི་བུ་མོའི་ཉལ་ཁང་ནང་སྐྱེས་པ་གཉིས་འཛུལ་ཏེ། རང་ལོ་ ༡༤ མ་སོན་པའི་བུ་མོ་ ༦ ལ་སྐེ་བདམས་པ་དང་འགྲམ་ལྕགས་གཞུས་པ་སོགས་བྱས་ནས་བུ་མོ་ ༥ ལ་བཙན་གཡེམ་བྱས་པར་བརྟེན། ཉེས་ཅན་གཙོ་བོ་ཏུང་ཆི་ཟེར་བ་དེར་ལོ་ ༢ དུས་འགྱངས་ཀྱིས་སྲོག་ཁྲིམས་གཅོད་རྒྱུ་དང་། ཡང་པེ་ཅིན་གྲོང་ཁྱེར་དུ་གནས་སྡོད་བྱས་ཡོད་པའི་ཝེ་ལིན་ཀྲི་ཟེར་བ་ཞིག་གིས། ལོ་ན་ཆུང་བའི་བུ་ ༦ ལ་དངུལ་སོགས་ཀྱིས་བསླུ་བྲིད་བཏང་སྟེ་འདོད་སྲིད་འཆལ་སྤྱོད་སྤེལ་བར་ལོ་ ༥ ཡི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་པ་དང་། གཞན་ཡང་། ལུའུ་ཀྲིན་ཧྥང་ཟེར་བ་སོགས་མི་བརྒྱད་ཀྱིས་ཕྱི་ལོ་ ༢༠༡༢ ནས་ ༢༠༡༣ བར་ནར་མ་སོན་པའི་བུ་མོ་ ༡༡ ཆང་ཁང་དང་མགྲོན་ཁང་སོགས་སུ་ཁྲིད་དེ། ངོ་སྤྲོད་ཀྱི་བར་ཁེ་བླངས་ནས་གཞང་བཙོང་དུ་བཅུག་ཡོད་སྟབས། ཉེས་ཅན་དེ་དག་ལ་ཟླ་ ༦ ནས་ལོ་ ༦ བར་གྱི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་པ་སོགས་བརྗོད་འདུག རྒྱ་ནག་གཞུང་ཕྱོགས་ནས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཕྱི་ལོ་ ༢༠༡༢ ནས་ ༢༠༡༤ བར་རྒྱ་ནག་རྒྱལ་ཡོངས་ཀྱི་ཁྲིམས་ཁང་ཁག་ནས་བཙན་གཡེམ་འཆལ་སྤྱོད་དང་འབྲེལ་བའི་ཉེས་དོན་རིགས་ ༧༡༤༥ ཐག་བཅད་ཡོད་པ་དང་། དེའི་ནང་ཕྱི་ལོ་ ༢༠༡༢ ལོར་ཉེས་དོན་ ༢༠༡༧ དང་། ༢༠༡༣ ལོར་ ༢༣༠༠། ༢༠༡༤ ལོར་༢༨༢༨ བྱུང་བ་བཅས་ལོ་རེ་བཞིན་གྲངས་འབོར་འཕར་བཞིན་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
