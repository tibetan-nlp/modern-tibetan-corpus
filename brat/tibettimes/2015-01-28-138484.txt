བོད་རིཊ་ལས་བྱེད་པ་གསུམ་ཏང་ནང་ནས་ཕུད།

རྒྱ་ནག་དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ཁང་གིས་མཚོ་སྔོན་ཞིང་ཆེན་དུ་རྟོག་ཞིབ་བྱས་པ་བརྒྱུད་བོད་མིའི་རང་ལུས་མེར་བསྲེཊ་ཀྱི་དོན་རྐྱེན་བྱུང་སའི་རེབ་གོང་མདོ་བ་ཤང་གི་བོད་རིགས་ལས་བྱེད་པ་ ༣ ཏང་ནང་ནས་ཕྱིར་འབུད་བཏང་ཡོད་འདུག ཟླ་འདིའི་ཕྱི་ཚེས་ ༢༥ ཉིན་རྒྱ་ནག་དབུས་སྒྲིག་ཁྲིམས་ཞིབ་བཤེར་ཁང་གི་དྲ་བའི་ཐོག་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། འདས་ལོའི་ཕྱི་ཟླ་ ༨ པ་ནས་བཟུང་མཚོ་སྔོན་ཞིང་ཆེན་ནང་སྒྲིག་ཁྲིམས་རྟོག་ཞིབ་བྱས་པ་བརྒྱུད། གནད་དོན་གྱི་ཁུངས་སྣེ་ ༧༠༡ རྙེད་དེ་ཉེས་དོན་ ༢༤༡ ཐོ་འགོད་ཁྲིམས་བཤེར་བྱས་ཡོད་པ་མ་ཟད། འབྲེལ་ཡོད་ཉེས་ཅན་ ༢༧༧ ལ་ཐག་གཅོད་བྱས་ཡོད་པ་དང་། དེའི་ནང་དུ་ཁུལ་རིམ་པའི་ལས་བྱེད་པ་ ༡༠ དང་། རྫོང་རིམ་པའི་ལས་བྱེད་པ་ ༤༥། གྲོང་ཚོ་རིམ་པ་མན་གྱི་ལས་བྱེད་པ་ ༢༢༢ བཅས་ཡོད་ཅིང་། ཁུལ་དང་ཞིང་ཆེན་རིམ་པའི་ལས་བྱེད་པ་ ༡༥ ཁྲིམས་འཛིན་ལྷན་ཁང་ལ་རྩིས་སྤྲད་ཡོད་འདུག ཐེངས་འདིར་ཐག་གཅོད་བྱས་པའི་ཉེས་དོན་མང་ཆེ་བ་ཡུལ་ཤུལ་བསྐྱར་འཛུགས་སོགས་འཛུགས་སྐྲུན་ལས་གཞི་དང་ལས་བྱེད་བསྐོ་འཐེན། གཏེར་ལས་དབང་ཆ་ཉོ་ཚོང་། དེ་བཞིན་དུས་ཆེན་དང་གཉེན་སྟོན་སོགས་སུ་ཐུམ་རྒྱག་རྔན་པ་སྤྲད་པ་སོགས་ལྐོག་ཟ་རུལ་སུངས་དང་ལྐོག་རྔན་བླངས་བའི་གནད་དོན་རིགས་ཡིན་ཡང་། རྟོག་ཞིབ་ཁྲོད་གྲངས་ཉུང་མི་རིགས་ཀྱི་ཏང་ཡོན་ལས་བྱེད་པའི་ཆབ་སྲིད་འདུ་ཤེས་དམན་པར་ཆེད་དུ་དམིགས་ཏེ། ཆབ་སྲིད་ཀྱི་སྒྲིག་ཁྲིམས་དང་འགལ་བའི་ཏང་ཡོན་ལས་བྱེད་པར་སྐྱོན་བརྗོད་དང་སློབ་གསོ་བཏང་ཡོད་པ་མ་ཟད། ཉེ་བའི་ལོ་ཤས་ནང་བསྟུད་མར་བྱུང་བའི་བོད་ཁུལ་བརྟན་ལྷིང་མེད་པའི་གནད་དོན་རིགས་ཐག་གཅོད་བྱས་པར་ཕྱིར་ལྟ་ཞེས་པའི་བྱ་འགུལ་སྤེལ་ཏེ། གྲོང་ཁྱེར་དང་ཞིང་ཆེན་ནང་ཁུལ་ནས་ཐག་གཅོད་བྱས་པའི་ཉེས་དོན་ ༨ དང་དེ་འབྲེལ་གྱི་ཉེས་ཅན་ ༣༠ ལ་བསྐྱར་དུ་རྟོག་ཞིབ་བྱས་ཏེ། བོད་དོན་ཆེད་སྔ་གཞུག་ཏུ་བོད་མི་ ༥ ཡིས་རང་ལུས་མེར་བསྲེགས་གནང་སའི་རྨ་ལྷོ་ཁུལ་རེབ་གོང་ཐུན་རིན་རྫོང་། མདོ་བ་ཞང་གི་སྡེ་བ་ ༥ ཡི་ཏང་ཡོན་ལས་བྱེད་པའི་ཆབ་སྲིད་ཀྱི་གནད་དོན་ཐག་གཅོད་བྱས་པའི་ནང་། བོད་རིགས་ཏང་ཡོན་ལས་བྱེད་པ་སྒྲོལ་མ་སྐྱབས་དང་རྒྱ་རྡོ(加多)། ལོ་བརྒྱ་(洛加)བཅས་ལ་བསྐྱར་དུ་ཐག་གཅོད་བྱས་ཏེ་ཏང་ནང་ནས་ཕྱིར་འབུད་བཏང་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
