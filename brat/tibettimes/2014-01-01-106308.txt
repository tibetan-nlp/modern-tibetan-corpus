ཡུ་གུར་བ་ ༨ ལ་མེ་མདའ་བརྒྱབ་ནས་བསད།

གཟའ་ཟླ་བའི་ཞོགས་པར་ཤིན་ཅང་ཡཱར་ཁན་ཌི་རྫོང་ཉེན་རྟོག་ཁང་དུ་འཇབ་རྒོལ་དོན་རྐྱེན་ཞིག་བྱུང་སྟེ་ངོ་རྒོལ་བ་ ༨ བསད་པ་དང་གཅིག་འཛིན་བཟུང་བྱས་ཡོད་འདུག ཤིན་ཅང་ས་གནས་སྲིད་གཞུང་གི་དྲ་བའི་ཐོག་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། གཟའ་ཟླ་བའི་ཞོགས་པ་པེ་ཅིན་གྱི་ཆུ་ཚོད་ ༦།༣༠ ཐོག ཤིན་ཅང་ལྷོ་ཕྱོགས་ཁཱ་ཞི་གྷར་གྲོང་ཁྱེར་ཡཱར་ཁན་ཌི་ (Yarkand )རྫོང་ཉེན་རྟོག་ཁང་དུ། འཇབ་རྒོལ་བ་ ༩ ཡིས་ལག་ཏུ་གྲི་དང་འགད་རྫས་བཟུང་སྟེ་གཏོར་རྒོལ་གཏོང་རྩིས་བྱེད་སྐབས། ཉེན་རྟོག་པས་ངོ་རྒོལ་བ་ ༨ ཡུལ་དངོས་ནས་མེ་མདའ་བརྒྱབ་སྟེ་བསད་པ་དང་། མི་ ༡ འཛིན་བཟུང་བྱས་ཡོད་པ་མ་ཟད།  ས་གནས་ཉེན་རྟོག་ཁང་གིས་གཏོར་རྒོལ་བ་ཚང་མ་ཡུ་གུར་རིགས་ཡིན་ལུགས་བརྗོད་འདུག འདི་ལོར་ཤིན་ཅང་ནང་ཉེན་རྟོག་ཁང་གཏོར་རྒོལ་དང་གནམ་བདེ་སྒོ་མོ་ཆེའི་རླངས་འཁོར་འབར་གས་དོན་རྐྱེན་སོགས་མི་གསོད་ཁྲག་སྦྱོར་གྱི་དོན་རྐྱེན་བསྟུད་མར་བྱུང་བའི་ཁྲོད་ཁྱོན་མི་གྲངས་ ༡༠༠ ལྷག་དམར་གསོད་དང་། མི་ ༩༠ ཙམ་ལ་རྨས་སྐྱོན་ཕོག་ཡོད་འདུག ཁ་སང་འཇར་མན་དུ་ཡོད་པའི་འཛམ་གླིང་ཡུ་གུར་མཐུན་ཚོགས་ཀྱི་མགྲིན་ཚབ་པ་ཌི་ལི་ཞཱ་ཌི་ལགས་ཀྱིས་གསར་འགོད་པར། རྒྱ་གཞུང་ཉེན་རྟོག་པས་ངོ་རྒོལ་བར་ཐད་ཀར་མེ་མདའ་བརྒྱབ་སྟེ་དམར་གསོད་བཏང་བ་མ་ཟད། ངོ་རྒོལ་བ་ཚང་མ་འཇིགས་སྐུལ་དྲག་སྤྱོད་པ་ཡིན་ལུགས་ཀྱི་སྐྱོན་འཛུགས་བྱས་པ་དེ། དམར་ཤོག་ཚོགས་པས་ཁྲིམས་འཛིན་ལམ་ལུགས་ལ་བསྒྱུར་བཅོས་བཏང་རྗེས་ཀྱི་བྱེད་ཐབས་གསར་པ་ཞིག་ཡིན་ལ། ཐེངས་འདིར་སླར་ཡང་རྒྱ་གཞུང་གིས་ཡུ་གུར་བ་ཚོར་དྲག་གནོན་བྱེད་པའི་སྲིད་བྱུས་དེ་རྒྱལ་ཁབ་ཀྱི་ཁྲིམས་འཛིན་ལམ་ལུགས་དང་བསྟུན་མི་དགོས་པ་བདེན་དཔང་བྱས་ཡོད་སྟབས། རྒྱལ་སྤྱིའི་སྤྱི་ཚོགས་ཀྱིས་རྒྱ་གཞུང་གི་དྲག་གནོན་སྲིད་བྱུས་མཚམས་འཇོག་ཡོང་བར་གནོན་ཤུགས་སྤྲོད་དགོས་པའི་འབོད་སྐུལ་བྱས་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
