ཕོ་བྲང་ཡུམ་བུ་བླ་སྒང་ལ་ཉམས་གསོ།

མི་ལོ་ཉིས་སྟོང་ལྷག་གི་ལོ་རྒྱུས་ལྡན་པའི་བོད་རྗེ་བཙན་པོ་རིམ་བྱོན་མང་པོའི་བཞུགས་གནས་ཕོ་བྲང་ཡུམ་བུ་བླ་སྒང་ལ་ཉམས་གསོ་བྱེད་བཞིན་ཡོད་འདུག ཁ་སང་རྒྱ་ནག་གཞུང་འབྲེལ་གསར་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཟླ་སྔོན་མའི་ཟླ་མཇུག་ཙམ་ནས་བཟུང་བོད་ལྗོངས་ལྷོ་ཁ་ས་ཁུལ་སྣེ་གདོང་རྫོང་ཁོངས་སུ་ཡོད་པའི་བོད་ཀྱི་སྐུ་མཁར་ཐོག་མ་ཕོ་བྲང་ཡུམ་བུ་བླ་སྒང་ལ་ཉམས་གསོ་བྱེད་མགོ་བཙུགས་ཏེ། གནས་སྐབས་རིང་གནས་སྐོར་བ་དང་སྤྲོ་འཆམ་པ་ཚོ་འགྲོ་མི་ཆོག་པ་བཟོས་ཡོད་འདུག བོད་ལྗོངས་ལྷོ་ཁ་གྲོང་ཁྱེར་རིག་དངོས་ལས་ཁུངས་ཀྱི་མགོ་ཁྲིད་བྱམས་པ་ཚེ་རིང་ཟེར་བ་ཞིག་གིས། ཕོ་བྲང་ཡུམ་བུ་བླ་སྒང་གི་ཤིང་ཆས་ལྷུ་ལག་རུལ་སྐྱོན་ཤོར་བ་དང་། ལྷ་ཁང་གི་རྩིག་གྱང་ལ་སེར་ཁ་མང་པོ་གས་པ། རྩིག་རྨང་བརྡིབས་ཡོད་པའི་རྐྱེན་གྱིས། ཕོ་བྲང་རྡིབ་སྐྱོན་འབྱུང་བའི་ཉེན་ཁ་ཡོད་སྟབས། གོང་རིམ་ནས་རྒྱ་སྒོར་བྱེ་བ་གཅིག་སྤྲད་དེ་ཕོ་བྲང་ཉམས་གསོ་བྱེད་པའི་རྩ་དོན་གཞིར་བཟུང་གིས་ཁྱོན་ཡོངས་ནས་ཕོ་བྲང་ཡུམ་བུ་བླ་སྒང་སྲུང་སྐྱོབ་དང་ཉམས་གསོ་བྱ་རྒྱུ་ཡིན་པ་དང་། ལོ་རྗེས་མའི་ཕྱི་ཟླ་ ༤ ནང་ཞིག་གསོ་མཇུག་སྒྲིལ་ཐུབ་པའི་སྔོན་དཔག་བྱེད་ཀྱི་ཡོད་སྐོར་བརྗོད་འདུག ཡུམ་བུ་བླ་སྒང་ནི་བོད་ལྗོངས་ལྷོ་ཁ་གྲོང་ཁྱེར་སྣེ་གདོང་རྫོང་ཁོངས་རྩེད་ཐང་གྲོང་རྡང་དང་བར་ཐག་སྤྱི་ལེ་ ༡༡ ཙམ་གྱི་མཚམས་སུ་ཡོད་པའི་ཇོ་མོ་བཀྲ་ཤིས་ཚེ་རིང་ཞེས་པའི་རི་སྟེང་དུ་ཆགས་ཡོད་པ་དང་། ལོ་རྒྱུས་ཡིག་ཆ་ལྟར་ན། ཡུམ་བུ་བླ་སྒང་ནི་ཕྱི་ལོའི་སྔོན་གྱི་དུས་རབས་གཉིས་པའི་ནང་བོད་རྗེ་ཐོག་མ་གཉའ་ཁྲི་བཙན་པོའི་དུས་སུ་བཞེངས་པའི་ཕོ་བྲང་ཆེས་སྔ་ཤོས་ཡིན་པ་དང་། ད་བར་མི་ལོ་ཉིས་སྟོང་ལྷག་གི་ལོ་རྒྱུས་ཡོད་པར་མ་ཟད། ཡུམ་བུ་བླ་སྒང་ནི་བོད་རྗེ་ཐོག་མ་གཉའ་ཁྲི་བཙན་པོ་ནས་རྒྱལ་རབས་སོ་གཉིས་པ་གནམ་རི་སྲོང་བཙན་བར་གྱི་བཙན་པོ་རིམ་བྱོན་རྣམས་བཞུགས་སའི་ཕོ་བྲང་ཡིན་སྟབས། དེ་ནི་བོད་ཀྱི་ལོ་རྒྱུས་ཀྱི་གནས་རྙིང་གྲགས་ཅན་ཞིག་ཡིན་པ་མ་ཟད། ཕོ་བྲང་ཕྱི་ནང་གི་བཀོད་པ་ནི་དེ་སྔའི་བོད་ཀྱི་རིག་གནས་དང་ལོ་རྒྱུས་ཞིབ་འཇུག་གི་རྒྱུ་ཆ་གལ་ཆེན་ཞིག་ཆགས་ཡོད་པ་རེད། ད་ལྟའི་ཡུམ་བུ་བླ་སྒང་རྒྱབ་མདུན་ཁག་གཉིས་སུ་དབྱེ་བའི་མདུན་ངོས་ཀྱི་ཐོག་རྩེག་གསུམ་ཅན་གྱི་ཁང་པ་དེའི་འོག་ལྷ་ཁང་ཡིན་པ་དང་། ནང་རྟེན་གཙོ་བོ་སྟོན་པ་ཤཱཀྱ་ཐུབ་པ་དང་། པུར་རྒྱལ་བཙན་པོ་རིམ་བྱོན་ལས་གཉའ་ཁྲི་བཙན་པོ་དང་། ལྷ་ཐོ་ཐོ་རི་གཉན་བཙན། ཆོས་རྒྱལ་སྲོང་བཙན་སྒམ་པོ། ཁྲི་སྲོང་ལྡེའུ་བཙན། ཁྲི་རལ་བ་ཅན་བཅས་བོད་རྗེ་གྲགས་ཅན་འགའ་དང་། སྲོང་བཙན་གྱི་བཙུན་མོ་རྒྱ་བཟའ་བལ་བཟའ་རྣམ་གཉིས་དང་། བློན་ཆེན་ཐུ་མི། མགར་སྟོང་བཙན་བཅས་རྒྱལ་བློན་དྲིན་ཆེ་བ་འགའི་ལྡེབས་སྐུ་བཞུགས་ཡོད་འདུག ཐོག་རྩེག་གཉིས་པ་ནི་རྒྱལ་དབང་སྐུ་ཕྲེང་རིམ་བྱོན་གྱི་ཕྱག་འཚལ་ཁང་ཡིན་པ་དང་། ཕོ་བྲང་རྒྱབ་ཕྱོགས་ཀྱི་བྱང་ངོས་སུ་གསེར་ཟངས་ལས་གྲུབ་པའི་རྒྱལ་བ་བྱམས་མགོན་དང་། རྗེ་ཙོང་ཁ་པ། སློབ་དཔོན་པདྨ་འབྱུང་གནས། འཇམ་དཔལ་དབྱངས་སོགས་ཀྱི་སྐུ་རྟེན་ཆེ་ཆུང་སྣ་ཚོགས་དང་། རྒྱལ་ཆེན་རིགས་བཞི་བཅས་བཞུགས་ཡོད་འདུག ཐོག་རྩེག་གསུམ་པའི་མདུན་ཕྱོགས་ནི་གཡབ་སྟེང་དང་། རྒྱབ་ངོས་ནི་ཁྱམས་ཡིན་པ་དང་། མདུན་ཁང་དང་འབྲེལ་བའི་རྒྱབ་ཕྱོགས་དཀྱིལ་དབུས་མཐོ་ཞིང་གཟར་བའི་ཁང་པ་ཐོག་རྩེག་གསུམ་ཅན་ཞིག་ཡོད་པ་དེས་ཡུམ་བུ་བླ་སྒང་ཐོག་མར་སྐྲུན་སྐབས་ཀྱི་འཛུགས་སྐྲུན་དང་ལོ་རྒྱུས་ཀྱི་དཔང་རྟགས་མཚོན་པར་བྱས་ཡོད་འདུག ཕྱི་ལོ་ ༡༩༦༢ ལོ་ནས་རྒྱ་ནག་གཞུང་གིས་ཕོ་བྲང་ཡུམ་བུ་བླ་སྒང་རྒྱལ་ཁབ་རིམ་པའི་རིག་དངོས་སྲུང་སྐྱོབ་གྲས་སུ་བཞག་ཡོད་པ་དང་། དེ་སྔོན་ཕོ་བྲང་དེའི་ནང་གི་ལྡེབས་རིས་ངོས་སུ་བོད་རྗེ་གཉའ་ཁྲི་བཙན་པོ་ཇི་ལྟར་བྱོན་ཚུལ་བྲིས་ཡོད་ཀྱང་ད་ལྟ་མཇལ་རྒྱུ་མེད་པ་སོགས་ལོ་རྒྱུས་ཀྱི་བྱུང་བ་དངོས་དང་། རྩ་ཆེན་གྱི་རིག་དངོས་མང་པོ་ད་ལྟ་མཐོང་རྒྱུ་མེད་པར་གྱུར་ཡོད་འདུག གསར་འགོད་པ། པདྨ་རྒྱལ།
