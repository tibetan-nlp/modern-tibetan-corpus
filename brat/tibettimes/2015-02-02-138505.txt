གྲྭ་བ་ཞིག་ཁྲིན་ཏུའུ་ནས་བོར་འདུག

ཟླ་སྔོན་མའི་ཟླ་མགོ་ཙམ་ལ་ཕ་ཡུལ་མཚོ་སྔོན་མང་ར་ནས་ཡིན་པའི་ཟླ་ཚང་དགེ་འདུན་རྒྱ་མཚོ་ལགས་ཁྲིན་ཏུའུ་ནས་རྒྱ་གཞུང་གིས་འཛིན་བཟུང་བྱས་ཡོད་འདུག དེ་རིང་བོད་ནས་གནས་ཚུལ་འབྱོར་བ་ལྟར་ན། ཟླ་སྔོན་མའི་ཚེས་ ༡༠ ཡས་མས་ཤིག་ལ་ཁྲིན་ཏུའུ་གྲོང་ཁྱེར་དུ། མཚོ་སྔོན་ཞིང་ཆེན་མཚོ་ལྷོ་ཁུལ། མང་ར་རྫོང་ཐར་ཤུལ་ཡུལ་ཚོའི་ཟླ་ཚང་དགེ་འདུན་རྒྱ་མཚོ་ཞུ་བར་རྒྱུ་མཚན་གསལ་ཁ་མེད་པར་རྒྱ་གཞུང་གིས་འཛིན་བཟུང་བྱས་ཡོད་འདུག དགེ་འདུན་རྒྱ་མཚོ་ལགས་ཀྱིས་ ༢༠༠༨ ལོའི་སྔོན་ལ་ཐར་ཤུལ་དགོན་པ་དང་ལྷ་ས་འབྲས་སྤུངས་དགོན་པ་བཅས་སུ་སློབ་གཉེར་གནང་ཡོད་པ་དང་། ༢༠༠༨ ལོར་ཁོང་ལྷ་ས་ནས་འཛིན་བཟུང་གིས་ཟླ་གསུམ་ལྷག་ལ་བཀག་ཉར་བྱས་ཡོད་ཅིང་། རྗེས་སུ་ཀློད་གྲོལ་ཐོབ་རྗེས་མཚོ་ལྷོ་མང་ར་རྫོང་ཐོ་ལེ་དགོན་པར་སློབ་གཉེར་བྱས་ཏེ་རྒྱུན་སྡོད་གནང་ཡོད་འདུག ཉེ་ལམ་དགོན་པའི་དགུན་ཁའི་གུང་གསེང་དང་བསྟུན་ཏེ་རོགས་པ་གྲྭ་བ་གཞན་གཉིས་དང་མཉམ་དུ་ཁྲིན་ཏུའུ་རུ་རྒྱ་ཡིག་སྦྱོང་དུ་ཕྱིན་ཡོད་འདུག ཉིན་ཉི་ཤུ་ཙམ་གྱི་སྔོན་ལ་ཁྲིན་ཏུའུ་རུ་རྒྱ་ཡིག་འཛིན་གྲྭ་གྲོལ་ནས་ཕྱིར་ཡོང་བའི་ལམ་བར་དུ། མིག་ཤེལ་ནག་པོ་གྱོན་ཡོད་པའི་མི་ཁག་ཅིག་གིས་ཁོང་འཛིན་བཟུང་གིས་བཙན་འཁྲིད་བྱས་ཡོད་པ་དང་། འཛིན་བཟུང་བྱེད་མཁན་དེ་དག་གིས་དགེ་འདུན་རྒྱ་མཚོ་ལགས་ཀྱི་གློག་ཀླད་དང་། རྩོམ་ཡིག་སོགས་ཡིག་ཆ་ཁག་ཅིག་ཡོད་པ་ཚང་མ་འཁྱེར་ཡོད་འདུག ཁོ་རོགས་པ་གཉིས་ཀྱིས་ཉིན་དེར་དགེ་འདུན་རྒྱ་མཚོ་འཛིན་བཟུང་བྱེད་མཁན་དེ་དག་རྒྱ་གཞུང་གསང་བའི་ལས་ཁུངས་ཀྱི་མི་སྣ་ཡིན་པའི་ངོས་འཛིན་བྱེད་བཞིན་ཡོད་པ་དང་། ཉིན་དེ་ནས་ད་བར་ཁོང་གར་ཡོད་ཅི་ཆགས་ཀྱི་གནས་ཚུལ་གང་ཡང་ཤེས་རྟོགས་མེད་པར་གྱུར་ཡོད་འདུག དགེ་འདུན་རྒྱ་མཚོ་ལགས་ནི་མཚོ་སྔོན་མང་ར་རྫོང་ཐར་ཤུལ་ཡུལ་མཚོ་ནས་ཡིན་པ་དང་། ཁོང་གི་སྨྱུག་མིང་ལ་ཟླ་ལྡེ་མིག་དང་། དྲ་མིང་ལ་བདེན་པའི་མེ་ལྕེ་ཞུ་བཞིན་ཡོད་འདུག རྩོམ་པ་པོ་དགེ་འདུན་རྒྱ་མཚོ་ལགས་དུས་རྒྱུན་བོད་མི་རིགས་ཀྱི་བདེ་སྡུག་ལ་སེམས་ཁུར་གནང་མཁན་ཞིག་ཡིན་འདུག་ལ་ཁོང་ད་ལྟ་རང་ལོ་ ༢༧ ཡིན་འདུག
