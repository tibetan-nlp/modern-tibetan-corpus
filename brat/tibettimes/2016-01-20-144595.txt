བོད་ཀྱི་རིགས་རྒྱུད་ཆད་ཁུངས་རྒྱ་གར་ཡིན།

  བོད་ཀྱི་གནད་དོན་འདི་བོད་དང་རྒྱ་གར་གཉིས་ཀའི་གནད་དོན་ཞིག་ཆགས་ཀྱི་ཡོད་སྟབས་ང་ཚོས་བོད་དོན་བདེན་མཐའ་མ་གསལ་བར་དུ་མུ་མཐུད་རྒྱབ་སྐྱོར་བྱ་རྒྱུ་ཡིན་ཟེར།   དེ་རིང་སྔ་དྲོའི་ཆུ་ཚོད་ ༡༡།༣༠ སྟེང་། གངས་སྐྱིད་ཕྱི་དྲིལ་ལྷག་པ་ཚེ་རིང་ཚོགས་ཁང་དུ། རྒྱ་གར་བྱང་ཤར་ཧི་མ་ཅལ་ཁུལ་གྱི་སློབ་ཕྲུག་སྐོར་ཞིག་གིས་བོད་པར་རྒྱུས་ལོན་བྱས་པའི་སྐོར་ལ་གསར་འགོད་གསལ་བསྒྲགས་ཤིག་སྤེལ་སོང་། སྐབས་དེར་བོད་མིའི་སྒྲིག་འཛུགས་ཕྱི་དྲིལ་དྲུང་ཆེ་བཀྲ་ཤིས་ཕུན་ཚོགས་མཆོག་དང་བོད་མི་མང་སྤྱི་འཐུས་ཨ་ཅར་ཡ་ཡེ་ཤེས་ཕུན་ཚོགས་ལགས། དེ་བཞིན། རྒྱ་གར་བྱང་ཤར་ཧི་མ་ཅལ་གྱི་མཐོ་སློབ་ཁག་ནས་ཕེབས་པའི་སློབ་ཕྲུག་ཕོ་མོ་ ༣༠ ལྷག་བཅས་ལྷན་ཞུགས་གནང་སོང་།   ཕྱི་དྲིལ་དྲུང་ཆེ་མཆོག་གིས། བོད་ཀྱི་ཆོས་དང་རིག་གཞུང་། སྐད་ཡིག་དང་རིག་གནས། བོད་ཀྱི་ཁོར་ཡུག ལྷག་པར་དུ་བོད་རྒྱའི་མཐའ་མཚམས་ཁག་གི་འབྲེལ་བ་སོགས་ནི་བོད་དང་རྒྱ་གར་མི་རིགས་གཉིས་ཀྱི་གནའ་དེང་གི་བྱུང་རབས་ལོ་རྒྱུས་ཁྲོད་དུ་གལ་གནད་ཆེན་པོ་ཡིན་པ་དང་། བོད་རྒྱལ་ཁབ་ཤོར་བའི་སྔ་རྗེས་ཀྱི་གནས་ཚུལ་མདོར་བསྡུས་ངོ་སྤྲོད་ཞུས་སོང་།   སྤྱི་འཐུས་ཨཅརྱ་ཡེ་ཤེས་ཕུན་ཚོགས་ལགས་ཀྱིས། བོད་རྒྱ་གཉིས་བར་གདོད་མ་ནས་འབྲེལ་བ་ཤུགས་ཆེན་པོ་ཡོད་པ་དང་། གནའ་བོའི་རྒྱ་གར་མི་རིགས་ཕན་ཌ་བ་སི་ (Pandavas) དང་། ཀུ་ར་བ་སི་ (Kauravas) གཉིས་བར་དམག་འཁྲུག་བྱུང་སྟེ་ཀུ་ར་བ་སི་ཕམ་ནས་བོད་དུ་བྱོལ་པ་ལས་བོད་ཀྱི་མི་རྒྱུད་ཅིག་བྱུང་བའི་ལོ་རྒྱུས་ཡོད་པ། ཡང་བོད་ཀྱི་རིགས་རྒྱུད་གཞན་ཞིག་ནི་སྤྲེའུ་བྱང་ཆུབས་སེམས་དཔའ་ཡིན་པའི་ལོ་རྒྱུས་ཀྱང་ཡོད་པ་དང་། བོད་ཀྱི་བཙན་པོ་རིམ་བྱོན་སྐབས་ཀྱི་འབྲེལ་དྲིས་ཐད། བོད་རྒྱལ་གཉའ་ཁྲི་བཙན་པོ་དེ་རྒྱ་གར་གྱི་མ་ཀ་ཊའི་(makata) རིགས་རྒྱུད་དུ་བཤད་སྲོལ་ཡོད། ཅེས་སོགས་བརྗོད་སོང་།   ཨཅརྱ་ཡེ་ཤེས་ཕུ་ཚོགས་ལགས་ཀྱིས་མུ་མཐུད་དུ། བོད་དང་རྒྱ་གར་གཉིས་ལ་མཐའ་མཚམས་ཀྱི་འབྲེལ་བ་དང་ཚོང་ལས་ཀྱི་འབྲེལ་བ། རང་བྱུང་ཁོར་ཡུག་དང་གཙང་ཆུའི་འབྲེལ་བ། ལྷག་པར་དུ་ཆོས་དང་རིག་གཞུང་། སྐད་ཡིག་རིག་གནས་སོགས་ཕྱི་གཏོང་ནང་འདྲེན་གྱི་འབྲེལ་བ་གཏིང་ཟབ་ཡོད་པ་འདི་དག་ཤེས་ཡོང་ན་ད་ལྟའི་བོད་རྒྱལ་ཁབ་ཤོར་བའི་གནས་བབ་འདི་འདྲའི་འོག རྒྱ་གར་གྱིས་ལ་བོད་སྲུང་སྐྱོབ་གནང་དགོས་པའི་གནས་སྟངས་ཤིག་རང་བཞིན་གྱིས་ཆགས་ཡོད་སྐོར་སོགས་བརྗོད་སོང་།   དེའི་རྗེས་ཧི་མ་ཅལ་བྱང་ཤར་ཁུལ་གྱི་སློབ་ཕྲུག་ཚོགས་པའི་ཚོགས་གཙོ་ན་གྷ་ཤཱ་ཐ་ཀུར་ (Nagesh Thakur) ལགས་ཀྱིས། ད་ལྟའི་བོད་ཀྱི་རྩ་དོན་འདི་གནད་འགག་ཆེན་པོ་ཆགས་ཡོད་སྟབས། དེ་རིང་སློབ་ཕྲུག་ཚོགས་པའི་གོ་སྐབས་འདི་བརྒྱུད་ནས་ང་ཚོས་བོད་ཀྱི་དངོས་ཡོད་གནས་སྟངས་གང་འཚམ་ཞིག་ཤེས་རྟོགས་བྱུང་བ་དང་། རྒྱ་གར་ཧི་མ་ཅལ་བྱང་ཤར་ཁུལ་དུ་རྒྱ་ནག་གི་དཔལ་འབྱོར་འཛུགས་སྐྲུན་གྱི་ཉེན་ཁ་ཆེན་པོ་ཡོད་པ་དང་། རྒྱ་གར་བྱང་ཤར་ཁུལ་ནི་བོད་དང་རྒྱ་གར་གྱི་འབྲེལ་ལམ་གཙོ་བོ་ཞིག་ཆགས་ཡོད་སྟབས། གནད་དོན་འདི་བོད་དང་རྒྱ་གར་གཉིས་ཀར་འབྲེལ་བ་ཆགས་ཡོད་པ་དང་། ད་ལྟའི་བར་ང་ཚོ་སློབ་ཕྲུག་ཚོགས་པའི་ངོས་ནས་བོད་དོན་ལ་རྒྱབ་སྐྱོར་བྱེད་མུས་ཡིན་པ་ལྟར། མ་འོངས་པར་ང་ཚོས་མུ་མཐུད་དུ་བོད་དོན་བདེན་མཐའ་མ་གསལ་བར་དུ་རྒྱབ་སྐྱོར་གནང་རྒྱུ་ཡིན་སྐོར་བརྗོད་སོང་།   གསར་འགོད་པ། པདྨ་རྒྱལ།
