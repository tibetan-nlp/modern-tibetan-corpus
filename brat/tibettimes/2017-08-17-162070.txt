
བོད་སྐད་ཡིག་སྦྱོང་དོན་ནང་ཆོས་དང་སྐད་སྒྱུར་གྱི་ཆེད་དུ་རེད།

བོད་པའི་སློབ་གཉེར་བ་ཚོས་རྒྱུགས་འབྲས་གཙོ་བོར་འཛིན་པ་དང་། ཕྱི་རྒྱལ་གྱི་སློབ་གཉེར་བ་ཚོས་སྐད་ཡིག་སློབ་གསོའི་རྨང་གཞི་གཙོ་བོར་འཛིན་གྱི་འདུག་ཟེར། ལོ་རེ་བཞིན་ཕྱིའི་རྒྱལ་ཁབ་ཁག་ནས་ས་རཱ་བོད་ཀྱི་མཐོ་རིམ་སློབ་གཉེར་ཁང་དུ་བོད་ཡིག་བོད་སྐད་སྦྱོང་མཁན་སློབ་གཉེར་བ་མང་པོ་ཡོང་བཞིན་ཡོད་པ་ལྟར། འདི་ག་གསར་ཁང་གི་གསར་འགོད་པ་ས་རཱ་མཐོ་སློབ་ཏུ་བསྐྱོད་དེ། གནས་ཚུལ་དེའི་སྐོར་རྒྱུས་ལོན་བྱེད་སྐབས། ད་ལྟའི་ཆར་ས་རཱ་མཐོ་སློབ་ནང་ཕྱི་རྒྱལ་གྱི་སློབ་མ་ཚོར་བོད་ཡིག་བོད་སྐད་སློབ་ཁྲིད་བྱེད་པའི་འཛིན་གྲྭ་གསུམ་ཡོད་པ་དང་། སྔོན་འགྲོའི་འཛིན་གྲྭར་ཆུ་ཚོད་གཉིས་རིང་བོད་སྐད་གཅིག་པུ་བཤད་སྟངས་སློབ་ཀྱི་ཡོད་པ་དང་། འཛིན་གྲྭ་དང་པོའི་ནང་བོད་སྐད་བཤད་སྟངས་བོད་ཡིག་འབྲི་སྟངས་སློབ་ཀྱི་ཡོད། འཛིན་གྲྭ་གཉིས་པ་ནས་བོད་ཀྱི་བརྡ་སྤྲོད་སུམ་རྟགས་གཉིས་དང་ཡིག་གཟུགས་སོགས་སློབ་ཁྲིད་བྱེད་ཀྱི་འདུག བོད་པའི་སློབ་གཉེར་བ་དང་ཕྱི་རྒྱལ་གྱི་སློབ་གཉེར་བ་ཚོས་བོད་སྐད་ཡིག་སློབ་སྦྱོང་བྱེད་སྟངས་མི་འདྲ་བའི་སྐོར་ལ་ཕྱི་རྒྱལ་གྱི་སློབ་ཕྲུག་ཚོའི་དགེ་རྒན་དོན་འགྲུབ་སྒྲོལ་མ་ལགས་ཀྱིས། བོད་པའི་སློབ་ཕྲུག་དང་བསྡུར་ན་ཕྱི་རྒྱལ་གྱི་སློབ་མ་ཚོས་སློབ་སྦྱོང་ལ་འབད་བརྩོན་དང་དོ་སྣང་ཆེན་པོ་བྱེད་པ་མ་ཟད། ཁོ་ཚོས་སྐད་ཡིག་གང་སྦྱོང་བ་དེའི་རྨང་གཞི་ཡག་པོ་ཆགས་ཐབས་སུ་འབད་བརྩོན་བྱེད་ཀྱི་འདུག བོད་པའི་སློབ་ཕྲུག་མང་ཆེ་བས་ཡིག་རྒྱུགས་དང་སྐར་གྲངས་ལ་དང་དོད་ཆེན་པོ་བྱེད་ཀྱི་འདུག ཁྱད་པར་གཞན་ཞིག་ནི། ཕྱི་རྒྱལ་གྱི་སློབ་གཉེར་བ་ཚོས་འཛིན་གྲྭའི་ནང་འཁྲིད་གཞི་གང་ཞིག་གསལ་བོར་མ་ཤེས་བའི་སྐབས་དེར་དགེ་རྒན་ལ་ཐད་ཀར་དོགས་འདྲི་བྱེད་པ་དང། སློབ་མ་ཕན་ཚུན་བར་གོ་བསྡུར་བྱས་ཏེ་སོ་སོའི་གང་མ་ཤེས་པ་དེ་གསལ་བོ་ཤེས་ཐབས་བྱེད་ཀྱི་འདུག ཡིན་ནའང་། བོད་པའི་སློབ་གཉེར་བ་ཚོས་དགེ་སློབ་ཕན་ཚུན་བར་གོ་བསྡུར་དང་དོགས་འདྲི་བྱེད་པའི་གོམས་གཤིས་དེ་འདྲ་ཧ་ལམ་ཉུང་ཉུང་ཡིན་པའི་མཐོང་སྣང་བྱུང་སྐོར་བརྗོད་བྱུང་། ད་ལྟ་ས་རཱ་མཐོ་སློབ་ནང་བོད་ཡིག་བོད་སྐད་སྦྱོང་མཁན་ཕྱི་རྒྱལ་གྱི་སློབ་གཉེར་བ་དེ་ཚོ་ཉི་ཧོང་དང་སེང་གྷ་པུར། སོག་པོ། ཐའེ་ལན་ཌི། ཝེ་ཋི་ནམ། ཀོ་རི་ཡ། ཐའེ་ཝན། རྒྱ་གར། ཨ་རི། ཁེ་ན་ཊ། དེ་བཞིན། ཡོ་རོབ་རྒྱལ་ཁབ་ཁག་ནས་ཡིན་ཞིང་། ཁྱོན་སློབ་གཉེར་བ་ ༣༣ ཙམ་ཡོད་པ་དང་། ཁོ་ཚོས་བོད་ཡིག་བོད་སྐད་སྦྱོང་བའི་དམིགས་ཡུལ་གཙོ་བོ་ནི། བོད་ཀྱི་ནང་པའི་ལྟ་གྲུབ་དང་རིགས་ལམ་སྐོར་ལ་སློབ་གཉེར་བྱ་རྒྱུ་དང་། སྐད་ཡིག་ཕན་ཚུན་བར་ལོ་ཙཱ་བྱ་རྒྱུ་སོགས་ཡིན་འདུག གསར་འགོད་པ། སངས་རྒྱས་དཀོན་མཆོག
