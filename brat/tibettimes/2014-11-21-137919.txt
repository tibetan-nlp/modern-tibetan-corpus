ལྷོ་ཨ་རི་ཁུལ་གྱི་མཛེས་མ་ཞིག་བསད་འདུག

ཧོན་ཌོབ་རེ་སི་རྒྱལ་ཁབ་ཀྱི་ད་ལོའི་མཛེས་མ་ཨང་དང་པོ་མཱ་ར་ཡ་ཇོ་སེ་ལགས་དང་མོ་རང་གི་གཅེན་མོ་གཉིས་གཅེན་མོའི་དགའ་རོགས་ཀྱིས་དམར་གསོད་བྱས་ཏེ་ཕུང་བོ་ས་འོག་ཏུ་སྦས་ཡོད་པ་ཉེན་རྟོག་པས་ཤེས་རྟོགས་བྱུང་འདུག    ཟླ་འདིའི་ཚེས་ ༡༣ ཉིན་ཨ་རིའི་ལྷོ་ཁུལ་གྱི་ཧོན་ཌོབ་རེ་སི་ (Honduras)རྒྱལ་ཁབ་ཀྱི་རང་ལོ་ ༡༩ སོན་པའི་ ༢༠༡༤ ལོའི་མཛེས་མ་མཱ་ར་ཡ་ཇོ་སེ་ (Marie Jose Alvarado) དང་། ཁོང་མོའི་གཅེན་མོ་སོབ་ཧྥེ་ཡ་གཉིས་ཀྱིས། གཅེན་མོའི་དགའ་གྲོགས་རོ་ཛེ་ (Plutarco Ruiz)ཡི་སྐྱེས་སྐར་མཚན་ཚོགས་སུ་ཞུགས་ཏེ་ཕྱིར་ལོག་པའི་ལམ་བར་ནས་བོར་བརླག་ཕྱིན་ཏེ་གཟའ་འཁོར་གཅིག་ཕྱིན་རྗེས། ཁ་སང་ཉེན་རྟོག་པས་གཙང་བོ་ཞིག་གི་འགྲམ་བརྒྱུད་ནས་མོ་རང་དང་གཅེན་མོའི་ཕུང་བོ་ས་འོག་ཏུ་སྦས་ཡོད་པ་རྙེད་སོན་བྱུང་ཡོད་འདུག ཉེན་རྟོག་ལས་ཁུངས་ཀྱིས་གསལ་བསྒྲགས་སྤེལ་བའི་ནང་། ཁོང་གཉིས་ཨ་ར་ཌལ་གཙང་འགྲམ་དུ་བསད་ཡོད་སྟབས། ཁོང་གཉིས་གསོད་མཁན་སོབ་ཧྥེ་ཡ་ཡི་དགའ་རོགས་རོ་ཛེ་དང་ཁོའི་གྲོགས་པོ་གཉིས་ཡིན་པའི་དོགས་གཞིའི་འོག་ཁོ་རང་གཉིས་འཛིན་བཟུང་བྱས་ཡོད་པ་མ་ཟད། རོ་ཛེའི་ཁྱིམ་ནས་ཐུང་མདའ་གཅིག་དང་དེབ་སྐྱེལ་མེད་པའི་རླངས་འཁོར་གཉིས་ཀྱང་རག་ཡོད་ཚུལ་བརྗོད་འདུག རྒྱལ་ཡོངས་ཉེན་རྟོག་ལས་ཁུངས་སྤྱི་ཁྱབ་པས་གསར་འགོད་པར། རོ་ཛེ་ཡིས་གྲོགས་པོ་ཨ་རེ་སིའི་རོགས་རམ་འོག་བུ་མོ་གཅེན་གཅུང་གཉིས་བསད་ནས་ས་འོག་ཏུ་བཅུག་པའི་ནག་ཉེས་ཁས་ལེན་བྱས་ཡོད་སྟབས། ཁྲིམས་ཁང་གིས་རྒྱལ་ཁབ་ཀྱི་ཉེས་ཁྲིམས་ལྟར་ཁྲིམས་ཐག་གཅོད་བྱ་ངེས་རེད་ཅེས་བརྗོད་འདུག   དེ་གའི་ས་གནས་གསར་ལམ་ཁག་ཏུ། སྐྱེས་སྐར་མཚན་ཚོགས་དེའི་སྟེང་སོབ་ཧྥེ་ཡིས་ཕོ་གསར་གཞན་དང་མཉམ་དུ་ཞབས་བྲོ་འཁྲབ་པར་བརྟེན། མོའི་དགའ་རོགས་ཕྲག་དོག་ལངས་ཏེ་ཐོག་མར་རང་ཉིད་ཀྱི་དགའ་རོགས་བསད་པ་དང་། གནས་ཚུལ་ཕྱིར་བསྒྲགས་བྱེད་པར་དོགས་ཏེ་གཅུང་མོ་ཡང་བསད་ཡོད་ཚུལ་བཀོད་འདུག ཟླ་འདིའི་ཚེས་ ༢༢ ཉིན་ནས་ཟླ་རྗེས་མའི་ ༡༤ ཉིན་བར་དབྱིན་ཡུལ་རྒྱལ་ས་ལོན་ཊོན་ཏུ་འཚོགས་གཏན་འཁེལ་བའི་ ༢༠༡༤ ལོའི་འཛམ་གླིང་མཛེས་མའི་འགྲན་བསྡུར་ནང་ཞུགས་འཆར་ཡོད་སྟབས། ཐེངས་འདིའི་མཛེས་མ་འགྲན་བསྡུར་གོ་སྒྲིག་ཚོགས་གཙོས་མོའི་ནང་མི་ཚོར་སེམས་གསོ་དང་གདུང་སེམས་མཉམ་སྐྱེད་ཞུས་ཡོད་འདུག ཐེངས་འདིའི་གནས་ཚུལ་དེ་དང་བསྟུན་ཏེ། ལྷོ་ཨ་རིའི་ཁུལ་དུ་བུད་མེད་ཚོར་བརྙས་བཅོས་དང་མེ་མདའ་གང་བྱུང་བེད་སྤྱོད་ཀྱི་གནས་ཚུལ་ཐོག་གླེང་སློང་ཆེ་རུ་ཕྱིན་ཡོད་འདུག བརྙན་འཕྲིན་ལས་ཁུངས་ཤིག་གི་གཙོ་སྐྱོང་བས། བུ་མོ་མང་པོ་ཞིག་སྙན་གྲགས་གང་མེད་པས་དོན་རྐྱེན་འདི་ལྟ་བུའི་འོག་ཤི་འགྲོ་བཞིན་ཡོད་པ་མ་ཟད། མི་གསོད་ཉེས་ཅན་ཚོའང་ནག་ཉེས་མེད་པར་ཐར་འགྲོ་བཞིན་ཡོད། མོ་རང་གི་དོན་རྐྱེན་འདི་བརྒྱུད་ནས་མེ་མདའི་འོག་གི་རྩེད་མོ་དང་སྤྱི་ཚོགས་ནང་གི་འཚེ་བའི་ལམ་ལུགས་མཇུག་རྫོགས་པའི་རེ་བ་བྱེད་ཀྱིན་ཡོད། ཅེས་བརྗོད་འདུག མཱ་ར་ཡ་ཇོ་སེ་ནི་ ༡༩༩༥ ལོའི་ཟླ་ ༧ ཚེས་ ༡༩ ཉིན་སྐྱེས་ཤིང་། ད་ལོའི་ཟླ་ ༤ པའི་ནང་ལྷོ་ཨ་རི་ཁུལ་གྱི་རྒྱལ་ཁབ་ ༡༨ ཡི་མཛེས་མའི་འགྲན་བསྡུར་ཁྲོད་དང་པོའི་མཛེས་མའི་རྒྱལ་ཁ་ཐོབ་ཡོད། ད་ལྟའི་བར་བརྙན་འཕྲིན་ལས་ཁུངས་ནང་གྱོན་ཆས་འཁྲབ་སྟོན་བྱེད་བཞིན་ཡོད་འདུག གསར་འགོད་པ། ལྷ་སྐྱབས་རྒྱལ།
