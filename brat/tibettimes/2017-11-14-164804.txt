
རྒྱ་བལ་གྱི་ཆུ་གློག་གྲོས་ཆོད་བལ་གཞུང་གིས་རྩིས་མེད་བཏང་།

རྒྱ་གཞུང་གིས་འགྲོ་གྲོན་ཨ་སྒོར་ཐེར་འབུམ་ ༢.༥ ཙམ་བཏང་སྟེ་བོད་བལ་ས་མཚམས་སུ་ཆུ་མཛོད་གློག་འདོན་ཁང་ཞིག་རྒྱག་འཆར་ཡོད་པའི་ལས་གཞི་དེ་བལ་གཞུང་གིས་རྩིས་མེད་བཏང་ཡོད་འདུག   རྒྱ་ནག་ལྷོ་རྒྱུད་ཞོགས་པའི་ཆ་འཕྲིན་གྱིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། རྒྱ་ནག་གཞུང་འབྲེལ་གྱི་བཟོ་གྲྭ་ཀི་ཀྲུའུ་བྷ་(Gezhouba Group)ཚད་ཡོད་ཚོང་གཉེར་ཁང་ཟེར་བ་ཞིག་གིས། བལ་ཡུལ་གྱི་རྒྱལ་ས་ཀཏ་མན་ཌུ་ནས་སྤྱི་ལེ་ ༨༠ ཙམ་གྱི་སར་ཡོད་པའི་བོད་བལ་ས་མཚམས་སུ་ཆུ་མཛོད་གློག་འདོན་ཁང་ཆེན་པོ་ཞིག་རྒྱག་འཆར་ཡོད་ཀྱང་། བལ་ཡུལ་གཞུང་གིས་ལས་འཆར་དེའི་གྲོས་ཆོད་རྩིས་མེད་བཏང་ཡོད་འདུག   ཆུ་མཛོད་གློག་འདོན་ཁང་གི་ལས་འཆར་དེ་ད་ལོའི་ཕྱི་ཟླ་ ༦ པའི་ནང་རྒྱ་ནག་དང་བལ་ཡུལ་གཞུང་གཉིས་ནས་གྲོས་མོལ་བྱས་ཏེ་གྲོས་ཆོད་བཞག་ཡོད་པ་དང་། ལས་འཆར་དེ་ལེགས་འགྲུབ་བྱུང་ཚེ་གློག་ཤུགས་ ༡༢༠༠ MW མཁོ་སྤྲོད་བྱེད་ཐུབ་པའི་རྒྱ་བལ་གཉིས་བར་གྱི་ཆུ་མཛོད་གློག་འདོན་ཁང་ཆེ་ཤོས་ཆགས་ཀྱི་ཡོད་འདུག   ཁ་སང་ཉིན་གུང་ལས་འཆར་དེ་རྩིས་མེད་བཏང་བའི་སྐོར་ལ་བལ་ཡུལ་གྱི་སྲིད་བློན་གཞོན་པ་ཀ་མལ་ཐ་པ་(Kamal Thapa)ལགས་ཀྱིས། དེ་རིང་བློན་ཆེན་རྣམས་འདུ་འཛོམས་ཀྱིས་ཀི་ཀྲུའུ་བྷ་ཚད་ཡོད་ཚོང་གཉེར་ཁང་དང་གྲོས་མོལ་བརྒྱུད་ཐག་གཅོད་བྱུང་བའི་ཆུ་གློག་ཁང་གི་ལས་འཆར་བུ་དྷི་གན་དྷ་ཀི་(Budhi Gandaki)ཡི་སྐོར་ལ་གོ་སྡུར་བྱུང་སོང་། ང་ཚོས་ལས་འཆར་དེ་ནི་བསམ་བློ་མ་བཏང་བར་ཐག་གཅོད་བྱས་པ་ཞིག་མཐོང་བ་མ་ཟད། དེ་ནི་ནོར་འཁྲུལ་ཡིན་པ་ཤེས་རྟོགས་བྱུང་སྟབས། གྲོས་ཚོགས་ནས་ལས་འཆར་དེ་རྩིས་མེད་བཏང་བ་ཡིན། ཞེས་བརྗོད་འདུག   ད་ལན་བལ་ཡུལ་གཞུང་གིས་གྲོས་ཆོད་རྩིས་མེད་བཏང་བ་དེས་ཧི་མ་ལ་ཡའི་རི་རྒྱུད་ཀྱི་རྒྱལ་ཁབ་ཁག་གི་ནང་རྒྱ་ནག་གཞུང་གིས་བཟོ་གྲྭ་དང་གཞུང་ལམ་སོགས་འཛུགས་བསྐྲུན་གྱི་ལས་གཞི་སྣ་ཚོགས་མགོ་བརྩམས་ཏེ། ཁྱིམ་མཚེས་རྒྱལ་ཁབ་ཁག་ལ་ཤུགས་རྐྱེན་སྤྲོད་པའི་རྨང་གཞི་བསྐྲུན་འདོད་པའི་སྲིད་བྱུས་ལ་འགོག་རྐྱེན་ཞིག་ཆགས་ཡོད་པ་རེད།   བལ་ཡུལ་གློག་གི་བློན་ཆེན་ལས་ཚབ་པ་དི་ནེ་ཤ་ཀུ་མར་གྷི་མཱིར་(Dinesh Kumar Ghimire)ལགས་ཀྱིས། ཁ་སང་སྲིད་བློན་གཞོན་པས་བློན་ཆེན་ལྷན་ཚོགས་ནས་ཐག་གཅོད་གང་བྱུང་བ་འགྲེལ་བརྗོད་བྱས་ཡོད་པ་དང་། ཕལ་ཆེར་དེ་རིང་གྲོས་ཆོད་རྩིས་མེད་བཏང་པའི་ཐད་ལ་གཞུང་འབྲེལ་གྱི་གསལ་བསྒྲགས་གཏང་སྲིད་པ་རེད། ཅེས་བརྗོད་འདུག  བལ་ཡུལ་དུ་ཡོད་པའི་ཀི་ཀྲུའུ་བྷ་ཚད་ཡོད་ཚོང་གཉེར་ཁང་གི་འབྲེལ་མཐུད་པས་བརྗོད་པ་ལྟར་ན། ཀི་ཀྲུའུ་བྷ་ཚོང་གཉེར་ཁང་ལ་ད་ལྟ་བར་བལ་ཡུལ་གཞུང་ནས་ལས་འཆར་དེ་རྩིས་མེད་བཏང་པའི་སྐོར་ལ་བརྡ་ལན་གང་ཡང་འབྱོར་མེད་པ་དང་། ལས་འཆར་དེ་འཆར་ལོའི་ནང་མགོ་འཛུགས་བྱ་རྒྱུ་ཡིན་སྐོར་བརྗོད་འདུག བལ་ཡུལ་གཞུང་དང་ཀི་ཀྲུའུ་བྷ་ཚད་ཡོད་ཚོང་གཉེར་ཁང་གཉིས་ཀྱིས་ད་ལྟ་བར་ཉུང་མཐར་ཆུ་མཛོད་གློག་འདོན་ཁང་གི་ལས་འཆར་དེ་འདྲ་གཉིས་ཙམ་ལ་གྲོས་ཆོད་བཞག་ཡོད་པ་དང་། གཞན་ཡང་ཚོང་གཉེར་ཁང་དེས་བལ་ཡུལ་ནང་སྨན་ཁང་དང་ལམ་ཁ། དེ་བཞིན་གནམ་ཐང་སོགས་བརྒྱབ་པར་རོགས་རམ་བྱེད་ཀྱི་ཡོད་འདུག གསར་འགོད་པ། ཉི་མ་ཚེ་རིང་།
