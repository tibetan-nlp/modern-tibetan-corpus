ཨ་ཏི་ཤ་ལཊ་ཀྱི་སྐུ་པར་ལག་སོན་བྱུང་།

རྒྱ་ནག་གཞུང་གིས་ ༢༠༠༨ ལོའི་རྔ་བའི་ཞི་རྒོལ་སྐབས་དམར་གསོད་བཏང་བའི་བོད་མི་ཨ་ཏི་ཤ་ལཊ་ཀྱི་སྐུ་པར་ཉེ་ལམ་འབྲེལ་ཡོད་བརྒྱུད་ཕྱི་ཕྱོཊ་སུ་འབྱོར་ཡོད། ཕྱི་ལོ་ ༢༠༠༨ ལོའི་ཕྱི་ཟླ་ ༣ ཚེས་ ༡༦ ཉིན་བོད་མདོ་སྨད་རྔ་བ་རྫོང་གི་ཁྲོམ་གཞུང་དུ་དགོན་སྡེ་ཁག་གི་གྲྭ་པ་རྣམས་ཀྱིས་སྣེ་འཁྲིད་དེ་རྔ་ཁོག་སྟོད་སྨད་བར་གསུམ་གྱི་སེར་སྐྱ་མི་མང་ ༡༠༠༠༠ ལྷག་གིས་རྒྱ་གཞུང་ལ་ཞི་བའི་ངོ་རྒོལ་བྱེད་སྐབས། ཉེན་རྟོག་པ་དང་དམག་མིས་མི་མང་ཁྲོད་གང་བྱུང་མེ་མདའ་བརྒྱབ་ཏེ་བོད་མི་མང་པོ་ཞིག་དམར་གསོད་བཏང་ཡོད་པ་ལྟར། སྐབས་དེའི་གནས་ཚུལ་དང་སྐུ་པར་ཁག་ཅིག་ཕྱི་ཕྱོགས་སུ་འབྱོར་ཐུབ་ཡོད་ཀྱང་། ས་གནས་ཀྱི་གནས་སྟངས་དང་འབྲེལ་ལམ་གྱི་དབང་གིས་བོད་མི་ཁག་ཅིག་གིས་གནས་ཚུལ་ཕྱི་ཕྱོགས་སུ་འབྱོར་ཐུབ་མེད་སྟབས། ཉེ་ལམ་འབྲེལ་ཡོད་བརྒྱུད་སྐབས་དེར་རྒྱ་གཞུང་གིས་རྔ་བ་རྫོང་མངགས་གཏོང་ཁང་གི་ཉེ་འཁྲིས་སུ་མེ་མདའ་བརྒྱབ་ཏེ་དམར་གསོད་བཏང་བའི་བོད་མི་ཨ་ཏི་ཤ་ལགས་ཀྱི་སྐུ་པར་ཕྱི་ཕྱོགས་སུ་བཏང་འབྱོར་བྱུང་ཡོད། གྲྭ་གསུང་རབ་རྒྱ་མཚོ་ལགས་ཀྱིས་བརྗོད་པ་ལྟར་ན། རང་ལོ་ ༢༨ ཡིན་པའི་ཨ་ཏི་ཤ་ལགས་ནི། རྔ་སྟོད་སྒོ་མང་འཚོ་བ་དྲུག་གི་ཁོང་གཏོགས་སྔོ་ཤུལ་འཚོ་འདུ་སྒང་ནས་ཡིན་པ་དང་། ཕ་མིང་ལ་ཤེས་རབ་དང་མ་མིང་ལ་ཐང་གོ་ཞུ་བཞིན་ཡོད་འདུག་ལ། ཁོང་ཆུང་དུས་སྔོ་ཤུལ་དགོན་པའི་ནང་གྲྭ་བ་བྱས་ཏེ་ཕྱིས་སུ་སྐྱ་སར་བབས་ཡོད་འདུག ༢༠༠༨ ལོའི་ཕྱི་ཟླ་ ༣ ཚེས་ ༡༦ ཉིན་གྱི་བོད་ནང་གི་སྔ་དྲོའི་ཆུ་ཚོད་ ༡༡ ཡས་མས་སུ་ཁོང་རྫོང་ཐོག་ཏུ་ཕྱིན་སྟེ། རྔ་བ་རྫོང་གི་ཞི་བའི་ངོ་རྒོལ་ཁྲོད་མཉམ་ཞུགས་བྱེད་སྐབས་རྒྱ་གཞུང་ཉེན་རྟོག་དམག་མིས་མེ་མདའ་བརྒྱབ་ཏེ་འདས་གྲོངས་སུ་གྱུར་ཡོད་འདུག གཞན། གནས་ཚུལ་མཁོ་སྤྲོད་པས་བརྗོད་པ་ལྟར་ན། དེ་རིང་ནས་བཟུང་བོད་ནང་གི་ས་ཆ་མང་པོར་དཔྱིད་གསར་དུས་ཆེན་རོལ་གྱིན་པའི་སྐབས་འདིར། རྔ་བ་སྒོ་མང་ལྷ་སྡེའི་ཁྱབ་ཁོངས་ཀྱི་ས་གནས་ཁག་གིས་རྒྱལ་གཅེས་དཔའ་བོ་ཞི་ཆུང་ལཊ་གཙོས་པའི་བསྟན་སྲིད་ཆེད་རང་ལུས་མེར་བསྲེཊ་གནང་མཁན་རྒྱལ་གཅེས་དཔའ་བོ་དཔའ་མོ་རྣམས་ལ་གདུང་སེམས་མཉམ་སྐྱེད་ཆེད། ལོ་སར་སྐབས་གློག་མཆོད་དང་། ཁང་པར་ཚོན་རྩི་བྱུགས་པ། རྟ་རྒྱུགས་གཙོས་པའི་རྩེད་མོའི་རིཊ། ཕར་ཚུར་ལོ་སར་གྱི་འགྲོ་འོང་སོགས་མེད་པར་བཟོས་ཡོད་པ་མ་ཟད། རང་སར་ཁ་འདོན་དགེ་སྦྱོར་དང་བསྔོ་བ་སྨོན་ལམ་སོགས་གསུངས་ཏེ་བསྡད་ཡོད་འདུག་ལ། བོད་མི་མང་པོ་ཞིག་བླ་བྲང་བཀྲ་ཤིས་འཁྱིལ་སོགས་དགོན་སྡེ་དང་གནས་ཆེན་ཁག་ལ་མཆོད་མཇལ་དུ་འགྲོ་བཞིན་ཡོད་འདུག   གསར་འགོད་པ། དོན་གོ
