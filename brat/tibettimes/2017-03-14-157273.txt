རེབ་གོང་དུ་དམ་དྲག་སྔར་ལས་ཆེ་རུ་ཕྱིན།

རྒྱ་ནག་གཞུང་གིས་བོད་རེབ་གོང་ཁུལ་དུ་དྲག་ཆས་སྤྲས་ཤིང་མེ་གསོད་འཕྲུལ་ཆས་བཟུང་བའི་ཉེན་རྟོག་པ་དང་དམག་མི་སྔར་ལས་མང་པོ་འགྲེམས་འཇོག་བྱས་ཏེ་དམ་དྲག་ཆེན་པོ་བྱེད་བཞིན་ཡོད་འདུག འབྲེལ་ཡོད་བོད་མི་ཞིག་གིས་འདི་གར་གནས་ཚུལ་མཁོ་གཏོང་གནང་བ་ལྟར་ན། རྒྱ་ནག་གཞུང་གིས་གསུམ་བཅུའི་དུས་དྲན་སྔ་གཞུག་ཏུ་བོད་ས་ཡོངས་སུ་དམག་མི་མང་པོ་འགྲེམས་འཇོག་གིས་དམ་དྲག་ཤུགས་ཆེ་རུ་བཏང་ཡོད་པ་ལྟར། ཕྱི་ཚེས་༡༠ཉིན་ནས་བཟུང་གསུམ་བཅུའི་དུས་དྲན་དང་ཉིན་འདི་དག་རིང་རྒྱ་ནག་གཞུང་གིས་རྨ་ལྷོ་ཁུལ་ཐོག་དང་རོང་བོ་དགོན་པའི་ཉེ་འཁོར་ལམ་མདོ་དང་སྲིད་གཞུང་ལས་ཁུངས་ཁག་གི་སྒོ་འགྲམ་སོགས་སུ་དྲག་ཆས་དང་མེ་གསོད་འཕྲུལ་ཆས་སོགས་བཟུང་བའི་དམག་མི་དང་ཉེན་རྟོག་པ། དམིགས་བསལ་ཉེན་རྟོག་པ། དེ་བཞིན་ལྭ་བ་དཀྱུས་མ་གྱོན་པའི་ཉེན་རྟོག་པས་ཉིན་མཚན་གཉིས་ཀར་ལྟ་སྲུང་བྱེད་བཞིན་ཡོད་པ་མ་ཟད། ཟ་ཁང་དང་མགྲོན་ཁང་སོགས་སུ་ཉེན་རྟོག་པ་མང་པོ་ཕྱིན་ཏེ། མགྲོན་པོའི་ཐོ་གཞུང་ལ་ཞིབ་བཤེར་བྱས་ཏེ་རྫོང་ཁག་གཞན་ནས་ཕེབས་པའི་བོད་མི་ཚོར་དམིགས་བསལ་བལྟ་སྲུང་ནན་མོ་བྱེད་ཀྱི་ཡོད་པ་དང་། ཞིབ་བཤེར་སྐབས་དོགས་པ་བྱུང་ཚེ་མགྲོན་ཁང་ནང་ཕྱིན་ཏེ་སྔོག་བཤེར་འདྲི་གཅོད་བྱས་ཏེ་ཉེས་མེད་འགྲུལ་བ་ཚོར་སུན་ཙེར་དང་འཇིགས་སྣང་སྐུལ་གྱི་ཡོད་པ་དང་།  དེ་གཞིན་རྒྱ་གར་ནས་བོད་ལ་གཉེན་འཕྲད་དུ་ཕྱིན་པ་རྣམས་ལ་ཉེན་རྟོག་པས་རེབ་གོང་རྫོང་ཐོག་ཏུ་གང་འདོད་འགྲོ་མི་ཆོག་པ་དང་། གལ་ཏེ་འགྲོ་དགོས་བྱུང་ཚེ་ས་གནས་གྲོང་རྡལ་ལས་ཁུངས་ལ་ཡར་ཞུ་བྱེད་དགོས་པའི་བཀའ་བསྒྲགས་ཡོད་འདུག གཞན་ཡང་། རྒྱ་ནག་གཞུང་གིས་ཡར་རྒྱས་དང་འཛུགས་སྐྲུན་ལ་ཁ་གཡར་ཏེ་རེབ་གོང་གི་སྙན་ཐོག་དང་སྒོ་དམར་སྡེ་བའི་ས་ཞིང་ཨེ་ཀ་ཆིག་སྟོང་ལྷག་རྒྱལ་ཁབ་ལ་གོང་ཁེ་བོས་ངེས་པར་དུ་བཙོང་དགོས་པའི་བཙན་བཀའ་བཏང་བར་བརྟེན། ས་གནས་ཡུལ་མི་ཚོ་ལྷན་དུ་འཛོམས་ཏེ་ང་ཚོའི་སྐབས་ལ་ས་ཞིང་བཙོང་ནས་དངུལ་ཅུང་ཙམ་རག་ན་འཚོལ་བ་འཁྱོལ་ཙམ་ཡོང་གི་ཡོད་ཀྱང་། ང་ཚོའི་མི་རབས་རྗེས་མའི་བྱིས་པ་བུ་ཕྲུག་རྣམས་ཀྱིས་འཚོ་བ་རྟེན་སའི་ས་ཞིང་མེད་ན་འཚོ་བ་ཇི་ལྟར་རོལ་དགོས་སམ་ཞེས་ས་གནས་རྫོང་སྲིད་གཞུང་ལ་ཞུ་གཏུག་བྱས་ཡོད་ཀྱང་། ས་གནས་སྲིད་གཞུང་གིས་ཉན་འཇོག་མི་བྱེད་པར་ངེས་པར་དུ་བཙོང་དགོས། མ་བཙོང་ན་མི་ཆོག འདི་ནི་སྲིད་གཞུང་གི་བཀའ་ཡིན་པར་བརྗོད་དེ་སྡེ་མི་ཚང་མ་སེམས་ཁྲལ་གྱིས་གཉིད་ཀྱང་ཁུག་མི་ཐུབ་པའི་གནས་ཚུལ་ལྷུང་ཡོད་འདུག ད་དུང་ཉེ་འཁོར་གྱི་སེང་གཞོང་སྡེ་དང་རྒྱལ་ཚང་མ་སྡེ་བ། ཧོར་རྒྱ་སོགས་སུ་ས་ཞིང་གཞུང་ལམ་ཁ་དང་ཁང་རྩེག་ཉེ་འགྲམ་དུ་ཁེལ་བ་རྣམས་ལ་ཨེ་ཀ་རེར་རྒྱ་སྒོར་ཁྲི་༢༥་ཡས་མས་སྤྲད་ནས་མི་བཙོང་ཐབས་མེད་བྱུང་ཡོད་པ་དང་། སྒོ་དམར་སྡེ་བའི་ས་ཞིང་ལ་ཨེ་ཀ་གཅིག་༡ ལ་རྒྱ་སྒོར་ཁྱི་ ༡༣་དང་ཁྲི་ ༡༤ ཡས་མས་མ་གཏོགས་མི་སྤྲོད་སྐོར་བརྗོད་ཡོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
