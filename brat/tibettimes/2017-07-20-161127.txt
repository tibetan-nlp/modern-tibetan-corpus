མཱ་ལཱ་ལཱ་ཡིས་རྒྱ་ནག་ལ་སྐྱོན་བརྗོད་བྱས།

ཕ་ཀི་སི་ཐན་གྱི་ནོ་བྷེལ་ཞི་བདེའི་གཟེངས་རྟགས་བཞེས་མཁན་བུ་མོ་མཱ་ལཱ་ལཱ་(Malala)ཡིས་རྒྱ་ནག་གི་ནོ་བྷེལ་ཞི་བདེའི་གཟེངས་ཐོབ་མཁན་སྐུ་ཞབས་ལིའུ་ཞོ་པོ་ལ་སྨན་བཅོས་དང་ལྟ་སྐྱོང་ཡག་པོ་མ་ཐོབ་པར་སྐུ་གྲོངས་བར་སྐྱོན་བརྗོད་བྱས། ཁ་སང་གཟའ་ལྷག་པའི་ཉིན་རོའོ་ཊིར་གསལ་ཁང་གིས་གནས་ཚུལ་སྤེལ་བ་ལྟར་ན། ཕ་ཀི་སི་ཐན་གྱི་ ༢༠༡༣ ལོར་འཛམ་གླིང་ནོ་བྷེལ་ཞི་བདེའི་གཟེངས་རྟགས་ཐོབ་མྱོང་མཁན་དང་འགྲོ་བ་མིའི་ཐོབ་ཐང་དང་འཛམ་གླིང་གི་བྱིས་པའི་ཤེས་ཡོན་འཐབ་རྩོད་མཁན་བུ་མོ་མཱ་ལཱ་ལཱ་ཡིས་རྒྱ་ནག་གཞུང་གིས་མི་མང་གི་རང་དབང་ལ་བཀག་སྡོམ་བྱེད་བཞིན་པར་སྐྱོན་བརྗོད་བྱས་འདུག ཁོ་མོས་ “ངས་མི་མང་གི་རང་དབང་ལ་བཀག་སྡོམ་བྱེད་མཁན་གྱི་སྲིད་གཞུང་གང་ཞིག་ཡིན་ཡང་དེར་སྐྱོན་བརྗོད་བྱེད་བཞིན་ཡོད་ལ། ངའི་རེ་བར་རྒྱ་ནག་མི་མང་གིས་སྐུ་ཞབས་ལིའུ་ཞོ་པོ་ཡི་ལས་དོན་གང་བྱས་པ་མིག་དཔེར་བཟུང་སྟེ། ཚང་མས་མཉམ་དུ་རང་དབང་ལ་འཐབ་རྩོད་དང་། མི་མང་གི་ཁེ་དབང་དང་འདྲ་མཉམ་ལ་འཐབ་རྩོད་བྱ་ཐུབ་པའི་རེ་བ་ཡོད།” ཅེས་བརྗོད་འདུག ཁོ་མོ་ཉེ་སྔོན་ཨ་ཧྥི་རི་ཁའི་(Nigeria)རྒྱལ་ཁབ་ནང་ཤེས་ཡོན་ཐད་དཀའ་ངལ་འཕྲད་བཞིན་པར་གོ་རྟོག་སྤེལ་བར་སྐོར་སྐྱོད་གནང་ཡོད་པ་དང་། རྒྱལ་ཁབ་དེ་ནི་ཨ་ཧྥི་རི་ཁའི་ནང་གི་ལུང་པ་ཤེས་ཡོན་གནས་སྟངས་སྐྱོ་ཤོས་དང་བྱིས་པ་ས་ཡ་ ༡༠.༥ ལྷག་སློབ་གྲྭར་འགྲོ་མ་ཐུབ་པར་ལྷག་ཡོད་པ་རེད། དེ་ནི་འཛམ་གླིང་ཡོངས་ཀྱི་བྱིས་པའི་སློབ་སྦྱོང་གོ་སྐབས་ཤོར་བ་ཆེས་མང་བའི་རྒྱལ་ཁབ་དེ་ཆགས་ཡོད་པ་རེད། ཁོ་མོས་ “རྒྱལ་ཁབ་དེའི་བྱང་ཁུལ་དུ་ཁ་ཆེ་ཐལ་སྤྱོད་རིང་ལུགས་པ་ཚོས་དཀའ་རྙོག་བཟོ་བཞིན་ཡོད་སྟབས། ཡུལ་དེའི་ཤེས་ཡོན་གནས་སྟངས་སྐྱོ་བོ་ཡོད་པ་དང་ལོ་གསུམ་གྱི་གོང་དང་བསྡུར་ན་བརྒྱ་ཆ་ ༡༠ ནས་བརྒྱ་ཆ་ ༦ ལ་སྐྱོ་རུ་ཕྱིན་འདུག” ཅེས་བརྗོད་འདུག གཟའ་ཟླ་བའི་ཉིན་ཁོ་མོས་(Nigeria)ཡི་སྲིད་འཛིན་ལ་ཤེས་ཡོན་གནས་སྟངས་ཐད་ལ་མྱུར་མགྱོགས་ངང་ཐབས་ཤེས་འཐེན་དགོས་པ་དང་། རྒྱལ་ཡོངས་སུ་བྱིས་པའི་ཤེས་ཡོན་མྱུར་སྐྱོབ་གཉེན་བརྡ་གཏོང་དགོས་པར་སྐུལ་མ་གནང་ཡོད་པ་རེད། མཱ་ལཱ་ལཱ་ནི་ད་ལོ་ལོ་ ༢༠ ཡིན་པ་དང་། ༢༠༡༠ ལོར་པ་ཀི་སི་ཐན་གྱི་དྲག་སྤྱོད་པས་ཁོ་མོའི་མགོ་ལ་མེ་མདའ་བརྒྱབ་ནས་རྗེས་སུ་ཁོ་མོ་དབྱིན་ཡུལ་ལ་སྐྱབས་བཅོལ་དུ་འགྲོ་དགོས་བྱུང་ཡོད། ཁོ་མོ་ནི་འཛམ་གླིང་གི་ནོ་བྷེལ་ཞི་བདེ་གཟེངས་རྟགས་ཐོབ་མཁན་ལོ་ན་ཆུང་ཤོས་དེ་ཆགས་ཡོད་པ་རེད། གསར་འགོད་པ། སངས་རྒྱས་དཀོན་མཆོག
