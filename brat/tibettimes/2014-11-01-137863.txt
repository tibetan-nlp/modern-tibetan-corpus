རྒྱ་སྒོར་དུང་ཕྱུར་ ༢ ལག་སོན་བྱུང་།

རྒྱ་ནག་རྡོ་སོལ་ཚོང་གཉེར་ཁང་གི་མགོ་ཁྲིད་གཞོན་པ་ཝེ་ཕུང་ཡན་གྱི་ཁྱིམ་དུ་སྔོག་བཤེར་བྱས་ཏེ་རྒྱ་སྒོར་དུང་ཕྱུར་ ༢ ལྷག་ལག་སོན་བྱུང་ཡོད་འདུག ཁ་སང་རྒྱ་ནག་ཆེས་མཐོའི་མི་མང་ཞིབ་བཤེར་ཁང་གིས་གསར་འགོད་ལྷན་ཚོགས་ཤིག་སྐོང་ཚོགས་བྱས་ཏེ། འདི་ལོའི་ལྷན་ཁང་ཁག་གི་ལས་དོན་ཞིབ་བཤེར་དང་ཁྲིམས་འགལ་རིགས་ཀྱི་བྱ་བ་སྔོན་འགོག་གི་རྟོག་ཞིབ་སྙན་ཐོ་སྒྲོགས་སྦྱང་བྱས་ཡོད་པ་དང་། སྐབས་དེར་རྒྱལ་ཁབ་འཕེལ་རྒྱས་དང་བསྒྱུར་བཅོས་ལྷན་ཁང་རྡོ་སོལ་ཚོང་གཉེར་ཁང་གི་མགོ་ཁྲིད་གཞོན་པ་ཝེ་ཕུང་ཡན་གྱི་ཁྱིམ་དུ་སྔོག་བཤེར་བྱེད་སྐབས། དངུལ་རྐྱང་རྒྱ་སྒོར་དུང་ཕྱུར་ ༢ ལྷག་ལག་སོན་བྱུང་ཡོད་པ་དང་། དེ་ནི་རྒྱ་ནག་དམར་ཤོག་ཚོགས་པ་བཙུགས་ནས་ད་ལྟའི་བར་སྔོག་བཤེར་ཁྲོད་ལག་སོན་བྱུང་བའི་དངུལ་འབོར་མང་ཤོས་ཤིག་ཆགས་ཡོད་སྐོར་བསྒྲགས་འདུག གསར་འགོད་ལྷན་ཚོགས་ཐོག རྒྱ་ནག་ཆེས་མཐོའི་མི་མང་ཞིབ་བཤེར་ལྷན་ཁང་ལྐོག་རྔན་རྒོལ་རྡུང་ཚན་པའི་འགན་འཛིན་གཞོན་པ་ཞི་ཅིན་ཧིན་གྱིས། ལྐོག་རྔན་རྒོལ་རྡུང་ཚན་པས་པེ་ཅིན་དང་ཧུའུ་པེ། ཧེ་ལོང་ཅང་སོགས་སུ་ཐད་ཀར་ལྐོག་ཟ་རུལ་སུངས་ལ་རྟོག་ཞིབ་ཀྱི་ཚོགས་ཆུང་བཏང་སྟེ་ཞིབ་བཤེར་བྱས་པ་བརྒྱུད། ད་ལྟའི་བར་རྒྱལ་ཁབ་འཕེལ་རྒྱས་དང་བསྒྱུར་བཅོས་ལྷན་ཁང་གི་མི་སྣ་འགའ་ཡིས། དབང་ཆ་ལོག་སྤྱོད་ཀྱིས་ལྐོག་རྔན་བླངས་བའི་ཉེས་དོན་ ༡༡ བསྔོགས་ཏེ། འབྲེལ་ཡོད་དོགས་ཡོད་ཉེས་ཅན་ ༡༡ ཐེར་འདོན་བྱས་ཡོད་པའི་ནང་རྒྱལ་ཁབ་ནུས་ཁུངས་འདོན་སྤེལ་ཁང་ནས་མི་ ༥ དང་། དངོས་ཟོག་རིན་གོང་ཞིབ་འཇུག་གཏན་འབེབས་ཁང་གི་མི་ ༥། ལས་དོན་དང་ཡོང་སྒོ་བགོ་འགྲེམས་ཁང་གི་མི་གཅིག་བཅས་ཡོད་སྐོར་བརྗོད་འདུག ཡང་ཞི་ཅིན་ཧིན་གྱིས། རྒྱལ་ཁབ་འཕེལ་རྒྱས་དང་བསྒྱུར་བཅོས་ལྷན་ཁང་ནང་ལྐོག་རྔན་གྱི་གནད་དོན་བྱུང་བར་དམིགས་བསལ་གྱི་ཁྱད་ཆོས་འགའ་ཡོད་པ་ནི། གཅིག་སྡེ་ཁག་གི་དབང་ཆ་ཆེ་དྲགས་པ་དང་དབང་ཆ་གཅིག་བསྡུས་ནི་གནད་དོན་བྱུང་བའི་རྒྱུ་རྐྱེན་གཙོ་བོ་ཞིག་ཆགས་ཡོད་པ་དང་། གཉིས། དོ་དམ་དང་ལྟ་སྐུལ་གྱི་སྒྲིག་ལམ་ཚགས་ཚུད་མེད་པ། གསུམ། ལས་བསྒྲུབ་དངུལ་ལེན་གྱི་སྒྲིག་གཞི་ནང་མགོ་ཁྲིད་ཚོ་སིམ་འཛུལ་གྱིས། དུས་ཡུན་རིང་པོར་ཐུན་མོང་ལྐོག་རྔན་བླངས་ཏེ་ཉེས་དོན་སྤུང་གསོག་བྱས་པ། བཞི། ལྷན་ཁང་དེའི་ནང་ལྐོག་རྔན་གྱི་དངུལ་འབོར་དུང་ཕྱུར་ལས་བརྒལ་བའི་དོགས་ཡོད་ཉེས་ཅན་ ༦ ཡོད་པ་དང་། དེའི་ནང་ནས་རྡོ་སོལ་ཚོང་གཉེར་ཁང་གི་མགོ་ཁྲིད་གཞོན་པ་ཝེ་ཕུང་ཡན་གྱི་ཁྱིམ་དུ་སྔོག་བཤེར་བྱེད་སྐབས། དངུལ་རྐྱང་རྒྱ་སྒོར་དུང་ཕྱུར་ ༢ ལྷག་ལག་སོན་བྱུང་ཡོད་ལ། དེ་ནི་རྒྱ་ནག་དམར་ཤོག་ཚོགས་པ་བཙུགས་ནས་ད་ལྟའི་བར་གྱི་སྔོག་བཤེར་ཁྲོད་ལག་སོན་བྱུང་བའི་དངུལ་འབོར་མང་ཤོས་ཤིག་ཆགས་ཡོད་སྐོར་བརྗོད་འདུག གསར་འགོད་པ། པདྨ་མཚོ།
