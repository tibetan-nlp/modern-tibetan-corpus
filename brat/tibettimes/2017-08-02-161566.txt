བྱང་ཀོ་རི་ཡས་འཇིགས་སྣང་བསྐུལ།
ཊི་རུམ་ཕིས་གང་བྱེད་ཐུབ་ཀྱི་རེད།

སི་ཨིན་ཨིན་གསར་ཁང་གིས་ ༢༠༡༧།༧།༢༩ ཉིན་སྤེལ བའི གནས ཚུལ། བྱང་ཀོ་རི་ཡའི་རྒྱང་སྐྱོད་མེ་ཤུགས་འཕུར་མདའ་ཚོད་ལྟ་བྱས་པ་དེས་ཨེ་ཤ་ཡའི་གླིང་ཆེན་དང་ཨ་རིའི་རྒྱལ་ས་ཝ་ཤིན་ཊོན་ལ་སེམས་འཚབ་ཆེན་པོ་བསླངས་ཡོད་པ་མ་ཟད། བྱང་ཀོ་རི་ཡའི་གདུག་རྩུབ་ཆེ་བའི་མགོ་ཁྲིད་དང་ཨ་རིར་སྙོབ་ཐུབ་པའི་མེ་ཤུགས་འཕུར་མདའ་གཉིས་ནི། ད་ལྟ་ཊི་རུམ་ཕི་གཞུང་གི་ཕྱི་སྲིད་ལ་དཀའ་ཁག་ཆེ་ཤོས་སྤྲོད་མཁན་དེ་ཆགས་ཡོད། བྱང་ཀོ་རི་ཡས་ཨ་རིར་ཞེད་སྣང་སློང་གི་ཡོད་པ་མ་ཟད། འདས་པའི་ལོ་ཟླའི་ནང་བྱང་ཀོ་རི་ཡས་ཨ་རིའི་སྲིད་འཛིན་མང་པོར་གདོང་གཏད་མྱོང་ཡོད། བྱང་ཀོ་རི་ཡས་རྡུལ་ཕྲན་མཚོན་ཆ་ཚོད་ལྟ་བྱེད་སྐབས། ཊི་རུམ་ཕི་གཞུང་ངོས་ནས་ “འདམ་ག་ནི་ཅོག་རྩེའི་སྟེང་དུ་ཡོད་” ཅེས་བརྗོད་དེ་བྱང་ཀོ་རི་ཡར་ཉེན་བརྡ་བཏང་ཡོད། གལ་སྲིད་རྒྱལ་ཁབ་དེ་གཉིས་རྒྱལ་ཁབ་ཡག་པོ་ཞིག་མ་རེད་ཅེས་བརྗོད་ན། རྒྱལ་ཁབ་དེ་གཉིས་ཀྱིས་ལན་འདེབས་གང་ཞིག་བྱེད་སྲིད་དམ། ༡) དཔལ་འབྱོར་གནོན་ཤུགས། ཨ་རི་དང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་གཉིས་ཀྱིས་རྒྱལ་སྤྱིའི་སྟེང་བྱང་ཀོ་རི་ཡ་དང་ཚོང་འབྲེལ་བྱེད་རྒྱུ་བཀག་འགོག་བྱས་ཡོད། འོན་ཀྱང་། ཨ་རི་དང་མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱིས་བྱང་ཀོ་རི་ཡའི་རྡུལ་ཕྲན་མཚོན་ཆའི་འོད་མདངས་དེ་ད་དུང་འགེབ་ཐུབ་མེད། ད་ལྟ་ཨ་རིའི་དྲུང་ཆེ་ཊེ་ལར་སཱོན་གྱིས་བྱང་ཀོ་རི་ཡ་ལ་རྒྱལ་ཁབ་སུ་ཞིག་གིས་ཚོང་འབྲེལ་བྱས་ན། དེ་ལ་ཨ་རིས་ཉེས་ཆད་གཏོང་གི་ཡིན་ཞེས་བརྗོད་འདུག སྐད་ཆ་དེས་ཤུགས་ལས་རྒྱ་ནག་ལ་ཉེན་བརྡ་གཏོང་གི་ཡོད། གང་ཡིན་ཟེར་ན། བྱང་ཀོ་རི་ཡའི་ཕྱིར་ཚོང་བརྒྱ་ཆ་ ༩༠ ནི་རྒྱ་ནག་ཆགས་ཡོད་སྟབས། ཊེ་ལར་སཱོན་གྱི་ཐབས་ཇུས་དེ་ལེགས་འགྲུབ་འབྱུང་དཀའ་མོ་རེད། བྱང་ཀོ་རི་ཡར་རྡུལ་ཕྲན་མཚོན་ཆ་གསར་སྐྲུན་བྱེད་པའི་མ་དངུལ་ཡོང་སྒོ་འཛམ་གླིང་ས་ཕྱོགས་གང་སར་ཁྲིམས་འགལ་ནག་ཚོང་བྱེད་པ་ལས་ཡོང་གི་ཡོད། དྲ་རྒྱ་བརྒྱུད་དེ་དངུལ་ཁང་ཁག་ལ་ལྐོག་འཛུལ་དང་། དམག་གི་མཚོན་ཆ་དང་མྱོས་རྫས་རིགས་ལྐོག་ཚོང་བྱེད་པ། དངུལ་ལོ་རྫུན་བཟོ་བྱེད་པ། ད་དུང་ལྐོག་ཏུ་མི་ཚོང་རྒྱག་རྒྱུའི་ཉེན་ཁའང་ཆེན་པོ་འདུག བྱང་ཀོ་རི་ཡས་དེ་འདྲའི་ཁྲིམས་འགལ་གྱི་བྱ་བ་བྱས་ཏེ་ཨ་སྒོར་ས་ཡ་བརྒྱ་ཕྲག་ཁ་ཤས་ཀྱི་ཁེ་ཕན་ལེན་ཐུབ་ཀྱི་ཡོད་སྟབས། མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱིས་བྱང་ཀོ་རི་ཡར་ཚོང་འབྲེལ་བཀག་འགོག་བྱས་ཡོད། འོན་ཀྱང་། བྱང་ཀོ་རི་ཡ་གཞུང་གིས་མུ་མཐུད་དེ་རྡུལ་ཕྲན་མཚོན་ཆ་གསར་བཟོ་དང་ཚོད་ལྟ་བྱེད་ཐུབ་ཀྱི་ཡོད་ཚུལ་སི་ཨིན་ཨིན་གསར་ཁང་ལ་བརྗོད་ཡོད་འདུག ༢) དྲ་རྒྱ་དང་གནས་ཚུལ་གྱི་དམག་འཁྲུག གདུག་རྩུབ་ཆེ་བའི་བྱང་ཀོ་རི་ཡས་གཞན་གྱི་དྲ་རྒྱར་ལྐོག་འཛུལ་བྱེད་པ་མ་ཟད། གནས་ཚུལ་བདེན་རྫུན་སྣ་ཚོགས་ཀྱི་དམག་འཁྲུག་རྒྱག་བཞིན་པ་འཛམ་གླིང་ས་ཕྱོགས་གང་ས་ནས་ཤེས་ཀྱི་ཡོད། དེང་སང་བྱང་ཀོ་རི་ཡས་སུ་ལའང་བརྩི་འཇོག་དང་ཉན་འཇོག་མི་བྱེད་པར་རྡུལ་ཕྲན་མཚོན་ཆ་ཚོད་ལྟ་བྱེད་བཞིན་པའི་ལས་འགུལ་དེ་དྲ་རྒྱ་བརྒྱུད་བཀག་འགོག་བྱེད་ཐུབ་སྲིད་ཟེར། འོན་ཀྱང་། དྲ་རྒྱའི་ཁྱད་ལས་པ་ཚོས་བརྗོད་པ་ལྟར་ན། བྱང་ཀོ་རི་ཡའི་དྲ་རྒྱར་ལྐོག་འཛུལ་བྱས་ཏེ་རྡུལ་ཕྲན་མཚོན་ཆའི་ལས་འགུལ་གནས་སྐབས་རིང་འཐུས་ཤོར་གཏོང་ཐུབ་སྲིད་ཀྱང་། ལས་འགུལ་དེ་རྦད་དེ་བཀག་འགོག་བྱེད་མི་ཐུབ་ཅེས་བརྗོད་ཡོད་འདུག ༣) གཉིས་ཕན་གྱི་གྲོས་མོལ། འདས་པའི་ཨ་རིའི་གཞུང་གིས་བྱང་ཀོ་རི་ཡ་དང་གྲོས་མོལ་ཡོང་བའི་འབད་བརྩོན་བྱས་ཡོད། ལོ་སྔོན་མར་ལྷོ་ཀོ་རི་ཡ་དང་ཉི་ཧོང་། རྒྱ་ནག ཨུ་རུ་སུ་སོགས་རྒྱལ་ཁབ་ཁ་ཤས་ཀྱིས་མཉམ་དུ་ཚོགས་འདུ་ཚོགས་ཏེ། བྱང་ཀོ་རི་ཡར་ཁ་གཏད་ལྕོག་རྒྱུའི་གྲོས་མོལ་བྱས་ཡོད་ཀྱང་ལེགས་འགྲུབ་བྱུང་མེད། འོན་ཀྱང་། ཐེངས་ཤིག་བྱང་ཀོ་རི་ཡས་ལོ་ ༢༠ རིང་རྡུལ་ཕྲན་མཚོན་ཆའི་ལས་འགུལ་མཚམས་འཇོག་བྱེད་ཀྱི་ཡིན། དེ་སྔ་མཚམས་འཇོག་བྱས་ཡོད་པའི་ཚོང་འབྲེལ་དག་སླར་ཡང་མགོ་འཛུགས་བྱེད་རོགས་ཞེས་ཞུས་ཡོད་ཀྱང་། རྗེས་སུ་བྱང་ཀོ་རི་ཡ་དམ་བཅའ་དེ་ལས་བརྒལ་ཡོད་པ་རེད། ལོ་འདིའི་མགོར་ཨ་རིའི་སྲིད་འཛིན་ཊི་རུམ་ཕིས། གོ་སྐབས་བཟང་པོ་ཞིག་དང་བསྟུན་ཏེ་བྱང་ཀོ་རི་ཡའི་མགོ་ཁྲིད་ཀཱེམ་ཇོང་ཨུན་དང་ཐུག་འཕྲད་བྱས་ཏེ། བྱང་ཀོ་རི་ཡའི་རྡུལ་ཕྲན་མཚོན་ཆས་གང་སར་སེམས་འཚབ་སློང་བཞིན་པ་དེ་ལྷིང་འཇགས་བཟོ་འདོད་འདུག་ཅེས་བརྗོད་ཡོད། ༤) རྒྱལ་སྤྱིའི་གནོན་ཤུགས་ལས་འགུལ། ལས་འགུལ་དེ་ནི་གསར་པ་ཞིག་རེད། ང་ཚོས་ལས་འགུལ་དེ་ཇི་ཙམ་ལམ་ལྷོངས་ཡོང་མིན་ལྟ་ཆོག འདི་ལོའི་ཕྱི་ཟླ་ ༤ པར་ཨ་རིའི་དྲུང་ཆེ་ཊེ་ལར་སཱོན་གྱིས། མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཁོངས་སུ་ཡོད་པའི་རྒྱལ་ཁབ་དག་གིས་བྱང་ཀོ་རི་ཡ་དང་ཚོང་འབྲེལ་བྱེད་མི་ཆོག་པའི་བཀའ་བརྒྱ་བཏང་སྟེ། བྱང་ཀོ་རི་ཡ་ཁེར་རྐྱང་དུ་འཕུད་དགོས་པའི་རེ་འདུན་བཏོན་ཡོད་པ་མ་ཟད། མཉམ་འབྲེལ་རྒྱལ་ཚོགས་ཀྱི་རྒྱུན་ལས་ཚོས་དེ་ལ་ངེས་པར་དུ་བརྩི་འཇོག་བྱེད་དགོས་པའི་བཀའ་ནན་པོ་བཏང་ཡོད། འོན་ཀྱང་། སྐབས་དེར་ཊེ་ལར་སཱོན་གྱིས་བྱང་ཀོ་རི་ཡ་དང་ཨ་རི་གཉིས་ཀྱི་ཕྱིར་ཚོང་ལ་འབྲེལ་བ་མེད་པའི་རྒྱལ་ཁབ་དག་གིས་བཀའ་བརྒྱ་དེ་དང་ལེན་བྱས་ན་ཕན་ཐོགས་གང་ཡོད་མེད་སྐོར་བརྗོད་མེད། ༥) བྱང་ཀོ་རི་ཡའི་དམག་གཞུང་བརྗེ་བ། ཊི་རུམ་ཕི་གཞུང་གིས་བྱང་ཀོ་རི་ཡའི་དམག་གཞུང་དེ་བརྗེ་དགོས་པའི་སྐོར་ལ་མགོ་རྙོག་ཅན་གྱི་གནས་ཚུལ་ཞིག་བཏང་ཡོད། འདི་ལོའི་ཕྱི་ཟླ་ ༤ པར་སྐུ་ཞབས་ཊེ་ལར་སཱོན་གྱིས་བྱང་ཀོ་རི་ཡའི་དམག་གཞུང་བརྗེ་དགོས་ཞེས་བརྗོད་སྐབས། མགོ་ཁྲིད་ཀཱེམ་ཇོང་ཨུན་གྱི་དབང་ཆ་མེད་པ་བཟོ་དགོས་ཞེས་པའི་དོན་མ་རེད་ཅེས་འགྲེལ་བརྗོད་བྱས་ཡོད་འདུག ཕྱི་ཟླ་ ༧ པར་ཨ་རིའི་རྒྱལ་སྤྱིའི་གསང་བའི་ལས་ཁུངས་ཀྱི་འགན་འཛིན་མཱ་ཡེག་པཱོམ་པིའོ་(Mike Pompeo)ཡིས་གོང་གི་བསམ་ཚུལ་དེར་རྒྱབ་སྐྱོར་ཡོད་པའི་རྣམ་པ་བསྟན་ཡོད་པ་མ་ཟད། གལ་ཆེ་ཤོས་དེ་མི་དང་རྡུལ་ཕྲན་མཚོན་ཆ་གཉིས་དབྱེ་འབྱེད་བྱེད་ཐུབ་དགོས། གང་ཡིན་ཟེར་ན། རྡུལ་ཕྲན་མཚོན་ཆ་ལས་ཀྱང་འཇིགས་སྣང་སྐྱེ་དགོས་པ་ནི་རྡུལ་ཕྲན་མཚོན་ཆ་བེད་སྤྱོད་གཏོང་མཁན་གི་མི་ཡིན་བརྗོད་ཡོད་འདུག ༦) དམག་མིའི་སྟོབས་ཤུགས། ཊི་རུམ་ཕི་གཞུང་གིས་བྱང་ཀོ་རི་ཡ་དང་དམག་འཁྲུག་རྒྱག་རྒྱུར་གྲ་སྒྲིག་ཡོད་པའི་རྣམ་འགྱུར་སྟོན་གྱི་ཡོད། འོན་ཀྱང་། འཛམ་གླིང་འདིའི་སྒང་གི་རྒྱལ་ཁབ་སུས་ཀྱང་དམག་འཁྲུག་བྱུང་ན་འདོད་ཀྱི་མེད། གལ་ཏེ་ཨ་རི་དང་བྱང་ཀོ་རི་ཡ་གཉིས་དམག་འཁྲུག་ཤོར་ན། དོན་མེད་དུ་ཨེ་ཤ་ཡའི་རྒྱལ་ཁབ་ཁག་དང་ལྷག་པར་དུ་རྒྱ་ནག་ནང་ཟིང་འཁྲུག་དང་དེའི་ཤུགས་རྐྱེན་ཆེན་པོ་ཐེབས་ངེས་ཡིན། སྤྱིར་དམག་འཁྲུག་དེར་ཨ་རིར་རྒྱལ་ཁ་ཐོབ་སྲིད་ཀྱང་། དམག་འཁྲུག་གིས་རྐྱེན་བྱས་ཏེ་མང་ཚོགས་སྟོང་ཕྲག་མང་པོ་འདས་གྲོངས་སུ་འགྱུར་རྒྱུ་མ་ཟད། ལྷོ་ཀོ་རི་ཡ་ལའང་གྱོང་གུད་ཤུགས་ཆེན་པོ་ཐེབས་ངེས་ཡིན། གསར་འགོད་པ། ཉེ་མ་ཚེ་རིང་།
