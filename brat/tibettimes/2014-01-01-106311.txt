གྲྭ་བ་ཞིག་ལ་ལོ་ ༦ ཁྲིམས་ཐག་བཅད།

གྲྭ་བསོད་ནམས་ལགས་སུ་དེ་རིང་གསར་དུ་གནས་ཚུལ་འབྱོར་བར། ཟླ་སྔོན་མའི་ཕྱི་ཚེས་ ༡༡ ཉིན་བསང་ཆུ་རྫོང་མི་དམངས་ཁྲིམས་ཁང་གིས་བླ་བྲང་འབོ་ར་དགོན་པའི་གྲྭ་དགེ་འདུན་རྒྱ་མཚོ་ལགས་སུ། མི་བསད་པའི་ཉེས་འཛུགས་འོག་ལོ་ ༦ གི་བཙོན་འཇུག་ཁྲིམས་ཐག་བཅད་ཡོད་འདུག ཕྱི་ལོ་ ༢༠༡༢ ཟླ་ ༡༢ ཚེས་ ༢ ཉིན། བླ་བྲང་འབོ་ར་དགོན་པའི་ཉེ་འཁྲུས་གཞུང་ལམ་སྟེང་འབོ་ར་བརྡར་མདོ་སྡེ་བའི་རྒྱལ་གཅེས་དཔའ་བོ་གཟུངས་འདུས་སྐྱབས་ཀྱིས་རྒྱ་གཞུང་ལ་ངོ་རྒོལ་གྱིས་རང་ལུས་མེར་བསྲེགས་གནང་པའི་ཕྱི་ཉིན། རྒྱ་ནག་གི་ཉེན་རྟོག་པ་དང་འབོ་ར་ཞང་སྲིད་གཞུང་གི་ལས་བྱེད་འབོ་ར་དགོན་པར་ཡོང་སྟེ་གྲྭ་དགེ་འདུན་རྒྱ་མཚོ་དང་བློ་བཟང་འཕགས་པ། འཇམ་དབྱངས་བཟོད་པ། འཇམ་དབྱངས་བློ་གྲོས། འཇམ་དབྱངས་རྒྱ་མཚོ་བཅས་ལ་སྐད་ཆ་འདྲི་རྒྱུ་ཡོད་ཚུལ་བཤད་དེ་འཛིན་བཟུང་བྱས་ཡོད་འདུག རྒྱ་གཞུང་གིས། གཟུངས་འདུས་སྐྱབས་ཀྱིས་རང་ལུས་མེར་བསྲེགས་གཏོང་སྐབས། ཉེན་རྟོག་པས་མེ་གསོད་ཐབས་བྱས་ཀྱང་གྲྭ་དགེ་འདུན་རྒྱ་མཚོ་སོགས་ཀྱིས་མེ་གསོད་དུ་མ་བཅུག་སྟབས་མི་བསད་པའི་ནག་ཉེས་བསགས་ཡོད་ཅེས་ཉེས་འཛུགས་བྱས་ཡོད་ཀྱང༌། གྲྭ་དགེ་འདུན་རྒྱ་མཚོ་ལགས་ཀྱིས་ཉེས་འཛུགས་དེ་ངོས་ལེན་བྱས་མེད་པ་དང༌། ཁྲིམས་ཐག་གཅོད་པའི་ཉིན་མོར་ཁོང་གི་ནང་མི་ ༤ སྐད་བཏང་ཡོད་ཅིང༌། ཉིན་དེར་ཡང་ཁོང་གིས་ནག་ཉེས་དེ་ངོས་ལེན་བྱེད་ཀྱི་མེད་ཚུལ་ནང་མི་ཚོས་བརྗོད་འདུག གྲྭ་དགེ་འདུན་རྒྱ་མཚོ་ལགས་ནི། འབོ་ར་ལྷ་འབུམ་ཐང་རི་འགབ་མ་ནས་ཡིན་པ་དང༌། ད་ལྟ་རང་ལོ་ ༤༧ ། ཨ་མའི་མིང་ལ་བསོད་ནམས་འཚོ་ཟེར་གྱི་ཡོད་འདུག
