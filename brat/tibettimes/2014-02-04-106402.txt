བོད་ནང་གི་འགྲོ་བ་མིའི་ཐོབ་ཐང་གླེང་བ།

སང་ཉིན་ནས་ཉིན་བཞིའི་རིང་བོད་ཀྱི་འགྲོ་བ་མིའི་ཐོབ་ཐང་དང་མང་གཙོ་འཕེལ་རྒྱས་ལྟེ་གནས་ཁང་དང་ནོར་ཝེ་བོད་ཀྱི་རླུང་འཕྲིན་ཁང་། རི་མཐོ་ས་གཙང་དྲ་རྒྱ་བཅས་མཉམ་རུབ་ཐོག་བོད་ནང་གི་འགྲོ་བ་མིའི་ཐོབ་ཐང་སྐོར་ལ་ལས་འགུལ་སྤེལ་རྒྱུ་ཡིན་འདུག ཟླ་འདིའི་ཚེས་ ༥ ནས་ ༩ བར་ནོར་ཝེ་རྒྱལ་ས་ཨོ་སི་ལོར་རྟེན་གཞི་བྱས་པའི་དངོས་བྱུང་གློག་བརྙན་བཟོ་སྐྲུན་ཚོཊ་པ་དང་། ཡུ་རོབ་ནང་གི་འགྲོ་བ་མིའི་ཐོབ་ཐང་དང་འབྲེལ་བའི་ཚོཊ་ཁག་མཉམ་རུབ་ཐོག ནོར་ཝེ་རྒྱལ་ས་ཨོ་སི་ལོར་༼འགྲོ་བ་མིའི་ཐོབ་ཐང་རྡོག་རོལ་དང་སྲུང་སྐྱོབ༽ཞེས་པའི་གློག་བརྙན་དུས་སྟོན་ཞིག་གོ་སྒྲིག་གནང་འཆར་ཡོད་པ་དང་། དུས་སྟོན་དེའི་ཐོག་བོད་ཀྱི་འགྲོ་བ་མིའི་ཐོབ་ཐང་དང་མང་གཙོ་འཕེལ་རྒྱས་ལྟེ་གནས་ཁང་དང་ནོར་ཝེ་བོད་ཀྱི་རླུང་འཕྲིན་ཁང་། རི་མཐོ་ས་གཙང་དྲ་རྒྱ་བཅས་ཀྱིས་བོད་ནང་གི་སྨྲ་བརྗོད་རང་དབང་གནས་སྟངས་སྐོར་གྱི་དངོས་བྱུང་གློག་བརྙན་ཞིག་གཟིཊ་འབུལ་གནང་སྟེ། བོད་ནང་སྨྲ་བརྗོད་རང་དབང་སྤྲོད་དགོས་པའི་ཐད་རྒྱབ་སྐྱོར་ཡོང་བའི་རྒྱལ་སྤྱིའི་རང་བཞིན་གྱི་ལས་འགུལ་ཞིག་སྤེལ་བཞིན་ཡོད་འདུག ཉིན་གྲངས་བཞིའི་རིང་གི་དུས་སྟོན་འདིའི་ཐོག་ནོར་ཝེ་བོད་ཀྱི་རླུང་འཕྲིན་ཁང་གིས་བཟོ་བསྐྲུན་བྱས་པའི་རྒྱ་གཞུང་གིས་བོད་ནང་སྨྲ་བརྗོད་རང་དབང་རྡོག་རོལ་གཏོང་བཞིན་པའི་ཐད་ཀྱི་དངོས་བྱུང་གློག་བརྙན་༼བོད་ནང་སྨྲ་བརྗོད་རང་དབང་བཀག་སྡོམ༽ ཞེས་པ་ཐོག་མར་དབུ་འབྱེད་གཟིཊ་འབུལ་ཞུ་རྒྱུ་ཡིན་པ་དང་། རི་མཐོ་ས་གཙང་དྲ་རྒྱའི་རྩོམ་སྒྲིག་པ་བདེ་ཆེན་སྤེན་པ་ལཊ་ཀྱིས། བོད་ཀྱི་འགྲོ་བ་མིའི་ཐོབ་ཐང་དང་མང་གཙོ་འཕེལ་རྒྱས་ལྟེ་གནས་ཁང་གིས་པར་སྐྲུན་བྱས་པའི་ ༼སྨྲ་བརྗོད་རང་དབང་བཀག་སྡོམ༽ ཞེས་པའི་དེབ་ཀྱི་བརྗོད་དོན་ཁག་འགྲེལ་བརྗོད་དང་སྦྲགས། ནོར་ཝེ་ཡི་མང་ཚོགས་དང་གསར་འགོད་པ་རྣམས་ལ་བོད་ནང་གི་གནས་སྟངས་ངོ་སྤྲོད་དང་། ནོར་ཝེ་བོད་ཀྱི་རླུང་འཕྲིན་ཁང་གི་ལས་བྱེད་སྐུ་ཞབས་ཨོ་སི་ཌེན་ཨལ་མའེ་(Oystein Alme) ལགས་གཉིས་ནས་ལས་འགུལ་དང་འབྲེལ་བའི་གཏམ་བཤད། བོད་ཀྱི་གཞས་པ་གྲགས་ཅན་བློ་བསྟན་རྣམ་གླིང་ལགས་ཀྱིས་གཞས་སྣ་འཁྲབ་སྟོན་བཅས་གནང་རྒྱུ་ཡིན་འདུག ནོར་ཝེ་རྒྱལ་ས་ཨོ་སི་ལོར་སྤེལ་བཞིན་པའི་བོད་དོན་འཐབ་རྩོད་དང་འབྲེལ་བའི་བོད་ནང་སྨྲ་བརྗོད་རང་དབང་བཀག་སྡོམ་སྐོར་གྱི་རྒྱལ་སྤྱིའི་ལས་འགུལ་འདི་ནི། བོད་ཀྱི་འགྲོ་བ་མིའི་ཐོབ་ཐང་དང་མང་གཙོ་འཕེལ་རྒྱས་ལྟེ་གནས་ཁང་ནོར་ཝེ་བོད་ཀྱི་རླུང་འཕྲིན་ཁང་དང་རི་མཐོ་ས་གཙང་དྲ་རྒྱ་བཅས་མཉམ་རུབ་ཐོག་སྤེལ་གྱི་ཡོད་འདུག་ལ། དམིཊ་ཡུལ་གཙོ་བོ་དུས་སྟོན་དེ་བརྒྱུད་ནས་བོད་ནང་གི་རྩོམ་པ་པོ་དང་སྒྱུ་རྩལ་བ། ཤེས་ཡོན་པ་རྣམས་ཀྱིས་འཇིགས་ཞུམ་བྲལ་བའི་སྒོ་ནས། བོད་ནང་འགྲོ་བ་མིའི་ཐོབ་ཐང་རྡོག་རོལ་གཏོང་བཞིན་པའི་གནས་སྟངས་རྣམས་ཡིག་ངག་གཉིས་ཐོག་ནས་གསལ་འདོན་གནང་བཞིན་པའི་སྙིང་སྟོབས་ལ་གུས་ཞབས་མཚོན་ཆེད་དང་། རྒྱ་གཞུང་གིས་ཁོང་རྣམས་ལ་དྲག་གནོན་བྱེད་བཞིན་པའི་གནས་སྟངས་འཛམ་གླིང་ལ་གསལ་སྟོན་ཡོང་ཆེད་ཡིན་འདུག ༢༠༡༣ ཟླ་ ༡༢ ཚེས་ ༡༠ ཉིན་སྟེ། རྒྱལ་སྤྱིའི་འགྲོ་བ་མིའི་ཐོབ་ཐང་ཉིན་མོར་ངྷ་རམ་ས་ལར་བོད་ནང་སྨྲ་བརྗོད་རང་དབང་བཀག་སྡོམ་ཐད་ཀྱི་ལས་འགུལ་ཐོག་མ་དེ་སྤེལ་ཡོད་འདུག གསར་འགོད་པ། དོན་གོ
