ཚེ་རིང་དོན་གྲུབ་ཀྱི་བརྩམས་སྒྲུང་། ཝ་ནག་ལུང་བ། 2018/06/1009:28:58ཚེ་རིང་དོན་གྲུབ། ཀློག་གྲངས། 8849མཆན2ཡོད། 《ཝ་ནག་ལུང་བ》ཞེས་པའི་བརྩམས་སྒྲུང་འདི་དང་པོ2012ལོའི《སྦྲང་ཆར》དེབ་བཞི་བར་སྤེལ་ཅིང་། 2014ལོའི《མི་རིགས་ཀྱི་རྩོམ་རིག》དེབ་དང་པོར་བརྒྱུད་འགོད་བྱས། བརྩམས་སྒྲུང་འདི་དབྱིན་ཇི་དང་ཧྥ་རན་སི། ཉི་ཧོང་། རྒྱ་ནགསོག་པོ་སོགས་ཀྱི་ཡི་གེར་བསྒྱུར་ཡོད་པར་མ་ཟད་ད་དུང་ཡིག་རིགས་མང་པོར་སྒྱུར་བཞིན་ཡོད་པ་དང་། སློབ་ཆེན་ཁག་ཏུ་འཆད་ཁྲིད་དང་བགྲོ་གླེང་བྱེད་བཞིན་ཡོད་པ། དྲ་བ་དང་འཕྲིན་སྟེགས་སྣ་སོགས་སུ་ཡང་ཡང་སྤེལ་བཞིན་པ་སོགས་མདོར་ན་ཉེ་བའི་ལོ་འགའི་རིང་ཤུགས་རྐྱེན་ཆེས་ཆེ་བའི་བརྩམས་སྒྲུང་ཞིག་ཡིན་པས་ད་རེས་རྩོམ་མཁན་གྱི་ཆོག་མཆན་བརྒྱུད་ངེད་དྲ་བར་སྤེལ་བ་དགེའོ། ། གཅིགརྩེ་གཞུང་རྫོང་མཁར་དང་སྤྱི་ལེ་དྲུག་ཅུ་ཙམ་གྱི་བྱང་ཕྱོགས་ན། ལབ་ཙེ་ཆུང་ཆུང་ཞིག་དང་དར་ལྕོག་འགའ་འཐེན་ཡོད་པའི་ཉག་ག་ཞིག་ཡོད་ལ། ཉག་ག་དེ་ནས་ཕར་བལྟས་ན་ཁ་བྱང་ཕྱོགས་སུ་འཁོར་བའི་ལུང་བ་ཞིག་ཡོད་པ་དེར་རྩྭ་སྣ་མང་ཞིང་རྩྭ་རྫོང་མཐུག་པ། ལུང་རྐེད་ན་འདམ་ར་ལུག་ལྷས་ཙམ་ཞིག་ཡོད་པ་དེ་རུ་ག་ས་ག་ནས་ཆུ་མིག་མང་བོ་བརྡོལ་ཅིང་ཕན་ཚུན་འདྲེས་ཏེ་ཆུ་ཕྲན་དྭངས་མོ་ཞིག་ལུང་དཀྱིལ་བརྒྱུད་ནས་བཞུར་བཞིན་ཡོད། སྤྱི་ཟླ་བདུན་པ་དང་བརྒྱད་པའི་དུས་སུ་ལུང་བའི་མགོ་སྟོད་དུ་སྨག་ཤད་དང་། སྤེན་དཀར། སྤེན་ནགསུ་རུ་སོགས་ཚང་ཚིང་སྡོང་ཕྲན་གྱི་མགོ་ལ་མེ་ཏོག་སྣ་ཚོགས་བཞད་པ་དང་། སྒང་རྒྱུད་དུ་སྤྲ་དང་རེ་ལྕག་ཆུན་པོ་དང་ཆུན་པོར་སྐྱེ་བ། སྟོན་དུས་སུ་སྤང་རྒྱན་བཞད་པ། ལུང་དཀྱིལ་གྱི་ཆུ་ཕྲན་དེའི་འགྲམ་གཉིས་སུ་བོང་སེར་དང་ལུག་རུ་སེར་པོ། མ་ནིང་གླང་སྣ། སྔོ་ཁྱུང་སྡེར་སྨུག་པོ་སོགས་བཞད་པ། ལུང་མཐིལ་གྱི་བདེ་ཐང་སྤང་གཞོང་དུ་རི་ཤོ་དང་རམ་བུ། རྟ་ལྤགས་པ། སྔོ་ཁྱུང་སྡེར་དཀར་པོ། ཀྱི་ལྕེ་དཀར་ནགལུག་མིག་དཀར་སྔོ། ཁུར་མོང་། མཁན་དཀར། གྲོ་མ། གཞན་ཡང་སྐྱེ་དངོས་རིག་པའི་ཁྱུ་མཆོག་ཏུ་རློམ་པ་དག་གིས་ཀྱང་ངོས་ཟིན་དཀའ་སྙམ་པའི་མེ་ཏོག་རིགས་སྣ་ཚོགས་བཞད་དེ། ཉིན་འགའ་རེར་ས་གཞིའི་ཁ་དོག་རེ་བརྗེ་ཞིང་དྲི་ངད་མི་འདྲ་བ་རེས་ཁྱབ་ཀྱིན་འདུག་པས་ན། 《གླིང་སྒྲུང》དུ་བརྗོད་པའི“རྩྭ་ར་པདྨ་སྟོང་ལྡན”ཟེར་བ་འདི་ལོས་ཡིན་སྙམ་པར་བྱེད། དེའི་སྟེང་དུ་འབྲོག་ཁྱིམ་ཞིག་ཡོད་པ་དེའི་རྟ་ནོར་ལུག་གསུམ་ལྔ་བརྒྱ་དྲུག་བརྒྱ་ཙམ་གྱིས་མཚམས་རེར་འཚེར་སྐད་དང་ངུར་སྐད། འབའ་སྐད་རེ་སྒྲོག་ཞོར་དལ་ལྷོད་བག་ཕེབས་སུ་གནས་པའི་རྣམ་པ་འདི་མཐོང་ཚེ།
