ད་རུང་དྲོན་པོ་ཡིན། ད་ཨ་ཕ་ཨ་མ་གཉིས་ཀ་ཕྱིད་རྒྱུ་རེད”ཟེར་བ་དང་། གློ་བུར་ཅི་ཞིག་ཡིད་ལ་དྲན་པ་ལྟར“ཝེ། སྒོ་རའི་ནང་ནས་སྦྲ་ཕུབ་སྟེ་ལྕགས་ཐབ་བཙུགས་ན་མི་བཟང་ངམ”ཟེར། “དེ……ད་དེ་ལྟར་བྱེད་པ་ལས་ཐབས་མེད་པ་འདྲ། ཡིན་ནའང……”སངས་རྒྱས་ཀྱིས་ཁ་སྤུ་འཐོག་ཞོར“ཡིན་ནའང་མགོ་དམར་མས་ཅི་ཟེར་རམ”ཟེར། “རེད་ཡ། དེ་ལྟར་བཤད་ན་མགོ་དམར་མ་དེ་འུ་ཚོའི་འགན་འཁུར་པ་ཡིན་པ་འདྲ། མོ་བོས་ཡོང་ནས་ཁང་བ་འདིར་བལྟ་རུ་བཅུག་ན་བཤད་རྒྱུ་ཞིག་ལོས་ཡོད། དེ་འདྲའི‘ཙི་ཁྲིག’སྒོར་མོ་མང་བོ་སྤྲད་ནས་འདི་འདྲའི་ཁང་བ་ཞིག་དཔེ་མི་སྲིད། ”“ཡུམ་དུམ་པ་བཅུ་གཉིས། ང་མགོ་དམར་མ་དེ་ལ་སྐྲག་གི”“སྐྲག་རྒྱུ་ཅི་ཡོད། ཁྱོད་སྐྲག་ན་ང་འགྲོ། ”ཀླུ་སྒྲོན་གྱིས་དེ་ལྟར་བཤད་དེ་ཡར་ལངས་ནས་སྒོར་བུད། དྲང་མོར་བཤད་ན་མོར་ཡང་མགོ་དམར་མ་དང་སྐད་ཆ་བཤད་པའི་སྤོབས་པ་ཡོད་མེད་མི་ཤེས་ནའང་། ད་ནི་གང་ལྟར་ཁྱོ་གའི་མདུན་ནས་དཔའ་ཁས་ལེན་བྱས་ཟིན་པས། དེའི་གམ་ལ་མི་འགྲོ་ཐབས་མེད་བྱུང་། སྟབས་ལེགས་པ་ཞིག་ལ་མོ་དེར་བསླེབས་དུས། མགོ་དམར་མའི་གཞུང་ལས་ཁང་དུ་རྩེ་གཞུང་ཡུལ་ཚོའི་ཧྲུའུ་ཅི་སོགས་མགོ་ཁྲིད་རྣམས་དང་འབྲོག་པ་ཕོ་མོ་མང་བོ་འདུས་ཡོད་ལ། བོད་སྐད་ཤེས་པའི་མགོ་ཁྲིད་ཅིག་གིས་བཤད་ན་ཁོ་ཚོས་གནས་ཚུལ་འདི་སྔར་ནས་རྫོང་ཨུ་དང་རྫོང་སྲིད་གཞུང་ལ་ཡར་ཞུ་བྱས་ཟིན་པ་དང་། རྫོང་ཨུ་དང་རྫོང་སྲིད་གཞུང་གིས་ཀྱང་གནད་དོན་འདིར་ཧ་ཅང་མཐོང་ཆེན་གནང་གིན་ཡོད། དེ་རིང་སང་ཉིན་གྱི་གནམ་གཤིས་བཙོག་པོ་འདི་ཡལ་མ་ཐགཁང་བ་ཡོངས་ལ་ཉམས་གསོ་བྱ་ངེས་ཡིན་པས། རྒྱ་ཆེ་བའི་མང་ཚོགས་ཀྱིས་གནས་སྐབས་ཙམ་ལ་བཟོད་བསྲན་གནང་རོགས་ཟེར། “ད་དེས་ན་བཤད་རྒྱུ་མེད། ཏང་དང་རྒྱལ་ཁབ་ལ་བཀའ་དྲིན་ཞུ ། ”སྐད་ངག་ཧ་ཅང་སྦོམ་པའི་གསར་བུ་ཞིག་གིས་ཚང་མའི་ཚབ་བྱེད་པ་བཞིན་དེ་ལྟར་བཤད་ནས་སྒོར་བུད་པ་ན། གཞན་རྣམས་ཀྱང་གཅིག་རྗེས་གཉིས་མཐུད་ཀྱིས་ཕྱིར་བུད་སོང་། ཀླུ་སྒྲོན་འཛུམ་དང་བཅས་ཕྱིར་ཡོང་སྟེ“མགོ་ཁྲིད་ཚོས་ཉིན་འགའ་ལ་བཟོད་བསྲན་བྱོས་དང་། ཁང་བར་མྱུར་དུ་ཉམས་གསོ་བྱ་ཟེར”ཞེས་བཤད། “ཏང་དང་རྒྱལ་ཁབ་ཕ་མ་འདྲ་བོ་དེ་འདྲ་ཞིག་ལོས་ཡིན།
